33 ConstStateRef ZeroOrderAlignAutomaton::getState(StateId s) const {
34  State* sp = new State(s);
35 
36  u32 si(s);
37  u32 m(si % M);
38  u32 i(si / M / (J + 1));
39  u32 j(si / M % (J + 1));
40 
41  bool doHorizontal(m == diagonal || m == horizontal);
42  bool doVertical(m == diagonal || m == vertical);
43 
44  if (i < I && doVertical) { // when not at the TOP of lattice, make vertical movement
45  std::vector<std::string> x;
46  x.push_back("NULL");
47  x.push_back(target_[i]);
48 
49  StateId targetState(si + (J + 1) * M - m + vertical);
50 
51  f64 weight(lexicon_->getProb(vertical, x) * factorLexicon_);
52  weight += factorTransition_ * transitionProbs_[vertical];
53 
54  sp->newArc(targetState, Fsa::Weight(weight), Epsilon, outputSentence_[i]);
55  }
56  if (j < J && doHorizontal) { // when not at the RIGHT border of the lattice make horizontal movement
57  StateId targetState(si + M - m + horizontal);
58  for (u32 jt = 0; jt < J; ++jt) {
59  std::vector<std::string> x;
60  x.push_back(source_[jt]);
61  x.push_back("NULL");
62 
63  f64 weight(lexicon_->getProb(horizontal, x) * factorLexicon_);
64  weight += factorTransition_ * transitionProbs_[horizontal];
65 
66  sp->newArc(targetState, Fsa::Weight(weight), inputSentence_[jt], Epsilon);
67  }
68  }
69  if ((j < J) && (i < I)) { // when in the middle of the lattice, do diagonal movment
70  StateId targetState(si + (J + 1) * M + M - m + diagonal);
71  for (u32 jt = 0; jt < J; ++jt) {
72  std::vector<std::string> x;
73  x.push_back(source_[jt]);
74  x.push_back(target_[i]);
75 
76  f64 weight(lexicon_->getProb(diagonal, x) * factorLexicon_);
77  weight += factorTransition_ * transitionProbs_[diagonal];
78  sp->newArc(targetState, Fsa::Weight(weight), inputSentence_[jt], outputSentence_[i]);
79  }
80  }
81 
82  if (i == I && j == J) {
83  sp->setFinal(this->semiring()->one());
84  }
85  return ConstStateRef(sp);
86 }
