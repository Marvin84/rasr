245 // A thread-safe scorer cache
246 class ContextScorerCache : public Core::ReferenceCounted {
247  struct ScorerHash {
248  size_t operator()(const Core::Ref<const Feature>& feature) const {
249  return ((size_t)feature.get()) * 2654435761u;
250  }
251  };
252 
253 public:
254  ContextScorerCache(u64 maxSize)
255  : maxSize_(maxSize), epoch_(0) {
256  }
257 
258  // Every returned cached scorer must be released by calling uncacheScorer as soon as it is not needed any more
259  CachedContextScorer cacheScorer(Core::Ref<const Feature> feature, u32 nEmissions) {
260  Core::MutexLock lock(&mutex_);
261  Core::HashMap<Core::Ref<const Feature>, std::pair<CachedContextScorer, u64>, ScorerHash>::iterator it = map_.find(feature);
262  if (it != map_.end()) {
263  CachedContextScorer ret = it->second.first;
264  epochMap_.erase(it->second.second);
265  map_.erase(it);
266  return ret;
267  }
268  return CachedContextScorer(new CachedFeatureScorer::CachedContextScorerOverlay(nEmissions));
269  }
270 
271  // After this was called, only the returned scorer must be used, instead of the given cached scorer.
272  FeatureScorer::Scorer uncacheScorer(Core::Ref<const Feature> feature, CachedContextScorer cachedScorer) {
273  FeatureScorer::Scorer ret = cachedScorer->scorer();
274  const_cast<CachedFeatureScorer::CachedContextScorerOverlay&>(*cachedScorer).setScorer(FeatureScorer::Scorer());
275  Core::MutexLock lock(&mutex_);
276  Core::HashMap<Core::Ref<const Feature>, std::pair<CachedContextScorer, u64>, ScorerHash>::iterator it = map_.find(feature);
277  if (it != map_.end())
278  return ret;
279  map_.insert(std::make_pair(feature, std::make_pair(cachedScorer, epoch_)));
280  epochMap_.insert(std::make_pair(epoch_, feature));
281  epoch_ += 1;
282  if (map_.size() > maxSize_) {
283  map_.erase(epochMap_.begin()->second);
284  epochMap_.erase(epochMap_.begin());
285  }
286  return ret;
287  }
288 
289  void clear() {
290  Core::MutexLock lock(&mutex_);
291  map_.clear();
292  epoch_ = 0;
293  }
294 
295 private:
296  Core::Mutex mutex_;
297  u64 maxSize_, epoch_;
298  Core::HashMap<Core::Ref<const Feature>, std::pair<CachedContextScorer, u64>, ScorerHash> map_;
299  std::map<u64, Core::Ref<const Feature>> epochMap_;
300 };
