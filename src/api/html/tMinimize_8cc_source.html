102  class HashTargetMappedAndLabels {
103  private:
104  bool transducer_;
105  _ConstAutomatonRef fsa_;
106  const ClassMap& classMap_;
107  typename A::ConstSemiringRef semiring_;
108  const StatePotentials<_Weight>& statePotentials_;
109 
110  public:
111  HashTargetMappedAndLabels(_ConstAutomatonRef f,
112  const ClassMap& classMap,
113  const StatePotentials<_Weight>& statePotentials)
114  : transducer_(f->type() == Fsa::TypeTransducer), fsa_(f), classMap_(classMap), semiring_(f->semiring()), statePotentials_(statePotentials) {}
115  size_t operator()(Fsa::StateId s) const {
116  const _ConstStateRef sp = fsa_->getState(s);
117  size_t key = 100003 * semiring_->hash(statePotentials_[s]) + sp->nArcs();
118  for (typename A::State::const_iterator a = sp->begin();
119  a != sp->end(); ++a) {
120  key = 100003 * key + classMap_[a->target()];
121  key = 100003 * key + a->input();
122  if (transducer_)
123  key = 100003 * key + a->output();
124  }
125  return key;
126  }
127  };
128  class EqualTargetMappedAndLabels {
129  private:
130  _ConstAutomatonRef fsa_;
131  const ClassMap& classMap_;
132  typename A::ConstSemiringRef semiring_;
133  const StatePotentials<_Weight>& statePotentials_;
134 
135  public:
136  EqualTargetMappedAndLabels(_ConstAutomatonRef f,
137  const ClassMap& classMap,
138  const StatePotentials<_Weight>& statePotentials)
139  : fsa_(f), classMap_(classMap), semiring_(f->semiring()), statePotentials_(statePotentials) {
140  }
141  bool operator()(Fsa::StateId as, Fsa::StateId bs) const {
142  if (semiring_->compare(statePotentials_[as],
143  statePotentials_[bs]) != 0)
144  return false;
145  const _ConstStateRef a = fsa_->getState(as), b = fsa_->getState(bs);
146  if (a->nArcs() != b->nArcs())
147  return false;
148  for (typename A::State::const_iterator aa = a->begin(), ba = b->begin();
149  aa != a->end(); ++aa, ++ba)
150  if ((classMap_[aa->target()] != classMap_[ba->target()]) || (aa->input() != ba->input()) || (aa->output() != ba->output()) || (semiring_->compare(aa->weight(), ba->weight()) != 0))
151  return false;
152  return true;
153  }

157  _ConstAutomatonRef minimize(_ConstAutomatonRef f,
158  Fsa::OptimizationHint hint, bool progress) {
159  // 1. get state potentials
160  StatePotentials<_Weight> statePotentials = ssspBackward<A>(f,
161  progress);
162  Fsa::StateId maxStateId = statePotentials.size() - 1;
163 
164  // 2. prepare automaton
165  bool transducer = (f->type() == Fsa::TypeTransducer);
166  if (transducer)
167  f = sort<A>(f, Fsa::SortTypeByInputAndOutput);
168  else
169  f = sort<A>(f, Fsa::SortTypeByInput);
170  f = cache<A>(f);
171 
172  // 3. maintain two tables of fixed size:
173  // - the equivalence set: all active states sorted by the class id (start with 2 classes: final/non-final)
174  // - a direct map of all states onto their class id
175  EquivalenceSet es, nes;
176  ClassMap classMap(maxStateId + 1, -1);
177  Core::Vector<size_t> classSize(2, 0);
178  for (size_t s = 0; s <= maxStateId; ++s) {
179  typename A::ConstStateRef state = f->getState(s);
180  classSize[classMap[s] = state->isFinal() ? 1 : 0]++;
181  if (state)
182  es.push_back(ClassEntry(s, classMap[s]));
183  }
184  sort(es.begin(), es.end());
185  nes.reserve(es.size());
186 
187  // progress indicator for split loop
188  Core::ProgressIndicator* p = 0;
189  if (progress) {
190  p = new Core::ProgressIndicator("splitting equiv. sets", "states");
191  p->start();
192  }
193 
194  // 4. split equivalence set
195  // assume that allocated memory of classMap doesn't change
196  bool didSplit = true;
197  HashTargetMappedAndLabels hashTargetMappedAndLabels(f, classMap,
198  statePotentials);
199  EqualTargetMappedAndLabels equalTargetMappedAndLabels(f, classMap,
200  statePotentials);
201  while (didSplit) {
202  // iterate over all classes (they are sorted by class id) and build new equivalence set
203  nes.clear();
204  didSplit = false;
205  ClassId nClasses = 0;
206  for (typename EquivalenceSet::const_iterator s = es.begin(); s != es.end();) {
207  ClassId c = s->class_;
208  if (classSize[c] == 1) { // singleton classes can't be split anymore
209  //classMap[(s++)->id_] = -(nextSingletonId++);
210  nes.push_back(ClassEntry((s++)->id_, nClasses++));
211  if (p)
212  p->notify(); // report progress
213  }
214  else {
215  // The best size for the following hash table is of course a prime number, which may be costly to
216  // calculate. The following is already an optimization over previous versions:
217  //
218  // 1. A sorted vector for each class where splitted classes get merged again. Performed badly, because
219  // either the sorted vector used ConstStateRefs which was inefficient memory-wise or state ids which
220  // was painfully slow in case of packed automata.
221  //
222  // 2. A static global hash table which performed badly, because class sizes continously shrink and
223  // clearing the big same sized hash table all the time becomes a big burden.
224  size_t startOfNewSets = nes.size();
225  Fsa::Hash<Fsa::StateId, HashTargetMappedAndLabels, EqualTargetMappedAndLabels>
226  states(hashTargetMappedAndLabels,
227  equalTargetMappedAndLabels,
228  classSize[c]);
229  for (; (s != es.end()) && (s->class_ == c); ++s) {
230  nes.push_back(ClassEntry(s->id_, nClasses + states.insert(s->id_)));
231  if (p)
232  p->notify(); // report progress
233  }
234  if (states.size() > 1) {
235  // just another trick:
236  // 1. we only need to sort sets if we know there was more than 1 class
237  // 2. overall complexity of this is smaller: sum(N_i)*log(N_i) <= sum(N_i)*log(sum(N_i))
238  // 3. sorting works more local thus better for CPU cache
239  std::sort(nes.begin() + startOfNewSets, nes.end());
240  didSplit = true;
241  }
242  nClasses += states.size();
243  }
244  }
245 
246  // swap equivalence sets and update class map & sizes
247  std::swap(es, nes);
248  Core::Vector<size_t> newClassSize(nClasses, 0);
249  for (typename EquivalenceSet::const_iterator s = es.begin(); s != es.end(); ++s) {
250  classMap[s->id_] = s->class_;
251  newClassSize[s->class_]++;
252  }
253  std::swap(classSize, newClassSize);
254  }
255  if (p) {
256  p->finish();
257  delete p;
258  }
259 
260  // 5. convert equivalence set into state id map
261  Precursor::map_.resize(maxStateId + 1);
262  for (size_t s = 0; s <= maxStateId; ++s)
263  Precursor::map_[s] = s;
264  for (typename EquivalenceSet::const_iterator s = es.begin(); s != es.end();) {
265  ClassId c = s->class_;
266  Fsa::StateId rep = s->id_;
267  for (++s; (s != es.end()) && (s->class_ == c); ++s)
268  Precursor::map_[s->id_] = rep;
269  }
270 
271  return f;
272  }

281  }
