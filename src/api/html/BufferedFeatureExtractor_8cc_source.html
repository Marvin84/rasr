113 template<typename T>
114 void BufferedFeatureExtractor<T>::initShuffle() {
115  // Resize the index buffer to full size
116  if (shuffledIndices_.size() != nBufferedFeatures_)
117  shuffledIndices_.resize(nBufferedFeatures_);
118 
119  verify(maxBufferSize_ >= nBufferedFeatures_);
120 
121  // Fill the index vector with the corresponding index number
122  u32 curIndex = 0;
123  for (std::vector<u32>::iterator shuffleIndexIterator = shuffledIndices_.begin();
124  shuffleIndexIterator != shuffledIndices_.end(); ++shuffleIndexIterator, ++curIndex) {
125  (*shuffleIndexIterator) = curIndex;
126  }
127 
128  // Shuffle now the indices from 1 to nBufferedFeatures_
129  std::shuffle(shuffledIndices_.begin(), shuffledIndices_.end(), shuffleRandomEngine_);
130 
131  // log some of the random indices (helpful for debugging)
132  std::stringstream ss;
133  u32 nIndicesForLogging = std::min((size_t)5, shuffledIndices_.size() / 2);
134  for (u32 i = 0; i < nIndicesForLogging; i++)
135  ss << shuffledIndices_.at(i) << " ";
136  ss << " ... ";
137  for (u32 i = 0; i < nIndicesForLogging; i++)
138  ss << shuffledIndices_.at(shuffledIndices_.size() - nIndicesForLogging + i) << " ";
139  log("accessing data in random order. Shuffled indices: ") << ss.str();
140 }

142 /** Initialize the (feature) buffer.
143  *
144  * Initialize the (feature) buffer. The number of elements in the buffer is
145  * limited by @param nFrames * \@parame nFeatures
146  *
147  * @param nFeatures Dimension of the features
148  * @param nFrames Number of frames
149  */
150 template<typename T>
151 void BufferedFeatureExtractor<T>::initBuffer(std::vector<u32>& nFeatures) {
152  segmentIndexBuffer_.resize(maxBufferSize_);
153  featureBuffer_.resize(nFeatures.size());
154  if (slidingWindowSizeDerivatives_ > 0) {
155  derivativesBuffer_.resize(nFeatures.size());
156  }
157  for (u32 streamIndex = 0; streamIndex < nFeatures.size(); streamIndex++) {
158  featureBuffer_[streamIndex].resize(nFeatures[streamIndex], maxBufferSize_);
159  if (slidingWindowSizeDerivatives_ > 0) {
160  /* nFeatures[streamIndex] + 1: first derivative of each component (nFeatures[streamIndex])
161  * and second derivative of first component (+1)
162  */
163  derivativesBuffer_[streamIndex].resize(nFeatures[streamIndex] + 1, maxBufferSize_);
164  }
165  }
166 
167  nBufferedFeatures_ = 0;
168  needInit_ = false;
169 }

193  for (u32 streamIndex = 0; streamIndex < featureBuffer_.size(); streamIndex++) {
194  featureBuffer_[streamIndex].setToZero();

196  for (u32 streamIndex = 0; streamIndex < derivativesBuffer_.size(); streamIndex++) {

210 template<typename T>
211 void BufferedFeatureExtractor<T>::updateBufferedFeature(Core::Ref<const Speech::Feature> f) {
212  // Copy the features of each stream (not only the main stream) into the buffer
213  u32 streamIndex = 0;
214  for (Mm::Feature::Iterator streamIterator = f->begin(); streamIterator != f->end(); ++streamIterator, ++streamIndex) {
215  for (u32 i = 0; i < (*streamIterator)->size(); ++i) {
216  featureBuffer_[streamIndex].at(i, nBufferedFeatures_) = (T)(*streamIterator)->at(i);
217  }
218  }
219  // buffer segment index (for windowing)
220  segmentIndexBuffer_.at(nBufferedFeatures_) = segmentIndex_;
221  nBufferedFeatures_++;
222 }

224 template<typename T>
225 bool BufferedFeatureExtractor<T>::checkIsTooLongSegment() {
226  if (utteranceOverflow_)
227  // We had an overflow earlier. This will be reset in resetBuffer() via setEnteredSegment().
228  return true;
229 
230  if (nBufferedFeatures_ < maxBufferSize_)
231  // We have still room.
232  return false;
233 
234  // Can only happen in utterance mode. Otherwise we should have called processBuffer() already.
235  require_eq(bufferType_, utterance);
236 
237  log("Buffer too small for utterance, skip utterance.");
238  utteranceOverflow_ = true;
239  return true;

242 template<typename T>
243 void BufferedFeatureExtractor<T>::processFeature(Core::Ref<const Speech::Feature> f) {
244  if (needInit_)
245  initBuffer(f);
246 
247  // check for buffer overflow
248  if (checkIsTooLongSegment())
249  return;
250  // Fill/update the buffer
251  updateBufferedFeature(f);
252 
253  // Process the buffer?
254  if (nBufferedFeatures_ >= maxBufferSize_) {
255  // process full buffer only in online mode, else overflow error or enlarge buffer
256  switch (bufferType_) {
257  case BufferedFeatureExtractor::minibatch:
258  log("Process buffer since it is full. Processing ") << (nBufferedFeatures_ / batchSize_) << " mini-batches.";
259  processBuffer();
260  break;
261  case BufferedFeatureExtractor::batch:
262  case BufferedFeatureExtractor::utterance:
263  break;
264  }
265  }

268 template<typename T>
269 void BufferedFeatureExtractor<T>::getRelativePositionsInSlidingWindow(u32 indexOfCentralFeature, u32 windowSize, std::vector<s32>& positions) {
270  s32 maxPastSize = windowSize / 2;
271  s32 maxFutureSize = windowSize / 2;
272  // determine segment boundaries
273  s32 leftBoundary = 0;
274  s32 rightBoundary = 0;
275  u32 segmentIndex = segmentIndexBuffer_.at(indexOfCentralFeature);
276  for (s32 i = 0; i < maxFutureSize; i++) {
277  if (indexOfCentralFeature + i + 1 >= nBufferedFeatures_ || segmentIndexBuffer_.at(indexOfCentralFeature + i + 1) != segmentIndex)
278  break;
279  else {
280  rightBoundary++;
281  }
282  }
283 
284  for (s32 i = 0; i > -maxPastSize; i--) {
285  bool atBufferBoundary = (u32)(1 - i) > indexOfCentralFeature;
286  if (atBufferBoundary || segmentIndexBuffer_.at(indexOfCentralFeature + i - 1) != segmentIndex)
287  break;
288  else
289  leftBoundary--;
290  }
291  positions.resize(windowSize);
292  for (s32 i = -maxPastSize; i <= maxFutureSize; i++) {
293  positions.at(i + maxPastSize) = std::min(std::max(i, leftBoundary), rightBoundary);
294  }
295 }

298 void BufferedFeatureExtractor<T>::setWindowedFeature(u32 streamIndex, u32 indexOfCentralFeatureInBuffer, u32 indexInMiniBatch,

323 template<typename T>
324 void BufferedFeatureExtractor<T>::computeFirstDerivative(u32 streamIndex, u32 indexInBuffer) {
325  std::vector<s32> positions;
326  getRelativePositionsInSlidingWindow(indexInBuffer, regressionWindowSize_, positions);
327 
328  // iterate over all feature components
329  for (u32 c = 0; c < featureBuffer_.at(streamIndex).nRows(); c++) {
330  // compute derivative of component c
331  T numerator = 0.0;
332  T denominator = 0.0;
333  for (u32 i = 0; i < regressionWindowSize_; i++) {
334  T pos = i - (regressionWindowSize_ - 1) / 2.0;
335  numerator += pos * featureBuffer_.at(streamIndex).at(c, indexInBuffer + positions.at(i));
336  denominator += pos * pos;
337  }
338  derivativesBuffer_[streamIndex].at(c, indexInBuffer) = numerator / denominator;
339  }
340 }

342 // cf. src/Signal/Regression.cc, derivation of this part unknown, but coefficients are correct!
343 template<typename T>
344 void BufferedFeatureExtractor<T>::computeSecondDerivative(u32 streamIndex, u32 indexInBuffer) {
345  std::vector<s32> positions;
346  getRelativePositionsInSlidingWindow(indexInBuffer, regressionWindowSize_, positions);
347  // index for single second derivative component in derivativesBuffer_
348  u32 derivativeBufferIndex = derivativesBuffer_[streamIndex].nRows() - 1;
349 
350  T numerator = 0.0;
351  T denominator = 0.0;
352  T tm = 0.0;
353  for (u32 i = 0; i < regressionWindowSize_; i++) {
354  T pos = i - (regressionWindowSize_ - 1) / 2.0;
355  tm += pos * pos;
356  denominator += pos * pos * pos * pos;
357  }
358  denominator = tm * tm - regressionWindowSize_ * denominator;
359  for (u32 i = 0; i < regressionWindowSize_; i++) {
360  T pos = i - (regressionWindowSize_ - 1) / 2.0;
361  numerator += tm * featureBuffer_[streamIndex].at(0, indexInBuffer + positions.at(i));
362  numerator -= regressionWindowSize_ * pos * pos * featureBuffer_[streamIndex].at(0, indexInBuffer + positions.at(i));
363  }
364  derivativesBuffer_[streamIndex].at(derivativeBufferIndex, indexInBuffer) = 2.0 * numerator / denominator;
365 }

367 template<typename T>
368 void BufferedFeatureExtractor<T>::generateMiniBatch(std::vector<NnMatrix>& miniBatch, u32 batchSize) {
369  // resize mini batch to number of input streams
370  miniBatch.resize(featureBuffer_.size());
371  // resize each stream to correct size
372  for (u32 streamIndex = 0; streamIndex < miniBatch.size(); streamIndex++) {
373  u32 featureDim = slidingWindowSize_ * featureBuffer_.at(streamIndex).nRows() + slidingWindowSizeDerivatives_ * (featureBuffer_.at(streamIndex).nRows() + 1);
374  miniBatch.at(streamIndex).resize(featureDim, batchSize);
375  }
376  // fill mini batch
377  for (u32 streamIndex = 0; streamIndex < miniBatch.size(); streamIndex++) {
378  miniBatch.at(streamIndex).finishComputation(false);
379  for (u32 column = 0; column < batchSize; column++) {
380  u32 featureIndex = nProcessedFeatures_ + column;
381  if (shuffle_) {
382  verify_lt(featureIndex, shuffledIndices_.size());
383  featureIndex = shuffledIndices_.at(featureIndex);
384  }
385 
386  setWindowedFeature(streamIndex, featureIndex, column, miniBatch.at(streamIndex));
387  if (slidingWindowSizeDerivatives_ > 0)
388  setWindowedFeatureDerivatives(streamIndex, featureIndex, column, miniBatch.at(streamIndex));
389  }
390  }
391 }

393 template<typename T>
394 void BufferedFeatureExtractor<T>::prepareProcessBuffer() {
395  // create trainer, if not yet done
396  if (!trainer_) {
397  trainer_ = createTrainer(config);
398  }
399 
400  processRemainingFeatures_ = trainer_->needsToProcessAllFeatures() || (trainer_->hasEstimator() && trainer_->estimator().fullBatchMode());
401  // shuffle, if desired
402  if (shuffle_ && (trainer_->hasEstimator() && trainer_->estimator().fullBatchMode())) {
403  this->log("do not shuffle buffer, because shuffling is irrelevant with batch estimator");
404  shuffle_ = false;
405  }
406  if (shuffle_)
407  initShuffle();
408 
409  // compute derivatives
410  if ((slidingWindowSizeDerivatives_ > 0) && (!derivativesComputed_)) {
411  for (u32 streamIndex = 0; streamIndex < featureBuffer_.size(); streamIndex++) {
412  // compute first derivative
413  for (u32 indexInBuffer = 0; indexInBuffer < nBufferedFeatures_; indexInBuffer++) {
414  computeFirstDerivative(streamIndex, indexInBuffer);
415  }
416  // compute second derivative
417  for (u32 indexInBuffer = 0; indexInBuffer < nBufferedFeatures_; indexInBuffer++) {
418  computeSecondDerivative(streamIndex, indexInBuffer);
419  }
420  }
421  derivativesComputed_ = true;
422  }
423 }

425 template<typename T>
426 void BufferedFeatureExtractor<T>::processBuffer() {
427  prepareProcessBuffer();
428  std::vector<NnMatrix> miniBatch;
429 
430  while (nProcessedFeatures_ + batchSize_ <= nBufferedFeatures_) {
431  generateMiniBatch(miniBatch, batchSize_);
432  log("Process mini-batch ") << nProcessedMiniBatches_ + 1 << " with " << miniBatch.at(0).nColumns() << " features.";
433  // initialize trainer
434  if (!trainer_->isInitialized()) {
435  std::vector<u32> streamSizes;
436  for (u32 stream = 0; stream < miniBatch.size(); stream++)
437  streamSizes.push_back(miniBatch.at(stream).nRows());
438  trainer_->initializeTrainer(batchSize_, streamSizes);
439  }
440  // process mini batch
441  trainer_->processBatch_feedInput(miniBatch, NULL, getCurSegment());
442  trainer_->processBatch_finish();
443  nProcessedMiniBatches_++;
444  nProcessedFeatures_ += batchSize_;
445  }
446  // process the remaining feature with a smaller mini batch
447  // only done for algorithms where the mini batch size is not critical
448  u32 nRemainingFeatures = nBufferedFeatures_ - nProcessedFeatures_;
449  if (processRemainingFeatures_ && nRemainingFeatures > 0) {
450  generateMiniBatch(miniBatch, nRemainingFeatures);
451  log("Process mini-batch ") << nProcessedMiniBatches_ + 1 << " with " << miniBatch.at(0).nColumns() << " features.";
452  trainer_->setBatchSize(nRemainingFeatures);
453  // initialize trainer
454  if (!trainer_->isInitialized()) {
455  std::vector<u32> streamSizes;
456  for (u32 stream = 0; stream < miniBatch.size(); stream++)
457  streamSizes.push_back(miniBatch.at(stream).nRows());
458  trainer_->initializeTrainer(nRemainingFeatures, streamSizes);
459  }
460  // process mini batch
461  trainer_->processBatch_feedInput(miniBatch, NULL, getCurSegment());
462  trainer_->processBatch_finish();
463  nProcessedMiniBatches_++;
464  nProcessedFeatures_ += nRemainingFeatures;
465  // reset to old batch size
466  trainer_->setBatchSize(batchSize_);
467  }
468  finalizeProcessBuffer();
469 }
