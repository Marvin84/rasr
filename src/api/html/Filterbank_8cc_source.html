44 FilterBank::Filter::Filter(size_t start, size_t end, const std::vector<FilterWeight>& weights)

50 void FilterBank::Filter::normalize(FilterBank::NormalizationType normalizationType) {
51  if (normalizationType != FilterBank::normalizeNone) {
52  FilterWeight normalizationTerm = 1;
53  if (normalizationType == FilterBank::normalizeSurface) {
54  normalizationTerm = std::accumulate(weights_.begin(), weights_.end(), (FilterWeight)0);
55  }
56  else
57  defect();
58  std::transform(weights_.begin(), weights_.end(), weights_.begin(),
59  std::bind2nd(std::divides<FilterWeight>(), normalizationTerm));
60  }
61 }

63 FilterBank::Data FilterBank::Filter::apply(const std::vector<Data>& in) const {
64  require(end_ <= in.size());
65  FilterWeight result = 0;
66  for (size_t f = start_; f < end_; ++f)
67  result += in[f] * weights_[f - start_];
68  return result;
69 }

80 /** Base class for filter builders
81  * Override the following functions to implement a specific filter:
82  * 1) normalizedCenterPosition: return the position of the center of the filter in percent.
83  * -0 -> filter has no left flank.
84  * -1 -> filter has no right flank.
85  * 2) weight(frequency): returns weight for the given frequency.
86  */
87 class FilterBank::FilterBuilder : public virtual Core::Component {
88 protected:

142 Core::Ref<FilterBank::Filter> FilterBank::FilterBuilder::create(Frequency center,
143  Frequency width,
144  Frequency minimumFrequency,
145  Frequency maximumFrequency,
146  Math::UnaryAnalyticFunctionRef discreteToContinuousFunction,
147  Math::UnaryAnalyticFunctionRef warpingFunction,
148  bool shouldWarpDifferentialUnit) {
149  Core::Ref<FilterBank::Filter> result;
150  center_ = center;
151  width_ = width;
152  minimumFrequency_ = minimumFrequency;
153  maximumFrequency_ = maximumFrequency;
154 
155  discreteToContinuousFunction_ = Math::nest(warpingFunction, discreteToContinuousFunction);
156  continuousToDiscreteFunction_ = discreteToContinuousFunction_->invert();
157  if (continuousToDiscreteFunction_) {
158  if (shouldWarpDifferentialUnit) {
159  if (warpingFunction->derive())
160  derivedWarpingFunction_ = Math::nest(warpingFunction->derive(), discreteToContinuousFunction);
161  }
162  else {
163  derivedWarpingFunction_ = Math::AnalyticFunctionFactory::createConstant(1);
164  }
165  if (derivedWarpingFunction_) {
166  if (setStart() && setEnd() && setWeights())
167  result = Core::ref(new Filter(start_, end_, weights_));
168  }
169  else {
170  error("Warping function is not derivable.");
171  }
172  }
173  else {
174  error("Warping function is not invertible.");
175  }
176  return result;
177 }

179 bool FilterBank::FilterBuilder::setStart() {
180  Frequency start = continuousToDiscreteFunction_->value(std::max(center_ - normalizedCenterPosition() * width_, minimumFrequency_));
181  if (isAlmostInteger(start))
182  start = Core::round(start);
183  else
184  start = Core::ceil(start);
185  if (start >= 0) {
186  start_ = (size_t)start;
187  return true;
188  }
189  error("Start point of the filter at center %f became negative (%d).", center_, (int)start);
190  return false;
191 }
192 
193 bool FilterBank::FilterBuilder::setEnd() {
194  Frequency end = continuousToDiscreteFunction_->value(std::min(center_ + (1.0 - normalizedCenterPosition()) * width_, maximumFrequency_));
195  if (isAlmostInteger(end))
196  end = Core::round(end) + 1;
197  else
198  end = Core::ceil(end);
199  if (end > 0 && start_ < (size_t)end) {
200  end_ = (size_t)end;
201  return true;
202  }
203  error("Inconsistent end point of the filter at center %f: start=%zd end=%d.", center_, start_, (int)end);
204  return false;
205 }
206 
207 bool FilterBank::FilterBuilder::setWeights() {
208  weights_.resize(end_ - start_);
209  for (u32 f = start_; f < end_; ++f) {
210  verify(derivedWarpingFunction_->value(f) >= 0);
211  weights_[f - start_] = weight(discreteToContinuousFunction_->value(f)) * derivedWarpingFunction_->value(f);
212  verify(weights_[f - start_] >= 0);
213  }
214  return true;
215 }

233 FilterBank::FilterWeight SymmetricalTriangularFilterBuilder::weight(FilterBank::Frequency frequency) const {
234  FilterBank::FilterWeight result = (FilterBank::Frequency)1 - Core::abs(frequency - center_) / (width_ / 2);
235  /* start_ and end_ control that 'frequency' stays within the interval [center-halfWidth..center+halfWidth],
236  * thus the resulting weight has to be larger or equal to zero. Nevertheless rounding error can yield
237  * slightly negative weight which will be clipped to zero.
238  */
239  ensure(result > -Core::Type<FilterBank::FilterWeight>::epsilon);
240  return result >= 0 ? result : 0;
241 }

264 FilterBank::FilterWeight TrapezeFilterBuilder::weight(FilterBank::Frequency frequency) const {
265  FilterBank::Frequency relativeFrequency = frequency - center_;
266 
267  FilterBank::Frequency middleLeftBorder = -normalizedMiddleBorder() * width_;
268  if (relativeFrequency < middleLeftBorder)
269  return pow(10, relativeFrequency - middleLeftBorder);
270  else {
271  FilterBank::Frequency middleRightBorder = normalizedMiddleBorder() * width_;
272  if (relativeFrequency <= middleRightBorder)
273  return 1;
274  else
275  return pow(10, -2.5 * (relativeFrequency - middleRightBorder));
276  }
277 }

305 FilterBank::FilterWeight TrapezeRastaFilterBuilder::weight(FilterBank::Frequency frequency) const {
306  FilterBank::Frequency relativeFrequency = frequency - center_;
307  FilterBank::Frequency middleBorder = normalizedMiddleBorder() * width_; // => 0.5
308 
309  if (relativeFrequency <= -middleBorder)
310  return pow(10, relativeFrequency + middleBorder);
311  else if (relativeFrequency >= middleBorder)
312  return pow(10, -2.5 * (relativeFrequency - middleBorder));
313  else
314  return 1;
315 }

467 /**
468  * Include Boundary: left and right boundaries of input are included but with small weights.
469  * -First filter starts at center equals spacing.
470  * -Last filter is chosen such that its left border is smaller than maximal-frequency and
471  * its right border is larger or equal to the maximal-frequency.
472  * =>Number of filters = ceil(maximum-frequency - (1.0 - normalized-center) * filter-width) / spacing)
473  * -Sample rate of output vector: 1.0 / spacing.
474  * -Note that if spacing > normalized-center * filter-width, the leftmost few input values will
475  * not contribute to the results.
476  */
477 class IncludeBoundary : public FilterBank::Boundary {
478  typedef FilterBank::Boundary Precursor;
479 
480 public:
481  IncludeBoundary(const Core::Configuration& c)
482  : Core::Component(c), FilterBank::Boundary(c) {}
483  virtual Frequency center(size_t filterIndex) const {
484  return warpingFunction_->value(spacing_ * (filterIndex + 1));
485  }
486  virtual size_t getNumberOfFilters() const;
487 };

497 /**
498  * Stretch-To-Cover: filter width and spacing are stretched to cover the complete input exactly,
499  * thus left and right boundaries of input are included with weight 0.
500  * -Basic idea to ensure the complete coverage of the whole interval [0..maximal-frequency]
501  * with minimal number of filters.
502  * -First filter starts at left-border equals 0.
503  * -Last filter is chosen such that its right border is smaller or equal to maximal-frequency and
504  * the right-border of the next filter would be larger than maximal-frequency.
505  * =>Number of filters = floor((maximal-frequency - filter-width) / spacing + 1)
506  * -Finally, spacing and filter width are multiplied by a number > 1 such that right border of the
507  * last filter falls on the maximal-frequency.
508  * -Sample rate of output vector: since spacing is changed according to the input length,
509  * the output sample-rate cannot be retrieved before the first
510  * input arrives, thus it will be set to 1.
511  * -Pre-warping of centers is supported in this boundary type since pre-warping introduces non-linearities.
512  */
513 class StretchToCover : public FilterBank::Boundary {
514  typedef FilterBank::Boundary Precursor;
515 
516 public:
517  StretchToCover(const Core::Configuration& c)
518  : Core::Component(c), FilterBank::Boundary(c) {}
519 
520  virtual bool init(Frequency filterWidth,
521  Frequency spacing,
522  Frequency normalizedCenterPosition,
523  Frequency minimumFrequency,
524  Frequency maximumFrequency,
525  Math::UnaryAnalyticFunctionRef warpingFunction);
526 
527  virtual Frequency center(size_t filterIndex) const {
528  return minimumFrequency_ + spacing_ * filterIndex + normalizedCenterPosition_ * filterWidth_;
529  }
530  virtual size_t getNumberOfFilters() const {
531  return (size_t)Core::floor(postprocessNumberOfFilters(
532  (maximumFrequency_ - minimumFrequency_ - filterWidth_) / spacing_ + 1));
533  }
534  virtual Frequency outputSampleRate() const {
535  return (Frequency)1;
536  }
537 };
538 
539 bool StretchToCover::init(Frequency filterWidth,
540  Frequency spacing,
541  Frequency normalizedCenterPosition,
542  Frequency minimumFrequency,
543  Frequency maximumFrequency,
544  Math::UnaryAnalyticFunctionRef warpingFunction) {
545  if (warpingFunction) // unwarping of center not possible in this boundary type.
546  return false;
547 
548  Precursor::init(filterWidth, spacing, normalizedCenterPosition,
549  minimumFrequency, maximumFrequency, warpingFunction);
550 
551  size_t nFilters = getNumberOfFilters();
552  Frequency coverage = (spacing_ * (nFilters - 1) + filterWidth_) / (maximumFrequency_ - minimumFrequency_);
553  if (nFilters == 1 && Core::isSignificantlyGreater(coverage, 1))
554  return true; // If single filter ensures full coverage, then do not shrink it.
555 
556  require(Core::isAlmostEqual(coverage, 1) || coverage < 1); // If fails then nFilters was too high
557  filterWidth_ /= coverage;
558  spacing_ /= coverage;
559  return true;
560 }

639 bool FilterBank::init() {
640  if (!builder_ || !boundary_ || (maximumFrequency_ == 0) ||
641  !boundary_->init(filterWidth_,
642  spacing_,
643  builder_->normalizedCenterPosition(),
644  minimumFrequency_,
645  maximumFrequency_,
646  shouldWarpCenterPositions_ ? Math::UnaryAnalyticFunctionRef() : warpingFunction_)) {
647  return false;
648  }
649  filters_.resize(boundary_->getNumberOfFilters());
650  for (size_t i = 0; i < filters_.size(); ++i) {
651  filters_[i] = builder_->create(boundary_->center(i),
652  boundary_->filterWidth(),
653  minimumFrequency_,
654  maximumFrequency_,
655  discreteToContinuousFunction_,
656  warpingFunction_, shouldWarpDifferentialUnit());
657  if (!filters_[i])
658  return false;
659  filters_[i]->normalize(normalizationType_);
660  }
661  return !(needInit_ = false);
662 }

763 void FilterBankNode::init(u32 inputSize) {
764  if (inputSize == 0)
765  error("Empty input vector.");
766  inputSize_ = inputSize;
767  if (sampleRate_ <= 0)
768  error("Sample rate is not set.");
769  respondToDelayedErrors();
770  Math::UnaryAnalyticFunctionRef discreteToContinuousFunction;
771  Math::UnaryAnalyticFunctionRef warpingFunction;
772  createAnalyticFunction(discreteToContinuousFunction, warpingFunction);
773  respondToDelayedErrors();
774 
775  f64 maximumFrequency = Math::nest(warpingFunction, discreteToContinuousFunction)->value(inputSize_ - 1);
776  if (filteringInterval_ < Core::Type<f32>::max) {
777  if (maximumFrequency < filteringInterval_) {
778  error("Filter interval (%f) is broader than the the input vector (%f). ",
779  filteringInterval_, maximumFrequency);
780  }
781  else
782  maximumFrequency = filteringInterval_;
783  }
784 
785  setDiscreteToContinuousFunction(discreteToContinuousFunction);
786  setWarpingFunction(StringExpressionNode::value(), warpingFunction);
787  setMinimumFrequency(filteringIntervalStart_);
788  setMaximumFrequency(maximumFrequency);
789  if (!isConfigurationAllowed())
790  error("This configuration of the filter bank is not allowed.");
791  respondToDelayedErrors();
792  if (dumpChannel_.isOpen())
793  dump(dumpChannel_);
794  needInit_ = false;
795 }

797 void FilterBankNode::createAnalyticFunction(Math::UnaryAnalyticFunctionRef& discreteToContinuousFunction,
798  Math::UnaryAnalyticFunctionRef& warpingFunction) {
799  verify(inputSize_ > 0);
800 
801  Math::AnalyticFunctionFactory factory(select(paramWarpingFunction.name()));
802  factory.setSampleRate(sampleRate_);
803  factory.setDomainType(Math::AnalyticFunctionFactory::continuousDomain);
804 
805  // Discrete to continuous
806  discreteToContinuousFunction = factory.createScaling(1 / sampleRate_);
807  factory.setMaximalArgument(discreteToContinuousFunction->value(inputSize_ - 1));
808 
809  // Continuous to warped continuous
810  warpingFunction = factory.createIdentity();
811  std::string declaration = StringExpressionNode::value();
812  if (!declaration.empty()) {
813  warpingFunction = factory.createUnaryFunction(declaration);
814  if (!warpingFunction)
815  error("Failed to create warping function.");
816  }
817 }
