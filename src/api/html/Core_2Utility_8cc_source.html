24 int Core::getline(std::istream& is, std::string& str, std::string delim) {
25  int token;
26  std::string::size_type pos = std::string::npos;
27 
28  // check if end of stream is reached
29  if (is.get() == EOF)
30  return EOF;
31  is.unget();
32 
33  // tokenize stream contents
34  str = "";
35  while (((token = is.get()) != EOF) &&
36  ((pos = delim.find(token)) == std::string::npos)) {
37  str += char(token);
38  }
39 
40  if (pos == std::string::npos)
41  return 0;
42 
43  return pos + 1;
44 }
