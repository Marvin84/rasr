34 /**
35  * abstraction of an allophone HMM.
36  *
37  * a state sequence consists of
38  * - a sequence of states (state: emission model index, transition model index)
39  * - information about the word boundary of the allphone (initial, final)
40  */
41 class StateSequence {
42 private:
43  struct State {
44  Am::AcousticModel::EmissionIndex emission_;
45  Am::AcousticModel::StateTransitionIndex transition_;
46 
47  State(Am::AcousticModel::EmissionIndex e, Am::AcousticModel::StateTransitionIndex t)
48  : emission_(e), transition_(t) {
49  }
50  State()
51  : emission_(0), transition_(0) {
52  }
53  bool operator==(const State& o) const {
54  return (emission_ == o.emission_ && transition_ == o.transition_);
55  }
56  };
57  std::vector<State> states_;
58  u8 flags_;
59 
60 public:
61  StateSequence(u8 flags = 0)
62  : flags_(flags) {
63  }
64 
65  void appendState(Am::AcousticModel::EmissionIndex emission, Am::AcousticModel::StateTransitionIndex transition) {
66  states_.push_back(State(emission, transition));
67  }
68 
69  size_t nStates() const {
70  return states_.size();
71  }
72  const State& state(u32 s) const {
73  return states_[s];
74  }
75 
76  u8 flags() const {
77  return flags_;
78  }
79  void setFlags(u8 flags) {
80  flags_ = flags;
81  }
82  void addFlag(u8 flag) {
83  flags_ |= flag;
84  }
85  void setFinal() {
86  flags_ |= Am::Allophone::isFinalPhone;
87  }
88  void setInitial() {
89  flags_ |= Am::Allophone::isInitialPhone;
90  }
91  bool isInitial() const {
92  return flags_ & Am::Allophone::isInitialPhone;
93  }
94  bool isFinal() const {
95  return flags_ & Am::Allophone::isFinalPhone;
96  }
97 
98  bool operator==(const StateSequence& o) const {
99  return states_ == o.states_ && flags_ == o.flags_;
100  }
101  bool read(Core::BinaryInputStream& i);
102  bool write(Core::BinaryOutputStream& o) const;

104  struct Hash {
105  size_t operator()(const StateSequence& ss) const {
106  size_t key = 0;
107  for (std::vector<State>::const_iterator s = ss.states_.begin(); s != ss.states_.end(); ++s)
108  key = ((key << 7) | (key >> 25)) ^ (u32(s->emission_) | (u32(s->transition_) << 14));
109  return key;
110  }
111  };

113  struct IgnoreFlagsEqual {
114  bool operator()(const StateSequence& a, const StateSequence& b) const {
115  return a.states_ == b.states_;
116  }
117  };

119  size_t memoryUsage() const {
120  return states_.capacity() * sizeof(State);
121  }
