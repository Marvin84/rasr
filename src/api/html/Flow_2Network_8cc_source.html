137 bool Network::addLink(const std::string& fromNodeName, const std::string& fromPortName,
138  const std::string& toNodeName, const std::string& toPortName,
139  u32 buffer) {
140  Link* l = new Link();
141  bool result = true;
142  // connect to network
143  if (toNodeName == typeName_) {
144  if (fromNodeName == typeName_) {
145  if (!connectOutputFromInside(inputRepeaterPrefix + fromPortName, "data", toPortName, l))
146  result = false;
147  }
148  else {
149  if (!connectOutputFromInside(fromNodeName, fromPortName, toPortName, l))
150  result = false;
151  }
152  }
153  else {
154  // connect 'from' node
155  if (fromNodeName == typeName_) {
156  if (!connectInputFromInside(fromPortName, l, buffer))
157  result = false;
158  }
159  else {
160  if (!connectNodeOutput(fromNodeName, fromPortName, l))
161  result = false;
162  }
163  // connect 'to' node
164  if (!connectNodeInput(toNodeName, toPortName, l))
165  result = false;
166  }
167 
168  if (!result) {
169  delete l;
170  return result;
171  }
172 
173  l->setBuffer(buffer);
174  // save names of the connections for the dump
175  l->setNodeNames(fromNodeName, fromPortName, toNodeName, toPortName);
176  links_.push_back(l);
177 
178  return true;
179 }

183 bool Network::connectInputFromInside(const std::string& portName, Link* l, u32 buffer) {
184  PortId portId = getInput(portName);
185  if (portId == IllegalPortId) {
186  error("could not find input '%s' in network.", portName.c_str());
187  return false;
188  }
189  Port& port = inputs_[portId];
190 
191  AbstractNode* repeater = port.node();
192  verify(repeater != 0);
193  if (repeater->connectOutputPort(port.nodePort(), l) == IllegalPortId) {
194  error("could not connect input '%s' in network.", portName.c_str());
195  return false;
196  }
197 
198  // create link for putData
199  if (port.link() == 0) {
200  Link* portLink = new Link;
201  portLink->setBuffer(buffer);
202  port.setLink(portLink);
203  links_.push_back(portLink);
204  connectInputPortLink(portId);
205  }
206  return true;
207 }

211 bool Network::connectOutputFromInside(const std::string& fromNodeName,
212  const std::string& fromPortName,
213  const std::string& toPortName,
214  Link* l) {
215  PortId toPortId = getOutput(toPortName);
216  if (toPortId == IllegalPortId) {
217  error("could not find output '%s' in network.", toPortName.c_str());
218  return false;
219  }
220 
221  Port& port = outputs_[toPortId];
222  AbstractNode* fromNode = getNode(fromNodeName);
223  if (fromNode == 0) {
224  error("could not find node '%s' in network.", fromNodeName.c_str());
225  return false;
226  }
227 
228  PortId fromPort = fromNode->getOutput(fromPortName);
229  if (fromPort == IllegalPortId) {
230  error("could not connect output '%s' at node '%s'.",
231  fromPortName.c_str(), fromNodeName.c_str());
232  return false;
233  }
234 
235  if (port.node()) {
236  error("Output port '%s' has already been connected.", toPortName.c_str());
237  return false;
238  }
239  port.setNode(fromNode, fromPort);
240  port.setLink(l);
241  connectOutputPortLink(toPortId);
242 
243  return true;
244 }

359  ensure(out.linkConnected());

361  AbstractNode* lastNode = out.link()->getFromNode();

698 void Network::go() {
699  /*
700  for (list<AbstractNode*>::const_iterator n = nodes_.begin(); n != nodes_.end(); n++) {
701  (*n)->Start();
702  }
703  */
704  // gather sinks
705  std::list<AbstractNode*> sinks;
706  for (std::list<AbstractNode*>::const_iterator n = nodes_.begin(); n != nodes_.end(); n++) {
707  if (((*n)->nOutputs() == 0) && (!(*n)->isThreaded()))
708  sinks.push_back(*n);
709  }
710 
711  // loop over all sinks
712  while (sinks.size()) {
713  for (std::list<AbstractNode*>::iterator n = sinks.begin(); n != sinks.end(); n++) {
714  if (!(*n)->work(0)) {
715  sinks.erase(n);
716  break;
717  }
718  }
719  }

724 bool Network::dump(const bool initialCall, Core::XmlChannel& dumpChannel, std::set<std::string>* dumped) {
725  // <network> is the root element
726  if (initialCall)
727  dumpChannel << Core::XmlOpen("network") + Core::XmlAttribute("name", getTypeName());
728 
729  // remember the filenames to avoid multiple inclusions
730  if (!dumped)
731  dumped = new std::set<std::string>;
732 
733  // dump all nodes in the network which include extern flow-networks
734  for (std::list<AbstractNode*>::const_iterator it = nodes_.begin();
735  it != nodes_.end(); it++) {
736  if (Network* n = dynamic_cast<Network*>(*it)) {
737  // dump only if not dumped before
738  std::set<std::string>::const_iterator it_dumped = dumped->find(n->filename());
739  if (it_dumped == dumped->end()) {
740  n->dump(false, dumpChannel, dumped);
741  dumped->insert(n->filename());
742  }
743  }
744  }
745 
746  // inner networks are stored in <network-node> tags
747  if (!initialCall)
748  dumpChannel << Core::XmlOpen("network-node") + Core::XmlAttribute("filter", filename()) + Core::XmlAttribute("name", getTypeName());
749 
750  // dump <param>
751  for (std::list<Network::Parameter>::const_iterator it = params_.begin();
752  it != params_.end(); it++) {
753  dumpChannel << Core::XmlEmpty("param") + Core::XmlAttribute("name", it->name());
754  }
755 
756  // dump <in>
757  for (std::vector<Port>::const_iterator inputPort = inputs_.begin();
758  inputPort != inputs_.end(); ++inputPort) {
