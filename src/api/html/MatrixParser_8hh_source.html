78 template<class T>
79 void XmlMatrixElement<T>::start(const XmlAttributes atts) {
80  Predecessor::start(atts);
81 
82  buffer_ = std::string();
83  const char* nRows = atts["nRows"];
84 
85  if (nRows == 0)
86  this->parser()->error("Number of rows attribute (nRows) not given.");
87 
88  const char* nColumns = atts["nColumns"];
89 
90  if (nColumns == 0)
91  this->parser()->error("Number of columns attribute (nColumns) is not given.");
92 
93  Predecessor::product_->resize(atoi(nRows), atoi(nColumns));
94 }
95 
96 template<class T>
97 void XmlMatrixElement<T>::end() {
98  std::istringstream stream(buffer_);
99 
100  for (u32 row = 0; row < Predecessor::product_->nRows(); ++row) {
101  for (u32 column = 0; column < Predecessor::product_->nColumns(); ++column) {
102  stream >> (*Predecessor::product_)[row][column];
103  if (stream.eof()) {
104  this->parser()->error("Number of elements smaller than necesseary.");
105  Predecessor::end();
106  return;
107  }
108  }
109  }
110 
111  T tmp;
112  if (!(stream >> tmp).eof())
113  this->parser()->error("Number of elements larger than necesseary.");
114 
115  Predecessor::end();
116 }
