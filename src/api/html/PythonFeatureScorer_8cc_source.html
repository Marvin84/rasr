19 /*
20  * PythonFeatureScorer uses a Python interface to get the scores.
21  * The scores will we forwarded as-is. Negative log scores are expected, i.e. something like -log p(x|s).
22  * So if you have a NN with posteriors, you need to divide by the prior on the Python side.
23  *
24  * For an RNN, esp a bidirectional one, we need to first get all the features before we can
25  * forward it and get the scores. That is important for the design of this class.
26  *
27  * Also, this class supports buffering (isBuffered() == true), so both the
28  * addFeature()/flush() but also the (traditionally non-buffered) getScorer()
29  * has to be supported.
30  * getScorer() is used e.g. in the aligner.
31  * addFeature()/flush() is used by the recognizer.
32  *
33  * Python interface:
34  *
35  * PythonControl is used. See its documentation. In short:
36  * init(sprint_unit='PythonFeatureScorer', ...) from the Python module will get called and is expected to return some object.
37  * Member functions of this object will get called for the further communication.
38  *
39  * Member functions which will get called:
40  *
41  * init(input_dim: int, output_dim: int) # output-dim is the number of emission classes for us
42  * get_feature_buffer_size() # expected to return -1 for now
43  * add_feature(feature: numpy.ndarray, time: int) # feature is of shape (input_dim,)
44  * reset(num_frames: int) # signals that we can flush any buffers
45  * compute(num_frames: int) # all the features which we received so far should be evaluated
46  * get_scores(time: int) # expected to return a numpy.ndarray of shape (output_dim,)
47  *
48  */

58 PythonFeatureScorer::PythonFeatureScorer(const Core::Configuration& config, Core::Ref<const Mm::MixtureSet> mixtureSet)
59  : Core::Component(config),
60  Mm::FeatureScorer(config),
61  featureBufferSize_(0),
62  numFeaturesReceived_(0),
63  currentFeature_(0),
64  scoresComputed_(false),
65  scoresCachePosition_(u32(-1)),
66  nClasses_(mixtureSet->nMixtures()),
67  inputDimension_(paramFeatureDimension(config)),
68  batchIteration_(0),
69  pythonControl_(config, /* sprintUnit */ "PythonFeatureScorer", /* isOptional */ false) {
70  int outputDim = paramOutputDimension(config);
71  if (outputDim >= 0) {
72  log("PythonFeatureScorer: will ignore mixture-set number of classes %i but use %i instead", nClasses_, outputDim);
73  nClasses_ = outputDim;
74  }
75  log("PythonFeatureScorer: initialize with feature dimension %i, number of classes %i", inputDimension_, nClasses_);
76  require_gt(inputDimension_, 0);
77 
78  Python::ScopedGIL gil;
79  pythonControl_.run_custom("init", "{s:i,s:i}", "input_dim", (int)inputDimension_, "output_dim", (int)nClasses_);
80  {
81  PyObject* res = pythonControl_.run_custom_with_result("get_feature_buffer_size", "{}");
82  if (res) {
83  long resLong = PyLong_AsLong(res);
84  if ((resLong == -1) && PyErr_Occurred())
85  pythonControl_.pythonCriticalError("PythonFeatureScorer: get_feature_buffer_size did not return an integer");
86  featureBufferSize_ = (u32)resLong;
87  if (featureBufferSize_ == 0)
88  criticalError("PythonFeatureScorer: get_feature_buffer_size returned 0");
89  Py_CLEAR(res);
90  }
91  }

125 void PythonFeatureScorer::_addFeature(const Mm::FeatureVector& f) const {
126  require(!bufferFilled());
127  require(!scoresComputed_);
128  if (inputDimension_ != f.size()) {
129  criticalError("PythonFeatureScorer: was configured with input dimension %i but we got features with dimension %zu",
130  inputDimension_, f.size());
131  }
132 
133  Python::ScopedGIL gil;
134  PyObject* numpyArray = NULL;
135  if (!Python::stdVec2numpy(pythonControl_.getPythonCriticalErrorFunc(), numpyArray, f))
136  return;
137  pythonControl_.run_custom("add_feature", "{s:O,s:i}", "feature", numpyArray, "time", (int)numFeaturesReceived_);
138  numFeaturesReceived_++;
139 }

174 Mm::FeatureScorer::Scorer PythonFeatureScorer::flush() const {
175  require_lt(currentFeature_, numFeaturesReceived_);
176  Scorer scorer(new ContextScorer(this, currentFeature_, batchIteration_));
177  currentFeature_++;
178  // We must not call reset() here because the calls to getScore() will be delayed.

182 Mm::Score PythonFeatureScorer::getScore(Mm::EmissionIndex e, u32 position) const {
183  require_lt(position, numFeaturesReceived_);
184  require_lt(e, nClasses_);
185  // process buffer if needed
186  if (!scoresComputed_) {
187  pythonControl_.run_custom("compute", "{s:i}", "num_frames", (int)numFeaturesReceived_);
188  // mark computed
189  scoresComputed_ = true;
190  }
191  if (scoresCachePosition_ != position) {
192  Python::ScopedGIL gil;
193  PyObject* res = pythonControl_.run_custom_with_result("get_scores", "{s:i}", "time", (int)position);
194  if (!res)
195  return 0.0;
196  if (!Python::numpy2stdVec(pythonControl_.getPythonCriticalErrorFunc(), res, scoresCache_))
197  return 0.0;
198  Py_CLEAR(res);
199  if (scoresCache_.size() != nClasses_) {
200  criticalError("PythonFeatureScorer: get_scores returned vector of len %i but we expected len (num classes) %i",
201  (int)scoresCache_.size(), (int)nClasses_);
202  return 0.0;
203  }
204  scoresCachePosition_ = position;
205  }
206  // return score in -log space
207  return scoresCache_[e];
208 }
