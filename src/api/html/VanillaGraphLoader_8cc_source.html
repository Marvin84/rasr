26 std::unique_ptr<Graph> VanillaGraphLoader::load_graph() {
27  if (file_.empty()) {
28  criticalError("no graph-def-path set");
29  }
30 
31  tf::Env* env = tf::Env::Default();
32  tf::GraphDef graph_def;
33  tf::Status status = ReadBinaryProto(env, file_, &graph_def);
34  if (not status.ok()) {
35  criticalError("error reading graph def %s", status.ToString().c_str());
36  }
37 
38  std::unique_ptr<Graph> result(new Graph());
39  setGraphDef(*result, graph_def);
40 
41  for (int i = 0; i < graph_def.node_size(); i++) {
42  auto& node = graph_def.node(i);
43  if (node.op() == "Placeholder" or node.op() == "PlaceholderV2") {
44  result->addInput(node.name());
45  }
46  else if (node.op() == "Variable" or node.op() == "VariableV2") {
47  DataType dt = node.attr().find("dtype")->second.type();
48  std::string var_name = node.name() + ":0"; // to use the same name as is used inside collections we append ":0" here
49  auto const& shape = node.attr().find("_output_shapes")->second.list().shape(0);
50  std::vector<s64> dims(static_cast<size_t>(shape.dim_size()));
51  for (int i = 0; i < shape.dim_size(); i++) {
52  dims[i] = shape.dim(i).size();
53  }
54  result->addVariable({var_name, "", "", "", dt, dims});
55  }
56  }
57 
58  return result;
59 }
