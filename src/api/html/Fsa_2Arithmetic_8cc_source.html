84  : ModifyAutomaton(f1), fsa2_(f2) {

89  virtual void modifyState(State* sp) const {
90  ConstStateRef _sp2 = fsa2_->getState(sp->id());
91  require(sp->id() == _sp2->id() && sp->tags() == _sp2->tags());
92  require(sp->nArcs() == _sp2->nArcs());
93  if (sp->isFinal())
94  sp->weight_ = fsa_->semiring()->extend(sp->weight_, _sp2->weight_);
95  State::const_iterator a2 = _sp2->begin();
96  for (State::iterator a = sp->begin(); a != sp->end(); ++a, ++a2) {
97  require(a->target() == a2->target());
98  require(a->input() == a2->input() && a->output() == a2->output());
99  a->weight_ = fsa_->semiring()->extend(a->weight(), a2->weight());
100  }
101  }

103  virtual std::string describe() const {
104  return Core::form("extend-by-fsa(%s,%s)",
105  fsa_->describe().c_str(), fsa2_->describe().c_str());
106  }
