113 void Mixture::removeDensitiesWithLowWeight(Weight minWeight, bool normalizeWeights) {
114  const Weight _minWeight = exp(logExpNorm(logWeights_.begin(), logWeights_.end())) * minWeight;
115  DensityIndex dnsMax = densityIndexWithMaxWeight();
116  for (DensityIndex dnsInMix = 0; dnsInMix < nDensities();) {
117  if ((weight(dnsInMix) < _minWeight) and (dnsInMix != dnsMax)) {
118  removeDensity(dnsInMix);
119  if (dnsMax > dnsInMix)
120  --dnsMax;
121  }
122  else {
123  ++dnsInMix;
124  }
125  }

127  this->normalizeWeights();
