120 template<class I, class O>
121 bool CollectorNode<I, O>::putNullData() {
122  if (checkSameLength_) {
123  typename std::vector<InputFrame>::iterator hasData =
124  std::find_if(inputData_.begin(), inputData_.end(),
125  std::bind2nd(std::equal_to<bool>(), true));
126 
127  if (hasData != inputData_.end())
128  criticalError("Input streams have different length.");
129  }
130 
131  typename std::vector<InputFrame>::iterator noData =
132  std::find_if(inputData_.begin(), inputData_.end(),
133  std::bind2nd(std::equal_to<bool>(), false));
134 
135  if (noData == inputData_.end())
136  return putEos(0);
137  return putData(0, noData->get());
138 }

140 template<class I, class O>
141 bool CollectorNode<I, O>::getData(Time& startTime, Time& endTime) {
142  startTime = Core::Type<Time>::max;
143  endTime = Core::Type<Time>::min;
144 
145  bool result = true;
146  for (PortId i = 0; i < (PortId)inputData_.size(); i++) {
147  InputFrame& in(inputData_[i]);
148  if (!Precursor::getData(i, in)) {
149  result = false;
150  continue;
151  }
152 
153  if (timestampPortId_ == IllegalPortId) {
154  if (startTime > in->getStartTime())
155  startTime = in->getStartTime();
156  if (endTime < in->getEndTime())
157  endTime = in->getEndTime();
158  }
159  else if (timestampPortId_ == i) {
160  startTime = in->startTime();
161  endTime = in->endTime();
162  }
163  }
164  return result;
165 }
