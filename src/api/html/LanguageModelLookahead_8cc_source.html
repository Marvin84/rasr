26 /*
27  * To enhance opportunities for future reuse the look-ahead tree
28  * structure is independent from the HMM state-tree structure.
29  *
30  * Memory consumption of LM look-ahead is determined by two
31  * parameters: The size of the look-ahead tables, and the number of
32  * tables kept im memory.
33  *
34  * The look-ahead table size is determined by the level of detail
35  * reflected in the look-ahead structure. This can be controled the
36  * parameters tree-cutoff and minimum-representation. By default the
37  * look-ahead tables are not pruned, so memory consumption is
38  * (unnecessarily) large.
39  *
40  * The number of tables kept in memory is affected by history-limit,
41  * cache-size-low and cache-size-high.
42  *
43  * By default tables are deleted as soon as they are not immediately
44  * needed (i.e. all relevant trees are pruned). We say, the table
45  * becomes inactive. This saves memory but wastes computation time.
46  * The caching strategy works as follows: As long as there are less
47  * than cache-size-high tables in memory, no table is deleted. Above
48  * that limit the table that has least recently become inactive is
49  * freed whenwever a table is release. Note that cache-size-high is
50  * not a strict upper bound on the number of tables. It will be
51  * exceeded when the search process requires it.
52  *
53  * When a new table is requested, usually one of the inactive tables
54  * is re-used. But when there are less than cache-size-low tables in
55  * memory, a new table is created instead. cache-size-low is thus a
56  * (soft) lower bound on the number of tables in memory. (Of course
57  * there may be less tables, if the search never requires that many
58  * different contexts.)
59  *
60  * Tree cutoff: In the full look-ahead structure non-branching state
61  * sequences are represented by a single node. The MINIMUM depth of
62  * these states is used as pruning criterion: If it is larger than
63  * "tree-cutoff", the look-ahead node is merged with its parent.
64  * (This minimum criterion is believed to best reproduce the behavior
65  * of the old standard system. However, other criteria might be
66  * better.)
67  *
68  * TODO:
69  * Optimizations to be considered:
70  * - Different LA table cut-off criteria (may-be)
71  * - Lazy-evaluation of LA tables (may-be; partially done by "lazy lookahead")
72  * - Tune LM scale for look-ahead (questionable)
73  */

149 class LanguageModelLookahead::ConstructionNode {
150 public:
151  LookaheadId id;
152  struct {
153  StateTree::Depth min, max;
154  } depth;
155  typedef std::vector<ConstructionNode*> Successors;
156  typedef std::vector<StateTree::StateId> Represents;
157  Represents represents;
158 
159 private:
160  Ends ends_;
161  Successors successors_;
162 
163  enum Consolidation { dirty,
164  unique,
165  domineesValid,
166  hashValid };
167  mutable Consolidation consolidation_;
168  mutable Ends dominees_;
169  mutable u32 hash_;
170 
171 public:
172  bool isUnique() const {
173  return consolidation_ >= unique;
174  }
175 
176  void makeUnique() {
177  std::sort(ends_.begin(), ends_.end());
178  ends_.erase(std::unique(ends_.begin(), ends_.end()), ends_.end());
179  std::sort(successors_.begin(), successors_.end());
180  successors_.erase(std::unique(successors_.begin(), successors_.end()), successors_.end());
181  consolidation_ = unique;
182  }
183 
184 private:
185  void updateDominiees() const {
186  require(consolidation_ >= unique);
187  dominees_ = ends_;
188  Ends curr;
189  for (Successors::const_iterator s = successors_.begin(); s != successors_.end(); ++s) {
190  dominees_.swap(curr);
191  dominees_.clear();
192  const Ends& succ((*s)->dominees());
193  std::set_union(curr.begin(), curr.end(),
194  succ.begin(), succ.end(),
195  std::back_inserter(dominees_));
196  }
197  consolidation_ = domineesValid;
198  }
199 
200  void updateHash() const {
201  require(consolidation_ >= domineesValid);
202  hash_ = 0;
203  for (Ends::const_iterator e = dominees_.begin(); e != dominees_.end(); ++e)
204  hash_ = ((hash_ << 3) | (hash_ >> 29)) ^ u32(long(*e /* this is a pointer */));
205  consolidation_ = hashValid;
206  }

208  struct DominationEquality;
209  friend struct ConstructionNode::DominationEquality;
210  friend class LanguageModelLookahead::ConstructionTree;

212 public:
213  ConstructionNode()
214  : id(invalidId),
215  consolidation_(dirty) {
216  depth.min = Core::Type<StateTree::Depth>::max;
217  depth.max = Core::Type<StateTree::Depth>::min;
218  }

220  const Ends& dominees() const {
221  switch (consolidation_) {
222  case dirty: require(consolidation_ > dirty);
223  case unique: updateDominiees();
224  case domineesValid:;
225  case hashValid:;
226  }
227  verify_(consolidation_ >= domineesValid);
228  return dominees_;
229  }
230 
231  u32 hash() const {
232  switch (consolidation_) {
233  case dirty: require(consolidation_ > dirty);
234  case unique: updateDominiees();
235  case domineesValid: updateHash();
236  case hashValid:;
237  }
238  verify_(consolidation_ >= hashValid);
239  return hash_;
240  }
241 
242  const Ends& ends() const {
243  return ends_;
244  }
245  Ends& ends() {
246  consolidation_ = dirty;
247  return ends_;
248  }
249 
250  const Successors& successors() const {
251  return successors_;
252  }
253  Successors& successors() {
254  consolidation_ = dirty;
255  return successors_;
256  }

258 protected:
259  struct DominationHash {
260  u32 operator()(const ConstructionNode* n) const {
261  return n->hash();
262  }
263  };
264 
265 private:
266  struct DominationEquality {
267  bool operator()(const ConstructionNode* l, const ConstructionNode* r) const {
268  if (l->consolidation_ >= hashValid && r->consolidation_ >= hashValid)
269  if (l->hash_ != r->hash_)
270  return false;
271  const Ends& ld(l->dominees());
272  const Ends& rd(r->dominees());
273  if (ld.size() != rd.size())
274  return false;
275  return std::equal(ld.begin(), ld.end(), rd.begin());
276  }
277  };

314 bool LanguageModelLookahead::ConstructionTree::isWellOrdered() const {
315  bool result = true;
316  for (u32 ci = 0; ci < nodeList_.size(); ++ci) {
317  const ConstructionNode& cn(*nodeList_[ci]);
318  if (cn.id == invalidId)
319  continue;
320  verify(cn.id == ci);
321  for (ConstructionNode::Successors::const_iterator si = cn.successors().begin(); si != cn.successors().end(); ++si) {
322  result = result && ((*si)->id != invalidId);
323  result = result && ((*si)->id < ci);
324  }
325  }
326  return result;
327 }

333 void LanguageModelLookahead::ConstructionTree::writeStatistics(std::ostream& os) const {
334  typedef std::map<StateTree::Depth, LevelStatistics> LevelStatisticsMap;
335  LevelStatisticsMap levels;
336  for (u32 ci = 0; ci < nodeList_.size(); ++ci) {
337  const ConstructionNode& cn(*nodeList_[ci]);
338  if (cn.id == invalidId)
339  continue;
340  LevelStatistics& ls(levels[cn.depth.min]);
341  ls.nNodes += 1;
342  ls.nSuccessors += cn.successors().size();
343  ls.nEnds += cn.ends().size();
344  }
345  for (std::map<StateTree::Depth, LevelStatistics>::const_iterator l = levels.begin(); l != levels.end(); ++l) {
346  StateTree::Depth depth = l->first;
347  const LevelStatistics& thisLevel(l->second);
348  os << Core::form("level %3d: %6d nodes, branching factor %3.2f, %4d ends\n",
349  depth, thisLevel.nNodes, float(thisLevel.nSuccessors) / float(thisLevel.nNodes), thisLevel.nEnds);
350  }
351 }

353 void LanguageModelLookahead::ConstructionTree::build(const StateTree* st) {
354  std::vector<LookaheadId> nodeId(st->nStates(), invalidId);
355  typedef std::unordered_set<ConstructionNode*,
356  ConstructionNode::DominationHash,
357  ConstructionNode::DominationEquality>
358  NodeSet;
359  NodeSet nodeSet;
360 
361  for (StateTree::ReverseTopologicalStateIterator rtsi(st); rtsi; ++rtsi) {
362  const StateTree::StateId si(*rtsi);
363  verify(nodeId[si] == invalidId);
364  StateTree::const_ExitIterator e, e_end;
365  tie(e, e_end) = st->wordEnds(si);
366  bool hasEnds = (e != e_end);
367 
368  StateTree::SuccessorIterator s, s_end;
369  tie(s, s_end) = st->successors(si);
370  u32 nSuccessors = s_end - s;
371 
372  ConstructionNode* cn = 0;
373  if (nSuccessors == 1 && !hasEnds) {
374  verify(nodeId[s] != invalidId);
375  cn = nodeList_[nodeId[s]];
376  }
377  else {
378  cn = new ConstructionNode;
379  for (; e != e_end; ++e) {
380  if (e->pronunciation) {
381  cn->ends().push_back(e->pronunciation);
382  }
383  else {
384  StateTree::StateId te = e->transitEntry;
385  verify(nodeId[te] != invalidId);
386  cn->successors().push_back(nodeList_[nodeId[te]]);
387  }
388  }
389  for (; s != s_end; ++s) {
390  verify(nodeId[s] != invalidId);
391  cn->successors().push_back(nodeList_[nodeId[s]]);
392  }
393  cn->makeUnique();
394 
395  NodeSet::iterator cni = nodeSet.find(cn);
396  if (cni == nodeSet.end()) {
397  cn->id = nodeList_.size();
398  nodeList_.push_back(cn);
399  nodeSet.insert(cn);
400  }
401  else {
402  delete cn;
403  cn = *cni;
404  }
405  }
406  nodeId[si] = cn->id;
407  cn->depth.min = std::min(cn->depth.min, st->stateDepth(si));
408  cn->depth.max = std::max(cn->depth.max, st->stateDepth(si));
409  cn->represents.push_back(si);
410  }
411  ensure(isWellOrdered());
412 }

422 /**
423  * Merge nodes of construction tree which should be pruned with their
424  * parents. Proceed in in bottom-up direction. A pruned nodes is
425  * indicated by its @c id field being invalid. Call purge() to remove
426  * pruned nodes.
427  */

429 void LanguageModelLookahead::ConstructionTree::prune(const LanguageModelLookahead* master) {
430  for (NodeList::iterator ci = nodeList_.begin(); ci != nodeList_.end(); ++ci) {
431  ConstructionNode& cn(**ci);
432  verify(cn.successors().size() || cn.ends().size());
433  ConstructionNode::Successors newSuccessors;
434  for (ConstructionNode::Successors::iterator si = cn.successors().begin(); si != cn.successors().end(); ++si) {
435  ConstructionNode& sn(**si);
436  if (sn.id == invalidId || master->shouldPruneConstructionNode(sn)) {
437  std::copy(sn.ends().begin(), sn.ends().end(), std::back_insert_iterator<Ends>(cn.ends()));
438  std::copy(sn.successors().begin(), sn.successors().end(), std::back_insert_iterator<ConstructionNode::Successors>(newSuccessors));
439  std::copy(sn.represents.begin(), sn.represents.end(), std::back_insert_iterator<ConstructionNode::Represents>(cn.represents));
440  cn.depth.min = std::min(cn.depth.min, sn.depth.min);
441  cn.depth.max = std::max(cn.depth.max, sn.depth.max);
442  sn.id = invalidId;
443  }
444  else {
445  newSuccessors.push_back(*si);
446  }
447  }
448  cn.successors().swap(newSuccessors);
449  verify(cn.successors().size() || cn.ends().size());
450  cn.makeUnique();
451  }
452  ensure(isWellOrdered());
453 }

457 void LanguageModelLookahead::ConstructionTree::purge() {
458  for (NodeList::iterator ci = nodeList_.begin(); ci != nodeList_.end(); ++ci) {
459  if ((*ci)->id == invalidId) {
460  delete *ci;
461  *ci = 0;
462  }
463  }
464  // reassign ids
465  nodeList_.erase(std::remove(nodeList_.begin(), nodeList_.end(), (ConstructionNode*)0), nodeList_.end());
466  for (LookaheadId id = 0; id < nodeList_.size(); ++id) {
467  nodeList_[id]->id = id;
468  }
469  ensure(isWellOrdered());
470 }

474 void LanguageModelLookahead::buildCompressesLookaheadStructure(const StateTree* st, const ConstructionTree& ct) {
475  require(ct.isWellOrdered());
476  require(ct.nNodes() > 0);
477  nodeId_.resize(st->nStates(), invalidId);
478 
479  for (LookaheadId ci = 0; ci < ct.nNodes(); ++ci) {
480  const ConstructionNode& cn(ct.node(ci));
481  verify(ci == nodes_.size());
482  nodes_.push_back(Node());
483  Node& n(nodes_.back());
484  n.firstEnd = ends_.size();
485  n.firstSuccessor = successors_.size();
486  std::copy(cn.ends().begin(), cn.ends().end(), std::back_insert_iterator<Ends>(ends_));
487  for (ConstructionNode::Successors::const_iterator si = cn.successors().begin(); si != cn.successors().end(); ++si)
488  successors_.push_back((*si)->id);
489  for (std::vector<StateTree::StateId>::const_iterator ri = cn.represents.begin(); ri != cn.represents.end(); ++ri)
490  nodeId_[*ri] = ci;
491  }
492 
493  for (StateTree::StateId si = 0; si < st->nStates(); ++si)
494  verify(nodeId_[si] != invalidId);
495 
496  // add sentinel
497  nodes_.push_back(Node());
498  Node& n(nodes_.back());
499  n.firstEnd = ends_.size();
500  n.firstSuccessor = successors_.size();
501 
502  nEntries_ = nodes_.size() - 1;
503 }

505 void LanguageModelLookahead::buildBatchRequest() {
506  require(!batchRequest_);
507 
508  Lm::BatchRequest batch;
509  for (u32 n = 0; n < nEntries_; ++n) {
510  for (Ends::const_iterator e = ends_.begin() + nodes_[n].firstEnd, e_end = ends_.begin() + nodes_[n + 1].firstEnd; e != e_end; ++e) {
511  Lm::Request request((*e)->lemma()->syntacticTokenSequence(), n);
512  for (u32 ti = 0; ti < request.tokens.length(); ++ti)
513  request.offset += lm_->scale() * request.tokens[ti]->classEmissionScore();
514  request.offset += wpScale_ * (*e)->pronunciationScore();
515  batch.push_back(request);
516  }
517  }
518  batchRequest_ = lm_->compileBatchRequest(batch);
519 }

573 void LanguageModelLookahead::computeScores(const Lm::History& history, std::vector<Score>& scores) const {
574  require(scores.size() == nEntries_);
575 
576  // log("computing look-ahead table for history ") << lm_->formatHistory(history);
577 
578  std::fill(scores.begin(), scores.end(), Core::Type<Score>::max);
579 
580  lm_->getBatch(history, batchRequest_, scores);
581 
582  std::vector<Score>::iterator score = scores.begin();
583  for (std::vector<Node>::const_iterator n = nodes_.begin(); n != nodes_.end() - 1; ++n) {
584  Score minScore = *score;
585  for (Successors::const_iterator s = successors_.begin() + n->firstSuccessor, s_end = successors_.begin() + (n + 1)->firstSuccessor; s != s_end; ++s) {
586  verify_(*s < LookaheadId(score - scores.begin()));
587  if (minScore > scores[*s])
588  minScore = scores[*s];
589  }
590  *score++ = minScore;
591  }
592  verify(score == scores.end());
593 }

603 LanguageModelLookahead::ContextLookahead* LanguageModelLookahead::acquireTable(const Lm::History& h) const {
604  ContextLookahead* t = 0;
605  if ((nTables() < cacheSizeLowMark_) || !nFreeTables_) {
606  t = new ContextLookahead(this, h, nEntries_);
607  tables_.push_front(t);
608  t->pos_ = tables_.begin();
609  ++nTables_;
610  }
611  else {
612  t = freeTables_.back();
613  freeTables_.pop_back();
614  --nFreeTables_;
615  t->freePos_ = freeTables_.end();
616  map_.erase(t->history_);
617  t->history_ = h;
618  }
619  ensure(t->history_ == h);
620  ensure(t->isActive());
621  return t;
622 }

624 void LanguageModelLookahead::releaseTable(const ContextLookahead* ct) const {
625  ContextLookahead* t = const_cast<ContextLookahead*>(ct);
626  require(t->isActive());
627  if (nTables() > cacheSizeHighMark_) {
628  if (nFreeTables_) {
629  freeTables_.push_front(t);
630  t->freePos_ = freeTables_.begin();
631  t = freeTables_.back();
632  freeTables_.pop_back();
633  }
634  verify(*t->pos_ == t);
635  tables_.erase(t->pos_);
636  --nTables_;
637  map_.erase(t->history_);
638  delete t;
639  }
640  else {
641  freeTables_.push_front(t);
642  t->freePos_ = freeTables_.begin();
643  ++nFreeTables_;
644  }

648  ContextLookahead* t = 0;

650  Map::const_iterator i = map_.find(h);
651  if (i != map_.end()) {

654  cacheStatistics_->cacheEvents += CacheStatistics::freeCacheHit;
