51 bool FsaCache::setDependencies(const Core::DependencySet& dependencies) {
52  if (!archive_)
53  return true;
54 
55  bool isClean = false;
56  Core::ArchiveReader ar(*archive_, "DEPENDENCIES");
57  if (ar.isOpen()) {
58  Core::DependencySet cacheDependencies;
59  if (cacheDependencies.read(config, ar)) {
60  if (dependencies.satisfies(cacheDependencies))
61  isClean = true;
62  }
63  else
64  error("Failed to read dependencies from model acceptor cache.");
65  }
66 
67  /*! \warning Apparently dependencies on Alpha and Intel differ. */
68 
69  if (isClean) {
70  log("Using clean model acceptor cache.");
71  }
72  else {
73  if (readOnly_)
74  criticalError("Read-only flag set but the cache is not clean.");
75 
76  if (ar.isOpen())
77  warning("Cache is dirty.");
78 
79  Core::XmlOutputStream xw(new Core::ArchiveWriter(*archive_, "DEPENDENCIES"));
80  dependencies.write(xw);
81  }
82  return isClean;
83 }
