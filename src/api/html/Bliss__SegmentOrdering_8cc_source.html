116 void SegmentOrderingTest::createCorpus(bool subCorpus) {
117  recordings_.clear();
118  segments_.clear();
119  speakers_.clear();
120  conditions_.clear();
121  Core::XmlOutputStream os(corpusFile_);
122  os << Core::XmlOpen("corpus") + Core::XmlAttribute("name", corpusName);
123  std::string condition = "cond";
124  os << Core::XmlEmpty("condition-description") + Core::XmlAttribute("name", condition);
125  std::string speaker = Core::form("speaker");
126  os << Core::XmlEmpty("speaker-description") + Core::XmlAttribute("name", speaker);
127  int nSub = (subCorpus ? nSubCorpus : 1);
128  for (int c = 0; c < nSub; ++c) {
129  std::string corpus = "";
130  if (subCorpus) {
131  corpus = Core::form("sub-%d", c);
132  condition = Core::form("cond-%d", c);
133  os << Core::XmlOpen("subcorpus") + Core::XmlAttribute("name", corpus);
134  os << Core::XmlEmpty("condition-description") + Core::XmlAttribute("name", condition);
135  speaker = Core::form("speaker-%d", c);
136  os << Core::XmlEmpty("speaker-description") + Core::XmlAttribute("name", speaker);
137  }
138  for (uint r = 0; r < nRecordings; ++r) {
139  std::string recording = Core::form("recording-%d-%d", c, r);
140  os << Core::XmlOpen("recording") +
141  Core::XmlAttribute("name", recording) +
142  Core::XmlAttribute("audio", "none");
143 
144  if (subCorpus)
145  recording = corpusName + "/" + corpus + "/" + recording;
146  else
147  recording = corpusName + "/" + recording;
148  recordings_.push_back(recording);
149  std::string recSpeaker = Core::form("speaker-%d-%d", c, r);
150  os << Core::XmlEmpty("speaker-description") + Core::XmlAttribute("name", recSpeaker);
151  for (uint s = 0; s < nSegments; ++s) {
152  std::string segment = Core::form("segment-%d-%d-%d", c, r, s);
153  os << Core::XmlOpen("segment") + Core::XmlAttribute("name", segment);
154  os << Core::XmlEmpty("condition") + Core::XmlAttribute("name", condition);
155  const std::string& curSpeaker = (s ? speaker : recSpeaker);
156  os << Core::XmlEmpty("speaker") + Core::XmlAttribute("name", curSpeaker);
157  os << Core::XmlClose("segment");
158  segment = recording + "/" + segment;
159  std::string speakerId = corpusName + "/";
160  if (subCorpus)
161  speakerId += corpus + "/";
162  if (s)
163  speakerId += curSpeaker;
164  else
165  speakerId = recording + "/" + curSpeaker;
166 
167  std::string conditionId = corpusName + "/";
168  if (subCorpus)
169  conditionId += corpus + "/";
170  conditionId += condition;
171  segments_.push_back(segment);
172  speakers_.push_back(speakerId);
173  conditions_.push_back(conditionId);
174  }
175  os << Core::XmlClose("recording");
176  }
177  if (subCorpus) {
178  os << Core::XmlClose("subcorpus");
179  }
180  }
181  os << Core::XmlClose("corpus");
182 }
