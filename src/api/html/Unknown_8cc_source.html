22 /*
23  * This is a rather crude model: We allow any sequence of phonemes.
24  * Currently the selection of allowed phonemes is very
25  * unsophisticated: Only coarticulated phonemes are allowed. This is
26  * reasonable and sufficient for the time being, since context
27  * independent phonemes are used for silence and noise
28  * pseudo-phonemes. For the future I envision the Lexicon to support
29  * non-standard pronunciations. Maybe one could write
30  * <phon type="random">a b c d e f</phon> or even
31  * <phon type="regex">(a|b|c|d|e|f)*</phon>
32  */

34 Fsa::ConstAutomatonRef Bliss::createAnyPhonemeToUnknownTransducer(
35  Core::Ref<const Lexicon> lexicon,
36  f32 weight) {
37  Core::Ref<Fsa::StaticAutomaton> utap(new Fsa::StaticAutomaton);
38  utap->setType(Fsa::TypeTransducer);
39  utap->setSemiring(Fsa::TropicalSemiring);
40  utap->setInputAlphabet(lexicon->phonemeInventory()->phonemeAlphabet());
41  utap->setOutputAlphabet(lexicon->lemmaAlphabet());
42  Fsa::State* first = utap->newState();
43  Fsa::State* second = utap->newState();
44  utap->setInitialStateId(first->id());
45  Fsa::Weight one = Fsa::Weight(utap->semiring()->one());
46  Fsa::Weight pen = Fsa::Weight(weight);
47  const Lemma* unknown = lexicon->specialLemma("unknown");
48  if (unknown) {
49  PhonemeInventory::PhonemeIterator pi, pi_end;
50  for (Core::tie(pi, pi_end) = lexicon->phonemeInventory()->phonemes(); pi != pi_end; ++pi) {
51  if (!(*pi)->isContextDependent())
52  continue;
53  first->newArc(second->id(), pen, (*pi)->id(), unknown->id());
54  second->newArc(second->id(), pen, (*pi)->id(), Fsa::Epsilon);
55  }
56  }
57  second->newArc(first->id(), one,
58  lexicon->phonemeInventory()->phonemeAlphabet()->disambiguator(0),
59  Fsa::Epsilon);
60  utap->setStateFinal(first);
61  return utap;
62 }
