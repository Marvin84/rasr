299 /**
300  * Test for near-equality of floating point numbers.
301  * Due to finite precision, the bit-wise test (a == b) is almost
302  * always false. isAlmostEqual() compares the relative difference
303  * of a and b to the machine precision (epsilon) times the given
304  * tolerance factor.
305  *
306  * Deprecation warning: For new code you should prefer
307  * isAlmostEqualUlp.
308  *
309  * Remark:
310  * -A similar idea can be found under the name "chordal metric":
311  * chord(a, b) = |a - b| / (sqrt(1+a^2) * sqrt(1 + b^2)).
312  * -sorry for repeating the same implementation for each
313  * floating point type, but specialization to complex numbers
314  * seems to be a hard nut with templates.
315  */
316 inline bool isAlmostEqual(f32 a, f32 b, f32 tolerance = (f32)1) {
317  require_(tolerance > (f32)0);
318  f32 d = Core::abs(a - b);
319  f32 e = (Core::abs(a) + Core::abs(b) + Type<f32>::delta) * Type<f32>::epsilon * tolerance;
320  return (d < e);
321 }
322 inline bool isAlmostEqual(f64 a, f64 b, f64 tolerance = (f64)1) {
323  require_(tolerance > (f64)0);
324  f64 d = Core::abs(a - b);
325  f64 e = (Core::abs(a) + Core::abs(b) + Type<f64>::delta) * Type<f64>::epsilon * tolerance;
326  return (d < e);
327 }

365 /**
366  * Test for near-equality of floating point numbers with tolerance
367  * given in unit of least precision. Due to finite precision, the
368  * bit-wise test (a == b) should not be use for floating point
369  * values. This function was taken from "Comparing floating point
370  * numbers" by Bruce Dawson
371  * [http://www.cygnus-software.com/papers/comparingfloats/comparingfloats.htm].
372  * You will probably not be able to make sense of this code
373  * without reading the article.
374  *
375  * @param tolerance allowed difference between @c a and @c b
376  * measured in units of least precision (Ulp). This can be
377  * thought of as the number of different normalized floating point
378  * numbers between @c a and @c b.
379  *
380  * This method is preferrable to the older isAlmostEqual(),
381  * because it is faster and theoretically more sound. So maybe
382  * isAlmostEqual() should be removed some time.
383  */
384 inline bool isAlmostEqualUlp(f32 a, f32 b, s32 tolerance) {
385  require_(tolerance > 0);
386  require_(tolerance < 0x400000);
387  return (differenceUlp(a, b) <= tolerance);
388 }
