359  std::pair<SuccessorIterator, SuccessorIterator> successors(StateId s) const {
360  require_(0 <= s && s < nStates());

368  std::pair<SuccessorIterator, SuccessorIterator> successors2(StateId s) const {
369  require_(0 <= s && s < nStates());
370  if (!haveSuccessorBatches_) {
371  return std::make_pair(SuccessorIterator(0), SuccessorIterator(0));
372  }
373  BatchId b = states_[s].successors;
374  verify_(0 <= b && b < BatchId(successorBatches_.size()));
375  b = successorBatches_[b];
376  verify_(0 <= b && b + 1 < BatchId(batches_.size()));
377  ensure_(batches_[b] <= batches_[b + 1]);
378  return std::make_pair(SuccessorIterator(batches_[b]),
