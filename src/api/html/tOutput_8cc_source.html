92  else
93  o_ << " " << a->input() + 1;
94  }
95  else {
96  if (input_)
97  o_ << " eps";
98  else
99  o_ << " 0";
100  }
101  if (Precursor::fsa_->type() == Fsa::TypeTransducer) {
102  if (a->output() != Fsa::Epsilon) {
103  if (output_)
104  o_ << " " << output_->symbol(a->output());
105  else
106  o_ << " " << a->output() + 1;
107  }
108  else {
109  if (output_)
110  o_ << " eps";
111  else
112  o_ << " 0";
113  }
114  }
115  if (!Precursor::fsa_->semiring()->isDefault(a->weight())) {
116  std::string weightStr = Precursor::fsa_->semiring()->asString(a->weight());
117  // weight representation must not contain whitespaces
118  verify(weightStr.find_first_of(utf8::whitespace) == std::string::npos);
119  o_ << " " << weightStr;
120  }
121  o_ << std::endl;
122  }

187 template<class _Automaton>
188 bool writeBinary(const Resources<_Automaton>& resources, typename _Automaton::ConstRef f,
189  std::ostream& o, Fsa::StoredComponents what, bool progress) {
190  if (!o)
191  return false;
192 
193  Core::BinaryOutputStream bo(o);
194 
195  static const char* magic = "RWTHFSA";
196  static const u32 formatVersion = 2;
197  if (!bo.write(magic, 8))
198  return false;
199 
200  Fsa::StoredComponents canStore = Fsa::storeStates;
201  if (f->getInputAlphabet())
202  canStore |= Fsa::storeInputAlphabet;
203  if ((f->type() == Fsa::TypeTransducer) && f->getOutputAlphabet())
204  canStore |= Fsa::storeOutputAlphabet;
205  what &= canStore;
206 
207  bo << u32(what | (formatVersion << 24)) << u32(f->type());
208 
209  if (what & Fsa::storeStates)
210  bo << u32(f->properties())
211  << u32(f->knownProperties())
212  << u32(resources.getSemiringType(f->semiring()));
213 
214  if (!bo)
215  return false;
216  if (what & Fsa::storeInputAlphabet)
217  f->getInputAlphabet()->write(bo);
218  if (what & Fsa::storeOutputAlphabet)
219  f->getOutputAlphabet()->write(bo);
220  if (what & Fsa::storeStates) {
221  u32 tmp = f->initialStateId();
222  bo << tmp;
223  Core::ProgressIndicator* p = 0;
224  if (progress)
225  p = new Core::ProgressIndicator("writing", "states");
226  WriteBinaryDfsState<_Automaton> s(f, bo);
227  s.dfs(p);
228  delete p;
229  }
230 
231  return true;
232 }

366 bool writeXml(const Resources<_Automaton>& resources, typename _Automaton::ConstRef f,
367  std::ostream& o, Fsa::StoredComponents what, bool progress) {
368  if (!o)
369  return false;
370  Core::XmlWriter xo(o);
371  xo.putDeclaration("UTF-8");
372  xo << "\n";
373  Core::ProgressIndicator* p = 0;
374  if (f->initialStateId() != Fsa::InvalidStateId) {
375  xo << Core::XmlOpen("fsa") + Core::XmlAttribute("type", Fsa::TypeChoice[f->type()]) + Core::XmlAttribute("semiring", f->semiring()->name()) + Core::XmlAttribute("initial", f->initialStateId()) << "\n";
376  }
377  else {
378  xo << Core::XmlOpen("fsa") + Core::XmlAttribute("type", Fsa::TypeChoice[f->type()]) + Core::XmlAttribute("semiring", f->semiring()->name()) << "\n";
379  }
380 
381  xo << "\n"
382  << Core::XmlOpenComment() << "Properties were (only saved in bin format) '" << f->properties() << "'" << Core::XmlCloseComment() << "\n\n";
383 
384  if (what & Fsa::storeInputAlphabet)
385  if (f->getInputAlphabet()) {
386  xo << Core::XmlOpen("input-alphabet") << "\n";
387  f->getInputAlphabet()->writeXml(xo);
388  xo << Core::XmlClose("input-alphabet") << "\n";
389  }
390  if (what & Fsa::storeOutputAlphabet && f->type() == Fsa::TypeTransducer)
391  if (f->getOutputAlphabet()) {
392  xo << Core::XmlOpen("output-alphabet") << "\n";
393  f->getOutputAlphabet()->writeXml(xo);
394  xo << Core::XmlClose("output-alphabet") << "\n";
395  }
396  if (what & Fsa::storeStates) {
397  if (progress)
398  p = new Core::ProgressIndicator("writing", "states");
399  WriteXmlDfsState<_Automaton> s(f, xo);
400  s.dfs(p);
401  delete p;
402  }
403  xo << Core::XmlClose("fsa") << "\n";
404  return true;

452  end_id_ = max_id_;

462  }
