233 template<typename T>
234 template<typename S>
235 inline void LinearAndSoftmaxLayer<T>::removeLogPriorFromBias(const Prior<S>& priors) {
236  require(this->getBias());
237  require_eq(priors.size(), this->getBias()->size());
238  T prioriScale = priors.scale();
239  if (prioriScale != S(0.0)) {
240  this->log("removing scaled log-prior from bias parameters (scale: ") << prioriScale << ")";
241  // subtract
242  if (this->getBias()->isComputing()) {
243  typename Types<S>::NnVector priorVector;
244  priorVector.resize(priors.size());
245  priors.getVector(priorVector);
246  priorVector.initComputation();
247  this->getBias()->add(priorVector, -S(prioriScale));
248  }
249  else {
250  for (u32 c = 0; c < this->getBias()->size(); c++)
251  this->getBias()->at(c) -= prioriScale * priors.at(c);
252  }
253 
254  logPriorIsRemovedFromBias_ = true;
255  }
256 }
