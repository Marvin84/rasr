88  y->dump(criticalError("Frame="));
89 
90  if (sendVector(*y, estimation_error, B_tilde, A_tilde))

94 
95  if (!ret)
96  y->dump(criticalError("Frame="));
97 
98  return ret;

101 bool LinearPredictionCodingNode::sendVector(const Flow::Timestamp& time_stamp,
102  const f32& estimation_error,
103  const std::vector<f32>& B_tilde,
104  const std::vector<f32>& A_tilde) {
105  if (nOutputLinks(0) == 0)
106  return false;
107 
108  Flow::Vector<f32>* out = new Flow::Vector<f32>(1 + B_tilde.size() + A_tilde.size());
109 
110  u32 i = 0;
111  (*out)[i++] = sqrt(estimation_error);
112  for (u8 j = 0; j < B_tilde.size(); j++)
113  (*out)[i++] = B_tilde[j];
114  for (u8 j = 0; j < A_tilde.size(); j++)
115  (*out)[i++] = A_tilde[j];
116 
117  out->setTimestamp(time_stamp);
118 
119  return putData(0, out);
120 }
