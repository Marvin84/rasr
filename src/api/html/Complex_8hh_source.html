65 /** tranforms alternating complex vector into an arbitrary type
66  * (Alternating complex vectors are stored in a standard container
67  * by storing the real and the imaginary parts alternating.)
68  *
69  * Alternating complex numbers are first transformed to a complex<InputType> object.
70  * Operation is transforms the complex<InputType> object:
71  * operation must support the function: OutputType operator()(const complex<InputType> &).
72  * Operation is a unary function (derived from std::unary_function):
73  * Argument must have a constructor with two parameters
74  * Result is of arbitrary type
75  */
76 template<class InputIterator, class OutputIterator, class Operation>
77 void transformAlternatingComplex(InputIterator first, InputIterator last,
78  OutputIterator result, Operation operation) {
79  for (InputIterator real = first, imaginary = first + 1;
80  real != last; ++result) {
81  *result = operation(typename Operation::argument_type(*real, *imaginary));
82  real += 2;
83  if (real != last)
84  imaginary += 2;
85  }
86 }

88 /** tranforms a vector of arbitrary type into an alternating complex vector
89  * (Alternating complex vectors are stored in a standard container
90  * by storing the real and the imaginary parts alternating.)
91  * Operation is a unary function (derived from std::unary_function):
92  * Argument is of arbitrary type
93  * For the result type the functions real and imag must exist.
94  *
95  */
96 template<class InputIterator, class OutputIterator, class Operation>
97 void transformToAlternatingComplex(InputIterator first, InputIterator last,
98  OutputIterator result, Operation operation) {
99  for (; first != last; result += 2, ++first) {
100  typename Operation::result_type r = operation(*first);
101 
102  *result = real(r);
103  *(result + 1) = imag(r);
104  }
105 }

107 /** tranforms an alternating complex vector into an alternating complex vector
108  * (Alternating complex vectors are stored in a standard container
109  * by storing the real and the imaginary parts alternating.)
110  *
111  * Operation is a binary function derived from std::binary_function:
112  * Arguments must have a constructor with two parameters
113  * For the result type the functions real and imag must exist.
114  */
115 template<class InputIterator, class OutputIterator, class Operation>
116 void transformAlternatingComplexToAlternatingComplex(InputIterator first1, InputIterator last1,
117  InputIterator first2, OutputIterator result,
118  Operation operation) {
119  for (InputIterator real1 = first1, imaginary1 = first1 + 1,
120  real2 = first2, imaginary2 = first2 + 1;
121  real1 != last1;
122  result += 2, real1 += 2, imaginary1 += 2, real2 += 2, imaginary2 += 2) {
123  typename Operation::result_type r = operation(
124  typename Operation::first_argument_type(*real1, *imaginary1),
125  typename Operation::second_argument_type(*real2, *imaginary2));
126 
127  *result = real(r);
128  *(result + 1) = imag(r);
129  }
130 }
