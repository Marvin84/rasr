43  "for a Bliss corpus or a sequence of batches depending\n"
44  "on the chosen nodes.\n"
45  "The network has to be provided as Sprint configuration file,\n"
46  "i.e. the use of the option \"--config\" is mandatory for\n"
47  "running a network.\n"
48  "\n"
49  "The following command line commands are supported:\n"
50  " (none) runs the network\n"
51  " init initializes the network, but does not\n"
52  " run it\n"
53  " parse parses the network, but neither\n"
54  " initializes nor runs it\n"
55  " help gives some general help on how setting\n"
56  " up a network\n"
57  " help list lists all nodes; help is available for\n"
58  " each node\n"
59  " help NODE gives help to the specific node\n";

118  int main(const std::vector<std::string>& arguments) {
119  enum { Run,
120  Init,
121  Parse,
122  Help } cmd = Run;
123  Flf::Module::Instance& flfApp = Flf::Module::instance();
124  if (!arguments.empty()) {
125  if ((arguments.front() == "init") ||
126  (arguments.front() == "check"))
127  cmd = Init;
128  else if (arguments.front() == "parse")
129  cmd = Parse;
130  else if (arguments.front() == "help")
131  cmd = Help;
132  }
133  switch (cmd) {
134  case Run:
135  case Init:
136  case Parse: {
137  /*! @todo network, lexicon, processor should not be part of Flf::Module (?) */
138  flfApp.setNetwork(Flf::Network::createNetwork(select("network")));
139  if (cmd == Parse) {
140  std::cout << "Network was successfully parsed:" << std::endl
141  << std::endl;
142  flfApp.network()->dump(std::cout);
143  break;
144  }
145  log("Initialize network ...");
146  flfApp.init();
147  flfApp.setProcessor(createProcessor(flfApp.network()));
148  bool good = flfApp.processor()->init(arguments);
149  if (cmd == Init) {
150  std::cout << "Network was successfully initialized:" << std::endl
151  << std::endl;
152  flfApp.processor()->crawler().dump(std::cout);
153  break;
154  }
155  log("Process network...");
156  if (good)
157  flfApp.processor()->run();
158  flfApp.processor()->finalize();
159  } break;
160  case Help: {
161  if (arguments.size() == 1) {
162  printHelp(std::cout);
163  }
164  else {
165  Flf::NodeFactory factory;
166  if (arguments[1] == "list") {
167  factory.dumpNodeList(std::cout);
168  std::cout << "Try \"help <node> to get help for a specific node.\n"
169  << std::endl;
170  }
171  else {
172  for (u32 i = 1; i < arguments.size(); ++i) {
173  factory.dumpNodeDescription(std::cout, arguments[i]);
174  std::cout << std::endl;
175  }
176  }
177  }
178  } break;
179  }
180 
181  delete flfApp.processor();
182  delete flfApp.network();
183  return EXIT_SUCCESS;
184  }
