93 void NBestListLatticeProcessor::processWordLattice(Lattice::ConstWordLatticeRef lattice, Bliss::SpeechSegment* s) {
94  Lattice::ConstWordLatticeRef nBestList = extractor_.getNBestList(lattice);
95  if (statisticsChannel_.isOpen()) {
96  Fsa::ConstAutomatonRef list = nBestList->mainPart();
97  Fsa::ConstStateRef hypotheses = list->getState(list->initialStateId());
98  statisticsChannel_ << Core::XmlOpen("n-best-list-statistics") + Core::XmlAttribute("size", hypotheses->nArcs());
99  for (Fsa::State::const_iterator it = hypotheses->begin(); it != hypotheses->end(); ++it) {
100  statisticsChannel_ << Core::XmlOpen("hypothesis") + Core::XmlAttribute("rank", it->target()) + Core::XmlAttribute("score", f32(Fsa::bestscore(Fsa::partial(list, it->target()))));
101  statisticsChannel_ << Core::XmlClose("hypothesis");
102  }
103  statisticsChannel_ << Core::XmlClose("n-best-list-statistics");
104  }
105  Precursor::processWordLattice(nBestList, s);
106 }
