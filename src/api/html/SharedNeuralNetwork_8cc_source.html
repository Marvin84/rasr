40 void SharedNeuralNetwork::create(const Core::Configuration& c) {
41  if (!hasInstance()) {
42  network_ = new NeuralNetwork<f32>(c);
43  network_->initializeNetwork(1);
44  labelWrapper_ = new ClassLabelWrapper(Core::Configuration(c, "class-labels"));
45  prior_ = new Prior<f32>(c);
46 
47  auto* topLayer = &network().getTopLayer();
48  auto* softmaxTopLayer = dynamic_cast<Nn::LinearAndSoftmaxLayer<f32>*>(topLayer);
49  auto* biasTopLayer = dynamic_cast<Nn::BiasLayer<f32>*>(topLayer);
50  if (!softmaxTopLayer && !biasTopLayer) {
51  auto* maxoutLayer = dynamic_cast<Nn::MaxoutVarLayer<f32>*>(topLayer);
52  if (maxoutLayer)
53  softmaxTopLayer = dynamic_cast<Nn::LinearAndSoftmaxLayer<f32>*>(&network().getLayer(maxoutLayer->getPredecessor(0)));
54  }
55 
56  if (softmaxTopLayer) {
57  // forward until softmax only
58  // assume that log-prior is already removed from bias parameters of last layer
59  // assume that parameters are already scaled according to mixture-scale
60  if (softmaxTopLayer->evaluatesSoftmax())
61  softmaxTopLayer->setEvaluateSoftmax(false); // switch off the softmax
62  Core::printLog("SharedNeuralNetwork: switched off softmax eval on softmax-layer");
63  }
64  else if (biasTopLayer) {
65  Core::printLog("SharedNeuralNetwork: bias-layer is top layer, we assume it's in log space already");
66  }
67  else {
68  Core::printWarning("SharedNeuralNetwork: top layer type is unknown, we assume it's in log space");
69  }
70 
71  if (prior_->fileName() != "" && prior_->scale() != 0.0) {
72  prior_->read();
73  if (softmaxTopLayer) {
74  softmaxTopLayer->removeLogPriorFromBias(prior());
75  Core::printLog("SharedNeuralNetwork: substract log prior from softmax-layer bias");
76  }
77  else if (biasTopLayer) {
78  biasTopLayer->removeLogPriorFromBias(prior());
79  Core::printLog("SharedNeuralNetwork: substract log prior from bias-layer bias");
80  }
81  else {
82  Core::printWarning("SharedNeuralNetwork: cannot subtract prior");
83  }
84  }
85  else {
86  Core::printLog("SharedNeuralNetwork: not substracting log prior (either file-name not set or scale is zero)");
87  }
88  }
89 }
