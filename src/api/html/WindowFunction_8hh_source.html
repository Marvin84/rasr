77 template<class Iterator>
78 bool WindowFunction::work(const Iterator& begin, const Iterator& end) {
79  ensure(std::distance(begin, end) >= (s32)length());
80 
81  if (needInit_ && !init())
82  return false;
83 
84  std::transform(window_.begin(), window_.end(), begin, begin, std::multiplies<Float>());
85 
86  std::fill(begin + length(), end, 0);
87 
88  return true;
89 }
