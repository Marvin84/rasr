31  void buildHash() {
32  hash_ = Search::StateTree::StateDesc::Hash()(desc_);
33  u32 numSuccessors = successors_.size();
34  hash_ += (numSuccessors >> 3) + (numSuccessors << 11);
35  for (std::set<Search::StateId>::iterator it = successors_.begin(); it != successors_.end(); ++it)
36  hash_ += ((*it) >> 11) + ((*it) << 5);
37  }
38 
39 public:
40  StateWithSuccessors()
41  : hash_(0) {
42  }
43 
44  StateWithSuccessors(const Search::StateTree::StateDesc& _desc, const std::set<Search::StateId>& _successors)
45  : desc_(_desc), successors_(_successors) {
46  buildHash();
47  }
48 
49  bool operator==(const StateWithSuccessors& rhs) const {
50  return hash_ == rhs.hash_ && desc_ == rhs.desc_ && successors_ == rhs.successors_;
51  }
