47 bool WarpTimeFilterNode::work(PortId p) {
48  DataPtr<Timestamp> in;
49  while (getData(0, in)) {
50  in = DataPtr<Timestamp>(dynamic_cast<Timestamp*>(in->clone()));
51  verify(currentTime_ != Core::Type<Flow::Time>::max);
52  verify(in->startTime() >= currentTime_);
53 
54  Time offset = currentTime_ - in->startTime();
55 
56  if (warping_.empty() || offset != warping_.back().first - warping_.back().second)
57  warping_.push_back(std::make_pair(in->startTime() + offset, in->startTime()));
58 
59  in->setStartTime(in->startTime() + offset);
60  in->setEndTime(in->endTime() + offset);
61 
62  currentTime_ = in->endTime();
63  return putData(0, in.get());
64  }
65 
66  if (in.get() == Flow::Data::eos() && warping_.size()) {
67  Core::Component::Message msg = log();
68  msg << "warping map:";
69  for (std::vector<std::pair<Time, Time>>::iterator it = warping_.begin(); it != warping_.end(); ++it)
70  msg << " " << it->first << ":" << it->second;
71  // Write the warping map
72  warping_.clear();
73  currentTime_ = Core::Type<Flow::Time>::max;
74  }
75 
76  return putData(0, in.get());
77 }
