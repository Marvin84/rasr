46 void computeAllScores(std::vector<LMRequest>& requests, LanguageModelRef lm, bool renormalize) {
47  for (auto& r : requests) {
48  Lm::Score normalization = 0.0f;
49  if (renormalize) {
50  normalization = std::numeric_limits<Lm::Score>::infinity();
51  auto iters = lm->lexicon()->syntacticTokens();
52  for (auto tok = iters.first; tok != iters.second; ++tok) {
53  Lm::Score s = lm->score(r.history, *tok);
54  normalization = Math::scoreSum(normalization, s);
55  }
56  }
57  r.score = lm->score(r.history, r.token) - normalization;
58  }
59 }

135 void LmUtilityTool::computePerplexityFromTextFile() {
136  bool renormalize = paramRenormalize(config);
137  size_t batch_size = paramBatchSize(config);
138  Bliss::LexiconRef lexicon(Bliss::Lexicon::create(select("lexicon")));
139  LanguageModelRef lm(Lm::Module::instance().createLanguageModel(select("lm"), lexicon));
140  Core::TextInputStream tis(new Core::CompressedInputStream(paramFile(config)));
141  Core::TextOutputStream out;
142 
143  tis.setEncoding(paramEncoding(config));
144  std::string out_file = paramScoreFile(config);
145  if (not out_file.empty()) {
146  out.open(out_file);
147  }
148 
149  std::vector<LMRequest> requests;
150  size_t num_tokens = 0;
151  Lm::Score corpus_score = 0.0;
152 
153  do {
154  std::string line;
155  std::getline(tis, line);
156  if (tis.good()) {
157  std::stringstream ss(line);
158  Lm::History h = lm->startHistory();
159  while (ss.good()) {
160  std::string word;
161  ss >> word;
162  Bliss::Lemma const* lemma = lexicon->lemma(word);
163  if (lemma == nullptr) {
164  lemma = lexicon->specialLemma("unknown");
165  }
166  auto const tokens = lemma->syntacticTokenSequence();
167  for (auto const& t : tokens) {
168  requests.emplace_back(LMRequest({word, lemma, t, h, 0.0f}));
169  h = lm->extendedHistory(h, t);
170  }
171  }
172  Bliss::Lemma const* lemma = lexicon->specialLemma("sentence-end");
173  auto const tokens = lemma->syntacticTokenSequence();
174  for (auto const& t : tokens) {
175  requests.emplace_back(LMRequest({"\\n", lemma, t, h, 0.0f}));
176  h = lm->extendedHistory(h, t);
177  }
178  }
179 
180  if (not tis.good() or requests.size() >= batch_size) {
181  computeAllScores(requests, lm, renormalize);
182  for (auto const& r : requests) {
183  corpus_score += r.score;
184  num_tokens += 1ul;
185  if (out.good()) {
186  out << r.word << " " << r.lemma->preferredOrthographicForm().str() << " " << r.score << std::endl;
187  }
188  }
189  requests.clear();
190  }
191  } while (tis.good());
192 
193  Lm::Score ppl = std::exp(corpus_score / num_tokens);
194 
195  log() << Core::XmlOpen("corpus-score") << corpus_score << Core::XmlClose("corpus-score")
196  << Core::XmlOpen("num-tokens") << num_tokens << Core::XmlClose("num-tokens")
197  << Core::XmlOpen("perplexity") << ppl << Core::XmlClose("perplexity");
198 }
