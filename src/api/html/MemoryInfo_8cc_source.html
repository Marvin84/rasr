37 void MemoryInfo::updateLinux() {
38  pageSize_ = ::sysconf(_SC_PAGESIZE);
39  pid_t pid = getpid();
40  std::string statm = Core::form("/proc/%d/statm", pid);
41  std::ifstream stat(statm.c_str());
42  if (!stat) {
43  Core::Application::us()->warning("cannot read %s", statm.c_str());
44  return;
45  }
46  stat >> size_ >> rss_ >> share_ >> text_ >> lib_ >> data_;
47  size_ *= pageSize_;
48  rss_ *= pageSize_;
49  text_ *= pageSize_;
50  lib_ *= pageSize_;
51  data_ *= pageSize_;
52 }

54 std::string MemoryInfo::peak() const {
55 #ifndef OS_linux
56  return "unknown";
57 #endif
58  pid_t pid = getpid();
59  std::string status = Core::form("/proc/%d/status", pid);
60  std::ifstream stat(status.c_str());
61  if (!stat) {
62  Core::Application::us()->warning("cannot read %s", status.c_str());
63  return "unknown";
64  }
65  std::string buffer;
66  while (std::getline(stat, buffer).good()) {
67  if (buffer.substr(0, 6) == "VmPeak") {
68  buffer = buffer.substr(7);
69  Core::stripWhitespace(buffer);
70  return buffer;
71  }
72  }
73  return "unknown";
74 }
