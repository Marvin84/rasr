31 /**
32  * Fits the lattice into [begin, end]:
33  * - begin and end are relative to the lattices initial state
34  * - fitting lattice has exactly one initial state with time 0 and one
35  * final state with time end-begin, i.e. all pathes have the same length
36  * - the time of all states but the final state is less than end (and >= 0)
37  * - for each path the in origianl lattice there exist a path with
38  * the same score in the fitting lattice; those are the only pathes
39  * in the fitting lattice (i.e. according to the begin and end times
40  * words at the beginning or end of a path might be discarded).
41  * - optional: the last arc carries the sentence-end-symbol
42  * - if begin and end time are not given, then begin is 0 and end time
43  * is the max time over all states.
44  **/

132  ScoresRef sssp(Fsa::StateId initSid) const {
133  ScoresRef ssspScore = semiring_->zero();
134  ssspScores_.grow(topologicalOrderMap_->maxSid);
135  TopologicalOrderQueue& Q = *topologicalQeue_;
136  Q.insert(initSid);
137  ssspScores_[initSid] = semiring_->one();
138  const Lattice& l = *fsa_;
139  while (!Q.empty()) {
140  Fsa::StateId sid = Q.top();
141  Q.pop();
142  ConstStateRef sr = l.getState(sid);
143  ScoresRef score = ssspScores_[sid];
144  ssspScores_[sid].reset();
145  if (sr->isFinal())
146  ssspScore = semiring_->collect(
147  ssspScore, semiring_->extend(score, sr->weight()));
148  for (State::const_iterator a = sr->begin(), a_end = sr->end(); a != a_end; ++a) {
149  if (!ssspScores_[a->target()]) {
150  ssspScores_[a->target()] = semiring_->extend(score, a->weight());
151  Q.insert(a->target());
152  }
153  else
154  ssspScores_[a->target()] = semiring_->collect(
155  ssspScores_[a->target()], semiring_->extend(score, a->weight()));
156  }
157  }
158  return ssspScore;

161  inline void addArc(Action action, State* from, const State* to, ScoresRef score, Fsa::LabelId input, Fsa::LabelId output) const {
162  switch (action) {
163  case ActionAddArc:
164  from->newArc(to->id() + 2, score, input, output);
165  break;
166  case ActionAddFinalArc: {
167  ScoresRef finalScore = to->hasArcs() ? sssp(to->id()) : (to->isFinal() ? to->weight() : semiring_->zero());
168  // true if lattice is trim
169  // verify(semiring_->compare(finalScore, semiring_->zero()) != 0);
170  from->newArc(1, semiring_->extend(score, finalScore), input, output);
171  break;
172  }
173  case ActionAddSentenceEndArc: {
174  ScoresRef finalScore = to->hasArcs() ? sssp(to->id()) : (to->isFinal() ? to->weight() : semiring_->zero());
175  // true if lattice is trim
176  // verify(semiring_->compare(finalScore, semiring_->zero()) != 0);
177  from->newArc(1, semiring_->extend(score, finalScore), sentenceEndInput_, sentenceEndOutput_);
178  break;
179  }
180  default:
181  defect();
182  }

185  State* buildInitialState(State* sp) const {
186  Fsa::StateId srcInitialSid = fsa_->initialStateId();
187  s32 initialT = srcBoundaries_->time(srcInitialSid);
188  verify(initialT == 0);
189  if (startTime_ < 0) {
190  sp->newArc(srcInitialSid + 2, semiring_->clone(semiring_->one()), Fsa::Epsilon, Fsa::Epsilon);
191  }
192  else if (startTime_ == 0) {
193  buildState(sp, srcInitialSid);
194  }
195  else {
196  TopologicalOrderQueueRef initialTopologicalQeue = createTopologicalOrderQueue(fsa_, topologicalOrderMap_);
197  TopologicalOrderQueue& Q = *initialTopologicalQeue;
198  Core::Vector<ScoresRef> initialSsspScores(topologicalOrderMap_->maxSid + 1);
199  Q.insert(srcInitialSid);
200  initialSsspScores[srcInitialSid] = semiring_->one();
201  const Lattice& l = *fsa_;
202  bool hasFinal = false;
203  ScoresRef finalScore = semiring_->zero();
204  while (!Q.empty()) {
205  Fsa::StateId sid = Q.top();
206  Q.pop();
207  ConstStateRef sr = l.getState(sid);
208  s32 t = srcBoundaries_->time(sid);
209  if (t >= endTime_)
210  t = endTime_ - 1;
211  ScoresRef score = initialSsspScores[sid];
212  initialSsspScores[sid].reset();
213  if (sr->isFinal()) {
214  hasFinal = true;
215  finalScore = semiring_->collect(
216  finalScore, semiring_->extend(score, sr->weight()));
217  }
218  for (State::const_iterator a = sr->begin(), a_end = sr->end(); a != a_end; ++a) {
219  ConstStateRef targetSr = fsa_->getState(a->target());
220  s32 targetT = srcBoundaries_->time(a->target());
221  if (targetT > startTime_) {
222  ScoresRef arcScore = semiring_->extend(score, a->weight());
223  Action action = ActionNone;
224  if (forceSentenceEndSymbol_) {
225  if (t == endTime_ - 1)
226  action = ActionAddSentenceEndArc;
227  else if (((targetT >= endTime_) || (targetSr->isFinal() && !targetSr->hasArcs())) &&
228  ((a->input() == sentenceEndInput_) || (a->output() == sentenceEndOutput_)))
229  action = ActionAddFinalArc;
230  else
231  action = ActionAddArc;
232  }
233  else {
234  if ((targetT >= endTime_) ||
235  (targetSr->isFinal() && !targetSr->hasArcs() && ((a->input() == Fsa::Epsilon) && (a->output() == Fsa::Epsilon))))
236  action = ActionAddFinalArc;
237  else
238  action = ActionAddArc;
239  }
240  addArc(action, sp, targetSr.get(), arcScore, a->input(), a->output());
241  }
242  else {
243  if (!initialSsspScores[a->target()]) {
244  initialSsspScores[a->target()] = semiring_->extend(score, a->weight());
245  Q.insert(a->target());
246  }
247  else
248  initialSsspScores[a->target()] = semiring_->collect(
249  initialSsspScores[a->target()], semiring_->extend(score, a->weight()));
250  }
251  }
252  }
253  if (hasFinal)
254  sp->newArc(1, finalScore, sentenceEndInput_, sentenceEndOutput_);
255  }
256  return sp;
257  }

264  State* buildState(State* sp, Fsa::StateId srcSid) const {
265  const Lattice& srcL = *fsa_;
266  ConstStateRef srcSr = srcL.getState(srcSid);
267  s32 srcT = srcBoundaries_->time(srcSid);
268  if (srcT >= endTime_)
269  srcT = endTime_ - 1;
270  if (srcSr->isFinal())
271  sp->newArc(1, srcSr->weight(), sentenceEndInput_, sentenceEndOutput_);
272  for (State::const_iterator a = srcSr->begin(), a_end = srcSr->end(); a != a_end; ++a) {
273  ConstStateRef targetSr = srcL.getState(a->target());
274  s32 targetT = srcBoundaries_->time(a->target());
275  Action action = ActionNone;
276  if (forceSentenceEndSymbol_) {
277  if (srcT == endTime_ - 1)
278  action = ActionAddSentenceEndArc;
279  else if (((targetT >= endTime_) || (targetSr->isFinal() && !targetSr->hasArcs())) &&
280  ((a->input() == sentenceEndInput_) && (a->output() == sentenceEndOutput_)))
281  action = ActionAddFinalArc;
282  else
283  action = ActionAddArc;
284  }
285  else {
286  if ((targetT >= endTime_) ||
287  (targetSr->isFinal() && !targetSr->hasArcs() && ((a->input() == Fsa::Epsilon) && (a->output() == Fsa::Epsilon))))
288  action = ActionAddFinalArc;
289  else
290  action = ActionAddArc;
291  }
292  addArc(action, sp, targetSr.get(), a->weight(), a->input(), a->output());

298  FittingLattice(ConstLatticeRef l, s32 startTime, s32 endTime, bool forceSentenceEndSymbol)
299  : SlaveLattice(l), startTime_(startTime), endTime_(endTime), forceSentenceEndSymbol_(forceSentenceEndSymbol) {
300  topologicalOrderMap_ = findTopologicalOrder(l);
301  verify(topologicalOrderMap_ && (topologicalOrderMap_->maxSid != Fsa::InvalidStateId));
302  if (startTime_ == Core::Type<s32>::max)
303  startTime_ = 0;
304  if (endTime_ == Core::Type<s32>::max) {
305  endTime_ = 0;
306  Fsa::StateId sid = 0;
307  for (StateMap::const_iterator it = topologicalOrderMap_->begin(), it_end = topologicalOrderMap_->end();
308  it != it_end; ++it, ++sid)
309  if (*it != Fsa::InvalidStateId)
310  endTime_ = std::max(endTime_, s32(srcBoundaries_->time(sid)));
311  }
312  verify((0 <= endTime_) && (startTime_ <= endTime_));
313  if (forceSentenceEndSymbol_) {
314  switch (Lexicon::us()->alphabetId(l->getInputAlphabet())) {
315  case Lexicon::LemmaAlphabetId:
316  sentenceEndInput_ = Lexicon::us()->sentenceEndLemmaId();
317  break;
318  case Lexicon::LemmaPronunciationAlphabetId:
319  sentenceEndInput_ = Lexicon::us()->sentenceEndLemmaPronunciationId();
320  break;
321  default:
322  sentenceEndInput_ = Fsa::Epsilon;
323  }
324  sentenceEndOutput_ = (l->type() == Fsa::TypeAcceptor) ? sentenceEndInput_ : Fsa::Epsilon;
325  }
326  else
327  sentenceEndInput_ = sentenceEndOutput_ = Fsa::Epsilon;
328  semiring_ = l->semiring();
329  topologicalQeue_ = createTopologicalOrderQueue(l, topologicalOrderMap_);
330  srcBoundaries_ = l->getBoundaries();
331  setBoundaries(Flf::ConstBoundariesRef(new Boundaries(srcBoundaries_, startTime_, endTime_)));
332  if (startTime_ == endTime_) {
333  ScoresRef finalScores = sssp(fsa_->initialStateId());
334  if (semiring_->compare(finalScores, semiring_->one()) != 0)
335  Core::Application::us()->error(
336  "Fitting lattice: Result is a single state lattice with final weight %s != %s;\n"
337  "discard final weight.",
338  semiring_->describe(finalScores, Fsa::HintShowDetails | HintUnscaled).c_str(),
339  semiring_->describe(semiring_->one(), Fsa::HintShowDetails | HintUnscaled).c_str());
340  initialSr_ = ConstStateRef(new State(0, Fsa::StateTagFinal, semiring_->one()));
341  }
342  else if ((startTime_ + 1 == endTime_) && forceSentenceEndSymbol_) {
343  State* sp = new State(0);
344  sp->newArc(1, sssp(fsa_->initialStateId()), sentenceEndInput_, sentenceEndOutput_);
345  initialSr_ = ConstStateRef(sp);
346  finalSr_ = ConstStateRef(buildFinalState(new State(1)));
347  }
348  else {
349  initialSr_ = ConstStateRef(buildInitialState(new State(0)));
350  finalSr_ = ConstStateRef(buildFinalState(new State(1)));
351  }
352  }

475 StaticLatticeRef cleanUp(ConstLatticeRef l) {
476  if (l->initialStateId() == Fsa::InvalidStateId)
477  return StaticLatticeRef();
478 
479  StaticLattice* s = new StaticLattice;
480  s->setDescription("cleaned-up(" + l->describe() + ")");
481  s->setType(l->type());
482  s->setProperties(l->knownProperties(), l->properties());
483  s->addProperties(Fsa::PropertyAcyclic);
484  s->setInputAlphabet(l->getInputAlphabet());
485  if (l->type() == Fsa::TypeTransducer)
486  s->setOutputAlphabet(l->getOutputAlphabet());
487  s->setSemiring(l->semiring());
488  s->setBoundaries(l->getBoundaries());
489  s->setInitialStateId(l->initialStateId());
490 
491  const bool isAcceptor = (l->type() == Fsa::TypeAcceptor);
492  Fsa::ConstAlphabetRef inputA = l->getInputAlphabet();
493  Fsa::ConstAlphabetRef outputA = isAcceptor ? l->getInputAlphabet() : l->getOutputAlphabet();
494  ConstSemiringRef semiring = l->semiring();
495 
496  Core::Vector<Color> C;
497  Core::Vector<Fsa::StateId> S;
498  C.grow(l->initialStateId(), None);
499  S.push_back(l->initialStateId());
500  C[l->initialStateId()] = White;
501  std::vector<bool> isValidArc;
502  while (!S.empty()) {
503  Fsa::StateId sid = S.back();
504  if (C[sid] == White) {
505  C[sid] = Gray;
506  ConstStateRef sr = l->getState(sid);
507  if (!sr)
508  Core::Application::us()->criticalError("State %d does not exist.", sid);
509  State* sp = 0;
510  bool isValidArc = true;
511  for (State::const_iterator a = sr->begin(); a != sr->end(); ++a) {
512  if (isValidTarget(a->target())) {
513  C.grow(a->target(), None);
514  if (C[a->target()] == None) {
515  S.push_back(a->target());
516  C[a->target()] = White;
517  }
518  if (isValidArc) {
519  if ((C[a->target()] == Gray) || !isValidLabel(inputA, a->input()) || !isValidLabel(outputA, a->output()) || (isAcceptor && (a->input() != a->output())) || !isFinite(semiring, a->weight()))
520  isValidArc = false;
521  }
522  }
523  else
524  isValidArc = false;
525  }
526  if (!isValidArc) {
527  u32 nDiscarded = 0, nCorrected = 0;
528  sp = new State(sr->id(), sr->tags(), sr->weight());
529  for (State::const_iterator a = sr->begin(); a != sr->end(); ++a) {
530  bool discard = false;
531  if (!isValidTarget(a->target())) {
532  Core::Application::us()->warning("Arc has invalid target.");
533  discard = true;
534  }
535  else if (C[a->target()] == Gray) {
536  Core::Application::us()->warning("Arc closes a cycle.");
537  discard = true;
538  }
539  if (!isValidLabel(inputA, a->input())) {
540  Core::Application::us()->warning("Arc has invalid input.");
541  discard = true;
542  }
543  if (!isValidLabel(outputA, a->output())) {
544  Core::Application::us()->warning("Arc has invalid ouput.");
545  discard = true;
546  }
547  if (!isFinite(semiring, a->weight())) {
548  Core::Application::us()->warning("Arc weight is not finite.");
549  discard = true;
550  }
551  if (discard) {
552  ++nDiscarded;
553  Core::Application::us()->warning(
554  "Discard arc from=%d,to=%d,i/o=%s(%d):%s(%d),w=%s",
555  sr->id(), a->target(),
556  inputA->symbol(a->input()).c_str(), a->input(),
557  outputA->symbol(a->output()).c_str(), a->output(),
558  semiring->describe(a->weight(), Fsa::HintShowDetails | HintUnscaled).c_str());
559  }
560  else {
561  Arc& newArc = (*sp->newArc() = *a);
562  if (isAcceptor && (a->input() != a->output())) {
563  ++nCorrected;
564  Core::Application::us()->warning("Acceptor arc has different input and output; correct.");
565  newArc.output_ = newArc.input_;
566  }
567  }
568  }
569  verify(nDiscarded + nCorrected > 0);
570  Core::Application::us()->log(
571  "For state %d: %d arcs discarded and %d arcs corrected of %d original arcs",
572  sr->id(), sr->nArcs(), nDiscarded, nCorrected);
573  }
574  else
575  sp = new State(*sr);
576  s->setState(sp);
577  }
578  else {
579  verify(C[sid] == Gray);
580  C[sid] = Black;
581  S.pop_back();
582  }
583  }
584  StaticLatticeRef sRef = StaticLatticeRef(s);
585  trimInPlace(sRef);
586  if (sRef->initialStateId() == Fsa::InvalidStateId)
587  return StaticLatticeRef();
588  return sRef;
589 }
