49 u32 FlacInputNode::readTyped(u32 nSamples, Flow::Timestamp*& d) {
50  require(isFileOpen());
51  require(d == 0);
52  Flow::Vector<T>* v = new Flow::Vector<T>(trackCount_ * nSamples);
53  int samplesRead = fd_->read(nSamples, &*(v->begin()));
54  int bytesRead = samplesRead * trackCount_ * sizeof(T);
55  if (bytesRead <= 0) {
56  if (bytesRead < 0)
57  error("FlacDecode::read failed");
58  delete v;
59  d = 0;
60  return 0;
61  }
62  v->resize(bytesRead / sizeof(T));
63  d = v;
64  return v->size() / trackCount_;
65 }
