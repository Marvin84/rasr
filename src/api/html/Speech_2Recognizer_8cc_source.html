216 void OfflineRecognizer::processResult(Bliss::SpeechSegment* s) {
217  Traceback remainingTraceback;
218  recognizer_->getCurrentBestSentence(remainingTraceback);
219  addPartialToTraceback(remainingTraceback);
220 
221  Core::XmlWriter& os(clog());
222  os << Core::XmlOpen("traceback");
223  traceback_.write(os, lexicon_->phonemeInventory());
224  os << Core::XmlClose("traceback");
225  os << Core::XmlOpen("orth") + Core::XmlAttribute("source", "recognized");
226  for (u32 i = 0; i < traceback_.size(); ++i)
227  if (traceback_[i].pronunciation)
228  os << traceback_[i].pronunciation->lemma()->preferredOrthographicForm()
229  << Core::XmlBlank();
230  os << Core::XmlClose("orth");
231  if (tracebackChannel_.isOpen()) {
232  logTraceback(traceback_);
233  featureTimes_.clear();
234  }
235 
236  /*! \todo Should stop timer here: evaluation may take some time we don't want to count it. */
237 
238  Core::Ref<const Search::LatticeAdaptor> lattice = recognizer_->getCurrentWordLattice();
239  if (lattice && !lattice->empty()) {
240  if (timeConditionedLattice_) {
241  lattice = Core::ref(new Lattice::WordLatticeAdaptor(
242  Lattice::timeConditionedWordLattice(lattice->wordLattice(latticeHandler_))));
243  }
244  if (shouldStoreLattice_) {
245  if (!lattice->write(s->fullName(), latticeHandler_))
246  error("cannot write lattice '%s'", s->fullName().c_str());
247  }
248  }
249  if (tracebackArchiveWriter_) {
250  tracebackArchiveWriter_->store(s->fullName(), traceback_.wordLattice(lexicon_));
251  }
252 
253  if (shouldEvaluateResult_) {
254  evaluator_->setReferenceTranscription(s->orth());
255  evaluator_->evaluate(
256  traceback_.lemmaPronunciationAcceptor(lexicon_),
257  "single best");
258  if (lattice && !lattice->empty()) {
259  Lattice::ConstWordLatticeRef wl = lattice->wordLattice(latticeHandler_);
260  if (wl->nParts() > 0)
261  evaluator_->evaluate(wl->part(0), "lattice");
262  }
263  }
264 }

306 void OfflineRecognizer::logTraceback(const Recognizer::Traceback& traceback) {
307  tracebackChannel_ << Core::XmlOpen("traceback") + Core::XmlAttribute("type", "xml");
308  u32 previousIndex = traceback.begin()->time;
309  Search::SearchAlgorithm::ScoreVector previousScore(0.0, 0.0);
310  for (std::vector<Search::SearchAlgorithm::TracebackItem>::const_iterator tbi = traceback.begin(); tbi != traceback.end(); ++tbi) {
311  if (tbi->pronunciation) {
312  tracebackChannel_ << Core::XmlOpen("item") + Core::XmlAttribute("type", "pronunciation")
313  << Core::XmlFull("orth", tbi->pronunciation->lemma()->preferredOrthographicForm())
314  << Core::XmlFull("phon", tbi->pronunciation->pronunciation()->format(lexicon_->phonemeInventory()))
315  << Core::XmlFull("score", f32(tbi->score.acoustic - previousScore.acoustic)) + Core::XmlAttribute("type", "acoustic")
316  << Core::XmlFull("score", f32(tbi->score.lm - previousScore.lm)) + Core::XmlAttribute("type", "language");
317  if (previousIndex < tbi->time)
318  tracebackChannel_ << Core::XmlEmpty("samples") + Core::XmlAttribute("start", f32(featureTimes_[previousIndex].startTime())) +
319  Core::XmlAttribute("end", f32(featureTimes_[tbi->time - 1].endTime()))
320  << Core::XmlEmpty("features") + Core::XmlAttribute("start", previousIndex) +
321  Core::XmlAttribute("end", tbi->time - 1);
322  tracebackChannel_ << Core::XmlClose("item");
323  }
324  previousScore = tbi->score;
325  previousIndex = tbi->time;
326  }
327  tracebackChannel_ << Core::XmlClose("traceback");
328 }

371 void ConstrainedOfflineRecognizer::enterSpeechSegment(Bliss::SpeechSegment* s) {
372  verify(latticeArchiveReader_);
373  Fsa::ConstAutomatonRef f = latticeArchiveReader_->get(s->fullName(), fsaPrefix_)->mainPart();
374 
375  // map to syntactic tokens
376  if (f->getInputAlphabet() == lemmaPronunciationToLemmaTransducer_->getInputAlphabet()) {
377  f = Fsa::composeMatching(f, lemmaPronunciationToLemmaTransducer_);
378  }
379  require(f->getOutputAlphabet() == lemmaToSyntacticTokenTransducer_->getInputAlphabet());
380  f = Fsa::projectOutput(Fsa::composeMatching(f, lemmaToSyntacticTokenTransducer_));
381 
382  // restrict search space to word sequences in the automaton f
383  Fsa::ConstAutomatonRef g = Fsa::multiply(f, scale_);
384  /*
385  * not yet checked: is new implementation more efficient than the old minimize implementation,
386  * i.e., Fsa::determinize(Fsa::transpose(Fsa::determinize(Fsa::transpose(g))))?
387  */
388  g = Fsa::minimize(Fsa::determinize(Fsa::removeEpsilons(g)));
389  if (lmFsa_) {
390  g = Fsa::composeMatching(g, lmFsa_);
391  }
392 
393  verify(recognizer_);
394  recognizer_->setGrammar(Fsa::staticCopy(g));
395 
396  Precursor::enterSpeechSegment(s);
397 }
