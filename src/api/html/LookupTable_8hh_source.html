35 /**
36  * LookupTable: array of Values indexed by arbitrary numbers of type Index
37  *
38  * Maps the Index interval to the [0..(max_index - min_index) * bucketSize] integer inteval.
39  * Integer interval is automatically extended, if a new Index does not fit into it.
40  *
41  * Values can be reached by
42  * Value& operator[](Index i) (do not extend the integer interval)
43  * iterator (do not extend the integer interval)
44  * iterator insert(Index i, Value init) (extends the integer interval)
45  *
46  * getInverse creates a LookupTable of the inverse function.
47  *
48  */
49 
50 template<class Value, class Index>
51 class LookupTable {
52 public:
53  typedef Value ValueType;
54  typedef Index IndexType;
55 
56  typedef typename std::deque<ValueType>::iterator Iterator;
57  typedef typename std::deque<ValueType>::const_iterator ConstIterator;

59 private:
60  typedef s32 BucketIndex;

62 private:
63  Index bucketSize_; // interval size of indices, which are assigned to the same bucket
64  BucketIndex offset_;
65  std::deque<Value> f_;
66  bool grow_;
67 
68 private:
69  BucketIndex bucket(const Index index) const {
70  verify_(bucketSize() > 0);
71  return (BucketIndex)round(index / bucketSize()) + offset_;
72  }
73  Index index(const BucketIndex b) const {
74  verify_(bucketSize() > 0);
75  return (Index)(b - offset_) * bucketSize();
76  }
77 
78 public:
79  /**
80  * Creates a lookup table with variable amount of buckets.
81  */
82  LookupTable(const Index bucketSize = 0)
83  : bucketSize_(bucketSize), offset_(0), grow_(true) {}
84  /**
85  * Creates a lookup table with constant amount of buckets.
86  * Index values smaller than min or larger than max will be set back to min resp. max.
87  */
88  LookupTable(const Index bucketSize, const Index min, const Index max);

93  }

95  Iterator insert(const Index index, const Value init = 0);
96  ConstIterator find(const Index index, const Value defaultValue = 0) const {
97  BucketIndex b = bucket(index);

174 template<class Value, class Index>
175 typename LookupTable<Value, Index>::Iterator LookupTable<Value, Index>::insert(const Index index, const Value init) {
176  BucketIndex b = bucket(index);
177  if (!f_.empty()) {
178  if (b < 0) {
179  if (grow_) {
180  f_.insert(f_.begin(), -b, init);
181  offset_ -= b;
182  }
183  b = 0;
184  }
185  else if (b >= (BucketIndex)size()) {
186  if (grow_)
187  f_.insert(f_.end(), b - (BucketIndex)size() + 1, init);
188  b = size() - 1;
189  }
190  }
191  else {
192  verify_(grow_);
193  f_.insert(f_.end(), 1, init);
194  offset_ -= b;
195  b = 0;
196  }
197  return f_.begin() + b;
198 }

207 template<class Value, class Index>
208 int LookupTable<Value, Index>::isMonotonous(Value equalityTolerance) const {
209  require(!empty());
210  if (size() < 2)
211  return 1;
212 
213  ConstIterator previous = begin();
214  ConstIterator current = previous + 1;
215  for (; Core::isAlmostEqual(*current, *previous, equalityTolerance) && current != end();
216  ++previous, ++current)
217  ;
218  if (current == end())
219  return 1;
220 
221  bool increasing = (*current > *previous);
222  int result = 2;
223  for (; current != end(); ++previous, ++current) {
224  if (Core::isAlmostEqual(*current, *previous, equalityTolerance)) {
225  result = 1;
226  }
227  else if ((increasing && *current < *previous) ||
228  (!increasing && *current > *previous)) {
229  return (result = 0);
230  }
231  }
232  return result;
233 }

235 template<class Value, class Index>
236 void LookupTable<Value, Index>::getInverse(LookupTable<Index, Value>& inverse) const {
237  verify(isMonotonous());
238 
239  require(inverse.empty());
240  if (inverse.bucketSize() == 0)
241  inverse = LookupTable<Index, Value>(proposeBucketSizeForInverse());
242 
243  // Note: index is used here rather than iterator since iterators become invalid after each call to insert.
244  BucketIndex previousIndex = 0;
245  for (BucketIndex b = 0; b < (BucketIndex)size(); ++b) {
246  Iterator current, previous;
247  *(current = inverse.insert(f_[b])) = index(b);
248  previous = inverse.begin() + previousIndex;
249 
250  // fill skipped buckets
251  if (previous < current)
252  std::fill(previous + 1, current, *current);
253  else if (current < previous)
254  std::fill(current + 1, previous, *current);
255 
256  previousIndex = current - inverse.begin();
257  }
258 }

260 template<class Value, class Index>
261 Value LookupTable<Value, Index>::proposeBucketSizeForInverse() const {
262  static Value enlargementFactor = 2;
263  Value result = (*std::max_element(begin(), end()) - *std::min_element(begin(), end())) /
264  ((Value)size() * enlargementFactor);
265  verify(result > 0);
266  return result;
