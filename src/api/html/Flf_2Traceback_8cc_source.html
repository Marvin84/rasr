83  void dumpBlissTraceback(ConstLatticeRef l, Core::XmlWriter& os) {
84  if (l && (l->initialStateId() != Fsa::InvalidStateId)) {
85  Core::Ref<const Bliss::LemmaPronunciationAlphabet> lpAlphabet =
86  Lexicon::us()->lemmaPronunciationAlphabet();
87  Search::SearchAlgorithm::Traceback traceback;
88  Lexicon::AlphabetId alphabetId = Lexicon::us()->alphabetId(l->getInputAlphabet());
89  if (alphabetId != Lexicon::LemmaPronunciationAlphabetId)
90  warning("DumpTracebackNode: Input alphabet of \"%s\" "
91  "is not lemma pronunciation; map alphabet",
92  l->describe().c_str());
93  const Boundaries& boundaries = *l->getBoundaries();
94  const ScoreList& scales = l->semiring()->scales();
95  ConstStateRef initialSr = l->getState(l->initialStateId());
96  for (Fsa::StateId i = 1, n = initialSr->nArcs(); i <= n; ++i) {
97  ConstLatticeRef p;
98  ConstStateRef sr;
99  if (l->hasProperty(Fsa::PropertyLinear)) {
100  verify(n == 1);
101  p = l;
102  sr = initialSr;
103  }
104  else if (alphabetId == Lexicon::LemmaPronunciationAlphabetId) {
105  p = l;
106  sr = p->getState(i);
107  }
108  else {
109  p = best(mapInput(partial(l, i), MapToLemmaPronunciation));
110  sr = p->getState(p->initialStateId());
111  }
112  traceback.clear();
113  traceback.push_back(
114  Search::SearchAlgorithm::TracebackItem(
115  0,
116  0,
117  Search::SearchAlgorithm::ScoreVector(0.0, 0),
118  Search::SearchAlgorithm::TracebackItem::Transit()));
119  Score score = Semiring::One;
120  for (; sr->hasArcs(); sr = p->getState(sr->begin()->target())) {
121  verify(sr->nArcs() == 1);
122  const Arc& a = *sr->begin();
123  score += a.weight()->project(scales);
124  const Bliss::LemmaPronunciation* lemmaPron = 0;
125  if ((Fsa::FirstLabelId <= a.input()) && (a.input() <= Fsa::LastLabelId)) {
126  lemmaPron = lpAlphabet->lemmaPronunciation(a.input());
127  }
128  else if (a.input() != Fsa::Epsilon) {
129  require(unkLemmaPron_);
130  lemmaPron = unkLemmaPron_;
131  }
132  if (lemmaPron) {
133  const Boundary& b = boundaries.get(a.target());
134  traceback.push_back(
135  Search::SearchAlgorithm::TracebackItem(
136  lemmaPron,
137  b.time(),
138  Search::SearchAlgorithm::ScoreVector(score, 0),
139  Search::SearchAlgorithm::TracebackItem::Transit()));
140  }
141  }
142  verify(sr->isFinal());
143  score += sr->weight()->project(scales);
144  const Boundary& b = boundaries.get(sr->id());
145  traceback.push_back(
146  Search::SearchAlgorithm::TracebackItem(
147  0,
148  b.time(),
149  Search::SearchAlgorithm::ScoreVector(score, 0),
150  Search::SearchAlgorithm::TracebackItem::Transit()));
151  Core::XmlOpen tracebackOpen("traceback");
152  tracebackOpen + Core::XmlAttribute("source", "recognized");
153  tracebackOpen + Core::XmlAttribute("n", i);
154  if (segment_) {
155  if (segment_->hasSegmentId())
156  tracebackOpen + Core::XmlAttribute("name", segment_->segmentId());
157  if (segment_->hasTrack())
158  tracebackOpen + Core::XmlAttribute("track", segment_->track());
159  if (segment_->hasStartTime())
160  tracebackOpen + Core::XmlAttribute("begin", segment_->startTime());
161  if (segment_->hasEndTime())
162  tracebackOpen + Core::XmlAttribute("end", segment_->endTime());
163  }
164  os << tracebackOpen;
165  traceback.write(os, Lexicon::us()->phonemeInventory());
166  os << Core::XmlClose("traceback");
167  }
168  }
169  else {
170  Core::XmlEmpty tracebackEmpty("traceback");
171  tracebackEmpty + Core::XmlAttribute("source", "recognized");
172  tracebackEmpty + Core::XmlAttribute("n", 0);
173  if (segment_) {
174  if (segment_->hasSegmentId())
175  tracebackEmpty + Core::XmlAttribute("name", segment_->segmentId());
176  if (segment_->hasTrack())
177  tracebackEmpty + Core::XmlAttribute("track", segment_->track());
178  if (segment_->hasStartTime())
179  tracebackEmpty + Core::XmlAttribute("begin", segment_->startTime());
180  if (segment_->hasEndTime())
181  tracebackEmpty + Core::XmlAttribute("end", segment_->endTime());
182  }
183  os << tracebackEmpty;
184  }
185  }

226  void dumpCtmTraceback(ConstLatticeRef l, std::ostream& os) {
227  CtmPrinter cp(os);
228  f32 begin = 0.0;
229  f32 end = Core::Type<f32>::max;
230  if (segment_) {
231  if (segment_->hasRecordingId())
232  cp.name = segment_->recordingId();
233  else if (segment_->hasAudioFilename())
234  cp.name = Core::stripExtension(Core::baseName(segment_->audioFilename()));
235  else if (segment_->hasSegmentId())
236  cp.name = segment_->segmentId();
237  if (segment_->hasTrack())
238  cp.track = segment_->track() + 1;
239  if (segment_->hasStartTime())
240  begin = segment_->startTime();
241  if (segment_->hasEndTime())
242  end = segment_->endTime();
243  }
244  if (l && (l->initialStateId() != Fsa::InvalidStateId)) {
245  bool segment_is_empty = true;
246  bool header_printed = false;
247  Lexicon::AlphabetId alphabetId = Lexicon::us()->alphabetId(l->getInputAlphabet());
248  Lexicon::LemmaAlphabetRef lAlphabet;
249  Lexicon::LemmaPronunciationAlphabetRef lpAlphabet;
250  Fsa::ConstAlphabetRef alphabet;
251  LabelMapRef nonWordToEpsilonMap;
252  bool mapToLemma = false;
253  switch (alphabetId) {
254  case Lexicon::LemmaAlphabetId:
255  alphabet = lAlphabet = Lexicon::us()->lemmaAlphabet();
256  nonWordToEpsilonMap = LabelMap::createNonWordToEpsilonMap(Lexicon::LemmaAlphabetId);
257  break;
258  case Lexicon::LemmaPronunciationAlphabetId:
259  alphabet = lpAlphabet = Lexicon::us()->lemmaPronunciationAlphabet();
260  nonWordToEpsilonMap = LabelMap::createNonWordToEpsilonMap(Lexicon::LemmaPronunciationAlphabetId);
261  break;
262  default:
263  if (dumpPhonemeAlignment_ || dumpSubwordAlignment_)
264  criticalError("Acoustic alignment requires lemma or lemma-pronunciation as input, not %s",
265  Lexicon::us()->alphabetName(alphabetId).c_str());
266  if (dumpOrthography_) {
267  alphabet = lAlphabet = Lexicon::us()->lemmaAlphabet();
268  nonWordToEpsilonMap = LabelMap::createNonWordToEpsilonMap(Lexicon::LemmaAlphabetId);
269  mapToLemma = true;
270  }
271  else
272  alphabet = l->inputAlphabet();
273  }
274  ConstSemiringRef semiring = l->semiring();
275  for (KeyList::const_iterator itKey = scoreKeys_.begin(); itKey != scoreKeys_.end(); ++itKey) {
276  cp.scoreIds.push_back(semiring->id(*itKey));
277  if (cp.scoreIds.back() == Semiring::InvalidId)
278  criticalError("DumpTracebackNode: Dimension \"%s\" does not exist.",
279  itKey->c_str());
280  }
281  const Boundaries& boundaries = *l->getBoundaries();
282  ConstLatticeAlignmentRef latticeAlignment;
283  if (dumpPhonemeAlignment_ || dumpSubwordAlignment_) {
284  latticeAlignment = getLatticeAlignment(l);
285  }
286  ConstStateRef initialSr = l->getState(l->initialStateId());
287  if (!l->hasProperty(Fsa::PropertyLinear))
288  printAsText(os << ";; ", segment_) << " [1.." << initialSr->nArcs() << "]-best" << std::endl;
289  for (Fsa::StateId i = 1, n = initialSr->nArcs(); i <= n; ++i) {
290  ConstLatticeRef p;
291  ConstStateRef sr;
292  if (l->hasProperty(Fsa::PropertyLinear)) {
293  verify(n == 1);
294  p = l;
295  sr = initialSr;
296  cp.tail = "";
297  }
298  else {
299  if (!mapToLemma) {
300  p = l;
301  sr = p->getState(i);
302  }
303  else {
304  p = best(mapInput(partial(l, i), MapToLemmaPronunciation));
305  sr = p->getState(p->initialStateId());
306  }
307  cp.tail = Core::form(" %d-best", i);
308  }
309  cp.printHeader(segment_);
310  header_printed = true;
311  for (; sr->hasArcs(); sr = p->getState(sr->begin()->target())) {
312  verify(sr->nArcs() == 1);
313  const Arc& a = *sr->begin();
314  const Boundary &leftBoundary = boundaries.get(sr->id()), &rightBoundary = boundaries.get(a.target());
315  f32 wordBegin = f32(leftBoundary.time()) / 100.00;
316  f32 wordEnd = f32(rightBoundary.time()) / 100.00;
317  if (wordBegin < wordEnd) {
318  if (lAlphabet || lpAlphabet) {
319  std::string word;
320  CtmPrinter::Type type = CtmPrinter::TypeInvalid;
321  if ((Fsa::FirstLabelId <= a.input()) && (a.input() <= Fsa::LastLabelId)) {
322  if (dumpOrthography_) {
323  word = (lAlphabet) ? std::string(lAlphabet->lemma(a.input())->preferredOrthographicForm()) : lpAlphabet->lemmaPronunciation(a.input())->lemma()->preferredOrthographicForm();
324  }
325  else
326  word = alphabet->symbol(a.input());
327  if (dumpCoarticulation_) {
328  word = Core::form("/%s/%c %s %c/%s/",
329  ((leftBoundary.transit().initial == Bliss::Phoneme::term) ? "#" : Lexicon::us()->phonemeInventory()->phonemeAlphabet()->symbol(leftBoundary.transit().initial).c_str()),
330  ((leftBoundary.transit().boundary == AcrossWordBoundary) ? '|' : '+'),
331  word.c_str(),
332  ((rightBoundary.transit().boundary == AcrossWordBoundary) ? '|' : '+'),
333  ((rightBoundary.transit().initial == Bliss::Phoneme::term) ? "#" : Lexicon::us()->phonemeInventory()->phonemeAlphabet()->symbol(rightBoundary.transit().final).c_str()));
334  }
335  if ((*nonWordToEpsilonMap)[a.input()].empty()) {
336  if (leftBoundary.transit().boundary == AcrossWordBoundary) {
337  if (rightBoundary.transit().boundary == AcrossWordBoundary)
338  type = CtmPrinter::TypeLexical;
339  else
340  type = CtmPrinter::TypeSubLexicalBegin;
341  }
342  else {
343  if (rightBoundary.transit().boundary == AcrossWordBoundary)
344  type = CtmPrinter::TypeSubLexicalEnd;
345  else
346  type = CtmPrinter::TypeSubLexicalWithin;
347  }
348  }
349  else if (dumpNonWord_) {
350  verify_((*nonWordToEpsilonMap)[a.input()].front().label == Fsa::Epsilon);
351  type = CtmPrinter::TypeNonLexical;
352  }
353  }
354  else if (dumpEps_) {
355  word = epsSymbol_;
356  type = CtmPrinter::TypeNonLexical;
357  }
358  // check whether orthography is a noise
359  if (!dumpNoises_ && Core::startsWith(word, noisePrefix_)) {
360  continue;
361  }
362  // dump
363  if (type != CtmPrinter::TypeInvalid) {
364  f32 absoluteBegin = begin + wordBegin, duration = wordEnd - wordBegin;
365  cp.print(absoluteBegin, duration, word, *a.weight(), (dumpType_ ? type : CtmPrinter::TypeInvalid));
366  segment_is_empty = false;
367  if (dumpPhonemeAlignment_) {
368  if (!dumpPhonemeAlignment(cp, latticeAlignment, sr, absoluteBegin, duration))
369  warning() << "No phoneme alignment available for \"" << absoluteBegin << " " << duration << " " << word << "\"";
370  }
371  if (dumpSubwordAlignment_) {
372  if (!dumpSubwordAlignment(cp, latticeAlignment, sr, absoluteBegin, duration))
373  warning() << "No sub-word alignment available for \"" << absoluteBegin << " " << duration << " " << word << "\"";
374  }
375  }
376  }
377  else {
378  if ((Fsa::FirstLabelId <= a.input()) && (a.input() <= Fsa::LastLabelId)) {
379  cp.print(begin + wordBegin, wordEnd - wordBegin, alphabet->symbol(a.input()), *a.weight());
380  segment_is_empty = false;
381  }
382  else if (dumpEps_) {
383  cp.print(begin + wordBegin, wordEnd - wordBegin, epsSymbol_, *a.weight());
384  segment_is_empty = false;
385  }
386  }
387  }
388  }
389  // check (last) wordEnd against end
390  }
391  if (segment_is_empty and fillEmptySegments_) {
392  if (not header_printed) {
393  cp.printHeader(segment_);
394  }
395  cp.print(begin, end - begin, epsSymbol_, *l->semiring()->one());
396  }
397  }
398  else {
399  cp.printHeader(segment_);
400  if (fillEmptySegments_) {
401  cp.print(begin, end - begin, epsSymbol_, *l->semiring()->one());
402  }
403  }
404  }

419  void dumpOrthTraceback(ConstLatticeRef l, Core::XmlWriter& os) {
420  if (l && (l->initialStateId() != Fsa::InvalidStateId)) {
421  ConstStateRef initialSr = l->getState(l->initialStateId());
422  verify(initialSr->nArcs() == 1);
423  verify(l->hasProperty(Fsa::PropertyLinear));
424 
425  Fsa::ConstAlphabetRef alphabet = l->getInputAlphabet();
426 
427  ConstLatticeRef p = l;
428  os << Core::XmlOpen("orth");
429  for (ConstStateRef sr = initialSr; sr->hasArcs(); sr = p->getState(sr->begin()->target())) {
430  verify(sr->nArcs() == 1);
431  const Arc& a = *sr->begin();
432  if ((Fsa::FirstLabelId <= a.input()) && (a.input() <= Fsa::LastLabelId))
433  os << std::string(alphabet->symbol(a.input())) << " ";
434  }
435  os << Core::XmlClose("orth");
436  }

439  void dumpTraceback(ConstLatticeRef l, std::ostream& os) {
440  if (l && (l->initialStateId() != Fsa::InvalidStateId)) {
441  require(l->getBoundaries()->valid());
442  }
443  switch (format_) {
444  case 0: {
445  Core::XmlWriter xml(os);
446  xml.generateFormattingHints();
447  dumpBlissTraceback(l, xml);
448  break;
449  }
450  case 1:
451  dumpCtmTraceback(l, os);
452  break;
453  case 2: {
454  Core::XmlWriter xml(os);
455  xml.generateFormattingHints();
456  dumpOrthTraceback(l, xml);
457  break;
458  }
459  default:
460  defect();
461  }
462  }

464  virtual ConstLatticeRef filter(ConstLatticeRef l) {
465  if (connected(1))
466  segment_ = requestSegment(1);
467  if (dump_ && dump_.isOpen()) {
468  dumpTraceback(l, dump_);
469  }
470  else {
471  dumpTraceback(l, clog());
472  }
473  return l;
474  }

483  virtual void init(const std::vector<std::string>& arguments) {
484  unkLemma_ = (Lexicon::us()->unkLemmaId() == Fsa::InvalidLabelId) ? 0 : Lexicon::us()->lemmaAlphabet()->lemma(Lexicon::us()->unkLemmaId());
485  unkLemmaPron_ = (Lexicon::us()->unkLemmaPronunciationId() == Fsa::InvalidLabelId) ? 0 : Lexicon::us()->lemmaPronunciationAlphabet()->lemmaPronunciation(Lexicon::us()->unkLemmaPronunciationId());
486  format_ = paramTracebackFormat(config);
487  switch (format_) {
488  case 0:
489  break;
490  case 1: {
491  const Core::Configuration ctmConfig = select("ctm");
492  dumpOrthography_ = paramDumpOrthography(ctmConfig);
493  dumpCoarticulation_ = paramDumpCoarticulation(ctmConfig);
494  dumpNonWord_ = paramDumpNonWord(ctmConfig);
495  dumpNoises_ = paramDumpNoises(ctmConfig);
496  dumpEps_ = paramDumpEps(ctmConfig, dumpNonWord_);
497  nonWordSymbol_ = paramNonWordSymbol(ctmConfig);
498  noisePrefix_ = paramNoisePrefix(ctmConfig);
499  epsSymbol_ = nonWordSymbol_.empty() ? "!NULL" : nonWordSymbol_;
500  scoreKeys_ = paramScoreKeys(ctmConfig);
501  dumpType_ = paramDumpType(ctmConfig);
502  dumpPhonemeAlignment_ = paramDumpPhonemeAlignment(ctmConfig);
503  dumpSubwordAlignment_ = paramDumpSubwordAlignment(ctmConfig);
504  fillEmptySegments_ = paramFillEmptySegments(ctmConfig);
505  if (dumpPhonemeAlignment_ || dumpSubwordAlignment_) {
506  createAlignmentBuilder(ctmConfig);
507  }
508  if (dump_ && dump_.isOpen()) {
509  dump_ << ";; <name> <track> <start> <duration> <word>";
510  for (KeyList::const_iterator itKey = scoreKeys_.begin(); itKey != scoreKeys_.end(); ++itKey)
511  dump_ << " <" << *itKey << ">";
512  if (dumpType_)
513  dump_ << " <type>";
514  dump_ << " [<n-best>]" << std::endl;
515  }
516  break;
517  }
518  case 2:
519  break;
520  default:
521  defect();
522  }
523  }
