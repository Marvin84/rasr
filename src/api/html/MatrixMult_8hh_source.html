45 /**
46  * Matrix multiplication filter.
47  * Each input vector is multiplied with a constant matrix. The
48  * result is emitted on output port. The dimension of the input
49  * vector must match the number of columns of the matrix. The
50  * output vector dimension will be equal to the number of row of the
51  * matrix.
52  * Parameters:
53  * -@c file is a path name to load the matrix from.
54  * Since MatrixMultiplicationNode derives from Flow::StringExpressionNode, @c file can
55  * contain placeholders for input ports. Example: file="$input(warping-factor).matrix".
56  * In this case, node has got an extra input port called warping-factor. The placeholder
57  * $input(warping-factor) will be replaced by the values which arrives on the input port.
58  * For more details @see Flow::StringExpressionNode.
59  * -$c use-cache activates/deactivates a cache for the matrices.
60  * -$c multi-data <number of expected parallel vectors> activates/deactivates
61  * processing of more than one vector per timestamp
62  * In this case input ports have to be numerical names, i.e. 0 , 1 ,2 ...
63  * The result of the multiplication of input vector 0 is then available at output port 0.
64  * Not that all vectors will multiplied with the very same matrix.
65  *
66  * Inputs: one or more
67  * Outputs: one or more
68  */

190 template<class T>
191 bool MatrixMultiplicationNode<T>::work(Flow::PortId p) {
192  std::vector<Flow::DataPtr<Flow::Vector<T>>> in;
193  for (s32 i = 0; i < multiData_; ++i) {
194  if (!inputConnected((Flow::PortId)i)) {
195  continue;
196  }
197  Flow::DataPtr<Flow::Vector<T>> inData;
198  if (!getData((Flow::PortId)i, inData)) {
199  return putDataAllPorts(inData.get());
200  }
201  in.push_back(inData);
202  }
203 
204  bool doUpdate = false;
205  u32 portNumber = 0;
206  bool retrval = true;
207 
208  if (ignoreTimestamps_)
209  doUpdate = Flow::StringExpressionNode::update();
210  else
211  doUpdate = Flow::StringExpressionNode::update(*(in[0]));
212  if (doUpdate)
213  updateMatrix(StringExpressionNode::value());
214  if (!currentMatrix_) {
215  u32 portNumber = 0;
216  bool retrval = true;
217  error() << "no matrix for multiplication";
218  for (s32 i = 0; i < multiData_; ++i) {
219  if (nOutputLinks((Flow::PortId)i)) {
220  Flow::Vector<T>* out = new Flow::Vector<T>(*(in[portNumber]));
221  out->setTimestamp(*(in[portNumber]));
222  retrval = putData((Flow::PortId)portNumber, out) && retrval;
223  ++portNumber;
224  }
225  }
226  return retrval;
227  }
228 
229  for (s32 i = 0; i < multiData_; ++i) {
230  if (inputConnected((Flow::PortId)i)) {
231  if (in[portNumber]->size() != currentMatrix_->nColumns()) {
232  error() << "vector/matrix dimension mismatch: vector(at port " << portNumber << " ): " << in[portNumber]->size()
233  << ", matrix " << currentMatrix_->nColumns() << " columns";
234  return putDataAllPorts(Flow::Data::eos());
235  } // end inner if
236  ++portNumber;
237  } // end connected if
238  }
239 
240  portNumber = 0;
241  for (s32 i = 0; i < multiData_; ++i) {
242  if (nOutputLinks((Flow::PortId)i)) {
243  Flow::Vector<T>* out = new Flow::Vector<T>((*currentMatrix_) * (*(in[portNumber])));
244  out->setTimestamp(*(in[portNumber]));
245  retrval = putData((Flow::PortId)portNumber, out) && retrval;
246  ++portNumber;
247  }
248  }
249 
250  return retrval;
251 }

266 template<typename T>
267 bool MatrixMultiplicationNode<T>::updateMatrix(const std::string& filename) {
268  if (filename.empty()) {
269  error() << "update matrix filename is empty.";
270  }
271  if (filename == currentMatrixFile_)
272  return (bool)currentMatrix_;
273  if (matrixCache_) {
274  typename MatrixCache::const_iterator m = matrixCache_->find(filename);
275  if (m != matrixCache_->end())
276  currentMatrix_ = m->second;
277  else {
278  currentMatrix_ = loadMatrix(filename);
279  currentMatrixFile_ = filename;
280  if (currentMatrix_)
281  matrixCache_->insert(std::make_pair(filename, currentMatrix_));
282  }
283  }
284  else {
285  currentMatrix_ = loadMatrix(filename);
286  currentMatrixFile_ = filename;
287  }
288  return (bool)currentMatrix_;
289 }
