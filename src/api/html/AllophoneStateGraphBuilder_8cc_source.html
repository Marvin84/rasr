63 void AllophoneStateGraphBuilder::setSilencesAndNoises(const std::vector<std::string>& silencesAndNoises) {
64  verify(silencesAndNoises_.empty());
65  std::vector<std::string>::const_iterator noiseIt = silencesAndNoises.begin();

67  std::string noise(*noiseIt);

69  const Bliss::Lemma* lemma = lexicon_->lemma(noise);
70  if (lemma) {

72  addSilenceOrNoise(lemma);
73  }

75  warning("did not find a pronunciation");

77  }
78  else {

113 Core::Ref<Fsa::StaticAutomaton> AllophoneStateGraphBuilder::allophoneStateToPhonemeTransducer() {
114  if (!allophoneStateToPhonemeTransducer_) {
115  Core::Ref<const Bliss::PhonemeAlphabet> phonemeAlphabet(
116  dynamic_cast<const Bliss::PhonemeAlphabet*>(
117  phonemeToLemmaPronunciationTransducer()->getInputAlphabet().get()));
118 
119  Core::Ref<Am::TransducerBuilder> tb = acousticModel_->createTransducerBuilder();
120 
121  tb->setDisambiguators(phonemeAlphabet->nDisambiguators());
122  tb->selectAllophonesFromLexicon();
123 
124  // for efficiency reasons, precompute flat allophoneStateToPhoneme transducer
125  // without loop and skip transitions and apply transition model (i.e. loops
126  // and skips) afterwards on the final transducer
127  tb->selectFlatModel();
128 
129  tb->selectAllophoneStatesAsInput();
130  auto allophoneStateToPhonemeTransducer = tb->createPhonemeLoopTransducer();
131  allophoneStateToPhonemeTransducer_ =
132  Fsa::staticCompactCopy(Fsa::sort(allophoneStateToPhonemeTransducer, Fsa::SortTypeByOutput));
133 
134  // To accelerate the application of context dependency, it would be nice, if allophoneStateToPhonemeTransducer
135  // was deterministic wrt. its output symbols:
136  // allophoneStateToPhonemeTransducer_ =
137  // Fsa::ConstAutomatonRef(Fsa::staticCompactCopy(Fsa::invert(Fsa::determinize(Fsa::invert(allophoneStateToPhonemeTransducer_)))));
138  // Unfortunately, this is currently not possible due to ambiguities at word boundaries
139  }
140  return allophoneStateToPhonemeTransducer_;
141 }

194 Fsa::ConstAutomatonRef AllophoneStateGraphBuilder::buildTransducer(std::string const& orth, std::string const& leftContextOrth, std::string const& rightContextOrth) {
195  Core::Vector<Fsa::ConstAutomatonRef> lemma_acceptors;
196  if (not leftContextOrth.empty()) {
197  lemma_acceptors.push_back(Fsa::allSuffixes(orthographicParser().createLemmaAcceptor(rightContextOrth)));
198  }
199  lemma_acceptors.push_back(orthographicParser().createLemmaAcceptor(orth));
200  if (not rightContextOrth.empty()) {
201  lemma_acceptors.push_back(Fsa::allPrefixes(orthographicParser().createLemmaAcceptor(rightContextOrth)));
202  }
203 
204  if (lemma_acceptors.size() == 1) {
205  return buildTransducer(lemma_acceptors[0]);
206  }
207  else {
208  return buildTransducer(Fsa::determinize(Fsa::removeEpsilons(Fsa::concat(lemma_acceptors))));
209  }
210 }

216 Fsa::ConstAutomatonRef AllophoneStateGraphBuilder::buildTransducer(
217  Fsa::ConstAutomatonRef lemmaAcceptor) {
218  if (modelChannel_.isOpen()) {
219  Fsa::info(lemmaAcceptor, modelChannel_);
220  Fsa::drawDot(lemmaAcceptor, "/tmp/lemma-acceptor.dot");
221  Fsa::write(lemmaAcceptor, "bin:/tmp/lemma-acceptor.binfsa.gz");
222  }
223 
224  require(lemmaAcceptor->type() == Fsa::TypeAcceptor);
225  require(lemmaAcceptor->getInputAlphabet() == lexicon_->lemmaAlphabet());
226 
227  Fsa::AutomatonCounts counts;
228 
229  // remove silence and phrases by choosing the shortest path for a flat model acceptor
230  // we trim in order to check for not empty but incomplete graphs without final states
231  lemmaPronunciationToLemmaTransducer()->setSemiring(lemmaAcceptor->semiring());
232  Fsa::ConstAutomatonRef lemmaPronunciationAcceptor =
233  Fsa::projectOutput(
234  Fsa::trim(
235  Fsa::composeMatching(
236  flatModelAcceptor_ ? Fsa::best(Fsa::extend(lemmaAcceptor, Fsa::Weight(1.0))) : Fsa::ConstAutomatonRef(lemmaAcceptor),
237  Fsa::invert(lemmaPronunciationToLemmaTransducer()))));
238 
239  if (modelChannel_.isOpen()) {
240  Fsa::info(lemmaPronunciationAcceptor, modelChannel_);
241  Fsa::drawDot(lemmaPronunciationAcceptor, "/tmp/lemma-pronunciation.dot");
242  Fsa::write(lemmaPronunciationAcceptor, "bin:/tmp/lemma-pronunciation.binfsa.gz");
243  }
244 
245  if (lemmaPronunciationAcceptor->initialStateId() == Fsa::InvalidStateId)
246  criticalError("lemma-pronuncation graph is empty. Probably the current sentence contains a word that has no pronunciation.");
247 
248  phonemeToLemmaPronunciationTransducer()->setSemiring(lemmaAcceptor->semiring());
249  Fsa::ConstAutomatonRef phon = Fsa::trim(
250  Fsa::composeMatching(phonemeToLemmaPronunciationTransducer(), lemmaPronunciationAcceptor));
251  if (modelChannel_.isOpen()) {
252  Fsa::info(phon, modelChannel_);
253  Fsa::drawDot(phon, "/tmp/phon.dot");
254  Fsa::write(phon, "bin:/tmp/phon.binfsa.gz");
255  }
256 
257  if (phon->initialStateId() == Fsa::InvalidStateId)
258  criticalError("phoneme graph is empty. Probably the current sentence contains a word that has no pronunciation.");
259 
260  // remove pronunciation variants
261  if (flatModelAcceptor_)
262  phon = Fsa::best(phon);
263 
264  allophoneStateToPhonemeTransducer()->setSemiring(lemmaAcceptor->semiring());
265  Fsa::ConstAutomatonRef model = Fsa::trim(
266  Fsa::composeMatching(allophoneStateToPhonemeTransducer(), phon));
267  if (modelChannel_.isOpen()) {
268  Fsa::info(model, modelChannel_);
269  Fsa::drawDot(model, "/tmp/allophon.dot");
270  Fsa::write(model, "bin:/tmp/allophon.binfsa.gz");
271  }
272 
273  if (!flatModelAcceptor_) {
274  model = Fsa::cache(model);
275  Core::Ref<Am::TransducerBuilder> tb = acousticModel_->createTransducerBuilder();
276  tb->selectAllophoneStatesAsInput();
277  tb->selectTransitionModel();
278  tb->setDisambiguators(1); // word end disambiguators
279  model = tb->applyTransitionModel(model);
280  }
281 
282  if (modelChannel_.isOpen()) {
283  Fsa::info(model, log());
284  Fsa::drawDot(model, "/tmp/states.dot");
285  Fsa::write(model, "bin:/tmp/states.binfsa.gz", Fsa::storeStates);
286  auto modelNoEps = Fsa::removeEpsilons(Fsa::removeDisambiguationSymbols(Fsa::projectInput(model)));
287  Fsa::drawDot(modelNoEps, "/tmp/states-no-eps.dot");
288  Fsa::write(modelNoEps, "bin:/tmp/states-no-eps.binfsa.gz", Fsa::storeStates);
289  }
290  if (model->initialStateId() == Fsa::InvalidStateId)
291  criticalError("allophone-state graph is empty.");
292 
293  return model;
294 }

337 Fsa::ConstAutomatonRef AllophoneStateGraphBuilder::createAlignmentGraph(const Alignment& alignment) {
338  Core::Ref<Fsa::StaticAutomaton> f(new Fsa::StaticAutomaton());
339  f->setSemiring(Fsa::TropicalSemiring);
340  f->setInputAlphabet(acousticModel_->allophoneStateAlphabet());
341  f->setType(Fsa::TypeAcceptor);
342  f->addProperties(Fsa::PropertyStorage | Fsa::PropertySortedByInput |
343  Fsa::PropertySortedByOutput | Fsa::PropertyAcyclic);
344 
345  Fsa::State* sp = 0;
346  Fsa::StateId sid = f->newState(Fsa::StateTagNone, f->semiring()->one())->id();
347  f->setInitialStateId(sid);
348  for (Alignment::const_iterator al = alignment.begin(); al != alignment.end(); ++al) {
349  sp = f->fastState(sid);
350  sid = f->newState(Fsa::StateTagNone, f->semiring()->one())->id();
351  sp->newArc(sid, f->semiring()->one(), al->emission);
352  }
353  f->fastState(sid)->addTags(Fsa::StateTagFinal);
354  f->normalize();
355  return f;
356 }
