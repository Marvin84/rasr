75 template<typename T>
76 void PoolingLayer<T>::_forward(const std::vector<NnMatrix*>& input, NnMatrix& output, bool reset) {
77  timeval start, end;
78  gettimeofday(&start, NULL);
79 
80  if (poolingPnorm_ == 0) {
81  argmax_.resize(output.nRows(), output.nColumns());
82  output.addPoolingMax(*(input.at(0)), argmax_, poolingSize_, poolingAbs_);
83  }
84  else {
85  output.addPoolingPnorm(*(input.at(0)), poolingSize_, poolingPnorm_);
86  }
87 
88  gettimeofday(&end, NULL);
89  timeForwardLinear_ += Core::timeDiff(start, end);
90 }

92 template<typename T>
93 void PoolingLayer<T>::_backpropagateWeights(const NnMatrix& errorSignalIn, std::vector<NnMatrix*>& errorSignalOut) {
94  timeval start, end;
95  gettimeofday(&start, NULL);
96 
97  if (poolingPnorm_ == 0) {
98  require_eq(argmax_.nRows(), errorSignalIn.nRows());
99  errorSignalOut.at(0)->backpropPoolingMax(argmax_, errorSignalIn);
100  }
101  else {
102  errorSignalOut.at(0)->backpropPoolingPnorm(errorSignalIn, poolingSize_, poolingPnorm_);
103  }
104 
105  gettimeofday(&end, NULL);
106  timeBackward_ += Core::timeDiff(start, end);
107 }
