144 void composeBoundaries(ConstLatticeRef composition, ConstLatticeRef l, ConstLatticeRef r) {
145  ConstBoundariesRef b = InvalidBoundaries;
146  if (l->getBoundaries()->valid()) {
147  if (r->getBoundaries()->valid()) {
148  if (l->getBoundaries() != r->getBoundaries()) {
149  Core::Application::us()->warning(
150  "In \"%s\": Both child lattices define boundaries, "
151  "keep boundaries from the left lattice.",
152  composition->describe().c_str());
153  b = ConstBoundariesRef(new MappedBoundaries(
154  l->getBoundaries(),
155  FtlWrapper::mapToLeft(composition)));
156  }
157  else
158  b = ConstBoundariesRef(new MappedBoundaries(
159  l->getBoundaries(),
160  FtlWrapper::mapToLeft(composition)));
161  }
162  else
163  b = ConstBoundariesRef(new MappedBoundaries(
164  l->getBoundaries(),
165  FtlWrapper::mapToLeft(composition)));
166  }
167  else if (r->getBoundaries()->valid()) {
168  b = ConstBoundariesRef(new MappedBoundaries(
169  r->getBoundaries(),
170  FtlWrapper::mapToRight(composition)));
171  }
172  composition->setBoundaries(b);
173 }

264 ConstLatticeRef intersect(ConstLatticeRef l, ConstLatticeRef r, bool appendScores) {
265  if (appendScores) {
266  ConstSemiringRef semiring = appendSemiring(l->semiring(), r->semiring());
267  r = offsetSemiring(r, semiring, l->semiring()->size());
268  l = offsetSemiring(l, semiring, 0);
269  }
270  else
271  r = unifySemirings(l, r);
272  if (r->type() != Fsa::TypeAcceptor)
273  r = projectOutput(r);
274  ConstLatticeRef k = composeSequencing(l, r);
275  return k;
276 }

325 class ComposeWithFsaNode : public RescoreSingleDimensionNode {
326 public:
327  static const Core::ParameterString paramFile;
328  static const Core::ParameterString paramName;
329 
330 private:

336 protected:
337  virtual ConstLatticeRef rescore(ConstLatticeRef l1, ScoreId id) {
338  ConstLatticeRef l2;
339  if (connected(1)) {
340  Fsa::ConstAutomatonRef fsa = Fsa::ConstAutomatonRef(static_cast<const Fsa::Automaton*>(requestData(1)));
341  fsa->releaseReference();
342  if (scale() != Semiring::DefaultScale)
343  fsa = Fsa::multiply(fsa, Fsa::Weight(scale()));
344  l2 = fromFsa(fsa_, l1->semiring(), l1->semiring()->one(), id, inputAlphabetId_, outputAlphabetId_);
345  }
346  else {
347  verify(fsa_);
348  if (!cachedL2_ || (l1->semiring().get() != cachedL2_->semiring().get())) {
349  cachedL2_ = persistent(fromFsa(fsa_, l1->semiring(), l1->semiring()->one(), id, inputAlphabetId_, outputAlphabetId_));
350  }
351  l2 = cachedL2_;
352  }
353  return composeSequencing(l1, l2);
354  }
