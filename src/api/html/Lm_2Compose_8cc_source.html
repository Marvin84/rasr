33  struct State {
34  Fsa::StateId left;
35  Lm::History right;
36 
37  State(Fsa::StateId ll, Lm::History rr)
38  : left(ll), right(rr) {}

40  struct Hash {
41  size_t operator()(const State& st) const {
42  return (size_t(st.left) * 2239) + st.right.hashKey();
43  }
44  };
45 
46  bool operator==(const State& st) const {
47  return (left == st.left) && (right == st.right);
48  }
49  };

51  mutable Fsa::Hash<State, State::Hash> states_;
52 
53  Fsa::StateId insertState(Fsa::StateId ll, Lm::History rr) const {
54  Fsa::StateId id = states_.insert(State(ll, rr));
55  hope(id <= Fsa::StateIdMask);
56  return id;
57  }

59 protected:
60  /**

111  virtual void dumpState(Fsa::StateId si, std::ostream& os) const {
112  if (si >= states_.size()) {
113  os << "unknown";
114  return;
115  }
116  os << "(";
117  left_->dumpState(states_[si].left, os);
118  os << ", "
119  << right_->formatHistory(states_[si].right)
120  << ")";
121  }
122 
123  virtual Fsa::ConstStateRef getState(Fsa::StateId si) const {
124  require(si < states_.size());
125  State state(states_[si]);
126  Fsa::State* result = new Fsa::State(*left_->getState(state.left));
127  result->setId(si);
128  for (Fsa::State::iterator aa = result->begin(); aa != result->end(); ++aa) {
129  Lm::History history = state.right;
130  addScore(*aa, history);
131  aa->target_ = insertState(aa->target_, history);
132  }
133  if (result->isFinal()) {
134  result->weight_ = semiring()->extend(
135  result->weight_,
136  Fsa::Weight(lmScale_ * right_->sentenceEndScore(state.right)));
137  }
138  return Fsa::ConstStateRef(result);
139  }

146 class LemmaComposeAutomaton : public ComposeAutomaton {
147  typedef ComposeAutomaton Precursor;
148 
149 protected:
150  void addScore(Fsa::Arc& a, Lm::History& history) const {
151  if (a.output() != Fsa::Epsilon) {
152  require(a.output() <= Fsa::LastLabelId);
153  Lm::Score score = f32(a.weight_);
154  Lm::addLemmaScore(right_, lmScale_,
155  right_->lexicon()->lemmaAlphabet()->lemma(a.output()),
156  syntaxEmissionScale_,
157  history, score);
158  a.weight_ = Fsa::Weight(score);
159  }
160  }
161 
162 public:
163  LemmaComposeAutomaton(Fsa::ConstAutomatonRef left,
164  Core::Ref<const Lm::LanguageModel> right,
165  Score lmScale, Score syntaxEmissionScale)
166  : Precursor(left, right, lmScale, syntaxEmissionScale) {
167  left_ = Fsa::mapOutput(left_, right_->lexicon()->lemmaAlphabet(), true);
168  left_ = Fsa::cache(left_);
169  }
170 
171  virtual ~LemmaComposeAutomaton() {}
172 };

214 ComposeAutomaton* ComposeAutomaton::createComposeAutomaton(
215  Fsa::ConstAutomatonRef left,
216  Core::Ref<const Lm::LanguageModel> right,
217  Score lmScale, Score syntaxEmissionScale) {
218  ComposeAutomaton* composeAutomaton = 0;
219  if (left->getOutputAlphabet() == right->lexicon()->lemmaAlphabet()) {
220  composeAutomaton = new LemmaComposeAutomaton(left, right, lmScale, syntaxEmissionScale);
221  }
222  else if (left->getInputAlphabet() == right->lexicon()->lemmaPronunciationAlphabet()) {
223  composeAutomaton = new LemmaPronunciationComposeAutomaton(left, right, lmScale, 0.0, syntaxEmissionScale);
224  }
225  else {
226  defect();
227  }
228  return composeAutomaton;
229 }
