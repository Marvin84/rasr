31  f64 inverseNFrames = 1.0F / nFrames_;
32  Math::Vector<f64> squareVec(dim());
33  mean_ *= inverseNFrames;
34  variance_ *= inverseNFrames;
35  squareVec.squareVector(mean_);
36  variance_ -= squareVec;

110 const f32 FullCovMonoGaussianModel::relativeLikelihood(const FullCovMonoGaussianModel& x) const {
111  /* p(this|x) */
112  f64 likelihood = 0.0F;
113  Math::Vector<f64> tmpvec(mean_ - x.mean_);
114  Math::Matrix<f64> invmat(x.variance_);
115  if (!Math::Lapack::invert(invmat))
116  return Core::Type<f32>::max; // failure
117 
118  f64 det = Math::Lapack::logDeterminant(x.variance_);
119  if (det == Core::Type<f64>::max)
120  return Core::Type<f32>::max; // failure
121 
122  likelihood -= det;
123  likelihood -= invmat.productTrace(variance_);
124  likelihood -= invmat.vvt(tmpvec);
125  likelihood -= dim() * LN_2PI;
126  likelihood *= 0.5F;
127  likelihood *= nFrames_;
128  return (f32(likelihood));
129 }
