33  */

39  void nextStep(StateId s) const {
40  Step step;
41  step.state = s;
42  ConstStateRef origState = fsa_->getState(step.state);
43  s32 nChoices = origState->nArcs();
44  if (origState->isFinal())
45  nChoices += 1;
46  if (maximumSize_ && (steps_.size() + 1 >= maximumSize_)) {
47  // maximum length exceeded
48  step.arc = -3;
49  }
50  else if (nChoices < 1) {
51  // reached dead end of original automaton
52  step.arc = -2;
53  }
54  else {
55  if (samplingWeight_ != 0.0) {
56  std::vector<f64> probs;
57  if (origState->isFinal())
58  probs.push_back(exp(-f32(origState->weight_) * samplingWeight_));
59  for (State::const_iterator aa = origState->begin(); aa != origState->end(); ++aa)
60  probs.push_back(exp(-f32(aa->weight()) * samplingWeight_));
61  verify(nChoices == s32(probs.size()));
62  for (s32 ii = 1; ii < nChoices; ++ii)
63  probs[ii] += probs[ii - 1];
64  f64 rand = drand48() * probs.back();
65  step.arc = std::upper_bound(probs.begin(), probs.end(), rand) - probs.begin();
66  verify(rand < probs[step.arc]);
67  verify(((step.arc == 0) ? 0.0 : probs[step.arc - 1]) <= rand);
68  }
69  else {
70  step.arc = lrand48() % nChoices;
71  }
72  if (origState->isFinal())
73  step.arc -= 1;
74  ensure(step.arc < s32(origState->nArcs()));
75  ensure(step.arc >= (origState->isFinal() ? -1 : 0));
76  }
77  steps_.push_back(step);
78  }

95  virtual ConstStateRef getState(StateId s) const {
96  require(s < steps_.size());
97  const Step& step(steps_[s]);
98  ConstStateRef origState = fsa_->getState(step.state);
99 
100  State* state = new State(s);
101  if (step.arc >= 0) {
102  State::const_iterator origArc = origState->begin() + step.arc;
103  state->newArc(s + 1,
104  origArc->weight(),
105  origArc->input(),
106  origArc->output());
107  if (s + 1 >= steps_.size())
108  nextStep(origArc->target());
109  }
110  else if (step.arc == -1) {
111  verify(origState->isFinal());
112  state->setFinal(origState->weight_);
113  }
114  return ConstStateRef(state);
115  }
