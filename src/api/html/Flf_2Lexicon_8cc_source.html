27 class UnknownAlphabet : public Fsa::Alphabet {
28 public:
29  virtual std::string symbol(Fsa::LabelId index) const {
30  switch (index) {
31  case Fsa::Epsilon:
32  return "*EPS*";
33  case Fsa::Any:
34  return "*ANY*";
35  case Fsa::Failure:
36  return "*FAIL*";
37  case Fsa::Else:
38  return "*ELSE*";
39  default:
40  return Core::form("%d", index);
41  }
42  }
43  virtual Fsa::LabelId index(const std::string& symbol) const {
44  Fsa::LabelId id;
45  if (!Core::strconv(symbol, id)) {
46  if (symbol == "*EPS*")
47  return Fsa::Epsilon;
48  else if (symbol == "*ANY*")
49  return Fsa::Any;
50  else if (symbol == "*FAIL*")
51  return Fsa::Failure;
52  else if (symbol == "*ELSE*")
53  return Fsa::Else;
54  else
55  Core::Application::us()->error(
56  "Failed to convert \"%s\"; it is not a valid integer", symbol.c_str());
57  return Fsa::InvalidLabelId;
58  }
59  return id;
60  }
61  virtual const_iterator begin() const {
62  return const_iterator(Core::Ref<const Fsa::Alphabet>(this), 0);
63  }
64  virtual const_iterator end() const {
65  return const_iterator(Core::Ref<const Fsa::Alphabet>(this), 0);
66  }
67 };

90 Lexicon::Lexicon(const Core::Configuration& config)
91  : Precursor(config) {
92  verify(!us_);
93  us_ = LexiconRef(this);
94  us_->acquireReference();
95  if (!paramFile(config).empty()) {
96  load(paramFile(config));
97  for (LemmaIterator l = lemmas().first; l != lemmas().second; ++l) {
98  const Lemma* lemma = *l;
99  for (Bliss::OrthographicFormList::Iterator o = lemma->orthographicForms().begin(),
100  o_end = lemma->orthographicForms().end();
101  o != o_end; ++o) {
102  if (!orthographyMap_.insert(std::make_pair(std::string(*o), const_cast<Lemma*>(lemma))).second)
103  warning("Orthography \"%s\" occurs twice; a lookup for the orthography will map to lemma \"%s\".",
104  o->str(), lemma->symbol().str());
105  }
106  }
107  // hash orthographies
108  }
109  else
110  us_->setPhonemeInventory(Core::Ref<Bliss::PhonemeInventory>(new Bliss::PhonemeInventory));
111  unknownAlphabet_ = Fsa::ConstAlphabetRef(new UnknownAlphabet);
112  phonemeInventory_ = const_cast<Bliss::PhonemeInventory*>(phonemeInventory().get());
113  normalizePronunciationWeights_ = paramNormalizePronunciationWeights(config);
114  isReadOnly_ = paramReadOnly(config);
115  if (!isReadOnly_)
116  insertionChannel_ = new Core::XmlChannel(config, "insertion");
117  else
118  insertionChannel_ = 0;
119  nLemmaUpdates_ = 0;
120  emptyPron_ = Precursor::getOrCreatePronunciation(
121  std::vector<Bliss::Phoneme::Id>(1, Bliss::Phoneme::term));
122  initSpecialLemmas();
123 }

129 void Lexicon::initSpecialLemmas() {
130  /*
131  * silence lemma (pronunciation)
132  */
133  {
134  const Lemma* siLemma = Precursor::specialLemma("silence");
135  if (!siLemma)
136  siLemma = Precursor::specialLemma("sil");
137  if (siLemma) {
138  // sil lemma
139  siLemmaId_ = siLemma->id();
140  // sil lemma pronunciation
141  LemmaPronunciationRange unkPronRange = siLemma->pronunciations();
142  if (unkPronRange.first != unkPronRange.second)
143  siLemmaPronunciationId_ = unkPronRange.first->id();
144  else
145  siLemmaPronunciationId_ = Fsa::InvalidLabelId;
146  }
147  else
148  siLemmaId_ = siLemmaPronunciationId_ = Fsa::InvalidLabelId;
149  }
150  /*
151  * unknown lemma (pronunciation)
152  */
153  {
154  const Lemma* unkLemma = Precursor::specialLemma("unknown");
155  if (!unkLemma)
156  unkLemma = Precursor::specialLemma("unk");
157  if (unkLemma) {
158  // unk lemma
159  unkLemmaId_ = unkLemma->id();
160  // unk lemma pronunciation
161  LemmaPronunciationRange unkPronRange = unkLemma->pronunciations();
162  if (unkPronRange.first != unkPronRange.second)
163  unkLemmaPronunciationId_ = unkPronRange.first->id();
164  else
165  unkLemmaPronunciationId_ = Fsa::InvalidLabelId;
166  }
167  else
168  unkLemmaId_ = unkLemmaPronunciationId_ = Fsa::InvalidLabelId;
169  }
170  /*
171  * sentence end lemma (pronunciation)
172  */
173  {
174  const Lemma* sentenceEndLemma = Precursor::specialLemma("sentence-end");
175  if (!sentenceEndLemma)
176  sentenceEndLemma = Precursor::specialLemma("sentence-boundary");
177  if (sentenceEndLemma) {
178  // sentence end lemma
179  sentenceEndLemmaId_ = sentenceEndLemma->id();
180  // unk lemma pronunciation
181  LemmaPronunciationRange sentenceEndPronRange = sentenceEndLemma->pronunciations();
182  if (sentenceEndPronRange.first != sentenceEndPronRange.second)
183  sentenceEndLemmaPronunciationId_ = sentenceEndPronRange.first->id();
184  else
185  sentenceEndLemmaPronunciationId_ = Fsa::InvalidLabelId;
186  }
187  else
188  sentenceEndLemmaId_ = sentenceEndLemmaPronunciationId_ = Fsa::InvalidLabelId;
189  }
190 }

193  ConstLemmaPtrList _nonWordLemmas;

203  }
204  else

206  }

244 std::pair<Lemma*, bool> Lexicon::getLemma(const std::string& orth) {
245  LemmaMap::const_iterator it = orthographyMap_.find(orth);
246  if (it != orthographyMap_.end())
247  return std::make_pair(it->second, true);
248  if (isReadOnly_)
249  return std::make_pair((Lemma*)0, true);
250  Lemma* _lemma = Precursor::newLemma(orth);
251  Precursor::setOrthographicForms(_lemma, std::vector<std::string>(1, orth));
252  orthographyMap_.insert(std::make_pair(orth, _lemma));
253  return std::make_pair(_lemma, false);

256 void Lexicon::updateLemma(Lemma* _lemma) {
257  require(_lemma);
258  if (!_lemma->hasName())
259  Precursor::setDefaultLemmaName(_lemma);
260  if (!_lemma->hasSyntacticTokenSequence())

262  if (!_lemma->hasEvaluationTokenSequence())
263  Precursor::setDefaultEvaluationToken(_lemma);

296 Bliss::Pronunciation* Lexicon::getPronunciation(const std::string& phons) {
297  std::vector<Bliss::Phoneme::Id> phonemes;
298  std::string::size_type i, j;
299  i = phons.find_first_not_of(utf8::whitespace);
300  while (i != std::string::npos) {
301  j = phons.find_first_of(utf8::whitespace, i);
302  if (j == std::string::npos)
303  j = phons.length();
304  Fsa::LabelId label = phonemeId(phons.substr(i, j - i));
305  if (label == Fsa::InvalidLabelId)
306  criticalError("Invalid phoneme symbol \"%s\".",
307  phons.substr(i, j - i).c_str());
308  phonemes.push_back(label);
309  i = phons.find_first_not_of(utf8::whitespace, j);
310  }
311  phonemes.push_back(Bliss::Phoneme::term);
312  Bliss::Pronunciation* pron = Precursor::getOrCreatePronunciation(phonemes);
313  return pron;
314 }

316 Fsa::LabelId Lexicon::lemmaPronunciationId(const std::string& id) {
317  Fsa::LabelId label = lemmaPronunciationAlphabet()->specialIndex(id);
318  if (label != Fsa::InvalidLabelId)
319  return label;
320  // Parse
321  std::string::size_type i = id.find(" /");
322  std::string::size_type j = i + 4;
323  if (i == std::string::npos) { // for backward compatibility
324  i = id.find("/");
325  j = i + 1;
326  }
327  std::string::size_type k = id.rfind("/");
328  if ((i == std::string::npos) || (k == std::string::npos) || (k < j)) {
329  criticalError("Invalid lemma pronunciation symbol \"%s\".",
330  id.c_str());
331 
332  //dbg
333  defect();
334 
335  return Fsa::InvalidLabelId;
336  }
337  std::string orth = id.substr(0, i);
338  std::string phon = id.substr(j, k - j);
339  // Get lemma and pronunciation
340  Lemma* _lemma = getLemma(orth).first;
341  if (!_lemma) {
342  //dbg
343  defect();
344 
345  return Fsa::InvalidLabelId;
346  }
347  for (LemmaPronunciationRange lpRange = _lemma->pronunciations();
348  lpRange.first != lpRange.second; ++lpRange.first) {
349  if (phon == lpRange.first->pronunciation()->format(phonemeInventory()))
350  return lpRange.first->id();
351  }
352  if (isReadOnly_) {
353  //dbg
354  defect();
355 
356  return Fsa::InvalidLabelId;
357  }
358  Bliss::Pronunciation* pron = getPronunciation(phon);
359  Precursor::addPronunciation(_lemma, pron, 1.0);
360  updateLemma(_lemma);
361  for (LemmaPronunciationRange lpRange = _lemma->pronunciations();
362  lpRange.first != lpRange.second; ++lpRange.first) {
363  if (phon == lpRange.first->pronunciation()->format(phonemeInventory())) {
364  if (insertionChannel_->isOpen())
365  *insertionChannel_ << (Core::XmlFull(
366  "new-lemma-pronunciation",
367  lemmaPronunciationAlphabet()->symbol(lpRange.first->id())) +
368  Core::XmlAttribute("id", lpRange.first->id()));
369  return lpRange.first->id();
370  }
371  }
372 
373  // dbg
374  defect();
375 
376  return Fsa::InvalidLabelId;
377 }

441 const LemmaPronunciation* Lexicon::lemmaPronunciation(const Lemma* _lemma, s32 variant) {
442  require(_lemma);
443  LemmaPronunciationRange lpRange;
444  if (_lemma->nPronunciations() > 0) {
445  lpRange = _lemma->pronunciations();
446  if (variant == -1) {
447  for (; lpRange.first != lpRange.second; ++lpRange.first)
448  if (lpRange.first->pronunciation()->length() == 0)
449  return lpRange.first;
450  }
451  else {
452  verify(variant >= 0);
453  if (variant < (s32)_lemma->nPronunciations())
454  for (s32 i = 0; i < variant; ++i, ++lpRange.first)
455  ;
456  return lpRange.first;
457  }
458  }
459  if (isReadOnly_)
460  return 0;
461  Precursor::addPronunciation(const_cast<Lemma*>(_lemma), emptyPron_, 1.0);
462  updateLemma(const_cast<Lemma*>(_lemma));
463  lpRange = _lemma->pronunciations();
464  verify(lpRange.first != lpRange.second);
465  if (insertionChannel_->isOpen())
466  *insertionChannel_ << (Core::XmlFull(
467  "new-lemma-pronunciation",
468  lemmaPronunciationAlphabet()->symbol(lpRange.first->id())) +
469  Core::XmlAttribute("id", lpRange.first->id()));
470  return lpRange.first;
471 }

473 std::pair<const Lemma*, s32> Lexicon::lemmaPronunciationVariant(const Bliss::LemmaPronunciation* lp) {
474  require(lp);
475  std::pair<const Lemma*, s32> lv(lp->lemma(), -1);
476  if (lp->pronunciation()->length() > 0) {
477  s32 v = 0;
478  for (LemmaPronunciationRange lpRange = lv.first->pronunciations();
479  lpRange.first != lpRange.second; ++lpRange.first, ++v)
480  if (lpRange.first->id() == lp->id())

483  lv.second = v;
484  }
485  return lv;

600 Fsa::LabelId Lexicon::AlphabetMap::operator[](Fsa::LabelId label) {
601  if ((label < Fsa::FirstLabelId) || (Fsa::LastLabelId < label))
602  return label; // special label ids are fixed
603  Fsa::LabelId target;
604  if (label < Fsa::LabelId(size())) {
605  target = Precursor::operator[](label);
606  if (target != Unmapped)
607  return target;
608  }
609  else
610  grow(label, Unmapped);
611  target = Precursor::operator[](label) = index(label);

814 EditDistance::Cost EditDistance::operator()(Fsa::LabelId label1, Fsa::LabelId label2) const {
815  if (label1 == label2)
816  return 0;
817  const Bliss::Token *tok1 = toks_->token(label1), *tok2 = toks_->token(label2);
818  if (tok1 == tok2)
819  return 0;
820  if (!tok1)
821  return tok2->symbol().length();
822  if (!tok2)
823  return tok1->symbol().length();
824  Cost l1 = tok1->symbol().length();
825  const Bliss::Symbol::Char* begin_c1 = tok1->symbol().str();
826  const Bliss::Symbol::Char* end_c1 = begin_c1 + l1;
827  verify(*end_c1 == '\0');
828  if (lengthD_ <= l1) {
829  delete[] D_;
830  lengthD_ = l1 + 1;
831  D_ = new Cost[lengthD_];
832  }
833  Cost* d = D_;
834  for (Cost l = 0; l <= l1; ++l, ++d)
835  *d = l;
836  for (const Bliss::Symbol::Char* c2 = tok2->symbol().str(); *c2 != '\0'; ++c2) {
837  d = D_;
838  Cost p = *d + 1;
839  ++d;
840  for (const Bliss::Symbol::Char* c1 = begin_c1; c1 != end_c1; ++c1, ++d) {
841  verify_((d != D_) && (d != D_ + lengthD_));
842  Cost n = std::min(*(d - 1) + ((*c1 == *c2) ? 0 : 1), std::min(*d, p) + 1);
843  *(d - 1) = p;
844  p = n;
845  }
846  *(d - 1) = p;
847  }
848  return *(D_ + l1);
849 }

855 Score EditDistance::normCost(Fsa::LabelId label1, Fsa::LabelId label2) const {
856  EditDistance::Cost dist = operator()(label1, label2);
857  if (dist == 0)
858  return 0.0;
859  const Bliss::Token *tok1 = toks_->token(label1), *tok2 = toks_->token(label2);
860  EditDistance::Cost norm = 0;
861  if (tok1)
862  norm += tok1->symbol().length();
863  if (tok2)
864  norm += tok2->symbol().length();
865  verify(norm > 0);
866  return Score(dist) / Score(norm);
867 }
