54 /**
55  * Prints a list of values.
56  * Use either ListWriter directly or
57  * for vector, list, and set you may use the function
58  * str returning a streamable object.
59  */
60 template<class InputIterator>
61 struct ListWriter : public OStreamWriter {
62  const InputIterator begin;
63  const InputIterator end;
64  const std::string delim;
65  const std::string openingBracket;
66  const std::string closingBracket;
67 
68  ListWriter(const InputIterator& begin,
69  const InputIterator& end,
70  const std::string& delim = ", ",
71  const std::string& openingBracket = "",
72  const std::string& closingBracket = "")
73  : begin(begin),
74  end(end),
75  delim(delim),
76  openingBracket(openingBracket),
77  closingBracket(closingBracket) {
78  }
79 
80  void write(std::ostream& out) const {
81  out << openingBracket;
82  if (begin != end) {
83  InputIterator it = begin;
84  out << *(it++);
85  for (; it != end; ++it)
86  out << delim << *it;
87  }
88  out << closingBracket;
89  }
90 };
