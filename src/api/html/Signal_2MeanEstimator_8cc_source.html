51 bool MeanEstimator::finalize(Math::Vector<Data>& mean) const {
52  if (needInit_) {
53  error("No observation has been seen.");
54  return false;
55  }
56  mean.resize(featureDimension_);
57  std::transform(vectorSum_.begin(), vectorSum_.end(), mean.begin(), std::bind2nd(std::divides<Sum>(), count_));
58  return true;
59 }

106 bool MeanEstimatorNode::work(Flow::PortId p) {
107  Flow::DataPtr<Flow::Vector<f32>> in;
108  u32 count = 0;
109  std::vector<Flow::Timestamp> timestamps;
110  while (Precursor::getData(0, in)) {
111  if (count == 0)
112  MeanEstimator::setDimension(in->size());
113  MeanEstimator::accumulate(*in);
114  timestamps.push_back(*in);
115  ++count;
116  }
117  Math::Vector<f32> mean;
118  if (!MeanEstimator::finalize(mean))
119  return false;
120  for (u32 i = 0; i < count; ++i) {
121  Flow::DataPtr<Flow::Vector<f32>> out(new Flow::Vector<f32>(mean));
122  out->setTimestamp(timestamps[i]);
123  putData(0, out.get());
124  }
125  return putData(0, Flow::Data::eos());
126 }
