23 /** Aggregate of Flow objects
24  * Aggregate collects objects of different streams into one object.
25  * It supports polymorfism:
26  * -The base class for the input objects is Timestamp.

29  */

63  /** Retrieves a reference to the i-th element.
64  * @return is false, if the i-th element is not an instance of class T or
65  * of one of its descendants.
66  */
67  template<class T>
68  bool get(size_t i, DataPtr<T>& d) {
69  return (bool)(d.set(operator[](i)));
70  }
71  /** Retrieves references to each elements.
72  * @return is false, if one of the elements is not an instance of class T or
73  * of one of its descendants.
74  */
75  template<class T>
76  bool get(std::vector<DataPtr<T>>& d) {
77  std::vector<DataPtr<T>> dd(begin(), end());
78  if (std::find_if(dd.begin(), dd.end(), std::bind2nd(std::equal_to<bool>(), false)) != dd.end())
79  return false;
80  d = dd;
81  return true;
82  }
