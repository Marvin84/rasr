31 bool ScaleUpdate::setMap(const Core::Configuration &c)
32 {
33  std::string filename = paramMapFilename(c);

37  ScaleMap scaleMap;
38  Core::XmlMapDocument<ScaleMap> parser(
39  c, scaleMap, "model-combination", "model", "name", "scale");
40  parser.parseFile(filename.c_str());

42  ScaleMap *extendedMap = new ScaleMap;
43  for(ScaleMap::const_iterator scale = scaleMap.begin(); scale != scaleMap.end(); ++ scale) {
44  std::string extendedKey = c.getSelection();
45  if (!scale->first.empty())
46  extendedKey = c.prepareResourceName(extendedKey, scale->first);

48  }
49  setMap(Core::Ref<const ScaleMap>(extendedMap));
50  return true;
51 }
