134 static void notifyHandler(int sig) {
135  // This handler is called for notification (e.g. on SIGUSR1).
136  // We want to continue with normal operation when the signal handler finished,
137  // so we need to be more careful here (in contrast to handler()).
138  // See signal() manpage about what is safe to call.
139  // Basically, avoid any malloc() and most other dynamic things.
140  PUTS(STDERR_FILENO, "\n\nRECEIVED NOTIFICATION SIGNAL:\n");
141  PUTS(STDERR_FILENO, strsignal(sig));
142  PUTS(STDERR_FILENO, "\n\n");
143  safe_stackTrace(STDERR_FILENO);
144  PUTS(STDERR_FILENO, "\n");
145  // Don't quit here. It's just a notification.
146 }
