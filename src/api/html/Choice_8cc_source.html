44 Choice::Choice(const char* ident, const Value value, ...)
45  : illegalIdentifier(Choice::IllegalIdentifier) {
46  int mode;
47  Value va_value;
48  const char* va_ident = 0;
49  va_list ap;
50 
51  addChoice(ident, value);
52 
53  mode = choice_parse_state_ident;
54  va_start(ap, value);
55  while (mode != choice_parse_state_exit) {
56  switch (mode) {
57  case choice_parse_state_ident:
58  va_ident = va_arg(ap, const char*);
59  if (va_ident == endMark())
60  mode = choice_parse_state_exit;
61  else
62  mode = choice_parse_state_value;
63  break;
64  case choice_parse_state_value:
65  va_value = va_arg(ap, Value);
66  addChoice(va_ident, va_value);
67  mode = choice_parse_state_ident;
68  break;
69  default:
70  break;
71  }
72  }
73  va_end(ap);
74 }
