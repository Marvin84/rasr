133  void build() {
134  if (l->getTopologicalSort())
135  stateIndex.grow(l->getTopologicalSort()->maxSid, 0);
136  traverse();
137  verify(!stateIndex.empty());
138  stateIndex.front() = 0;
139  for (Fsa::StateId sid = 0; sid < stateIndex.size() - 1; ++sid)
140  stateIndex[sid + 1] += stateIndex[sid];
141  }

143  ConfusionNetwork::ConstMapPropertiesRef getMapping(bool reduced) const {
144  ConfusionNetwork::MapProperties* mapProps = new ConfusionNetwork::MapProperties;
145 
146  Core::Vector<Fsa::StateId>& stateIndex = mapProps->stateIndex;
147  ConfusionNetwork::MapProperties::Map& lat2cn = mapProps->lat2cn;
148  {
149  StateIndexBuilder stateIndexBuilder(l_, stateIndex);
150  stateIndexBuilder.build();
151  lat2cn.resize(stateIndex.back(), ConfusionNetwork::MapProperties::Mapping(Fsa::InvalidStateId, 0));
152  }
153  if (reduced) {
154  // lattice arc -> slot
155  for (Fsa::StateId sid = 0; sid < slots_.size(); ++sid) {
156  Slot& pcnSlot = *slots_[sid];
157  for (ArcPtrList::const_iterator itArcPtr = pcnSlot.arcs.begin(), endArcPtr = pcnSlot.arcs.end();
158  itArcPtr != endArcPtr; ++itArcPtr) {
159  const Arc& pcnArc = **itArcPtr;
160  verify_(stateIndex[pcnArc.sid] + pcnArc.aid < lat2cn.size());
161  lat2cn[stateIndex[pcnArc.sid] + pcnArc.aid] = ConfusionNetwork::MapProperties::Mapping(sid, Fsa::InvalidStateId);
162  }
163  }
164  }
165  else {
166  // lattice arc <-> slot arc
167  Core::Vector<Fsa::StateId>& slotIndex = mapProps->slotIndex;
168  ConfusionNetwork::MapProperties::Map& cn2lat = mapProps->cn2lat;
169  {
170  slotIndex.resize(slots_.size() + 1);
171  u32 nCnArcs = 0;
172  for (Fsa::StateId sid = 0; sid < slots_.size(); ++sid) {
173  const Slot& pcnSlot = *slots_[sid];
174  slotIndex[sid] = nCnArcs;
175  nCnArcs += pcnSlot.arcs.size();
176  }
177  slotIndex.back() = nCnArcs;
178  cn2lat.resize(slotIndex.back(), ConfusionNetwork::MapProperties::Mapping(Fsa::InvalidStateId, 0));
179  }
180  for (Fsa::StateId sid = 0; sid < slots_.size(); ++sid) {
181  Slot& pcnSlot = *slots_[sid];
182  Fsa::StateId aid = 0;
183  for (ArcPtrList::const_iterator itArcPtr = pcnSlot.arcs.begin(), endArcPtr = pcnSlot.arcs.end();
184  itArcPtr != endArcPtr; ++itArcPtr, ++aid) {
185  const Arc& pcnArc = **itArcPtr;
186  verify_(slotIndex[sid] + aid < cn2lat.size());
187  cn2lat[slotIndex[sid] + aid] = ConfusionNetwork::MapProperties::Mapping(pcnArc.sid, pcnArc.aid);
188  verify_(stateIndex[pcnArc.sid] + pcnArc.aid < lat2cn.size());
189  lat2cn[stateIndex[pcnArc.sid] + pcnArc.aid] = ConfusionNetwork::MapProperties::Mapping(sid, aid);
190  }
191  }
192  }
193  return ConfusionNetwork::ConstMapPropertiesRef(mapProps);
194  }

197  ConstConfusionNetworkRef getCn(ScoreId posteriorId, bool mapping = false) const {
198  if (slots_.empty())
199  return ConstConfusionNetworkRef();
200  ConstSemiringRef semiring = l_->semiring();
201 
202  ConfusionNetwork* cn = new ConfusionNetwork(slots_.size());
203  cn->alphabet = l_->getInputAlphabet();
204  cn->semiring = semiring;
205  for (u32 i = 0; i < slots_.size(); ++i) {
206  Slot& pcnSlot = *slots_[i];
207  ConfusionNetwork::Slot& slot = (*cn)[i];
208  for (ArcPtrList::const_iterator itArcPtr = pcnSlot.arcs.begin(), endArcPtr = pcnSlot.arcs.end(); itArcPtr != endArcPtr; ++itArcPtr) {
209  const Arc& pcnArc = **itArcPtr;
210  const State& state = *l_->getState(pcnArc.sid);
211  const Flf::Arc& arc = *(state.begin() + pcnArc.aid);
212  ScoresRef scores = arc.weight();
213  if (posteriorId != Semiring::InvalidId) {
214  scores = semiring->clone(scores);
215  scores->set(posteriorId, ::exp(-pcnArc.posteriorScore));
216  }
217  slot.push_back(ConfusionNetwork::Arc(
218  arc.input(), scores,
219  pcnArc.start, pcnArc.end - pcnArc.start));
220  }
221  }
222  if (mapping)
223  cn->mapProperties = getMapping(mapping);
224  return ConstConfusionNetworkRef(cn);

232  Probability calcConfidence(ProbabilityList::const_iterator itP, ProbabilityList::const_iterator endP) const {
233  switch (u32(endP - itP)) {
234  case 0:
235  return 1.0;
236  case 1:
237  return *itP;
238  default:
239  Probability maxP = 0.0;
240  for (; itP != endP; ++itP)
241  if (*itP > maxP)
242  maxP = *itP;
243  return maxP;
244  }
245  }

248  std::pair<ConstConfusionNetworkRef, ConstLatticeRef> getNormalizedCn(ScoreId confidenceId, bool mapping = false) const {
249  if (slots_.empty())
250  return std::make_pair(ConstConfusionNetworkRef(), ConstLatticeRef());
251  ConstSemiringRef semiring = l_->semiring();
252 
253  // Posterior CN
254  ConfusionNetwork* cn = new ConfusionNetwork(slots_.size());
255  cn->alphabet = l_->getInputAlphabet();
256  cn->semiring = semiring;
257  cn->normalizedProperties = ConfusionNetwork::ConstNormalizedPropertiesRef(
258  new ConfusionNetwork::NormalizedProperties(confidenceId));
259  // best lattice
260  StaticBoundaries* b = new StaticBoundaries;
261  StaticLattice* s = new StaticLattice;
262  s->setDescription("decode-cn(" + l_->describe() + ",pivot-arc-cluster)");
263  s->setType(Fsa::TypeAcceptor);
264  s->setProperties(Fsa::PropertyAcyclic | Fsa::PropertyLinear, Fsa::PropertyAll);
265  s->setInputAlphabet(l_->getInputAlphabet());
266  s->setSemiring(semiring);
267  s->setBoundaries(ConstBoundariesRef(b));
268  s->setInitialStateId(0);
269  Flf::State* sp = new Flf::State(0);
270  s->setState(sp);
271  Time minEndTime = 0, preferredEndTime = 0; // Attention: preferredEndTime >= minEndTime
272  ProbabilityList pendingEpsConfidences;
273  Collector
274  *col = createCollector(Fsa::SemiringTypeLog),
275  *sumCol = createCollector(Fsa::SemiringTypeLog),
276  *colStart = createCollector(Fsa::SemiringTypeLog),
277  *colEnd = createCollector(Fsa::SemiringTypeLog);
278  for (u32 i = 0; i < slots_.size(); ++i) {
279  // distribute probabilities over words
280  ConfusionNetwork::Slot& slot = (*cn)[i];
281  ArcPtrList& arcs = slots_[i]->arcs;
282  std::sort(arcs.begin(), arcs.end(), Arc::SortByLabel());
283  const Arc* bestPcnArcPtr = 0;
284  ScoresRef bestPcnArcScores = ScoresRef();
285  f64 bestPosteriorScore = Core::Type<f64>::max;
286  const Arc* bestCurrentLabelPcnArcPtr = arcs.front();
287  Score epsConfidence = 0.0;
288  for (ArcPtrList::const_iterator itArcPtr = arcs.begin(), endArcPtr = arcs.end(); itArcPtr != endArcPtr; ++itArcPtr) {
289  const Arc* pcnArcPtr = *itArcPtr;
290  if (pcnArcPtr->label != bestCurrentLabelPcnArcPtr->label) {
291  f64 posteriorScore = col->get();
292  col->reset();
293  Score confidence = ::exp(-posteriorScore);
294  if (bestCurrentLabelPcnArcPtr->label == Fsa::Epsilon)
295  epsConfidence = confidence;
296  Score start = ::exp(posteriorScore - colStart->get()), end = ::exp(posteriorScore - colEnd->get());
297  colStart->reset();
298  colEnd->reset();
299  const Flf::State& state = *l_->getState(bestCurrentLabelPcnArcPtr->sid);
300  const Flf::Arc& arc = *(state.begin() + bestCurrentLabelPcnArcPtr->aid);
301  ScoresRef scores = arc.weight();
302  if (confidenceId != Semiring::InvalidId) {
303  scores = semiring->clone(scores);
304  scores->set(confidenceId, confidence);
305  }
306  slot.push_back(ConfusionNetwork::Arc(
307  bestCurrentLabelPcnArcPtr->label,
308  scores,
309  Time(Core::round(start)), std::max(Time(1), Time(Core::round(end - start)))));
310  if (posteriorScore < bestPosteriorScore) {
311  bestPosteriorScore = posteriorScore;
312  bestPcnArcPtr = bestCurrentLabelPcnArcPtr;
313  bestPcnArcScores = scores;
314  }
315  bestCurrentLabelPcnArcPtr = pcnArcPtr;
316  sumCol->feed(posteriorScore);
317  }
318  else if (pcnArcPtr->posteriorScore < bestCurrentLabelPcnArcPtr->posteriorScore) {
319  bestCurrentLabelPcnArcPtr = pcnArcPtr;
320  }
321  col->feed(pcnArcPtr->posteriorScore);
322  colStart->feed(pcnArcPtr->posteriorScore - ::log(f64(pcnArcPtr->start)));
323  colEnd->feed(pcnArcPtr->posteriorScore - ::log(f64(pcnArcPtr->end)));
324  }
325  f64 posteriorScore = col->get();
326  col->reset();
327  {
328  const Flf::State& state = *l_->getState(bestCurrentLabelPcnArcPtr->sid);
329  const Flf::Arc& arc = *(state.begin() + bestCurrentLabelPcnArcPtr->aid);
330  Score confidence = ::exp(-posteriorScore);
331  if (bestCurrentLabelPcnArcPtr->label == Fsa::Epsilon)
332  epsConfidence = confidence;
333  Score start = ::exp(posteriorScore - colStart->get()), end = ::exp(posteriorScore - colEnd->get());
334  colStart->reset();
335  colEnd->reset();
336  ScoresRef scores = arc.weight();
337  if (confidenceId != Semiring::InvalidId) {
338  scores = semiring->clone(scores);
339  scores->set(confidenceId, confidence);
340  }
341  slot.push_back(ConfusionNetwork::Arc(
342  bestCurrentLabelPcnArcPtr->label,
343  scores,
344  Time(Core::round(start)), std::max(Time(1), Time(Core::round(end - start)))));
345  if (posteriorScore < bestPosteriorScore) {
346  bestPosteriorScore = posteriorScore;
347  bestPcnArcPtr = bestCurrentLabelPcnArcPtr;
348  bestPcnArcScores = scores;
349  }
350  }
351  sumCol->feed(posteriorScore);
352  verify(bestPcnArcPtr);
353  // add remaining probability mass to eps arc
354  Probability bestConfidence = ::exp(-bestPosteriorScore);
355  Probability sum = ::exp(-sumCol->get());
356  sumCol->reset();
357  if (sum != 1.0) {
358  bool normalize = false;
359  Probability dEpsConfidence = 1.0 - sum;
360  if (dEpsConfidence < 0.0) {
361  if (dEpsConfidence < -0.01)
362  Core::Application::us()->warning("PivotArcCnBuilder: Expected 1.0, got %f", (1.0 - dEpsConfidence));
363  normalize = true;
364  }
365  else {
366  if (slot.front().label == Fsa::Epsilon) {
367  epsConfidence += dEpsConfidence;
368  if (confidenceId != Semiring::InvalidId)
369  slot.front().scores->set(confidenceId, epsConfidence);
370  }
371  else if (dEpsConfidence >= 0.001) {
372  epsConfidence += dEpsConfidence;
373  ScoresRef scores = semiring->one();
374  if (confidenceId != Semiring::InvalidId) {
375  scores = semiring->clone(semiring->one());
376  scores->set(confidenceId, epsConfidence);
377  }
378  slot.insert(slot.begin(), ConfusionNetwork::Arc(
379  Fsa::Epsilon,
380  scores,
381  slot.front().begin + slot.front().duration / 2, 0));
382  }
383  else
384  normalize = true;
385  if (epsConfidence > bestConfidence) {
386  bestConfidence = epsConfidence;
387  bestPcnArcPtr = 0;
388  }
389  }
390  if (normalize) {
391  Score norm = 1.0 / sum;
392  if (confidenceId != Semiring::InvalidId)
393  for (ConfusionNetwork::Slot::iterator itSlot = slot.begin(), endSlot = slot.end(); itSlot != endSlot; ++itSlot)
394  itSlot->scores->multiply(confidenceId, norm);
395  bestConfidence *= norm;
396  }
397  }
398  // add best arc
399  if ((bestPcnArcPtr) && (bestPcnArcPtr->label != Fsa::Epsilon)) {
400  if (preferredEndTime < bestPcnArcPtr->start) {
401  b->set(sp->id(), Boundary(preferredEndTime));
402  ScoresRef scores = semiring->one();
403  if (confidenceId != Semiring::InvalidId) {
404  Score epsConfidence = calcConfidence(pendingEpsConfidences.begin(), pendingEpsConfidences.end());
405  scores = semiring->clone(semiring->one());
406  scores->set(confidenceId, epsConfidence);
407  }
408  pendingEpsConfidences.clear();
409  sp->newArc(sp->id() + 1, scores, Fsa::Epsilon, Fsa::Epsilon);
410  sp = new Flf::State(sp->id() + 1);
411  s->setState(sp);
412  preferredEndTime = bestPcnArcPtr->start;
413  }
414  else
415  preferredEndTime = std::max(minEndTime, (preferredEndTime + bestPcnArcPtr->start) / 2);
416 
417  b->set(sp->id(), Boundary(preferredEndTime));
418  if (confidenceId != Semiring::InvalidId) {
419  bestPcnArcScores = semiring->clone(bestPcnArcScores);
420  bestPcnArcScores->set(confidenceId, bestConfidence);
421  }
422  sp->newArc(sp->id() + 1, bestPcnArcScores, bestPcnArcPtr->label, bestPcnArcPtr->label);
423  sp = new Flf::State(sp->id() + 1);
424  s->setState(sp);
425  minEndTime = preferredEndTime + 1;
426  preferredEndTime = std::max(minEndTime, bestPcnArcPtr->end);
427  }
428  else
429  pendingEpsConfidences.push_back(bestConfidence);
430  }
431  b->set(sp->id(), Boundary(preferredEndTime));
432  ScoresRef scores = semiring->one();
433  if ((confidenceId != Semiring::InvalidId) && !pendingEpsConfidences.empty()) {
434  scores = semiring->clone(scores);
435  scores->set(confidenceId, calcConfidence(pendingEpsConfidences.begin(), pendingEpsConfidences.end()));
436  pendingEpsConfidences.clear();
437  }
438  sp->setFinal(scores);
439  delete col;
440  delete sumCol;
441  delete colStart;
442  delete colEnd;
443  if (mapping)
444  cn->mapProperties = getMapping(true);
445  return std::make_pair(ConstConfusionNetworkRef(cn), ConstLatticeRef(s));
446  }

472  // min. distance over all arcs in slot
473  class WeightedTimeDistance : public Distance {
474  public:
475  Score impact;
476  ConstEditDistanceRef editDist;
477 
478  public:
479  WeightedTimeDistance(Score impact, ConstEditDistanceRef editDist)
480  : impact(impact), editDist(editDist) {}
481  virtual std::string describe() const {
482  return Core::form("weighted time, impact=%f, edit-distance=%s", impact, ((editDist) ? "yes" : "no"));
483  }
484  virtual Score distance(const PivotArcCn::Arc& ref, const PivotArcCn::Arc& hyp) const {
485  Time start = std::max(ref.start, hyp.start), end = std::min(ref.end, hyp.end);
486  verify(start < end);
487  Score score = (impact + ref.posteriorScore * hyp.posteriorScore) / impact;
488  if ((ref.isNonWord && hyp.isNonWord) || (!ref.isNonWord && !hyp.isNonWord)) {
489  Time dist = ((ref.start < hyp.start) ? hyp.start - ref.start : ref.start - hyp.start) +
490  ((ref.end < hyp.end) ? hyp.end - ref.end : ref.end - hyp.end);
491  Time norm = (ref.end - ref.start) + (hyp.end - hyp.start);
492  score *= Score(dist) / Score(norm);
493  }
494  if (ref.label != hyp.label)
495  score *= (editDist) ? (1.0 + editDist->normCost(ref.label, hyp.label)) : 2.0;
496  return score;
497  }
498  static ConstDistanceRef create(Score impact = 10.0, ConstEditDistanceRef editDist = ConstEditDistanceRef()) {
499  return ConstDistanceRef(new WeightedTimeDistance(impact, editDist));
500  }

549  bool isConsistent(PivotArcCn::SlotPtrList& slots) const {
550  bool consistent = true;
551  if (slots.size() > 1) {
552  Time pStart = 0, pEnd = 0, tStart = slots.front()->start, tEnd = slots.front()->end;
553  if (!(tStart < tEnd)) {
554  dbg(1);
555  consistent = false;
556  }
557  for (PivotArcCn::SlotPtrList::const_iterator itSlotPtr = slots.begin() + 1; (itSlotPtr != slots.end()) && consistent; ++itSlotPtr) {
558  const PivotArcCn::Slot& slot = **itSlotPtr;
559  pStart = tStart;
560  pEnd = tEnd;
561  tStart = slot.start;
562  tEnd = slot.end;
563  if (!(tStart < tEnd)) {
564  consistent = false;
565  break;
566  }
567  if (!(pEnd <= tStart)) {
568  consistent = false;
569  break;
570  }
571  if (slot.arcs.empty()) {
572  consistent = false;
573  break;
574  }
575  for (PivotArcCn::ArcPtrList::const_iterator itArcPtr = slot.arcs.begin(); itArcPtr != slot.arcs.end(); ++itArcPtr) {
576  const PivotArcCn::Arc& arc = **itArcPtr;
577  if (!((arc.start <= tStart) && (tEnd <= arc.end))) {
578  consistent = false;
579  break;
580  }
581  }
582  }
583  }
584  if (!consistent)
585  dump(slots);
586  return consistent;
587  }

589  bool isArcListCleanedUp(u8 threshold) const {
590  PivotArcCn::ArcPtrList::iterator itArcPtr = beginArcPtr_;
591  for (; itArcPtr != endArcPtr_; ++itArcPtr) {
592  verify((*itArcPtr)->state < threshold);
593  verify(((*itArcPtr)->slotDist == Core::Type<Score>::max) && ((*itArcPtr)->slotId == Core::Type<u32>::max));
594  }
595  for (PivotArcCn::ArcPtrList::iterator endArcPtr = pcn_->arcs_.end(); itArcPtr != endArcPtr; ++itArcPtr) {
596  verify((*itArcPtr)->state >= threshold);
597  verify((*itArcPtr)->slotDist != Core::Type<Score>::max);
598  }
599  return true;
600  }

602  * End: For Debugging

605  // move all arcs to the end of the arc vector, where state reaches or exceeds the threshold
606  void cleanUpArcs(u8 threshold = 1) const {
607  for (; (endArcPtr_ != beginArcPtr_) && ((*(endArcPtr_ - 1))->state >= threshold); --endArcPtr_)
608  ;
609  for (PivotArcCn::ArcPtrList::iterator itArcPtr = beginArcPtr_; itArcPtr != endArcPtr_; ++itArcPtr) {
610  if ((*itArcPtr)->state >= threshold) {
611  std::swap(*itArcPtr, *(endArcPtr_ - 1));
612  for (; (*(endArcPtr_ - 1))->state >= threshold; --endArcPtr_)
613  ;
614  }
615  (*itArcPtr)->state = 0;
616  (*itArcPtr)->slotDist = Core::Type<Score>::max;
617  (*itArcPtr)->slotId = Core::Type<u32>::max;
618  }
619  verify(isArcListCleanedUp(threshold));
620  }

622  void cleanUpArcsAndSlots() const {
623  endArcPtr_ = endPivotArcPtr_;
624  cleanUpArcs(2);
625  endPivotArcPtr_ = endArcPtr_;
626  for (PivotArcCn::SlotPtrList::iterator itSlotPtr = pcn_->slots_.begin(), endSlotPtr = pcn_->slots_.end(); itSlotPtr != endSlotPtr; ++itSlotPtr) {
627  PivotArcCn::ArcPtrList& arcs = (*itSlotPtr)->arcs;
628  arcs.erase(arcs.begin() + 1, arcs.end());
629  verify(arcs.front()->state == 2);
630  }

633  struct TraceElement {
634  Score score;
635  Fsa::StateId bptr;
636  PivotArcCn::Arc* arc;
637  TraceElement()
638  : score(Semiring::Max), bptr(Fsa::InvalidStateId), arc(0) {}
639  };
640  typedef std::vector<TraceElement> Traceback;
641  u32 init(ConstLatticeRef l, ConstFwdBwdRef fb) const {
642  LabelMapRef nonWordToEpsilonMap = LabelMap::createNonWordToEpsilonMap(Lexicon::us()->alphabetId(l->getInputAlphabet(), true));
643  ConstStateMapRef topologicalSort = sortTopologically(l);
644  Fsa::StateId initialSid = topologicalSort->front();
645 
646  ConstBoundariesRef b = l->getBoundaries();
647  ConstSemiringRef semiring = l->semiring();
648  PivotArcCn::ArcPtrList& arcs = pcn_->arcs_;
649  PivotArcCn::SlotPtrList& slots = pcn_->slots_;
650  Traceback traceback(topologicalSort->maxSid + 1);
651  traceback[initialSid].score = 0.0;
652  TraceElement bestTrace;
653  for (u32 i = 0; i < topologicalSort->size(); ++i) {
654  Fsa::StateId sid = (*topologicalSort)[i];
655  ConstStateRef sr = l->getState(sid);
656  Time startTime = b->time(sid);
657  const TraceElement& currentTrace = traceback[sid];
658  if (sr->isFinal()) {
659  Score score = currentTrace.score + semiring->project(sr->weight());
660  if (score < bestTrace.score) {
661  bestTrace.score = score;
662  bestTrace.bptr = sid;
663  }
664  }
665  for (State::const_iterator a = sr->begin(), a_begin = sr->begin(), a_end = sr->end(); a != a_end; ++a) {
666  PivotArcCn::Arc* arc = 0;
667  Score posteriorScore = fb->arc(sr, a).score();
668  Time endTime = b->time(a->target());
669  /*
670  * if ((startTime < endTime) && (a->input() != Fsa::Epsilon)) {
671  *
672  * Seems to make not much a difference with the given distance function;
673  * for other distance functions things might like different.
674  *
675  * Todo: still crashes sometimes, especially for system combination
676  * (the reason is probably that the algorithm assumes that the pivot elements/slots
677  * occupy the complete time span ...)
678  * perhaps it is interesting to make it run for system combination
679  */
680  if (startTime < endTime) {
681  arc = new PivotArcCn::Arc;
682  arc->sid = sid;
683  arc->aid = a - a_begin;
684  arc->posteriorScore = posteriorScore;
685  arc->start = startTime;
686  arc->end = endTime;
687  arc->label = a->input();
688  arc->isNonWord = ((arc->label != Fsa::Epsilon) && (*nonWordToEpsilonMap)[arc->label].empty()) ? false : true;
689  arcs.push_back(arc);
690  }
691  else if (a->input() != Fsa::Epsilon)
692  Core::Application::us()->warning(
693  "Non-epsilon arc of length 0 detected: \"%s\", %d-%d, score=%f; arc is discarded",
694  l->getInputAlphabet()->symbol(a->input()).c_str(), startTime, endTime, semiring->project(a->weight()));
695  TraceElement& trace = traceback[a->target()];
696  /*
697  * Score score = currentTrace.score + semiring->project(a->weight());
698  *
699  * Use best posterior sentence instead of Viterbi sentence for initialization.
700  * Seems to make no difference no average;
701  * sometimes some errors better, very sometimes a few errors worse.
702  * But it works, too, for fwd/bwd combination experiments.
703  */
704  if (posteriorScore < trace.score) {
705  trace.score = posteriorScore;
706  trace.bptr = sid;
707  trace.arc = arc;
708  }
709  }
710  }
711  beginArcPtr_ = pcn_->arcs_.begin();
712  endArcPtr_ = pcn_->arcs_.end();
713  verify(bestTrace.bptr != Fsa::InvalidStateId);
714  u32 nAdded = 0;
715  Fsa::StateId bestSid = bestTrace.bptr;
716  while (bestSid != initialSid) {
717  const TraceElement& trace = traceback[bestSid];
718  if (trace.arc) {
719  PivotArcCn::Arc& arc = *trace.arc;
720  PivotArcCn::Slot* slot = new PivotArcCn::Slot;
721  slot->start = arc.start;
722  slot->end = arc.end;
723  slot->arcs.push_back(&arc);
724  slots.push_back(slot);
725  arc.state = 2;
726  arc.slotDist = Core::Type<Score>::min;
727  ++nAdded;
728  }
729  bestSid = trace.bptr;
730  }
731  std::reverse(slots.begin(), slots.end());
732  verify(isConsistent(slots));
733  cleanUpArcs(2);
734  endPivotArcPtr_ = endArcPtr_;
735  return nAdded;
736  }
737 
738  // find pivot arcs, initialize a new slot for each pivot arc, and insert new slot preserving time order
739  u32 addPivotArcs(PivotArcCn::SlotPtrList& slots) const {
740  u32 nAdded = 0;
741  std::sort(beginArcPtr_, endArcPtr_, PivotArcCn::Arc::SortByPosteriorScore());
742  for (PivotArcCn::ArcPtrList::iterator itArcPtr = beginArcPtr_; itArcPtr != endArcPtr_; ++itArcPtr) {
743  PivotArcCn::Arc& arc = **itArcPtr;
744  // binary search for finding correct position for new pivot element; slots are sorted by time
745  bool insertSlot = false;
746  PivotArcCn::SlotPtrList::iterator itInsertSlot = slots.end();
747  if (!slots.empty()) {
748  s32 l = 0, r = slots.end() - slots.begin() - 1, m, maxM = slots.size() - 1;
749  for (;;) {
750  // m = (l + r) / 2;
751  m = (s32)((u32)(l + r) >> 1);
752  verify((l <= r) && (0 <= m) && (m <= maxM));
753  const PivotArcCn::Slot& slot = **(slots.begin() + m);
754  if (arc.start > slot.start) {
755  if (m == r) {
756  if ((slot.end <= arc.start) && ((m == maxM) || (arc.end <= (*(slots.begin() + m + 1))->start))) {
757  insertSlot = true;
758  itInsertSlot = slots.begin() + m + 1;
759  }
760  break;
761  }
762  l = m + 1;
763  }
764  else if (arc.start < slot.start) {
765  if (l == m) {
766  if ((arc.end <= slot.start) && ((m == 0) || ((*(slots.begin() + m - 1))->end <= arc.start))) {
767  insertSlot = true;
768  itInsertSlot = slots.begin() + m;
769  }
770  break;
771  }
772  r = m - 1;
773  }
774  else
775  break;
776  }
777  }
778  else
779  insertSlot = true;
780  if (insertSlot) {
781  PivotArcCn::Slot* newSlot = new PivotArcCn::Slot;
782  newSlot->start = arc.start;
783  newSlot->end = arc.end;
784  newSlot->arcs.push_back(&arc);
785  slots.insert(itInsertSlot, newSlot);
786  arc.slotDist = Core::Type<Score>::min;
787  arc.state = 2;
788  ++nAdded;
789  }
790  }
791  verify_(!slots.empty() && isConsistent(slots));
792  return nAdded;
793  }
794 
795  // find best matching slot for all arcs; some arcs might be left because of violation of time contraints
796  u32 addArcs(PivotArcCn::SlotPtrList& slots) const {
797  const Distance& distFcn = *distFcn_;
798  u32 nAdded = 0, lastNAdded = 0;
799  do {
800  lastNAdded = nAdded;
801  std::sort(beginArcPtr_, endArcPtr_, PivotArcCn::Arc::SortByTime());
802  PivotArcCn::ArcPtrList::iterator beginArcPtr = beginArcPtr_;
803  u32 slotId = 0;
804  for (PivotArcCn::SlotPtrList::const_iterator itSlotPtr = slots.begin(), endSlotPtr = slots.end(); itSlotPtr != endSlotPtr; ++itSlotPtr, ++slotId) {
805  const PivotArcCn::Slot& slot = **itSlotPtr;
806  for (; (beginArcPtr != endArcPtr_) && ((*beginArcPtr)->end <= slot.start); ++beginArcPtr)
807  ;
808  for (PivotArcCn::ArcPtrList::iterator itArcPtr = beginArcPtr; (itArcPtr != endArcPtr_) && ((*itArcPtr)->start < slot.end); ++itArcPtr) {
809  PivotArcCn::Arc& arc = **itArcPtr;
810  Score slotDist = distFcn.distance(slot, arc);
811  if (slotDist < arc.slotDist) {
812  arc.slotDist = slotDist;
813  arc.slotId = slotId;
814  }
815  }
816  }
817 
818  std::sort(beginArcPtr_, endArcPtr_, PivotArcCn::Arc::SortByDistance());
819  for (PivotArcCn::ArcPtrList::iterator itArcPtr = beginArcPtr_; itArcPtr != endArcPtr_; ++itArcPtr) {
820  PivotArcCn::Arc& arc = **itArcPtr;
821  if (arc.slotId < slots.size()) {
822  PivotArcCn::Slot& slot = *slots[arc.slotId];
823  Time start = std::max(arc.start, slot.start), end = std::min(arc.end, slot.end);
824  if (start < end) {
825  slot.arcs.push_back(&arc);
826  slot.start = start;
827  slot.end = end;
828  arc.state = 1;
829  ++nAdded;
830  }
831  }
832  }
833 
834  cleanUpArcs();
835 
836  } while (lastNAdded < nAdded);
837  verify_(isConsistent(slots));
838  return nAdded;
839  }
840 
841  void addSlots(PivotArcCn::SlotPtrList& slots, PivotArcCn::SlotPtrList& addSlots) const {
842  PivotArcCn::SlotPtrList::iterator itSlotPtr = slots.begin(), itAddSlotPtr = addSlots.begin();
843  while ((itSlotPtr != slots.end()) && (itAddSlotPtr != addSlots.end())) {
844  if ((*itAddSlotPtr)->start < (*itSlotPtr)->start) {
845  itSlotPtr = slots.insert(itSlotPtr, *itAddSlotPtr);
846  ++itSlotPtr;
847  ++itAddSlotPtr;
848  }
849  else
850  ++itSlotPtr;
851  }
852  if (itAddSlotPtr != addSlots.end())
853  slots.insert(slots.end(), itAddSlotPtr, addSlots.end());
854  }
855 
856  /**
857  * Algorithm 1 (fast):
858  * 1) Initialize CN with top best arcs; remove top best aka pivot arcs
859  * 2) for all arcs finds best matching slot
860  * 3) sort arcs by similarity and assign arc to slot until all remaining arcs violate time contraints;
861  * remove all assigned arcs
862  * goto 2) until no more arcs are assigned
863  * 4) sort arcs by posterior and find new pivot arcs; remove pivot arcs
864  * goto 2) until no more arcs are left
865  *
866  * Algorithm 2:
867  * 1) Initialize CN with top best arcs; remove top best aka pivot arcs
868  * 2) for all arcs finds best matching slot
869  * 3) sort arcs by similarity and assign arc to slot until all remaining arcs violate time contraints;
870  * temporarily remove all assigned arcs
871  * goto 2) until no more arcs are assigned
872  * 4) sort arcs by posterior and find new pivot arcs; remove pivot arcs, put back all temporarily removed arcs
873  * goto 2) until no more arcs are left
874  **/
875  void buildPivotArcCn(ConstLatticeRef l, ConstFwdBwdRef fb) const {
876  // initialize CN, use viterbi path for initialization
877  init(l, fb);
878  // build CN
879  PivotArcCn::SlotPtrList& slots = pcn_->slots_;
880  PivotArcCn::SlotPtrList newSlots;
881  addArcs(slots);
882  while (beginArcPtr_ != endArcPtr_) {
883  verify_(isConsistent(slots));
884 
885  addPivotArcs(newSlots);
886  if (fast_) {
887  // add only non-member arcs
888  cleanUpArcs();
889  addArcs(newSlots);
890  addSlots(slots, newSlots);
891  }
892  else {
893  // (re-)assign all member- and non-member-arcs
894  addSlots(slots, newSlots);
895  cleanUpArcsAndSlots();
896  addArcs(slots);
897  }
898  newSlots.clear();
899  }
900  verify(isConsistent(pcn_->slots_));
901  }
