230 class CorpusDescription::SegmentPartitionVisitorAdaptor : public CorpusVisitor {
231 private:
232  u32 segmentIndex_, recordingIndex_, nPartitions_, selectedPartition_, nSkippedSegments_;
233  Core::StringHashSet segmentsToSkip_; // blacklist
234  Core::StringHashSet segmentsToKeep_; // whitelist
235  Recording* currentRecording_;
236  bool isVisitorInCurrentRecording_;
237  bool recordingBasedPartitions_;
238  CorpusVisitor* visitor_;
239 
240  bool shouldVisit(Segment* s) {
241  u32 index = recordingBasedPartitions_ ? recordingIndex_ : segmentIndex_;
242  bool isSelectedPartition = (index % nPartitions_) == selectedPartition_;
243  bool hasSkippedEnoughSegments = ((segmentIndex_ / nPartitions_) >= nSkippedSegments_);
244  bool shouldNotSkip = segmentsToSkip_.empty() || (segmentsToSkip_.find(s->fullName()) == segmentsToSkip_.end());
245  bool shouldKeep = segmentsToKeep_.empty() || (segmentsToKeep_.find(s->fullName()) != segmentsToKeep_.end()) || (segmentsToKeep_.find(s->name()) != segmentsToKeep_.end());
246  ++segmentIndex_;
247  if (isSelectedPartition && hasSkippedEnoughSegments && shouldNotSkip && shouldKeep) {
248  if (!isVisitorInCurrentRecording_) {
249  visitor_->enterRecording(currentRecording_);
250  isVisitorInCurrentRecording_ = true;
251  }
252  return true;
253  }
254  else {
255  return false;
256  }
257  }
258 
259 public:
260  SegmentPartitionVisitorAdaptor()
261  : visitor_(0) {
262  nPartitions_ = 1;
263  selectedPartition_ = 0;
264  recordingBasedPartitions_ = false;
265  nSkippedSegments_ = 0;
266  }
267  void loadSegmentList(const std::string& filename, const std::string& encoding);
268  const Core::StringHashSet& segmentsToKeep() const {
269  return segmentsToKeep_;
270  }
271  void setPartitioning(u32 nPartitions, u32 selectedPartition, bool recordingBased = false) {
272  require(selectedPartition < nPartitions);
273  nPartitions_ = nPartitions;
274  selectedPartition_ = selectedPartition;
275  recordingBasedPartitions_ = recordingBased;
276  }
277  void setSkippedSegments(u32 nSkippedSegments) {
278  nSkippedSegments_ = nSkippedSegments;
279  }
280  void setSegmentsToSkip(const Core::StringHashSet& segmentsToSkip) {
281  segmentsToSkip_ = segmentsToSkip;
282  }
283  void setVisitor(CorpusVisitor* v) {
284  visitor_ = v;
285  }
286  virtual void visitSegment(Segment* s) {
287  if (shouldVisit(s))
288  visitor_->visitSegment(s);
289  }
290  virtual void visitSpeechSegment(SpeechSegment* s) {
291  if (shouldVisit(s))
292  visitor_->visitSpeechSegment(s);
293  }

298  }

316 void CorpusDescription::SegmentPartitionVisitorAdaptor::loadSegmentList(const std::string& filename, const std::string& encoding) {
317  if (!filename.empty()) {
318  Core::CompressedInputStream* cis = new Core::CompressedInputStream(filename.c_str());
319  Core::TextInputStream is(cis);
320  is.setEncoding(encoding);
321  if (!is)
322  Core::Application::us()->criticalError("Failed to open segment list file \"%s\".", filename.c_str());
323  std::string s;
324  while (Core::getline(is, s) != EOF) {
325  if ((s.size() == 0) || (s.at(0) == '#'))
326  continue;
327  Core::stripWhitespace(s);
328  segmentsToKeep_.insert(s);
329  }
330  }
331 }

466 CorpusDescription::CorpusDescription(const Core::Configuration& c)
467  : Component(c),
468  selector_(0),
469  progressChannel_(c, "progress"),
470  reporter_(0),
471  indicator_(0),
472  ordering_(0) {
473  filename_ = paramFilename(config);
474 
475  s32 partitioning = paramPartition(config);
476  u32 skipFirstSegments = paramSkipFirstSegments(config);
477  const std::vector<std::string> segmentFullNames = paramSegmentsToSkip(config);
478  std::string segmentsFilename = paramFilename(select("segments"));
479  if (partitioning || skipFirstSegments || !segmentFullNames.empty() || !segmentsFilename.empty()) {
480  selector_ = new SegmentPartitionVisitorAdaptor;
481  }
482 
483  if (partitioning) {
484  s32 selectedPartition = paramPartitionSelection(config);
485  bool recordingBasedPartition = paramRecordingBasedPartition(config);
486  if (selectedPartition == partitioning)
487  selectedPartition = 0; // This convention is useful for SGE array jobs
488  else if (selectedPartition > partitioning)
489  error("Invalid partition %d (should be 0 - %d).", selectedPartition, partitioning);
490  selector_->setPartitioning(partitioning, selectedPartition, recordingBasedPartition);
491  }
492  if (skipFirstSegments) {
493  selector_->setSkippedSegments(skipFirstSegments);
494  }
495  if (!segmentFullNames.empty()) {
496  Core::StringHashSet segmentsToSkip;
497  for (u32 i = 0; i < segmentFullNames.size(); ++i) {
498  segmentsToSkip.insert(segmentFullNames[i]);
499  }
500  selector_->setSegmentsToSkip(segmentsToSkip);
501  }
502  if (!segmentsFilename.empty()) {
503  selector_->loadSegmentList(segmentsFilename, paramEncoding(select("segments")));
504  if (selector_->segmentsToKeep().empty() && !paramAllowEmptyWhitelist(select("segments")))
505  error("Discard segment whitelist, because file is empty or does not exist: ") << segmentsFilename;
506  else
507  log("Use a segment whitelist with %d entries, keep only listed segments.", u32(selector_->segmentsToKeep().size()));
508  }
509 
510  // Handle the ordering.
511  {
512  if (paramTheanoSegmentOrder(config)) {
513 #ifdef MODULE_THEANO_INTERFACE
514  verify(!ordering_);
515  ordering_ = new TheanoSegmentOrderingVisitor();
516  log("Using Theano segment ordering");
517 #else
518  criticalError("theano-segment-order not possible, MODULE_THEANO_INTERFACE disabled.");
519 #endif
520  }
521  if (paramPythonSegmentOrder(config)) {
522 #ifdef MODULE_PYTHON
523  if (ordering_)
524  criticalError("python-segment-order not possible, another ordering (theano?) already used");
525  std::string pyModPath = paramPythonSegmentOrderModPath(config);
526  std::string pyModName = paramPythonSegmentOrderModName(config);
527  std::string pyConfig = paramPythonSegmentOrderConfig(config);
528  if (pyModName.empty()) {
529  criticalError("python-segment-order: need Python module name (%s)",
530  paramPythonSegmentOrderModPath.name().c_str());
531  return;
532  }
533  ordering_ = new PythonSegmentOrderingVisitor(pyModPath, pyModName, pyConfig, *this);
534 #else
535  criticalError("python-segment-order not possible, MODULE_PYTHON disabled.");
536 #endif
537  }
538  std::string segmentOrder = paramSegmentOrder(config);
539  if (!segmentOrder.empty()) {
540  if (!ordering_)
541  ordering_ = new SegmentOrderingVisitor();
542  log("Using segment order list '%s'", segmentOrder.c_str());
543  ordering_->setSegmentList(segmentOrder);
544  }
545  bool segmentOrderShuffle = paramSegmentOrderShuffle(config);
546  if (segmentOrderShuffle) {
547  if (!ordering_)
548  ordering_ = new SegmentOrderingVisitor();
549  ordering_->setAutoShuffle(true);
550  s32 seed = paramSegmentOrderShuffleSeed(config);
551  if (seed == -1)
552  // Fallback to other seed config setting which is used e.g. by NnTrainer and others.
553  seed = Core::ParameterInt("seed", "seed", -1)(config);
554  if (seed == -1)
555  seed = 0;
556  ordering_->shuffleRandomSeed((u32)seed);
557  log("Using segment order shuffling with seed %i", seed);
558  }
559  if (paramSegmentOrderSortByTimeLength(config)) {
560  if (!ordering_)
561  ordering_ = new SegmentOrderingVisitor();
562  s32 chunkSize = paramSegmentOrderSortByTimeLengthChunkSize(config);
563  ordering_->setSortByTimeLength(true, chunkSize);
564  log("Using segment order sort-by-time-length with chunk-size %i", chunkSize);
565  }
566 
567  if (ordering_) {
568  ordering_->setShortNameLookup(paramSegmentOrderLookupName(config));
569  }
570  }
571 
572  progressIndicationMode_ = ProgressIndcationMode(paramProgressIndication(config));
573  u32 nSegments = 0;
574  switch (progressIndicationMode_) {
575  case globalProgress: {
576  nSegments = totalSegmentCount();
577  } // fall through
578  case localProgress:
579  indicator_ = new ProgressIndicationVisitorAdaptor();
580  indicator_->setTotal(nSegments);
581  break;
582  case noProgress: break;
583  default: defect();
584  }
585 
586  if (progressChannel_.isOpen()) {
587  reporter_ = new ProgressReportingVisitorAdaptor(progressChannel_, paramProgressReportingSegmentOrth(config));
588  }
589 }

619 u32 CorpusDescription::totalSegmentCount() {
620  SegmentCountingVisitor* counter = new SegmentCountingVisitor();
621  counter->reset();
622  // Note: An accept() call is problematic because we don't
623  // want certain side-effects like the reporter or the indicator.
624  // Also, it would change the internal state of ordering,
625  // thus we need to have our own copy here to not change the state.
626  SegmentOrderingVisitor* ordering = ordering_ ? ordering_->copy() : 0;
627  {
628  CorpusDescriptionParser parser(config);
629  CorpusVisitor* visitor = counter;
630  if (selector_) {
631  selector_->setVisitor(visitor);
632  visitor = selector_;
633  }
634  if (ordering) {
635  ordering->setVisitor(visitor);
636  visitor = ordering;
637  }
638  parser.accept(file(), visitor);
639  }
640  delete ordering;
641  u32 nSegments = counter->nSegments();
642  delete counter;
643  return nSegments;
644 }
