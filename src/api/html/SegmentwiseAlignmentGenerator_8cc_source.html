83 void SegmentwiseAlignmentGenerator::update(const std::string& segmentId) {
84  acousticModel()->setKey(segmentId);
85  if (segmentwiseFeatureExtractor_ && checkCompatibility_) {
86  segmentwiseFeatureExtractor_->checkCompatibility(portId_, acousticModel());
87  }
88  features_.reset();
89 
90  Core::Ref<const Mm::FeatureScorer> featureScorer(modelCombination()->acousticModel()->featureScorer());
91  if (featureScorer->hasTimeIndexedCache() && featureScorer->isBuffered()) {
92  // We need to provide all features now.
93  featureScorer->reset();
94  size_t t = 0;
95  for (SegmentwiseFeatures::const_iterator f = features()->begin(), featureEnd = features()->end(); f != featureEnd; ++f, ++t) {
96  if (featureScorer->bufferFilled()) {
97  criticalError("Buffer filled after %zu/%zu frames.", t, features()->size());
98  break;
99  }
100  featureScorer->addFeature(*f);
101  }
102  }
103 }

126 void SegmentwiseAlignmentGenerator::getScorers(TimeframeIndex tbeg, TimeframeIndex tend, std::vector<Mm::FeatureScorer::Scorer>& scorers) const {
127  if (features()->size() < tend) {
128  if (features()->size() <= tbeg) {
129  criticalError("Requested alignment is out of range, [%d,%d] not in [0,%zu].",
130  tbeg, tend, features()->size());
131  }
132  else {
133  warning("Requested alignment is out of range, [%d,%d] not in [0,%zu]; align [%d,%zu].",
134  tbeg, tend, features()->size(), tbeg, features()->size());
135  tend = features()->size();
136  }
137  }
138 
139  /*
140  * Do not use iterators here because pointer arithmetic
141  * and polymorphism (e.g. Sparse::SegmentwiseFeatures)
142  * are not compatible.
143  */
144  require(tbeg <= tend);
145  scorers.resize(tend - tbeg);
146  Core::Ref<const Mm::FeatureScorer> featureScorer(modelCombination()->acousticModel()->featureScorer());
147  if (featureScorer->hasTimeIndexedCache()) {
148  for (u32 t = tbeg; t < tend; ++t) {
149  scorers[t - tbeg] = featureScorer->getTimeIndexedScorer(t);
150  }
151  }
152  else {
153  for (u32 t = tbeg; t < tend; ++t) {
154  Core::Ref<Mm::Feature> f = (*features())[t];
155  scorers[t - tbeg] = featureScorer->getScorer(f);
156  }
157  }
158 }

208 Core::Archive* OrthographyAlignmentGenerator::createAlignmentCache(const Core::Configuration& config) {
209  Core::Archive* alignmentCache = Core::Archive::create(config, paramPath(config), Core::Archive::AccessModeReadWrite);
210  Core::ArchiveReader ar(*alignmentCache, "DEPENDENCIES");
211  if (ar.isOpen()) {
212  Core::DependencySet archiveDependencies;
213  if (!archiveDependencies.read(config, ar)) {
214  error("Failed to read dependencies from alignment cache \"%s\".", alignmentCache->path().c_str());
215  }
216  else if (archiveDependencies == dependencies()) {
217  warning("Archive \"%s\" exists already and will be overwritten.", alignmentCache->path().c_str());
218  }
219  }
220  else {
221  Core::XmlOutputStream xw(new Core::ArchiveWriter(*alignmentCache, "DEPENDENCIES"));
222  dependencies().write(xw);
223  }
224  return alignmentCache;
225 }

234 Fsa::ConstAutomatonRef OrthographyAlignmentGenerator::getAlignmentFsa() {
235  verify(!orthography_.empty());

237  // allophone state to lemma-pronunciation transducer
238  Fsa::ConstAutomatonRef transducer = transducerCache_->get(allophoneStateGraphBuilder_->createTransducerFunctor(segmentId_, orthography_));
239  wordLatticeBuilder_->setModelTransducer(transducer);
240  // ???
241  Fsa::ConstAutomatonRef modelAcceptor = modelAcceptorCache_->get(allophoneStateGraphBuilder_->createFinalizationFunctor(segmentId_, transducer));
242  aligner_.setModel(modelAcceptor, acousticModel());
243 
244  std::vector<Mm::FeatureScorer::Scorer> scorers;
245  getScorers(scorers);
246  aligner_.feed(scorers);
247  if (aligner_.reachedFinalState()) {
248  alignmentFsa_ = aligner_.getAlignmentFsa();
249  verify(alignmentFsa_);

251  else
252  warning("Alignment did not reach any final state in segment '%s'", segmentId_.c_str());

258 const Alignment* OrthographyAlignmentGenerator::getAlignment(const std::string& orthography) {
259  setOrthography(orthography);
260  if (alignmentCache_) {
261  Core::ArchiveReader reader(*alignmentCache_, segmentId_);
262  if (reader.isOpen()) {
263  Alignment* alignment = new Alignment;
264  Core::BinaryInputStream bi(reader);
265  bi >> *alignment;
266  if (!bi)
267  error("Failed to read alignment \"%s\" from cache \"%s\".", segmentId_.c_str(), alignmentCache_->path().c_str());
268  return alignment;
269  }
270  }
271  Fsa::ConstAutomatonRef alignmentFsa = getAlignmentFsa();
272  if (!alignmentFsa)
273  return 0;
274  std::pair<Fsa::ConstAutomatonRef, Fsa::Weight> alignmentPosteriorFsa = aligner_.getAlignmentPosteriorFsa(alignmentFsa);
275  Alignment* alignment = new Alignment;
276  aligner_.getAlignment(*alignment, alignmentPosteriorFsa);
277  if (alignmentCache_) {
278  Core::ArchiveWriter writer(*alignmentCache_, segmentId_);
279  if (writer.isOpen()) {
280  Core::BinaryOutputStream bo(writer);
281  bo << *alignment;
282  if (!bo)
283  error("Failed to write alignment \"%s\" to cache \"%s\".", segmentId_.c_str(), alignmentCache_->path().c_str());
284  }
285  else {
286  error("Could not open file \"%s\" for writing in cache \"%s\".", segmentId_.c_str(), alignmentCache_->path().c_str());
287  }
288  }
289  return alignment;
290 }
