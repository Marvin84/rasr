42 /** log ( sum_i exp(v_i) ) */
43 template<class InputIterator>
44 double logExpNorm(InputIterator begin, InputIterator end) {
45  InputIterator maxIt = std::max_element(begin, end);
46  double result = 0;
47  for (; begin != end; ++begin)
48  if (begin != maxIt)
49  result += exp(*begin - *maxIt);
50  return log1p(result) + *maxIt;
51 };

61 /** sum_i ( weight_i * log(v_i) )*/
62 template<class InputIterator, class WeightIterator>
63 double logNorm(InputIterator beginDiagonal, InputIterator endDiagonal, WeightIterator beginWeights, WeightIterator endWeights) {
64  double result = 0;
65  for (; beginDiagonal != endDiagonal; ++beginDiagonal) {
66  result += (*beginWeights) * log(Core::abs(*beginDiagonal));
67  ++beginWeights;
68  }
69  return result;
70 };

245  DistanceType df, score = 0;

247  switch ((dimension - 1) % 8) {

249  case 7:

331 template<class O>
332 class Cache2D {
333  std::vector<std::pair<bool, std::vector<bool>>> isCalculated_;
334  O** objects_;
335  bool freedStorage_;
336 
337 public:
338  Cache2D(size_t first_level_size)
339  : isCalculated_(first_level_size, std::make_pair(false, std::vector<bool>())),
340  freedStorage_(false) {
341  objects_ = new O*[first_level_size];
342  for (size_t i = 0; i < first_level_size; ++i) {
343  objects_[i] = 0;
344  }

346 
347  /** deep copy constructor to prevent side effects in combination with STL **/
348  Cache2D(const Cache2D<O>& other) {
349  // copy is calculated
350  isCalculated_ = other.isCalculated_;
351  size_t first_level_size = (size_t)(other.isCalculated_.size());
352 
353  // allocate memory for first level
354  objects_ = new O*[first_level_size];
355  // allocate memory for second level and copy contents
356  for (size_t i = 0; i < first_level_size; ++i) {
357  if (other.objects_[i]) {
358  // get correct size
359  u32 tsize = isCalculated_[i].second.size();
360  objects_[i] = new O[tsize];
361  O*& obj = objects_[i];
362  O*& otherObj = other.objects_[i];
363  for (u32 j = 0; j < tsize; ++j) {
364  obj[j] = otherObj[j];
365  }
366  }
367  else {
368  objects_[i] = 0;
369  }
370  }
371  }

373  ~Cache2D() {
374  if (!freedStorage_) {
375  for (size_t i = 0; i < isCalculated_.size(); ++i) {
376  if (objects_[i]) {
377  delete[](objects_[i]);
378  }
