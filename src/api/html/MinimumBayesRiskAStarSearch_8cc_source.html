73 /**
74  * Computes one column of the DP levenshtein distance algorithm.
75  */
76 inline void computeLevenshteinColumn(const Fsa::LabelId symbol,
77  const std::vector<Fsa::LabelId>& hypothesis,
78  const std::vector<u32>& oldScores,
79  std::vector<u32>& newScores,
80  const u32 initialNewScore) {
81  require(oldScores.size() == newScores.size());
82  require(newScores.size() == hypothesis.size() + 1);
83 
84  newScores.at(0) = initialNewScore;
85  for (u32 n = 1; n < hypothesis.size() + 1; ++n) {
86  u32 del = oldScores.at(n) + 1;
87  u32 ins = newScores.at(n - 1) + 1;
88  u32 sub = oldScores.at(n - 1) + (hypothesis.at(n - 1) == symbol ? 0 : 1);
89  newScores.at(n) = (del <= ins ? (del <= sub ? del : (ins <= sub ? ins : sub)) : (ins <= sub ? ins : sub));
90  }
91 }

94  u32 minimum = Core::Type<u32>::max;

259 void MinimumBayesRiskAStarSearch::expand() {
260  ++numberEvaluations_;
261 
262  // As the values of the expansion node are needed for the construction of the new grid nodes,
263  // this reference serves as a temporary copy of the expansion node. Processing this way, the
264  // deletion can be safely carried out without further effort.
265  SearchNodeRef expansionNode = minimalIncompleteNode_;
266 
267  expansionNode->isExplorable_ = false;
268 
269  if (fsa_->getState(expansionNode->begin()->first)->isFinal()) {
270  expansionNode->setFinal();
271  }
272 
273  // The comparison and optimality test make use of null compares. As a result, the corresponding
274  // references have to be initialized by null.
275  minimalNode_ = SearchNodeRef();
276  secondMinimalNode_ = SearchNodeRef();
277  minimalIncompleteNode_ = SearchNodeRef();
278 
279  // In case the current expanded node is also final, it must not be deleted from the stack
280  // but instead, its overestimate has to be computed. In fact, this is the only case during
281  // which the initial overestimate will be computed.
282  // In addition, a heuristic has been prepared to stop computation if the minimal node remains the
283  // same over 50 iterations.
284  if (expansionNode->isFinal()) {
285  if (!newMinimalNode_ || newMinimalNode_ != expansionNode) {
286  newMinimalNode_ = expansionNode;
287  newMinimalNodeCounter_ = 1;
288  }
289  else {
290  if (++newMinimalNodeCounter_ > 50) {
291  minimalNode_ = newMinimalNode_;
292  stack_->clear();
293  return;
294  }
295  }
296 
297  // needed for prefix tree search.
298  Fsa::Accumulator* collector = Fsa::LogSemiring->getCollector(Fsa::LogSemiring->zero());
299  collector->feed(expansionNode->finalProbability_);
300  std::map<Fsa::StateId, Fsa::Weight>::const_iterator stateId = expansionNode->begin();
301  for (; stateId != expansionNode->end(); ++stateId) {
302  collector->feed(
303  semiring_->extend(
304  semiring_->extend(
305  stateId->second,
306  fsa_->getState(stateId->first)->weight()),
307  inverseBackwardPotentialsNormalizationConstant_));
308  }
309  expansionNode->totalProbability_ = collector->get();
310  delete collector;
311 
312  // one-half-criterion
313  if (f32(expansionNode->totalProbability_) < -std::log(0.5)) {
314  minimalNode_ = expansionNode;
315  stack_->clear();
316  return;
317  }
318 
319  overestimateFinal(expansionNode);
320  }
321  else {
322  // In case the current expansion node is not a final one, it has to be deleted from the stack. In addition, all
323  // score entries have to be deleted for all other stack nodes. As expansionNode is a copy of the deleted node,
324  // all values for creation of new nodes are available.
325  // In addition, the old entry has to be deleted from the pruning stack.
326  stack_->erase(stack_->begin() + expansionNode->index_);
327  std::vector<SearchNodeRef>::iterator stackNode = stack_->begin();
328  for (; stackNode != stack_->end(); ++stackNode) {
329  (*stackNode)->levenshteinColumns_.erase((*stackNode)->levenshteinColumns_.begin() + expansionNode->index_);
330  (*stackNode)->levenshteinScores_.erase((*stackNode)->levenshteinScores_.begin() + expansionNode->index_);
331  }
332 
333  if (!stackEntries_.empty()) {
334  std::vector<SearchNodeRef>& oldEntries = stackEntries_.at(expansionNode->hypothesis_.size());
335  std::vector<SearchNodeRef>::iterator oldNodeToDelete = oldEntries.begin();
336  for (; oldNodeToDelete != oldEntries.end(); ++oldNodeToDelete) {
337  if ((*oldNodeToDelete)->index_ == expansionNode->index_) {
338  oldEntries.erase(oldNodeToDelete);
339  break;
340  }
341  }
342  }
343  }
344 
345  std::vector<SearchNodeRef> newSearchNodes = createNewSearchNodes(expansionNode);
346 
347  u32 newLength_ = expansionNode->hypothesis_.size() + 1;
348  if (stackEntries_.size() < newLength_ + 1) {
349  stackEntries_.resize(newLength_ + 2);
350  }
351 
352  updateStack(expansionNode, newSearchNodes);
353  generateNewGridNodes(expansionNode, newSearchNodes);
354  numberComputations_ += stack_->size() * stack_->size();
355  pruneStack();
356  newMinimalNode_ = minimalNode_;
357  if (dumpStack_) {
358  dump(clog(), fsa_->inputAlphabet());
359  }
360 }

362 std::vector<SearchNodeRef> MinimumBayesRiskAStarSearch::createNewSearchNodes(ConstSearchNodeRef expansionNode) {
363  std::vector<SearchNodeRef> result;
364 
365  // This will be needed for the prefix tree search. All new created nodes will share the string of the expansion node
366  // as prefix. In case the lattice is nondeterministic, there may exist several paths belonging to this prefix, so all
367  // target states have to be determined for the same prefix. As the new hypothesis differ in exactly one word/label,
368  // the corresponding data structure was chosen to be a map.
369  std::map<Fsa::LabelId, SearchNodeRef> newNodes;
370 
371  u32 exIndex = (expansionNode->isFinal() ? Core::Type<u32>::max : expansionNode->index_);
372 
373  // generate new Stack nodes from expansion node.
374  std::map<Fsa::StateId, Fsa::Weight>::const_iterator stateId = expansionNode->begin();
375  for (; stateId != expansionNode->end(); ++stateId) {
376  Fsa::ConstStateRef state = fsa_->getState(stateId->first);
377  Fsa::State::const_iterator arc = state->begin();
378  for (; arc != state->end(); ++arc) {
379  NodeMap::iterator mapEntry = newNodes.find(arc->input());
380  if (mapEntry == newNodes.end()) {
381  mapEntry = newNodes.insert(mapEntry,
382  std::make_pair(arc->input(),
383  SearchNodeRef(new SearchNode(expansionNode->hypothesis_,
384  arc->input(),
385  Fsa::LogSemiring->zero(),
386  expansionNode->levenshteinColumns_,
387  expansionNode->levenshteinScores_,
388  Fsa::LogSemiring->max(),
389  exIndex))));
390  }
391 
392  Fsa::ConstStateRef targetState = fsa_->getState(arc->target());
393  Fsa::Weight forwardProbability = semiring_->extend(stateId->second, arc->weight());
394 
395  mapEntry->second->addState(arc->target(),
396  distances_.at(arc->target()),
397  forwardProbability,
398  backwardPotentials_.at(arc->target()),
399  inverseBackwardPotentialsNormalizationConstant_);
400  }
401  }
402 
403  // In case a prefix tree search will be performed, the probability mass has to be accumulated,
404  // because there may exist several paths for the string, a search node represents.
405  NodeMap::iterator newSearchNode = newNodes.begin();
406  for (; newSearchNode != newNodes.end(); ++newSearchNode) {
407  newSearchNode->second->computeTotalProbability();
408  result.push_back(newSearchNode->second);
409  }
410 
411  return result;
412 }

414 void MinimumBayesRiskAStarSearch::updateStack(
415  SearchNodeRef expansionNode, std::vector<SearchNodeRef>& newNodes) {
416  if (stack_->size() < 1) {
417  return;
418  }
419  for (u32 n = 0; n < stack_->size(); ++n) {
420  SearchNodeRef competingSearchNode = stack_->at(n);
421  competingSearchNode->index_ = n;
422  if (competingSearchNode->isFinal()) {
423  ++numberOverestimates_;
424  }
425 
426  std::vector<u32> competingLevenshteinColumn = competingSearchNode->levenshteinColumns_.at(expansionNode->index_);
427 
428  std::vector<SearchNodeRef>::iterator newSearchNode = newNodes.begin();
429  for (; newSearchNode != newNodes.end(); ++newSearchNode) {
430  std::vector<u32> levenshteinColumn(competingSearchNode->hypothesis_.size() + 1);
431  computeLevenshteinColumn(
432  (*newSearchNode)->hypothesis_.back(),
433  competingSearchNode->hypothesis_,
434  competingLevenshteinColumn,
435  levenshteinColumn,
436  expansionNode->hypothesis_.size() + 1);
437 
438  // compute levenshtein score
439  if (competingSearchNode->isFinal()) {
440  if (!exactEstimate_) {
441  u32 longestDistance = (*newSearchNode)->longestDistance_;
442  std::vector<u32> newLevenshteinColumn = levenshteinColumn;
443  std::vector<u32> tmpLevenshteinColumn;
444  for (u32 l = 0; l < longestDistance; ++l) {
445  tmpLevenshteinColumn = newLevenshteinColumn;
446  computeLevenshteinColumn(
447  Fsa::InvalidLabelId,
448  competingSearchNode->hypothesis_,
449  tmpLevenshteinColumn,
450  newLevenshteinColumn,
451  (*newSearchNode)->hypothesis_.size() + l + 1);
452  }
453  Fsa::Weight levenshteinScoreOverestimate = Fsa::LogSemiring->extend(LogWeights.at(newLevenshteinColumn.back()),
454  (*newSearchNode)->totalProbability_);
455  competingSearchNode->levenshteinScoresOverestimate_.push_back(levenshteinScoreOverestimate);
456  }
457  }
458 
459  Fsa::Weight levenshteinScore = Fsa::LogSemiring->extend(LogWeights.at(minimum(levenshteinColumn)),
460  (*newSearchNode)->totalProbability_);
461  competingSearchNode->levenshteinScores_.push_back(levenshteinScore);
462 
463  (*newSearchNode)->levenshteinColumns_.at(n).push_back(levenshteinColumn.back());
464 
465  (*newSearchNode)->levenshteinScores_.at(n) = Fsa::LogSemiring->extend(Fsa::Weight(LogWeights.at(minimum((*newSearchNode)->levenshteinColumns_.at(n)))),
466  competingSearchNode->totalProbability_);
467 
468  competingSearchNode->levenshteinColumns_.push_back(levenshteinColumn);
469  }
470 
471  if (competingSearchNode->isExplorable_ && (!minimalIncompleteNode_ || competingSearchNode < minimalIncompleteNode_)) {
472  minimalIncompleteNode_ = competingSearchNode;
473  }
474 
475  Fsa::Accumulator* collector = Fsa::LogSemiring->getCollector(Fsa::LogSemiring->zero());
476  for (u32 i = 0; i < competingSearchNode->levenshteinScores_.size(); ++i) {
477  collector->feed(competingSearchNode->levenshteinScores_[i]);
478  }
479  competingSearchNode->estimate_ = collector->get();
480  delete collector;
481 
482  if (competingSearchNode->isFinal()) {
483  Fsa::Accumulator* collector = Fsa::LogSemiring->getCollector(Fsa::LogSemiring->zero());
484  for (u32 i = 0; i < competingSearchNode->levenshteinScoresOverestimate_.size(); ++i) {
485  collector->feed(competingSearchNode->levenshteinScoresOverestimate_[i]);
486  }
487  competingSearchNode->overestimate_ = collector->get();
488  delete collector;
489  }
490 
491  if (!minimalNode_ || operator<(competingSearchNode, minimalNode_)) {
492  secondMinimalNode_ = minimalNode_;
493  minimalNode_ = competingSearchNode;
494  }
495  else if (!secondMinimalNode_ || competingSearchNode < secondMinimalNode_) {
496  secondMinimalNode_ = competingSearchNode;
497  }
498  }

501 void MinimumBayesRiskAStarSearch::overestimateFinal(SearchNodeRef finalNode) {
502  verify(finalNode->isFinal());
503  if (exactEstimate_) {
504  finalNode->estimate_ = posteriorExpectedRisk(
505  createLinearAutomatonFromVector(
506  finalNode->hypothesis_,
507  Fsa::Weight(0.0), //(*newSearchNode)->second->totalProbability_,
508  fsa_->inputAlphabet(),
509  fsa_->outputAlphabet(),
510  Fsa::TropicalSemiring),
511  fsa_);
512  finalNode->overestimate_ = finalNode->estimate_;
513  }
514  else {
515  // overestimate for final node.
516  {
517  for (u32 n = 0; n < stack_->size(); ++n) {
518  SearchNodeRef competingSearchNode = stack_->at(n);
519  if (!exactEstimate_) {
520  Fsa::Weight levenshteinScoreOverestimate;
521  if (competingSearchNode != finalNode) {
522  u32 longestDistance = competingSearchNode->longestDistance_;
523  std::vector<u32> newLevenshteinColumn = competingSearchNode->levenshteinColumns_.at(finalNode->index_);
524  std::vector<u32> tmpLevenshteinColumn;

527  computeLevenshteinColumn(Fsa::InvalidLabelId,
528  competingSearchNode->hypothesis_,
529  tmpLevenshteinColumn,

540  }
541  }

546 void MinimumBayesRiskAStarSearch::generateNewGridNodes(SearchNodeRef expansionNode, std::vector<SearchNodeRef>& newNodes) {
547  // generate grid nodes for every pair ( new, new )
548  std::vector<u32> newNodeLevenshteinEqualColumn(expansionNode->hypothesis_.size() + 2);
549  std::vector<u32> newNodeLevenshteinDifferColumn(expansionNode->hypothesis_.size() + 2);
550 
551  for (u32 s = 0; s < expansionNode->hypothesis_.size() + 2; ++s) {
552  newNodeLevenshteinEqualColumn.at(s) = expansionNode->hypothesis_.size() + 1 - s;
553  newNodeLevenshteinDifferColumn.at(s) = expansionNode->hypothesis_.size() + 1 - s;
554  }
555 
556  newNodeLevenshteinDifferColumn.at(expansionNode->hypothesis_.size() + 1) = 1;
557 
558  // add grid node for every new stack node to every new stack node
559  std::vector<SearchNodeRef>::iterator newSearchNode = newNodes.begin();
560  for (; newSearchNode != newNodes.end(); ++newSearchNode) {
561  std::vector<SearchNodeRef>::iterator newCompetingSearchNode = newNodes.begin();
562  for (; newCompetingSearchNode != newNodes.end(); ++newCompetingSearchNode) {
563  if ((*newSearchNode)->hypothesis_.back() == (*newCompetingSearchNode)->hypothesis_.back()) {
564  (*newSearchNode)->levenshteinColumns_.push_back(newNodeLevenshteinEqualColumn);
565  (*newSearchNode)->levenshteinScores_.push_back(semiring_->zero());
566  }
567  else {
568  (*newSearchNode)->levenshteinColumns_.push_back(newNodeLevenshteinDifferColumn);
569  (*newSearchNode)->levenshteinScores_.push_back((*newCompetingSearchNode)->totalProbability_);
570  }
571  }
572 
573  Fsa::Accumulator* collector = Fsa::LogSemiring->getCollector(Fsa::LogSemiring->zero());
574  for (u32 i = 0; i < (*newSearchNode)->levenshteinScores_.size(); ++i) {
575  collector->feed((*newSearchNode)->levenshteinScores_[i]);
576  }
577  (*newSearchNode)->estimate_ = collector->get();
578  delete collector;
579 
580  (*newSearchNode)->index_ = stack_->size();
581  stack_->push_back(*newSearchNode);
582  stackEntries_.at(newLength_).push_back(*newSearchNode);
583 
584  if (!minimalNode_ || (*newSearchNode) < minimalNode_) {
585  secondMinimalNode_ = minimalNode_;
586  minimalNode_ = *newSearchNode;
587  }
588  else if (!secondMinimalNode_ || (*newSearchNode) < secondMinimalNode_) {
589  secondMinimalNode_ = (*newSearchNode);
590  }
591 
592  if ((*newSearchNode)->isExplorable_ && (!minimalIncompleteNode_ || (*newSearchNode) < minimalIncompleteNode_)) {
593  minimalIncompleteNode_ = (*newSearchNode);
594  }
595  }

598 void MinimumBayesRiskAStarSearch::pruneStack() {
599  if (stackEntries_.at(newLength_).size() > maxStackSize_) {
600  sort(stackEntries_.at(newLength_).begin(), stackEntries_.at(newLength_).end());
601 
602  std::vector<u32> pruneIndices;
603  bool addIncomplete = 0;
604  std::vector<SearchNodeRef>::iterator pruneNode = stackEntries_.at(newLength_).begin() + maxStackSize_;
605  for (; pruneNode != stackEntries_.at(newLength_).end(); ++pruneNode) {
606  if ((*pruneNode)->index_ != minimalIncompleteNode_->index_) {
607  pruneIndices.push_back((*pruneNode)->index_);
608  }
609  else {
610  addIncomplete = true;
611  }
612  }
613 
614  sort(pruneIndices.begin(), pruneIndices.end());
615 
616  std::vector<SearchNodeRef>::iterator stackNode = stack_->begin();
617  for (; stackNode != stack_->end(); ++stackNode) {
618  std::vector<u32>::reverse_iterator pruneIndex = pruneIndices.rbegin();
619  for (; pruneIndex != pruneIndices.rend(); ++pruneIndex) {
620  (*stackNode)->levenshteinScores_.erase((*stackNode)->levenshteinScores_.begin() + *pruneIndex);
621  (*stackNode)->levenshteinColumns_.erase((*stackNode)->levenshteinColumns_.begin() + *pruneIndex);
622  if ((*stackNode)->isFinal()) {
623  (*stackNode)->levenshteinScoresOverestimate_.erase((*stackNode)->levenshteinScoresOverestimate_.begin() + *pruneIndex);
624  }
625  }
626  }
627 
628  std::vector<u32>::reverse_iterator pruneIndex = pruneIndices.rbegin();
629  for (; pruneIndex != pruneIndices.rend(); ++pruneIndex) {
630  if ((*pruneIndex) < minimalIncompleteNode_->index_) {
631  --minimalIncompleteNode_->index_;
632  }
633  stack_->erase(stack_->begin() + (*pruneIndex));
634  }
635  stackEntries_.at(newLength_).resize(maxStackSize_);
636  if (addIncomplete) {
637  stackEntries_.at(newLength_).push_back(minimalIncompleteNode_);
638  }
639  }
640 }

642 bool MinimumBayesRiskAStarSearch::stackIsOptimal() {
643  if (minimalNode_->isFinal()) {
644  if (f32(minimalNode_->totalProbability_) < -std::log(0.5)) {
645  clog() << Core::XmlFull("stack-optimal", "0");
646  return true;

653  clog() << Core::XmlFull("stack-optimal", "2");
654  return true;
655  }

657  clog() << Core::XmlFull("stack-optimal", "3");
658  return true;
659  }
660  }

664 void MinimumBayesRiskAStarSearch::performSearch(Fsa::ConstAutomatonRef fsa) {
665  fsa_ = fsa;
666  evaluationSpaceSize_ = Fsa::countPaths(fsa_);
667  summationSpaceSize_ = evaluationSpaceSize_;
668 
669  LogWeights.at(0) = semiring_->zero();
670  for (u32 n = 1; n < MinimumBayesRiskAStarSearch::MaxLogWeights; ++n) {
671  LogWeights.at(n) = Fsa::Weight(-f32(std::log(f32(n))));
672  }
673 
674  exactEstimate_ = paramExactEstimate(config);
675  backwardPotentials_ = Fsa::sssp(Fsa::transpose(fsa_));
676  inverseBackwardPotentialsNormalizationConstant_ = Fsa::LogSemiring->invert(backwardPotentials_.at(fsa_->initialStateId()));
677  distances_ = getDistances(Fsa::transpose(fsa_));
678  for (u32 n = 0; n < distances_.size(); ++n) {
679  --distances_.at(n);
680  }
681  u32 longestDistance = distances_.at(fsa_->initialStateId()) - 1;
682  fsa_ = Fsa::changeSemiring(fsa_, Fsa::LogSemiring);
683 
684  f32 threshold = initialPruningThreshold_;
685 
686  evaluationSpaceSize_ = u32(Fsa::countPaths(fsa_));
687  while (threshold > 0 && evaluationSpaceSize_ > maximumNumberHypotheses_) {
688  threshold *= thresholdFactor_;
689  fsa_ = Fsa::prunePosterior(fsa_, Fsa::Weight(threshold));
690  evaluationSpaceSize_ = u32(Fsa::countPaths(fsa_));
691  }
692  summationSpaceSize_ = evaluationSpaceSize_;
693 
694  SearchNodeRef initialSearchNode(new SearchNode(longestDistance, fsa_->initialStateId()));
695  if (!fsa_->getState(fsa_->initialStateId())->hasArcs()) {
696  initialSearchNode->setInexplorable();
697  }
698  if (fsa_->getState(fsa_->initialStateId())->isFinal()) {
699  initialSearchNode->setFinal();
700  }
701 
702  stack_ = new std::vector<SearchNodeRef>();
703  stack_->push_back(initialSearchNode);
704  minimalNode_ = (*stack_->begin());
705  minimalIncompleteNode_ = (*stack_->begin());
706 
707  while (!stackIsOptimal()) {
708  expand();
709  }
710 
711  clog() << Core::XmlFull("mbr-risk-underestimate", exp(-f32(minimalNode_->estimate_)));
712  clog() << Core::XmlFull("mbr-risk-overestimate", exp(-f32(minimalNode_->overestimate_)));
713  clog() << Core::XmlFull("mbr-probability", exp(-f32(minimalNode_->totalProbability_)));
714  clog() << Core::XmlFull("overestimates", numberOverestimates_);
715 
716  // this has to changed to all paths in the original Fsa belonging to the best hypothesis.
717  // therefore, either the interface of the search node has to be changed or the automaton
718  // has to be created by a contouring algorithm.
719  bestAutomaton_ = createLinearAutomatonFromVector(
720  minimalNode_->hypothesis_,
721  Fsa::Weight(minimalNode_->estimate_),
722  fsa_->getInputAlphabet(),
723  fsa_->getOutputAlphabet(),
724  fsa_->semiring());
725 
726  delete stack_;
727 }
