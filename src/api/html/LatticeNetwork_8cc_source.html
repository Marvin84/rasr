52 bool LatticeNetwork::loadLexicon(const std::string& file) {
53  delete l_;
54  LexicalFstFactory::Options options;
55  // options.grammarType = AbstractGrammarFst::TypeVector;
56  options.accumulatorType = LexicalFstFactory::DefaultAccumulator;
57  options.lookAhead = LabelLookAhead;
58  log("loading lexicon fst: %s", file.c_str());
59  LexicalFstFactory factory(select("lexicon-fst"));
60  if (paramRemoveLexiconWeights(config)) {
61  FstLib::StdVectorFst* base = FstLib::StdVectorFst::Read(file);
62  log("removing lexicon weights");
63  FstLib::ArcMap(base, FstLib::RmWeightMapper<OpenFst::Arc>());
64  l_ = factory.convert(base, options, 0);
65  delete base;
66  }
67  else {
68  l_ = factory.load(file, options, 0);
69  }
70  return l_;
71 }

93 void LatticeNetwork::createNetwork(const FstLib::StdFst& lmLattice) {
94  if (l_) {
95  GrammarFst g(OpenFst::VectorFst(FstLib::StdProjectFst(lmLattice, FstLib::PROJECT_OUTPUT)));
96  l_->relabel(&g);

107 bool LatticeNetwork::loadLattice(const std::string& name) {
108  const Lattice* lattice = archive_->read(name, false);
109  if (!lattice) {
110  error("cannot load lattice '%s'", name.c_str());
111  return false;
112  }
113  bool flagSymbols = FLAGS_fst_compat_symbols;
114  FLAGS_fst_compat_symbols = false;
115  FstLib::Fst<OpenFst::Arc>* lmLattice = getLmLattice(*lattice);
116  createNetwork(*lmLattice);
117  delete lmLattice;
118  delete lattice;
119  if (g_ || l_)
120  FstLib::Connect(f_);
121  FstLib::ArcSort(f_, FstLib::StdILabelCompare());
122  FLAGS_fst_compat_symbols = flagSymbols;
123  return true;
