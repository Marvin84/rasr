33  std::vector<std::string> numberedSource = numberTokensVector(source_);
34  StaticAlphabet* ia = new StaticAlphabet();
35  for (std::vector<std::string>::const_iterator i = numberedSource.begin(); i != numberedSource.end(); ++i) {
36  inputSentence_.push_back(ia->addSymbol(*i));
37  }
38  inputAlphabet_ = ConstAlphabetRef(ia);
39 
40  std::vector<std::string> numberedTarget = numberTokensVector(target_);
41  StaticAlphabet* oa = new StaticAlphabet();
42  for (std::vector<std::string>::const_iterator i = numberedTarget.begin(); i != numberedTarget.end(); ++i) {
43  outputSentence_.push_back(oa->addSymbol(*i));
44  }
45  outputAlphabet_ = ConstAlphabetRef(oa);
46 
47  // generate state (there is only one)
48  // it contains all the arcs of translations from source words to target words
49  // as permitted by the lexicon and additionally all the translations to the
50  // empty word from any source word
51 
52  // regular words
53  for (u32 i = 0; i < numberedTarget.size(); ++i) {
54  for (u32 j = 0; j < numberedSource.size(); ++j) {
55  std::vector<std::string> x;
56  x.push_back(source_[j]);
57  x.push_back(target_[i]);
58  state_->newArc(state_->id(), Fsa::Weight(lexicon_->getProb(x)), inputSentence_[j], outputSentence_[i]);
59  }

89 class ReorderAutomaton : public SlaveAutomaton {
90 private:
91  StateRef initial_;
92  StateId nextFreeStateId_;
93  std::map<StateId, StateRef> statemap_;
94 
95 public:
96  ReorderAutomaton(ConstAutomatonRef f)
97  : SlaveAutomaton(f),
98  initial_(new State()),
99  nextFreeStateId_(1) {
100  this->setProperties(PropertyAcyclic || PropertyLinear, PropertyAcyclic || PropertyLinear);
101  statemap_[0] = initial_;
102 
103  ConstStateRef initial = fsa_->getState(initialStateId());
104  // iterate over all hypotheses to generate nbest reordering graphs
105  for (State::const_iterator i = initial->begin(); i != initial->end(); ++i) {
106  ConstStateRef ts = fsa_->getState(i->target());
107  std::map<LabelId, Vector<LabelId>> sentence;
108  // traverse this hypotheses and create mapping in target word order
109  while (!(ts->isFinal())) {
110  LabelId out(ts->begin()->output());
111  LabelId in(ts->begin()->input());
112 
113  if (in != Epsilon && out != Epsilon) {
114  if (sentence.find(out) == sentence.end()) {
115  sentence[out] = Vector<LabelId>(1, in);
116  }
117  else {
118  sentence[out].push_back(in);
119  }
120  }
121  ts = fsa_->getState(ts->begin()->target());
122  }
123  // create permutation graph in target word order
124  StateRef new_state = initial_;
125  for (std::map<LabelId, Vector<LabelId>>::const_iterator i = sentence.begin(); i != sentence.end(); ++i) {
126  for (Vector<LabelId>::const_iterator j = i->second.begin(); j != i->second.end(); ++j) {
127  LabelId in = *j;
128  bool foundArc(false);
129  for (State::const_iterator k = new_state->begin(); k != new_state->end(); ++k) {
130  if (k->input() == in) {
131  new_state = statemap_[k->target()];
132  foundArc = true;
133  break;
134  }
135  }
136  // this part after the for loop is only reached, when no arc exists in the current state
137  // of the permutation graph that corresponds to the current input symbol
138  //
139  // create new arc that corresponds to the current input symbol and traverse
140  if (!foundArc) {
141  new_state->newArc(nextFreeStateId_, this->semiring()->one(), in);
142  statemap_[nextFreeStateId_] = StateRef(new State(nextFreeStateId_));
143  new_state = statemap_[nextFreeStateId_];
144  ++nextFreeStateId_;
145  }
146  }
147  }
148  new_state->setFinal(this->semiring()->one());
149  }
150  }
151  virtual ConstStateRef getState(StateId s) const {
152  return Fsa::ConstStateRef(statemap_.find(s)->second);
153  }
154  virtual std::string describe() const {
155  return std::string("reorderAutomaton(" + fsa_->describe() + ")");
156  }
157  virtual StateId initialStateId() const {
158  return 0;
159  }
160  virtual Type type() const {
161  return TypeAcceptor;
162  }
163  virtual ConstAlphabetRef getOutputAlphabet() const {
164  return fsa_->getInputAlphabet();

189  AlignLinearNbestAutomaton(ConstAutomatonRef sourceAutomaton, ConstAutomatonRef alignAutomaton)
190  : SlaveAutomaton(sourceAutomaton) {
191  for (State::const_iterator i = sourceAutomaton->getState(sourceAutomaton->initialStateId())->begin();
192  i != sourceAutomaton->getState(sourceAutomaton->initialStateId())->end(); ++i) {
193  ConstAutomatonRef bestTrace = best(composeMatching(partial(sourceAutomaton, i->target()), alignAutomaton));
194  ConstStateRef s = bestTrace->getState(bestTrace->initialStateId());
195  do {
196  State* sp = new State(nextFreeStateId_, s->tags(), s->weight());
197  statemap_[nextFreeStateId_] = StateRef(sp);
198  ++nextFreeStateId_;
199  sp->newArc(nextFreeStateId_,
200  s->begin()->weight(),
201  s->begin()->input(),
202  s->begin()->output());
203  s = bestTrace->getState(s->begin()->target());
204  } while (!s->isFinal());
205  State* sp = new State(nextFreeStateId_, s->tags(), s->weight());
206  sp->setFinal(this->semiring()->one());
207  statemap_[nextFreeStateId_] = StateRef(sp);
208  ++nextFreeStateId_;
209  }
210  }
