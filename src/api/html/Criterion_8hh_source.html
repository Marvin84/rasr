28 /*
29 The NnTrainer tool will use the Buffered*Feature(Extractor|Processor),
30 which extracts features for a mini-batch or segment,
31 and also might provide the alignment for a segment.
32 
33 This calls the NeuralNetworkTrainer, which has an instance of Criterion.
34 
35 That could be to train the network (FeedForwardTrainer),
36 but also to calculate mean-and-variance (MeanAndVarianceTrainer),
37 or to evaluate the frame-error-eval (FrameErrorEvaluator).
38 
39 The network training must forward the network for some input,
40 and then calculate the error signal + objective function via criterion,
41 and then backprop + collect gradients + estimate new network.
42 
43 Some criterions must know the alignment (current implementation of CE, etc.),
44 some must operate on full segments (CTC).
45 In some cases (e.g. CTC), it needs the segment transcription.
46 
47 Mean-and-variance: custom trainer, no criterion, weights (via alignment).
48 Frame-error-eval: custom trainer, aligned criterion.
49 Aligned-trainer: aligned criterion.
50 Segment-basic-trainer: no lattice, no alignment, segment, generic criterion.
51 Segment-lattice-trainer: -> MMI/ME/MPE, see SegmentwiseNnTrainer.
52 
53 */

93  // Interface:
94  // First call one (and only one) of the input* functions, then you can read
95  // out the objective function value and error signal via
96  // the get* functions.

98  /** Override for an unsupervised criterion */
99  virtual void input(NnMatrix& nnOutput, NnVector* weights = NULL) {
100  nnOutput_ = &nnOutput;
101  weights_ = weights;
102  needRecalc_objectiveFunction_ = true;
103  needRecalc_errorSignal_ = true;
104  }

106  /** Override for an alignment-based criterion (usually frame-wise, e.g. Cross-Entropy) */
107  virtual void inputAlignment(Math::CudaVector<u32>& alignment, NnMatrix& nnOutput, NnVector* weights = NULL) {
108  alignment_ = &alignment;
109  input(nnOutput, weights);
110  }

112  // Note: Some inputWithReference() or so would also make sense here, which would be
113  // a generic variant of inputAlignment().
114 
115  /** Override for a segment (without lattice) criterion (e.g. CTC) */

117  segment_ = &segment;
118  input(nnOutput, weights);
119  }
120 
121  /** Override to calculate the objective function. Some input* function was called before. */
122  virtual void getObjectiveFunction(FloatT& value) {
123  value = 0;
124  }
125 
126  /** Override to calculate the error signal. Some input* function was called before.
127  * This is thus the error signal of the NN output.
128  */
129  virtual void getErrorSignal(NnMatrix& errorSignal) {
130  errorSignal.setToZero();
131  }
132 
133  /** Override to calculate the error signal with natural pairing of the last layer activation function.
134  * This is thus the error signal of the linear part of the last layer.
135  * This function is usually used in frame-wise training with a fixed alignment,
136  * via BufferedAlignedFeatureProcessor.
137  */
138  virtual void getErrorSignal_naturalPairing(NnMatrix& errorSignal, NeuralNetworkLayer<FloatT>& lastLayer) {
139  // This default implementation just uses the backprop implementation of the layer
140  // and the standard getErrorSignal().
141  // We don't do the weighting here (if we have weights) because we expect that
142  // getErrorSignal() does it already, and the backprop is multiplicative, thus
143  // the weighting is just passed down to us.
144  verify(nnOutput_);
145  NnMatrix imtmErrorSignal(errorSignal.nRows(), errorSignal.nColumns());
146  imtmErrorSignal.initComputation(false);
147  imtmErrorSignal.setToZero();
148  getErrorSignal(imtmErrorSignal);
149  lastLayer.backpropagateActivations(imtmErrorSignal, errorSignal, *nnOutput_);
150  }
151 
152  /** Overrwrite if you want to signal any trainer to ignore this input. */
153  virtual bool discardCurrentInput() {
154  return false;
155  }
156 
157  virtual void reset() {
158  nnOutput_ = NULL;
159  weights_ = NULL;
160  alignment_ = NULL;
161  segment_ = NULL;
162  needRecalc_objectiveFunction_ = true;
163  needRecalc_errorSignal_ = true;
164  }
165 
166  // Calls the input* function again, which was called last time,
167  // with all the same parameters, except a new nnOutput.
168  virtual void reinputWithNewNnOutput(NnMatrix& nnOutput) {
169  needRecalc_objectiveFunction_ = true;
170  needRecalc_errorSignal_ = true;
171  if (alignment_) {
172  segment_ = NULL;
173  inputAlignment(*alignment_, nnOutput, weights_);
174  }
175  else if (segment_) {
176  alignment_ = NULL;
177  inputSpeechSegment(*segment_, nnOutput, weights_);
178  }
179  else {
180  alignment_ = NULL;
181  segment_ = NULL;
182  input(nnOutput, weights_);
183  }
184  }
185 
186  /** Some criterions will calculate some kind of pseudo targets
187  * and have the gradient -\hat{y} / y w.r.t. y, and when y = softmax(a),
188  * they have the gradient y - \hat{y} w.r.t. a.
189  * In that case, we call \hat{y} the pseudo targets.
190  * Examples: Cross-Entropy and CTC.
191  * Can return NULL. Otherwise the pointer will be valid until the next input.
192  */
193  virtual NnMatrix* getPseudoTargets() {
194  return NULL;
195  }
196 
197  virtual Type getType() const {
198  return criterionType_;
199  }
