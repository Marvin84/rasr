99 /*****************************************************************************/
100 History FsaLm::extendedHistory(const History& h, Token w) const
101 /*****************************************************************************/
102 {
103  Fsa::ConstStateRef sp(descriptor<Self>(h));
104  if (sp == invalidHistory) {
105  invalidHistory->acquireReference();
106  return history(invalidHistory.get());
107  }
108  while (sp) {
109  Fsa::Arc tmp;
110  tmp.input_ = w->id();
111  Fsa::State::const_iterator a = sp->lower_bound(tmp, Fsa::byInput());
112  if ((a == sp->end()) || (a->input() != w->id())) {
113  a = sp->begin();
114  if ((a == sp->end()) || (a->input() != Fsa::Epsilon)) {
115  invalidHistory->acquireReference();
116  return history(invalidHistory.get());
117  }
118  sp = fsa_->getState(a->target());
119  }
120  else {
121  Fsa::ConstStateRef ts = fsa_->getState(a->target());
122  ts->acquireReference();
123  return history(ts.get());
124  }
125  }
126  return h;
127 }

160 /*****************************************************************************/
161 Lm::Score FsaLm::sentenceEndScore(const History& h) const
162 /*****************************************************************************/
163 {
164  Fsa::ConstStateRef sp(descriptor<Self>(h));
165  if (sp == invalidHistory) {
166  return infinityScore();
167  }
168  Score score = 0.0;
169  while (sp) {
170  if (sp->isFinal())
171  return score + Score(sp->weight_);
172  Fsa::State::const_iterator a = sp->begin();
173  if ((a == sp->end()) || (a->input() != Fsa::Epsilon)) {
174  return infinityScore();
175  }

177  score += Score(a->weight());
178  }
