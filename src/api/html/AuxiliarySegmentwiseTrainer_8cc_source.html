37 f32 SigmoidFunction::argument(const Fsa::Weight& totalInvNum, const Fsa::Weight& totalInvDen) const {
38  f32 x = Core::Type<f32>::max;
39  if ((f32(totalInvDen) - f32(totalInvNum)) > Core::Type<f64>::delta) {
40  x = f32(totalInvNum);
41  x -= (f32(totalInvDen) + log1p(-exp(f32(totalInvNum) - f32(totalInvDen))));
42  }
43  return x;
44 }

51 Fsa::Weight SigmoidFunction::df(const Fsa::Weight& totalInvNum, const Fsa::Weight& totalInvDen) const {
52  f32 x = argument(totalInvNum, totalInvDen);

65 private:

69 public:
70  DenominatorWeightsAutomaton(PosteriorFsa num, PosteriorFsa den)
71  : Precursor(Fsa::extend(den.fsa, den.fsa->semiring()->invert(den.totalInv))),
72  fsaNum_(Fsa::extend(num.fsa, den.fsa->semiring()->invert(num.totalInv))) {
73  totalInv_ = semiring()->extend(den.totalInv, Fsa::Weight(log1p(-exp(f32(num.totalInv) - f32(den.totalInv)))));
74  }
75  virtual ~DenominatorWeightsAutomaton() {}
76 
77  virtual std::string describe() const {
78  return "denominator-weights(" + fsa_->describe() + ")";
79  }
80  virtual void modifyState(Fsa::State* sp) const {
81  Fsa::ConstStateRef spNum = fsaNum_->getState(sp->id());
82  Fsa::State::const_iterator aNum = spNum->begin();
83  for (Fsa::State::iterator a = sp->begin(); a != sp->end(); ++a, ++aNum) {
84  /*
85  * a->weight() contains the unnormalized denominator arc posterior probability.
86  */
87  if (semiring()->compare(aNum->weight(), semiring()->zero()) < 0) {
88  a->weight_ =
89  semiring()->extend(
90  a->weight(),
91  Fsa::Weight(-log1p(-exp(f32(a->weight()) - f32(aNum->weight())))));
92  }
93  a->weight_ = semiring()->extend(a->weight(), totalInv_);
94  }
95  }
96 };
