27 void MixtureSetBuilder::build(MixtureSet& toBuild, ComponentIndex dimension) {
28  toBuild.clear();
29  toBuild.setDimension(dimension);
30 
31  for (MixtureIndex mix = 0; mix < nMixtures_; ++mix) {
32  Mixture* mixture = new Mixture;
33  for (size_t dns = 0; dns < nDensitiesPerMixture_; ++dns) {
34  if (densityIndex(mix, dns) >= toBuild.nDensities()) {
35  if (meanIndex(mix, dns) >= toBuild.nMeans())
36  toBuild.addMean(meanIndex(mix, dns), new Mean(dimension));
37  if (covarianceIndex(mix, dns) >= toBuild.nCovariances())
38  toBuild.addCovariance(covarianceIndex(mix, dns), new DiagonalCovariance(dimension));
39 
40  toBuild.addDensity(densityIndex(mix, dns), new GaussDensity(meanIndex(mix, dns), covarianceIndex(mix, dns)));
41  }
42  mixture->addDensity(densityIndex(mix, dns));
43  }
44  toBuild.addMixture(mix, mixture);
