20 bool NthOrderFeatures::SecondsSelection::load(
21  const std::string& filename) {
22  verify(!filename.empty());
23  Core::CompressedInputStream cis(filename);
24  std::string line;
25  u32 prevI = 0;
26  u32 prevJ = 0;
27  nSeconds_ = 0;
28  while (cis) {
29  std::getline(cis, line);
30  Core::stripWhitespace(line);
31  if (!line.empty() and (*line.c_str() != '#')) {
32  std::vector<std::string> fields = Core::split(line, " ");
33  require(fields.size() == 2);
34  u32 currI, currJ;
35  Core::strconv(fields[0], currI);
36  Core::strconv(fields[1], currJ);
37  require(currI <= currJ);
38  require((prevI < currI) or ((prevI == currI) and (prevJ <= currJ)));
39  selection_[currI][currJ] = nSeconds_;
40  prevI = currI;
41  prevJ = currJ;
42  ++nSeconds_;
43  }
44  }
45  return true;
46 }

53 void NthOrderFeatures::setOutputSize(size_t inputSize) {
54  outputSize_ = 0;
55  if (order_ & zeroth) {
56  outputSize_ += 1;

58  if (order_ & first) {
59  outputSize_ += inputSize;
60  }
61  if (order_ & diagonalSecond) {
62  outputSize_ += inputSize;
63  }
64  if (order_ & second) {
65  outputSize_ += ((inputSize + 1) * inputSize) / 2;
66  }
67  if (order_ & selectedSecond) {

71  outputSize_ += inputSize;

77  outputSize_ += inputSize * ((inputSize + 1) * inputSize / 2);

99 void NthOrderFeatures::apply(const std::vector<f32>& in, std::vector<f32>& out) {
100  require(out.size() == 0);
101  if (order_ & first) {
102  out.resize(out.size() + in.size());
103  std::copy(in.rbegin(), in.rend(), out.rbegin());
104  }
105  if (order_ & diagonalSecond) {
106  out.resize(out.size() + in.size());
107  std::transform(in.rbegin(), in.rend(), out.rbegin(), power<f32>(2));
108  }
109  if (order_ & second) {
110  for (size_t i = 0; i < in.size(); ++i) {
111  for (size_t j = i; j < in.size(); ++j) {
112  out.push_back(in[i] * in[j]);
113  }
114  }
115  }
116  if (order_ & selectedSecond) {
117  for (size_t i = 0; i < in.size(); ++i) {
118  if (secondsSelection_.setI(i)) {
119  for (size_t j = i; j < in.size(); ++j) {
120  if (secondsSelection_.setJ(j)) {
121  out.push_back(in[i] * in[j]);
122  }
123  }
124  }
125  }
126  }
127  if (order_ & diagonalThird) {
128  out.resize(out.size() + in.size());
129  std::transform(in.rbegin(), in.rend(), out.rbegin(), power<f32>(3));
130  }
131  if (order_ & third) {
132  for (size_t i = 0; i < in.size(); ++i) {
133  for (size_t j = i; j < in.size(); ++j) {
134  for (size_t k = j; k < in.size(); ++k) {
135  out.push_back(in[i] * in[j] * in[k]);
136  }
137  }
138  }
139  }
140  if (order_ & asymmetricThird) {
141  for (size_t i = 0; i < in.size(); ++i) {
142  for (size_t j = 0; j < in.size(); ++j) {
143  for (size_t k = j; k < in.size(); ++k) {
144  out.push_back(in[i] * in[j] * in[k]);
145  }
146  }
147  }
148  }
149  if (order_ & diagonalFourth) {
150  out.resize(out.size() + in.size());
151  std::transform(in.rbegin(), in.rend(), out.rbegin(), power<f32>(4));
152  }
153  if (order_ & diagonalFifth) {
154  out.resize(out.size() + in.size());
155  std::transform(in.rbegin(), in.rend(), out.rbegin(), power<f32>(5));
156  }
157  if (order_ & diagonalSixth) {
158  out.resize(out.size() + in.size());
159  std::transform(in.rbegin(), in.rend(), out.rbegin(), power<f32>(6));
160  }
161  if (order_ & diagonalSeventh) {
162  out.resize(out.size() + in.size());
163  std::transform(in.rbegin(), in.rend(), out.rbegin(), power<f32>(7));
164  }
165  if (order_ & diagonalEighth) {
166  out.resize(out.size() + in.size());
167  std::transform(in.rbegin(), in.rend(), out.rbegin(), power<f32>(8));
168  }
169  if (order_ & diagonalNinth) {
170  out.resize(out.size() + in.size());
171  std::transform(in.rbegin(), in.rend(), out.rbegin(), power<f32>(9));
172  }
173  if (order_ & zeroth) {
174  out.push_back(1);
175  }
176  verify(out.size() == outputSize());
177 }

218 bool NthOrderFeaturesNode::setParameter(const std::string& name, const std::string& value) {
219  if (paramOrderType.match(name)) {
220  int order = (int)NthOrderFeatures::none;
221  const std::vector<std::string> fields = Core::split(value, "-and-");
222  for (std::vector<std::string>::const_iterator it = fields.begin(); it != fields.end(); ++it) {
223  order |= choiceOrderType[*it];
224  }
225  if (!nthOrder_) {
226  nthOrder_ = createNthOrderFeatures();
227  nthOrder_->setOrder((NthOrderFeatures::OrderType)order);
228  nthOrder_->loadSecondsSelection(paramSecondsSelectionFile(config));
229  }
230  }
231  else {
232  return false;
233  }
234  return true;
235 }
