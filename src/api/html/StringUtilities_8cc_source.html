24 bool Core::isWhitespaceNormalized(const std::string& s, char trailingBlank) {
25  std::string::size_type i, b;
26 
27  b = 1;
28  for (i = 0; i < s.length(); ++i) {
29  if (s[i] == utf8::blank) {
30  if (++b > 1)
31  return false;
32  }
33  else {
34  b = 0;
35  }
36  }
37  switch (trailingBlank) {
38  case requireTrailingBlank: return (b == 1);
39  case prohibitTrailingBlank: return (b == 0) || (s.length() == 0);
40  case tolerateTrailingBlank: return true;
41  default: require(false);
42  }
43 }

45 void Core::normalizeWhitespace(std::string& s) {
46  std::string::size_type i, j;
47 
48  // remove leading whitespace
49  j = s.find_first_not_of(utf8::whitespace);
50  s.erase(0, j);
51 
52  // replace consecutive whitespace by single blanks
53  for (i = 0;; ++i) {
54  i = s.find_first_of(utf8::whitespace, i);
55  if (i == std::string::npos)
56  break;
57  j = s.find_first_not_of(utf8::whitespace, i);
58  if (j == std::string::npos)
59  j = s.length();
60  s.replace(i, j - i, 1, utf8::blank);
61  }
62 }

74 void Core::stripWhitespace(std::string& s) {
75  std::string::size_type i;
76 
77  // remove leading whitespace
78  i = s.find_first_not_of(utf8::whitespace);
79  if (i == std::string::npos)
80  i = s.length();
81  s.erase(0, i);
82 
83  // remove trailing whitespace
84  i = s.find_last_not_of(utf8::whitespace);
85  if (i != std::string::npos)
86  s.erase(i + 1);
87 
88  ensure(s.empty() || s.find_first_of(utf8::whitespace) != 0);
89  ensure(s.empty() || s.find_last_of(utf8::whitespace) != s.size() - 1);
90 }

106 std::string Core::form(const char* format, ...) {
107  va_list ap;
108  static size_t buf_size = 0;
109  static char* buf = 0;
110  int n;
111 
112  for (;;) {
113  va_start(ap, format);
114  n = vsnprintf(buf, buf_size, format, ap);
115  va_end(ap);
116 
117  if (n < 0) {
118  // This implementation of vsnprintf returns -1 if the
119  // output was truncated. We use the simple heuristic of
120  // doubling the buffer size
121  n = (buf_size > 64) ? 2 * buf_size : 64;
122  }
123  else {
124  // Newer implementations (>= glibc 2.1) return the number
125  // of characters needed.
126  if (n + 1 <= (int)buf_size)
127  break; // +1 for terminating null byte
128  }
129 
130  if (buf)
131  delete[] buf;
132  buf = new char[buf_size = n + 1];
133  }
134 
135  return std::string(buf, n);
136 }

167 std::vector<std::string> Core::split(const std::string& string, const std::string& separator) {
168  std::vector<std::string> result;
169  if (string.empty())
170  return result;
171 
172  std::string::size_type start = 0;
173  std::string::size_type end = 0;
174  while (end != std::string::npos) {
175  end = string.find(separator, start);
176  if (end == std::string::npos) {
177  result.push_back(string.substr(start));
178  stripWhitespace(result.back());
179  }
180  else {
181  if (end > start) {
182  result.push_back(string.substr(start, end - start));
183  stripWhitespace(result.back());
184  }
185  start = end + separator.size();
186  }
187  }
188  return result;
189 }
