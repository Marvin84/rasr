61 bool AnalyticFunctionFactory::accept(const std::string& s, const std::string& declaration, std::string::size_type& position) const {
62  acceptWhitespaces(declaration, position);
63  if (position != std::string::npos) {
64  if (declaration.find(s, position) == position) {
65  position += s.size();
66  require(position <= declaration.size());
67  if (position == declaration.size())
68  position = std::string::npos;
69  return true;
70  }
71  }
72  return false;
73 }

75 bool AnalyticFunctionFactory::accept(UnaryAnalyticFunction::Argument& a, const std::string& declaration, std::string::size_type& position) const {
76  acceptWhitespaces(declaration, position);
77  if (position != std::string::npos) {
78  std::string s = declaration.substr(position);
79  char* endptr;
80  UnaryAnalyticFunction::Argument newA = strtod(s.c_str(), &endptr);
81  if (endptr == 0) {
82  position = std::string::npos;
83  a = newA;
84  return true;
85  }
86  if (s.c_str() != endptr) {
87  position += (endptr - s.c_str());
88  a = newA;
89  return true;
90  }
91  }
92  return false;
93 }

120 AnalyticFunctionFactory::AnalyticFunctionRef AnalyticFunctionFactory::create(const std::string& declaration) const {
121  AnalyticFunctionRef result;
122  if (!declaration.empty()) {
123  std::string::size_type position = 0;
124  if (result = create(declaration, position)) {
125  acceptWhitespaces(declaration, position);
126  if (position == std::string::npos)
127  return result;
128  }
129  if (position != std::string::npos) {
130  error("Failed to parse function declaration from position %zd.\nUnparsed ending '%s'.",
131  position, declaration.substr(position).c_str());
132  }
133  else {
134  error("End of string reached during parsing '%s'.", declaration.c_str());
135  }
136  }
137  else {
138  error("Function declaration is empty.");
139  }
140  return result;
141 }

143 AnalyticFunctionFactory::AnalyticFunctionRef AnalyticFunctionFactory::create(const std::string& declaration,
144  std::string::size_type& position) const {
145  AnalyticFunctionRef result;
146  std::string::size_type maximalPosition = 0;
147  for (size_t i = 0; i < parserFunctions_.size(); ++i) {
148  std::string::size_type newPosition = position;
149  if (result = (this->*parserFunctions_[i])(declaration, newPosition)) {
150  position = newPosition;
151  return result;
152  }
153  else {
154  maximalPosition = std::max(maximalPosition, newPosition);
155  }
156  }
157  position = maximalPosition;
158  return result;
159 }

161 AnalyticFunctionFactory::AnalyticFunctionRef AnalyticFunctionFactory::parseNesting(const std::string& declaration,
162  std::string::size_type& position) const {
163  UnaryAnalyticFunctionRef result;
164  if (accept("nest(", declaration, position)) {
165  UnaryAnalyticFunctionRef g = castToUnaryFunction(create(declaration, position));
166  if (g) {
167  if (accept(",", declaration, position)) {
168  AnalyticFunctionFactory factory(*this);
169  factory.setMaximalArgument(g->value(maximalArgument_));
170 
171  UnaryAnalyticFunctionRef f = castToUnaryFunction(factory.create(declaration, position));
172  if (f) {
173  if (accept(")", declaration, position))
174  result = nest(f, g);
175  }
176  }
177  }
178  }
179  return result;
180 }

338 UnaryAnalyticFunctionRef AnalyticFunctionFactory::createMelWarpingFunction(bool discretizeArgument) const {
339  switch (domainType_) {
340  case continuousDomain:
341  return nest(createScaling(2595.0), createMelWarpingCore());
342  case discreteDomain: {
343  // sampleRate_: sample rate of the frequency axis
344  UnaryAnalyticFunctionRef scaleInput = createScaling(1 / sampleRate_);
345  if (discretizeArgument)
346  scaleInput = nest(createFloorFloat(), scaleInput); // creates result uninvertible
347  return nest(createScaling(2595.0 * sampleRate_), nest(createMelWarpingCore(), scaleInput));
348  }
349  case normalizedOmegaDomain:
350  // sampleRate_: sample rate in time domain
351  return nest(createScaling(2595.0 * 2 * M_PI / sampleRate_),
352  nest(createMelWarpingCore(), createScaling(sampleRate_ / (2 * M_PI))));
353  default:
354  error("Input type not supported by mel warping.");
355  return UnaryAnalyticFunctionRef();
356  }
357 }

394 UnaryAnalyticFunctionRef AnalyticFunctionFactory::createTwoPieceLinearFunction(UnaryAnalyticFunction::Argument warpingFactor,
395  UnaryAnalyticFunction::Argument limit) const {
396  require(0 < limit && limit < 1);
397  if (warpingFactor <= 1) {
398  PiecewiseLinearFunction* result = new PiecewiseLinearFunction;
399  result->add(limit * maximalArgument_, warpingFactor);
400  result->normalize(maximalArgument_);
401  return UnaryAnalyticFunctionRef(result);
402  }
403  else {
404  PiecewiseLinearFunction inverse;
405  inverse.add(limit * maximalArgument_, 1 / warpingFactor);
406  inverse.normalize(maximalArgument_);
407  return inverse.invert();

457 UnaryAnalyticFunctionRef AnalyticFunctionFactory::createThreePieceAffineFunction(UnaryAnalyticFunction::Argument warpingFactor,
458  UnaryAnalyticFunction::Argument aShift,
459  UnaryAnalyticFunction::Argument lowerLimit,
460  UnaryAnalyticFunction::Argument upperLimit) const {
461  require(0 < lowerLimit && lowerLimit < upperLimit && upperLimit < 1 && aShift >= 0);
462  if (warpingFactor <= 1) {
463  PiecewiseLinearFunction* result = new PiecewiseLinearFunction;
464  result->add(lowerLimit * maximalArgument_,
465  (warpingFactor * lowerLimit * maximalArgument_ +
466  (aShift * (warpingFactor - 1))) /
467  (lowerLimit * maximalArgument_));
468  result->add(upperLimit * maximalArgument_, warpingFactor);
469  result->normalize(maximalArgument_);
470  return UnaryAnalyticFunctionRef(result);
471  }
472  else {
473  PiecewiseLinearFunction inverse;
474  inverse.add(lowerLimit * maximalArgument_,
475  ((1 / warpingFactor) * lowerLimit * maximalArgument_ + aShift *
476  ((1 / warpingFactor) - 1)) /
477  (lowerLimit * maximalArgument_));
478  inverse.add(upperLimit * maximalArgument_, 1 / warpingFactor);
479  inverse.normalize(maximalArgument_);
480  return inverse.invert();
481  }
482 }

484 AnalyticFunctionFactory::AnalyticFunctionRef AnalyticFunctionFactory::parseThreePieceAffineFunction(const std::string& declaration,
485  std::string::size_type& position) const {
486  UnaryAnalyticFunctionRef result;
487  if (accept("affine-3(", declaration, position)) {
488  UnaryAnalyticFunction::Argument warpingFactor;
489  if (accept(warpingFactor, declaration, position)) {
490  if (0.7 < warpingFactor && warpingFactor < 1.3) {
491  if (accept(",", declaration, position)) {
492  UnaryAnalyticFunction::Argument aShift;
493  if (accept(aShift, declaration, position)) {
494  if (0 <= aShift && aShift <= 1500) {
495  if (accept(",", declaration, position)) {
496  UnaryAnalyticFunction::Argument lowerLimit;
497  if (accept(lowerLimit, declaration, position)) {
498  if (0.0350 < lowerLimit && lowerLimit < 1) {
499  if (accept(",", declaration, position)) {
500  UnaryAnalyticFunction::Argument upperLimit;
501  if (accept(upperLimit, declaration, position)) {
502  if (lowerLimit < upperLimit && upperLimit < 1) {
503  if (accept(")", declaration, position)) {
504  result = createThreePieceAffineFunction(
505  warpingFactor, aShift, lowerLimit, upperLimit);
506  }
507  }
508  else {
509  error("Upper-Limit has to be greater than 'lower-limit' "
510  "and less than 1.");
511  }
512  }
513  }
514  }
515  else {
516  error("Lower-Limit has to lie in the interval (0.0350, 1).");
517  }
518  }
519  }
520  }
521  else {
522  error("Reasonable a-shift are usually expected in the interval [0, 1500].");
523  }
524  }
525  }
526  }
527  else {
528  error("Warping factor has to lie in the interval (0.7, 1.3).");
