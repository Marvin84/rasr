23 /** VectorResizeNode resizes input vector to a given size
24  * If new size is larger than the current size, the vector is extended by initial-value.
25  *
26  * Size is given in continuous units.
27  */
28 extern const Core::ParameterFloat paramVectorResizeNewSize;
29 extern const Core::ParameterInt paramVectorResizeNewDiscreteSize;
30 extern const Core::ParameterFloat paramVectorResizeInitialValue;
31 extern const Core::ParameterBool paramVectorResizeChangeFront;
32 extern const Core::ParameterBool paramVectorResizeRelativeChange;

34 template<class T>
35 class VectorResizeNode : public SleeveNode {
36 public:
37  typedef SleeveNode Precursor;
38 
39 private:
40  s32 newSize_;
41  f64 continuousNewSize_;
42  u32 discreteNewSize_;
43  T initialValue_;
44  bool changeFront_;
45  bool relativeChange_;
46 
47 public:
48  static std::string filterName() {
49  return std::string("signal-vector-") + Core::Type<T>::name + "-resize";
50  }
51  VectorResizeNode(const Core::Configuration& c)
52  : Core::Component(c), Precursor(c), newSize_(0) {
53  continuousNewSize_ = paramVectorResizeNewSize(c);
54  discreteNewSize_ = paramVectorResizeNewDiscreteSize(c);
55  initialValue_ = paramVectorResizeInitialValue(c);
56  changeFront_ = paramVectorResizeChangeFront(c);
57  relativeChange_ = paramVectorResizeRelativeChange(c);
58  }
59  virtual ~VectorResizeNode() {}

61  virtual bool configure() {
62  Core::Ref<const Flow::Attributes> a = getInputAttributes(0);
63  if (!configureDatatype(a, Flow::Vector<T>::type()))
64  return false;
65  std::string sampleRateAttribute = a->get("sample-rate");
66  f64 sampleRate = sampleRateAttribute.empty() ? 1 : atof(sampleRateAttribute.c_str());
67  if (sampleRate == 0)
68  sampleRate = 1;
69  newSize_ = (discreteNewSize_ != 0) ? discreteNewSize_ : (u32)rint(continuousNewSize_ * sampleRate);
70  if (discreteNewSize_ != 0 && continuousNewSize_ != 0.0)
71  warning("Continuous units resize will be overwritten by discrete units");
72  return putOutputAttributes(0, a);
73  }
74 
75  virtual bool setParameter(const std::string& name, const std::string& value) {
76  if (paramVectorResizeNewSize.match(name))
77  continuousNewSize_ = paramVectorResizeNewSize(value);
78  else if (paramVectorResizeNewDiscreteSize.match(name))
79  discreteNewSize_ = paramVectorResizeNewDiscreteSize(value);
80  else if (paramVectorResizeInitialValue.match(name))
81  initialValue_ = paramVectorResizeInitialValue(value);
82  else if (paramVectorResizeChangeFront.match(name))
83  changeFront_ = paramVectorResizeChangeFront(value);
84  else if (paramVectorResizeRelativeChange.match(name))
85  relativeChange_ = paramVectorResizeRelativeChange(value);
86  else
87  return false;
88  return true;
89  }

91  virtual bool work(Flow::PortId p) {
92  Flow::DataPtr<Flow::Vector<T>> in;
93  if (!getData(0, in))
94  return putData(0, in.get());
95  in.makePrivate();
96  s32 difference = relativeChange_ ? newSize_ : (s32)newSize_ - (s32)in->size();
97  if (((s32)in->size() + difference) < 0) {
98  warning("Negative size set to zero");
99  difference = -(s32)in->size();
100  }
101  if (difference <= 0)
102  if (changeFront_)
103  in->erase(in->begin(), in->begin() - difference);
104  else
105  in->erase(in->end() + difference, in->end());
106  else if (changeFront_)
107  in->insert(in->begin(), difference, initialValue_);
108  else
109  in->insert(in->end(), difference, initialValue_);
110  return putData(0, in.get());
111  }
