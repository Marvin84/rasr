289 template<typename T>
290 void FastVector<T>::resize(u32 newSize, T value, bool allocOnly) {
291  u32 oldSize = nRows_;
292  nRows_ = newSize;
293  if (newSize == oldSize)
294  return;
295  else if (newSize == 0) { // free memory
296  if (elem_) {
297  delete[] elem_;
298  }
299  elem_ = 0;
300  }
301  else {
302  // allocate new array
303  T* newElem = 0;
304  try {
305  newElem = new T[newSize];
306  }
307  catch (std::bad_alloc& ba) {
308  Core::Application::us()->criticalError("failed to allocate memory for vector of size ") << nRows_;
309  }
310  if (!allocOnly) {
311  // ... copy old elements and free old memory
312  u32 nElements = std::min(oldSize, newSize);
313  std::copy(elem_, elem_ + nElements, newElem);
314  if (elem_)
315  delete[] elem_;
316  // initialize new memory
317  if (newSize > oldSize)
318  std::fill(newElem + oldSize, newElem + newSize, value);
319  }
320  elem_ = newElem;
321  }
322 }
