79 bool StringExpression::value(std::string& v, bool ignore_unknown_parameters) const {
80  std::string result;
81  size_t idx = 0;
82  for (Tokens::const_iterator t = tokens_.begin(); t != tokens_.end(); ++t) {
83  if (t->set()) {
84  result += t->value();
85  }
86  else if (not ignore_unknown_parameters) {
87  return false;
88  }
89  else {
90  for (Variables::const_iterator iter_v = variables_.begin(); iter_v != variables_.end(); ++iter_v) {
91  TokenReferences const& tokref = iter_v->second;
92  for (TokenReferences::const_iterator iter_r = tokref.begin(); iter_r != tokref.end(); ++iter_r) {
93  if (*iter_r == idx) {
94  result += "$(" + iter_v->first + ")";
95  goto search_end;
96  }
97  }
98  }
99  }
100  search_end:
101  idx++;
102  }
103  v = result;
104  return true;
105 }

109 bool StringExpressionParser::accept(const std::string& expression) {
110  clear();
111 
112  std::string::size_type pos = 0, begin, end;
113  while (pos < expression.size()) {
114  begin = expression.find(openTag_, pos);
115  if (begin == std::string::npos) {
116  toBuild_.pushBackToken(expression.substr(pos));
117  pos = begin;
118  }
119  else {
120  if (begin != pos)
121  toBuild_.pushBackToken(expression.substr(pos, begin - pos));
122  pos = begin + openTag_.size();
123  end = expression.find(closeTag_, pos);
124  if (end == std::string::npos || end == pos)
125  return false;
126  toBuild_.pushBackVariableToken(expression.substr(pos, end - pos));
127  pos = end + closeTag_.size();
128  }
129  }
130  return true;
131 }
