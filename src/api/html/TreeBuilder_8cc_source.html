69 TreeBuilder::TreeBuilder(Core::Configuration config,
70  const Bliss::Lexicon& lexicon,
71  const Am::AcousticModel& acousticModel,
72  Search::PersistentStateTree& network,
73  bool initialize,
74  bool arcBased)
75  : lexicon_(lexicon),
76  acousticModel_(acousticModel),
77  network_(network),
78  config_(config),
79  minPhones_(paramMinPhones(config)),
80  forceExactWordEnds_(paramForceExactWordEnds(config)),
81  keepRoots_(paramKeepRoots(config)),
82  allowCrossWordSkips_(paramAllowCrossWordSkips(config)),
83  repeatSilence_(paramRepeatSilence(config)),
84  reverse_(isBackwardRecognition(config)),
85  arcBased_(arcBased) {
86  if (allowCrossWordSkips_) {
87  Score skipPenalty = acousticModel_.stateTransition(0)->operator[](Am::StateTransitionModel::skip);
88  Score forwardPenalty = acousticModel_.stateTransition(0)->operator[](Am::StateTransitionModel::forward);
89  for (u32 t = 0; t < acousticModel_.nStateTransitions(); ++t) {
90  Score modelPenalty = acousticModel_.stateTransition(t)->operator[](Am::StateTransitionModel::skip);
91  Score modelForwardPenalty = acousticModel_.stateTransition(t)->operator[](Am::StateTransitionModel::forward);
92  if (modelPenalty != skipPenalty)
93  Core::Application::us()->warning() << "Inconsistency for forward/backward decoding: Transition model " << t
94  << ": skip penalty differs from previous value: " << modelPenalty
95  << " (previous value " << skipPenalty << ")";
96  if (modelForwardPenalty != forwardPenalty)
97  Core::Application::us()->warning() << "Inconsistency for forward/backward decoding: Transition model " << t
98  << ": forward penalty differs from previous value: " << modelForwardPenalty
99  << " (previous value " << forwardPenalty << ")";
100  }
101  }
102 
103  if (reverse_)
104  log() << "building backward network";
105  else
106  log() << "building forward network";
107 
108  if (initialize) {
109  verify(!network_.rootState);
110  network_.masterTree = network_.structure.allocateTree();
111 
112  // Non-coarticulated root state
113  network_.ciRootState = network_.rootState = createRoot(Bliss::Phoneme::term, Bliss::Phoneme::term, 0);
114  }

153 void TreeBuilder::hmmFromAllophone(TreeBuilder::HMMSequence& ret,
154  Bliss::Phoneme::Id left,
155  Bliss::Phoneme::Id central,
156  Bliss::Phoneme::Id right,
157  u32 boundary,
158  bool allowNonStandard) {
159  verify(ret.length == 0);
160  verify(central != Bliss::Phoneme::term);
161  verify(acousticModel_.phonemeInventory()->isValidPhonemeId(central));
162  Bliss::ContextPhonology::SemiContext history, future;
163 
164  if (reverse_) {
165  std::swap(left, right);
166  if (boundary == Am::Allophone::isFinalPhone)
167  boundary = Am::Allophone::isInitialPhone;
168  else if (boundary == Am::Allophone::isInitialPhone)
169  boundary = Am::Allophone::isFinalPhone;
170  }
171 
172  if (isContextDependent(central)) {
173  if (acousticModel_.phonemeInventory()->isValidPhonemeId(left) && isContextDependent(left))
174  history.append(1, left);
175  if (acousticModel_.phonemeInventory()->isValidPhonemeId(right) && isContextDependent(right))
176  future.append(1, right);
177  }
178 
179  const Am::Allophone* allophone = acousticModel_.allophoneAlphabet()->allophone(Am::Allophone(Bliss::ContextPhonology::PhonemeInContext(central, history, future), boundary));
180 
181  const Am::ClassicHmmTopology* hmmTopology = acousticModel_.hmmTopology(central);
182 
183  for (u32 phoneState = 0; phoneState < hmmTopology->nPhoneStates(); ++phoneState) {
184  Am::AllophoneState alloState = acousticModel_.allophoneStateAlphabet()->allophoneState(allophone, phoneState);
185  StateTree::StateDesc desc;
186  desc.acousticModel = acousticModel_.emissionIndex(alloState); // Decision tree look-up for CART id.
187 
188  for (u32 subState = 0; subState < hmmTopology->nSubStates(); ++subState) {
189  desc.transitionModelIndex = acousticModel_.stateTransitionIndex(alloState, subState);
190  verify(desc.transitionModelIndex < Core::Type<StateTree::StateDesc::TransitionModelIndex>::max);
191 
192  verify(ret.length < HMMSequence::MaxLength); // So far hard-wired to MaxLength = 12.
193  ret.hmm[ret.length] = desc;
194  ++ret.length;
195  }
196  }
197 
198  if (arcBased_) {
199  HMMSequence newRet;
200  newRet.length = 1;
201  newRet.hmm[0].transitionModelIndex = boundary;
202 
203  ArcSequenceHash::const_iterator it = arcSequencesHash_.find(ret);
204  if (it != arcSequencesHash_.end()) {
205  newRet.hmm[0].acousticModel = it->second;
206  if (arcDescs_[it->second].central != central)
207  arcDescs_[it->second].central = Core::Type<Bliss::Phoneme::Id>::max;
208  if (arcDescs_[it->second].left != left)
209  arcDescs_[it->second].left = Core::Type<Bliss::Phoneme::Id>::max;
210  if (arcDescs_[it->second].right != right)
211  arcDescs_[it->second].right = Core::Type<Bliss::Phoneme::Id>::max;
212  }
213  else {
214  newRet.hmm[0].acousticModel = arcSequences_.size();
215  verify(newRet.hmm[0].acousticModel == arcSequences_.size());
216  arcSequencesHash_[ret] = arcSequences_.size();
217  arcSequences_.push_back(ret);
218  ArcDesc desc;
219  desc.left = left;
220  desc.central = central;
221  desc.right = right;
222  arcDescs_.push_back(desc);
223  }
224  ret = newRet;
225  }
226  if (reverse_)
227  ret.reverse();
228 
229  if (repeatSilence_ && ret.length == 1 && central == acousticModel_.silence()) {
230  ret.hmm[1] = ret.hmm[0];
231  ret.length = 2;
232  }

235 void TreeBuilder::build() {
236  std::pair<Bliss::Lexicon::PronunciationIterator, Bliss::Lexicon::PronunciationIterator> prons = lexicon_.pronunciations();
237 
238  u32 coarticulatedInitial = 0, uncoarticulatedInitial = 0, coarticulatedFinal = 0, uncoarticulatedFinal = 0;
239 
240  // Collect initial/final phonemes
241  for (Bliss::Lexicon::PronunciationIterator pronIt = prons.first; pronIt != prons.second; ++pronIt) {
242  const Bliss::Pronunciation& pron(**pronIt);
243  if (pron.length()) {
244  Bliss::Phoneme::Id initial = pron[0], fin = pron[pron.length() - 1];
245  if (reverse_)
246  std::swap(initial, fin);
247 
248  if (!initialPhonemes_.count(initial)) {
249  initialPhonemes_.insert(initial);
250  if (isContextDependent(initial))
251  coarticulatedInitial += 1;
252  else
253  uncoarticulatedInitial += 1;
254  }
255 
256  if (!finalPhonemes_.count(fin)) {
257  if (isContextDependent(fin))
258  coarticulatedFinal += 1;
259  else
260  uncoarticulatedFinal += 1;
261 
262  finalPhonemes_.insert(fin);
263  }
264  }
265  else {
266  log() << "Ignoring 0-length pronunciation in state-network: '" << pron.format(acousticModel_.phonemeInventory()) << "'";
267  }
268  }
269 
270  if ((uncoarticulatedFinal == 0 || uncoarticulatedInitial == 0) && !paramAddCiTransitions(config_))
271  Core::Application::us()->error() << "There are no context-independent initial or final phonemes in the lexicon, word-end detection will not work properly. Consider adding context-independent phonemes, or setting add-ci-transitions=true";
272 
273  log() << "coarticulated initial phones: " << coarticulatedInitial
274  << " uncoarticulated: " << uncoarticulatedInitial
275  << ", coarticulated final phones: " << uncoarticulatedFinal
276  << " uncoarticulated: " << uncoarticulatedFinal;
277 
278  bool useRootForCiExits = paramUseRootForCiExits(config_) && !paramAddCiTransitions(config_);
279 
280  // Build the network-like non-coarticulated portion starting at the context-independent root
281  log() << "building";
282  for (Bliss::Lexicon::PronunciationIterator pronIt = prons.first; pronIt != prons.second; ++pronIt) {
283  const Bliss::Pronunciation& pron(**pronIt);
284  u32 pronLength = pron.length();
285  if (pronLength == 0)
286  continue;
287 
288  std::pair<StateId, StateId> currentState(0, network_.rootState);
289 
290  std::vector<Bliss::Phoneme::Id> phones;
291  for (u32 phoneIndex = 0; phoneIndex < pronLength; ++phoneIndex)
292  phones.push_back(pron[phoneIndex]);
293 
294  if (reverse_)
295  std::reverse(phones.begin(), phones.end());
296 
297  for (u32 phoneIndex = 0; phoneIndex < pronLength - 1; ++phoneIndex)
298  currentState = extendPhone(currentState.second, phoneIndex, phones);
299 
300  std::pair<Bliss::Pronunciation::LemmaIterator, Bliss::Pronunciation::LemmaIterator> lemmaProns = pron.lemmas();
301 
302  if (pronLength - 1 < minPhones_ || !isContextDependent(phones[pronLength - 1])) {
303  // Statically expand the fan-out.
304  for (std::set<Bliss::Phoneme::Id>::iterator initialIt = initialPhonemes_.begin(); initialIt != initialPhonemes_.end(); ++initialIt) {
305  std::pair<StateId, StateId> tail = extendPhone(currentState.second, pronLength - 1, phones, Bliss::Phoneme::term, *initialIt);
306  for (Bliss::Pronunciation::LemmaIterator lemmaPron = lemmaProns.first; lemmaPron != lemmaProns.second; ++lemmaPron) {
307  u32 exit;
308  if (!isContextDependent(phones[pronLength - 1]) && useRootForCiExits)
309  exit = addExit(tail.first, tail.second, Bliss::Phoneme::term, Bliss::Phoneme::term, 0, lemmaPron->id()); // Use the non-coarticulated root node
310  else
311  exit = addExit(tail.first, tail.second, phones[pronLength - 1], *initialIt, 0, lemmaPron->id());
312  if (pronLength == 1)
313  initialFinalPhoneSuffix_[RootKey(phones[0], *initialIt, 1)].insert(ID_FROM_LABEL(exit));
314  }
315  }
316  }
317  else {
318  // Minimize the remaining phoneme, insert corresponding word-ends.
319  for (Bliss::Pronunciation::LemmaIterator lemmaPron = lemmaProns.first; lemmaPron != lemmaProns.second; ++lemmaPron) {
320  if (pronLength == 1) {
321  addExit(currentState.first, currentState.second, Bliss::Phoneme::term, phones[0], -1, lemmaPron->id());
322 
323  for (std::set<Bliss::Phoneme::Id>::const_iterator finalIt = finalPhonemes_.begin(); finalIt != finalPhonemes_.end(); ++finalIt) {
324  Search::PersistentStateTree::Exit exit;
325  exit.transitState = createRoot(*finalIt, phones[0], -1);
326  exit.pronunciation = lemmaPron->id();
327  addSuccessor(createRoot(*finalIt, phones[0], 0), ID_FROM_LABEL(createExit(exit)));
328  }
329  }
330  else {
331  u32 exit = addExit(currentState.first, currentState.second, phones[pronLength - 2], phones[pronLength - 1], -1, lemmaPron->id());
332  if (pronLength == 2)
333  initialPhoneSuffix_[RootKey(phones[0], phones[1], 1)].insert(ID_FROM_LABEL(exit));
334  }
335  }
336  }
337  }
338 
339  log() << "states: " << network_.structure.stateCount() << " exits: " << network_.exits.size() << " roots: " << roots_.size();
340 
341  buildFanInOutStructure();
342 
343  skipRootTransitions();
344 
345  u32 it = paramMinimizeIterations(config_);
346  for (u32 i = 0; i < it; ++i)
347  minimize();
348 
349  if (allowCrossWordSkips_)
350  addCrossWordSkips();
351 
352  log() << "building ready";
353 }

355 void TreeBuilder::addCrossWordSkips() {
356  log() << "adding cross-word skips";
357  u32 oldNodes = network_.structure.stateCount();
358 
359  for (StateId node = 1; node < oldNodes; ++node) {
360  {
361  bool hasWordEnd = false;
362  bool hasSuccessor = false;
363  for (HMMStateNetwork::SuccessorIterator target = network_.structure.successors(node); target; ++target) {
364  if (!target.isLabel())
365  hasSuccessor = true;
366  if (target.isLabel() && network_.structure.state(network_.exits[target.label()].transitState).stateDesc.transitionModelIndex != Am::TransitionModel::entryM2)
367  hasWordEnd = true;
368  }
369  verify(hasSuccessor || hasWordEnd);
370  }
371 
372  std::set<PersistentStateTree::Exit> skipRoots;
373 
374  for (HMMStateNetwork::SuccessorIterator target = network_.structure.successors(node); target; ++target) {
375  if (target.isLabel())
376  continue;
377  for (HMMStateNetwork::SuccessorIterator target2 = network_.structure.successors(*target); target2; ++target2)
378  if (target2.isLabel())
379  skipRoots.insert(network_.exits[target2.label()]);
380  }
381 
382  if (skipRoots.size()) {
383  for (std::set<PersistentStateTree::Exit>::iterator it = skipRoots.begin(); it != skipRoots.end(); ++it) {
384  PersistentStateTree::Exit e(*it);
385  verify(e.pronunciation != Bliss::LemmaPronunciation::invalidId);
386  if (network_.structure.state(e.transitState).stateDesc.transitionModelIndex == Am::TransitionModel::entryM2)
387  continue;
388  e.transitState = createSkipRoot(e.transitState);
389  network_.structure.addOutputToNode(node, createExit(e));
390  }
391  }
392 
393  {
394  bool hasWordEnd = false;
395  bool hasSuccessor = false;
396  for (HMMStateNetwork::SuccessorIterator target = network_.structure.successors(node); target; ++target) {
397  if (!target.isLabel())
398  hasSuccessor = true;
399  if (target.isLabel() && network_.structure.state(network_.exits[target.label()].transitState).stateDesc.transitionModelIndex != Am::TransitionModel::entryM2)
400  hasWordEnd = true;
401  }
402  verify(hasSuccessor || hasWordEnd);
403  }
404  }
405 
406  for (StateId node = 1; node < oldNodes; ++node) {
407  bool hasWordEnd = false;
408  bool hasSuccessor = false;
409  for (HMMStateNetwork::SuccessorIterator target = network_.structure.successors(node); target; ++target) {
410  if (!target.isLabel())
411  hasSuccessor = true;
412  if (target.isLabel() && network_.structure.state(network_.exits[target.label()].transitState).stateDesc.transitionModelIndex != Am::TransitionModel::entryM2)
413  hasWordEnd = true;
414  }
415  verify(hasSuccessor || hasWordEnd);
416  }
417 
418  log() << "added " << network_.structure.stateCount() - oldNodes << " skip-roots";
419 
420  network_.cleanup();
421 }

423 void TreeBuilder::skipRootTransitions() {
424  for (StateId node = 1; node < network_.structure.stateCount(); ++node) {
425  if (network_.structure.state(node).stateDesc.acousticModel == Search::StateTree::invalidAcousticModel)
426  continue;
427 
428  HMMStateNetwork::ChangePlan change = network_.structure.change(node);
429  for (HMMStateNetwork::SuccessorIterator target = network_.structure.successors(node); target; ++target) {
430  if (target.isLabel())
431  continue;
432 
433  if (network_.structure.state(*target).stateDesc.acousticModel == Search::StateTree::invalidAcousticModel) {
434  change.removeSuccessor(*target);
435  for (HMMStateNetwork::SuccessorIterator target2 = network_.structure.successors(*target); target2; ++target2)
436  change.addSuccessor(*target2);
437  }
438  }
439  change.apply();
440  }

443 std::vector<TreeBuilder::StateId> TreeBuilder::minimize(bool forceDeterminization, bool onlyMinimizeBackwards, bool allowLost) {
444  log() << "minimizing";
445 
446  if (forceExactWordEnds_)
447  log() << "forcing exact word-ends";
448 
449  for (std::set<StateId>::iterator it = network_.unpushedCoarticulatedRootStates.begin(); it != network_.unpushedCoarticulatedRootStates.end(); ++it)
450  verify(network_.coarticulatedRootStates.count(*it));
451 
452  std::set<StateId> usedRoots;
453  std::deque<StateId> active;
454 
455  std::vector<u32> fanIn(network_.structure.stateCount(), 0);
456 
457  for (StateId node = 1; node < network_.structure.stateCount(); ++node) {
458  active.push_back(node);
459  for (HMMStateNetwork::SuccessorIterator target = network_.structure.successors(node); target; ++target) {
460  if (target.isLabel()) {
461  usedRoots.insert(network_.exits[target.label()].transitState);
462  fanIn[network_.exits[target.label()].transitState] += 1;
463  }
464  else {
465  fanIn[*target] += 1;
466  }
467  }
468  }
469 
470  log() << "keeping " << usedRoots.size() << " out of " << network_.coarticulatedRootStates.size() << " roots";
471  std::set<StateId> oldCoarticulatedRoots = network_.coarticulatedRootStates;
472  for (std::set<StateId>::iterator it = oldCoarticulatedRoots.begin(); it != oldCoarticulatedRoots.end(); ++it) {
473  if (usedRoots.count(*it) == 0) {
474  network_.coarticulatedRootStates.erase(*it);
475  network_.rootTransitDescriptions.erase(*it);
476  network_.unpushedCoarticulatedRootStates.erase(*it);
477  network_.structure.clearOutputEdges(*it);
478  }
479  }
480 
481  std::vector<StateId> determinizeMap(network_.structure.stateCount(), 0);
482 
483  u32 determinizeClashes = 0;
484 
485  if (onlyMinimizeBackwards) {
486  log() << "skipping determinization";
487  for (StateId node = 1; node < network_.structure.stateCount(); ++node)
488  determinizeMap[node] = node;
489  }
490  else {
491  // Determinize states: Join successor states with the same state-desc
492  while (!active.empty()) {
493  StateId state = active.front();
494  active.pop_front();
495  HMMStateNetwork::ChangePlan change = network_.structure.change(state);
496  typedef std::unordered_multimap<StateTree::StateDesc, StateId, StateTree::StateDesc::Hash> SuccessorHash;
497  SuccessorHash successors;
498  for (HMMStateNetwork::SuccessorIterator target = network_.structure.successors(state); target; ++target)
499  if (!target.isLabel() && (forceDeterminization || fanIn[*target] == 1))
500  successors.insert(std::make_pair(network_.structure.state(*target).stateDesc, *target));
501 
502  while (!successors.empty()) {
503  std::pair<SuccessorHash::iterator, SuccessorHash::iterator> items = successors.equal_range(successors.begin()->first);
504 
505  SuccessorHash::iterator it = items.first;
506  if (++it != items.second) {
507  StateId newNode = network_.structure.allocateTreeNode(network_.masterTree);
508  if (newNode >= determinizeMap.size())
509  determinizeMap.resize(newNode + 1, 0);
510  network_.structure.state(newNode).stateDesc = items.first->first;
511  if (network_.uncoarticulatedWordEndStates.count(items.first->second))
512  network_.uncoarticulatedWordEndStates.insert(newNode);
513  HMMStateNetwork::ChangePlan newChange = network_.structure.change(newNode);
514  // There are multiple successors with the same state-desc, join them
515  for (it = items.first; it != items.second; ++it) {
516  verify(it->second < determinizeMap.size());
517  if (forceExactWordEnds_ && network_.uncoarticulatedWordEndStates.count(it->second))
518  network_.uncoarticulatedWordEndStates.insert(newNode);
519  if (determinizeMap[it->second])
520  ++determinizeClashes;
521  determinizeMap[it->second] = newNode;
522  for (HMMStateNetwork::SuccessorIterator target2 = network_.structure.successors(it->second); target2; ++target2)
523  newChange.addSuccessor(*target2);
524  change.removeSuccessor(it->second);
525  }
526  newChange.apply();
527  change.addSuccessor(newNode);
528  active.push_back(newNode);
529  }
530 
531  successors.erase(items.first, items.second);
532  }
533  change.apply();
534  }
535  log() << "clashes during determinization: " << determinizeClashes;
536  }
537 
538  // Minimize: Join states with the same successors/exits
539  predecessors_.clear();
540 
541  std::vector<StateId> minimizeMap(network_.structure.stateCount(), 0);
542 
543  minimizeState(network_.rootState, minimizeMap);
544  for (std::set<StateId>::iterator it = network_.coarticulatedRootStates.begin(); it != network_.coarticulatedRootStates.end(); ++it)
545  minimizeState(*it, minimizeMap);
546  for (std::set<StateId>::iterator it = skipRootSet_.begin(); it != skipRootSet_.end(); ++it)
547  minimizeState(*it, minimizeMap);
548 
549  verify(minimizeMap[network_.rootState] == network_.rootState);
550 
551  if (!keepRoots_) {
552  std::vector<u32> minimizeExitsMap(network_.exits.size(), Core::Type<u32>::max);
553  {
554  std::vector<PersistentStateTree::Exit> oldExits;
555  oldExits.swap(network_.exits);
556  exitHash_.clear();

561  minimizeExitsMap[exitIndex] = createExit(exit);

567  log() << "joining exits, coarticulated roots before: " << network_.coarticulatedRootStates.size();

571  minimizeExits(state, minimizeExitsMap);
572  else
573  network_.structure.clearOutputEdges(state);
574  }
575  }
576 
577  log() << "coarticulated roots after joining: " << network_.coarticulatedRootStates.size();
578 
579  network_.ciRootState = network_.rootState = minimizeMap[network_.rootState];
580 
581  mapSet(network_.coarticulatedRootStates, minimizeMap, true);
582  mapSet(network_.unpushedCoarticulatedRootStates, minimizeMap, true);
583  mapSet(skipRootSet_, minimizeMap, true);
584  mapSet(network_.uncoarticulatedWordEndStates, minimizeMap, forceExactWordEnds_);

586  {
587  PersistentStateTree::RootTransitDescriptions oldTransitDescs;
588  oldTransitDescs.swap(network_.rootTransitDescriptions);

590  for (std::map<StateId, std::pair<Bliss::Phoneme::Id, Bliss::Phoneme::Id>>::iterator it = oldTransitDescs.begin(); it != oldTransitDescs.end(); ++it) {
591  StateId orig = it->first;
592  if (orig == network_.rootState || orig >= minimizeMap.size()) {
593  if (orig == network_.rootState || network_.coarticulatedRootStates.count(orig))
594  network_.rootTransitDescriptions.insert(*it);
595  }
596  else {
597  StateId mapped = minimizeMap[it->first];
598  verify(mapped);
599  verify(network_.coarticulatedRootStates.count(mapped));
600 
601  if (mapped == network_.rootState) {
602  network_.coarticulatedRootStates.erase(network_.rootState);
603  network_.unpushedCoarticulatedRootStates.erase(network_.rootState);
604  continue;
605  }
606  network_.rootTransitDescriptions.insert(std::make_pair(mapped, it->second));
607  }
608  }
609  }
610 
611  log() << "cleaning";
612  u32 lost = 0, kept = 0;
613  for (StateId state = 1; state < determinizeMap.size(); ++state) {
614  if (determinizeMap[state])
615  determinizeMap[state] = minimizeMap[determinizeMap[state]];
616  else
617  determinizeMap[state] = minimizeMap[state];
618  }
619  minimizeMap = determinizeMap;
620 
621  HMMStateNetwork::CleanupResult cleanupResult = network_.cleanup();
622  for (std::vector<StateId>::iterator it = minimizeMap.begin(); it != minimizeMap.end(); ++it) {
623  if (*it) {
624  if (cleanupResult.nodeMap.count(*it)) {
625  *it = cleanupResult.nodeMap[*it];
626  kept += 1;
627  verify(*it);
628  }
629  else {
630  lost += 1;
631  *it = 0;
632  }
633  }
634  }
635  log() << "transformed states: " << kept << " lost: " << lost;
636  // verify( allowLost || !lost );
637 
638  printStats("after minimization");

657 void TreeBuilder::minimizeState(StateId state, std::vector<StateId>& minimizeMap) {
658  verify(state < minimizeMap.size());
659  if (minimizeMap[state])
660  return;
661 
662  minimizeMap[state] = Core::Type<u32>::max;
663 
664  verify(state && state < network_.structure.stateCount());
665  std::set<StateId> successors;
666  for (HMMStateNetwork::SuccessorIterator target = network_.structure.successors(state); target; ++target) {
667  if (target.isLabel()) {
668  successors.insert(*target);
669  continue;
670  }
671  verify(*target < minimizeMap.size());
672  minimizeState(*target, minimizeMap);
673  verify(minimizeMap[*target] != 0);
674  if (minimizeMap[*target] == Core::Type<u32>::max) {
675  // std::cout << "detected recursion while minimization on " << *target << std::endl;
676  successors.insert(*target);
677  }
678  else {
679  successors.insert(minimizeMap[*target]);
680  }
681  }
682 
683  network_.structure.clearOutputEdges(state);
684 
685  StatePredecessor pred(successors, network_.structure.state(state).stateDesc, forceExactWordEnds_ && network_.uncoarticulatedWordEndStates.count(state));
686  std::unordered_map<StatePredecessor, StateId, StatePredecessor::Hash>::iterator it = predecessors_.find(pred);
687  if (it != predecessors_.end()) {
688  minimizeMap[state] = it->second;
689  }
690  else {
691  minimizeMap[state] = state;
692  predecessors_.insert(std::make_pair(pred, state));
693  for (std::set<StateId>::iterator succIt = successors.begin(); succIt != successors.end(); ++succIt)
694  network_.structure.addTargetToNode(state, *succIt);
695  }

698 void TreeBuilder::minimizeExits(StateId state, const std::vector<u32>& minimizeExitsMap) {
699  typedef std::multimap<Bliss::LemmaPronunciation::Id, u32> ExitMap;
700  ExitMap successorExits;
701 
702  {
703  std::set<StateId> successorStates;
704  for (HMMStateNetwork::SuccessorIterator target = network_.structure.successors(state); target; ++target) {
705  if (target.isLabel()) {
706  successorExits.insert(std::make_pair(network_.exits[minimizeExitsMap[target.label()]].pronunciation, minimizeExitsMap[target.label()]));
707  continue;
708  }
709  successorStates.insert(*target);
710  }
711 
712  if (successorExits.empty())
713  return;
714 
715  network_.structure.clearOutputEdges(state);
716  for (std::set<StateId>::iterator it = successorStates.begin(); it != successorStates.end(); ++it)
717  network_.structure.addTargetToNode(state, *it);
718  }
719 
720  // Join multiple exits for the same pronunciation to one
721  while (successorExits.size()) {
722  std::pair<ExitMap::iterator, ExitMap::iterator> range = successorExits.equal_range(successorExits.begin()->first);
723  ExitMap::iterator i = range.first;
724  if (++i == range.second) {
725  network_.structure.addOutputToNode(state, range.first->second);
726  }
727  else {
728  // Join
729  std::set<StateId> newRootSuccessors;
730  std::set<Bliss::Phoneme::Id> left, right;
731  for (i = range.first; i != range.second; ++i) {
732  for (HMMStateNetwork::SuccessorIterator target = network_.structure.successors(network_.exits[i->second].transitState); target; ++target)
733  newRootSuccessors.insert(*target);
734  left.insert(network_.rootTransitDescriptions[network_.exits[i->second].transitState].first);
735  right.insert(network_.rootTransitDescriptions[network_.exits[i->second].transitState].second);
736  }
737 
738  PersistentStateTree::Exit exit;
739  exit.pronunciation = range.first->first;
740  u32 newNodeLimit = network_.structure.stateCount();
741  exit.transitState = extendFanIn(newRootSuccessors, rootDesc());
742  network_.structure.addOutputToNode(state, createExit(exit));
743  if (exit.transitState >= newNodeLimit) {
744  network_.coarticulatedRootStates.insert(exit.transitState);
745  network_.rootTransitDescriptions.insert(std::make_pair(exit.transitState, std::make_pair(left.size() == 1 ? *left.begin() : Bliss::Phoneme::term, right.size() == 1 ? *right.begin() : Bliss::Phoneme::term)));
746  for (i = range.first; i != range.second; ++i) {
747  verify(i->second < network_.exits.size());
748  if (network_.unpushedCoarticulatedRootStates.count(network_.exits[i->second].transitState))
749  network_.unpushedCoarticulatedRootStates.insert(exit.transitState);
750  if (network_.uncoarticulatedWordEndStates.count(network_.exits[i->second].transitState))
751  network_.uncoarticulatedWordEndStates.insert(exit.transitState);
752  }
753  }
754  }
755  successorExits.erase(range.first, range.second);
756  }
757 }

759 void TreeBuilder::buildFanInOutStructure() {
760  bool ciTransitions = paramAddCiTransitions(config_);
761 
762  // Create temporary coarticulated roots
763  for (std::set<Bliss::Phoneme::Id>::const_iterator finalIt = finalPhonemes_.begin(); finalIt != finalPhonemes_.end(); ++finalIt)
764  for (std::set<Bliss::Phoneme::Id>::const_iterator initialIt = initialPhonemes_.begin(); initialIt != initialPhonemes_.end(); ++initialIt)
765  createRoot(*finalIt, *initialIt, 0);
766 
767  log() << "building fan-in";
768  // Build the fan-in structure (eg. the HMM structure representing the initial word phonemes, behind roots, up to the joints)
769  for (RootHash::const_iterator rootIt = roots_.begin(); rootIt != roots_.end(); ++rootIt) {
770  if (rootIt->first.depth != 0 || rootIt->second == network_.rootState)
771  continue;
772  Bliss::Phoneme::Id initial = rootIt->first.right;
773  verify(initialPhonemes_.count(initial));
774  verify(initial != Bliss::Phoneme::term);
775  u32 paths = 0;
776 
777  for (CoarticulationJointHash::const_iterator initialSuffixIt = initialPhoneSuffix_.begin(); initialSuffixIt != initialPhoneSuffix_.end(); ++initialSuffixIt) {
778  if (initialSuffixIt->first.left != initial)
779  continue;
780  ++paths;
781  HMMSequence hmm;
782  hmmFromAllophone(hmm, rootIt->first.left, initial, initialSuffixIt->first.right, Am::Allophone::isInitialPhone);
783  verify(hmm.length > 0);
784  StateId currentNode = extendFanIn(initialSuffixIt->second, hmm[hmm.length - 1]);
785  for (s32 s = hmm.length - 2; s >= 0; --s)
786  currentNode = extendFanIn(currentNode, hmm[s]);
787 
788  addSuccessor(rootIt->second, currentNode);
789  }
790 
791  for (CoarticulationJointHash::const_iterator initialSuffixIt = initialFinalPhoneSuffix_.begin(); initialSuffixIt != initialFinalPhoneSuffix_.end(); ++initialSuffixIt) {
792  if (initialSuffixIt->first.left != initial)
793  continue;
794  ++paths;
795  HMMSequence hmm;
796  hmmFromAllophone(hmm, rootIt->first.left, initial, initialSuffixIt->first.right, Am::Allophone::isInitialPhone | Am::Allophone::isFinalPhone);
797  verify(hmm.length > 0);
798  StateId currentNode = extendFanIn(initialSuffixIt->second, hmm[hmm.length - 1]);
799  for (s32 s = hmm.length - 2; s >= 0; --s)
800  currentNode = extendFanIn(currentNode, hmm[s]);
801 
802  addSuccessor(rootIt->second, currentNode);
803  }
804  }
805 
806  log() << "states: " << network_.structure.stateCount() << " exits: " << network_.exits.size() << " roots: " << roots_.size();
807 
808  log() << "building fan-out";
809 
810  // Build the fan-out structure (eg. the HMM structure representing the final word phonemes, behind special roots)
811  // At the left side delimited by the roots of depth -1, and at the right side by the roots of depth 0
812  for (RootHash::const_iterator leftRootIt = roots_.begin(); leftRootIt != roots_.end(); ++leftRootIt) {
813  if (leftRootIt->first.depth != -1)
814  continue;
815  Bliss::Phoneme::Id fin = leftRootIt->first.right;
816  verify(finalPhonemes_.count(fin));
817 
818  u32 paths = 0;
819  for (RootHash::const_iterator rightRootIt = roots_.begin(); rightRootIt != roots_.end(); ++rightRootIt) {
820  if (rightRootIt->first.depth != 0 || (rightRootIt->first.left != fin && (!ciTransitions || rightRootIt->first.left != Bliss::Phoneme::term)))
821  continue;
822  ++paths;
823  HMMSequence hmm;
824  hmmFromAllophone(hmm, leftRootIt->first.left, fin, rightRootIt->first.right, Am::Allophone::isFinalPhone, false);
825  verify(hmm.length > 0);
826 
827  // The last state in the pushed fan-in is equivalent with the corresponding root state
828 
829  StateId lastNode = extendFanIn(network_.structure.targetSet(rightRootIt->second), hmm[hmm.length - 1]);
830  StateId currentNode = lastNode;
831  for (s32 s = hmm.length - 2; s >= 0; --s)
832  currentNode = extendFanIn(currentNode, hmm[s]);
833 
834  if (rightRootIt->first.right == Bliss::Phoneme::term || !isContextDependent(rightRootIt->first.right))
835  network_.uncoarticulatedWordEndStates.insert(lastNode);
836 
837  addSuccessor(leftRootIt->second, currentNode);
838  }
839  verify(paths);
840  }
841 
842  printStats("after fan-in/out structure");
843 }
