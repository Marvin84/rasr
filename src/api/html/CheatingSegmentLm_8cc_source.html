37 void CheatingSegmentLm::setSegment(Bliss::SpeechSegment const* s) {
38  std::string orth = s->orth();
39  Core::normalizeWhitespace(orth);
40  Core::enforceTrailingBlank(orth);
41 
42  Core::Ref<Bliss::LemmaAcceptor> orth_automaton = orthParser_->createLemmaAcceptor(orth); // automaton using orth-alphabet
43  auto alphabet = orth_automaton->inputAlphabet();
44 
45  // build automaton using syntactic-token alphabet
46  Fsa::StaticAutomaton* synt_automaton = new Fsa::StaticAutomaton();
47  synt_automaton->setType(Fsa::TypeAcceptor);
48  synt_automaton->setInputAlphabet(lexicon_->syntacticTokenAlphabet());
49  synt_automaton->setSemiring(Fsa::TropicalSemiring);
50  synt_automaton->setDescription(std::string("cheatingLm(") + s->fullName() + std::string(")"));
51 
52  std::vector<Fsa::StateId> id_map;
53  id_map.reserve(orth_automaton->size());
54  for (Fsa::StateId sid = 0u; sid <= orth_automaton->maxStateId(); sid++) {
55  if (orth_automaton->hasState(sid)) {
56  auto* orth_state = orth_automaton->fastState(sid);
57  auto* synt_state = synt_automaton->newState(orth_state->tags(), orth_state->weight());
58  id_map.push_back(synt_state->id());
59  }
60  else {
61  id_map.push_back(0u);
62  }
63  }
64  synt_automaton->setInitialStateId(id_map[orth_automaton->initialStateId()]);
65  for (Fsa::StateId sid = 0u; sid <= orth_automaton->maxStateId(); sid++) {
66  if (!orth_automaton->hasState(sid)) {
67  continue;
68  }
69  auto* orth_state = orth_automaton->fastState(sid);
70  for (auto arc = orth_state->begin(); arc != orth_state->end(); ++arc) {
71  Bliss::Lemma const* lemma = lexicon_->lemma(alphabet->symbol(arc->input()));
72  require(lemma != nullptr);
73  if (lexicon_->specialLemma("silence") != lemma and lemma->hasSyntacticTokenSequence()) {
74  auto& tokens = lemma->syntacticTokenSequence();
75  if (tokens.size() > 0ul) {
76  Fsa::StateId source_id = id_map[sid];
77  for (size_t idx = 0ul; idx + 1 < tokens.size(); idx++) {
78  auto* source_state = synt_automaton->fastState(source_id);
79  auto* next_state = synt_automaton->newState();
80  source_state->newArc(next_state->id(), Fsa::StaticAutomaton::Weight(), tokens[idx]->id());
81  source_id = next_state->id();
82  }
83  auto* synt_state = synt_automaton->fastState(source_id);
84  synt_state->newArc(id_map[arc->target()], arc->weight(), tokens[tokens.size() - 1ul]->id());
85  }
86  }
87  }
88  }
89  auto synt_ref = Fsa::ConstAutomatonRef(synt_automaton);
90  this->setFsa(synt_ref);
91 }
