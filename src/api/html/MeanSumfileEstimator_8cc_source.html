53 int MeanSumfileCreator::main(const vector<string>& arguments) {
54  select("lexicon");
55 
56  Bliss::LexiconRef lexicon = Bliss::Lexicon::create(select("lexicon"));
57  if (!lexicon)
58  criticalError("failed to initialize lexicon");
59 
60  const Legacy::PhoneticDecisionTree* dectree = new Legacy::PhoneticDecisionTree(select("decision-tree"),
61  lexicon->phonemeInventory());
62  dectree->respondToDelayedErrors();
63 
64  Core::Ref<Legacy::MixtureSet> mixtureSet = Legacy::createMixtureSet(paramMixtureSetFilename(config));
65  if (!mixtureSet)
66  criticalError("failed to load mixture set");
67 
68  std::vector<std::set<u32>> phonemeToMixtureIndizes = dectree->PhonemeToMixtureIndizes();
69  std::vector<Bliss::Phoneme::Id> mixtureToPhoneme(mixtureSet->nMixtures(), -1);
70  for (Bliss::Phoneme::Id p = 1; p < phonemeToMixtureIndizes.size(); ++p) {
71  for (std::set<u32>::const_iterator m = phonemeToMixtureIndizes[p].begin(); m != phonemeToMixtureIndizes[p].end(); ++m) {
72  mixtureToPhoneme[*m] = p;
73  }
74  }
75 
76  for (std::vector<Bliss::Phoneme::Id>::const_iterator m = mixtureToPhoneme.begin();
77  m != mixtureToPhoneme.end(); ++m)
78  if (*m == -1)
79  error("no entry in mixtureToPhoneme for mixture ") << *m;
80 
81  Legacy::MeanSumfileEstimator meanSumfileEstimator(mixtureSet, mixtureToPhoneme, lexicon->phonemeInventory());
82 
83  std::string test(paramSumfileFilename(config));
84  meanSumfileEstimator.write(test);
85 
86  delete dectree;
87  return 0;
88 }
