32  for (u32 layer = 0; layer < network.nLayers(); layer++) {
33  if (network.getLayer(layer).isTrainable()) {

46 /**
47  * estimation with mean-normalized SGD
48  * see Wiesler, Richard, Schlüter & Ney: Mean-Normalized SGD, ICASSP 2014
49  */
50 template<typename T>
51 void MeanNormalizedSgd<T>::estimate(NeuralNetwork<T>& network, Statistics<T>& statistics) {
52  if (firstEstimation_) {
53  checkForStatistics(network);
54  firstEstimation_ = false;
55  }
56 
57  T learningRate = initialLearningRate_;
58 
59  /*
60  * estimation of parameters:
61  *
62  * W -= eta * delta_W (update weight matrix)
63  * a -= eta * delta_a (update bias vector)
64  *
65  * with
66  *
67  * delta_W = gradient_W + b . gradient_a^T
68  * delta_a = \gradient_W^T . b + (1 + b^T b) \gradient_a
69  *
70  * The shift b is the negative of the (smoothed) activation mean.
71  * The update matrices are stored in place, i.e. the gradients are replaced by the update terms.
72  *
73  */
74  require(statistics.hasGradient());
75  std::vector<T> stepSizes(network.nLayers());
76 
77  for (u32 layer = 0; layer < network.nLayers(); layer++) {
78  if (network.getLayer(layer).isTrainable()) {
79  /* modify weights gradient and update weights */
80  for (u32 stream = 0; stream < network.getLayer(layer).nPredecessors(); stream++) {
81  // if layer has predecessors with activation statistics...
82  u32 predecessor = network.getLayer(layer).getPredecessor(stream);
83  if (network.getLayer(predecessor).hasActivationStatistics()) {
84  // ... modify gradient of weights for the current input stream
85  // delta_W = gradient_W + b . gradient_a^T
86  statistics.gradientWeights(layer)[stream].addOuterProduct(network.getLayer(predecessor).getActivationMean(),
87  statistics.gradientBias(layer), -1.0);
88  }
89  // update weights
90  NnMatrix* weights = network.getLayer(layer).getWeights(stream);
91  require(weights);
92  T localLearningRate = learningRate * network.getLayer(layer).learningRate();
93  weights->add(statistics.gradientWeights(layer)[stream], (T)-localLearningRate);
94 
95  // log step size
96  if (logStepSize_)
97  stepSizes[layer] += statistics.gradientWeights(layer)[stream].l1norm() * localLearningRate;
98  }
99 
100  /* modify bias gradient and update bias */
101  for (u32 stream = 0; stream < network.getLayer(layer).nPredecessors(); stream++) {
102  // if layer has predecessors with activation statistics...
103  u32 predecessor = network.getLayer(layer).getPredecessor(stream);
104  if (network.getLayer(predecessor).hasActivationStatistics()) {
105  // ... modify gradient of bias
106  // delta_a = \gradient_W^T . b + (1 + b^T b) \gradient_a
107  // = delta_a + \delta_W^T b
108  statistics.gradientWeights(layer)[stream].multiply(network.getLayer(predecessor).getActivationMean(),
109  statistics.gradientBias(layer), true, -1.0, 1.0);
110  }
111  }
112  // update bias
113  NnVector* bias = network.getLayer(layer).getBias();
114  require(bias);
115  T localLearningRate = learningRate * biasLearningRate_ * network.getLayer(layer).learningRate();
116  bias->add(statistics.gradientBias(layer), (T)-localLearningRate);
117  /* log step size */
118  if (logStepSize_)
119  stepSizes[layer] += statistics.gradientBias(layer).l1norm() * localLearningRate;
120  }
121  }
122  if (logStepSize_ && statisticsChannel_.isOpen()) {
123  T stepSize = Math::asum<T>(stepSizes.size(), &stepSizes.at(0), 1);
124  statisticsChannel_ << "step-size: " << stepSize << " (" << Core::vector2str(stepSizes, ",") << ")";
125  }
126 }

135 template<typename T>
136 void MeanNormalizedSgdL1Clipping<T>::estimate(NeuralNetwork<T>& network, Statistics<T>& statistics) {
137  Precursor::estimate(network, statistics);
138 
139  T learningRate = this->initialLearningRate_;
140 
141  for (u32 layer = 0; layer < network.nLayers(); layer++) {
142  if (network.getLayer(layer).isTrainable()) {
143  if (network.getLayer(layer).nInputActivations() != 1) {
144  Core::Application::us()->criticalError("Estimation for multiple streams not yet implemented.");
145  }
146  NnMatrix* weights = network.getLayer(layer).getWeights(0);
147  NnVector* bias = network.getLayer(layer).getBias();
148  require(weights);
149  require(bias);
150  weights->l1clipping(network.getLayer(layer).regularizationConstant() * learningRate * network.getLayer(layer).learningRate());
151  bias->l1clipping(network.getLayer(layer).regularizationConstant() * learningRate * network.getLayer(layer).learningRate());
152  }
153  }
154 
155  if (logStepSize_ && statisticsChannel_.isOpen()) {
156  statisticsChannel_ << "step size does not include l1-regularization";
157  }
158 }
