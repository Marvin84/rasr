64 void WindowingFirFilter::getFilterResponse(std::vector<Sample>& h) {
65  verify(!cutOff_.empty());
66 
67  h.resize(M_ + 1);
68 
69  for (u32 n = 0; n <= M_ / 2; n++) {
70  h[n] = h[M_ - n] = 0;
71 
72  for (u8 k = 1; k <= cutOff_.size(); k++) {
73  f64 omega = cutOff_[k - 1].second * 2 * M_PI / (f64)sampleRate();
74 
75  h[n] = h[M_ - n] += (cutOff_[k - 1].first - (k != cutOff_.size() ? cutOff_[k].first : 0.0)) *
76  (omega * sinc(omega * ((Sample)n - ((Sample)M_ / (Sample)2))) / (Sample)M_PI);
77  }
78  }
79 
80  window_.setBeta(getBeta(overshoot_));
81  window_.setLength(M_ + 1);
82 
83  if (!window_.work(h.begin(), h.end()))
84  defect();
85 }

87 f64 WindowingFirFilter::getBeta(f64 overshoot) {
88  f64 beta;
89 
90  f64 A = -20.0 * log10(overshoot);
91 
92  if (A > 50.0)
93  beta = 0.1102 * (A - 8.7);
94  else if (A >= 21.0)
95  beta = 0.5842 * exp(0.4 * std::log(A - 21.0)) + 0.07886 * (A - 21.0);
96  else
97  beta = 0;
98 
99  return beta;
100 }

102 u32 WindowingFirFilter::getM(f64 overshoot, f64 deltaOmega) {
103  f64 A = -20.0 * log10(overshoot);

105  u32 M = (u32)((A - 8) / 2.285 / deltaOmega);

273 bool WindowingFirFilterNode::parseCutOff(const std::string& value, std::vector<std::pair<f64, f64>>& cutOff) {
274  std::string str(value);
275  std::string element;
276  std::string::size_type pos;
277 
278  char* error;
279  std::pair<f64, f64> gainFrequencyPair;
280  cutOff.clear();
281  while (!str.empty()) {
282  pos = str.find_first_of(";"); // semicolon-separated elements
283 
284  if (pos != 0) {
285  element = str.substr(0, pos); // element is gain:frequency
286 
287  gainFrequencyPair.first = strtod(element.c_str(), &error);
288  if (*error != ':')
289  return false;
290  element = error + 1;
291  gainFrequencyPair.second = strtod(element.c_str(), &error);
292 
293  cutOff.push_back(gainFrequencyPair);
294  }
295 
296  str.erase(0, (pos == std::string::npos) ? pos : pos + 1);
297  }
298 
299  return cutOff.size();
300 }
