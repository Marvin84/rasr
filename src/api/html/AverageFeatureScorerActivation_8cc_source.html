36 void AverageFeatureScorerActivation::leaveSpeechSegment(Bliss::SpeechSegment* segment) {
37  for (size_t i = 0ul; i < scorers_.size(); i++) {
38  if (scores_.empty()) {
39  scores_.resize(scorers_[i]->nEmissions(), 0.0);
40  }
41  for (size_t j = 0ul; j < scores_.size(); j++) {
42  Score s = scorers_[i]->score(j);
43  if (transform_to_probabilities_) {
44  scores_[j] += std::exp(-s);
45  }
46  else {
47  scores_[j] += s;
48  }
49  }
50  }
51  n_frames_ += scorers_.size();
52  scorers_.clear();
53  feature_scorer_->reset();
54 
55  Precursor::leaveSpeechSegment(segment);
56 }
