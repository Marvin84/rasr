23 ConstWordLatticeRef normalize(ConstWordLatticeRef lattice) {
24  if (lattice) {
25  WordLattice* normalized = new WordLattice;
26  Fsa::ConstAutomatonRef n;
27  for (size_t i = 0; i < lattice->nParts(); ++i) {
28  n = Fsa::normalize(lattice->part(i));
29  normalized->setFsa(Fsa::staticCopy(n), lattice->name(i));

33  lattice->wordBoundaries(),
34  Fsa::mapNormalized(n));

37  return ConstWordLatticeRef(normalized);

118  virtual void discoverState(Fsa::ConstStateRef sp) {
119  const Speech::TimeframeIndex tbeg = wordBoundaries_->time(sp->id());
120  if (tbeg != Speech::InvalidTimeframeIndex) {
121  minTimeframeIndex_ = std::min(minTimeframeIndex_, tbeg);
122  maxTimeframeIndex_ = std::max(maxTimeframeIndex_, tbeg);
123  }
124  for (Fsa::State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
125  if (a->input() != Fsa::Epsilon) {
126  const Speech::TimeframeIndex tend =
127  wordBoundaries_->time(
128  fsa_->getState(a->target())->id());
129  nArcs_ += tend - tbeg;
130  }
131  }
132  }

134  virtual void finish() {
135  density_ = (nTimeframes() > 0) ? (f32)nArcs() / nTimeframes() : 0;
136  }
