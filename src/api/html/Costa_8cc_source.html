65 void CorpusVisitor::enterRecording(Bliss::Recording* r) {
66  std::string inputFile(r->video());
67  if (inputFile.empty())
68  inputFile = r->audio();
69 
70  recordingsChannel_ << inputFile << "\n";
71  if (extraction_) {
72  extraction_->setParameter("id", r->fullName());
73  extraction_->setParameter("input-file", inputFile);
74 
75  std::istringstream durationAttr(extraction_->getAttribute(dataPort_, "total-duration"));
76  Bliss::Time duration;
77  if (durationAttr >> duration) {
78  r->setDuration(duration);
79  }
80  }
81  Precursor::enterRecording(r);
82 }

112 int Costa::main(const std::vector<std::string>& arguments) {
113  const Core::Configuration statsConfig(config, "statistics");
114  Bliss::LexiconRef lexicon;
115 
116  if (paramLexcialStatistics(config) || paramLmStatistics(config)) {
117  const Core::Configuration lexConfig(statsConfig, "lexicon");
118  lexicon = Bliss::Lexicon::create(lexConfig);
119  if (lexicon) {
120  Core::XmlChannel ch(lexConfig, "dump");
121  if (ch.isOpen())
122  lexicon->writeXml(ch);
123  }
124  else
125  error("failed to initialize lexicon");
126  }
127 
128  const Core::Configuration corpusConfig(statsConfig, "corpus");
129  Bliss::CorpusDescription corpus(corpusConfig);
130 
131  Bliss::CompositeCorpusStatisticsVisitor* stat = 0;
132 
133  stat = new CorpusVisitor(statsConfig);
134  stat->add(new Bliss::CorpusSizeStatisticsVisitor(statsConfig));
135  stat->add(new Bliss::CorpusSpeakerStatisticsVisitor(statsConfig));
136  stat->add(new Bliss::CorpusConditionStatisticsVisitor(statsConfig));
137  if (paramLexcialStatistics(config) && lexicon) {
138  stat->add(new Bliss::CorpusLexicalStatisticsVisitor(statsConfig, lexicon));
139  }
140  if (paramLmStatistics(config) && lexicon) {
141  const Core::Configuration lmConfig(statsConfig, "lm");
142  stat->add(new Lm::CorpusStatisticsVisitor(lmConfig, lexicon));
143  }
144 
145  stat->reset();
146  corpus.accept(stat);
147  stat->writeReport(clog());
148 
149  delete stat;
150 
151  return 0;
152 }
