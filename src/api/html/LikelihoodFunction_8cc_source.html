34 bool IndependentSequenceLikelihood::setClasses(const std::vector<std::string>& classLabels) {
35  if (logLikelihoodFunctions_ == 0)
36  return false;
37  if (logLikelihoodFunctions_->nMixtures() != classLabels.size()) {
38  error("Number of mixtures (%d) does not match to the number of classes (%zd).",
39  logLikelihoodFunctions_->nMixtures(), classLabels.size());
40  return false;
41  }
42  scores_.resize(classLabels.size());
43  reset();
44  return true;

66 void IndependentSequenceLikelihood::feed(const std::vector<Data>& featureVector,
67  Weight featureScoreWeight,
68  ScoreVector* currentScores) {
69  verify_(logLikelihoodFunctions_ != 0);
70 
71  Core::Ref<const Mm::Feature> feature(new Mm::Feature(featureVector));
72  Mm::FeatureScorer::Scorer scorer = logLikelihoodFunctions_->getScorer(feature);
73  size_t nClasses = logLikelihoodFunctions_->nMixtures();
74  if (currentScores)
75  currentScores->resize(nClasses);
76  for (u32 c = 0; c < nClasses; ++c) {
77  Score score = featureScoreWeight * scorer->score(c);
78  if (currentScores)
79  (*currentScores)[c] = score;
80  scores_[c] += score;
81  }
82  Precursor::feed(featureVector, featureScoreWeight);
83 }
