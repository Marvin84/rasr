43 /* Helper functions to realize a UTF-8-like encoding scheme for u32 values.
44  * The best compression rate is achieved if there are many small values
45  * and only some huge values.
46  */
47 void writePackedU32(Core::BinaryOutputStream& out, u32 value) {
48  do {
49  if (value < 128) {
50  out << (u8)value;
51  }
52  else {
53  out << (u8)((value & 127) | 128);
54  }
55  value >>= 7;
56  } while (value);
57 }
58 
59 u32 readPackedU32(Core::BinaryInputStream& in) {
60  u32 result = 0;
61  u8 b;
62  u8 s = 0;
63  do {
64  in >> b;
65  if (b & 128) {
66  result |= ((b & 127) << s);
67  }
68  else {
69  result |= (b << s);
70  }
71  s += 7;
72  } while (b & 128);
73  return result;
74 }

83 /*
84  * Alignments use a simple run-length compression scheme.
85  * (Run-length encoding is a sensible choice because of the HMM
86  * loop-transitions, esp. in silence portions.)
87  * A one byte code n indicated one of the following cases:
88  * n > 0: the next n shorts are mixtures indices
89  * n < 0: the next short should be repeated |n| times
90  * In case time indices are not contiguous, the 0 code is used:
91  * n = 0: the next long is the new time frame index
92  *
93  * Weighted alignments use a different compression scheme:
94  * u32 values like sizes and allophone state indices are compressed
95  * by an utf-8-like encoding. Weights are expected to be inside
96  * the interval [0..1] and quantized by a u16 value.
97  */
98 
99 Core::BinaryOutputStream& Speech::operator<<(Core::BinaryOutputStream& os, const Alignment& a) {
100  if (a.labelType() == Alignment::allophoneStateIds) {
101  os.write(a.alphabet_ ? Alignment::magic_alphabet : Alignment::magic, Alignment::magicSize);
102  }
103  else {
104  os.write(Alignment::magic_emission, Alignment::magicSize);
105  }
106 
107  std::map<Fsa::LabelId, std::string> alphabet;
108 
109  TimeframeIndex time = 0;
110  if (!a.hasWeights()) {
111  // viterbi compression scheme with run-length encoding
112  os << u32(a.size());
113  for (s32 i = 0; os && i < s32(a.size());) {
114  if (time != a[i].time) {
115  os << s8(0) << u32(time = a[i].time);
116  }
117  s32 n;
118 
119  if (a.alphabet_ && !alphabet.count(a[i].emission)) {
120  std::string symbol = a.alphabet_->symbol(a[i].emission);
121  verify(symbol.length());
122  alphabet.insert(std::make_pair(a[i].emission, symbol));
123  }
124 
125  if (i + 1 < s32(a.size()) && a[i].emission == a[i + 1].emission) {
126  for (n = 1; i + n < s32(a.size()); ++n) {
127  if (a[i + n].emission != a[i].emission)
128  break;
129  if (a[i + n].time != a[i].time + n)
130  break;
131  if (-n <= Core::Type<s8>::min)
132  break;
133  }
134  hope(a[i].emission <= Fsa::LastLabelId);
135  os << s8(-n) << u32(a[i].emission);
136  i += n;
137  }
138  else {
139  for (n = 1; i + n < s32(a.size()); ++n) {
140  if (a[i + n].emission == a[i + n - 1].emission)
141  break;
142  if (a[i + n].time != a[i].time + n)
143  break;
144  if (n >= Core::Type<s8>::max)
145  break;
146  }
147  os << s8(n);
148  while (n--) {
149  if (a.alphabet_ && !alphabet.count(a[i].emission)) {
150  std::string symbol = a.alphabet_->symbol(a[i].emission);
151  verify(symbol.length());
152  alphabet.insert(std::make_pair(a[i].emission, symbol));
153  }
154  hope(a[i].emission <= Fsa::LastLabelId);
155  os << u32(a[i++].emission);
156  }
157  }
158  time += n;
159  }
160  }
161  else {
162  // compression scheme for weighted alignments
163  const u32 version = 2;
164  const u32 highestBitU32 = 1 << 31;
165  os << u32(version | highestBitU32); // use highest bit to distinguish this scheme from the RLE scheme
166  writePackedU32(os, a.size());
167  Alignment::const_iterator i, i_end, j;
168  i = j = a.begin();
169  i_end = a.end();
170  TimeframeIndex lastTimePlusOne = 0;
171  while (i != i_end) {
172  TimeframeIndex time = j->time;
173  while (i != i_end && i->time == time)
174  ++i;
175  u32 nItemsCurrentFrame = u32(i - j);
176  if (time == lastTimePlusOne) {
177  writePackedU32(os, nItemsCurrentFrame * 2);
178  }
179  else {
180  writePackedU32(os, nItemsCurrentFrame * 2 + 1);
181  writePackedU32(os, time);
182  }
183  for (; j != i; ++j) {
184  if (a.alphabet_ && !alphabet.count(j->emission)) {
185  std::string symbol = a.alphabet_->symbol(j->emission);
186  verify(symbol.length());
187  alphabet.insert(std::make_pair(j->emission, symbol));
188  }
189 
190  writePackedU32(os, (u32)(j->emission));
191  f32 tmp = j->weight;
192  os << tmp;
193  }
194  lastTimePlusOne = time + 1;
195  }
196  }
197  if (a.alphabet_) {
198  writePackedU32(os, alphabet.size());
199  for (std::map<Fsa::LabelId, std::string>::const_iterator it = alphabet.begin(); it != alphabet.end(); ++it) {
200  writePackedU32(os, it->first);
201  os << it->second;
202  }
203  }
204  return os;
205 }

207 Core::BinaryInputStream& Speech::operator>>(Core::BinaryInputStream& is, Alignment& a) {
208  a.clear();
209 
210  char header[Alignment::magicSize];
211  is.read(header, Alignment::magicSize);
212  bool matchedHeader = strncmp(header, Alignment::magic, Alignment::magicSize) == 0;
213  bool matchedAlphabetHeader = strncmp(header, Alignment::magic_alphabet, Alignment::magicSize) == 0;
214  bool matchedEmissionLabelHeader = strncmp(header, Alignment::magic_emission, Alignment::magicSize) == 0;
215 
216  if (!matchedHeader && !matchedAlphabetHeader && !matchedEmissionLabelHeader) {
217  is.addState(std::ios::badbit);
218  return is;
219  }
220  if (matchedEmissionLabelHeader)
221  a.setLabelType(Alignment::emissionIds);
222  u32 size = 0;
223  is >> size;
224  if (size < ((u32)1 << 31)) {
225  // RLE scheme
226  u32 time = 0;
227  u32 mixture;
228  s8 n;
229  for (; is && a.size() < size;) {
230  is >> n;
231  if (n > 0) {
232  // read N different mixtures
233  while (n--) {
234  is >> mixture;
235  a.push_back(AlignmentItem(time++, mixture));
236  }
237  }
238  else if (n < 0) {
239  // repeat mixture N times
240  is >> mixture;
241  while (n++) {
242  a.push_back(AlignmentItem(time++, mixture));
243  }
244  }
245  else /* if (n == 0) */ {
246  // change time index
247  is >> time;
248  }
249  }
250  }
251  else {
252  // compression scheme for weighted alignments
253  u32 version = size;
254  version &= (((u32)1 << 31) - 1);
255  require(version <= 2);
256  a.resize(readPackedU32(is));
257  TimeframeIndex time = 0;
258  Alignment::iterator i, i_end;
259  i = a.begin();
260  i_end = a.end();
261  while (i != i_end) {
262  u32 nItemsCurrentFrame = readPackedU32(is);
263  if (nItemsCurrentFrame & 1) {
264  time = readPackedU32(is);
265  }
266  nItemsCurrentFrame /= 2;
267  for (; nItemsCurrentFrame; --nItemsCurrentFrame, ++i) {
268  require(i != i_end);
269  i->time = time;
270  i->emission = readPackedU32(is);
271  if (version == 1) {
272  u16 quantizedWeight;
273  is >> quantizedWeight;
274  i->weight = (Mm::Weight)(quantizedWeight) / 65535;
275  }
276  else if (version == 2) {
277  f32 tmp;
278  is >> tmp;
279  i->weight = tmp;
280  }
281  }
282  ++time;
283  }
284  }
285 
286  if (matchedAlphabetHeader) {
287  u32 count = readPackedU32(is);
288  for (u32 i = 0; i < count; ++i) {
289  u32 emission = readPackedU32(is);
290  std::string target;
291  is >> target;
292  a.archiveAlphabet_.insert(std::make_pair(emission, target));
293  }
294  for (Alignment::iterator it = a.begin(); it != a.end(); ++it)
295  verify(a.archiveAlphabet_.count(it->emission));
296  if (a.alphabet_)
297  a.mapAlphabet(false);
298  }
299 
300  return is;
301 }

303 void Alignment::mapAlphabet(bool skipMismatch) {
304  verify(alphabet_);
305  const Am::AllophoneStateAlphabet* alphabet = dynamic_cast<const Am::AllophoneStateAlphabet*>(alphabet_.get());
306  verify(alphabet);
307  std::set<std::string> missing;
308  u32 missingCount = 0;
309  for (Alignment::iterator it = begin(); it != end(); ++it) {
310  std::map<u32, std::string>::const_iterator mapIt = archiveAlphabet_.find(it->emission);
311  verify(mapIt != archiveAlphabet_.end());
312  verify(!mapIt->second.empty());
313  it->emission = alphabet->index(mapIt->second, false);
314  if (it->emission == Fsa::InvalidLabelId) {
315  missing.insert(mapIt->second);
316  ++missingCount;
317  continue;
318  }
319  verify(it->emission != Fsa::InvalidLabelId);
320  verify(alphabet_->symbol(it->emission) == mapIt->second);
321  }
322  archiveAlphabet_.clear();
323  if (missingCount) {
324  {
325  Core::Component::Message msg = Core::Application::us()->warning();
326  msg << "failed mapping " << missingCount << " out of " << size() << " alignment-items\n";
327  msg << "mismatched allophones:";
328  for (std::set<std::string>::const_iterator it = missing.begin(); it != missing.end(); ++it)
329  msg << " " << *it;
330  }
331  if (!skipMismatch) {
332  Core::Application::us()->error("Alignment: mapping of some allophones failed, and skipping is disabled");
333  }
334  Alignment::iterator itOut = begin();
335  for (Alignment::iterator it = begin(); it != end(); ++it) {
336  if (it->emission != Fsa::InvalidLabelId) {
337  *itOut = *it;
338  ++itOut;
339  }
340  }
341  erase(itOut, end());
342  }
343 }

345 bool Alignment::setAlphabet(Core::Ref<const Fsa::Alphabet> alphabet, bool skipMismatch) {
346  if (alphabet_ == alphabet)
347  return false;
348 
349  if (alphabet_) {
350  // Backup the relevant part of the old alphabet, to perform the mapping aftwards
351  verify(archiveAlphabet_.empty());
352  for (Alignment::iterator it = begin(); it != end(); ++it) {
353  std::string symbol = alphabet_->symbol(it->emission);
354  verify(!symbol.empty());
355  archiveAlphabet_.insert(std::make_pair(it->emission, symbol));
356  }
357  }
358 
359  alphabet_ = alphabet;
360  if (archiveAlphabet_.size()) {
361  mapAlphabet(skipMismatch);
362  return true;
363  }
364  else {
365  return false;
366  }
