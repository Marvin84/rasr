141 void FirstBestTraceRecorder::createBestPath(const WordEndDetector& wordEnds, bool ignoreLast, TraceRef end, BestPath* path) {
142  path->clear();
143  // insert trace back item for sentence end
144  const Trace& last = data_[end];
145  path->append(Epsilon, last.time, last.score);
146  std::deque<Label> outputs;
147  std::deque<BestPath::Item> timeAndScore;
148  bool ignoreOutput = ignoreLast;
149  for (TraceRef trace = end; trace != InvalidTraceRef; trace = data_[trace].predecessor) {
150  const Trace& currentTrace = data_[trace];
151  if (wordEnds.isNonWord(currentTrace.input)) {
152  path->append(Epsilon, currentTrace.time, currentTrace.score);
153  }
154  else if (wordEnds.isWordEnd(currentTrace))
155  timeAndScore.push_back(BestPath::Item(Epsilon, currentTrace.time, currentTrace.score));
156  if (currentTrace.output != Epsilon) {
157  if (!ignoreOutput)
158  outputs.push_back(currentTrace.output);
159  else
160  ignoreOutput = false;
161  }
162  if (!outputs.empty() && !timeAndScore.empty()) {
163  BestPath::Item item = timeAndScore.front();
164  item.word = outputs.front();
165  path->append(item);
166  outputs.pop_front();
167  timeAndScore.pop_front();
168  }
169  }
170  std::reverse(path->begin(), path->end());
171  std::sort(path->begin(), path->end(), BestPath::CompareTime());
172 }
