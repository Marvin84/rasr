28 /**
29  * Cache for Fsa object.
30  *
31  * Cache stores FSA objects in a archive.
32  * Use 'find', to get an existing object.
33  * Use 'insert' to add a new object.
34  * Use 'get', to retrieve object which will be created if not existent.
35  *
36  * Remark: This is a simple implementation, once Core::ObjectCache supports
37  * reading of Core::Ref, this implementation could be easly merged with it.
38  */
39 class FsaCache : public Core::Component {
40 private:
41  static const Core::ParameterString paramPath;
42  static const Core::ParameterBool paramReadOnly;
43 
44 private:
45  Core::Archive* archive_;
46  /** If valid, input alphabet of new transducers is overwritten by this one. */
47  Fsa::ConstAlphabetRef inputAlphabet_;
48  /** If valid, output alphabet of new transducers is overwritten by this one. */
49  Fsa::ConstAlphabetRef outputAlphabet_;
50  Fsa::StoredComponents whatToStore_;
51  bool readOnly_;
52 
53 private:
54  Fsa::ConstAutomatonRef read(const std::string& id);
55  void write(const std::string& id, Fsa::ConstAutomatonRef);
56 
57 public:
58  FsaCache(const Core::Configuration&, Fsa::StoredComponents = Fsa::storeAll);
59  virtual ~FsaCache();
60 
61  bool setDependencies(const Core::DependencySet&);
62 
63  /** Input alphabet of read transducer will be overwritten by @param input. */
64  void forceInputAlphabet(Fsa::ConstAlphabetRef input) {
65  inputAlphabet_ = input;
66  }
67  /** Output alphabet of read transducer will be overwritten by @param output. */
68  void forceOutputAlphabet(Fsa::ConstAlphabetRef output) {
69  outputAlphabet_ = output;
70  }
71 
72  Fsa::ConstAutomatonRef find(const std::string& id) {
73  return archive_ ? read(id) : Fsa::ConstAutomatonRef();
74  }
75  void insert(const std::string& id, Fsa::ConstAutomatonRef f) {
76  if (!readOnly_ && archive_)
77  write(id, f);
78  }
79 
80  template<class Builder>

84 template<class Builder>
85 Fsa::ConstAutomatonRef FsaCache::get(Builder builder) {
86  Fsa::ConstAutomatonRef result;
87  const std::string& id(builder.id());
88  if (!(result = find(id))) {
89  log("Building transducer for \"%s\".", id.c_str());
90  insert(id, result = builder.build());
91  }
92  return result;
