118  /*
119  * In case of a valid label id:
120  * Before:
121  * am_score = emission_scale * emission + transition_scale * transition
122  * am_scale = 1.0
123  * lm_score = pronunciation_scale * pronunciation + lm_scale * lm
124  * lm_scale = 1.0
125  * Afterwards:
126  * am_score = emission_scale * emission + transition_scale * transition + pronunciation_scale * pronunciation
127  * am_scale = 1.0
128  * lm_score = lm
129  * lm_scale = lm_scale
130  */
131  ScoresRef buildScore(Fsa::LabelId label, Score amRecogScore, Score lmRecogScore) {
132  verify(amRecogScore != Semiring::Zero);
133  verify(lmRecogScore != Semiring::Zero);
134 
135  Score amScore, pronScore, lmScore;
136  if ((Fsa::FirstLabelId <= label) && (label <= Fsa::LastLabelId)) {
137  amScore = amRecogScore;
138  pronScore = lpAlphabet_->lemmaPronunciation(label)->pronunciationScore();
139  verify(pronScore != Semiring::Zero);
140  lmScore = (lmRecogScore - pronScale_ * pronScore) / lmScale_;
141  }
142  else {
143  amScore = amRecogScore;
144  pronScore = Semiring::One;
145  lmScore = lmRecogScore / lmScale_;
146  }
147 
148  ScoresRef scores = semiring_->create();
149  if (addPronunciationScores_) {
150  scores->set(amId_, amScore);
151  scores->set(pronunciationId_, pronScore);
152  }
153  else
154  scores->set(amId_, amScore + pronScale_ * pronScore);
155  scores->set(lmId_, lmScore);
156  if (addConfidenceScores_)
157  scores->set(confidenceId_, 0.0);
158  verify(semiring_->project(scores) != Semiring::Zero);
159 
160  return scores;
161  }

163  ConstLatticeRef buildLattice(Core::Ref<const Search::LatticeAdaptor> la, bool zeroStartTime) {
164  Flf::LatticeHandler* handler = Module::instance().createLatticeHandler(config);
165  handler->setLexicon(Lexicon::us());
166  if (la->empty())
167  return ConstLatticeRef();
168  ::Lattice::ConstWordLatticeRef lattice = la->wordLattice(handler);
169  Core::Ref<const ::Lattice::WordBoundaries> boundaries = lattice->wordBoundaries();
170  Fsa::ConstAutomatonRef amFsa = lattice->part(::Lattice::WordLattice::acousticFsa);
171  Fsa::ConstAutomatonRef lmFsa = lattice->part(::Lattice::WordLattice::lmFsa);
172  require_(Fsa::isAcyclic(amFsa) && Fsa::isAcyclic(lmFsa));
173 
174  StaticBoundariesRef b = StaticBoundariesRef(new StaticBoundaries);
175  StaticLatticeRef s = StaticLatticeRef(new StaticLattice);
176  s->setType(Fsa::TypeAcceptor);
177  s->setProperties(Fsa::PropertyAcyclic | PropertyCrossWord, Fsa::PropertyAll);
178  s->setInputAlphabet(lpAlphabet_);
179  s->setSemiring(semiring_);
180  s->setDescription(Core::form("recog(%s)", segment_->name().c_str()));
181  s->setBoundaries(ConstBoundariesRef(b));
182  s->setInitialStateId(0);
183 
184  Time timeOffset = zeroStartTime ? (*boundaries)[amFsa->initialStateId()].time() : 0;
185 
186  Fsa::Stack<Fsa::StateId> S;
187  Core::Vector<Fsa::StateId> sidMap(amFsa->initialStateId() + 1, Fsa::InvalidStateId);
188  sidMap[amFsa->initialStateId()] = 0;
189  S.push_back(amFsa->initialStateId());
190  Fsa::StateId nextSid = 2;
191  Time finalTime = 0;
192  while (!S.isEmpty()) {
193  Fsa::StateId sid = S.pop();
194  verify(sid < sidMap.size());
195  const ::Lattice::WordBoundary& boundary((*boundaries)[sid]);
196  Fsa::ConstStateRef amSr = amFsa->getState(sid);
197  Fsa::ConstStateRef lmSr = lmFsa->getState(sid);
198  State* sp = new State(sidMap[sid]);
199  s->setState(sp);
200  b->set(sp->id(), Boundary(boundary.time() - timeOffset,
201  Boundary::Transit(boundary.transit().final, boundary.transit().initial)));
202  if (amSr->isFinal()) {
203  sp->newArc(1, buildScore(Fsa::InvalidLabelId, amSr->weight(), lmSr->weight()), sentenceEndLabel_);
204  finalTime = std::max(finalTime, boundary.time() - timeOffset);
205  }
206  for (Fsa::State::const_iterator am_a = amSr->begin(), lm_a = lmSr->begin(); (am_a != amSr->end()) && (lm_a != lmSr->end()); ++am_a, ++lm_a) {
207  sidMap.grow(am_a->target(), Fsa::InvalidStateId);
208  if (sidMap[am_a->target()] == Fsa::InvalidStateId) {
209  sidMap[am_a->target()] = nextSid++;
210  S.push(am_a->target());
211  }
212  Fsa::ConstStateRef targetAmSr = amFsa->getState(am_a->target());
213  Fsa::ConstStateRef targetLmSr = amFsa->getState(lm_a->target());
214  if (targetAmSr->isFinal() && targetLmSr->isFinal()) {
215  if (am_a->input() == Fsa::Epsilon) {
216  ScoresRef scores = buildScore(am_a->input(), am_a->weight(), lm_a->weight());
217  scores->add(amId_, Score(targetAmSr->weight()));
218  scores->add(lmId_, Score(targetLmSr->weight()) / lmScale_);
219  sp->newArc(1, scores, sentenceEndLabel_);
220  }
221  else
222  sp->newArc(sidMap[am_a->target()], buildScore(am_a->input(), am_a->weight(), lm_a->weight()), am_a->input());
223  }
224  else
225  sp->newArc(sidMap[am_a->target()], buildScore(am_a->input(), am_a->weight(), lm_a->weight()), am_a->input());
226  }
227  }
228  State* sp = new State(1);
229  sp->setFinal(semiring_->clone(semiring_->one()));
230  s->setState(sp);
231  b->set(sp->id(), Boundary(finalTime));
232  ConstLatticeRef l = ConstLatticeRef(s);
233  if (applyNonWordClosureFilter_) {
234  l = transducer(l);
235  l = applyOneToOneLabelMap(l, nonWordToEpsilonMap_);
236  StaticLatticeRef filteredLattice = applyEpsClosureWeakDeterminizationFilter(l);
237  trimInPlace(filteredLattice);
238  l = projectOutput(filteredLattice);
239  }
240  if (addConfidenceScores_ || applyPosteriorPruning_) {
241  std::pair<ConstLatticeRef, ConstFwdBwdRef> latAndFb = FwdBwd::build(l, posteriorSemiring_);
242  l = latAndFb.first;
243  ConstFwdBwdRef fb = latAndFb.second;
244  if (addConfidenceScores_) {
245  ConstPosteriorCnRef cn = buildFramePosteriorCn(l, fb);
246  l = extendByFCnConfidence(l, cn, confidenceId_, RescoreModeInPlaceCache);
247  l = persistent(l);
248  }
249  if (applyPosteriorPruning_) {
250  l = pruneByFwdBwdScores(l, fb, fwdBwdThreshold_);
251  StaticLatticeRef trimmedLattice = StaticLatticeRef(new StaticLattice);
252  copy(l, trimmedLattice.get(), 0);
253  trimInPlace(trimmedLattice);
254  trimmedLattice->setBoundaries(l->getBoundaries());
255  l = normalizeCopy(trimmedLattice);
256  }
257  }
258  return l;
259  }

261  void logTraceback(const Search::SearchAlgorithm::Traceback& traceback) {
262  tracebackChannel_ << Core::XmlOpen("traceback") + Core::XmlAttribute("type", "xml");
263  u32 previousIndex = traceback.begin()->time;
264  Search::SearchAlgorithm::ScoreVector previousScore(0.0, 0.0);
265  for (std::vector<Search::SearchAlgorithm::TracebackItem>::const_iterator tbi = traceback.begin(); tbi != traceback.end(); ++tbi) {
266  if (tbi->pronunciation) {
267  tracebackChannel_ << Core::XmlOpen("item") + Core::XmlAttribute("type", "pronunciation")
268  << Core::XmlFull("orth", tbi->pronunciation->lemma()->preferredOrthographicForm())
269  << Core::XmlFull("phon", tbi->pronunciation->pronunciation()->format(lexicon_->phonemeInventory()))
270  << Core::XmlFull("score", f32(tbi->score.acoustic - previousScore.acoustic)) + Core::XmlAttribute("type", "acoustic")
271  << Core::XmlFull("score", f32(tbi->score.lm - previousScore.lm)) + Core::XmlAttribute("type", "language");
272  if (previousIndex < tbi->time)
273  tracebackChannel_ << Core::XmlEmpty("samples") + Core::XmlAttribute("start", f32(featureTimes_[previousIndex].startTime())) +
274  Core::XmlAttribute("end", f32(featureTimes_[tbi->time - 1].endTime()))
275  << Core::XmlEmpty("features") + Core::XmlAttribute("start", previousIndex) +
276  Core::XmlAttribute("end", tbi->time - 1);
277  tracebackChannel_ << Core::XmlClose("item");
278  }
279  previousScore = tbi->score;
280  previousIndex = tbi->time;
281  }
282  tracebackChannel_ << Core::XmlClose("traceback");
283  }

296  featureExtractor_ = SegmentwiseFeatureExtractorRef(new SegmentwiseFeatureExtractor(featureExtractionConfig, dataSource));
297 
298  require(mc_);

301  addPronunciationScores_ = paramPronunciationScore(config);
302  addConfidenceScores_ = paramConfidenceScore(config);
303  applyNonWordClosureFilter_ = paramApplyNonWordClosureFilter(config);
304  applyPosteriorPruning_ = paramApplyPosteriorPruning(config);
305  allowSkips_ = paramAllowSkip(config);
306  {
307  Core::Component::Message msg(log());
308  lpAlphabet_ = mc_->lexicon()->lemmaPronunciationAlphabet();
309  sentenceEndLabel_ = Fsa::Epsilon;
310  const Bliss::Lemma* special = mc_->lexicon()->specialLemma("sentence-end");
311  if (special) {
312  Bliss::Lemma::LemmaPronunciationRange lpRange = special->pronunciations();
313  if (lpRange.first != lpRange.second)
314  sentenceEndLabel_ = lpRange.first->id();
315  }
316  msg << "Sentence end symbol is \"" << lpAlphabet_->symbol(sentenceEndLabel_) << "\".\n";
317 
318  u32 dim = 0;
319  amId_ = dim++;
320  lmId_ = dim++;
321  if (addPronunciationScores_)
322  pronunciationId_ = dim++;
323  else
324  pronunciationId_ = Semiring::InvalidId;
325  if (addConfidenceScores_)
326  confidenceId_ = dim++;
327  else
328  confidenceId_ = Semiring::InvalidId;
329  semiring_ = Semiring::create(Fsa::SemiringTypeTropical, dim);
330  semiring_->setKey(amId_, "am");
331  semiring_->setScale(amId_, 1.0);
332  semiring_->setKey(lmId_, "lm");
333  semiring_->setScale(lmId_, lmScale_);
334  if (addPronunciationScores_) {
335  semiring_->setKey(pronunciationId_, "pronunciation");
336  semiring_->setScale(pronunciationId_, pronScale_);
337  }
338  if (addConfidenceScores_) {
339  semiring_->setKey(confidenceId_, "confidence");
340  semiring_->setScale(confidenceId_, 0.0);
341  }
342  msg << "Semiring is " << semiring_->name() << ".\n";
343  if (addConfidenceScores_ || applyPosteriorPruning_) {
344  posteriorSemiring_ = toLogSemiring(semiring_, paramAlpha(select("fb")));
345  msg << "Posterior-semiring is " << posteriorSemiring_->name() << ".\n";
346  }
347  if (applyNonWordClosureFilter_) {
348  nonWordToEpsilonMap_ = LabelMap::createNonWordToEpsilonMap(Lexicon::LemmaPronunciationAlphabetId);
349  msg << "Non-word-closure filter is active.\n";
350  }
351  if (addConfidenceScores_) {
352  msg << "Confidence score calculation is active (Attention: Confidence scores are calculated on lemma pronunciations). \n";
353  }
354  if (applyPosteriorPruning_) {
355  fwdBwdThreshold_ = paramThreshold(select("posterior-pruning"));
356  msg << "Posterior pruning is active (threshold=" << fwdBwdThreshold_ << ").\n";
357  }
358  }
359  initializeRecognizer(*mc_);
360  delayedRecognition_.reset(new Speech::RecognizerDelayHandler(recognizer_, acousticModel_));
361  verify(recognizer_);
362  }

418  std::pair<ConstLatticeRef, ConstSegmentRef> buildLatticeAndSegment(Core::Ref<const Search::LatticeAdaptor> la) {
419  verify(segment_);
420 
421  Speech::TimeframeIndex startTime;
422  {
423  Flf::LatticeHandler* handler = Module::instance().createLatticeHandler(config);
424  ::Lattice::ConstWordLatticeRef lattice = la->wordLattice(handler);
425  Core::Ref<const ::Lattice::WordBoundaries> boundaries = lattice->wordBoundaries();
426  Fsa::ConstAutomatonRef amFsa = lattice->part(::Lattice::WordLattice::acousticFsa);
427  startTime = (*boundaries)[amFsa->initialStateId()].time();
428  }
429 
430  ConstLatticeRef partialLattice = buildLattice(la, true);
431 
432  Fsa::StateId endState = partialLattice->initialStateId();
433  while (partialLattice->getState(endState)->nArcs())
434  endState = partialLattice->getState(endState)->getArc(0)->target();
435  Speech::TimeframeIndex endTime = partialLattice->boundary(endState).time() + startTime;
436 
437  log() << "got partial lattice for interval " << startTime << " -> " << endTime;
438 
439  verify(startTime < featureTimes_.size());
440  if (endTime >= featureTimes_.size()) {
441  log() << "end-time is too high: " << endTime << " max. " << featureTimes_.size() - 1 << ", truncated!";
442  endTime = featureTimes_.size() - 1;
443  }
444 
445  verify(startTime < endTime);
446  verify(endTime < featureTimes_.size());
447 
448  SegmentRef newSegment(new Flf::Segment(segment_));
449  newSegment->setOrthography("");
450  newSegment->setStartTime(featureTimes_[startTime].startTime());
451  newSegment->setEndTime(featureTimes_[endTime].endTime());
452  verify(newSegment->segmentId().size());
453  {
454  std::string::size_type timeStart = newSegment->segmentId().rfind("_");
455  std::string::size_type timeGap = newSegment->segmentId().rfind("-");
456  std::ostringstream os;
457  if (timeStart != std::string::npos && timeGap != std::string::npos && timeGap > timeStart) {
458  // Create a new segment name with corrected time information in the identifier
459  os << newSegment->segmentId().substr(0, timeStart + 1);
460  os << std::setiosflags(std::ios::fixed) << std::setprecision(3) << newSegment->startTime() << "-" << newSegment->endTime();
461  }
462  else {
463  // Create a new segment name by appending "_$subsegment"
464  os << newSegment->segmentId() << "_" << subSegment_;
465  }
466  newSegment->setSegmentId(os.str());
467  }
468  log() << "created segment " << newSegment->segmentId();
469  info(partialLattice, clog());
470  subSegment_ += 1;
471  return std::make_pair(partialLattice, newSegment);
472  }

478  std::pair<ConstLatticeRef, ConstSegmentRef> recognize() {
479  auto timer_start = std::chrono::steady_clock::now();
480  if (!segment_)
481  return std::pair<ConstLatticeRef, ConstSegmentRef>(ConstLatticeRef(), ConstSegmentRef());
482 
483  FeatureRef feature;
484  while (dataSource_->getData(feature)) {
485  putFeature(feature);
486  Core::Ref<const Search::LatticeAdaptor> la = recognizer_->getPartialWordLattice();
487  if (la)
488  return buildLatticeAndSegment(la);
489  }
490 
491  while (delayedRecognition_->flush())
492  ;
493 
494  std::pair<ConstLatticeRef, ConstSegmentRef> ret;
495 
496  if (subSegment_) {
497  ret = buildLatticeAndSegment(recognizer_->getCurrentWordLattice());
498  }
499  else {
500  ret = std::make_pair(buildLattice(recognizer_->getCurrentWordLattice(), false), SegmentRef(new Flf::Segment(segment_)));
501  info(ret.first, clog());
502  processResult();
503  }
504 
505  auto timer_end = std::chrono::steady_clock::now();
506  double duration = std::chrono::duration<double, std::milli>(timer_end - timer_start).count();
507  double signal_duration = (featureTimes_.back().endTime() - featureTimes_.front().startTime()) * 1000.; // convert duration to ms
508  clog() << Core::XmlOpen("flf-recognizer-time") + Core::XmlAttribute("unit", "milliseconds") << duration << Core::XmlClose("flf-recognizer-time");
509  clog() << Core::XmlOpen("flf-recognizer-rtf") << (duration / signal_duration) << Core::XmlClose("flf-recognizer-rtf");
510 
511  finishRecognition();
512  return ret;
513  }

515  void finishRecognition() {
516  dataSource_->finalize();

518  modelAdaptor_->leaveSegment(segment_);

524  delayedRecognition_->reset();

593  Fsa::ConstAutomatonRef buildFsaGrammar(ConstLatticeRef l) {
594  if (!l || (l->initialStateId() == Fsa::InvalidStateId))
595  return Fsa::ConstAutomatonRef();
596 
597  grammarChannel_ << Core::XmlOpen("grammar");
598 
599  /*
600  * the initial grammar
601  */
602  LabelMapRef nonWordToEpsilonMap = LabelMap::createNonWordToEpsilonMap(Lexicon::us()->alphabetId(l->getInputAlphabet()));
603  LatticeCounts bestCounts = count(applyOneToOneLabelMap(best(l), nonWordToEpsilonMap));
604  u32 nBestWords = bestCounts.nArcs_ - bestCounts.nIEps_;
605  LatticeCounts latticeCounts = count(l);
606  f32 latticeDensity = (nBestWords == 0) ? f32(0) : f32(latticeCounts.nArcs_) / f32(nBestWords);
607  grammarChannel_ << Core::XmlOpen("lattice");
608  info(l, grammarChannel_);
609  grammarChannel_ << Core::XmlFull("density", latticeDensity);
610  grammarChannel_ << Core::XmlClose("lattice");
611 
612  f32 desiredDensity = std::min(latticeDensity, grammarDensity_);
613  if ((desiredDensity < latticeDensity) || (latticeCounts.nArcs_ > grammarMaxArcs_)) {
614  StaticLatticeRef
615  tmpL = StaticLatticeRef(new StaticLattice),
616  prunedL = StaticLatticeRef(new StaticLattice);
617  std::pair<ConstLatticeRef, ConstFwdBwdRef> fbResult = FwdBwd::build(l, toLogSemiring(l->semiring()));
618  l = fbResult.first;
619  Score threshold = grammarPruningStepWidth_;
620  LatticeCounts prunedCounts;
621  prunedCounts.nArcs_ = 0;
622  f32 prunedDensity = 0.0;
623  for (;;) {
624  persistent(pruneByFwdBwdScores(l, fbResult.second, threshold), tmpL.get(), 0);
625  trimInPlace(tmpL);
626  LatticeCounts tmpCounts;
627  f32 tmpDensity;
628  if (tmpL->initialStateId() == Fsa::InvalidStateId) {
629  tmpCounts.nArcs_ = 0;
630  tmpDensity = 0.0;
631  }
632  else {
633  tmpCounts = count(tmpL);
634  tmpDensity = (nBestWords == 0) ? f32(0) : f32(tmpCounts.nArcs_) / f32(nBestWords);
635  }
636  grammarChannel_ << Core::XmlFull(
637  "dynamic-pruning", Core::form("threshold = %f, #arcs = %d, density = %f", threshold, s32(tmpCounts.nArcs_), tmpDensity));
638  if (tmpCounts.nArcs_ >= grammarMaxArcs_) {
639  if ((prunedCounts.nArcs_ == 0) || ((tmpCounts.nArcs_ - grammarMaxArcs_) < (grammarMaxArcs_ - prunedCounts.nArcs_))) {
640  prunedCounts = tmpCounts;
641  prunedDensity = tmpDensity;
642  prunedL = tmpL;
643  }
644  break;
645  }
646  prunedCounts = tmpCounts;
647  prunedDensity = tmpDensity;
648  StaticLatticeRef tmp = prunedL;
649  prunedL = tmpL;
650  tmpL = tmp;
651  tmpL->clear();
652  if (tmpDensity >= desiredDensity)
653  break;
654  threshold += grammarPruningStepWidth_;
655  }
656  grammarChannel_ << Core::XmlFull(
657  "dynamic-pruning", Core::form("final threshold = %f, #arcs = %d, density = %f", threshold, s32(prunedCounts.nArcs_), prunedDensity));
658  l = prunedL;
659  latticeCounts = prunedCounts;
660  latticeDensity = prunedDensity;
661  grammarChannel_ << Core::XmlOpen("lattice-after-dynamic-pruning");
662  info(l, grammarChannel_);
663  grammarChannel_ << Core::XmlFull("density", latticeDensity);
664  grammarChannel_ << Core::XmlClose("lattice-after-dynamic-pruning");
665  }
666 
667  /*
668  * make epsilon-free syntactic token acceptor
669  */
670  l = projectInput(mapInput(l, MapToSyntacticTokenSequence));
671  l = fastRemoveEpsilons(l);
672  l = persistent(l);
673  latticeCounts = count(l);
674  latticeDensity = (nBestWords == 0) ? f32(0) : f32(latticeCounts.nArcs_) / f32(nBestWords);
675  grammarChannel_ << Core::XmlOpen("lattice-after-epsilon-removal");
676  info(l, grammarChannel_);
677  grammarChannel_ << Core::XmlFull("density", latticeDensity);
678  grammarChannel_ << Core::XmlClose("lattice-after-epsilon-removal");
679 
680  /*
681  * to fsa, determinization or minimization
682  */
683  Core::Ref<Fsa::StaticAutomaton> g;
684  if (!grammarKey_.empty()) {
685  g = Fsa::staticCopy(toFsa(l, l->semiring()->id(grammarKey_)));
686  }
687  else {
688  g = Fsa::staticCopy(toFsa(l));
689  }
690  // minimize
691  Core::Ref<Fsa::StaticAutomaton> gT;
692  gT = Fsa::staticCopy(Fsa::transpose(g));
693  gT->addProperties(Fsa::PropertyAcyclic); // just to make sure
694  gT = Fsa::staticCopy(Fsa::determinize(gT));
695  Fsa::trimInPlace(gT);
696  g = Fsa::staticCopy(Fsa::transpose(gT));
697  g->addProperties(Fsa::PropertyAcyclic); // just to make sure
698  g = Fsa::staticCopy(Fsa::determinize(g));
699  Fsa::trimInPlace(g);
700  grammarChannel_ << Core::XmlOpen("fsa-after-minimization");
701  Fsa::info(g, grammarChannel_);
702  grammarChannel_ << Core::XmlClose("fsa-after-minimization");
703 
704  grammarChannel_ << Core::XmlClose("grammar");
705 
706  if (!g || (g->initialStateId() == Fsa::InvalidStateId))
707  return Fsa::ConstAutomatonRef();
708  return g;
709  }
