73 template<typename T>
74 T L1Regularizer<T>::objectiveFunction(NeuralNetwork<T>& network, T factor) {
75  T objectiveFunction = 0;
76  for (u32 layer = 0; layer < network.nLayers(); layer++) {
77  if ((network.getLayer(layer).isTrainable()) &&
78  (network.getLayer(layer).regularizationConstant() > 0)) {
79  T tmpObjectiveFunction = 0;
80 
81  NnVector* bias = network.getLayer(layer).getBias();
82  require(bias);
83  tmpObjectiveFunction = bias->l1norm();
84  for (u32 stream = 0; stream < network.getLayer(layer).nInputActivations(); stream++) {
85  NnMatrix* matrix = network.getLayer(layer).getWeights(stream);
86  require(matrix);
87  tmpObjectiveFunction += matrix->l1norm();
88  }
89  tmpObjectiveFunction *= network.getLayer(layer).regularizationConstant();
90  objectiveFunction += tmpObjectiveFunction;
91  }
92  }
93  return factor * objectiveFunction;
94 }

185  if ((network.getLayer(layer).isTrainable()) && (network.getLayer(layer).regularizationConstant() > 0)) {
186  T tmpObjectiveFunction = 0;
187 
188  NnVector* bias = network.getLayer(layer).getBias();
189  NnVector* centerBias = centerNetwork_.getLayer(layer).getBias();
190  require(bias);

194  diffVector_.copy(*bias);

199  NnMatrix* centerWeightMatrix = centerNetwork_.getLayer(layer).getWeights(stream);
