132 template<class T>
133 bool DataSource::getData(Flow::PortId portId, Flow::DataPtr<T>& d) {
134  Flow::DataPtr<Flow::Data> out;
135  if (Precursor::Precursor::getData(portId, out))
136  updateProgressStatus(portId, out);
