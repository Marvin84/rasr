50 u32 BinaryTree::checkTree(const Node& node) {
51  checkNode(node);
52  if (node.isLeaf()) {
53  ++leafN_;
54  return 1;
55  }
56  else {
57  ++internalNodeN_;
58  return 1 + std::max(checkTree(node.leftChild()),
59  checkTree(node.rightChild()));
