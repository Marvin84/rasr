123  void uncompress() {
124  Core::Vector<u8>::const_iterator tmp = container_.substates_.begin() + current_;
125  u8 partition = *tmp;
126  if (partition != 0xff) {
127  ++tmp;
128  if (partition & 0x80)
129  substate_.predecessor_ = Fsa::getBytesAndIncrement(tmp, sizeof(Cursor));
130  else
131  substate_.predecessor_ = NoPredecessor;
132  Fsa::StateId state = Fsa::getBytesAndIncrement(tmp, ((partition >> 5) & 0x03) + 1);
133  substate_.state_ = state >> 1;
134  substate_.disconnect_ = (state & 0x1);
135  substate_.output_ = Fsa::getBytesAndIncrement(tmp, (partition >> 2) & 0x07);
136  switch (partition & 0x03) {
137  case 0x00:
138  substate_.weight_ = container_.semiring_->one();
139  break;
140  case 0x01:
141  substate_.weight_ = container_.semiring_->uncompress(tmp);
142  break;
143  case 0x02:
144  break;
145  }
146  }
147  skip_ = tmp - container_.substates_.begin() - current_;
148  }

191  Cursor addPredecessor(Cursor p1, Cursor p2) {
192  //if (p2 == None) return p1; // does not occur
193  if (p1 == NoPredecessor)
194  return p2;
195  if (!(p1 & Overflow)) {
196  predecessors_.push_back(Core::Vector<Cursor>(1, p1));
197  p1 = predecessors_.size() - 1;
198  }
199  if (!(p2 & Overflow))
200  predecessors_[p1].push_back(p2);
201  else

203  predecessors_[p2 & ~Overflow].end());
204  return (p1 | Overflow);
205  }

207  Cursor size(u8 partition) const {
208  if (partition == 0xff)
209  return 0;
210  return (partition & 0x80 ? sizeof(Cursor) : 0) + ((partition >> 5) & 0x03) + 1 +
211  ((partition >> 2) & 0x07) + (partition & 0x01 ? semiring_->compressedSize() : 0);
212  }
213  Fsa::StateId state(Cursor pos) const {
214  Core::Vector<u8>::const_iterator tmp = substates_.begin() + pos;
215  u8 partition = *(tmp++);
216  if (partition & 0x80)
217  tmp += sizeof(Cursor);
218  return Fsa::getBytesAndIncrement(tmp, ((partition >> 5) & 0x03) + 1) >> 1;
219  }
220  Cursor predecessor(Cursor pos) const {
221  Core::Vector<u8>::const_iterator tmp = substates_.begin() + pos;
222  u8 partition = *(tmp++);
223  Cursor predecessor = NoPredecessor;
224  if (partition & 0x80)
225  predecessor = Fsa::getBytesAndIncrement(tmp, sizeof(Cursor));

228  u32 hash(Cursor pos) const {
229  u32 value = 0;
230  for (Core::Vector<u8>::const_iterator i = substates_.begin() + pos + sizeof(Cursor); *i != 0xff;) {
231  Core::Vector<u8>::const_iterator end = i + 1 + size(*i);
232  for (; i != end; ++i)
233  value = 337 * value + *i;
234  }
235  return value;
236  }
237  bool equal(Cursor pos1, Cursor pos2) const {
238  /*! \todo compare without the user state tags = cycle indicators */
239  Core::Vector<u8>::const_iterator i1 = substates_.begin() + pos1 + sizeof(Cursor);
240  Core::Vector<u8>::const_iterator i2 = substates_.begin() + pos2 + sizeof(Cursor);
241  for (; *i1 != 0xff;) {
242  u8 partition = *(i1++);
243  if (partition != *(i2))
244  return false;
245  require(*i2 != 0xff);
246  ++i2;
247  Cursor len = size(partition);
248  if (memcmp(&*i1, &*i2, len) != 0)
249  return false;
250  i1 += len;
251  i2 += len;
252  }
253  if (*i2 != 0xff)
254  return false;
255  return true;
256  }
257  std::pair<Cursor, bool> insert(const Core::Vector<Substate>& v) {
258  Cursor start = substates_.size();
259  Fsa::appendBytes(substates_, 0, sizeof(Cursor));
260  Weight previous = semiring_->one();
261  for (typename Core::Vector<Substate>::const_iterator s = v.begin(); s != v.end(); ++s) {
262  u8 partition = (s->predecessor_ != NoPredecessor ? 0x80 : 0x00);
263  Fsa::StateId state = (s->state_ << 1) | (s->disconnect_ ? 1 : 0);
264  partition |= (Fsa::estimateBytes(state) - 1) << 5;
265  if (s->output_ > 0)
266  partition |= Fsa::estimateBytes(s->output_) << 2;
267  if (previous == s->weight_)
268  partition |= 0x02;
269  else if (!semiring_->isDefault(s->weight_))
270  partition |= 0x01;
271  substates_.push_back(partition);
272  if (partition & 0x80)
273  Fsa::appendBytes(substates_, s->predecessor_, sizeof(Cursor));
274  Fsa::appendBytes(substates_, state, ((partition >> 5) & 0x03) + 1);
275  Fsa::appendBytes(substates_, s->output_, (partition >> 2) & 0x07);
276  if (partition & 0x01)
277  semiring_->compress(substates_, s->weight_);
278  previous = s->weight_;
279  }
280  substates_.push_back(0xff);
281 
282  // get hash value
283  u32 key = hash(start);
284  Cursor i = bins_[key % bins_.size()];
285  for (; (i != Core::Type<Cursor>::max) && (!equal(start, i));
286  i = Cursor(Fsa::getBytes(substates_.begin() + i, sizeof(Cursor))))
287  ;
288 
289  // if existing: resize vector, else: add to hash table
290  if (i != Core::Type<Cursor>::max) {
291  substates_.resize(start);
292  return std::make_pair(i, true);
293  }
294 
295  // resize hash table on demand
296  if (nSubstates_ > 2 * bins_.size()) {
297  std::fill(bins_.begin(), bins_.end(), Core::Type<Cursor>::max);
298  bins_.grow(2 * bins_.size() - 1, Core::Type<Cursor>::max);
299  for (Cursor i = 0; i != start; ++i) {
300  u32 key = hash(i) % bins_.size();
301  Fsa::setBytes(substates_.begin() + i, bins_[key], sizeof(Cursor));
302  bins_[key] = i;
303  for (i += sizeof(Cursor); substates_[i] != 0xff; i += 1 + size(substates_[i]))
304  ;
305  }
306  }
307  Fsa::setBytes(substates_.begin() + start, bins_[key % bins_.size()], sizeof(Cursor));
308  bins_[key % bins_.size()] = start;
309  ++nSubstates_;
310  return std::make_pair(start, false);
311  }

357  bool operator()(const State_& s1, const State_& s2) const {
358  if (s1.output_ != s2.output_)
359  return false;
360  else if (s1.subset_ != s2.subset_) {
361  typename Substates::const_iterator sub1 = d_.substates_.begin(s1.subset_);
362  typename Substates::const_iterator sub2 = d_.substates_.begin(s2.subset_);
363  for (; sub1.valid() && sub2.valid(); ++sub1, ++sub2) {
364  if (sub1->state_ != sub2->state_)
365  return false;
366  if (d_.semiring_->compare(sub1->weight_, sub2->weight_) != 0)
367  return false;
368  if (sub1->output_ != sub2->output_)
369  return false;
370  }
371  if (sub1.valid() != sub2.valid())
372  return false;
373  }
374  return true;

440  void subsetMarkSiblingsWithAmbiguousCycles() const {
441  // we follow the predecessor links until we find the same original state (= cycle)
442  for (typename Core::Vector<Substate>::iterator sub = tmpSubstates_.begin(); sub != tmpSubstates_.end(); ++sub)
443  if ((*outputs_.begin(sub->output_) != Fsa::InvalidLabelId) || (!semiring_->isDefault(sub->weight_))) {
444  Fsa::Stack<typename Substate::Cursor> S;
445  if (sub->predecessor_ & Substates::Overflow) {
446  for (typename Substates::const_predecessor_iterator p = substates_.predecessorBegin(sub->predecessor_);
447  p != substates_.predecessorEnd(sub->predecessor_); ++p)
448  S.push(*p);
449  }
450  else if (sub->predecessor_ != Substates::NoPredecessor)
451  S.push(sub->predecessor_);
452  while (!S.empty()) {
453  typename Substate::Cursor substate = S.pop();
454  if (substates_.state(substate) == sub->state_) {
455  sub->disconnect_ = true;
456  break;
457  }
458  else {
459  substate = substates_.predecessor(substate);
460  if (substate & Substates::Overflow)
461  for (typename Substates::const_predecessor_iterator p = substates_.predecessorBegin(substate);
462  p != substates_.predecessorEnd(substate); ++p)
463  S.push(*p);
464  else if (substate != Substates::NoPredecessor)
465  S.push(substate);
466  }
467  }
468  }
469  }
470  typename Substates::Cursor subsetStop() const {
471  size_t nExternalPredecessors = 0;
472  if (disambiguate_)
473  nExternalPredecessors = substates_.nExternalPredecessors();
474  std::sort(tmpSubstates_.begin(), tmpSubstates_.end());
475  typename Core::Vector<Substate>::iterator j = tmpSubstates_.begin();
476  for (typename Core::Vector<Substate>::const_iterator i = tmpSubstates_.begin() + 1; i != tmpSubstates_.end(); ++i) {
477  if ((j->state_ == i->state_) && (j->output_ == i->output_)) {
478  j->weight_ = semiring()->collect(j->weight_, i->weight_);
479  if (disambiguate_)
480  j->predecessor_ = substates_.addPredecessor(j->predecessor_, i->predecessor_);
481  }
482  else if (++j < i)
483  *j = *i;
484  }
485  tmpSubstates_.resize(j - tmpSubstates_.begin() + 1);
486  if (disambiguate_)
487  subsetMarkSiblingsWithAmbiguousCycles();
488  std::pair<typename Substates::Cursor, bool> id = substates_.insert(tmpSubstates_);
489  if (disambiguate_) {
490  if (id.second) {
491  for (typename Core::Vector<Substate>::const_iterator i = tmpSubstates_.begin() + 1; i != tmpSubstates_.end(); ++i)
492  j->predecessor_ = substates_.addPredecessor(j->predecessor_, i->predecessor_);
493  substates_.resizeExternalPrecessors(nExternalPredecessors);
494  }
495  }
496  return id.first;
497  }

522  virtual _ConstStateRef getState(Fsa::StateId s) const {
523  if (s < states_.size()) {
524  const State_* state = &states_[s];
525  const Fsa::LabelIdStrings::Id output = state->output_;
526  _State* sp = new _State(s);
527 
528  if (*outputs_.begin(output) == Fsa::InvalidLabelId) { // zero length string
529  Fsa::StateTag tags = Fsa::StateTagNone;
530  Weight fw = semiring_->zero();
531 
532  // extend weights and gather arcs
533  tmpArcs_.resize(0);
534  tmpOutputs_.resize(0);
535  typename Substates::Cursor predecessor = Substates::NoPredecessor;
536  for (typename Substates::const_iterator sub = substates_.begin(state->subset_); sub.valid(); ++sub)
537  if (!sub->disconnect_) {
538  _ConstStateRef _sp = fsa_->getState(sub->state_);
539  const Fsa::LabelIdStrings::Id output = sub->output_;
540  for (typename _State::const_iterator a = _sp->begin(); a != _sp->end(); ++a) {
541  if (disambiguate_)
542  predecessor = sub.cursor();
543  tmpArcs_.push_back(Arc_(predecessor, a->input(), a->target(),
544  semiring_->extend(sub->weight_, a->weight()), tmpOutputs_.size()));
545  for (Core::Vector<Fsa::LabelId>::const_iterator i = outputs_.begin(output);
546  *i != Fsa::InvalidLabelId; ++i)
547  tmpOutputs_.push_back(*i);
548  if (a->output() != Fsa::Epsilon)
549  tmpOutputs_.push_back(a->output());
550  tmpOutputs_.push_back(Fsa::InvalidLabelId);
551  }
552  // subsequential final strings at final states (yet not determinized on output side)
553  if (_sp->isFinal() && (*outputs_.begin(output) != Fsa::InvalidLabelId)) {
554  _Arc* a = sp->newArc();
555  a->output_ = *outputs_.begin(output);
556  if (fsa_->type() == Fsa::TypeTransducer)
557  a->input_ = Fsa::Epsilon;
558  else
559  a->input_ = a->output();
560  a->weight_ = semiring_->one();
561  subsetStart();
562  subsetInsert(Substate(sub->state_, outputs_.insert(output + 1), sub->weight_));
563  a->target_ = insertState(subsetStop(), state->output_);
564  state = &states_[s];
565  tags |= (_sp->tags() & ~Fsa::StateTagFinal);
566  }
567  else
568  tags |= _sp->tags();
569  if (_sp->isFinal()) {
570  fw = semiring_->collect(fw, semiring_->extend(sub->weight_, _sp->weight_));
571  }
572  }
573  else { // disambiguate automaton => disconnect detected, ambiguous cycles
574  subsetStart();
575  subsetInsert(Substate(sub->state_, 0, semiring_->one()));
576  Fsa::LabelIdStrings::Id id = outputs_.insert(sub->output_ + 1);
577  Fsa::StateId target = insertState(subsetStop(), *outputs_.begin(sub->output_) == Fsa::InvalidLabelId ? sub->output_ : id);
578  Fsa::LabelId label = *outputs_.begin(sub->output_) == Fsa::InvalidLabelId ? Fsa::Epsilon : *outputs_.begin(sub->output_);
579  sp->newArc(target, sub->weight_,
580  (fsa_->type() == Fsa::TypeTransducer) ? Fsa::Epsilon : label, label);
581  }
582  sp->setTags(tags);
583  if (sp->isFinal()) {
584  sp->weight_ = fw;
585  }
586  std::sort(tmpArcs_.begin(), tmpArcs_.end());
587 
588  // create arcs and collect weights
589  // one iteration of the following loop creates one arc with a unique input label
590  Core::Vector<Fsa::LabelId> outputArc, outputTarget;
591  for (typename Core::Vector<Arc_>::const_iterator a = tmpArcs_.begin(); a != tmpArcs_.end();) {
592  typename Core::Vector<Arc_>::const_iterator first = a, last;
593  _Arc* arc = sp->newArc();
594 
595  // determinize arcs and collect arc weights
596  arc->input_ = first->input_;
597  arc->weight_ = a->weight_;
598  outputArc.resize(0);
599  for (Core::Vector<Fsa::LabelId>::const_iterator i = tmpOutputs_.begin() + a->start_;
600  *i != Fsa::InvalidLabelId; ++i)
601  outputArc.push_back(*i);
602  outputArc.push_back(Fsa::InvalidLabelId);
603  for (++a; (a != tmpArcs_.end()) && (a->input_ == first->input_); ++a) {
604  arc->weight_ = semiring_->collect(arc->weight_, a->weight_);
605  lcp(outputArc, tmpOutputs_.begin() + a->start_);
606  }
607  last = a;
608 
609  // partial arc label
610  Fsa::LabelIdStrings::Id output = 0;
611  if (outputArc.size() > 1) {
612  arc->output_ = outputArc.front();
613  if (outputArc.size() > 2)
614  output = outputs_.append(outputArc.begin() + 1);
615  }
616  else
617  arc->output_ = Fsa::Epsilon;
618 
619  // gather target weights and outputs and create target state
620  Weight inv = semiring_->invert(arc->weight_);
621  subsetStart();
622  for (a = first; a < last; ++a) {
623  Fsa::LabelIdStrings::Id output = 0;
624  Core::Vector<Fsa::LabelId>::const_iterator start = tmpOutputs_.begin() + a->start_;
625  if (isPrefix(outputArc, start))
626  output = outputs_.append(start + outputArc.size() - 1);
627  subsetInsert(Substate(a->predecessor_, a->target_, output,
628  semiring_->extend(inv, a->weight_)));
629  }
630  arc->target_ = insertState(subsetStop(), output);
631  state = &states_[s]; // update pointer to state (might change due to resize of hash vector)
632  }
633  }
634  else {
635  // create new intermediate state with a single arc (also used for subsequential final strings)
636  sp->newArc(insertState(state->subset_, outputs_.insert(output + 1)), semiring_->one(),
637  (fsa_->type() == Fsa::TypeTransducer) ? Fsa::Epsilon : *outputs_.begin(output),
638  *outputs_.begin(output));
639  }
640  return _ConstStateRef(sp);
641  }
642  return _ConstStateRef();
643  }
644  void dumpOutput(Fsa::LabelIdStrings::Id output, std::ostream& o) const {
645  const Fsa::LabelIdStrings::Id s = output;
646  if (*outputs_.begin(s) == Fsa::InvalidLabelId)
647  o << "eps";
648  else {
649  Fsa::ConstAlphabetRef a = fsa_->getOutputAlphabet();
650  for (Core::Vector<Fsa::LabelId>::const_iterator i = outputs_.begin(s); *i != Fsa::InvalidLabelId; ++i) {
651  if (i > outputs_.begin(s))
652  o << " ";
653  if (a)
654  o << a->symbol(*i);
655  else
656  o << *i;
657  }
658  }

661  if (s < states_.size()) {
662  const State_* state = &states_[s];
663  typename Substates::const_iterator begin = substates_.begin(state->subset_);
664  for (typename Substates::const_iterator sub = begin; sub.valid(); ++sub) {

669  o << "+";

671  if (!semiring_->isDefault(sub->weight_))

673  if (*outputs_.begin(sub->output_) != Fsa::InvalidLabelId) {

683  }

687  }
