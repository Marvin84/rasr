34 // We don't use the BufferedFeatureExtractor::processBuffer because it just forwards
35 // the segment features ^= mini batch to the NN trainer.
36 // The NN must also know about the segment transcription, though.
37 template<typename FloatT>
38 void BufferedSegmentFeatureProcessor<FloatT>::processBuffer() {
39  typedef typename Math::FastMatrix<FloatT> FastMatrix;
40  typedef typename Precursor::NnMatrix NnMatrix;
41 
42  // We expect that the underlying BufferedFeatureProcessor
43  // is in utterance buffer mode and this is only called indirectly
44  // from leaveSpeechSegment().
45  verify(Precursor::nBufferedFeatures_ > 0);
46  auto* currentSegment = dynamic_cast<Bliss::SpeechSegment*>(Precursor::getCurSegment());
47  require(currentSegment);
48 
49  Precursor::prepareProcessBuffer();
50 
51  // This is the number of time frames of the current segment.
52  verify_eq(Precursor::batchSize_, Precursor::nBufferedFeatures_);
53 
54  std::vector<NnMatrix> miniBatch;
55  Precursor::generateMiniBatch(miniBatch, Precursor::batchSize_);
56  verify(!miniBatch.empty());
57  verify_eq(Precursor::batchSize_, miniBatch.at(0).nColumns());
58 
59  // Every buffer is exactly one mini-batch.
60  verify_eq(Precursor::nProcessedMiniBatches_, 0);
61 
62  Precursor::log("Process segment ")
63  << "with " << miniBatch.at(0).nColumns() << " features.";
64 
65  if (!Precursor::trainer_->isInitialized()) {
66  std::vector<u32> streamSizes;
67  for (u32 stream = 0; stream < miniBatch.size(); stream++)
68  streamSizes.push_back(miniBatch.at(stream).nRows());
69  Precursor::trainer_->initializeTrainer(Precursor::batchSize_, streamSizes);
70  }
71  else
72  Precursor::trainer_->setBatchSize(Precursor::batchSize_);
73 
74  // Process mini batch.
75  Precursor::trainer_->processBatch_feedInput(miniBatch, NULL, currentSegment);
76  Precursor::trainer_->processBatch_finishWithSpeechSegment(*currentSegment);
77 
78  Precursor::nProcessedMiniBatches_++;
79  Precursor::nProcessedFeatures_ += Precursor::batchSize_;
80 
81  Precursor::finalizeProcessBuffer();
82 }
