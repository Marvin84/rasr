87 template<class T>
88 void VectorSelectNode<T>::applyFeatureSelection(Flow::Vector<T>& in, Flow::Vector<T>& out) {
89  if (featureRange_.size() == 0)
90  out = in;
91  else
92  for (u32 i = 0; i < featureRange_.size(); i++) {
93  if ((u32)featureRange_[i] >= in.size())
94  criticalError() << "Selection mismatch: " << featureRange_[i] << " >= " << in.size();
95  require(i < out.size());
96  out[i] = in[featureRange_[i]];
97  }
98 }

113 template<class T>
114 bool VectorSelectNode<T>::work(Flow::PortId p) {
115  Flow::DataPtr<Flow::Vector<T>> ptrFeatures;
116 
117  if (getData(0, ptrFeatures)) {
118  // select the features
119  require(getOutputSize()); // If this fails, then no selection was specified
120  Flow::Vector<T>* out = new Flow::Vector<T>(getOutputSize());
121  applyFeatureSelection(*(ptrFeatures.get()), *out);
122 
123  out->setTimestamp(*ptrFeatures);
124  return putData(0, out);
125  }
126  return putData(0, ptrFeatures.get());
127 }

129 template<class T>
130 void VectorSelectNode<T>::setFeatureRange(std::vector<std::string> rangeToken) {
131  std::vector<Range> featureRange;
132 
133  if (rangeToken.size() != 0) { // get the range:
134  for (u32 i = 0; i < rangeToken.size(); i++) {
135  featureRange = paramFeatureSelectTemp(rangeToken[i]);
136 
137  switch (featureRange.size()) {
138  case 1: {
139  featureRange_.push_back(featureRange[0]);
140  } break;
141  case 2: {
142  for (Range j = featureRange[0]; j <= (Range)featureRange[1]; j++) {
143  featureRange_.push_back(j);
144  }
145  } break;
146  default:
147  error("parse error in feature selection list");
148  }
149  }
150  }
151 }
