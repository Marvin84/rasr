31 /**
32  * The HistoryManager serves as a memory manger for the
33  * HistoryDescriptors used by a LanguageModel. Each LanguageModel
34  * creates an appropriate HistoryManager. Clients should never
35  * bother about this class. HistoryManager also plays the role of
36  * a virtual method table for HistoryHandle.
37  *
38  */
39 class HistoryManager {
40 protected:
41  HistoryManager(){};
42 
43  friend class History;
44 
45  /**
46  * Create a new reference to this history.
47  * This function is called whenever a new pointer to this
48  * history is required. You may instantiate a copy, increase
49  * a reference count, or whatever
50  */
51  virtual HistoryHandle acquire(HistoryHandle) = 0;
52 
53  /**
54  * Destroy a reference to this history.
55  * This function should be redefined to complement the chosen
56  * strategy fo acquire()
57  */
58  virtual void release(HistoryHandle) = 0;
59 
60  /**
61  * If isEquivalent() is true For two descriptors, hashKey()
62  * must return the same value for both of them.
63  */
64  virtual HistoryHash hashKey(HistoryHandle) const = 0;
65 
66  /**
67  * Test if two descriptors represent the same context.
68  * Note: The "check for equal pointers" optimization is
69  * already done before this function is called.
70  */
71  virtual bool isEquivalent(HistoryHandle, HistoryHandle) const = 0;
72 
73  virtual std::string format(HistoryHandle hd) const {
74  return Core::form("n%p", hd);
75  }
76 
77 public:
78  virtual ~HistoryManager(){};
79 };

98 /**
99  * Manager for statically allocated HistoryHandles.
100  * This is very trivial: acquire() and release() are no-ops.
101  * hashKey() and isEquivalent() use the descriptors' address in
102  * memory.
103  **/

105 class StaticHistoryManager : public HistoryManager {

122 /**
123  * HistoryManager for reference counted descriptors.
124  * The concrete HistoryHandle must inherit Core::ReferenceCounted.
125  */
126 class ReferenceCountingHistoryManager : public HistoryManager {
127 protected:
128  virtual HistoryHandle acquire(HistoryHandle hd) {
129  const Core::ReferenceCounted* rc = static_cast<const Core::ReferenceCounted*>(hd);
130  rc->acquireReference();
131  return hd;
132  }
133 
134  virtual void release(HistoryHandle hd) {
135  const Core::ReferenceCounted* rc = static_cast<const Core::ReferenceCounted*>(hd);
136  if (rc->releaseReference())
137  rc->free();
138  }
139 };
