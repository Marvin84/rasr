45 void SegmentwiseFeatureExtractor::enterRecording(const Bliss::Recording* recording) {
46  if (recording->fullName() == lastRecordingName_)
47  return;
48  std::string inputFile;
49  if (!recording->audio().empty()) {
50  inputFile = recording->audio();
51  dataSource_->setParameter("input-audio-file", recording->audio());
52  dataSource_->setParameter("input-audio-name", recording->name());
53  }
54  if (!recording->video().empty()) {
55  inputFile = recording->video();
56  dataSource_->setParameter("input-video-file", recording->video());
57  }
58  require(!inputFile.empty());
59  dataSource_->setParameter("input-file", inputFile);
60  dataSource_->setParameter("recording-index", Core::form("%d", u32(nRecordings_)));
61  lastRecordingName_ = recording->fullName();
62  ++nRecordings_;
63 }

65 void SegmentwiseFeatureExtractor::enterSegment(const Bliss::SpeechSegment* segment) {
66  enterRecording(segment->recording());
67  dataSource_->setParameter("id", segment->fullName());
68  dataSource_->setParameter("segment-index", Core::form("%d", u32(nSegments_)));
69  dataSource_->setParameter("segment-type", std::string(Bliss::Segment::typeId[segment->type()]));
70  dataSource_->setParameter("acoustic-condition", segment->condition() ? segment->condition()->name() : "");
71  dataSource_->setParameter("start-time", Core::form("%g", segment->start()));
72  dataSource_->setParameter("end-time", Core::form("%g", segment->end()));
73  dataSource_->setParameter("track", Core::form("%d", segment->track()));
74  // disassemble segment fullname: .../segment-1/segment-0 and corpus-0/corpus-1/...
75  dataSource_->setParameter("segment", segment->fullName());
76  dataSource_->setParameter("segment-0", segment->name());
77  u32 segmentLevel = 1;
78  Bliss::CorpusSection* corpusSection = segment->parent();
79  while (corpusSection) {
80  dataSource_->setParameter("corpus-" + Core::form("%d", corpusSection->level()),
81  corpusSection->name());
82  dataSource_->setParameter("segment-" + Core::form("%d", segmentLevel),
83  corpusSection->name());
84  corpusSection = corpusSection->parent();
85  segmentLevel++;
86  }
87  if (segment->speaker()) {
88  dataSource_->setParameter("speaker", segment->speaker()->name());
89  dataSource_->setParameter("gender", Bliss::Speaker::genderId[segment->speaker()->gender()]);
90  }
91  dataSource_->setParameter("orthography", segment->orth());
92  lastSegmentName_ = segment->fullName();
93  ++nSegments_;
94 }

96 void SegmentwiseFeatureExtractor::leaveSegment(const Bliss::SpeechSegment* segment) {
97  if (statisticsChannel_.isOpen()) {
98  const std::vector<size_t>& nFrames(dataSource_->nFrames());
99  for (size_t i = nFrames_.size(); i < nFrames.size(); ++i) {
100  portNames_.push_back(dataSource_->outputName(i));
101  nFrames_.push_back(0);
102  }
103  statisticsChannel_ << Core::XmlOpen("statistics");
104  for (size_t i = 0; i < nFrames_.size(); ++i) {
105  nFrames_[i] += nFrames[i];
106  statisticsChannel_ << Core::XmlEmpty("frames") + Core::XmlAttribute("port", portNames_[i]) + Core::XmlAttribute("number", nFrames[i]);
107  }
108  statisticsChannel_ << Core::XmlClose("statistics");
109  }
110  dataSource_->setParameter("speaker", "");
111  dataSource_->setParameter("gender", "");
112  dataSource_->setParameter("orthography", "");
113 }

118  statisticsChannel_ << Core::XmlEmpty("recordings") + Core::XmlAttribute("number", nRecordings_);

146 void SegmentwiseSpeechProcessor::processSegment(const Bliss::SpeechSegment* segment) {
147  modelAdaptor_->enterSegment(segment);
148  featureExtractor_->enterSegment(segment);
149  DataSourceRef dataSource = featureExtractor_->extractor();
150  FeatureList features;
151  FeatureRef feature;
152  dataSource->initialize(const_cast<Bliss::SpeechSegment*>(segment));
153  if (dataSource->getData(feature)) {
154  // try to check the dimension only once for each segment
155  AcousticModelRef acousticModel = modelAdaptor_->modelCombination()->acousticModel();
156  if (acousticModel) {
157  Mm::FeatureDescription* description = feature->getDescription(*featureExtractor_);
158  if (!acousticModel->isCompatible(*description))
159  acousticModel->respondToDelayedErrors();
160  delete description;
161  }
162  features.push_back(feature);
163  while (dataSource->getData(feature))
164  features.push_back(feature);
165  }
166  dataSource->finalize();
167  featureExtractor_->leaveSegment(segment);
168  process(features);
169  modelAdaptor_->leaveSegment(segment);
170 }
