114 /*
115  * AcousticSegmentwiseTrainer

119  * split into denominator and numerator accumulation.

130  delete acc_;
131  acc_ = 0;

134  if (!acc_)
135  acc_ = createAcc();
136  return acc_;
137  }
138  virtual A* createAcc() = 0;
139  virtual A* createNumAcc() = 0;
140  virtual A* createDenAcc() = 0;
141  virtual A* createMleAcc() = 0;
142 
143  void accumulateNumerator(Fsa::ConstAutomatonRef fsa, Core::Ref<const Lattice::WordBoundaries> wb) {
144  A* acc = createNumAcc();
145  acc->setWordBoundaries(wb);
146  acc->setFsa(fsa);

151  A* acc = createDenAcc();
