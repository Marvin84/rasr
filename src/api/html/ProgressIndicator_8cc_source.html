95  void putPercentage(int pos, f32 percentage) {

100  void putBar(int start, int end, f32 percentage) {
101  line[start] = '[';
102  line[end] = ']';
103  int pos = start + 1 + int(roundf(f32(end - start - 1) / 100.0 * percentage));
104  for (int i = start + 1; i < end; ++i)
105  line[i] = (i < pos) ? '=' : ' ';
106  }

108  void putItems(int pos, u32 done, const std::string& unit) {
109  int z = snprintf(info, length, "%10d %s", done, unit.c_str());
110  strncpy(line + pos, info, std::min(length - pos, std::max(0, z)));
111  }

130 void ProgressIndicator::DrawBuffer::drawBar(const std::string& task, ProgressIndicator::Alignment align, const std::string& unit,
131  u32 done, f32 percentage, f32 eta) {
132  int etaPos = length - 7;
133  int unitPos = etaPos - 1 - std::min(int(unit.length()), length / 10);
134  int itemsPos = unitPos - 1 - 10;
135  int barEnd = itemsPos - 2;
136  int percentagePos = std::min(std::max(int(task.length()), length / 5), length / 3) + 1;
137  int barStart = percentagePos + 5;
138 
139  clear();
140 
141  switch (align) {
142  case Left:
143  strncpy(line, task.data(), std::min(int(task.length()), percentagePos - 1));
144  break;
145  case Right:
146  strncpy(line, task.data() + std::max(0, int(task.length()) - percentagePos - 1),
147  std::min(size_t(percentagePos - 1), task.length()));
148  break;
149  default: defect();
150  }
151  putPercentage(percentagePos, percentage);
152  putBar(barStart, barEnd, percentage);
153  putItems(itemsPos, done, unit);
154 
155  if (eta) {
156  int z;
157  if (eta < 60.0) {
158  z = snprintf(info, length, "~ %4.0fs ", eta);
159  }
160  else if (eta < 3600.0) {
161  z = snprintf(info, length, "~ %2.0fmin ", eta / 60.0);
162  }
163  else if (eta < 36000) {
164  z = snprintf(info, length, "~ %2.1fh ", eta / 3600.0);
165  }
166  else {
167  z = snprintf(info, length, "~ %4.0fh ", eta / 3600.0);
168  }
169  strncpy(line + etaPos, info, std::min(length - etaPos, std::max(0, z)));
170  }
171 }

199 void ProgressIndicator::sigStopContinueHandler(int sig) {
200  if (!activeInstance)
201  return;
202  switch (sig) {
203  case SIGTSTP:
204  activeInstance->deactivate();
205  activeInstance->hide();
206  signal(SIGCONT, sigStopContinueHandler);
207  raise(sig);
208  break;
209  case SIGCONT:

255 void ProgressIndicator::start(u32 total) {
256  require(activeInstance != this);
257  tick_ = done_ = doneAtLastUpdate_ = 0;
258  total_ = total;
259  startTime_ = time(0);
260 
261  if (!isEnabled() || activeInstance)
262  return;
263 
264  activeInstance = this;
265  Core::Channel::Manager::us()->blockTty();
266  if (!draw)
267  draw = new DrawBuffer(windowSize());
268 
269  // Note: An active progress bar
270  // prevents GDB from stepping through the program code. GDB stops correctly at the breakpoints,
271  // but if you use the "step" or "next" command afterwards, the timer signal lets GDB jump into
272  // the sigAlarmHandler method instead of the desired program line. For that reason, the progress
273  // indicator is deactivated in debug mode.
274 #ifndef DEBUG
275  if (!AmIBeingDebugged())
276  activate();
277 #endif
278 }

342 void ProgressIndicator::print() {
343  require(activeInstance == this);
344  if (total_) {
345  f32 percentage = 100.0 * f32(done_) / f32(total_);
346  f32 eta = 0.0;
347  if (done_ && done_ < total_)
348  eta = f32(time(0) - startTime_) / f32(done_) * f32(total_ - done_);
349  draw->drawBar(task_, align_, unit_, done_, percentage, eta);
350  }
351  else {
352  draw->drawSpinner(task_, unit_, done_, tick_);
353  }
354  draw->write(fd);
355  isVisible_ = true;
356 }
