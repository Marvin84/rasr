21 template<class _Automaton>
22 bool isLinear(typename _Automaton::ConstRef la) {
23  Fsa::StateId si = la->initialStateId();
24  if (si == Fsa::InvalidStateId)
25  return true;
26  std::unordered_set<Fsa::StateId> seen;
27  seen.insert(si);
28  typename _Automaton::ConstStateRef s = la->getState(si);
29  while (!s->isFinal()) {
30  if (s->nArcs() != 1)
31  return false;
32  si = s->begin()->target();
33  if (!seen.insert(si).second)
34  return false;
35  s = la->getState(si);
36  }
37  return (s->nArcs() == 0);
38 }
