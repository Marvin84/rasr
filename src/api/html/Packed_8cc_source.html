41 void PackedAutomaton::setState(State* sp) {
42  static PackedState init(Core::Type<size_t>::max);
43  states_.grow(sp->id(), init);
44  require(states_[sp->id()].start_ == Core::Type<size_t>::max);
45 
46  if (arcs_.size() > 0)
47  arcs_.pop_back(); // rewind as we can overwrite the automaton end marker
48  states_[sp->id()].start_ = arcs_.size();
49  u8 partition = (sp->tags() >> StateIdBits) << 5;
50  if (sp->hasTags(StateTagFinal))
51  if (!semiring()->isDefault(sp->weight_))
52  partition |= 0x01;
53  arcs_.push_back(partition | 0x1c);
54  if (partition & 0x01)
55  appendBytes(arcs_, sp->weight_, sizeof(Weight));
56  size_t start = arcs_.size();
57  pStates_ += start - states_[sp->id()].start_;
58 
59  for (State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
60  partition = (estimateBytes(a->target()) - 1) << 6;
61  if (!semiring()->isDefault(a->weight()))
62  partition |= 0x20;
63  int i = 0;
64  if (a->input() != Epsilon)
65  i = estimateBytes(a->input());
66  int o = 0;
67  if (type() != TypeAcceptor)
68  if (a->output() != Epsilon)
69  o = estimateBytes(a->output());
70  partition |= (5 * o) + i;
71  arcs_.push_back(partition);
72 
73  appendBytes(arcs_, a->target(), (partition >> 6) + 1);
74  if (partition & 0x20)
75  appendBytes(arcs_, a->weight(), sizeof(Weight));
76  appendBytes(arcs_, a->input(), i);
77  if (type() != TypeAcceptor)
78  appendBytes(arcs_, a->output(), o);
79  nArcs_++;
80  }
81  pArcs_ += arcs_.size() - start;
82  arcs_.push_back(0xff); // automaton end marker
83  nStates_++;
84  delete sp;
85 }

92 ConstStateRef PackedAutomaton::getState(StateId s) const {
93  if (s < states_.size()) {
94  Core::Vector<u8>::const_iterator a = arcs_.begin() + states_[s].start_;
95  u8 partition = *(a++);
96  StateTag tags = StateTag(partition & 0xe0) << (StateIdBits - 5);
97  State* sp = new State(s, tags);
98  if (partition & 0x01)
99  sp->weight_ = Weight(getBytesAndIncrement(a, sizeof(Weight)));
100  else
101  sp->weight_ = semiring()->one();
102  for (; ((partition = *(a++)) & 0x1c) != 0x1c;) {
103  Arc* arc = sp->newArc();
104  arc->target_ = getBytesAndIncrement(a, (partition >> 6) + 1);
105  if (partition & 0x20)
106  arc->weight_ = Weight(u32(getBytesAndIncrement(a, sizeof(Weight))));
107  else
108  arc->weight_ = semiring()->one();
109  u8 io = partition & 0x1f;
110  if (io % 5)
111  arc->input_ = getBytesAndIncrement(a, io % 5);
112  else
113  arc->input_ = Epsilon;
114  if (type() != TypeAcceptor) {
115  if (io / 5)
116  arc->output_ = getBytesAndIncrement(a, io / 5);
117  else
118  arc->output_ = Epsilon;
119  }
120  else
121  arc->output_ = arc->input();
122  }
123  return ConstStateRef(sp);
124  }
125  return ConstStateRef();
126 }
