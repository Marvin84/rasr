36 void HistogramNormalization::setTrainingHistograms(const std::vector<HistogramVector<Value>>& trainingHistograms,
37  const std::vector<HistogramWeight>& scales,
38  Probability probabilityBucketSize) {
39  size_t nScales = trainingHistograms.size();
40  require(nScales == scales.size());
41  require(nScales > 0);
42 
43  size_t dimension = trainingHistograms[0].size();
44 
45  Value minimalBucketSize = Core::Type<Value>::max;
46  for (size_t i = 0; i < nScales; ++i)
47  minimalBucketSize = std::min(minimalBucketSize, trainingHistograms[i].minimalBucketSize());
48 
49  HistogramVector<Value> interpolatedHistograms(dimension, minimalBucketSize);
50  for (size_t i = 0; i < nScales; ++i) {
51  require(trainingHistograms[i].size() == dimension);
52  for (size_t d = 0; d < dimension; ++d) {
53  Histogram<Value> toAdd(trainingHistograms[i][d]);
54  toAdd.normalizeSurface();
55  toAdd *= scales[i];
56  interpolatedHistograms[d] += toAdd;
57  }
58  }
59  setTrainingHistograms(interpolatedHistograms, probabilityBucketSize);
60 }

77 bool HistogramNormalization::areScalesWellDefined(const std::vector<HistogramWeight>& scales) {
78  // 0 <= scale <= 1
79  if (std::find_if(scales.begin(), scales.end(), std::bind2nd(std::less<HistogramWeight>(), 0)) != scales.end())
80  return false;
81  if (std::find_if(scales.begin(), scales.end(), std::bind2nd(std::greater<HistogramWeight>(), 1)) != scales.end())
82  return false;
83  return true;
84 }

86 bool HistogramNormalization::areScalesNormalized(const std::vector<HistogramWeight>& scales) {
87  HistogramWeight sum = std::accumulate(scales.begin(), scales.end(), 0.0, std::plus<HistogramWeight>());
88  return Core::isAlmostEqual(sum, (HistogramWeight)1.0);

176 void HistogramNormalizationNode::init(size_t featureDimension) {
177  if (nTestHistograms() != featureDimension) {
178  error() << "Mismatch between #test-histograms(" << nTestHistograms()
179  << ") and feature dimension(" << featureDimension << ").";
180  }
181 
182  verify(nInputs() >= firstScalePortId_);
183  histogramScales_.resize(nInputs() - firstScalePortId_);
184  if ((histogramScales_.size() + 1) != trainingHistogramFilenames_.size()) {
185  error() << "Mismatch between #training-histograms(" << trainingHistogramFilenames_.size()
186  << ") and #scale-ports(" << histogramScales_.size() << ").";
187  }
188 
189  loadTrainingHistograms(featureDimension);
190 
191  respondToDelayedErrors();
192 
193  if (trainingHistograms_.size() == 1)
194  HistogramNormalization::setTrainingHistograms(trainingHistograms_[0], probabilityBucketSize_);
195 
196  reset();
197  needInit_ = false;
198 }

200 void HistogramNormalizationNode::loadTrainingHistograms(size_t featureDimension) {
201  trainingHistograms_.resize(trainingHistogramFilenames_.size());
202  for (size_t i = 0; i < trainingHistograms_.size(); ++i) {
203  const std::string& filename(trainingHistogramFilenames_[i]);

205  Core::BinaryInputStream is(filename);
206  bool success = is.isOpen();
207  if (success) {
208  trainingHistograms_[i].read(is);
209  success = is.good();
210  if (success) {

212  error() << "Mismatch between #training-histograms(" << trainingHistograms_[i].size()
213  << ") and feature dimension(" << featureDimension << ").";
214  }
215  }
216  }
217  if (!success)
218  error() << "Failed to read training histogram from file '" << filename << "'.";
219  }

238 bool HistogramNormalizationNode::updateScales(const Flow::Timestamp& timestamp) {
239  bool changed = false;
240  for (size_t i = 0; i < histogramScales_.size(); ++i) {
241  while (!histogramScales_[i].contains(timestamp)) {
242  Flow::DataPtr<Flow::DataAdaptor<HistogramWeight>> in;
243  Flow::PortId portId = firstScalePortId_ + i;
244  if (getData(portId, in)) {
245  histogramScales_[i] = *in;
246  changed = true;
247  }
248  else {
249  criticalError() << "The " << scalePortname_ << portId
250  << " stream stopped before start-time ("
251  << timestamp.startTime() << ").";
252  }
253  }
254  }
255  return changed;
256 }

258 bool HistogramNormalizationNode::updateTrainingHistograms() {
259  bool result = true;
260  std::vector<HistogramWeight> histogramScales(histogramScales_.size());
261  for (size_t i = 0; i < histogramScales_.size(); ++i)
262  histogramScales[i] = histogramScales_[i].data();
263 
264  normalizeScales(histogramScales);
265  if (!(result = areScalesWellDefined(histogramScales)))
266  error() << "One or more histogram scales are smaller than zero or larger than 1.";
267 
268  verify(areScalesNormalized(histogramScales));
269  HistogramNormalization::setTrainingHistograms(trainingHistograms_, histogramScales, probabilityBucketSize_);
270  return result;
