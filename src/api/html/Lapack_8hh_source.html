67 /** getrf: computes an LU factorization of a general m-by-n matrix A
68  * using partial pivoting with row interchanges.
69  *
70  * @param A:
71  * -On entry a general m-by-n matrix;
72  * -On exit the factors L and U from the factorization
73  * A = P*L*U; the unit diagonal elements of L are not stored
74  *
75  * @param pivotIndices: for 1 <= i <= min(M,N), row i of the
76  * matrix was interchanged with row pivotIndices(i).
77  *
78  * @return:
79  * 0: successful exit
80  * i<0: the -i-th argument of the Lapack call had an illegal value
81  * i>0: U(i,i) is exactly zero. The factorization
82  * has been completed, but the factor U is exactly
83  * singular, and division by zero will occur if it is used
84  * to solve a system of equations.
85  */
86 void getrf(int* m, int* n, float* A, int* lda, int* ipiv, int* info);

90 template<class T>
91 int getrf(Matrix<T>& A, Vector<int>& pivotIndices) {
92  int m = A.nRows();
93  int n = A.nColumns();
94  int lda = A.nRows();
95  int info;
96 
97  pivotIndices.resize(std::min(A.nRows(), A.nColumns()));
98 
99  getrf(&m, &n, A.buffer(), &lda, pivotIndices.buffer(), &info);
100 
101  return info;
102 }

128 template<class T>
129 int getrs(Matrix<T>& LU, Matrix<T>& B, Vector<int>& pivotIndices) {
130  char trans = 'N';
131  int n = std::min(std::min(LU.nColumns(), LU.nRows()), B.nRows());
132  int nrhs = B.nColumns();
133  int lda = LU.nRows();
134  int ldb = B.nRows();
135  int info = 1;
136 
137  getrs(&trans, &n, &nrhs, LU.buffer(), &lda, pivotIndices.buffer(), B.buffer(), &ldb, &info);
138 
139  return info;
140 }
