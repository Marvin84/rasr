22 void ludcmp(Math::Matrix<DP>& a, Math::Vector<int>& indx, DP& d) {
23  require(a.nRows() == a.nColumns());
24  require(indx.size() == a.nColumns());
25 
26  const DP TINY = 1.0e-20;
27  int i, imax = 0, j, k;
28  DP big, dum, sum, temp;
29 
30  int n = a.nRows();
31  Math::Vector<DP> vv(n);
32  d = 1.0;
33  for (i = 0; i < n; i++) {
34  big = 0.0;
35  for (j = 0; j < n; j++)
36  if ((temp = fabs(a[i][j])) > big)
37  big = temp;
38  if (big == 0.0)
39  nrerror("Singular matrix in routine ludcmp");
40  vv[i] = 1.0 / big;
41  }
42  for (j = 0; j < n; j++) {
43  for (i = 0; i < j; i++) {
44  sum = a[i][j];
45  for (k = 0; k < i; k++)
46  sum -= a[i][k] * a[k][j];
47  a[i][j] = sum;
48  }
49  big = 0.0;
50  for (i = j; i < n; i++) {
51  sum = a[i][j];
52  for (k = 0; k < j; k++)
53  sum -= a[i][k] * a[k][j];
54  a[i][j] = sum;
55  if ((dum = vv[i] * fabs(sum)) >= big) {
56  big = dum;
57  imax = i;
58  }
59  }
60  if (j != imax) {
61  for (k = 0; k < n; k++) {
62  dum = a[imax][k];
63  a[imax][k] = a[j][k];
64  a[j][k] = dum;
65  }
66  d = -d;
67  vv[imax] = vv[j];
68  }
69  indx[j] = imax;
70  if (a[j][j] == 0.0)
71  a[j][j] = TINY;
72  if (j != n - 1) {
73  dum = 1.0 / (a[j][j]);
74  for (i = j + 1; i < n; i++)
75  a[i][j] *= dum;
76  }
77  }
78 }
