50 void Session::addGraph(Graph const& graph) {
51  auto timer_start = std::chrono::steady_clock::now();
52  tf::Env* env = tf::Env::Default();
53  for (std::string const& lib : graph.libraries()) {
54  void* handle = nullptr;
55 #if TF_MAJOR_VERSION < 2 || (TF_MAJOR_VERSION == 2 && TF_MINOR_VERSION < 4)
56  tf::Status status = env->LoadLibrary(lib.c_str(), &handle);
57 #else
58  tf::Status status = env->LoadDynamicLibrary(lib.c_str(), &handle);
59 #endif
60  log("Loading library: ") << lib.c_str();
61  if (!status.ok()) {
62  criticalError("error loading library: %s", status.ToString().c_str());
63  }
64  }
65 
66  tf::Status status = session_->Create(graph.graph_def_);
67  if (not status.ok()) {
68  criticalError("error creating session %s", status.ToString().c_str());
69  }
70 
71  std::vector<std::string> var_init;
72  for (auto const& v : graph.variables()) {
73  if (not v.second.initializer_name.empty()) {
74  var_init.push_back(v.second.initializer_name);
75  }
76  }
77  if (not var_init.empty()) {
78  run({}, var_init);
79  }
80  auto timer_end = std::chrono::steady_clock::now();
81  log("Session::addGraph ") << std::chrono::duration<double, std::milli>(timer_end - timer_start).count() << "ms";
82 }

84 bool Session::run(std::vector<std::pair<std::string, Tensor>> const& inputs,
85  std::vector<std::string> const& output_tensor_names,
86  std::vector<std::string> const& target_node_names,
87  std::vector<Tensor>& outputs) {
88  std::vector<std::pair<std::string, tf::Tensor>> tf_inputs;
89  tf_inputs.reserve(inputs.size());
90  std::vector<tf::Tensor> tf_outputs;
91 
92  for (auto& input : inputs) {
93  tf_inputs.push_back(std::make_pair(input.first, *input.second.tensor_));
94  }
95 
96  tf::Status status;
97 
98  if (profileRun_) {
99  tf::RunOptions options;
100  options.set_trace_level(tf::RunOptions::SOFTWARE_TRACE);
101  tf::RunMetadata meta_data;
102  status = session_->Run(options, tf_inputs, output_tensor_names, target_node_names, &tf_outputs, &meta_data);
103  std::ofstream out(profilePrefix_ + std::to_string(profileCounter_++), std::ios::out | std::ios::trunc);
104  meta_data.SerializeToOstream(&out);
105  }
106  else {
107  status = session_->Run(tf_inputs, output_tensor_names, target_node_names, &tf_outputs);
108  }
109 
110  if (!status.ok()) {
111  std::string target;
112  for (auto const& t : target_node_names) {
113  target += t + " ";
114  }
115  criticalError("error calling Session::Run (target: %s): %s", target.c_str(), status.ToString().c_str());
116  }
117 
118  outputs.clear();
119  for (auto& tensor : tf_outputs) {
120  outputs.emplace_back(Tensor(std::move(tensor)));
121  }
122 
123  return status.ok();
124 }
