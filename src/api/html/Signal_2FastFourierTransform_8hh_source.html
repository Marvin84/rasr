256 template<class Algorithm>
257 bool FastFourierTransformNode<Algorithm>::configure() {
258  Core::Ref<Flow::Attributes> a(new Flow::Attributes());
259  getInputAttributes(0, *a);
260  if (!configureDatatype(a, Flow::Vector<f32>::type()))
261  return false;
262 
263  f64 sampleRate = atof(a->get("sample-rate").c_str());
264  if (sampleRate <= 0)
265  criticalError("Sample rate (%f) is smaller or equal to 0.", sampleRate);
266 
267  algorithm_.setInputSampleRate(sampleRate);
268  algorithm_.setLength(length(sampleRate));
269  a->set("sample-rate", algorithm_.outputSampleRate());
270 
271  return putOutputAttributes(0, a);
272 }

274 template<class Algorithm>
275 u32 FastFourierTransformNode<Algorithm>::length(f64 sampleRate) const {
276  u32 maximumLength = (u32)ceil(maximumInputSize_ * sampleRate);
277  if (length_ == 0)
278  return maximumLength;
279  else if (maximumLength != 0) {
280  warning("FFT length given by maximum-input-size (%d) will be overwitten by parameter length (%d).",
281  maximumLength, length_);
282  }
283  return length_;
284 }
