39 bool TemporalPattern::init() {
40  // init/check all the member values
41  if ((nFrames() == 0) || (nFeatures() == 0) || (dctSize() == 0) || (!(dctSize() < nFrames())))

43 
44  // init the window/cosine function
45  windowFunction()->setLength((u32)nFrames());
46  cosineTransform().init(Signal::CosineTransform::NplusOneData, nFrames(), (size_t)dctSize(), true);

51 
52  // all inits done

54

58 bool TemporalPattern::apply(std::vector<Value>& in, std::vector<Value>& out) {
59  // init is requested, initialization failed?
60  if ((needInit_) && (!init()))
61  return false;
62 
63  for (size_t band = 0; band < nFeatures(); band++) {
64  getBand(band, in, bandVec);
65  applyWindow(bandVec);
66  applyDCT(bandVec, dctVec);
67  setBand(band, dctVec, out);
68  }
69 
70  return true;
71 }
