44  /** Feeds new data into the Algorithm and reads data out of the Algorithm if existant.
45  * Steps:
46  * 1) Read data out of Algorithm and if successful send it on in the network, and exit.
47  * 2) If Algorithm cannot deliver an output, read new input from the network and
48  * feed it into the algorithm and try step 1) again.
49  *
50  * Error handling:
51  * 1) If EOS arrives from the network, flush the next output of the algorithm.
52  * 2) If Algorithm cannot accept new input, flush all output of the algorithm,
53  * put them into the network and put the new input into the algorithm again.
54  */
55  virtual bool work(Flow::PortId p);
56 };

58 template<class Algorithm>
59 bool SlidingAlgorithmNode<Algorithm>::work(Flow::PortId p) {
60  Flow::DataPtr<InputData> in;
61  Flow::DataPtr<OutputData> out(new OutputData);
62 
63  while (!Algorithm::get(*out)) {
64  if (!getData(0, in)) {
65  if (in == Flow::Data::eos()) {
66  if (Algorithm::flush(*out))
67  return putData(0, out.get());
68  }
69  return putData(0, in.get());
70  }
71  if (!Algorithm::put(*in) && flushAllAndPut(p, *in))
72  return true;
73  }
74  return putData(0, out.get());
75 }

77 template<class Algorithm>
78 bool SlidingAlgorithmNode<Algorithm>::flushAllAndPut(Flow::PortId p, const InputData& in) {
79  bool result = true;
80  Flow::DataPtr<OutputData> out;
81 
82  do {
83  out = Flow::dataPtr(new OutputData());
84  if (!Algorithm::flush(*out))
85  break;
86  if (!putData(0, out.get()))
87  result = false;
88  } while (true);
89 
90  if (!Algorithm::put(in))
91  defect();
92  return result;
93 }
