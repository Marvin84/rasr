37 class SsspQueue {
38 protected:
39  Fsa::StateId head_;
40  Core::Vector<Fsa::StateId> queue_;
41 
42 public:
43  SsspQueue()
44  : head_(Fsa::InvalidStateId) {
45  }
46  bool empty() const {
47  return head_ == Fsa::InvalidStateId;
48  }
49  Fsa::StateId dequeue() {
50  require_(!empty());
51  Fsa::StateId s = head_;
52  head_ = queue_[s];
53  if (head_ == s)
54  head_ = Fsa::InvalidStateId;
55  queue_[s] = Fsa::InvalidStateId;
56  return s;
57  }
58  Fsa::StateId maxStateId() const {
59  return Fsa::InvalidStateId;
60  }
61 };

63 class FifoSsspQueue : public SsspQueue {
64 public:
65  void enqueue(Fsa::StateId s) {
66  queue_.grow(s, Fsa::InvalidStateId);
67  if (queue_[s] == Fsa::InvalidStateId) {
68  if (head_ != Fsa::InvalidStateId)
69  queue_[s] = head_;
70  else
71  queue_[s] = s;
72  head_ = s;
73  }
74  }
75 };
