26 void Warping::setWarpingFunction(Math::UnaryAnalyticFunctionRef warpingFunction, size_t inputSize,
27  MergeType mergeType, InterpolationType interpolationType) {
28  require(warpingFunction);
29 
30  inputSize_ = inputSize;
31  deleteInverseItems();
32  inverseWarpingFunction_.resize((size_t)Core::floor(warpingFunction->value(inputSize - 1) + 1));
33 
34  u32 index, previousIndex = 0;
35  u32 warpingIndex, previousWarpingIndex = (u32)Core::floor(warpingFunction->value(previousIndex));
36 
37  for (index = previousIndex + 1; index < inputSize; ++index) {
38  warpingIndex = (u32)Core::floor(warpingFunction->value(index));
39  // only monoton increasing warping functions are supported
40  verify(warpingIndex >= previousWarpingIndex);
41 
42  if (warpingIndex != previousWarpingIndex) {
43  verify(inverseWarpingFunction_.size() > warpingIndex);
44 
45  inverseWarpingFunction_[previousWarpingIndex] = createMerger(mergeType, previousIndex, index);
46  ensure(inverseWarpingFunction_[previousWarpingIndex] != 0);
47 
48  for (u32 w = previousWarpingIndex + 1; w < warpingIndex; ++w) {
49  f64 relativePosition = f32(w - previousWarpingIndex) / f32(warpingIndex - previousWarpingIndex);
50 
51  inverseWarpingFunction_[w] = createInterpolator(interpolationType, previousIndex, index, relativePosition);
52  ensure(inverseWarpingFunction_[w] != 0);
53  }
54 
55  previousWarpingIndex = warpingIndex;
56  previousIndex = index;
57  }
58  }
59  inverseWarpingFunction_[previousWarpingIndex] = createMerger(mergeType, previousIndex, index);
60  ensure(inverseWarpingFunction_[previousWarpingIndex] != 0);
61 }

63 void Warping::setInverseWarpingFunction(Math::UnaryAnalyticFunctionRef inverseWarpingFunction,
64  size_t inputSize, InterpolationType interpolationType) {
65  require(inverseWarpingFunction);
66 
67  inputSize_ = inputSize;
68  deleteInverseItems();
69 
70  for (u32 warpingIndex = 0;; ++warpingIndex) {
71  FloatIndex floatIndex = inverseWarpingFunction->value(warpingIndex);
72  ensure(floatIndex >= 0);
73 
74  u32 leftOfIndex = (u32)Core::floor(floatIndex);
75  f64 relativePosition = floatIndex - leftOfIndex;
76 
77  Warping::InverseItem* interpolator = createInterpolator(interpolationType,
78  leftOfIndex,
79  leftOfIndex + 1,
80  relativePosition);
81  if (interpolator != 0)
82  inverseWarpingFunction_.push_back(interpolator);
83  else
84  return;
85  }
86 }

94 Warping::InverseItem* Warping::createMerger(MergeType type, u32 begin, u32 end) const {
95  require(begin < end);
96 
97  if (begin >= inputSize_ || end > inputSize_)
98  return 0;
99  else if (begin + 1 == end) {
100  return new CopyInverseItem(begin);
101  }
102  else {
103  switch (type) {
104  case AritmeticMean: return new AritmeticMeanInverseItem(begin, end);
105  case SelectBegin: return new SelectBeginInverseItem(begin, end);
106  }
107  }
108  defect();

111 Warping::InverseItem* Warping::createInterpolator(InterpolationType type, u32 first, u32 last, f64 relativePosition) const {
112  require(first < last);
113  require(0.0 <= relativePosition && relativePosition <= 1.0);
114 
115  if (first >= inputSize_) {
116  return 0;
117  }
118  else if (last >= inputSize_) {
119  if (equalRelativePosition(relativePosition, 0)) {
120  return new CopyInverseItem(first);
121  }
122  else
123  return 0;
124  }
125  else if (relativePosition == 0) {
126  // using equalRelativePosition here could result in more CopyInverseItems thus faster warping
127  return new CopyInverseItem(first);
128  }
129  else if (relativePosition == 1) {
130  // using equalRelativePosition here could result in more CopyInverseItems thus faster warping
131  return new CopyInverseItem(last);
132  }
133  else {
134  switch (type) {
135  case InsertZero: return new InsertZeroInverseItem(first, last, relativePosition);
136  case KeepEnd: return new KeepEndInverseItem(first, last, relativePosition);
137  case LinearInterpolation: return new LinearInterpolationInverseItem(first, last, relativePosition);
138  }
139  }
140  defect();
141 }
