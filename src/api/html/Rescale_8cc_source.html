24  * - Rescale lattice, i.e. change semiring scales (and semiring)
25  */
26 class RescaleNode : public FilterNode {
27 public:
28  static const Core::ParameterFloat paramScale;
29  static const Core::ParameterString paramKey;
30 
31 private:
32  ConstSemiringRef lastSemiring_;
33  ConstSemiringRef lastRescaledSemiring_;
34 
35 private:
36  virtual ConstLatticeRef filter(ConstLatticeRef l) {
37  if (!l)
38  return l;
39  if (!lastRescaledSemiring_ || (l->semiring().get() != lastSemiring_.get())) {
40  lastSemiring_ = l->semiring();
41  lastRescaledSemiring_ = cloneSemiring(l->semiring());
42  ScoreId id = 0;
43  for (KeyList::const_iterator itKey = l->semiring()->keys().begin();
44  itKey != l->semiring()->keys().end(); ++itKey, ++id) {
45  if (itKey->empty())
46  warning("RescaleNode: Dimension %zu has no symbolic name", id);
47  else {
48  Score scale = paramScale(select(*itKey));
49  if (scale != Semiring::UndefinedScale)
50  lastRescaledSemiring_->setScale(id, scale);
51  Key key = paramKey(select(*itKey));
52  if (!key.empty())
53  lastRescaledSemiring_->setKey(id, key);
54  }
55  }
56  }
57  return changeSemiring(l, lastRescaledSemiring_);
58  }
59 
60 public:
61  RescaleNode(const std::string& name, const Core::Configuration& config)
62  : FilterNode(name, config) {}
63  virtual ~RescaleNode() {}
64 };
