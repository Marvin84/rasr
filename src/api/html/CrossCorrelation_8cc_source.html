26 void CrossCorrelation::crossCorrelation(const std::vector<Data>& x,
27  const std::vector<Data>& y,
28  std::vector<Data>& Rxy) {
29  u32 K = std::max(Core::abs(begin_), Core::abs(end_ - 1));
30 
31  RealFastFourierTransform fft(std::max(x.size(), y.size()) + K);
32 
33  if (!fft.transform(X_ = x))
34  defect();
35 
36  if (&x == &y) { // autocorrelation
37  Math::transformAlternatingComplexToAlternatingComplex(X_.begin(),
38  X_.end(),
39  X_.begin(),
40  X_.begin(),
41  Math::conjugateMultiplies<Data>());
42  }
43  else {
44  fft.transform(Y_ = y);
45  Math::transformAlternatingComplexToAlternatingComplex(X_.begin(),
46  X_.end(),
47  Y_.begin(),
48  X_.begin(),
49  Math::conjugateMultiplies<Data>());
50  }
51 
52  RealInverseFastFourierTransform iFft(fft.length(), fft.outputSampleRate());
53  if (!iFft.transform(X_))
54  defect();
55 
56  Rxy.resize(end_ - begin_);
57  for (s32 m = begin_; m < end_; ++m)
58  Rxy[m - begin_] = m >= 0 ? *(X_.begin() + m) : *(X_.end() + m);
59 }

106 void CrossCorrelation::apply(const std::vector<Data>& x,
107  const std::vector<Data>& y,
108  std::vector<Data>& Rxy) {
109  if (similarityFunctionType_ == Multiplication)
110  crossCorrelationWithMultiplication(x, y, Rxy);
111  else if (similarityFunctionType_ == AbsoluteDifference)
112  crossCorrelationWithAbsoluteDifference(x, y, Rxy);
113  else
114  defect();
115 
116  if (normalizationType_ == UnbiasedEstimate)
117  normalize(x, y, Rxy, normalizeCrossCorrelationEstimate<Data>(x, y));
118  else if (normalizationType_ == UpperBound)
119  normalizeUpperBound(x, y, Rxy);
120  else if (normalizationType_ != None)
121  defect();
122 }

124 void CrossCorrelation::apply(const std::vector<Data>& x,
125  const std::vector<Data>& y,
126  std::vector<Data>& Rxy,
127  Core::Component& errorChannels) {
128  if (begin_ > end_)
129  errorChannels.error("Discrete begin time (%d) is larger then Discrete end time (%d).", begin_, end_);
130  if ((similarityFunctionType_ != Multiplication || !useFastFourierTransform_) &&
131  (begin_ <= -((s32)y.size()) || end_ > (s32)x.size())) {
132  errorChannels.error("Cannot calculate the interval [%d..%d) with x.size (%zd) and y.size (%zd).",
133  begin_, end_, x.size(), y.size());
134  }
135  if (Core::isMalformed(x.begin(), x.end()) || Core::isMalformed(y.begin(), y.end()))
136  errorChannels.error("One of the inputs contains malformed element.");
137 
138  apply(x, y, Rxy);
139 }

142 bool BandpassAutocorrelation::work(u32 bandBegin, u32 bandEnd, std::vector<Data>& autocorrelation) {
143  size_t nComponents = R_.nRows();
144  require_(autocorrelation.size() == nComponents);

150 bool BandpassAutocorrelation::init(size_t nComponents, const std::vector<Data>& amplitude) {
151  require(nComponents > 0);
152  require(!amplitude.empty());
153 
154  amplitudeSquare_.resize(amplitude.size());
155  std::transform(amplitude.begin(), amplitude.end(), amplitude.begin(),
156  amplitudeSquare_.begin(), std::multiplies<Data>());
157 
158  R_.resize(nComponents, amplitudeSquare_.size());
159  u32 nSamples = 2 * (amplitudeSquare_.size() - 1);
160  for (size_t n = 0; n < nComponents; ++n) {
161  /* Integral from 0 to pi must be done on both sides
162  * The amplitude spectrum is symetric, so the same data [0..N/2] can be used,
163  * just shifted ones: positive discrete freq. = [0..N/2-1],
164  * negative negative discrete freq. [1..N/2] */
165  f64 kernel = 1.0;
166  f64 conjugateKernel;
167 
168  R_[n][0] = 0.0;
169  size_t discreteFrequency;
170  for (discreteFrequency = 0; discreteFrequency < nSamples / 2; discreteFrequency++) {
171  conjugateKernel = cos(2.0 * M_PI / nSamples * (discreteFrequency + 1) * n);
172  R_[n][discreteFrequency + 1] = R_[n][discreteFrequency] +
173  1.0 / nSamples * (amplitudeSquare_[discreteFrequency] * kernel + amplitudeSquare_[discreteFrequency + 1] * conjugateKernel);
174  kernel = conjugateKernel;
175  }
176  }
177  return true;
178 }

214 void CrossCorrelationNode::init() {
215  if (sampleRate_ <= 0)
216  error("Sample rate (%f) is smaller or equal to 0.", sampleRate_);
217 
218  s32 begin = (s32)rint(continuousBegin_ * sampleRate_);
219  s32 end = (s32)rint(continuousEnd_ * sampleRate_);
220  if (nCoefficients_ != 0) {
221  if (continuousBegin_ != 0 || continuousEnd_ != 0) {
222  warning("Ambiguous parameters: begin=%f, end=%f vs. nr-coefficients=%zd.\n"
223  "Nr-coefficients will be used.",
224  continuousBegin_, continuousEnd_, nCoefficients_);
225  }
226  begin = 0;
227  end = nCoefficients_;
228  }
229  setBegin(begin);
230  setEnd(end);
231  needInit_ = false;
232 }
