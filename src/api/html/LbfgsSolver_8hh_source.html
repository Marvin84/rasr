84 template<class S, typename T>
85 void LbfgsSolver<S, T>::setRhos() {
86  require_eq(gradients_.size(), iterates_.size());
87  require_gt(iterates_.size(), 1);
88  int nIterates = iterates_.size();
89  rhos_.resize(nIterates - 1);
90  int m = getHistorySize();
91  for (int i = nIterates - 2; i >= nIterates - 1 - m; i--)
92  setRhoi(i);
93 }

95 template<class S, typename T>
96 void LbfgsSolver<S, T>::setRhoi(u32 i) {
97  require_lt(i, iterates_.size() - 1);
98  require_lt(i, rhos_.size());
99  T rhoInverse = dotyi(*iterates_.at(i + 1), i);
100  rhoInverse -= dotyi(*iterates_.at(i), i);
101  rhos_.at(i) = 1.0 / rhoInverse;
102 }

104 template<class S, typename T>
105 void LbfgsSolver<S, T>::firstLoop(S& out) {
106  require_ge(iterates_.size(), 2);
107  int nIterates = iterates_.size();
108  alphas_.resize(nIterates - 1);
109  int m = getHistorySize();
110  for (int i = nIterates - 2; i >= nIterates - 1 - m; i--) {
111  alphas_.at(i) = rhos_.at(i) * dotsi(out, i);
112  addyi(i, -alphas_.at(i), out);
113  }
114 }

116 template<class S, typename T>
117 void LbfgsSolver<S, T>::secondLoop(S& out) {
118  require_ge(gradients_.size(), 2);
119  require_eq(rhos_.size(), gradients_.size() - 1);
120  int nIterates = gradients_.size();
121  int m = getHistorySize();
122  for (int i = nIterates - 1 - m; i < nIterates - 1; i++) {
123  T beta = rhos_.at(i) * dotyi(out, i);
124  addsi(i, alphas_.at(i) - beta, out);
125  }
126 }

128 template<class S, typename T>
129 void LbfgsSolver<S, T>::solve(const S& in, S& out) {
130  require_eq(iterates_.size(), gradients_.size());
131  out.copy(in);
132  if (iterates_.size() <= 1) // no preconditioning possible
133  return;
134  // compute rhos
135  setRhos();
136  // run first loop
137  firstLoop(out);
138  // apply initial Hessian approximation
139  multiplyWithInitialApproximation(out);
140  // run second loop
141  secondLoop(out);
142 }

144 template<class S, typename T>
145 void LbfgsSolver<S, T>::multiplyWithInitialApproximation(S& vector) const {
146  require_ge(iterates_.size(), 2);
147  require_ge(gradients_.size(), 2);
148  int nIterates = iterates_.size();
149  T factor = dotyi(*iterates_.at(nIterates - 1), nIterates - 2);
150  factor -= dotyi(*iterates_.at(nIterates - 2), nIterates - 2);
151  T den = dotyi(*gradients_.at(nIterates - 1), nIterates - 2);
152  den -= dotyi(*gradients_.at(nIterates - 2), nIterates - 2);
153  factor /= den;
154  vector.scale(factor);
155 }
