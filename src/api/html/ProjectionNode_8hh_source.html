93 template<class T>
94 bool ProjectionNode<T>::work(Flow::PortId p) {
95  Flow::DataPtr<Flow::Vector<T>> in;
96  if (!getData(0, in))
97  return putData(0, in.get());
98 
99  Flow::Vector<T>* out = new Flow::Vector<T>;
100 
101  //loop over components
102  u32 cmp = 0;
103  for (u32 i = 0; i < nComponents_; i++) {
104  cmp = components_[i];
105  out->push_back((*in)[cmp]);
106  }
107  out->setTimestamp(*in);
108  return putData(0, out);
109 }

111 template<typename T>
112 void ProjectionNode<T>::loadComponents(const std::string& filename) {
113  if (filename.empty()) {
114  error() << "components filename is empty.";
115  }
116  else {
117  components_.clear();
118  nComponents_ = 0;
119  s32 cmp = 0;
120 
121  Core::TextInputStream tis(filename);
122  if (!tis.good()) {
123  error() << "failed to read from components file";
124  }
125  while (tis >> cmp) {
126  if (nComponents_ >= 1) {
127  require(components_.back() < cmp);
128  }
129  components_.push_back(cmp);
130  nComponents_++;
131  }
132  }
133 }
