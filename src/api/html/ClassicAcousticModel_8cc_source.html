71 bool ClassicAcousticModel::loadStateModel() {
72  HmmTopologySet* hmmTopologySet =
73  new HmmTopologySet(
74  select("hmm"),
75  silence_);
76  HmmTopologySetRef hmmTopologySetRef(hmmTopologySet);
77  Phonology* phonology =
78  new Phonology(
79  select("phonology"),
80  lexiconRef_->phonemeInventory());
81  phonology->setCrossWord(hmmTopologySetRef->isAcrossWordModelEnabled());
82  ConstPhonologyRef phonologyRef(phonology);
83  AllophoneAlphabet* allophoneAlphabet =
84  new AllophoneAlphabet(
85  select("allophones"),
86  phonologyRef,
87  lexiconRef_);
88  ConstAllophoneAlphabetRef allophoneAlphabetRef(allophoneAlphabet);
89  AllophoneStateAlphabet* allophoneStateAlphabet =
90  new AllophoneStateAlphabet(
91  select("allophone-states"),
92  allophoneAlphabetRef,
93  hmmTopologySetRef);
94  ConstAllophoneStateAlphabetRef allophoneStateAlphabetRef(allophoneStateAlphabet);
95  stateModelRef_ = ClassicStateModelRef(
96  new ClassicStateModel(
97  phonologyRef,
98  allophoneAlphabetRef,
99  allophoneStateAlphabetRef,
100  hmmTopologySetRef));
101  determineSilenceAllophoneStateIndex();
102  return true;
103 }
