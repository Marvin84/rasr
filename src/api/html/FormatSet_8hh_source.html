30 /**
31  * Set of format-qualifiers.
32  * Supports filenames with format qualifiers, e.g. bin:foo.matrix, xml:foo.vector, etc.
33  * Usage:
34  * 1) Instantiate this class in every library, tipically in Application classes.
35  * 2) For your class Foo, derive from the Format<Foo> class your own formats, e.g.
36  * FooFormat : FormatSet::Format<Foo> {..}.
37  * 3) Implement the required read/write methods in FooFormat class.R
38  * 4) Register the your format at your library FormatSet object.
39  * 5) To read or store foo objects call the I/O function of the fomat set,
40  * e.g. Application::us()->formatSet()->read("xml:foo.xml", foo);
41  * Note:
42  * -You can have different types of format objects in one format set.
43  */

155 template<class T>
156 void FormatSet::getTypeSpecificFormats(Ref<TypeSpecificFormats<T>>& typeSpecificFormats) {
157  NameHelper<T> nameHelper;
158  Formats::iterator f = formats_.find(nameHelper);
159  if (f == formats_.end())
160  f = formats_.insert(std::make_pair(nameHelper, ref(new TypeSpecificFormats<T>()))).first;
161  typeSpecificFormats = ref(dynamic_cast<TypeSpecificFormats<T>*>(f->second.get()));
162  require(typeSpecificFormats);
163 }

174 template<class T>
175 void FormatSet::registerFormat(const std::string& qualifier,
176  const Format<T>* format,
177  bool isDefault) {
178  Ref<TypeSpecificFormats<T>> formats;
179  getTypeSpecificFormats(formats);
180  require(formats->find(qualifier) == formats->end());
181  (*formats)[qualifier] = ref(format);
182  if (isDefault)
183  registerFormat(defaultQualifier, format, false);
184 }

186 template<class T>
187 bool FormatSet::getFormat(const std::string& qualifier,
188  Ref<const Format<T>>& format) const {
189  Ref<TypeSpecificFormats<T>> formats;
190  getTypeSpecificFormats(formats);
191  std::string q(qualifier);
192  if (q.empty())
193  q = defaultQualifier;
194 
195  typename TypeSpecificFormats<T>::const_iterator found = formats->find(q);
196  if (found == formats->end()) {
197  NameHelper<T> nameHelper;
198  error("Format '%s' is not supported by type '%s'.", q.c_str(), nameHelper.c_str());
199  return false;
200  }
201  format = found->second;
202  return true;
203 }
