57 template<typename T>
58 bool NetworkTopology<T>::addConnection(const std::string& layerNameFrom, const std::string& layerNameTo, u32 targetPortId) {
59  if (layerMapping_.find(layerNameFrom) == layerMapping_.end())
60  return false;
61  if (layerMapping_.find(layerNameTo) == layerMapping_.end())
62  return false;
63 
64  NetworkTopologyElement<T>* nodeFrom = &(layers_.at(layerMapping_[layerNameFrom]));
65  s32 nodeTo = layerMapping_[layerNameTo];
66  // add connection
67  nodeFrom->addConnection(nodeTo);
68  // add predecessor
69  if (!layers_.at(nodeTo).addPredecessor(nodeFrom->id(), targetPortId))
70  return false;
71  return true;

85 template<typename T>
86 void NetworkTopology<T>::setTopologicalOrdering(std::vector<NeuralNetworkLayer<T>*>& result) {
87  std::vector<bool> discovered(nLayers(), false);
88  std::vector<bool> finished(nLayers(), false);
89 
90  std::vector<NetworkTopologyElement<T>*> topologicalOrdering;
91  for (u32 layerIdx = 0; layerIdx < layers_.size(); ++layerIdx) {
92  if (!discovered.at(layerIdx)) {
93  dfsTopologicalOrderingVisit(discovered, finished, topologicalOrdering, layerIdx);
94  }
95  }
96 
97  std::reverse(topologicalOrdering.begin(), topologicalOrdering.end());
98 
99  result.clear();
100  for (u32 i = 0; i < topologicalOrdering.size(); i++) {
101  topologicalOrdering.at(i)->setTopologicalId(i);
102  result.push_back(topologicalOrdering.at(i)->layer());
103  }
104 }

106 template<typename T>
107 void NetworkTopology<T>::dfsTopologicalOrderingVisit(std::vector<bool>& discovered,
108  std::vector<bool>& finished,
109  std::vector<NetworkTopologyElement<T>*>& topologicalOrdering,
110  u32 layerIdx) {
111  verify_lt(layerIdx, nLayers());
112 
113  discovered.at(layerIdx) = true;
114  NetworkTopologyElement<T>* layer = &layers_.at(layerIdx);
115 
116  for (u32 i = 0; i < layer->nConnections(); i++) {
117  NetworkTopologyElement<T>* target = &layers_.at(layer->connection(i));
118  s32 targetId = target->id();
119  if (!discovered.at(targetId)) {
120  dfsTopologicalOrderingVisit(discovered, finished, topologicalOrdering, targetId);
121  }
122  }
123  finished.at(layerIdx) = true;
124  topologicalOrdering.push_back(layer);
125 }
