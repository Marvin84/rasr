36 Speech::PosteriorFsa MmiSegmentwiseNnTrainer<T>::getDenominatorPosterior(Lattice::ConstWordLatticeRef lattice) {
37  Speech::PosteriorFsa result;
38  result.fsa = Fsa::posterior64(Fsa::changeSemiring(lattice->part(this->part_), Fsa::LogSemiring),
39  result.totalInv,
40  this->posteriorTolerance());
41  if (Core::isAlmostEqualUlp(f32(result.totalInv), Core::Type<f32>::min, this->posteriorTolerance())) {
42  this->log("discard segment because it has vanishing total flow");
43  return Speech::PosteriorFsa();
44  }
45  result.fsa = Fsa::expm(result.fsa);

62 template<typename T>
63 bool MmiSegmentwiseNnTrainer<T>::computeInitialErrorSignal(Lattice::ConstWordLatticeRef lattice, Lattice::ConstWordLatticeRef numeratorLattice,
64  Bliss::SpeechSegment* segment, T& objectiveFunction, bool objectiveFunctionOnly) {
65  require(numeratorLattice);
66  Speech::PosteriorFsa numeratorPosterior;
67  Speech::PosteriorFsa denominatorPosterior;
68  denominatorPosterior = getDenominatorPosterior(lattice);
69  if (!denominatorPosterior) {
70  this->log("failed to compute denominator posterior FSA, skipping segment");
71  return false;
72  }
73  objectiveFunction = f32(denominatorPosterior.totalInv);
74  u32 nRejectedObsInSeq = 0;
75  if (!objectiveFunctionOnly) {
76  this->accumulateStatisticsOnLattice(denominatorPosterior.fsa, lattice->wordBoundaries(), 1.0);
77  // frame rejection heuristic described in Vesely et al: Sequence-discriminative training of DNNs, in Interspeech 2013
78  if (this->frameRejectionThreshold_ > 0) {
79  for (u32 t = 0; t < alignment_.size(); ++t) {
80  verify_ge(errorSignal_.back().at(alignment_.at(t), t), 0);
81  if (errorSignal_.back().at(alignment_.at(t), t) < this->frameRejectionThreshold_) {
82  weights_.at(t) = 0.0;
83  ++nRejectedObsInSeq;
84  }
85  }
86  }
87  }
88  this->numberOfRejectedObservations_ += nRejectedObsInSeq;
89  this->log("denominator-lattice-objective-function: ") << objectiveFunction;
90 
91  numeratorPosterior = getNumeratorPosterior(numeratorLattice);
92  if (!numeratorPosterior) {
93  this->log("failed to compute numerator posterior FSA, skipping segment");
94  return false;
95  }
96  objectiveFunction -= f32(numeratorPosterior.totalInv);
97  if (!objectiveFunctionOnly)
98  this->accumulateStatisticsOnLattice(numeratorPosterior.fsa, numeratorLattice->wordBoundaries(), -1.0);
99 
100  if (!objectiveFunctionOnly && this->frameRejectionThreshold_ > 0)
101  this->log("rejected ") << nRejectedObsInSeq << " out of " << alignment_.size() << " observations (" << 100.0 * nRejectedObsInSeq / alignment_.size() << "%)";
102  this->log("numerator-lattice-objective-function: ") << f32(numeratorPosterior.totalInv);
103  this->log("MMI-objective-function: ") << objectiveFunction;
104  return true;
105 }
