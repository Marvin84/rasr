176 template<>
177 inline void FeedForwardTrainer<f32>::initializeDoublePrecisionStatistics() {
178  u32 statisticsType = estimator().requiredStatistics();
179  if (estimator().fullBatchMode() || statisticsChannel_.isOpen())
180  statisticsType |= Statistics<f32>::BASE_STATISTICS;
181  doublePrecisionStatistics_ = new Statistics<f64>(network().nLayers(), statisticsType);
182  doublePrecisionStatistics_->copyStructure(statistics());
183  doublePrecisionStatistics_->initComputation();
184 }
