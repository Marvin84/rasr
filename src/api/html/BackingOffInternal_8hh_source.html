31 const Node* binarySearch(const Node* l, const Node* r, TokenIndex t) {
32  while (l <= r) {
33  const Node* m = l + (r - l) / 2;
34  if (t < m->token()) {
35  r = m - 1;
36  }
37  else if (t > m->token()) {
38  l = m + 1;
39  }
40  else /* t == m->token() */ {
41  return m;
42  }
43  }
44  return 0;
45 }
