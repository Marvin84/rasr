48  void startFsa(const Core::XmlAttributes atts) {
49  const char* tmp = atts["type"];
50  if (!tmp) {
51  parser()->warning("no type specified. set type to transducer.");
52  fsa_->setType(Fsa::TypeTransducer);
53  }
54  else {
55  Core::Choice::Value type = Fsa::TypeChoice[std::string(tmp)];
56  if (type != Core::Choice::IllegalValue)
57  fsa_->setType(Fsa::Type(type));
58  else {
59  parser()->warning("unknown type \"%s\". set type to transducer.", tmp);
60  fsa_->setType(Fsa::TypeTransducer);
61  }
62  }
63 
64  _ConstSemiringRef semiring;
65  tmp = atts["semiring"];
66  if (tmp) {
67  semiring = resources_.getSemiring(tmp);
68  if (semiring)
69  fsa_->setSemiring(semiring);
70  else
71  parser()->warning("%s semiring not found", tmp);
72  }
73  if (!(semiring)) {
74  if (fsa_->semiring())
75  parser()->warning("semiring not specified or not found. use %s semiring.",
76  fsa_->semiring()->name().c_str());
77  else
78  parser()->error("semiring not specified or not found. no semiring set.");
79  }
80 
81  tmp = atts["initial"];
82  if (!tmp) {
83  parser()->warning("no initial state id specified for fsa.");
84  fsa_->setInitialStateId(Fsa::InvalidStateId);
85  }
86  else {
87  char* end;
88  Fsa::StateId id = strtoll(tmp, &end, 0);
89  if (*end != '\0') {
90  parser()->error("invalid initial state id \"%s\"", tmp);
91  fsa_->setInitialStateId(Fsa::InvalidStateId);
92  }
93  else
94  fsa_->setInitialStateId(id);
95  }
96  }

215  StorageAutomatonXmlParser(const _Resources& resources, _StorageAutomaton* fsa)
216  : Precursor(resources.getConfiguration()), resources_(resources), fsa_(fsa) {
217  fsa_->setType(Fsa::TypeTransducer);
218  Core::XmlMixedElement* arcElement = new Core::XmlMixedElementRelay("arc", this, startHandler(&Self::startArc), 0, 0,
219  XML_CHILD(new Core::XmlMixedElementRelay("in", this, startHandler(&Self::startIn),
220  endHandler(&Self::endIn), charactersHandler(&Self::charactersFsa),
221  XML_NO_MORE_CHILDREN)),
222  XML_CHILD(new Core::XmlMixedElementRelay("out", this, startHandler(&Self::startOut),
223  endHandler(&Self::endOut), charactersHandler(&Self::charactersFsa),
224  XML_NO_MORE_CHILDREN)),
225  XML_CHILD(new Core::XmlMixedElementRelay("weight", this, startHandler(&Self::startArcWeight),
226  endHandler(&Self::endArcWeight), charactersHandler(&Self::charactersFsa),
227  XML_NO_MORE_CHILDREN)),
228  XML_NO_MORE_CHILDREN);
229  Core::XmlMixedElement* stateElement = new Core::XmlMixedElementRelay("state", this, startHandler(&Self::startState), endHandler(&Self::endState), 0,
230  XML_CHILD(new Core::XmlEmptyElementRelay("user", this, startHandler(&Self::startUser))),
231  XML_CHILD(new Core::XmlEmptyElementRelay("final", this, startHandler(&Self::startFinal))),
232  XML_CHILD(new Core::XmlMixedElementRelay("weight", this, startHandler(&Self::startFinalWeight),
233  endHandler(&Self::endFinalWeight), charactersHandler(&Self::charactersFsa),
234  XML_NO_MORE_CHILDREN)),
235  arcElement,
236  XML_NO_MORE_CHILDREN);
237 
238  if (fsa_->getInputAlphabet()) {
239  if (fsa_->type() == Fsa::TypeTransducer)
240  verify(fsa_->getOutputAlphabet());
241  Core::XmlMixedElement* rootElement = new Core::XmlMixedElementRelay("fsa", this, startHandler(&Self::startFsa), 0, 0,
242  XML_CHILD(new Core::XmlIgnoreElement("input-alphabet", this)),
243  XML_CHILD(new Core::XmlIgnoreElement("output-alphabet", this)),
244  stateElement,
245  XML_NO_MORE_CHILDREN);
246  setRoot(rootElement);
247  }
248  else {
249  Core::Ref<Fsa::StaticAlphabet> input(new Fsa::StaticAlphabet), output(new Fsa::StaticAlphabet);
250  fsa_->setInputAlphabet(input);
251  if (fsa_->type() == Fsa::TypeTransducer)
252  fsa_->setOutputAlphabet(output);
253  Core::XmlMixedElement* rootElement = new Core::XmlMixedElementRelay("fsa", this, startHandler(&Self::startFsa), 0, 0,
254  XML_CHILD(new Fsa::AlphabetXmlParser("input-alphabet", this, input)),
255  XML_CHILD(new Fsa::AlphabetXmlParser("output-alphabet", this, output)),
256  stateElement,
257  XML_NO_MORE_CHILDREN);
258  setRoot(rootElement);
259  }
260  }
