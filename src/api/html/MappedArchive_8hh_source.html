152 template<class T>

155  ConstantVector()

159  : data_(mapped), size_(size) {}

162  : editable_(data),

165  }
166 
167  inline const T& operator[](size_t index) const {
168  return data_[index];
169  }
170 
171  /// Edit an element of the vector. After editing an element,
172  /// the data is not constant any more.
173  T& edit(size_t index) {
174  makeEditable();
175  return editable_[index];
176  }
177 
178  size_t size() const {
179  return size_;
180  }
181 
182  const T* data() const {
183  return data_;
184  }
185 
186  typedef const T* const_iterator;
187 
188  const_iterator begin() const {
189  return data_;
190  }
191 
192  const_iterator end() const {
193  return data_ + size_;
194  }
195 
196  /// Resize the vector. After resizing the internal data
197  /// is not constant and shared any more.
198  void resize(size_t size, const T& value = T()) {
199  if (size_ == size)
200  return;
201  makeEditable();
202  editable_.resize(size, value);
203  size_ = size;
204  data_ = editable_.data();
205  }
206 
207  /// Add an element to the vector. Afterwards,
208  /// the data is not constant and shared any more.
209  void push_back(const T& value) {
210  resize(size_ + 1, value);
211  }
212 
213  const T& back() const {
214  return *(data_ + size_ - 1);
215  }
216 
217  bool isConstant() const {
218  return data_ != editable_.data();
219  }
220 
221 private:
222  void makeEditable() {
223  if (isConstant()) {
224  editable_.resize(size_);
225  memcpy(editable_.data(), data_, size_ * sizeof(T));
226  data_ = editable_.data();
227  }
228  }
229 
230  std::vector<T> editable_;
231  const T* data_;
232  size_t size_;
233 };
