60 /**
61  * Calculates the n best paths through a weighted automaton.
62  * The result is an acyclic automaton where all state except the
63  * initial one have only a single successor:
64  * 0: initial state
65  * 1 .. n: start states for n-best paths
66  * So you can extract the i-th best path using partial(i).
67  * Category: on-demand with precalculations
68  * Complexity: O(sssp(determinize(remove-epsilons(f)))
69  * @param f the original automaton
70  * @param n is an upper limit on the number of paths to generate.
71  * The actual number of paths in the result automaton may be
72  * smaller (if the original automaton does not contain that many
73  * paths.) The actual number of n-bests can be determined by
74  * counting the number of arcs leaving the initial state.
75  * @param bestSequences set to false if you want to calculate the n-best
76  * paths instead of the n-best unique sequences, true by default
77  * @return an automaton representing the n-best sequences
78  **/
79 template<class _Automaton>
80 typename _Automaton::ConstRef nbest(typename _Automaton::ConstRef f, size_t n = 1, bool bestSequences = true);
