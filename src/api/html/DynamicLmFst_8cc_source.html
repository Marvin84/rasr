23 class DynamicLmFstScoreCache {
24 public:
25  typedef std::vector<float> Cache;
26  DynamicLmFstScoreCache(size_t maxElements)
27  : maxElements_(maxElements), elements_(0) {}
28  ~DynamicLmFstScoreCache();
29  Cache* Get(u32 s) {
30  if (s < data_.size()) {
31  data_[s].second = true;
32  return data_[s].first;
33  }
34  else {
35  return 0;
36  }
37  }
38  void Set(u32 s, Cache* cache) {
39  if (s < data_.size() && data_[s].first) {
40  delete data_[s].first;
41  data_[s].first = cache;
42  }
43  else {
44  if ((elements_ + 1) > maxElements_)
45  CleanCache(false);
46  ++elements_;
47  if (s >= data_.size())
48  data_.resize(s + 1, Element(0, false));
49  data_[s].first = cache;
50  }
51  data_[s].second = true;
52  }
53  void Clear() {
54  elements_ = 0;
55  for (std::vector<Element>::iterator i = data_.begin(); i != data_.end(); ++i)
56  delete i->first;
57  data_.clear();
58  }
59  u32 Size() const {
60  return elements_;
61  }
62 
63 private:
64  typedef std::pair<Cache*, bool> Element;
65  void CleanCache(bool);
66  size_t maxElements_;
67  size_t elements_;
68  std::vector<Element> data_;
69 
70  void operator=(const DynamicLmFstScoreCache&);
71  DynamicLmFstScoreCache(const DynamicLmFstScoreCache&);
72 };

79 void DynamicLmFstScoreCache::CleanCache(bool freeRecent) {
80  size_t targetSize = (2 * maxElements_) / 3 + 1;
81  std::vector<Element>::iterator i = data_.begin();
82  while (i != data_.end() && elements_ > targetSize) {
83  if (i->first) {
84  if (freeRecent || !i->second) {
85  delete i->first;
86  i->first = 0;
87  --elements_;
88  }
89  i->second = false;
90  }
91  ++i;
92  }
93  if (!freeRecent && elements_ > targetSize)
94  CleanCache(true);
95 }

229 void DynamicLmFstImpl::CreateArc(StateId source, Label label, Arc* arc, bool cache) {
230  if (GetCachedArc(source, label, arc)) {
231  ++nCached_;
232  return;
233  }
234  const ScoreCache* scores = GetScores(source);
235  ++nCalculated_;
236  Lm::History history = state2History_[source];
237  Label wordLabel = GetLabel(label);
238  verify(wordLabel != FstLib::kNoLabel);
239  Lm::Score score = 0;
240  const Bliss::SyntacticTokenSequence& tokenSequence = SyntacticToken(wordLabel, &score);
241  for (u32 ti = 0; ti < tokenSequence.length(); ++ti) {
242  const Bliss::SyntacticToken* st = tokenSequence[ti];
243  score += scores ? scores->at(label) : lm_->score(history, st);
244  history = lm_->extendedHistory(history, st);
245  }
246  arc->ilabel = label;
247  arc->olabel = wordLabel;
248  arc->weight = score;
249  arc->nextstate = GetState(history);
250  if (cache) {
251  CacheArc(source, *arc);
252  }
253 }

271  if (!map.empty()) {
272  relabeling_.resize(InputSymbols()->NumSymbols(), FstLib::kNoLabel);
