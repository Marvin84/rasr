78 Fsa::ConstAutomatonRef ClassicStateTying::createMixtureToAllophoneStateTransducer(s32 nDisambiguators) const {
79  ConstEmissionAlphabetRef emissionAlphabet = ConstEmissionAlphabetRef(new EmissionAlphabet(nClasses()));
80 
81  Fsa::StaticAutomaton* f = new Fsa::StaticAutomaton(Fsa::TypeTransducer);
82  f->setSemiring(Fsa::TropicalSemiring);
83  f->setInputAlphabet(emissionAlphabet);
84  f->setOutputAlphabet(alphabetRef_);
85  f->setProperties(Fsa::PropertyAcyclic | Fsa::PropertyLinear, Fsa::PropertyNone);
86  const Fsa::Weight One(f->semiring()->one());
87 
88  Fsa::State *initial, *final, *root;
89  if (nDisambiguators > 0) {
90  initial = final = f->newState();
91  root = f->newState();
92  for (Fsa::LabelId d = 0; d < Fsa::LabelId(nDisambiguators); ++d) {
93  initial->newArc(root->id(), One, emissionAlphabet->disambiguator(d), alphabetRef_->disambiguator(d));
94  root->newArc(final->id(), One, emissionAlphabet->disambiguator(d), alphabetRef_->disambiguator(d));
95  }
96  }
97  else
98  initial = final = root = f->newState();
99  f->setInitialStateId(initial->id());
100  final->setFinal(One);
101 
102  for (std::pair<AllophoneStateIterator, AllophoneStateIterator> it = alphabetRef_->allophoneStates(); it.first != it.second; ++it.first)
103  root->newArc(root->id(), One, classify(it.first.allophoneState()), it.first.id());
104  return Fsa::ConstAutomatonRef(f);
105 }

137 bool LutStateTying::loadLut(const std::string& filename) {
138  /*
139  * nClasses_ and lut_ are set
140  */
141  verify(!filename.empty());
142  Core::CompressedInputStream cis(filename);
143  if (!cis.isOpen()) {
144  criticalError("cannot open lookup table '%s'", filename.c_str());
145  }
146  std::string line;
147  Mm::MixtureIndex topMixtureId = 0;
148  Core::ProgressIndicator* progress = new Core::ProgressIndicator("state tying lookup from file");
149  progress->start();
150  u32 nLine = 0;
151  while (cis) {
152  ++nLine;
153  std::getline(cis, line);
154  Core::stripWhitespace(line);
155  if (!line.empty() && (*line.c_str() != '#')) {
156  Mm::MixtureIndex mixtureId;
157  std::string allophoneStateString;
158  std::vector<std::string> fields = Core::split(line, " ");
159  if (fields.size() != 2) {
160  criticalError("failure while reading line %d from file \"%s\"", nLine, filename.c_str());
161  return false;
162  }
163  allophoneStateString = fields[0],
164  Core::strconv(fields[1], mixtureId);
165  AllophoneStateIndex allophoneStateId = alphabetRef_->index(allophoneStateString);
166 
167  // TODO: check if already set
168  lut_[allophoneStateId] = mixtureId;
169  if (mixtureId > topMixtureId)
170  topMixtureId = mixtureId;
171 
172  progress->notify();
173  }
174  }
175  nClasses_ = topMixtureId + 1;
176  progress->finish();
177  delete progress;
178  return true;
179 }

188  : Core::Component(config),

211 Mm::MixtureIndex NoStateTyingDense::classify(const AllophoneState& a) const {
212  require_lt(a.allophone()->boundary, numBoundaryClasses_);
213  require_le(0, a.state());
214  require_lt(u32(a.state()), numStates_);
215  u32 result = 0;
216 
217  u32 phoneIdx = a.allophone()->phoneme(0);
218  require_lt(phoneIdx, numPhoneClasses_);
219  result += phoneIdx;
220 
221  result *= numStates_;
222  result += u32(a.state());
223 
224  for (u32 i = 1; i < 2 * contextLength_ + 1; ++i) { // context len is usually 1
225  // pos sequence: -1, 1, [-2, 2, ...]
226  s16 pos = i / 2;
227  if (i % 2 == 1)
228  pos = -pos - 1;
229  result *= numPhoneClasses_;
230  u32 phoneIdx = a.allophone()->phoneme(pos);
231  require_lt(phoneIdx, numPhoneClasses_);
232  result += phoneIdx;
233  }
234 
235  if (useBoundaryClasses_){
236  result *= numBoundaryClasses_;
237  result += a.allophone()->boundary;
238  }
239  require_lt(result, nClasses_);
240 
241  return result;
242 }
