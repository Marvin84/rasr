147 /** Class template for Data smart pointers
148  * Extends the features of Core::TsRef by:
149  * -up and down cast using dynamic_cast
150  * -makePrivate() which ensures that the object is referenced only by this DataPtr.
151  *
152  * Note: for comments more important, please @see Core::Ref and Core::TsRef.
153  *
154  * Remark: Objects referenced by DataPtr can be referenced by Core::TsRef as well.
155  * However, in certain applications multithread safety is not important, thus
156  * usage of the mutex_ is wasted time. In these application, thread safe object
157  * needs to be copied into a object derived from ReferenceCounted.
158  * For an examles @see Speech::Feature.
159  */
160 template<class T>
161 class DataPtr : public Core::TsRef<T> {
162  typedef Core::TsRef<T> Precursor;
163  typedef typename Precursor::Object Object;
164  friend class Link;
165  friend class AbstractNode;
166 
167 private:
168  void take(Data* p) {
169  this->unreference();
170  Precursor::object_ = convert(p);
171  }
172 
173  template<class S>
174  static Object* convert(S* o) {
175  if (Object::isSentinel(o))
176  return static_cast<Object*>(o);
177  Object* result = dynamic_cast<Object*>(o);
178  return (result == 0) ? Precursor::sentinel() : result;
179  }
180 
181 public:
182  DataPtr()
183  : Precursor() {}
184  DataPtr(const DataPtr& p)
185  : Precursor(p) {}
186  template<class S>
187  explicit DataPtr(const DataPtr<S>& p) {
188  Precursor::object_ = convert(p._get());
189  Precursor::reference();
190  }
191  explicit DataPtr(Object& d) {
192  Precursor::object_ = dynamic_cast<T*>(d->clone());
193  ensure_(Precursor::object_ != 0);
194  Precursor::reference();
195  }
196  explicit DataPtr(Object* d)
197  : Precursor(d) {}
198  virtual ~DataPtr() {}
199 
200  template<class S>
201  DataPtr& operator=(const DataPtr<S>& rhs) {
202  this->set(convert(rhs._get()));
203  return *this;
204  }

206  /** Test for identity
207  * Remark: overriding of the precursor function is necessary because
208  * in comparisons the compiler uses Precursor::operator bool() instead of Precursor::operator==.
209  */
210  bool operator==(const DataPtr& r) const {
211  return Precursor::operator==(r);
212  }
213  template<class S>
214  bool operator==(const DataPtr<S>& r) const {
215  return Precursor::operator==(r);
216  }
217  bool operator==(const Data* o) const {
218  return Precursor::object_ == o;
219  }
220 
221  Object* get() const {
222  return Precursor::_get();
223  }
224 
225  Object* release() {
226  Object* o = Precursor::object_;
227  Precursor::object_ = this->sentinel();
228  Precursor::reference();
229  if (Object::isSentinel(o)) {
230  o->decrement();
231  return 0;
232  }
233  else {
234  if (!o->decrement()) {
235  // If reference count of 'd' is larger than 0 then
236  // the object may get destroyed by one of the other owners.
237  // Use makePrivate() before calling release.
238  defect();
239  }
240  return o;
241  }
242  }
243 
244  void makePrivate() {
245  if (Object::isSentinel(Precursor::object_))
246  return;
247  Precursor::object_->lock();
248  Object* o = Precursor::object_;
249  if (o->refCount() > 1) {
250  Precursor::object_ = dynamic_cast<Object*>(o->clone());
251  ensure_(Precursor::object_ != 0);
252  Precursor::object_->increment();
253  o->decrement();
254  }
255  o->release();
256  }
257 };
