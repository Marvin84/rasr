66 /** Initialize the Gaussian filters.
67  *
68  * u = (cMax - 1) / 2 => 50
69  * c = 0..cMax
70  * o[i] = {0.8, 1.2, 1.8, 2.7, 4.0, 6.0, 9.0, 13.0}
71  *
72  * G0 = exp( -(c-u)^2 / (2 o[i]^2 ) )
73  * G1 = -(c-u) / o[i]^2 * G0
74  * G2 = ( (c-u)^2 / s[i]^4 - 1 / o[i]^2 ) * G0
75  */
76 void MrastaFiltering::initGaussianFilters(size_t nFilter, size_t nFrames) {
77  // if necessary, resize the filter matrices
78  resizeFilters(nFilter, nFrames);
79 
80  // fill the filter with the corresponding Gaussian values
81  for (int i = 0; i < (int)nFilter; i++) {
82  for (int j = 0; j < (int)nFrames; j++) {
83  // Gaussian filter (G0)
84  getG0()[i][j] = exp(
85  -(Value)((j - (int)(nFrames / 2)) * (j - (int)(nFrames / 2)) / (2.0 * sigma(i) * sigma(i))));
86  // first (time) derivative of the Gaussian filter (G1)
87  getG1()[i][j] = (Value) - ((j - (int)(nFrames / 2)) / (sigma(i) * sigma(i))) * getG0()[i][j];
88 
89  // second derivative of the Gaussian filter (G2)
90  getG2()[i][j] = ((Value)(j - (int)(nFrames / 2)) * (j - (int)(nFrames / 2)) / (sigma(i) * sigma(i) * sigma(i) * sigma(i)) - 1.0 / (sigma(i) * sigma(i))) * getG0()[i][j];
91  }
92  // divide each Gaussian filter (G1, G2) by absolute maximum (over frames)
93  normalizeFilterResponse(getG1(), i);
94  normalizeFilterResponse(getG2(), i);
95  }
96 }

106 inline void MrastaFiltering::normalizeFilterResponse(Math::Matrix<Value>& G, int filter_num) {
107  Math::Vector<Value> vector = G.row(filter_num);
108  // max(|v|) = max(-min(v), max(v))
109  Value maxAbsVal = std::max(-(Value)*std::min_element(vector.begin(), vector.end()),
110  *std::max_element(vector.begin(), vector.end()));
111  // normalize vector and write back into the matrix
112  std::transform(vector.begin(), vector.end(), vector.begin(),
113  std::bind2nd(std::multiplies<Value>(), (Value)1 / maxAbsVal));
114  G.setRow(filter_num, vector);
115 }

141 void MrastaFiltering::appendDerivatives(std::vector<Value>& in) {
142  // Pointer to start of the values for the derivative calculation
143  Math::Vector<Value>::iterator itBand;
144  // Pointer to the start of the derivatives
145  Math::Vector<Value>::iterator itDeriv;
146 
147  // add first derivative
148  if (nDerivatives_ > 0) {
149  // set the iterators
150  itBand = in.begin();
151  itDeriv = in.begin() + (int)nFeatures_ * 2 * nFilters_;
152 
153  // for each Gaussian filter [(G1+G2) * nFilter()]
154  for (int filter = 0; filter < 2 * (int)nFilters_; filter++) {
155  // first derivative (no derivatives for first/last band)
156  for (int i = 1; i < (int)nFeatures_ - 1; i++, itDeriv++, itBand++)
157  *itDeriv = *(itBand + 2) - *itBand;
158 
159  // set the iterator to start of next band (no derivatives for first last band)
160  itBand += 2;
161  }
162  }
163 
164  // add second derivative
165  if (nDerivatives_ > 1) {
166  // reset the iterators (itDeriv should be correct)
167  itBand = in.begin();
168  itDeriv = in.begin() + (int)2 * nFilters_ * (nFeatures_ + (nFeatures_ - 2));
169 
170  // for each Gaussian filter [(G1+G2) * nFilter()]
171  for (int filter = 0; filter < 2 * (int)nFilters_; filter++) {
172  // second derivative (no derivatives for first/last band)
173  for (int i = 1; i < (int)nFeatures_ - 1; i++, itDeriv++, itBand++)
174  *itDeriv = *(itBand + 2) / 2 - *(itBand + 1) + *itBand / 2;
175 
176  // set the iterator to start of next band (no derivatives for first last band)
177  itBand += 2;
178  }
179  }
180 }

182 /** apply the Gaussian filtering to the values
183  *
184  * @param in input feature vector of size nFrames(101) * nBands(19)
185  * @param out output feature vector of size nFrames(101) * 2 * nFilter(6) + derivatives
186  *
187  * @return false if initialization fails, true on success
188  * */
189 bool MrastaFiltering::apply(std::vector<Value>& in, std::vector<Value>& out) {
190  // init is requested, initialization failed?
191  if (needInit_ && !init())
192  return false;
193 
194  // get the Gaussian filter response for each band
195  for (size_t band = 0; band < nFeatures_; band++) {
196  getBand(band, in, inputBuffer_);
197  filterEnergyBand(inputBuffer_, outputBuffer_);
198  setBand(band, outputBuffer_, out);
199  }
200 
201  // apply derivatives for the Gaussian filter response
202  appendDerivatives(out);
203  return true;
204 }
