209 void AlignmentResetWeightsNode::transform(Alignment& a) {
210  if (restrict_) {
211  f64 transformation = 0;
212  u32 selected = 0, transformed = 0, alreadyTransformed = 0, skipped = 0;
213  if (mode_ != modeSmallerThan && mode_ != modeLargerThan)
214  error() << "restricted weight transformation is only implemented for 'smaller-than' and 'larger-than' modes, actual: " << mode_;
215 
216  std::cout << "mode " << mode_ << " previous " << previousWeight_ << " new " << newWeight_ << std::endl;
217  std::vector<Alignment::Frame> frames;
218  a.getFrames(frames);
219  std::vector<Alignment::Frame>::const_iterator frame_i;
220  for (frame_i = frames.begin(); frame_i != frames.end(); ++frame_i) {
221  Alignment::iterator state_i, state_i_end; //start- and end-states for timeframe i
222  for (Core::tie(state_i, state_i_end) = *frame_i; state_i != state_i_end; ++state_i) {
223  Am::AllophoneState alloState = acousticModel_->allophoneStateAlphabet()->allophoneState(state_i->emission);
224  if (selectPhones_.count(alloState.allophone()->central())) {
225  ++selected;
226  if ((mode_ == modeSmallerThan && state_i->weight < previousWeight_) ||
227  (mode_ == modeLargerThan && state_i->weight > previousWeight_)) {
228  transformation += newWeight_ - state_i->weight;
229  if (state_i->weight != newWeight_) {
230  state_i->weight = newWeight_;
231  ++transformed;
232  }
233  else {
234  ++alreadyTransformed;
235  }
236  }
237  }
238  ++skipped;
239  }
240  }
241  log() << "selected alignment items: " << selected << " transformed: " << transformed << " were already transformed: " << alreadyTransformed << " skipped: " << skipped;
242  if (transformed)
243  log() << "average alignment weight transformation: " << transformation / transformed;
244 
245  return;
246  }
247  if (mode_ == modeSmallerThan)
248  a.resetWeightsSmallerThan(Mm::Weight(previousWeight_), Mm::Weight(newWeight_));
249  else if (mode_ == modeLargerThan)
250  a.resetWeightsLargerThan(Mm::Weight(previousWeight_), Mm::Weight(newWeight_));
251  else if (mode_ == modeTiedSmallerThan ||
252  mode_ == modeTiedLargerThan) {
253  std::vector<Alignment::Frame> frames;
254  a.getFrames(frames);
255 
256  std::vector<Alignment::Frame>::const_iterator frame_i;
257  for (frame_i = frames.begin(); frame_i != frames.end(); ++frame_i) {
258  Alignment::iterator state_i, state_i_end; //start- and end-states for timeframe i
259  std::unordered_map<u32, f32> frameStateProb;
260  std::unordered_map<u32, u32> frameStateCounts;
261  for (Core::tie(state_i, state_i_end) = *frame_i; state_i != state_i_end; ++state_i) {
262  u32 stateId = acousticModel_->emissionIndex(state_i->emission);
263  if (frameStateProb.count(stateId) > 0) {
264  frameStateProb[stateId] += f32(state_i->weight);
265  frameStateCounts[stateId]++;
266  }
267  else {
268  frameStateProb[stateId] = f32(state_i->weight);
269  frameStateCounts[stateId] = 1;
270  }
271  }
272 
273  if (mode_ == modeTiedSmallerThan)
274  for (Core::tie(state_i, state_i_end) = *frame_i; state_i != state_i_end; ++state_i) {
275  u32 stateId = acousticModel_->emissionIndex(state_i->emission);
276  if (frameStateProb[stateId] < previousWeight_)
277  state_i->weight = newWeight_ / f32(frameStateCounts[stateId]);
278  }
279  if (mode_ == modeTiedLargerThan)
280  for (Core::tie(state_i, state_i_end) = *frame_i; state_i != state_i_end; ++state_i) {
281  u32 stateId = acousticModel_->emissionIndex(state_i->emission);
282  if (frameStateProb[stateId] > previousWeight_)
283  state_i->weight = newWeight_ / f32(frameStateCounts[stateId]);
284  }
285  }
286  }
287 }

344  Flow::DataPtr<Flow::DataAdaptor<Alignment>> in1;
345  Flow::DataPtr<Flow::DataAdaptor<Alignment>> in2;
346  if (!getData(0, in1)) {
347  error("could not read port 0");

356  putData(0, in1.get());

475 void AlignmentRemoveEmissionScoreNode::transform(Alignment& a) {
476  TimeframeIndex currentFeatureId = 0;
477  Alignment::iterator currentAlignmentItem = a.begin();
478  Flow::DataPtr<Feature::FlowFeature> in;
479  bool firstFeature = true;
480  while (getData(1, in)) {
481  Core::Ref<const Feature> f(new Feature(in));
482  if (firstFeature) { // try to check the dimension only once for each segment
483  checkFeatureDependencies(*f);
484  firstFeature = false;
485  }
486  Mm::FeatureScorer::Scorer scorer(acousticModel_->featureScorer()->getScorer(f));
487  while (currentAlignmentItem != a.end() && currentAlignmentItem->time == currentFeatureId) {
488  verify(currentAlignmentItem == a.begin() ? true : (currentAlignmentItem - 1)->time <= currentAlignmentItem->time);
489  f32 emScore = scorer->score(acousticModel_->emissionIndex(currentAlignmentItem->emission));
490  currentAlignmentItem->weight -= emScore;
491  ++currentAlignmentItem;
492  }
493  ++currentFeatureId;
494  }
495 }

530 bool SetAlignmentWeightsByTiedStateAlignmentWeightsNode::work(Flow::PortId p) {
531  Flow::DataPtr<Flow::DataAdaptor<Alignment>> ina;
532  Flow::DataPtr<Flow::DataAdaptor<Alignment>> inb;
533 
534  while (getData(0, ina) && getData(1, inb)) {
535  const Alignment& aa = ina->data();
536  const Alignment& bb = inb->data();
537  Alignment a(aa);
538  Alignment b(bb);
539 
540  // Sometimes the aligner may create new allophones in the allophone state alphabet.
541  // This works around that bug by removing allophones which don't match our current allophone alphabet (which is maintained separately).
542  a.setAlphabet(acousticModel_->allophoneStateAlphabet(), true);
543  b.setAlphabet(acousticModel_->allophoneStateAlphabet(), true);
544 
545  std::vector<Alignment::Frame> framesa;
546  std::vector<Alignment::Frame> framesb;
547  a.combineItems(Fsa::ProbabilitySemiring);
548  a.getFrames(framesa);
549  b.getFrames(framesb);
550 
551  std::vector<Alignment::Frame>::const_iterator i;
552  Alignment::iterator s;
553  s = b.begin();
554  for (i = framesa.begin(); i != framesa.end() && s != b.end(); ++i) {
555  Alignment::iterator j, j_end;
556 
557  s->weight = 0.0;
558  for (Core::tie(j, j_end) = *i; j != j_end; ++j) {
559  if (j->time != s->time) {
560  warning("alignment times differ: single-path=%d multi-path=%d", s->time, j->time);
561  }
562  else {
563  if (acousticModel_->emissionIndex(s->emission) == acousticModel_->emissionIndex(j->emission))
564  s->weight += j->weight;
565  }
566  }
567  s++;
568  }
569 
570  Flow::DataAdaptor<Alignment>* alignment = new Flow::DataAdaptor<Alignment>(b);
571  putData(0, alignment);
572  }
573 
574  return putData(0, ina.get());
575 }
