146 bool readBinary(StaticBoundaries* b, std::istream& is) {
147  Core::BinaryInputStream bis(is);
148  char magic[BinaryHeader::magicSize];
149  u32 version;
150  bis.read(magic, BinaryHeader::magicSize);
151  if (!bis)
152  return false;
153  if (strncmp(magic, BinaryHeader::magic, BinaryHeader::magicSize) != 0) {
154  // backward compatibility: old format without any header
155 #if defined(BISANI)
156  version = 1;
157 #else
158  version = 2;
159 #endif
160  bis.seek(-BinaryHeader::magicSize, std::ios::cur);
161  }
162  else {
163  if (!(bis >> version))
164  return false;
165  }
166  u32 size;
167  if (!(bis >> size))
168  return false;
169  b->resize(size);
170  switch (version) {
171  case 1: {
172  for (StaticBoundaries::iterator it = b->begin(); (it != b->end()) && (bis); ++it) {
173  u32 time, transitId;
174  if (!(bis >> time >> transitId))
175  return false;
176  it->setTime(time);
177  }
178  } break;
179  case 2:
180  case 3: {
181  for (StaticBoundaries::iterator it = b->begin(); (it != b->end()) && (bis); ++it) {
182  u32 time;
183  u16 final, initial;
184  if (!(bis >> time >> final >> initial))
185  return false;
186  it->setTime(time);
187  it->setTransit(Boundary::Transit(final, initial, AcrossWordBoundary));
188  }
189  } break;
190  case 4: {
191  for (StaticBoundaries::iterator it = b->begin(); (it != b->end()) && (bis); ++it) {
192  u32 time;
193  u16 final, initial;
194  u8 boundary;
195  if (!(bis >> time >> final >> initial >> boundary))
196  return false;
197  it->setTime(time);
198  it->setTransit(Boundary::Transit(final, initial, boundary));
199  }
200  } break;
201  default:
202  defect();
203  }
204  return bis;
205 }

215 bool writeBinary(const StaticBoundaries* b, std::ostream& os) {
216  Core::BinaryOutputStream bos(os);
217  bos.write(BinaryHeader::magic, BinaryHeader::magicSize);
218  bos << (u32)BinaryHeader::version;
219  bos << (u32)b->size();
220  switch (BinaryHeader::version) {
221  case 3:
222  for (StaticBoundaries::const_iterator it = b->begin(); it != b->end(); ++it) {
223  bos << (u32)it->time();
224  bos << (u16)it->transit().final << (u16)it->transit().initial;
225  }
226  break;
227  case 4:
228  for (StaticBoundaries::const_iterator it = b->begin(); it != b->end(); ++it) {
229  bos << (u32)it->time();
230  bos << (u16)it->transit().final << (u16)it->transit().initial << (u8)it->transit().boundary;
231  }
232  break;
233  default:
234  defect();
235  }
236  return bos;
237 }

648  void load() {
649  LexiconRef lexicon = Lexicon::us();
650  Fsa::LabelId unkId = lexicon->unkLemmaId();
651  const std::string path = paramPath(config);
652  if (!Core::isValidPath(path))
653  criticalError("CtmReaderNode: Could not find \"%s\"",
654  path.c_str());
655  CtmRecording emptyCtmRecording;
656  CtmEntry ctmEntry;
657  TextFileParser tf(path, paramEncoding(config));
658  if (!tf)
659  criticalError("CtmReaderNode: Could not open \"%s\"",
660  path.c_str());
661  std::string lastName;
662  CtmCorpus::iterator it = corpus_.end();
663  u32 nWords = 0;
664  for (;; ++nWords) {
665  const TextFileParser::StringList& cols = tf.next();
666  if (!tf)
667  break;
668  if (cols.size() < 5)
669  criticalError("CtmReaderNode: Require at least five columns per line.");
670  ctmEntry.scores = str2scores(cols);
671  ctmEntry.start = str2time(cols[2]);
672  ctmEntry.end = ctmEntry.start + str2time(cols[3]);
673  if (cols[4] == "@")
674  ctmEntry.label = Fsa::Epsilon;
675  else {
676  ctmEntry.label = lexicon->lemmaId(cols[4]);
677  if (ctmEntry.label == Fsa::InvalidLabelId)
678  ctmEntry.label = unkId;
679  }
680  if (lastName != cols[0]) {
681  lastName = cols[0];
682  it = corpus_.find(cols[0]);
683  }
684  if (it != corpus_.end()) {
685  if (ctmEntry.start < it->second.back().end) {
686  if ((it->second.back().start < ctmEntry.start) && ((it->second.back().end - ctmEntry.start) <= 1)) {
687  it->second.back().end = ctmEntry.start;
688  }
689  else
690  criticalError("CtmReaderNode: Chronological order violated %s:%d",
691  path.c_str(), tf.currentLineNumber());
692  }
