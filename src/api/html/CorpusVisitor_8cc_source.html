73 void setParameter(size_t index, Bliss::Recording* recording, ParameterAdaptor parameterAdaptor) {
74  std::string inputFile(recording->video());
75  if (inputFile.empty())
76  inputFile = recording->audio();
77  parameterAdaptor.set("input-file", inputFile);
78  if (!recording->audio().empty()) {
79  parameterAdaptor.set("input-audio-file", recording->audio());
80  parameterAdaptor.set("input-audio-name", recording->name());
81  }
82  if (!recording->video().empty()) {
83  parameterAdaptor.set("input-video-file", recording->video());
84  parameterAdaptor.set("input-video-name", recording->name());
85  }
86  parameterAdaptor.set("recording-index", Core::form("%zd", index));

89 template<class ParameterAdaptor>
90 void setParameter(size_t index, Bliss::Segment* segment, ParameterAdaptor parameterAdaptor) {
91  parameterAdaptor.set("id", segment->fullName());
92  parameterAdaptor.set("segment-index", Core::form("%zd", index));
93  parameterAdaptor.set("segment-type", std::string(Bliss::Segment::typeId[segment->type()]));
94  parameterAdaptor.set("acoustic-condition", segment->condition() ? segment->condition()->name() : "");
95  parameterAdaptor.set("start-time", Core::form("%g", segment->start()));
96  parameterAdaptor.set("end-time", Core::form("%g", segment->end()));
97  parameterAdaptor.set("track", Core::form("%d", segment->track()));
98 
99  // disassemble segment fullname: .../segment-1/segment-0 and corpus-0/corpus-1/...
100  parameterAdaptor.set("segment", segment->fullName());
101  parameterAdaptor.set("segment-0", segment->name());
102  u32 segmentLevel = 1;
103  Bliss::CorpusSection* corpusSection = segment->parent();
104  while (corpusSection) {
105  parameterAdaptor.set("corpus-" + Core::form("%d", corpusSection->level()),
106  corpusSection->name());
107  parameterAdaptor.set("segment-" + Core::form("%d", segmentLevel),
108  corpusSection->name());
109  corpusSection = corpusSection->parent();
110  segmentLevel++;
111  }
112 }

181 void CorpusVisitor::visitSegment(Bliss::Segment* segment) {
182  setParameter(segmentIndex_, segment, DataSourceParameterAdaptor(dataSources_));
183  setParameter(segmentIndex_, segment, StringExpressionAdaptor(corpusKeys_));
184 
185  for (size_t i = 0; i < corpusProcessors_.size(); ++i)
186  corpusProcessors_[i]->enterSegment(segment);
187  for (size_t i = 0; i < corpusProcessors_.size(); ++i)
188  corpusProcessors_[i]->processSegment(segment);
189  for (size_t i = 0; i < corpusProcessors_.size(); ++i)
190  corpusProcessors_[i]->leaveSegment(segment);
191  ++segmentIndex_;
192 }

194 void CorpusVisitor::visitSpeechSegment(Bliss::SpeechSegment* speechSegment) {
195  setParameter(segmentIndex_, speechSegment, DataSourceParameterAdaptor(dataSources_));
196  setParameter(segmentIndex_, speechSegment, StringExpressionAdaptor(corpusKeys_));
197 
198  size_t i;
199  for (i = 0; i < corpusProcessors_.size(); ++i)
200  corpusProcessors_[i]->enterSpeechSegment(speechSegment);
201  for (i = 0; i < corpusProcessors_.size(); ++i)

203  for (i = 0; i < corpusProcessors_.size(); ++i)
204  corpusProcessors_[i]->leaveSpeechSegment(speechSegment);

207  clearParameter(speechSegment, StringExpressionAdaptor(corpusKeys_));
