46  void createPhoneSequence(const Am::Allophone& a, OpenFst::VectorFst* seq) const {
47  Bliss::Phoneme::Id history = (a.history().size() ? a.history()[0] : Bliss::Phoneme::term);
48  Bliss::Phoneme::Id central = a.central();
49  Bliss::Phoneme::Id future = (a.future().size() ? a.future()[0] : Bliss::Phoneme::term);
50  OpenFst::Label labels[4];
51  labels[0] = history == Bliss::Phoneme::term ? sil_ : history;
52  labels[1] = central;
53  labels[2] = future == Bliss::Phoneme::term ? sil_ : future;
54  labels[3] = seqEnd_;
55  for (int i = 0; i < 4; ++i)
56  labels[i] = labels[i];
57  labels[0] += initialOffset_;
58 
59  if (a.boundary & Am::Allophone::isInitialPhone)
60  labels[1] += initialOffset_;
61  if (a.boundary & Am::Allophone::isFinalPhone)
62  labels[2] += initialOffset_;
63 
64  if (verbose_) {
65  for (int i = 0; i < 4; ++i)
66  std::cout << labels[i] << "=" << phoneSymbols_->Find(labels[i]) << " ";
67  std::cout << std::endl;
68  }
69  seq->DeleteStates();
70  OpenFst::StateId s = seq->AddState(), ns;
71  seq->SetStart(s);
72  for (int i = 0; i < 4; ++i) {
73  ns = seq->AddState();
74  seq->AddArc(s, OpenFst::Arc(labels[i], labels[i], OpenFst::Weight::One(), ns));
75  s = ns;
76  }
77  seq->SetFinal(s, OpenFst::Weight::One());
78  }

80  void createPhoneSequence(const Bliss::Phoneme::Id* pron, OpenFst::VectorFst* seq) const {
81  seq->DeleteStates();
82  OpenFst::StateId s = seq->AddState(), ns = OpenFst::InvalidStateId;
83  seq->SetStart(s);
84  OpenFst::Label offset = initialOffset_;
85  while (*pron != Bliss::Phoneme::term) {
86  ns = seq->AddState();
87  OpenFst::Label l = *pron + offset;
88  seq->AddArc(s, OpenFst::Arc(l, l, OpenFst::Weight::One(), ns));
89  if (verbose_) {
90  std::cout << phoneSymbols_->Find(l) << " ";
91  }
92  s = ns;
93  ++pron;
94  offset = 0;
95  }
96  ns = seq->AddState();
97  OpenFst::Label endPhone = seqEnd_;
98  if (verbose_) {
99  std::cout << phoneSymbols_->Find(endPhone) << std::endl;
100  }
101  seq->AddArc(s, OpenFst::Arc(endPhone, endPhone, OpenFst::Weight::One(), ns));
102  seq->SetFinal(ns, OpenFst::Weight::One());
103  }

105  int checkAllophones(Core::Ref<Am::AcousticModel> am, const OpenFst::VectorFst* f) const {
106  bool error = false;
107  int nChecked = 0;
108  Core::Ref<const Am::AllophoneAlphabet> allophones = am->allophoneAlphabet();
109  const Am::AllophoneAlphabet::AllophoneList& allophoneList = allophones->allophones();
110  for (Am::AllophoneAlphabet::AllophoneList::const_iterator ai = allophoneList.begin();
111  ai != allophoneList.end() && !error; ++ai) {
112  const Am::Allophone* a = *ai;
113  if (verbose_)
114  std::cout << "allophone " << allophones->symbol(allophones->index(a)) << std::endl;
115  OpenFst::VectorFst phoneSeq, result;
116  phoneSeq.SetProperties(FstLib::kAcceptor | FstLib::kNotAcceptor, FstLib::kAcceptor | ~FstLib::kNotAcceptor);
117  createPhoneSequence(*a, &phoneSeq);
118  FLAGS_fst_compat_symbols = false;
119  FstLib::Compose(*f, phoneSeq, &result);
120  if (verbose_)
121  printAllophoneSequence(allophones, result);
122  if (result.NumStates() < 5) {
123  error = true;
124  std::cout << "ERROR: wrong number of states: " << result.NumStates() << std::endl;
125  }
126  if (!isLinearFst(result)) {
127  error = true;
128  std::cout << "ERROR: result not linear" << std::endl;
129  }
130  OpenFst::StateId s = result.Start();
131  u32 inputPos = 0;
132  while (s != OpenFst::InvalidStateId && inputPos < 2) {
133  OpenFst::ArcIterator aiter(result, s);
134  if (aiter.Done()) {
135  std::cout << "ERROR: result too short" << std::endl;
136  error = true;
137  break;
138  }
139  const OpenFst::Arc& arc = aiter.Value();
140  if (verbose_) {
141  std::cout << inputPos << " " << arc.ilabel << " " << allophones->symbol(OpenFst::convertLabelToFsa(arc.ilabel)) << std::endl;
142  }
143  if (arc.ilabel != OpenFst::Epsilon) {
144  ++inputPos;
145  if (inputPos == 2 && OpenFst::convertLabelToFsa(arc.ilabel) != allophones->index(*a)) {
146  error = true;
147  std::cout << "ERROR: wrong input label: " << arc.ilabel << " "
148  << allophones->symbol(OpenFst::convertLabelToFsa(arc.ilabel)) << std::endl;
149  break;
150  }
151  }
152  s = arc.nextstate;
153  }
154  if (error) {
155  Fsa::LabelId id = allophones->index(*a);
156  std::cout << "allophone: " << id << " "
157  << allophones->symbol(id) << std::endl;
158  result.SetOutputSymbols(phoneSymbols_);
159  result.Write(Core::form("/tmp/%d_result.fst", id));
160  phoneSeq.SetInputSymbols(phoneSymbols_);
161  phoneSeq.Write(Core::form("/tmp/%d_seq.fst", id));
162  }
163  ++nChecked;
164  }
165  return nChecked;
166  }

192  bool checkWord(Core::Ref<const Am::AllophoneAlphabet> allophones, Core::Ref<const Bliss::PhonemeInventory> pi,
193  const Bliss::Phoneme::Id* pron, const OpenFst::VectorFst& result) const {
194  OpenFst::StateId s = result.Start();
195  Bliss::Phoneme::Id prev = Bliss::Phoneme::term;
196  u32 phoneIndex = 0;
197  while (*pron != Bliss::Phoneme::term) {
198  OpenFst::ArcIterator aiter(result, s);
199  if (aiter.Done()) {
200  std::cout << "ERROR: result too short" << std::endl;
201  return false;
202  }
203  const OpenFst::Arc& arc = aiter.Value();
204  if (arc.ilabel != OpenFst::Epsilon) {
205  const Am::Allophone* allophone = allophones->allophone(OpenFst::convertLabelToFsa(arc.ilabel));
206  verify(allophone);
207  Bliss::Phoneme::Id phoneseq[3];
208  phoneseq[0] = (allophone->history().size() ? allophone->history()[0] : Bliss::Phoneme::term);
209  phoneseq[1] = allophone->central();
210  phoneseq[2] = (allophone->future().size() ? allophone->future()[0] : Bliss::Phoneme::term);
211  Bliss::Phoneme::Id ref[3];
212  for (u32 i = 0; i < 3; ++i) {
213  ref[i] = (i ? *(pron + i - 1) : prev);
214  const bool isCiPhone = (ref[i] != Bliss::Phoneme::term && !pi->phoneme(ref[i])->isContextDependent());
215  if (isCiPhone) {
216  if (i == 1) {
217  ref[0] = ref[2] = Bliss::Phoneme::term;
218  break;
219  }
220  else {
221  ref[i] = Bliss::Phoneme::term;
222  }
223  }
224  }
225  for (u32 i = 0; i < 3; ++i) {
226  if (ref[i] != phoneseq[i]) {
227  std::cout << "ERROR: wrong phone pos=" << phoneIndex << " c=" << i << " "
228  << "expected " << static_cast<u32>(ref[i]) << " "
229  << (ref[i] == Bliss::Phoneme::term ? "#" : pi->phoneme(ref[i])->symbol().str()) << " "
230  << "found " << static_cast<u32>(phoneseq[i]) << " "
231  << (phoneseq[i] == Bliss::Phoneme::term ? "#" : pi->phoneme(phoneseq[i])->symbol().str())
232  << std::endl;
233  return false;
234  }
235  }
236  prev = *pron;
237  ++pron;
238  ++phoneIndex;
239  }
240  s = arc.nextstate;
241  }
242  return true;
243  }

245  int checkWords(Core::Ref<Am::AcousticModel> am, Bliss::LexiconRef lexicon, const OpenFst::VectorFst* f) const {
246  int nChecked = 0;
247  Bliss::Lexicon::PronunciationIterator pronIter, pronEnd;
248  Core::Ref<const Bliss::PhonemeInventory> pi = lexicon->phonemeInventory();
249  Core::tie(pronIter, pronEnd) = lexicon->pronunciations();
250  Core::Ref<const Am::AllophoneAlphabet> allophones = am->allophoneAlphabet();
251  OpenFst::VectorFst phoneSeq, result;
252  bool error = false;
253  for (; pronIter != pronEnd; ++pronIter) {
254  const Bliss::Pronunciation* pron = *pronIter;
255  if (verbose_) {
256  std::cout << "pronunciation: ";
257  std::cout << pron->format(pi) << std::endl;
258  }
259  const Bliss::Phoneme::Id* phones = pron->phonemes();
260  createPhoneSequence(phones, &phoneSeq);
261  phoneSeq.SetInputSymbols(phoneSymbols_);
262  FLAGS_fst_compat_symbols = false;
263  FstLib::Compose(*f, phoneSeq, &result);
264  if (verbose_ && result.NumStates()) {
265  printAllophoneSequence(allophones, result);
266  }
267  if (!isLinearFst(result)) {
268  std::cout << "ERROR: result not linear" << std::endl;
269  error = true;
270  }
271  if (result.NumStates() != phoneSeq.NumStates()) {
272  std::cout << "ERROR: wrong number of states. "
273  << "expected: " << phoneSeq.NumStates() << " "
274  << "found: " << result.NumStates() << std::endl;
275  error = true;
276  }
277  if (!error) {
278  if (!checkWord(allophones, pi, phones, result)) {
279  std::cout << "ERROR: wrong allophone sequence " << std::endl;
280  error = true;
281  }
282  }
283  if (error) {
284  break;
285  }
286  ++nChecked;
287  }
288  return nChecked;
289  }
