39 void MatrixTool::usage(std::string action) const {
40  if (action == "write") {
41  std::cout << "action write: write parameter to file" << std::endl;
42  std::cout << "parameter: \"new-file\"" << std::endl;
43  }
44  else if (action == "scale") {
45  std::cout << "action scale: scale matrix" << std::endl;
46  std::cout << "parameter: \"scaling-factor\"" << std::endl;
47  }
48  else if (action == "max") {
49  std::cout << "action max: compute abs max of matrix" << std::endl;
50  std::cout << " no parameters" << std::endl;
51  }
52  else if (action == "l2norm") {
53  std::cout << "action l2norm: compute l2norm of matrix" << std::endl;
54  std::cout << " no parameters" << std::endl;
55  }
56  else if (action == "add") {
57  std::cout << "action add: add another matrix" << std::endl;
58  std::cout << "parameter \"file"
59  "\" (name of summand matrix file)"
60  << std::endl;
61  }
62  else if (action == "expand") {
63  std::cout << "action expand: expand matrix with zeros" << std::endl;
64  std::cout << "parameters \"number-of-rows\", \"number-of-columns\"" << std::endl;
65  }
66  else if (action == "exp") {
67  std::cout << "action exp: take exp of all entries" << std::endl;
68  }
69  else if (action == "log") {
70  std::cout << "action exp: take log of all entries" << std::endl;
71  }
72  else {
73  std::cout << "usage: " << basename_ << " [filename] [precision] [actions]" << std::endl;
74  std::cout << "\t where" << std::endl;
75  std::cout << "\t precision is \"f32\" or \"f64\"" << std::endl;
76  std::cout << "\t actions is a comma-separated sequence of commands" << std::endl;
77  std::cout << "\t available actions:" << std::endl;
78  std::cout << "\t\twrite, scale, max, l2norm, add, expand, exp, log" << std::endl;
79  }
80 }

82 int MatrixTool::main(const std::vector<std::string>& arguments) {
83  if (arguments.size() == 2 && arguments.at(0) == "help") {
84  usage(arguments.at(1));
85  return 1;
86  }
87  else if (arguments.size() < 3) {
88  usage();
89  return 1;
90  }
91  std::string filename = arguments.at(0);
92  std::string precision = arguments.at(1);
93  std::vector<std::string> actions = Core::split(arguments.at(2), ",");
94 
95  bool success = true;
96  if (precision == "f32") {
97  Math::Matrix<f32> matrix;
98  if (!Math::Module::instance().formats().read(filename, matrix)) {
99  error("could not read matrix from file ") << matrix;
100  success = 1;
101  }
102  else {
103  success = actionLoop(actions, matrix);
104  }
105  }
106  else if (precision == "f64") {
107  Math::Matrix<f64> matrix;
108  if (!Math::Module::instance().formats().read(filename, matrix)) {
109  error("could not read matrix from file ") << matrix;
110  success = 1;
111  }
112  else {
113  success = actionLoop(actions, matrix);
114  }
115  }
116  else {
117  error("unknown precision: ") << precision;
118  return 1;
119  }
120  int returnVal = success ? 0 : 1;
121  return returnVal;
122 }
