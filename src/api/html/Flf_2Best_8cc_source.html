44 std::pair<ConstLatticeRef, Score> bestProjection(ConstLatticeRef l) {
45  const Semiring& semiring = *l->semiring();
46  ConstStateMapRef topologicalSort = sortTopologically(l);
47  Fsa::StateId initialSid = topologicalSort->front();
48  Traceback traceback(topologicalSort->maxSid + 1);
49  traceback[initialSid].score = 0.0;
50  TraceElement bestTrace;
51  for (u32 i = 0; i < topologicalSort->size(); ++i) {
52  const Fsa::StateId sid = (*topologicalSort)[i];
53  ConstStateRef sr = l->getState(sid);
54 
55  const TraceElement& currentTrace = traceback[sid];
56  if (sr->isFinal()) {
57  const Score score = add(currentTrace.score, semiring.project(sr->weight()));
58  if (score < bestTrace.score) {
59  bestTrace.score = score;
60  bestTrace.sid = sid;
61  }
62  }
63  Fsa::StateId aid = 0;
64  for (State::const_iterator a = sr->begin(), end = sr->end(); a != end; ++a, ++aid) {
65  const Score score = add(currentTrace.score, semiring.project(a->weight()));
66  TraceElement& trace = traceback[a->target()];
67  if (score < trace.score) {
68  trace.score = score;
69  trace.sid = sid;
70  trace.aid = aid;
71  }
72  }
73  }
74  if (bestTrace.sid == Fsa::InvalidStateId) {
75  Core::Application::us()->error("No best path found; all pathes have infinite score.");
76  return std::make_pair(ConstLatticeRef(), Semiring::Max);
77  }
78  StaticLattice* s = new StaticLattice(l->type());
79  s->setProperties(l->knownProperties() | Fsa::PropertyLinear, l->properties() | Fsa::PropertyLinear);
80  s->setInputAlphabet(l->getInputAlphabet());
81  if (l->type() != Fsa::TypeAcceptor)
82  s->setOutputAlphabet(l->getOutputAlphabet());
83  s->setSemiring(l->semiring());
84  s->setBoundaries(l->getBoundaries());
85  s->setDescription(Core::form("best-projection(%s)", l->describe().c_str()));
86  Fsa::StateId bestSid = bestTrace.sid;
87  ConstStateRef sr = l->getState(bestSid);
88  State* sp = new State(sr->id(), sr->tags(), sr->weight());
89  s->setState(sp);
90  verify(sp->isFinal());
91  while (bestSid != initialSid) {
92  const TraceElement& trace = traceback[bestSid];
93  sr = l->getState(trace.sid);
94  sp = new State(sr->id(), sr->tags(), sr->weight());
95  s->setState(sp);
96  *sp->newArc() = *(sr->begin() + trace.aid);
97  bestSid = trace.sid;
98  }
99  s->setInitialStateId(sp->id());
100  return std::make_pair(ConstLatticeRef(s), bestTrace.score);
101 }

105 void dumpBest(ConstLatticeRef l, std::ostream& os) {
106  ConstSemiringRef semiring = l->semiring();
107  ScoresRef sum = semiring->one();
108  os << "\"";
109  if (l && (l->initialStateId() != Fsa::InvalidStateId)) {
110  ConstLatticeRef b = best(l);
111  Fsa::ConstAlphabetRef alphabet = b->getInputAlphabet();
112  ConstStateRef sr = b->getState(b->initialStateId());
113  for (; sr->hasArcs(); sr = b->getState(sr->begin()->target())) {
114  sum = semiring->extend(sum, sr->begin()->weight());
115  os << alphabet->symbol(sr->begin()->input()) << " ";
116  }
117  sum = semiring->extend(sum, sr->weight());
118  }
119  os << "\""
120  << ", score=" << semiring->project(sum)
121  << ", scores=" << semiring->describe(sum, Fsa::HintShowDetails | HintUnscaled)
122  << std::endl;
123 }

127 ConstLatticeRef best(ConstLatticeRef l, SingleSourceShortestPathAlgorithm algo) {
128  if (!l || (l->initialStateId() == Fsa::InvalidStateId))
129  return ConstLatticeRef();
130  switch (algo) {
131  case Dijkstra:
132  return FtlWrapper::firstbest(l);
133  case BellmanFord:
134  return FtlWrapper::best(l);
135  case ProjectingBellmanFord:
136  return bestProjection(l).first;
137  default:
138  defect();
139  return ConstLatticeRef();
140  }
141 }

276 ConstAllPairsShortestDistanceRef AllPairsShortestDistance::create(ConstLatticeRef lRef, Time timeThreshold) {
277  Internal* internal = new Internal;
278  ConstStateMapRef topologicalSort = sortTopologically(lRef);
279  const Lattice& l = *lRef;
280  const Semiring& semiring = *l.semiring();
281  const Boundaries& b = *l.getBoundaries();
282  std::vector<StateIdScorePairList> D(topologicalSort->maxSid + 1);
283  for (StateMap::const_iterator itSid = topologicalSort->begin(),
284  endSid = topologicalSort->end();
285  itSid != endSid; ++itSid) {
286  Fsa::StateId sid = *itSid;
287  ConstStateRef sr = l.getState(sid);
288  StateIdScorePairList& from = D[sid];
289  // update self
290  StateIdScorePair d(sid, Semiring::One);
291  StateIdScorePairList::iterator itPair = std::lower_bound(from.begin(), from.end(), d);
292  from.insert(itPair, d);
293  // update successors
294  for (State::const_iterator a = sr->begin(), a_end = sr->end(); a != a_end; ++a) {
295  StateIdScorePairList& to = D[a->target()];
296  Time toTime = b.time(sr->id());
297  Score score = semiring.project(a->weight());
298  for (StateIdScorePairList::const_iterator itFrom = from.begin(), endFrom = from.end(); itFrom != endFrom; ++itFrom) {
299  Time fromTime = b.time(itFrom->sid);
300  verify(fromTime <= toTime);
301  if (toTime - fromTime <= timeThreshold) {
302  d.sid = itFrom->sid;
303  d.score = itFrom->score + score;
304  itPair = std::lower_bound(to.begin(), to.end(), d);
305  if ((itPair == to.end()) || (itPair->sid != d.sid))
306  to.insert(itPair, d);
307  else if (d.score < itPair->score)
308  itPair->score = d.score;
309  }
310  }
311  }
312  // store into final data structure
313  for (StateIdScorePairList::const_iterator itFrom = from.begin(), endFrom = from.end(); itFrom != endFrom; ++itFrom)
314  internal->insert(std::make_pair(StateIdPair(itFrom->sid, sid), itFrom->score));
315  }
316  return ConstAllPairsShortestDistanceRef(new AllPairsShortestDistance(internal));
317 }
