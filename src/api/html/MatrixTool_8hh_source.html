182 template<typename T>
183 void MatrixTool::join(Math::Matrix<T>& matrix) const {
184  Math::Matrix<T> B;
185  if (!Math::Module::instance().formats().read(paramFile(select("join")), B))
186  error("could not read matrix from file ") << paramFile(select("join"));
187 
188  if (B.nRows() != matrix.nRows()) {
189  error("dimension mismatch");
190  }
191  u32 oldColumns = matrix.nColumns();
192  matrix.resize(matrix.nRows(), matrix.nColumns() + B.nColumns());
193  for (u32 i = 0; i < B.nRows(); ++i) {
194  for (u32 j = 0; j < B.nColumns(); ++j)
195  matrix[i][j + oldColumns] = B[i][j];
196  }
197  log("matrix ") << paramFile(select("join")) << " joined (to the right)";
198 }

200 template<typename T>
201 bool MatrixTool::actionLoop(const std::vector<std::string>& actions, Math::Matrix<T>& matrix) const {
202  log("processing matrix of size: ") << matrix.nRows() << "x" << matrix.nColumns();
203  for (std::vector<std::string>::const_iterator action = actions.begin(); action != actions.end(); action++) {
204  log("action: ") << *action;
205  if (*action == "write") {
206  write(matrix);
207  }
208  else if (*action == "scale") {
209  scale(matrix);
210  }
211  else if (*action == "max") {
212  max(matrix);
213  }
214  else if (*action == "add") {
215  add(matrix);
216  }
217  else if (*action == "add-multiple") {
218  addMultiple(matrix);
219  }
220  else if (*action == "mult") {
221  mult(matrix);
222  }
223  else if (*action == "l2-norm") {
224  l2norm(matrix);
225  }
226  else if (*action == "exp") {
227  exp(matrix);
228  }
229  else if (*action == "log") {
230  logarithm(matrix);
231  }
232  else if (*action == "get-columns") {
233  getColumns(matrix);
234  }
235  else if (*action == "join") {
236  join(matrix);
237  }
238  else if (*action == "expand") {
239  expand(matrix);
240  }
241  else {
242  error("unknown action: ") << *action;
243  return false;
244  }
245  }
246  return true;
247 }
