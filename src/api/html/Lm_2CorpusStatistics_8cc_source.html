46  SentenceStatistics()

48  nSyntacticTokens(0),
49  avgEvaluationTokens(0.0),
50  nUnkLemmas(0),
51  nUnkSyntacticTokens(0),
52  avgUnkEvaluationTokens(0.0),
53  score(0.0) {}
54 
55  f32 lemmaPerplexity() const {
56  return (nLemmas) ? exp(score / f32(nLemmas)) : Core::Type<f32>::max;
57  }
58  f32 lemmaCrossEntropy() const {
59  return (nLemmas) ? score / f32(nLemmas) / ln2 : Core::Type<f32>::max;
60  }
61  f32 syntPerplexity() const {
62  return (nSyntacticTokens) ? exp(score / f32(nSyntacticTokens)) : Core::Type<f32>::max;
63  }
64  f32 syntCrossEntropy() const {
65  return (nSyntacticTokens) ? score / f32(nSyntacticTokens) / ln2 : Core::Type<f32>::max;
66  }
67  f32 evalPerplexity() const {
68  return (avgEvaluationTokens) ? exp(score / avgEvaluationTokens) : Core::Type<f32>::max;
69  }
70  f32 evalCrossEntropy() const {
71  return (avgEvaluationTokens) ? score / avgEvaluationTokens / ln2 : Core::Type<f32>::max;
72  }

149 /**
150  * @page Perplexity On computing perplexity / cross entropy
151  *
152  Definitions:
153  cross-entropy in bit
154  H_p(T) = - \frac{1}{W} \log_2 p(T)
155  perplexity
156  PP_p(T) = \exp( - \frac{1}{W} \log p(T) )
157 
158  where $T$ is the text and $W$ is the length of the text measured in
159  words. If the $s_1, \ldots, s_K$ are the sentences of the text, then
160 
161  p(T) = \prod_{k=1}^K p(s_k)
162 
163  * A recurring issue with perplexity is whether or not to count the
164  * sentence end. The option chosen here, is to treat the sentence end
165  * as a genuine event. I.e. the term p(sentence-end | final words) is
166  * included in $p(s)$, and $W$ is the number of "ordinary" words plus
167  * one for each sentences. (This is consistent with well-known
168  * publications such as [Chen & Goodman 1999].)
169  *
170  * Calculating the perplexity is less straight forward than you might
171  * think, due to the concept of language model phrases. The idea is
172  * that multiple words are treated as a single event by the language
173  * model. In Bliss such a thing is represented as an ortographic form
174  * containing one (or more) blank character(s). Now let us consider
175  * the following example: first the Bliss lexicon:
176 
177  <lemma>
178  <orth>going</orth>
179  <phon>g @U I N</phon>
180  </lemma>
181  <lemma>
182  <orth>to</orth>
183  <phon>t u:</phon>
184  </lemma>
185  <lemma>
186  <orth>going to</orth>
187  <phon>g O n @</phon>
188  <synt><tok>going_to</tok></synt>
189  <eval><tok>going</tok><tok>to</tok></eval>
190  </lemma>
191  <lemma>
192  <orth>New York</orth>
193  <phon>n u: j O: k</phon>
194  <synt><tok>[CITY]</tok></synt>
195  </lemma>
196  <lemma>
197  <orth>[breath]</orth>
198  <synt/>
199  <eval/>
200  </lemma>
201 
202  * Now what about the sentence "I'am going to [breath] visit New York". We
203  * observe that the phrase "going to" allows for two parsings, namely
204  * as one or as two lemmata, while "New York" can only be read as a
205  * single token. (Let's assume that "New" with captial N does not
206  * occur in the lexicon.) So there are two possible syntactic token
207  * sequences:
208 
209  I'am going to visit [CITY] </s>
210  I'am going_to visit [CITY] </s>
211 
212  * The first question when computing perplexity is: What is the
213  * log-likelihood LL = ln(p(W))? The solution implemented below is:
214  * Use the most likely token sequence, i.e. LL = max_W ln(p(W)).
215  *
216  * The second question is what is the number of words $W$? We might
217  * use the length of the most likely token sequence, but this would
218  * bias the result in favour of long phrases. We could count the
219  * number of blanks (plus one), but this would violate an important
220  * design principle of Bliss and moreover you can argue that "New
221  * York" should be counted as a single word unless "New" and "York"
222  * individually also have the status of a word, which is not true for
223  * "New". Non-lexical events like '[breath]" pose another
224  * complication: Some system setups model them as language model
225  * events others don't. A common formal characterisitic property is
226  * that they do not count in evaluation. The solution to these issues
227  * as implemented below is: Take the most likely lemma sequence,
228  * according to the language model, and count the number of evaluation
229  * tokens (plus one for the sentence end). In this example the result
230  * is six. Addtionally we provide numbers on the basis of syntactic
231  * tokens.
232  */
233 CorpusStatisticsVisitor::SentenceStatistics CorpusStatisticsVisitor::buildSentenceStatistics(
234  Fsa::ConstAutomatonRef sentence,
235  Lm::Score syntaxEmissionScale,
236  Lm::Score classEmissionScale) {
237  SentenceStatistics stats;
238 
239  // determine best lemma sequence (in case the orthography was ambigious)
240  if (classLm_)
241  classLm_->setClassEmissionScale(classEmissionScale);
242  sentence = Lm::compose(sentence, lm_, 1.0, syntaxEmissionScale);
243  sentence = Fsa::firstbest(sentence);
244  if (Fsa::isEmpty(sentence))
245  return stats;
246 
247  // convert to sequence of lemmas
248  std::vector<Fsa::LabelId> labels;
249  Fsa::getLinearInput(sentence, labels);
250  Core::Ref<const Bliss::LemmaAlphabet> lemmaAlphabet = ref(dynamic_cast<const Bliss::LemmaAlphabet*>(sentence->getInputAlphabet().get()));
251  std::vector<const Bliss::Lemma*> lemmas;
252  for (std::vector<Fsa::LabelId>::const_iterator li = labels.begin(); li != labels.end(); ++li)
253  lemmas.push_back(lemmaAlphabet->lemma(*li));
254 
255  // process LM events
256  History h = lm_->startHistory();
257  for (std::vector<const Bliss::Lemma*>::const_iterator ll = lemmas.begin(); ll != lemmas.end(); ++ll) {
258  const Bliss::SyntacticTokenSequence syntacticTokens((*ll)->syntacticTokenSequence());
259  u32 nSyntacticTokens = syntacticTokens.length();
260  u32 nEvaluationTokens = 0;
261  u32 nEvaluationTokenSequences = 0;
262  Bliss::Lemma::EvaluationTokenSequenceIterator e, e_end;
263  for (Core::tie(e, e_end) = (*ll)->evaluationTokenSequences(); e != e_end; ++e) {
264  nEvaluationTokens += e->length();
265  ++nEvaluationTokenSequences;
266  }
267  f32 avgEvaluationTokens = nEvaluationTokenSequences ? f32(nEvaluationTokens) / f32(nEvaluationTokenSequences) : 0;
268  bool hasAnythingToReport = (nSyntacticTokens > 0) || (nEvaluationTokens > 0);
269 
270  if (sentenceChannel_.isOpen() && hasAnythingToReport)
271  sentenceChannel_
272  << (Core::XmlOpen("lemma") + Core::XmlAttribute("name", (*ll)->name().str()));
273  Lm::Score lemmaScore = 0.0;
274  bool isUnk = false;
275  for (u32 ti = 0; ti < syntacticTokens.length(); ++ti) {
276  const Bliss::SyntacticToken* st = syntacticTokens[ti];
277  Lm::Score lmScore = lm_->score(h, st);
278  if (lmScore != Core::Type<Lm::Score>::max) {
279  lmScore += syntaxEmissionScale * st->classEmissionScore();
280  if (classMapping_)
281  lmScore += classEmissionScale * classMapping_->classEmissionScore(st);
282  }
283  else
284  isUnk = true;
285  if (sentenceChannel_.isOpen()) {
286  sentenceChannel_
287  << Core::XmlOpen("event")
288  << Core::XmlFull("token", st->symbol());
289  if (classMapping_)
290  sentenceChannel_
291  << Core::XmlFull("class", classMapping_->classToken(st)->symbol());
292  sentenceChannel_
293  << Core::XmlFull("history", lm_->formatHistory(h))
294  << Core::XmlFull("lm-score", lmScore)
295  << Core::XmlClose("event");
296  }
297  lemmaScore += lmScore;
298  h = lm_->extendedHistory(h, st);
299  }
300  if (isUnk) {
301  ++stats.nUnkLemmas;
302  stats.nUnkSyntacticTokens += nSyntacticTokens;
303  stats.avgUnkEvaluationTokens += avgEvaluationTokens;
304  }
305  if (!(isUnk && ignoreUnk_)) {
306  ++stats.nLemmas;
307  stats.nSyntacticTokens += nSyntacticTokens;
308  stats.avgEvaluationTokens += avgEvaluationTokens;
309  stats.score += lemmaScore;
310  }
311  if (sentenceChannel_.isOpen() && hasAnythingToReport)
312  sentenceChannel_
313  << Core::XmlFull("words", avgEvaluationTokens)
314  // << Core::XmlFull("score", stats.score)
315  << Core::XmlClose("lemma");
316  }
317  Lm::Score exitScore = lm_->sentenceEndScore(h);
318  stats.avgEvaluationTokens += 1.0;
319  stats.score += exitScore;
320  if (sentenceChannel_.isOpen()) {
321  sentenceChannel_
322  << Core::XmlOpen("sentence-end")
323  << Core::XmlOpen("event");
324  if (lm_->sentenceEndToken()) {
325  sentenceChannel_
326  << Core::XmlFull("token", lm_->sentenceEndToken()->symbol());
327  if (classMapping_)
328  sentenceChannel_
329  << Core::XmlFull("class", lm_->sentenceEndToken()->symbol());
330  }
331  else
332  sentenceChannel_
333  << Core::XmlFull("token", "n/a");
334  sentenceChannel_
335  << Core::XmlFull("history", lm_->formatHistory(h))
336  << Core::XmlFull("sentence-end-score", exitScore)
337  << Core::XmlClose("event")
338  << Core::XmlFull("words", 1)
339  // << Core::XmlFull("score", stats.score)
340  << Core::XmlClose("sentence-end");
341  }
342  if (sentenceChannel_.isOpen()) {
343  sentenceChannel_
344  << (Core::XmlFull("log-likelihood", -stats.score) + Core::XmlAttribute("base", "e"))
345  << Core::XmlFull("words", stats.avgEvaluationTokens)
346  << (Core::XmlFull("perplexity", stats.evalPerplexity()) + Core::XmlAttribute("base", "word"))
347  << (Core::XmlFull("cross-entropy", stats.evalCrossEntropy()) + Core::XmlAttribute("base", "word") + Core::XmlAttribute("unit", "bit"))
348  << Core::XmlFull("events", stats.nSyntacticTokens)
349  << (Core::XmlFull("perplexity", stats.syntPerplexity()) + Core::XmlAttribute("base", "event"))
350  << (Core::XmlFull("cross-entropy", stats.syntCrossEntropy()) + Core::XmlAttribute("type", "event") + Core::XmlAttribute("unit", "bit"));
351  }
352 
353  // check against FSA result
354  if (!ignoreUnk_) {
355  Fsa::Weight scoreFsa = Fsa::getLinearWeight(sentence);
356  if (!Core::isAlmostEqualUlp(f32(stats.score), f32(scoreFsa), 10))
357  warning("Lm/Fsa discrepancy: sentence log-likelihood %g != %g",
358  f32(stats.score), f32(scoreFsa));
359  }
360 
361  return stats;
362 }

364 void CorpusStatisticsVisitor::visitSpeechSegment(Bliss::SpeechSegment* segment) {
365  sentenceChannel_
366  << Core::XmlOpen("orth") + Core::XmlAttribute("source", "reference")
367  << segment->orth()
368  << Core::XmlClose("orth");
369 
370  Fsa::ConstAutomatonRef sentence = orthographicParser_.createLemmaAcceptor(segment->orth());
371 
372  if (what_ & ReportLanguageModel) {
373  if (sentenceChannel_.isOpen())
374  sentenceChannel_
375  << (Core::XmlOpen("sentence-statistics") + Core::XmlAttribute("model", "language model"));
376  SentenceStatistics lmStats = buildSentenceStatistics(sentence, syntaxEmissionScale_, classEmissionScale_);
377  if (lmStats.nLemmas > 0)
378  *lmStats_ += lmStats;
379  else
380  error("failed to determine best lemma sequence for orthography \"%s\"",
381  segment->orth().c_str());
382  if (sentenceChannel_.isOpen())
383  sentenceChannel_ << Core::XmlClose("sentence-statistics");
384  }
385  if (what_ & ReportClassModel) {
386  if (sentenceChannel_.isOpen())
387  sentenceChannel_
388  << (Core::XmlOpen("sentence-statistics") + Core::XmlAttribute("model", (classLm_ ? "class model" : "syntax model")));
389  SentenceStatistics cmStats = buildSentenceStatistics(sentence, 0.0, 0.0);
390  if (cmStats.nLemmas > 0)
391  *cmStats_ += cmStats;
392  else
393  error("failed to determine best lemma sequence for orthography \"%s\"",
394  segment->orth().c_str());
395  if (sentenceChannel_.isOpen())
396  sentenceChannel_ << Core::XmlClose("sentence-statistics");
397  }
