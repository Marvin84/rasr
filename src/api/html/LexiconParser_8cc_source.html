66 void PhonemeInventoryElement::phonemedefSymbol(const std::string& s) {
67  std::string symbol(s);
68  stripWhitespace(symbol);
69 
70  const Phoneme* test = product_->phoneme(symbol);
71  if (test == phoneme_) {
72  parser()->warning("Redundant definition of phonetic symbol \"%s\"", symbol.c_str());

74  else if (test != 0) {
75  parser()->error("Phonetic symbol \"%s\" already assigned to a different phoneme", symbol.c_str());
76  }
77  else {
78  product_->assignSymbol(phoneme_, symbol);

118 void PronunciationElement::start(const Core::XmlAttributes atts) {
119  Precursor::start(atts);
120  const char* weightStr = atts["weight"];
121  const char* scoreStr = atts["score"];
122  if (weightStr && scoreStr) {
123  parser()->criticalError("not possible to set both, pronunciation weight and score");
124  return;
125  }
126  f32 weight = (weightStr) ? atof(weightStr) : 1.0;
127  product_.weight = (scoreStr) ? exp(-atof(scoreStr)) : weight;
128  if (product_.weight < 0.0) {
129  parser()->error("pronunciation weight must be non-negative");
130  product_.weight = 0.0;
131  }
132 }

222 void LexiconElement::addOrth(const std::string& _orth) {
223  std::string orth(_orth);
224  normalizeWhitespace(orth);
225  suppressTrailingBlank(orth);
226  if (specialLemmaName_.empty() && !whitelist_.empty() && (whitelist_.find(orth) == whitelist_.end()))
227  return;
228  orths_.push_back(orth);
229  if (!lemma_) {
230  if (!lemmaName_.empty()) {
231  std::string name(lemmaName_);
232  normalizeWhitespace(lemmaName_);
233  suppressTrailingBlank(lemmaName_);
234  if (product_->lemma(lemmaName_)) {
235  parser()->error("Lemma name \"%s\" already taken", lemmaName_.c_str());
236  lemma_ = product_->newLemma();
237  }
238  else {
239  lemma_ = product_->newLemma(lemmaName_);
240  }
241  }
242  else {
243  lemma_ = product_->newLemma();
244  }
245  verify(lemma_);
246  if (!specialLemmaName_.empty()) {
247  if (product_->specialLemma(specialLemmaName_))
248  parser()->error("Special lemma \"%s\" already defined", specialLemmaName_.c_str());
249  else
250  product_->defineSpecialLemma(specialLemmaName_, lemma_);
251  }
252  }
253 }

321 void LexiconElement::endLemma() {
322  if (!lemma_)

361 void LexiconParser::loadWhitelist(const Core::Configuration& config, Core::StringHashSet& whitelist) {
362  std::string filename = paramFile(config);
363  if (!filename.empty()) {
364  Core::CompressedInputStream* cis = new Core::CompressedInputStream(filename.c_str());
365  Core::TextInputStream is(cis);
366  is.setEncoding(paramEncoding(config));
367  if (!is)
368  criticalError("Failed to open vocab file \"%s\".", filename.c_str());
369  std::string s;
370  while (Core::getline(is, s) != EOF) {
371  if ((s.size() == 0) || (s.at(0) == '#'))
372  continue;
373  Core::normalizeWhitespace(s);
374  Core::suppressTrailingBlank(s);
375  whitelist.insert(s);
376  }
377  if (!whitelist.empty())
378  log("Use a vocab list with %d entries.", u32(whitelist.size()));
379  }
380 }
