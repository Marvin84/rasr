73 LatticeHandler::ConstWordLatticeRef LatticeHandler::convert(const FlfLatticeAdaptor& la) const {
74  require(lexicon());
75  ConstLatticeRef l = la.get();
76  ScoreId amId = l->semiring()->id("am");
77  require(l->semiring()->hasId(amId));
78  ScoreId lmId = l->semiring()->id("lm");
79  require(l->semiring()->hasId(lmId));
80 
81  l = projectInput(mapInput(l, MapToLemmaPronunciation));
82  if (l->semiring()->scale(amId) != Semiring::DefaultScale)
83  l = multiply(l, amId, l->semiring()->scale(amId), RescoreModeInPlaceCache);
84  if (l->semiring()->scale(lmId) != Semiring::DefaultScale)
85  l = multiply(l, lmId, l->semiring()->scale(lmId), RescoreModeInPlaceCache);
86  l = extendByPronunciationScore(l, lexicon()->lemmaPronunciationAlphabet(), lmId, Semiring::UndefinedScale, RescoreModeInPlaceCache);
87  l = persistent(l);
88 
89  ::Lattice::WordLattice* wl = new ::Lattice::WordLattice;
90  wl->setFsa(toFsa(l, amId), ::Lattice::WordLattice::acousticFsa);
91  wl->setFsa(toFsa(l, lmId), ::Lattice::WordLattice::lmFsa);
92  if (l->getBoundaries()->valid()) {
93  StaticBoundaries boundaries;
94  copyBoundaries(l, &boundaries);
95  Core::Ref<::Lattice::WordBoundaries> wordBoundaries(new ::Lattice::WordBoundaries);
96  for (Fsa::StateId s = 0; s < boundaries.size(); ++s) {
97  if (boundaries.valid(s)) {
98  const Boundary& bnd = boundaries.get(s);
99  wordBoundaries->set(s,
100  ::Lattice::WordBoundary(bnd.time(),
101  ::Lattice::WordBoundary::Transit(bnd.transit().final, bnd.transit().initial)));
102  }
103  }
104  }
105  return ConstWordLatticeRef(wl);
106 }
