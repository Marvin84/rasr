73 void mapAlphabet(ConstAlphabetRef from, ConstAlphabetRef to, AlphabetMapping& mapping,
74  LabelId unknownId, u32 maximumUnknownsReported) {
75  u32 nUnknowns = 0;
76 
77  mapping.clear();
78  mapping.from_ = from;
79  mapping.to_ = to;
80  mapping.unkId_ = unknownId;
81 
82  if (!from || !to || from == to) {
83  mapping.type_ |= AlphabetMapping::typeIdentity;
84  }
85  else {
86  for (Alphabet::const_iterator i = from->begin(); i != from->end(); ++i) {
87  LabelId id = to->index(*i);
88  if (id != InvalidLabelId) {
89  mapping.map(LabelId(i), id);
90  }
91  else {
92  mapping.del(LabelId(i));
93  if (nUnknowns < maximumUnknownsReported && maximumUnknownsReported > 0)
94  Core::Application::us()->warning(
95  "Fsa::mapAlphabet: \"%s\" is not in second alphabet",
96  (*i).c_str());
97  else if (nUnknowns == maximumUnknownsReported && maximumUnknownsReported > 0)
98  Core::Application::us()->warning(
99  "Fsa::mapAlphabet: further unknown symbols not reported");
100  ++nUnknowns;
101  }
102  }
103  if (!mapping.isModifyingType())
104  mapping.clear();
105  }
106  if (nUnknowns > 0 && maximumUnknownsReported > 0)
107  Core::Application::us()->warning(
108  "Fsa::mapAlphabet: %d symbols were mapped to \"%s\"",
109  nUnknowns, ((unknownId == InvalidLabelId) ? "*INVALID*" : to->symbol(unknownId).c_str()));
110 }
