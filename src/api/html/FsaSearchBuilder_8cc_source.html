46  std::pair<std::string, std::string> getOperationAndName(const std::string& spec) const {
47  std::string op, name;

58  void getOperationList(Builder::Resources& r, OperationList& l) {
59  l.clear();
60  std::vector<std::string> operationNames = paramOperations(config);
61 
62  const Search::Wfst::Module_& module = Search::Wfst::Module::instance();
63  for (std::vector<std::string>::const_iterator i = operationNames.begin(); i != operationNames.end(); ++i) {
64  std::string op, name;
65  Core::tie(op, name) = getOperationAndName(*i);
66  Builder::Operation* o = module.getBuilderOperation(op, select(name), r);
67  if (!o) {
68  error("unknown operation '%s'", op.c_str());
69  }
70  else {
71  l.push_back(o);
72  }
73  }
74  log("%d operations", int(l.size()));
75  }

77  bool runOperations(OperationList& ops) {
78  typedef std::vector<Builder::Operation::AutomatonRef> AutomatonStack;
79  AutomatonStack stack;
80  for (OperationList::iterator iOp = ops.begin(); iOp != ops.end(); ++iOp) {
81  Builder::Operation& op = **iOp;
82  u32 input = op.nInputAutomata();
83 
84  if (stack.size() < input) {
85  error("operation %s requires %d operands, but stack size is: %d", op.name().c_str(), input, int(stack.size()));
86  return false;
87  }
88  for (u32 i = 0; i < input; ++i) {
89  if (!op.addInput(stack[stack.size() - (i + 1)])) {
90  error("cannot set input for operation '%s'", op.name().c_str());
91  return false;
92  }
93  }
94  if (op.consumeInput()) {
95  for (u32 i = 0; i < input; ++i) {
96  stack.pop_back();
97  }
98  }
99  Builder::Operation::AutomatonRef result = op.getResult();
100  if (!result) {
101  if (op.hasOutput()) {
102  error("operation '%s' could not produce any output", op.name().c_str());
103  return false;
104  }
105  }
106  else {
107  stack.push_back(result);
108  }
109  }
110  log("%d automata remaining", int(stack.size()));
111  while (!stack.empty()) {
112  delete stack.back();
113  stack.pop_back();
114  }
115  return true;
116  }

128  Builder::Resources resources(config);
129  OperationList ops;

132  return 0;
