107 class FilterArcNode : public FilterNode {

116 protected:
117  virtual ConstLatticeRef filter(ConstLatticeRef l) {
118  if (!l)
119  return ConstLatticeRef();
120  if (!input_.empty()) {
121  Fsa::LabelId inputLabel = l->getInputAlphabet()->index(input_);
122  if (inputLabel != Fsa::InvalidLabelId)
123  l = filterByInput(l, inputLabel);
124  else
125  criticalError(
126  "Label \"%s\" not in intput \"%s\" alphabet.",
127  input_.c_str(),
128  Lexicon::us()->alphabetName(Lexicon::us()->alphabetId(l->getInputAlphabet())).c_str());
129  }
130  if (!output_.empty()) {
131  Fsa::LabelId outputLabel = l->getOutputAlphabet()->index(output_);
132  if (outputLabel != Fsa::InvalidLabelId)
133  l = filterByOutput(l, outputLabel);
134  else
135  criticalError(
136  "Label \"%s\" not in intput \"%s\" alphabet.",
137  output_.c_str(),
138  Lexicon::us()->alphabetName(Lexicon::us()->alphabetId(l->getOutputAlphabet())).c_str());
139  }
140  return l;
141  }
