37  Math::LinearConjugateGradient<Math::FastVector<T>, T>::allocate(vector);
38  }

40  void setMatrix(Math::FastMatrix<T>& matrix) {
41  require(matrix.nRows() == matrix.nColumns());
42  matrix_.swap(matrix);
43  }
44 
45  void setPreconditioner(Math::FastVector<T>& diagonalVector) {
46  require(diagonalVector.size() == matrix_.nRows());
47  Math::DiagonalCgPreconditioner<Math::FastVector<T>, T>* preconditioner = new Math::DiagonalCgPreconditioner<Math::FastVector<T>, T>();
48  preconditioner->swap(diagonalVector);
49  Precursor::setPreconditioner(preconditioner);
50  }
51 
52  // not efficient, since symmetry is not exploited, but ok for testing purposes
53  virtual void applyMatrix(const Math::FastVector<T>& in, Math::FastVector<T>& out) {
54  matrix_.multiply(in, out);
55  }
