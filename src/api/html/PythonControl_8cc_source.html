15 /*
16 The PythonControl is a generic Python control interface.
17 It creates a Python interpreter via the standard CPython API.
18 It provides some generic commands which can be called through a single callback().
19 
20 It loads a Python module (pymod-name; add path via pymod-path if needed).
21 It expects these general functions in the Python module:
22 
23  def init(name, sprint_unit, reference, config)
24 
25 `name` is "Sprint.PythonControl" and `reference` is any Python object which
26 you can use as a reference (in case the module functions are used from multiple
27 sources at the same time). `sprint_unit` is whatever you used in the PythonControl constructor.
28 `init` should return an object which has these methods:
29 
30  def run_control_loop(callback)
31  def exit()
32 
33 `run_control_loop` will be called once when Sprint leaves the control to Python.
34 In `run_control_loop`, all the control can be done through `callback`, which is a function like:
35 
36  def callback(cmd, ...)
37 
38 `cmd` has to be a string.
39 See the code below for reference about supported commands.
40 */

110  void extractViterbiAlignment(const Speech::Alignment& alignment, Python::ObjRef& pyAlignment) {
111  u32 time = 0;
112  u32 t_offset = 0;
113  std::vector<u32> alignmentVec;
114  alignmentVec.reserve(alignment.size());
115  for (auto iter = alignment.begin(); iter != alignment.end(); ++iter) {
116  const Speech::AlignmentItem& item = *iter;
117  if (features_ && item.time >= features_->nColumns() + t_offset) {
118  parent_->error("Viterbi alignment: got time frame %u but sequence length is %u", item.time, features_->nColumns() + t_offset);
119  return;
120  }
121  if (item.time < time + t_offset) {
122  parent_->error("Viterbi alignment: expected time frame %u, got %u. (maybe Baum-Welch alignment?)", time + t_offset, item.time);
123  return;
124  }
125  while (item.time > time + t_offset) {
126  parent_->warning("Viterbi alignment: skipped time frame %u, got %u", time + t_offset, item.time);
127  if (features_)
128  features_->removeColumn(time);
129  t_offset++;
130  nSkippedAlignmentFrames_++;
131  }
132  require_eq(item.time, time + t_offset);
133  u32 classIdx = 0;
134  if (!_alignmentLabelIndex(alignment, item.emission, classIdx)) {
135  if (nSkippedAlignmentFrames_ == 0)
136  parent_->log("Viterbi alignment: we skip some frames because of the class label wrapper");
137  if (features_)
138  features_->removeColumn(time);
139  t_offset++;
140  nSkippedAlignmentFrames_++;
141  continue;
142  }
143  alignmentVec.push_back(classIdx);
144  ++time;
145  }
146  require_eq(time, alignmentVec.size());
147  if (features_) {
148  require_eq(time, features_->nColumns());
149  }
150  pyAlignment.clear();
151  Python::stdVec2numpy(criticalErrorFunc_, pyAlignment.obj, alignmentVec);
152  }

154  void extractSoftAlignment(const Speech::Alignment& alignment, Python::ObjRef& pySoftAlignment) {
155  u32 time = 0;
156  std::vector<u32> alignmentTime;
157  std::vector<u32> alignmentClassIdx;
158  std::vector<f32> alignmentWeight;
159  alignmentTime.reserve(alignment.size());
160  alignmentClassIdx.reserve(alignment.size());
161  alignmentWeight.reserve(alignment.size());
162  for (auto iter = alignment.begin(); iter != alignment.end(); ++iter) {
163  const Speech::AlignmentItem& item = *iter;
164  if (features_ && item.time >= features_->nColumns()) {
165  parent_->error("Soft alignment: got time frame %u but sequence length is %u", item.time, features_->nColumns());
166  return;
167  }
168  if (item.time < time) {
169  parent_->error("Soft alignment: expected time frame %u, got %u", time, item.time);
170  return;
171  }
172  if (item.time > time) {
173  if (item.time == time + 1)
174  ++time;
175  else {
176  parent_->warning("Soft alignment: skipped time frame %u, got %u", time, item.time);
177  time = item.time;
178  }
179  }
180  require_eq(item.time, time);
181  u32 classIdx = 0;
182  if (!_alignmentLabelIndex(alignment, item.emission, classIdx))
183  continue;
184  alignmentTime.push_back(time);
185  alignmentClassIdx.push_back(classIdx);
186  alignmentWeight.push_back(item.weight); // std space in [0,1]
187  }
188  require_eq(alignmentTime.size(), alignmentClassIdx.size());
189  require_eq(alignmentTime.size(), alignmentWeight.size());
190  // like sparse matrix in COOrdinate format
191  // http://docs.scipy.org/doc/scipy/reference/sparse.html
192  // https://github.com/scipy/scipy/blob/master/scipy/sparse/coo.py
193  Python::ObjRef pyI, pyJ, pyData;
194  if (!Python::stdVec2numpy(criticalErrorFunc_, pyI.obj, alignmentTime))
195  return;
196  if (!Python::stdVec2numpy(criticalErrorFunc_, pyJ.obj, alignmentClassIdx))
197  return;
198  if (!Python::stdVec2numpy(criticalErrorFunc_, pyData.obj, alignmentWeight))
199  return;
200  pySoftAlignment.takeOver(PyTuple_Pack(3, pyI.obj, pyJ.obj, pyData.obj));
201  }

346  PyObject* calculateCriterion(PyObject* args, PyObject* kws) {
347  static const char* kwlist[] = {
348  "command",
349  "posteriors",
350  "orthography",
351  "alignment",
352  "output_error_type",
353  "segment_name",
354  NULL};
355  const char* _cmd = NULL;
356  PyObject* posteriorsPy = NULL; // borrowed
357  const char* orthography = NULL;
358  PyObject* alignmentPy = NULL; // borrowed
359  const char* outputErrorTypeC = NULL;
360  const char* segmentName = NULL;
361 
362  if (!PyArg_ParseTupleAndKeywords(
363  args, kws, "sO|sOss:callback", (char**)kwlist,
364  &_cmd, &posteriorsPy, &orthography, &alignmentPy, &outputErrorTypeC, &segmentName))
365  return NULL;
366 
367  enum {
368  ET_None,
369  ET_ErrorSignal,
370  ET_ErrorSignalBeforeSoftmax,
371  ET_PseudoTargets
372  } outputErrorType = ET_None;
373  if (!outputErrorTypeC || strcmp(outputErrorTypeC, "none") == 0)
374  outputErrorType = ET_None;
375  else if (strcmp(outputErrorTypeC, "error-signal") == 0)
376  outputErrorType = ET_ErrorSignal;
377  else if (strcmp(outputErrorTypeC, "error-signal-before-softmax") == 0)
378  outputErrorType = ET_ErrorSignalBeforeSoftmax;
379  else if (strcmp(outputErrorTypeC, "pseudo-targets") == 0)
380  outputErrorType = ET_PseudoTargets;
381  else {
382  PyErr_Format(PyExc_ValueError, "PythonControl callback(): calculate_criterion: unknown output_error_type '%s'", outputErrorTypeC);
383  return NULL;
384  }
385 
386  Math::CudaMatrix<f32> posteriors;
387  if (!Python::numpy2nnMatrix(getPythonCriticalErrorFunc(), posteriorsPy, posteriors))
388  return NULL;
389  posteriors.initComputation(true);
390 
391  Math::CudaVector<u32> alignment;
392  if (alignmentPy) {
393  if (orthography) {
394  PyErr_Format(PyExc_ValueError, "PythonControl callback(): calculate_criterion: you should provide either an alignment, or the orthography, but not both");
395  return NULL;
396  }
397 
398  if (!Python::numpy2nnVector(getPythonCriticalErrorFunc(), alignmentPy, alignment))
399  return NULL;
400  alignment.initComputation(true);
401  }
402 
403  _initCriterion();
404 
405  Bliss::Corpus dummyCorpus;
406  Bliss::Recording dummyRecording(&dummyCorpus);
407  Bliss::SpeechSegment speechSegment(&dummyRecording); // must be in scope until end when used
408  // Note that segmentName is the full segment name, so setName() is not perfectly correct.
409  // We would have to split it by "/" and set the corpus-name and recording-name.
410  if (segmentName)
411  speechSegment.setName(segmentName);
412  if (orthography)
413  speechSegment.setOrth(orthography);
414 
415  if (orthography)
416  criterion_->inputSpeechSegment(speechSegment, posteriors);
417  else if (alignmentPy)
418  criterion_->inputAlignment(alignment, posteriors);
419  else
420  criterion_->input(posteriors);
421 
422  if (criterion_->discardCurrentInput())
423  return Py_BuildValue("(OO)", Py_None, Py_None);
424 
425  f32 resLoss = 0;
426  criterion_->getObjectiveFunction(resLoss);
427 
428  Python::ObjRef resOutputError;
429  switch (outputErrorType) {
430  case ET_None:
431  resOutputError.copyRef(Py_None);
432  break;
433  case ET_ErrorSignal: {
434  Math::CudaMatrix<f32> errorSignal(posteriors.nRows(), posteriors.nColumns());
435  errorSignal.initComputation(false);
436  errorSignal.setToZero();
437  criterion_->getErrorSignal(errorSignal);
438  errorSignal.finishComputation(true);
439  if (!Python::nnMatrix2numpy(getPythonCriticalErrorFunc(), resOutputError.obj, errorSignal))
440  return NULL;
441  break;
442  }
443  case ET_ErrorSignalBeforeSoftmax: {
444  Math::CudaMatrix<f32> errorSignal(posteriors.nRows(), posteriors.nColumns());
445  errorSignal.initComputation(false);
446  errorSignal.setToZero();
447  SoftmaxLayer<f32> dummyLayer(getConfiguration());
448  criterion_->getErrorSignal_naturalPairing(errorSignal, dummyLayer);
449  errorSignal.finishComputation(true);
450  if (!Python::nnMatrix2numpy(getPythonCriticalErrorFunc(), resOutputError.obj, errorSignal))
451  return NULL;
452  break;
453  }
454  case ET_PseudoTargets: {
455  Math::CudaMatrix<f32>* targets = criterion_->getPseudoTargets();
456  if (targets) {
457  targets->finishComputation(true);
458  if (!Python::nnMatrix2numpy(getPythonCriticalErrorFunc(), resOutputError.obj, *targets))
459  return NULL;
460  targets->initComputation(false);
461  }
462  else
463  resOutputError.copyRef(Py_None);
464  break;
465  }
466  }
467 
468  PyObject* res = Py_BuildValue("(fO)", resLoss, resOutputError.obj);
469  return res;
470  }

472  PyObject* getCtcAlignment(PyObject* args, PyObject* kws) {

607  bool _readAlignmentFromCacheArchive(const std::shared_ptr<Core::Archive>& a,
608  const std::string& segmentName,
609  Flow::DataPtr<Flow::DataAdaptor<Speech::Alignment>>& alignmentRef) {
610  // Be sure that the necessary Flow datatypes are registered.
611  Flow::Module::instance();
612  Speech::Module::instance();
613 
614  Core::ArchiveReader reader(*a, segmentName);
615  if (!reader.isOpen()) {
616  PyErr_Format(PyExc_ValueError, "PythonControl: cannot read entry (segment-name) '%s' in archive",
617  segmentName.c_str());
618  return false;
619  }
620 
621  Core::BinaryInputStream b(reader);
622  std::string datatypeName;
623  if (!(b >> datatypeName)) {

721  PyObject* analyzeAlignmentFromCache(PyObject* args, PyObject* kws) {
722  static const char* kwlist[] = {
723  "command",
724  "cache_filename",
725  "segment_name",
726  "silence_allophone_state_idx",
727  NULL};
728  const char* _cmd = NULL;
729  const char* cache_filename_c = NULL;
730  const char* segment_name_c = NULL;
731  int silence_allophone_state_idx = -1;
732 
733  if (!PyArg_ParseTupleAndKeywords(args, kws, "ss|si:callback", (char**)kwlist,
734  &_cmd, &cache_filename_c, &segment_name_c, &silence_allophone_state_idx))
735  return NULL;
736 
737  Am::AllophoneStateIndex silenceAllophoneStateIdx = 0;
738  if (silence_allophone_state_idx >= 0)
739  silenceAllophoneStateIdx = silence_allophone_state_idx;
740  else {
741  _initAcousticModel();
742  silenceAllophoneStateIdx = acousticModel_->silenceAllophoneStateIndex();
743  }
744 
745  std::shared_ptr<Core::Archive> a = getCacheArchive(cache_filename_c);
746  if (!a) {
747  PyErr_Format(PyExc_ValueError, "PythonControl: cannot open cache archive '%s'",
748  cache_filename_c);
749  return NULL;
750  }
751 
752  size_t nTime = 0;
753  size_t nSilForward = 0, nSilLoop = 0;
754  size_t nPhonForward = 0, nPhonLoop = 0;
755 
756  std::string segmentName;
757  Core::Archive::const_iterator archive_iter;
758  if (!segment_name_c)
759  archive_iter = a->files();
760  while (true) {
761  if (!segment_name_c) {
762  // Skip over *.attribs entries in the cache archive.
763  while (archive_iter) {
764  const std::string& entryName = archive_iter.name();
765  if (entryName.size() >= strlen(".attribs") && entryName.substr(entryName.size() - strlen(".attribs")) == ".attribs")
766  ++archive_iter;
767  else
768  break;
769  }
770  if (!archive_iter)
771  break;
772  segmentName = archive_iter.name();
773  }
774  else
775  segmentName = segment_name_c;
776 
777  Flow::DataPtr<Flow::DataAdaptor<Speech::Alignment>> alignmentRef;
778  if (!_readAlignmentFromCacheArchive(a, segmentName, alignmentRef))
779  return NULL;
780 
781  require(alignmentRef);
782  const Speech::Alignment& alignment = alignmentRef->data();
783 
784  if (alignment.labelType() != Speech::Alignment::allophoneStateIds) {
785  PyErr_Format(PyExc_ValueError, "PythonControl: alignment label type is not allophone-state-id");
786  return NULL;
787  }
788 
789  size_t time = 0;
790  Am::AllophoneStateIndex lastAllophoneStateIdx = silenceAllophoneStateIdx;
791  for (auto iter = alignment.begin(); iter != alignment.end(); ++iter) {
792  const Speech::AlignmentItem& item = *iter;
793  if (item.time < time) {
794  PyErr_Format(PyExc_ValueError, "PythonControl: Viterbi alignment: expected time frame %zu, got %u."
795  " (maybe Baum-Welch alignment?)",
796  time, item.time);
797  return NULL;
798  }
799  if (item.time > time) {
800  PyErr_Format(PyExc_ValueError, "PythonControl: Viterbi alignment: skipped time frame %zu, got %u",
801  time, item.time);
802  return NULL;
803  }
804  require_eq(item.time, time);
805  Am::AllophoneStateIndex allophoneStateIdx = item.emission;
806  if (time > 0) {
807  if (lastAllophoneStateIdx == silenceAllophoneStateIdx) {
808  if (allophoneStateIdx != silenceAllophoneStateIdx)
809  nSilForward++;
810  else
811  nSilLoop++;
812  }
813  else {
814  if (allophoneStateIdx != lastAllophoneStateIdx)
815  nPhonForward++;
816  else
817  nPhonLoop++;
818  }
819  }
820 
821  ++time;
822  lastAllophoneStateIdx = allophoneStateIdx;
823  }
824  nTime += time;
825 
826  if (!segment_name_c)
827  ++archive_iter;
828  else
829  break;
830  }
831 
832  Python::ObjRef res;
833  res.takeOver(PyDict_New());
834  if (!res)
835  return NULL;
836  Python::dict_SetItemString(res.obj, "length", nTime);
837  Python::dict_SetItemString(res.obj, "nSilForward", nSilForward);
838  Python::dict_SetItemString(res.obj, "nSilLoop", nSilLoop);
839  Python::dict_SetItemString(res.obj, "nPhonForward", nPhonForward);
840  Python::dict_SetItemString(res.obj, "nPhonLoop", nPhonLoop);
841  return res.release();
842  }

844  PyObject* callback(PyObject* args, PyObject* kws) {
845  Py_ssize_t nargs = PyTuple_Size(args);
846  if (nargs < 1) {
847  PyErr_SetString(PyExc_TypeError, "PythonControl callback(): requires at least one arg. try callback('help')");
848  return NULL;
849  }
850 
851  PyObject* cmd = PyTuple_GetItem(args, 0); // borrowed
852  if (PyUnicode_KIND(cmd) != PyUnicode_1BYTE_KIND){
853  PyErr_SetString(PyExc_TypeError, "PythonControl callback(): first arg is not a 1BYTE unicode string");
854  return NULL;
855  }
856  const char* cmd_cs = (const char*)PyUnicode_1BYTE_DATA(cmd);
857  if (!cmd_cs) {
858  PyErr_SetString(PyExc_TypeError, "PythonControl callback(): first arg must be a string");
859  return NULL;
860  }
861  std::string cmd_s(cmd_cs);
862 
863  if (cmd_s == "version")
864  return version();
865  if (cmd_s == "version_number")
866  return versionNumber();
867  if (cmd_s == "help")
868  return help();
869  if (cmd_s == "init_criterion")
870  return initCriterion();
871  if (cmd_s == "calculate_criterion")
872  return calculateCriterion(args, kws);
873  if (cmd_s == "get_ctc_alignment")
874  return getCtcAlignment(args, kws);
875  if (cmd_s == "export_allophone_state_fsa_by_orthography")
876  return exportAllophoneStateFsaByOrthography(args, kws);
877  if (cmd_s == "export_allophone_state_fsa_by_segment_name")
878  return exportAllophoneStateFsaBySegName(args, kws);
879  if (cmd_s == "get_orthography_by_segment_name")
880  return getOrthographyBySegmentName(args, kws);
881  if (cmd_s == "get_alignment_from_cache")
882  return getAlignmentFromCache(args, kws);
883  if (cmd_s == "analyze_alignment_from_cache")
884  return analyzeAlignmentFromCache(args, kws);
885 
886  PyErr_Format(PyExc_ValueError, "PythonControl callback(): unknown command '%s'", cmd_cs);
887  return NULL;
888  }
