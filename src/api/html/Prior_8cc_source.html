67 template<typename T>
68 static void updatePriors(
69  const typename Types<T>::NnVector& logPriors,
70  T logPriorScale,
71  typename Types<T>::NnVector& priors,
72  T& logNorm) {
73  priors.copy(logPriors);
74  priors.scale(logPriorScale);
75  // apply softmax
76  priors.exp();
77  T norm = priors.l1norm();
78  norm = std::max(norm, Core::Type<T>::epsilon);
79  norm = std::min(norm, T(1.0) / Core::Type<T>::epsilon);
80  priors.scale(1.0 / norm);
81  logNorm = std::log(norm);
82 }

84 template<typename T>
85 void Prior<T>::trainSoftmax(const NnVector& errorSignal, T errFactor) {
86  u32 nClasses = size();
87  require_eq(errorSignal.nRows(), nClasses);
88 
89  logPrior_.initComputation(false); // no need to sync, GPU-mem is already correct
90 
91  T logNorm;
92  NnVector priors(nClasses);
93  priors.initComputation(false);
94  updatePriors(logPrior_, scale_, priors, logNorm);
95 
96  NnVector p(nClasses);
97  p.initComputation(false);
98  p.copy(errorSignal);
99 
100  NnVector grad(nClasses);
101  grad.initComputation(false);
102  grad.setToZero();
103  grad.add(p, (T)-1);
104 
105  // XXX: This would probably be more stable in log space.
106  // P''' = P'' .* 1/p(a)
107  p.elementwiseDivision(priors);
108 
109  // XXX: This syncs to CPU. Bottleneck?
110  // = P''' * p(a)
111  T f = p.dot(priors);
112 
113  // grad = -P'' + p(a) * (P''' * p(a))
114  grad.add(priors, f);
115 
116  // XXX: This would probably be more stable in log space.
117  // errFactor is the error factor of some outer error function. (L(P))
118  // This is the gradient descent step.
119  logPrior_.add(grad, -errFactor * learningRate_ / scale_);
120 
121  updatePriors(logPrior_, scale_, priors, logNorm);
122  logPrior_.addConstantElementwise(-logNorm / scale_);
123 
124  logPrior_.finishComputation(true);
125 }

127 template<typename T>
128 void Prior<T>::setFromClassCounts(const Statistics<T>& statistics, const Math::Vector<T>& classWeights) {
129  logPrior_.resize(classWeights.size());
130  this->log("calculating prior from class counts");
131  this->log("using back-off-count ") << backOffCount_;
132  T totalWeight = 0.0;
133  for (u32 c = 0; c < logPrior_.size(); c++) {
134  if (statistics.classCount(c) == 0)
135  this->warning("zero observations for class: ") << c;
136  logPrior_.at(c) = std::max(statistics.classCount(c), backOffCount_);
137  logPrior_.at(c) *= classWeights.at(c);
138  if (statisticsChannel_.isOpen()) {
139  std::stringstream ss;
140  ss << "class-" << c;
141  std::string xmlName = ss.str();
142  statisticsChannel_ << Core::XmlOpen(xmlName.c_str())
143  << Core::XmlFull("number-of-observations", statistics.classCount(c))
144  << Core::XmlFull("weighted-number-of-observations", logPrior_.at(c))
145  << Core::XmlClose(xmlName.c_str());
146  }
147  totalWeight += logPrior_.at(c);
148  }
149  for (u32 c = 0; c < logPrior_.size(); c++)
150  logPrior_.at(c) = logPrior_.at(c) == 0 ? Core::Type<T>::min : std::log(logPrior_.at(c) / totalWeight);
151  if (statisticsChannel_.isOpen())
152  statisticsChannel_ << logPrior_;
153  // sync to GPU memory. train() expects it to be up-to-date
154  logPrior_.initComputation(true);
155  logPrior_.finishComputation(false);
156 }

158 template<typename T>
159 void Prior<T>::setFromMixtureSet(Core::Ref<const Mm::MixtureSet> mixtureSet, const ClassLabelWrapper& labelWrapper) {
160  log("calculating prior from mixture set");
161  // get counts from mixture set
162  std::vector<f32> priorFromMixtureSet;
163  priorFromMixtureSet.resize(mixtureSet->nMixtures());
164  require_eq(priorFromMixtureSet.size(), labelWrapper.nClasses());
165  for (size_t m = 0; m < mixtureSet->nMixtures(); ++m) {
166  const Mm::Mixture* mixture = mixtureSet->mixture(m);
167  size_t nDensities = mixture->nDensities();
168  for (size_t dns = 0; dns < nDensities; ++dns) {
169  priorFromMixtureSet.at(m) += mixture->weight(dns); /// returns exp(logWeights_[densityInMixture]), @see Mm/Mixture.cc
170  }
171  }
172 
173  // map counts corresponding to order in output layer
174  logPrior_.resize(labelWrapper.nClassesToAccumulate());
175  for (u32 m = 0; m < mixtureSet->nMixtures(); m++) {
176  if (labelWrapper.isClassToAccumulate(m))
177  logPrior_.at(labelWrapper.getOutputIndexFromClassIndex(m)) = priorFromMixtureSet.at(m);
178  }
179 
180  f32 observationWeight = 0.0;
181  // normalize and apply log
182  if (compatibilityMode_)
183  observationWeight = std::accumulate(priorFromMixtureSet.begin(), priorFromMixtureSet.end(), 0.0);
184  else
185  observationWeight = std::accumulate(logPrior_.begin(), logPrior_.end(), 0.0);
186 
187  for (u32 c = 0; c < logPrior_.size(); ++c)
188  logPrior_.at(c) = std::log(logPrior_.at(c) / observationWeight);
189  if (statisticsChannel_.isOpen())
190  statisticsChannel_ << logPrior_;
191  // sync to GPU memory. train() expects it to be up-to-date
192  logPrior_.initComputation(true);
193  logPrior_.finishComputation(false);
194 }
