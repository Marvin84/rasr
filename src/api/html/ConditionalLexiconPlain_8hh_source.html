67  virtual Translation::Cost getCost(const size_t index, const std::vector<Fsa::LabelId>& key) const {
68  bool exists = true;
69  const Lexicon::Node* currentNode = lexica_[index]->rootNode();
70  std::vector<Fsa::LabelId>::const_iterator i = key.begin();
71  while (exists && i != key.end()) {
72  currentNode = currentNode->follow(*i);
73  if (currentNode == 0) {
74  exists = false;

76  ++i;
77  }
78 
79  Translation::Cost result;
80  if (exists) {
81  result = currentNode->getData();
82  }
83  else {

126  virtual void addValue(const size_t index, const std::vector<Fsa::LabelId>& key, Translation::Cost value) {
127  /*
128  * traverse tree to store key and create arcs that do not exist so far.
129  * existence has to be memorized in order to decide whether to store or
130  * to increase the current key.
131  */
132 
133  if (lexica_.size() < index + 1) {
134  //initialize new lexica
135  size_t oldSize = lexica_.size();
136  lexica_.resize(index + 1);
137  for (size_t i = oldSize; i <= index; ++i) {
138  lexica_[i] = LexiconRef(new Lexicon);
139  }
140  }
141 
142  Lexicon::Node* currentNode = lexica_[index]->rootNode();
143  currentNode->setData(currentNode->getData() + value);
144  std::vector<Fsa::LabelId>::const_iterator i = key.begin();
145  while (i != key.end()) {
146  Lexicon::Node* newNode = currentNode->follow(*i);
147  if (newNode == 0) {
148  newNode = currentNode->followOrExpand(*i);
149  newNode->setData(value);
150  }
151 
152  else {
153  newNode->setData(newNode->getData() + value);
154  }
155 
156  currentNode = newNode;
157  ++i;
158  }
159  }
