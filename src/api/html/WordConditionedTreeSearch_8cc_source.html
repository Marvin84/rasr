45 /*
46  * Delayed tree deletion:
47  *
48  * Normally trees without active hypotheses would be deleted
49  * immediatelly. With tree deletion latency, empty trees will stay
50  * around for @c treeDeletionLatency_ time-frames. This is meant to
51  * reduce to over-head of immediate reactivation.
52  *
53  *
54  * Language model look-ahead laziness
55  *
56  * Normally when a new tree is activated, the corresponding LM
57  * look-ahead table is calculated immediately. This may cause many
58  * wasteful calculations, because the tree can be pruned soon after
59  * that. With look-ahead laziness the strategy is as follows: As long
60  * as the tree contains less than @c lmLookaheadLaziness_ states,
61  * the LM look-ahead is only used if the table is already in the
62  * cache. When the number of states in the tree grows beyond that
63  * limit, calculation of the table is enforced.
64  */

426 /**
427  * create or update score and trace of a state hypothesis.
428  * a state hypothesis is updated if a state hypothesis for the same exists
429  * with a higher score
430  */
431 inline void WordConditionedTreeSearch::SearchSpace::activateOrUpdateStateHypothesis(
432  StateTree::StateId state,
433  TraceId trace,
434  Score score) {
435  Score prospect = score;
436  if (anticipatedLookaheadPruning_) {
437  if (currentLookahead_)
438  prospect += currentLookahead_->score(state);
439  if (prospect > currentStateTreshold_)
440  return;
441  }
442  StateHypothesisIndex shi = stateHypothesisForTreeState[state];
443  if (shi >= StateHypothesisIndex(newStateHypotheses.size()) ||
444  shi < currentTreeFirstNewStateHypothesis ||
445  newStateHypotheses[shi].state != state) {
446  stateHypothesisForTreeState[state] = newStateHypotheses.size();
447  newStateHypotheses.push_back(StateHypothesis(state, trace, score, prospect));
448  }
449  else {
450  StateHypothesis& sh(newStateHypotheses[shi]);
451  verify_(sh.state == state);
452  if (sh.score >= score) {
453  sh.score = score;
454  sh.prospect = prospect;
455  sh.trace = trace;
456  }
457  }
458  if (anticipatedLookaheadPruning_ && prospect < currentBestScore_) {
459  currentBestScore_ = prospect;
460  currentStateTreshold_ = prospect + pruningThreshold_;
461  }
462 }

464 /**
465  * expand state @c sh using forward, loop, and skip transitions.
466  * new state hypotheses are stored in newStateHypotheses
467  */
468 inline void WordConditionedTreeSearch::SearchSpace::expandState(const StateHypothesis& sh) {
469  const Am::StateTransitionModel& tdp(*tree_->transitionModel(sh.state));
470  StateTree::SuccessorIterator state, state_end;
471  Score score;
472 
473  // self-loop
474  score = sh.score + tdp[Am::StateTransitionModel::loop];
475  if (score < Core::Type<Score>::max) {
476  state = sh.state;
477  activateOrUpdateStateHypothesis(state, sh.trace, score);
478  }
479 
480  // forward
481  score = sh.score + tdp[Am::StateTransitionModel::forward];
482  if (score < Core::Type<Score>::max)
483  for (tie(state, state_end) = tree_->successors(sh.state); state < state_end; ++state)
484  activateOrUpdateStateHypothesis(*state, sh.trace, score);
485 
486  // skip
487  score = sh.score + tdp[Am::StateTransitionModel::skip];
488  if (score < Core::Type<Score>::max)
489  for (tie(state, state_end) = tree_->successors2(sh.state); state < state_end; ++state)
490  activateOrUpdateStateHypothesis(*state, sh.trace, score);
491 }

558 /**
559  * calculate s.prospect (using LM lookahead) and find best state hypothesis in all active trees.
560  * result: bestStateHypothesis_
561  */
562 void WordConditionedTreeSearch::SearchSpace::determinePruningCriterion() {
563  if (anticipatedLookaheadPruning_) {
564  // lookahead scores are already computed in expand()
565  // bestStateHypothesis will be computed using bestStateHypothesis()
566  bestStateHypothesis_ = stateHypotheses.end();
567  return;
568  }
569  bestStateHypothesis_ = stateHypotheses.begin();
570  Score bestScore = Core::Type<Score>::max;
571  for (TreeInstanceList::const_iterator t = activeTrees.begin(); t != activeTrees.end(); ++t) {
572  const TreeInstance& at(**t);
573  StateHypothesesList::iterator sh_begin = stateHypotheses.begin() + at.stateHypotheses.begin;
574  StateHypothesesList::iterator sh_end = stateHypotheses.begin() + at.stateHypotheses.end;
575 
576  for (StateHypothesesList::iterator sh = sh_begin; sh != sh_end; ++sh) {
577  sh->prospect = sh->score;
578  }
579 
580  if (at.lookahead) {
581  const LanguageModelLookahead::ContextLookahead& la(*at.lookahead);
582  for (StateHypothesesList::iterator sh = sh_begin; sh != sh_end; ++sh)
583  sh->prospect += la.score(sh->state);
584  }
585 
586  for (StateHypothesesList::iterator sh = sh_begin; sh != sh_end; ++sh)
587  if (bestScore > sh->prospect)
588  bestScore = (bestStateHypothesis_ = sh)->prospect;
589  }
590 }

632  TreeInstance* at(activeTrees[treeIn]);
633  verify_(hypIn == hypBegin + at->stateHypotheses.begin);
634  at->stateHypotheses.begin = hypOut - hypBegin;
635  for (treeHypEnd = hypBegin + at->stateHypotheses.end; hypIn < treeHypEnd; ++hypIn) {
636  verify_(hypIn < stateHypotheses.end());
637  if (hypIn->prospect <= threshold)
638  *(hypOut++) = *hypIn;
639  }
640  at->stateHypotheses.end = hypOut - hypBegin;
641  if (!mayDeactivateTree(at))
642  activeTrees[treeOut++] = at;
643  }
644  stateHypotheses.erase(hypOut, stateHypotheses.end());
645  activeTrees.resize(treeOut);
646  bestStateHypothesis_ = stateHypotheses.end();
647 }

921 /**
922  * Fall back strategy for finding the best sentence hypothesis when
923  * there is no active word end hypothesis. Typically used when
924  * getBestSentenceEnd() fails. This problem can occur when the
925  * recording is truncated in the middle of a word and pruning is
926  * tight. The strategy is to select the best state from each active
927  * tree and consider it as a "word end" without a pronunciation.
928  */
929 
930 Ref<WordConditionedTreeSearch::Trace> WordConditionedTreeSearch::SearchSpace::getSentenceEndFallBack(Core::Ref<const Lm::ScaledLanguageModel> lm,
931  TimeframeIndex time,
932  bool shallCreateLattice) {

934  for (TreeInstanceList::const_iterator ti = activeTrees.begin(); ti != activeTrees.end(); ++ti) {
935  const TreeInstance* at(*ti);
936  if (!at->nStateHypotheses())
937  continue;

939  StateHypothesis bestStateHyp(stateHypotheses[s]);
940  while (++s < at->stateHypotheses.end) {
941  const StateHypothesis& sh(stateHypotheses[s]);
942  if (bestStateHyp.score > sh.score)
943  bestStateHyp = sh;
944  }
945  Ref<Trace> pre(at->trace(bestStateHyp));
946  Ref<Trace> t(new Trace(pre, 0, time, pre->score,
947  tree_->describeRootState(bestStateHyp.state)));
948  t->score.acoustic = globalScoreOffset_ + bestStateHyp.score - pre->score.lm;
949  t->score.lm += lm->sentenceEndScore(at->history);
950  if (!best || best->score > t->score) {
951  if (shallCreateLattice)
952  t->sibling = best;
953  best = t;
954  }
955  else {
956  if (shallCreateLattice) {
957  t->sibling = best->sibling;
958  best->sibling = t;
959  }
960  }
961  }
962  return best;

965 Ref<WordConditionedTreeSearch::Trace> WordConditionedTreeSearch::SearchSpace::getLastUnambiguousTrace() const {
966  std::vector<Trace*> traces;
967  for (WordEndHypothesisList::const_iterator weh = wordEndHypotheses.begin(); weh != wordEndHypotheses.end(); ++weh)
968  traces.push_back(weh->trace.get());
969  for (TreeInstanceList::const_iterator at = activeTrees.begin(); at != activeTrees.end(); ++at)
970  for (std::vector<Ref<Trace>>::const_iterator t = (*at)->traces.begin(); t != (*at)->traces.end(); ++t)
971  traces.push_back(t->get());
972 
973  std::unordered_map<const Trace*, u32, Core::conversion<const Trace*, size_t>> m;
974  std::vector<Trace*> chain;
975  for (Trace* t = traces.back(); t; t = t->predecessor.get()) {
976  m[t] = 2 + chain.size();
977  chain.push_back(t);
978  }
979  traces.pop_back();
980 
981  u32 mLimit = 2;
982  while (traces.size()) {
983  for (Trace* t = traces.back(); t; t = t->predecessor.get()) {
984  u32 mt = m[t];
985  if (mt == 1)
986  break;
987  m[t] = 1;
988  if (mLimit < mt) {
989  mLimit = mt;
990  break;
991  }
992  }
993  traces.pop_back();
994  }
995 
996  verify(mLimit >= 2);
997  mLimit -= 2;
998  verify(mLimit < chain.size());
999  return Ref<Trace>(chain[mLimit]);
1000 }
