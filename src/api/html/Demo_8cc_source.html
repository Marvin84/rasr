101 bool DemoNode::configure() {
102  if (demoChannel_.isOpen()) {
103  demoChannel_ << Core::XmlOpen("configure method of " + fullName());
104  demoChannel_ << Core::XmlComment("configure method of a Demo Node is called");
105 
106  // require that the number of input ports equals the number of output
107  // ports
108  demoChannel_ << Core::XmlEmpty("checking number of input ports == number of output ports");
109  demoChannel_ << (Core::XmlEmpty("") + Core::XmlAttribute("nInput", nInputs()) + Core::XmlAttribute("nOutput", nOutputs()));
110  }
111  // various methods like warning, error and so forth are defined in
112  // Core:Component and hence inherited in all classes !
113  if (nInputs() != nOutputs()) {
114  error("Number of input and output ports differs !");
115  }
116 
117  if (demoChannel_.isOpen()) {
118  demoChannel_ << Core::XmlComment("number of input ports matches number of output ports");
119  demoChannel_ << (Core::XmlEmpty("ports") + Core::XmlAttribute("number of ports", nInputs()));
120  }
121  bool status = true;
122  // configure input port attributes
123  // note due to the generic number of possible input ports
124  // and the variety of possible attribute types it is not
125  // possible to check whehter or not the attribute types
126  // are correct; at least not in a reasonable way
127  for (PortId port = 0; port < nInputs(); ++port) {
128  if (demoChannel_.isOpen()) {
129  demoChannel_ << Core::XmlOpen("fetching attributes")
130  << Core::XmlFull("from input port", port)
131  << Core::XmlClose("fetching attributes");
132  }
133  Core::Ref<const Attributes> attr = getInputAttributes(port);
134  // note the output of attributes into an xml file is done
135  // in the putOutputAttributes method inherited from Flow::Node
136  if (!putOutputAttributes(port, attr)) {
137  if (demoChannel_.isOpen()) {
138  demoChannel_ << (Core::XmlEmpty("forwarding of attributes failed") + Core::XmlAttribute("port", port));
139  }
140  status = false;
141  }
142  }
143  if (demoChannel_.isOpen()) {
144  demoChannel_ << (Core::XmlEmpty("forwarding status of attributes") + Core::XmlAttribute("status", status));
145  demoChannel_ << Core::XmlClose("configure method of " + fullName());
146  }
147  return status;
148 }

150 bool DemoNode::work(PortId output) {
151  if (demoChannel_.isOpen()) {
152  demoChannel_ << Core::XmlOpen("work method of " + fullName());
153  }
154  require(nInputs() > 0);
155  require(nInputs() == nOutputs());
156 
157  // create necessary smart data pointer
158  DataPtr<Data> data;
159  if (demoChannel_.isOpen()) {
160  demoChannel_ << (Core::XmlEmpty("fetching data from input port") +
161  Core::XmlAttribute("port", output));
162  }
163  // note: because we require that the number of output ports equalls the
164  // the number of input notes and that the data of input port x should be
165  // send to output port x the getData method is called with output
166  // in a "real" flow node you have to use the correct input port id !
167  if (!getData(output, data)) {
168  // put "empty" object into the queue of the flow network
169  // the following node won't get data and signal this
170  // Flow is a pull network
171  if (demoChannel_.isOpen()) {
172  demoChannel_ << Core::XmlEmpty("no data fetched");
173  demoChannel_ << Core::XmlEmpty("provding empty data for successor nodes");
174  }
175  /* in cases you require an output port to have a specific number of links
176  *
177  * if(nOutputLinks(output) > 0){
178  * "do some stuff like providing data"
179  * } else {
180  * "do some error handling like: "
181  * demoChannel_ << (Core::XmlEmpty("no activ links from output port")
182  * + Core::XmlAttribute("output port",output))
183  * << Core::XmlClose("work method of " + fullName());
184  * return false;
185  * }
186  */
187 
188  // basically you send an END OF SEQUENCE signal
189  bool success = putData(output, data.get());
190  if (demoChannel_.isOpen()) {
191  demoChannel_ << (Core::XmlEmpty("data provided") + Core::XmlAttribute("output port", output))
192  << Core::XmlClose("work method of " + fullName());
193  }
194  return success;
195  }
196 
197  // forwarding data
198  // note: similar to the putAttributes method the data is dumped in the
199  // putData method.
200  if (demoChannel_.isOpen()) {
201  demoChannel_ << (Core::XmlEmpty("providing data from input port") + Core::XmlAttribute("port", output));
202  }
203  if (!putData(output, data.get())) {
204  if (demoChannel_.isOpen()) {
205  demoChannel_ << (Core::XmlEmpty("no data provided") + Core::XmlAttribute("output port", output))
206  << Core::XmlClose("work method of " + fullName());
207  }
208  return false;
209  }
210  if (demoChannel_.isOpen()) {
211  demoChannel_ << (Core::XmlEmpty("data provided") + Core::XmlAttribute("output port", output))
212  << Core::XmlClose("work method of " + fullName());
213  }
214  return true;
215 }
