122 Core::Ref<MixtureSet> Module_::createMixtureSet(
123  CovarianceTyingType covarianceTying,
124  size_t nMixtures, size_t dimension, size_t nDensitiesPerMixture) const {
125  Core::Ref<MixtureSet> result;
126  MixtureSetBuilder* builder = 0;
127  switch (covarianceTying) {
128  case pooledCovariance:
129  builder = new PooledCovarianceBuilder(nMixtures, nDensitiesPerMixture);
130  break;
131  case mixtureSpecificCovariance:
132  builder = new MixtureSpecificCovarianceBuilder(nMixtures, nDensitiesPerMixture);
133  break;
134  case noCovarianceTying:
135  builder = new MixtureSetBuilder(nMixtures, nDensitiesPerMixture);
136  break;
137  default:
138  Core::Application::us()->error("Covariance tying type not given.");
139  }
140  if (builder != 0) {
141  result = Core::ref(new MixtureSet);
142  builder->build(*result, dimension);
143  delete builder;
144  }
145  return result;
146 }

152 Core::Ref<MixtureSet> Module_::readMixtureSet(const std::string& filename, const Core::Configuration& configuration) const {
153  Core::Ref<MixtureSet> result;
154  if (filename.empty()) {
155  return result;
156  }
157  Core::Application::us()->log("Loading mixture set from file \"%s\"", filename.c_str());
158  result = MixtureSetReader(configuration).read<MixtureSet>(filename);
159  if (result)
160  Core::Application::us()->log("mixtureset read from \"%s\"", filename.c_str());
161  else
162  Core::Application::us()->error("failed to read mixtureset from \"%s\"", filename.c_str());
163 
164  if (result) {
165  ComponentIndex offset = paramReducedMixtureSetDimensionOffset(configuration);
166  if (offset > 0) {
167  result->setOffset(offset);
168  Core::Application::us()->log("offset feature space dimension of the mixture set to %d", offset);
169  }
170  ComponentIndex dim = paramReducedMixtureSetDimension(configuration);
171 
172  if (dim > 0) {
173  result->setDimension(dim);
174  Core::Application::us()->log("reduced dimension of mixture set to %d", dim);
175  }
176 
177  Core::XmlChannel logNormStatisticsChannel(configuration, "log-norm-factor-statistics");
178  if (logNormStatisticsChannel.isOpen())
179  result->writeLogNormStatistics(logNormStatisticsChannel);
180  }
181  return result;
182 }

273 bool Module_::writeMixtureSetEstimator(const std::string& filename, AbstractMixtureSetEstimator& estimator) {
274  require(!filename.empty());
275 
276  Core::Application::us()->log("Saving mixture set estimator to file \"%s\" ...", filename.c_str());
277 
278  std::string backupName = filename + ".bak";
279  if (Core::isRegularFile(filename) && rename(filename.c_str(), backupName.c_str()) != 0) {
280  Core::Application::us()->error("Could not create backup file \"%s\".", backupName.c_str());
281  return false;
282  }
283 
284  Core::BinaryOutputStream bos(filename);
285  if (!bos) {
286  Core::Application::us()->error("Failed to open \"%s\" for writing", filename.c_str());
287  return false;
288  }
289  estimator.write(bos);
290  if (!bos) {
291  Core::Application::us()->error("Failed to write mixture estimator");
292  return false;
293  }
294 
295  if (Core::isRegularFile(backupName) && remove(backupName.c_str()) != 0)
296  Core::Application::us()->error("Could not remove backup file \"%s\".", backupName.c_str());
297 
298  return true;
299 }
