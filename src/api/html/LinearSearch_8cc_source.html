32 LinearSearch::Pronunciation::Pronunciation(const Bliss::LemmaPronunciation* const lemma, const Am::AcousticModel* acousticModel)
33  : lemma_(lemma) {
34  const Bliss::Pronunciation* pron = lemma->pronunciation();
35  const Am::Phonology& phonology = *(acousticModel->phonology());
36  isRegularWord_ = isRegularWordPrivate();
37 
38  for (u32 a = 0; a < pron->length(); a++) {
39  const Am::Allophone* allo = 0;
40 
41  ///Get the allophone:
42  {
43  u16 bound = 0;
44  if (a == 0)
45  bound |= Am::Allophone::isInitialPhone;
46  if (a == pron->length() - 1)
47  bound |= Am::Allophone::isFinalPhone;
48 
49  Am::Allophone allophone(phonology(*pron, a), bound);
50 
51  verify(acousticModel->phonemeInventory()->phoneme(allophone.phoneme(0))->isContextDependent() || (allophone.history().size() == 0 && allophone.future().size() == 0));
52 
53  allo = acousticModel->allophoneAlphabet()->allophone(allophone);
54  }
55 
56  verify(allo);
57 
58  const Am::ClassicHmmTopology* hmmTopology = acousticModel->hmmTopology((*pron)[a]);
59  verify(hmmTopology != 0);
60  int nPhoneStates = hmmTopology->nPhoneStates();
61  int nReps = hmmTopology->nSubStates();
62  verify(nPhoneStates != 0);
63  verify(nReps != 0);
64 
65  bool isSilence = ((*pron)[a] == acousticModel->silence());
66 
67  verify(!isSilence || (nReps == 1 && nPhoneStates == 1));
68 
69  for (int a = 0; a < nPhoneStates; a++) {
70  for (int b = 0; b < nReps; b++) {
71  MixtureItem i;
72  Am::AllophoneState alloState = acousticModel->allophoneStateAlphabet()->allophoneState(allo, a);
73  i.mixture = acousticModel->emissionIndex(alloState);
74 
75  if (isSilence)
76  i.stateTransitionModel = acousticModel->stateTransition(Am::TransitionModel::silence);
77  else
78  i.stateTransitionModel = acousticModel->stateTransition(Am::TransitionModel::phone0 + b);
79 
80  mixtures_.push_back(i);
81  }
82  }
83  }
84 }

227 void LinearSearch::feed(const Mm::FeatureScorer::Scorer& emissionScores) {
228  require(emissionScores);
229  require(emissionScores->nEmissions() >= acousticModel_->nEmissions());
230 
231  ++time_;
232  WordPronunciationState::HypoVector hypTmp;
233 
234  //emissionScores represents the scorer for only one single mixture/state
235  u32 wordNumber = 0;
236 
237  for (WordPronunciationStateVector::iterator it = state_.begin(); it != state_.end(); ++it, ++wordNumber) {
238 #if defined SEARCH_DEBUG
239  std::cout << "------------------------------------------------------------" << std::endl
240  << " word " << (*it)->pronunciation()->lemma()->lemma()->preferredOrthographicForm()
241  << " at time frame " << time_ << std::endl
242  << "------------------------------------------------------------" << std::endl;
243 #endif
244  Book* last = 0;
245 
246  if (!book_.empty())
247  last = &book_.back();
248 
249  verify((*it)->pronunciation()->lemma()->lemma() != silence_ || !(*it)->pronunciation()->isRegularWord());
250 
251  if ((singleWordRecognition_ && last && last->hadRegularWord && (*it)->pronunciation()->isRegularWord()) || (*it)->irregularChain()) {
252  //Only irregular words are allowed before this one
253  if (irregularBook_.empty()) {
254  last = 0;
255  }
256  else {
257  last = &irregularBook_.back();
258  }
259  }
260 
261  WordPronunciationState::HypoVector& hyp = (*it)->hyp();
262  Pronunciation* pron = (*it)->pronunciation();
263 
264  hyp[0].bkp = last;
265 
266  if (last) {
267  Score localLmScore = 0; // CHANGED from f32
268  if (isUnigram()) {
269  localLmScore = (*it)->unigramScore();
270  }
271  else {
272  ///Bigram
273  Lm::History h(lm_->startHistory());
274  Lm::extendHistoryByLemmaPronunciation(lm_, last->word->pronunciation()->lemma(), h);
275  Lm::addLemmaPronunciationScore(lm_, pron->lemma(), pronunciationScale_, lm_->scale(), h, localLmScore);
276  }
277 
278  hyp[0].lmScore = localLmScore + last->lmScore;
279  hyp[0].score = last->score;
280  }
281  else {
282  f32 localLmScore = (*it)->unigramScore();
283 
284  hyp[0].lmScore = localLmScore;
285  hyp[0].score = 0;
286  }
287  hyp[0].score += hyp[0].lmScore; //The complete lm-score must be added, because last is of type Book, and in Book "lmScore" is not included in "score"
288 
289  hyp[0].mixture = 0;
290 #if defined SEARCH_DEBUG
291  std::cout << "OLD HYP FROM TIME FRAME " << time_ - 1 << std::endl;
292  for (u32 i = 0; i < hyp.size(); ++i) {
293  std::cout << "hyp[" << i << "] = " << hyp[i].score << std::endl;
294  }
295 #endif
296  hypTmp.resize(hyp.size());
297 #if defined SEARCH_DEBUG
298  std::vector<u32> buffer(hyp.size());
299 #endif
300  for (u32 sta = 1; sta < hyp.size(); ++sta) {
301  hypTmp[sta].score = Core::Type<Score>::max;
302  hypTmp[sta].lmScore = 0;
303 
304  u32 pre = sta >= 2 ? sta - 2 : 0;
305 
306  for (; pre <= sta; pre++) {
307  Score scoTmp = hyp[pre].score;
308 
309  const Am::StateTransitionModel* m;
310  if (pre != 0) {
311  verify(hyp[pre].mixture != 0);
312  m = hyp[pre].mixture->stateTransitionModel;
313  }
314  else {
315  m = acousticModel_->stateTransition(Am::TransitionModel::entryM1); //This is the first real state, so we use the StateTransitionModel entryM1
316  }
317 
318  scoTmp += (*m)[sta - pre];
319 
320  // std::cout << "state transition score from " << pre << " to state " << sta << " = " << (*m)[ sta - pre ] << std::endl;
321  if (scoTmp < hypTmp[sta].score) {
322  hypTmp[sta].score = scoTmp;
323  hypTmp[sta].bkp = hyp[pre].bkp;
324  hypTmp[sta].lmScore = hyp[pre].lmScore;
325 #if defined SEARCH_DEBUG
326  // DEBUG
327  buffer[sta] = sta - pre;
328 #endif
329  }
330  }
331  }
332 #if defined SEARCH_DEBUG
333  std::cout << "NEW HYP FOR TIME " << time_ << std::endl;
334 #endif
335  for (u32 sta = 1; sta < hyp.size(); sta++) {
336  hyp[sta].bkp = hypTmp[sta].bkp;
337  hyp[sta].score = hypTmp[sta].score + emissionScores->score(hyp[sta].mixture->mixture);
338  hyp[sta].lmScore = hypTmp[sta].lmScore;
339 #if defined SEARCH_DEBUG
340  //DEBUG
341  std::cout << "tmpHyp[" << sta << "] = " << hyp[sta].score << " | transition: " << buffer[sta] << " local score: "
342  << emissionScores->score(hyp[sta].mixture->mixture)
343  << " lmScore " << hyp[sta].lmScore
344  << std::endl;
345 #endif
346  }
347  }
348 
349  book_.push_back(Book());
350 
351  Book& newBook(book_.back());
352  newBook.score = Core::Type<Score>::max;
353 
354  bookKeeping(newBook);
355 
356  if (newBook.score == Core::Type<Score>::max)
357  book_.pop_back();
358 
359  if (singleWordRecognition_) {
360  //When single-word recognition is enabled it is also neccessary to keep track of irregular-book-keeping
361  irregularBook_.push_back(Book());
362 
363  Book& newBook(irregularBook_.back());
364  newBook.score = Core::Type<Score>::max;
365 
366  bookKeeping(newBook, true);
367 
368  if (newBook.score == Core::Type<Score>::max)
369  irregularBook_.pop_back();
370  else
371  verify(!irregularBook_.back().hadRegularWord);
372  }

375 void LinearSearch::bookKeeping(Book& newBook, bool irregular) {
376 #if defined SEARCH_DEBUG
377  // DEBUG
378  u32 wordNumber = 0;
379  u32 chosen = 0;
380  Score exit = 0;
381 #endif
382  for (WordPronunciationStateVector::iterator it = state_.begin(); it != state_.end(); ++it) //, ++wordNumber )
383  {
384  WordPronunciationStatePointer p = (*it);
385  verify(!p->hyp().empty());
386  WordPronunciationState::Hypo& hyp = p->hyp().back();
387  if (irregular && p->pronunciation()->isRegularWord()) {
388  continue; //Drop regular words when only irregular are wanted
389  }
390  if (irregular && hyp.bkp && hyp.bkp->hadRegularWord) {
391  continue; //Drop hypotheses whose histories contain regular words when only irregular are wanted
392  }
393 
394  Score tmpScore = hyp.score;
395  if (hyp.mixture)
396  tmpScore += (*hyp.mixture->stateTransitionModel)[Am::StateTransitionModel::exit];
397  else {
398  continue;
399  }
400 
401  if (tmpScore < newBook.score + newBook.lmScore) {
402  newBook.score = tmpScore - hyp.lmScore;
403  newBook.lmScore = hyp.lmScore;
404  newBook.bkp = hyp.bkp;
405  newBook.word = *it;
406 
407  if (newBook.word->pronunciation()->isRegularWord()) {
408  verify(!singleWordRecognition_ || (!newBook.bkp || newBook.bkp->hadRegularWord == false));
409  newBook.hadRegularWord = true;
410  }
411  else if (newBook.bkp) {
412  newBook.hadRegularWord = newBook.bkp->hadRegularWord;
413  }
414  else {
415  newBook.hadRegularWord = false;
416  }
417 
418  newBook.time = time_;
419 #if defined SEARCH_DEBUG
420  // DEBUG
421  chosen = wordNumber;
422  exit = (*hyp.mixture->stateTransitionModel)[Am::StateTransitionModel::exit];
423 #endif
424  }
425  }

432 void LinearSearch::getCurrentBestSentence(Traceback& result) const {
433  result.clear();
434 
435  if (book_.empty())
436  return;
437 
438  Lm::History h(lm_->startHistory());
439 
440  if (book_.back().bkp)
441  Lm::extendHistoryByLemmaPronunciation(lm_, book_.back().bkp->word->pronunciation()->lemma(), h);
442 
443  Lm::extendHistoryByLemmaPronunciation(lm_, book_.back().word->pronunciation()->lemma(), h);
444 
445  result.push_back(TracebackItem(0, time_, ScoreVector(book_.back().score, book_.back().lmScore + lm_->sentenceEndScore(h)), TracebackItem::Transit()));
446 
447  const Book* bkp = &book_.back();
448  int cnt = 0;
449 
450  while (bkp != 0) {
451  cnt++;
452  TracebackItem item(bkp->word->pronunciation()->lemma(), bkp->time, ScoreVector(bkp->score, bkp->lmScore), TracebackItem::Transit());
453  result.push_back(item);
454  bkp = bkp->bkp;
455  }
456  if (cnt != 0)
457  log("returning %i words", cnt);
458 
459  result.push_back(TracebackItem(0, 0, ScoreVector(0, 0), TracebackItem::Transit()));
460 
461  std::reverse(result.begin(), result.end());
462 }
