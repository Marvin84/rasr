35 Score ID3::entropy(ExamplePtrList::const_iterator begin, ExamplePtrList::const_iterator end) const {
36  Score entropy = 0.0;
37  u32 nObs = 0;
38  for (ExamplePtrList::const_iterator it = begin; it != end; ++it) {
39  const Example& example = **it;
40  CountMap::iterator itt = map_.find(example.values);
41  if (itt == map_.end())
42  map_.insert(CountMap::value_type(example.values, example.nObs));
43  else
44  itt->second += example.nObs;
45  nObs += example.nObs;
46  }
47  if (nObs == 0)
48  return 0.0;
49 
50  Score denominator = nObs;
51  for (CountMap::const_iterator it = map_.begin(); it != map_.end(); ++it) {
52  Score p = it->second;
53  if (p > 0.0) {
54  p /= denominator;
55  entropy -= p * ::log10(p);
56  }
57  }
58  entropy /= ::log10(2.0); // -> log2
59  map_.clear();
60  return entropy;
61 }

67 Score ID3::operator()(const ExamplePtrRange& leftExamples, const ExamplePtrRange& rightExamples,
68  const Score fatherEntropy,
69  Score& leftChildEntropy, Score& rightChildEntropy) const {
70  leftChildEntropy = entropy(leftExamples.begin, leftExamples.end);
71  rightChildEntropy = entropy(rightExamples.begin, rightExamples.end);
72  Score nObs = leftExamples.size() + rightExamples.size();
73  Score leftNObs = leftExamples.size();
74  Score rightNObs = rightExamples.size();
75  return fatherEntropy - (leftNObs / nObs) * leftChildEntropy - (rightNObs / nObs) * rightChildEntropy;
76 }

244  /**
245  * Keep the N-best hypotheses,
246  * use a simple bubble-sort to insert new hypothesis.
247  * For small N this will be very efficient.
248  */
249  bool push(const SplitHypothesis& hyp) {
250  if (nHyps_ == 0) {
251  nHyps_ = 1;
252  *hyps_.front() = hyp;
253  }
254  else if (hyps_.size() == 1) {
255  if (hyp.gain <= hyps_.front()->gain)
256  return false;
257  *hyps_.front() = hyp;
258  }
259  else {
260  SplitHypothesisPtrList::iterator itHyp = hyps_.begin() + (nHyps_ - 1);
261  if (nHyps_ == hyps_.size()) {
262  if (hyp.gain <= (*itHyp)->gain)
263  return false;
264  }
265  else {
266  ++nHyps_;
267  ++itHyp;
268  }
269  **itHyp = hyp;
270  for (SplitHypothesisPtrList::iterator itPrevHyp = itHyp - 1;
271  (itHyp != hyps_.begin()) && ((*itPrevHyp)->gain < (*itHyp)->gain);
272  --itPrevHyp, --itHyp)
273  std::swap(*itPrevHyp, *itHyp);
274  }
275  return true;
276  }

394  /*
395  * finds best split of a node, if any
396  */
397  Split* splitNode(Node* node, QuestionIdList* questionIds, bool strict = false) {
398  if (node->nObs < 2 * step_->minObs)
399  return 0;
400 
401  ExamplePtrList* examplePtrs = node->examplePtrs;
402 
403  ExamplePtrList leftExamplePtrs(examplePtrs->size(), 0);
404  ExamplePtrList rightExamplePtrs(examplePtrs->size(), 0);
405  const ExamplePtrList::iterator beginLeftExamplePtr = leftExamplePtrs.begin();
406  const ExamplePtrList::iterator beginRightExamplePtr = rightExamplePtrs.begin();
407  SplitHypothesis splitHyp;
408  verify(splitHyps_.empty());
409 
410  for (splitHyp.questionIdId = 0; splitHyp.questionIdId < questionIds->size(); ++splitHyp.questionIdId) {
411  const Question& question = *(questionRefs_[(*questionIds)[splitHyp.questionIdId]]);
412  ExamplePtrList::iterator endLeftExamplePtr = beginLeftExamplePtr;
413  ExamplePtrList::iterator endRightExamplePtr = beginRightExamplePtr;
414 
415  if (parallel_) {
416  f64 nLeftObs = 0;
417  f64 nRightObs = 0;
418 #pragma omp parallel reduction(+ \
419  : nLeftObs, nRightObs)
420  {
421  // some thread private variables
422  ExamplePtrList leftExamplePtrs_private((examplePtrs->size() + Core::omp::get_num_threads() - 1) / Core::omp::get_num_threads(), 0);
423  ExamplePtrList rightExamplePtrs_private((examplePtrs->size() + Core::omp::get_num_threads() - 1) / Core::omp::get_num_threads(), 0);
424  const ExamplePtrList::iterator beginLeftExamplePtr_private = leftExamplePtrs_private.begin();
425  const ExamplePtrList::iterator beginRightExamplePtr_private = rightExamplePtrs_private.begin();
426  ExamplePtrList::iterator endLeftExamplePtr_private = beginLeftExamplePtr_private;
427  ExamplePtrList::iterator endRightExamplePtr_private = beginRightExamplePtr_private;
428 
429 #pragma omp for nowait
430  for (size_t tmpCounter = 0; tmpCounter < examplePtrs->size(); ++tmpCounter) {
431  if (question(*((*examplePtrs)[tmpCounter]->properties)) == RASR_TRUE) {
432  *(endLeftExamplePtr_private++) = (*examplePtrs)[tmpCounter];
433  nLeftObs += (*examplePtrs)[tmpCounter]->nObs;
434  }
435  else {
436  *(endRightExamplePtr_private++) = (*examplePtrs)[tmpCounter];
437  nRightObs += (*examplePtrs)[tmpCounter]->nObs;
438  }
439  }
440 
441 #pragma omp critical // concatenate ptr vectors, order does not matter, beware: sum-up will be non-deterministic!
442  {
443  std::copy(beginLeftExamplePtr_private, endLeftExamplePtr_private, endLeftExamplePtr);
444  std::copy(beginRightExamplePtr_private, endRightExamplePtr_private, endRightExamplePtr);
445  std::advance(endLeftExamplePtr, std::distance(beginLeftExamplePtr_private, endLeftExamplePtr_private));
446  std::advance(endRightExamplePtr, std::distance(beginRightExamplePtr_private, endRightExamplePtr_private));
447  }
448  }
449  splitHyp.nLeftObs = nLeftObs; // see header f64->u32, is this really intended?
450  splitHyp.nRightObs = nRightObs;
451  }
452  else { // due to backward compatibility
453  splitHyp.nLeftObs = 0;
454  splitHyp.nRightObs = 0;
455  for (ExamplePtrList::const_iterator itExamplePtr = examplePtrs->begin(), endExamplePtr = examplePtrs->end();
456  itExamplePtr != endExamplePtr; ++itExamplePtr) {
457  if (question(*((*itExamplePtr)->properties)) == RASR_TRUE) {
458  *(endLeftExamplePtr++) = *itExamplePtr;
459  splitHyp.nLeftObs += (*itExamplePtr)->nObs;
460  }
461  else {
462  *(endRightExamplePtr++) = *itExamplePtr;
463  splitHyp.nRightObs += (*itExamplePtr)->nObs;
464  }
465  }
466  }
467 
468  splitHyp.nLeftExamples = u32(endLeftExamplePtr - beginLeftExamplePtr);
469  splitHyp.nRightExamples = u32(endRightExamplePtr - beginRightExamplePtr);
470 
471  verify(splitHyp.nLeftExamples + splitHyp.nRightExamples == node->examplePtrs->size());
472  if ((splitHyp.nLeftObs < step_->minObs) || (splitHyp.nRightObs < step_->minObs))
473  continue;
474  if (strict && ((splitHyp.nLeftObs == 0) || (splitHyp.nRightObs == 0)))
475  continue;
476  splitHyp.gain = scorer_(ExamplePtrRange(beginLeftExamplePtr, endLeftExamplePtr),
477  ExamplePtrRange(beginRightExamplePtr, endRightExamplePtr),
478  node->score,
479  splitHyp.leftScore, splitHyp.rightScore);
480  if (splitHyp.gain < 0.0) {
481  if (!Core::isAlmostEqualUlp(f32(splitHyp.gain), f32(0.0), 20))
482  error() << "negative split gain of " << splitHyp.gain << "; gain must be positive";
483  continue;
484  }
485  if (splitHyp.gain < step_->minGain)
486  continue;
487  if (strict && (splitHyp.gain == 0.0))
488  continue;
489  splitHyps_.push(splitHyp);
490  }
491  Split* split = 0;
492  if (!splitHyps_.empty()) {
493  const SplitHypothesis& bestSplitHyp = splitHyps_.get();
494  const Question& question = *(questionRefs_[(*questionIds)[bestSplitHyp.questionIdId]]);
495  ExamplePtrList* leftExamplePtrs = new ExamplePtrList(bestSplitHyp.nLeftExamples);
496  ExamplePtrList* rightExamplePtrs = new ExamplePtrList(bestSplitHyp.nRightExamples);
497  ExamplePtrList::iterator itLeftExamplePtr = leftExamplePtrs->begin();
498  ExamplePtrList::iterator itRightExamplePtr = rightExamplePtrs->begin();
499  for (ExamplePtrList::const_iterator itExamplePtr = node->examplePtrs->begin(), endExamplePtr = node->examplePtrs->end();
500  itExamplePtr != endExamplePtr; ++itExamplePtr) {
501  if (question(*((*itExamplePtr)->properties)) == RASR_TRUE)
502  *(itLeftExamplePtr++) = *itExamplePtr;
503  else
504  *(itRightExamplePtr++) = *itExamplePtr;
505  }
506  verify(itLeftExamplePtr == leftExamplePtrs->end());
507  verify(itRightExamplePtr == rightExamplePtrs->end());
508  delete node->examplePtrs;
509  node->examplePtrs = 0;
510  node->left = new Node(node->depth + 1,
511  bestSplitHyp.leftScore,
512  bestSplitHyp.nLeftObs,
513  leftExamplePtrs);
514  node->right = new Node(node->depth + 1,
515  bestSplitHyp.rightScore,
516  bestSplitHyp.nRightObs,
517  rightExamplePtrs);
518  split = new Split(node,
519  questionIds,
520  bestSplitHyp.questionIdId,
521  bestSplitHyp.gain);
522  }
523  splitHyps_.reset();
524  return split;
525  }

617  /**
618  * action: split
619  * find tuple of node and question causing best split,
620  * split using best split,
621  * repeat procedure until no splits left or max. leaf number is reached
622  */
623  void split() {
624  QuestionId questionId = questionRefs_.size();
625  QuestionIdList questionIds(step_->questionRefs->size());
626  questionRefs_.resize(questionRefs_.size() + step_->questionRefs->size());
627  questionStats_.resize(questionRefs_.size());
628  for (size_t i = 0; i < step_->questionRefs->size(); ++i, ++questionId) {
629  questionRefs_[questionId] = (*(step_->questionRefs))[i];
630  questionIds[i] = questionId;
631  }
632 
633  for (size_t i = nodes_.size(); i > 0; nodes_.pop_front(), --i)
634  suggestSplit(nodes_.front(), new QuestionIdList(questionIds));
635 
636  while (!splits_.empty() && (nLeaf_ + nodes_.size() + splits_.size() < plan_.maxLeaves)) {
637  Split* split = splits_.top();
638  splits_.pop();
639  commitSplit(split);
640  }
641  }

643  /**
644  * action: partition(->finalizeLeftNodes=false) or cluster(->finalizeLeftNodes=true)
645  * same as split, but put only right node back into queue for further splitting;
646  * if finalizeLeftNodes=false, then put left nodes back into queue for the next step,
647  * otherwise each left node will become a leaf
648  */
649  void halfSplit(bool finalizeLeftNodes = false) {
650  QuestionId questionId = questionRefs_.size();
651  QuestionIdList questionIds(step_->questionRefs->size());
652  questionRefs_.resize(questionRefs_.size() + step_->questionRefs->size());
653  questionStats_.resize(questionRefs_.size());
654  for (size_t i = 0; i < step_->questionRefs->size(); ++i, ++questionId) {
655  questionRefs_[questionId] = (*(step_->questionRefs))[i];
656  questionIds[i] = questionId;
657  }
658 
659  for (size_t i = nodes_.size(); i > 0; nodes_.pop_front(), --i)
660  suggestSplit(nodes_.front(), new QuestionIdList(questionIds));
661 
662  if (finalizeLeftNodes) {
663  while (!splits_.empty() && (nLeaf_ + nodes_.size() + splits_.size() < plan_.maxLeaves)) {
664  Split* split = splits_.top();
665  splits_.pop();
666  commitHalfSplit(split);
667  ++nLeaf_;
668  }
669  }
670  else {
671  while (!splits_.empty() && (nLeaf_ + nodes_.size() + splits_.size() < plan_.maxLeaves)) {
672  Split* split = splits_.top();
673  splits_.pop();
674  nodes_.push_back(commitHalfSplit(split));
675  }
676  }
677  }
678 
679  void start(Node* root) {
680  nodes_.push_back(root);
681  for (TrainingPlan::StepPtrList::const_iterator it = plan_.steps.begin();
682  (it != plan_.steps.end()) && (nLeaf_ + nodes_.size() < plan_.maxLeaves); ++it) {
683  step_ = *it;
684  step_->write(log());
685  splitHyps_.init(step_->nRandomQuestion);
686  if (step_->action == "split")
687  split();
688  else if (step_->action == "partition")
689  halfSplit(false);
690  else if (step_->action == "cluster")
691  halfSplit(true);
692  else
693  error("Unknown action \"%s\"", step_->action.c_str());
694 
695  while (!splits_.empty()) {
696  Split* split = splits_.top();
697  splits_.pop();
698  rollbackSplit(split);
699  }
700  writeXml(log());
701  }
702  nLeaf_ += nodes_.size();
703  nodes_.clear();
704  step_ = 0;
705  }
706 
707  DecisionTree::Node* commitNode(Node* node) {
708  verify(node->order != DecisionTreeTrainer::Information::InvalidOrder);
709  DecisionTree::Node* _node = new DecisionTree::Node(
710  0, 0, new DecisionTreeTrainer::Information(node->order, node->nObs, node->score));
711  if ((node->left) && (node->right)) {
712  _node->id_ = questionIdMap_[node->questionId];
713  _node->setChilds(
714  commitNode(node->left),
715  commitNode(node->right));
716  }
717  else {
718  verify(!(node->left) && !(node->right));
719  _node->id_ = ClassId(nCluster_++);
720  ConstExampleRefList* exampleRefs = new ConstExampleRefList();
721  exampleRefs->reserve(node->examplePtrs->size());
722  clusters_.push_back(new Cluster(_node, exampleRefs));
723  }
724  delete node->examplePtrs;
725  delete node;
726  return _node;
727  }
728 
729  ClusterList* finish(Node* root) {
730  tree_->setMap(plan_.map);
731 
732  // reduce list of questions to used ones
733  QuestionRefList* usedQuestionRefs = new QuestionRefList();
734  questionIdMap_.resize(questionRefs_.size(), DecisionTree::Node::InvalidId);
735  QuestionId questionId = 0;
736  QuestionStatisticList::iterator questionStatIt = questionStats_.begin();
737  QuestionRefList ::iterator questionRefIt = questionRefs_.begin();
738  QuestionIdList ::iterator questionIdMapIt = questionIdMap_.begin();
739  for (; questionStatIt != questionStats_.end();
740  ++questionStatIt, ++questionRefIt, ++questionIdMapIt) {
741  if (questionStatIt->count > 0) {
742  usedQuestionRefs->push_back(*questionRefIt);
743  *questionIdMapIt = questionId++;
744  }
745  }
746  tree_->setQuestions(usedQuestionRefs);
747 
748  // create decision tree
749  clusters_.reserve(nLeaf_);
750  nCluster_ = 0;
751  if (root->order == DecisionTreeTrainer::Information::InvalidOrder)
752  root->order = 0;
753  tree_->setRoot(commitNode(root));
754  verify(nLeaf_ == nCluster_);
755 
756  // create cluster
757  for (ExampleList::const_iterator it = examples_.begin();
758  it != examples_.end(); ++it)
759  clusters_[tree_->classify(*((*it)->properties))]->exampleRefs->push_back(*it);
760  ClusterList* clusters = new ClusterList(config, plan_.map);
761  for (ClusterPtrList::const_iterator it = clusters_.begin();
762  it != clusters_.end(); ++it)
763  clusters->add(*it);
764 
765  clusters_.clear();
766  return clusters;
767  }
