51 bool DiscriminativeMixtureSetEstimator::distributePreviousMixtureSet(const MixtureSet& previousMixtureSet) {
52  if (dimension() != previousMixtureSet.dimension()) {
53  return false;
54  }
55  MixtureSetEstimatorIndexMap indexMaps(*this);
56  if (previousMixtureSet.nMixtures() != nMixtures()) {
57  return false;
58  }
59  for (MixtureIndex i = 0; i < previousMixtureSet.nMixtures(); ++i) {
60  mixtureEstimator(i).setPreviousMixture(previousMixtureSet.mixture(i));
61  }
62  if (previousMixtureSet.nDensities() != indexMaps.densityMap().size()) {
63  return false;
64  }
65  for (DensityIndex i = 0; i < previousMixtureSet.nDensities(); ++i) {
66  required_cast(DiscriminativeGaussDensityEstimator*, indexMaps.densityMap()[i].get())
67  ->setPreviousDensity(previousMixtureSet.density(i));
68  }
69  if (previousMixtureSet.nMeans() != indexMaps.meanMap().size()) {
70  return false;
71  }
72  for (MeanIndex i = 0; i < previousMixtureSet.nMeans(); ++i) {
73  required_cast(DiscriminativeMeanEstimator*, indexMaps.meanMap()[i].get())
74  ->setPreviousMean(previousMixtureSet.mean(i));
75  }
76  if (previousMixtureSet.nCovariances() != indexMaps.covarianceMap().size()) {
77  return false;
78  }
79  CovarianceToMeanSetMap meanSetMap(indexMaps.densityMap().indexToPointerMap());
80  for (CovarianceIndex i = 0; i < previousMixtureSet.nCovariances(); ++i) {
81  required_cast(DiscriminativeCovarianceEstimator*, indexMaps.covarianceMap()[i].get())
82  ->setPreviousCovariance(previousMixtureSet.covariance(i));
83  }
84  log("distribute previous mixture set ...done");
85  return true;
86 }

136 void DiscriminativeMixtureSetEstimator::estimate(MixtureSet& toEstimate) {
137  load();
138  if (minObservationWeight() != 0 or minRelativeWeight() != 0) {
139  removeDensitiesWithLowWeight(minObservationWeight(), minRelativeWeight());
140  }
141  toEstimate.clear();
142  toEstimate.setDimension(dimension_);
143  MixtureSetEstimatorIndexMap indexMaps(*this);
144  CovarianceToMeanSetMap meanSetMap(indexMaps.densityMap().indexToPointerMap());
145  initialize(indexMaps, meanSetMap);
146  for (MixtureIndex i = 0; i < mixtureEstimators_.size(); ++i) {
147  toEstimate.addMixture(i, mixtureEstimators_[i]->estimate(indexMaps.densityMap(), normalizeMixtureWeights_));
148  }
149  for (DensityIndex i = 0; i < indexMaps.densityMap().size(); ++i) {
150  toEstimate.addDensity(i, indexMaps.densityMap()[i]->estimate(
151  indexMaps.meanMap(), indexMaps.covarianceMap()));
152  }
153  for (MeanIndex i = 0; i < indexMaps.meanMap().size(); ++i) {
154  toEstimate.addMean(i, indexMaps.meanMap()[i]->estimate());
155  }
156  for (CovarianceIndex i = 0; i < indexMaps.covarianceMap().size(); ++i) {
157  toEstimate.addCovariance(i, indexMaps.covarianceMap()[i]->estimate(meanSetMap, minVariance_));
158  }
159  finalize(toEstimate, indexMaps, meanSetMap);
160 }

187 void DiscriminativeMixtureSetEstimator::accumulateDenominator(
188  MixtureIndex mixtureIndex, Core::Ref<const Feature::Vector> featureVector, Weight weight) {
189  if (viterbi_) {
190  DensityIndex index = densityIndex(mixtureIndex, featureVector);
191  mixtureEstimator(mixtureIndex).accumulateDenominator(index, *featureVector, weight);
192  }
193  else {
194  std::vector<Mm::Weight> dnsPosteriors;
195  getDensityPosteriorProbabilities(mixtureIndex, featureVector, dnsPosteriors);
196  for (DensityIndex index = 0; index < dnsPosteriors.size(); ++index) {
197  Weight finalWeight = weight * dnsPosteriors[index];
198  if (finalWeight > weightThreshold_) {
199  mixtureEstimator(mixtureIndex).accumulateDenominator(index, *featureVector, finalWeight);
200  }
201  }
202  }
203 }
