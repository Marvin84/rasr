58 void SpectralIntegration::apply(const Flow::Vector<Sample>& in, Flow::Vector<Sample>& out) {
59  windowFunction_->setLength(length_);
60  /*** calculate the number of channels of the output signal ***/
61  u32 channels = (in[0].size() - length_) / shift_ + 1;
62 
63  out.resize(in.size());
64  out.setTimestamp(in);
65  for (u32 i = 0; i < in.size(); i++) {
66  out[i].resize(channels);
67 
68  for (u32 ch = 0; ch < channels; ch++) {
69  for (u32 w = 0; w < length_; w++) {
70  verify(ch * shift_ + w < in[0].size());
71  out[i][ch] += (windowFunction_->getWindow())[w] * in[i][ch * shift_ + w];
72  }
73  }
74  }
75 }
