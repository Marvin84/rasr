317 void ContextTransducerBuilder::Builder::initialize() {
318  size_t maxHistory = 0, maxFuture = 0;
319  for (Am::AllophoneAlphabet::AllophoneList::const_iterator ai = allophoneList_->begin();
320  ai != allophoneList_->end(); ++ai) {
321  const Allophone* a = *ai;
322  maxHistory = std::max(maxHistory, a->history().size());
323  maxFuture = std::max(maxFuture, a->future().size());
324  const Bliss::Phoneme* phoneme = model_->phonemeInventory()->phoneme(a->phoneme());
325  if (!phoneme->isContextDependent()) {
326  if ((*ai)->boundary & (Allophone::isInitialPhone | Allophone::isFinalPhone))
327  ciPhones_[phoneme->id()] = a;
328  else
329  contextIndependentInnerPhones_.push_back(phoneme->id());
330  }
331  if (a->boundary & Allophone::isInitialPhone) {
332  if (a->history().size() == 0)
333  initialNonCoartPhones_.push_back(phoneme->id());
334  else
335  initialCoartPhones_.push_back(phoneme->id());
336  }
337  if (a->history().size() == 0 && !(a->boundary & Allophone::isInitialPhone))
338  innerNonCoartPhones_.push_back(phoneme->id());
339  }
340  removeDuplicates(initialCoartPhones_);
341  removeDuplicates(initialNonCoartPhones_);
342  removeDuplicates(innerNonCoartPhones_);
343  removeDuplicates(contextIndependentInnerPhones_);
344 
345  if (nonWordTokens_) {
346  nonWordTokens_->createAllophones(allophones_);
347  const NonWordTokens::AllophoneMap& nonWordAllophones = nonWordTokens_->allophones();
348  for (NonWordTokens::AllophoneMap::const_iterator nw = nonWordAllophones.begin();
349  nw != nonWordAllophones.end(); ++nw) {
350  verify(!phonemes_->phonemeInventory()->isValidPhonemeId(nw->first));
351  ciPhones_[nw->first] = nw->second;
352  initialNonCoartPhones_.push_back(nw->first);
353  }
354  }
355 
356  /* require triphone context */
357  require(maxHistory <= 1);
358  require(maxFuture <= 1);
359 }

361 OpenFst::VectorFst* ContextTransducerBuilder::Builder::build() {
362  verify(phoneSymbols_);
363  verify(allophoneSymbols_);
364  stateMap_.clear();
365  c_ = new OpenFst::VectorFst();
366  prepare();
367  for (AllophoneAlphabet::AllophoneList::const_iterator ai = allophoneList_->begin();
368  ai != allophoneList_->end(); ++ai) {
369  buildAllophone(*ai);
370  }
371  if (nonWordTokens_) {
372  const NonWordTokens::AllophoneMap& nonWordAllophones = nonWordTokens_->allophones();
373  for (NonWordTokens::AllophoneMap::const_iterator ai = nonWordAllophones.begin();
374  ai != nonWordAllophones.end(); ++ai) {
375  buildAllophone(ai->second);
376  }
377  }
378  finalize();
379  c_->SetInputSymbols(allophoneSymbols_);
380  c_->SetOutputSymbols(phoneSymbols_);
381  FstLib::Connect(c_);
382  return c_;
383 }

402 /**
403  * builds a deterministic context dependency transducer for triphones.
404  *
405  * the states store information about the two previously read phones
406  * and if a final allophone was the output of the incoming arc
407  *
408  * the first phone P of a word in the lexicon transducer is assumed to have
409  * a special index: index(P) + initialPhoneOffset_
410  *
411  * in general a transition looks like this:
412  * (A,B) -- C : B{A+C} --> (B,C)
413  *
414  * allophones with the final tag:
415  * (A,B) -- C! : B{A+C}@f --> (B,C,boundary=true)
416  * where C! is the word start phone C (see above)
417  *
418  * allophones with the initial tag:
419  * (B,C,boundary=true) -- D : C{B+D}@i --> (C,D)
420  *
421  * in the resulting transducer, the allophones are on the input and
422  * phones on the output (inverse of the above)
423  *
424  * the resulting transducer is deterministic on the output side (phones)
425  * if allowNonCrossWordTransitions == false
426  *
427  * phone disambiguators are converted to allophone disambiguators using
428  * loop transitions on all states with an incoming final allophone
429  *
430  * if sequenceEndSymbol_ == eps, a epsilon transition is created from
431  * every boundary state to a final state
432  *
433  * if sequenceEndSymbol_ != eps, the state (#,sequenceEndSymbol) is final.
434  * thereby the last phone symbol does not produce an allophone symbol.
435  *
436  * if exploitDisambiguators_ == true, an unshifted loop is created on
437  * the initial state for all CI phones. The disambiguation symbols occuring
438  * directly after a CI phone is deleted.
439  */
440 class ContextTransducerBuilder::AcrossWordBuilder : public ContextTransducerBuilder::Builder {
441 public:
442  AcrossWordBuilder(Core::Ref<const Am::AcousticModel> m, Core::Ref<const Bliss::Lexicon> l)
443  : Builder(m, l),
444  allowNonCrossWord_(false),
445  addSuperFinal_(false),
446  exploitDisambiguators_(false),
447  unshiftCiPhones_(false),
448  nonPhoneSequenceEnd_(false),
449  finalCiLoop_(false) {}
450 
451 protected:
452  virtual void prepare() {
453  disambiguatorStates_.clear();
454  stateMap_.clear();
455  iInitial_ = getStateId(Bliss::Phoneme::term, Bliss::Phoneme::term, 0, true);
456  c_->SetStart(iInitial_);
457  c_->SetFinal(iInitial_, OpenFst::Weight::One());
458  disambiguatorStates_.insert(iInitial_);
459  OpenFst::Label sequenceEnd = getSequenceEndSymbol();
460  if (sequenceEnd != OpenFst::Epsilon)
461  nonPhoneSequenceEnd_ = !model_->phonemeInventory()->isValidPhonemeId(sequenceEnd);
462  if (sequenceEnd == OpenFst::Epsilon || addSuperFinal_ || nonPhoneSequenceEnd_) {
463  iFinal_ = getStateId(Bliss::Phoneme::term, Bliss::Phoneme::term, 1, true);
464  c_->SetFinal(iFinal_, OpenFst::Weight::One());
465  }
466  else {
467  iFinal_ = OpenFst::InvalidStateId;
468  }
469 
470  OpenFst::StateId iCiState = OpenFst::InvalidStateId;
471  if (exploitDisambiguators_) {
472  iCiState = getStateId(Bliss::Phoneme::term, Bliss::Phoneme::term, 2, true);
473  addDisambiguatorArcs(iCiState, iInitial_);
474  }
475  if (nonPhoneSequenceEnd_ && (exploitDisambiguators_ || unshiftCiPhones_)) {
476  addArc(iInitial_, iFinal_, OpenFst::Epsilon, sequenceEnd);
477  }
478 
479  // for all initial non-coarticulated allophones A{#+X} build transition
480  // (#,#) -- EPS : A --> (#,A)
481  // if exploitDisambiguators_ build for all CI phones P
482  // (#,#) -- P{#,#} : P --> (CI)
483  // if unshiftCiPhones_ build for all CI phones P
484  // (#,#) -- P{#,#} : P --> (#,#)
485  for (PhoneList::const_iterator i = initialNonCoartPhones_.begin();
486  i != initialNonCoartPhones_.end(); ++i) {
487  PhoneMap::const_iterator ci = ciPhones_.find(*i);
488  const bool isCiPhone = ci != ciPhones_.end();
489  if (isCiPhone) {
490  DBG(1) << "initial CI allophone=" << allophones_->symbol(allophones_->index(ci->second)) << " phone=" << phonemes_->symbol(ci->first) << std::endl;
491  }
492  else {
493  DBG(1) << "initial phone: " << phonemes_->symbol(*i) << std::endl;
494  }
495  if (isCiPhone) {
496  if (exploitDisambiguators_) {
497  DBG(1) << "arc to CI-state" << std::endl;
498  addArc(iInitial_, iCiState, ci->second, ci->first, true);
499  }
500  else if (unshiftCiPhones_) {
501  DBG(1) << "CI loop" << std::endl;
502  addArc(iInitial_, iInitial_, ci->second, ci->first, true);
503  }
504  else {
505  DBG(1) << "CI initial" << std::endl;
506  addArc(iInitial_, getStateId(Bliss::Phoneme::term, *i, 0, true), 0, *i, true);
507  }
508  if (iFinal_ != OpenFst::InvalidStateId && finalCiLoop_) {
509  addArc(iFinal_, iFinal_, ci->second, ci->first, true);
510  }
511  }
512  else {
513  DBG(1) << "initial arc" << std::endl;
514  OpenFst::StateId iTo = getStateId(Bliss::Phoneme::term, *i, 0, true);
515  addArc(iInitial_, iTo, 0, *i, true);
516  }
517  }
518  }

520 private:
521  /**
522  * Build transitions for a final allophone with empty right context A{B,#}@f
523  */
524  void buildFinalRightCiAllophone(OpenFst::StateId iFrom, const Allophone* a, bool ciPhone) {
525  DBG(1) << "final right-ci" << std::endl;
526  if (!ciPhone) {
527  DBG(1) << "cd phone" << std::endl;
528  if (allowNonCrossWord_) {
529  // for all initial non-coarticulated phones X
530  // (A,B) -- X : B{A+#} --> (#,X)
531  for (PhoneList::const_iterator p = initialNonCoartPhones_.begin();
532  p != initialNonCoartPhones_.end(); ++p) {
533  OpenFst::StateId iTo = getStateId(Bliss::Phoneme::term, *p, 0, true);
534  addArc(iFrom, iTo, a, *p, true);
535  }
536  }
537  else {
538  // for all non-coarticulated phones X
539  // (A,B) -- X : B{A+#} --> (#,X)
540  DBG(1) << "for all ci phones" << std::endl;
541  for (PhoneMap::const_iterator p = ciPhones_.begin(); p != ciPhones_.end(); ++p) {
542  OpenFst::StateId iTo = getStateId(Bliss::Phoneme::term, p->first, 0, true);
543  addArc(iFrom, iTo, a, p->first, true);
544  }
545  }
546  }
547  else {
548  DBG(1) << "ci phone" << std::endl;
549  // non-coarticulated phone
550  verify(a->history().size() == 0);
551 
552  if (exploitDisambiguators_) {
553  // (#,A) -- #_ : A{#,#} --> (#,#)
554  addDisambiguatorArcs(iFrom, iInitial_, a);
555  }
556  else if (unshiftCiPhones_) {
557  // (#,A) -- eps : A{#,#} --> (#,#)
558  DBG(1) << "unshifting arc" << std::endl;
559  addArc(iFrom, iInitial_, a, Bliss::Phoneme::Id(-1), false);
560  }
561  else {
562  // (#,A) -- B : A{#,#} --> (#,B)
563  for (PhoneList::const_iterator p = initialNonCoartPhones_.begin();
564  p != initialNonCoartPhones_.end(); ++p) {
565  OpenFst::StateId iTo = getStateId(Bliss::Phoneme::term, *p, 0, true);
566  addArc(iFrom, iTo, a, *p, true);
567  }
568  }
569  }
570  if (getSequenceEndSymbol() == OpenFst::Epsilon) {
571  // (A,B) -- EPS : B{A+#} --> ((final))
572  addArc(iFrom, iFinal_, a, Bliss::Phoneme::Id(-1), false);
573  }
574  else if (nonPhoneSequenceEnd_ && !(unshiftCiPhones_ && ciPhone)) {
575  // (A,B) -- $ : B{A+#} --> ((final))
576  DBG(1) << "arc to final" << std::endl;
577  addArc(iFrom, iFinal_, getAllophoneLabel(a), getSequenceEndSymbol());
578  }
579  }

581  /**
582  * Build transitions for a non-final allophone with empty right context A{B,#}
583  */
584  void buildRightCiAllophone(OpenFst::StateId iFrom, const Allophone* a, bool ciPhone) {
585  if (!ciPhone) {
586  // empty future, but not a final allophone
587  // -> must be a phone before a CI phone inside a word
588  // add transitions for all CI phones X
589  // (A,B) --> X : B{A,#} --> (#,X)
590  for (PhoneList::const_iterator p = contextIndependentInnerPhones_.begin();
591  p != contextIndependentInnerPhones_.end(); ++p) {
592  OpenFst::StateId iTo = getStateId(Bliss::Phoneme::term, *p, 0, false);
593  addArc(iFrom, iTo, a, *p, false);
594  }
595  }
596  else {
597  // context-independent phone without the final tag.
598  // add transitions for intra-word non-coarticulated phones X
599  // (#,P) -- X : P{#,#} --> (#,X)
600  for (PhoneList::const_iterator p = innerNonCoartPhones_.begin();
601  p != innerNonCoartPhones_.end(); ++p) {
602  OpenFst::StateId iTo = getStateId(Bliss::Phoneme::term, *p, 0, false);
603  addArc(iFrom, iTo, a, *p, false);
604  }
605  }
606  }
607 
608 protected:
609  virtual void buildAllophone(const Allophone* a) {
610  DBG(1) << allophones_->symbol(allophones_->index(a)) << std::endl;
611  Bliss::Phoneme::Id history =
612  (a->history().size() ? a->history()[0] : Bliss::Phoneme::term);
613  Bliss::Phoneme::Id central = a->central();
614  Bliss::Phoneme::Id future =
615  (a->future().size() ? a->future()[0] : Bliss::Phoneme::term);
616  const bool ciPhone = isCiPhone(central);
617  const bool finalPhone = a->boundary & Allophone::isFinalPhone;
618  OpenFst::StateId iFrom = getStateId(history, central, 0,
619  a->boundary & Allophone::isInitialPhone);
620 
621  if (finalPhone && !(exploitDisambiguators_ && ciPhone))
622  disambiguatorStates_.insert(iFrom);
623  if (future == Bliss::Phoneme::term) {
624  if (finalPhone) {
625  // empty future and final allophone
626  // -> last phone of a word without across-word context
627  buildFinalRightCiAllophone(iFrom, a, ciPhone);
628  }
629  else {
630  // empty future but not final
631  // phone inside a word having context independent inner phones
632  // (for example silence inside a phrase)
633  buildRightCiAllophone(iFrom, a, ciPhone);
634  }
635  }
636  else {
637  // (A,B) -- C : B{A,C} --> (B,C)
638  OpenFst::StateId iTo = getStateId(central, future, 0,
639  finalPhone);
640  addArc(iFrom, iTo, a, future, finalPhone);
641  }
642  }

644  virtual void finalize() {
645  // create disambiguator loops
646  for (std::set<OpenFst::StateId>::const_iterator s = disambiguatorStates_.begin();
647  s != disambiguatorStates_.end(); ++s) {
648  addDisambiguatorArcs(*s, *s);
649  }
650  OpenFst::Label seqEnd = getSequenceEndSymbol();
651  if (seqEnd != OpenFst::Epsilon && !nonPhoneSequenceEnd_) {
652  // create final state on (#,si),
653  // i.e. use last silence symbol as sequence end symbol
654  // verify(isCiPhone(sequenceEndSymbol_));
655  TriphoneContextAndBoundary siState(Bliss::Phoneme::term, seqEnd, 0, true);
656  StateSet::const_iterator si = stateMap_.find(siState);
657  verify(si != stateMap_.end());
658  if (addSuperFinal_) {
659  verify(iFinal_ != Fsa::InvalidStateId);
660  const Allophone* a = allophones_->allophone(
661  Allophone(seqEnd, Allophone::isInitialPhone | Allophone::isFinalPhone));
662  addArc(si->second, iFinal_, a, Bliss::Phoneme::Id(-1), false);
663  }
664  else {
665  c_->SetFinal(si->second, OpenFst::Weight::One());
666  }
667  }

698 // builds a non-deterministic C transducer
699 // transitions:
700 // (A,B) -- B{A+C} : A --> (B,C)
701 // (A,B) -- B{A+C}@f : B --> (B,C,@i)
702 // (A,B,@i) -- B{A+C}@i : B@i --> (B,C)
703 // (#,#) -- B{#+C} : B --> (B,C)
704 // (A,B) -- B{A+#} : B --> (#,#,1)
705 // (#,#,1) -- CI{#+#} : CI --> (#,#,0)
706 // to enforce at least one CI phone between allophones A{B+#} C{#+D}
707 class ContextTransducerBuilder::NonDeterministicBuilder : public ContextTransducerBuilder::Builder {
708 public:
709  NonDeterministicBuilder(Core::Ref<const Am::AcousticModel> m, Core::Ref<const Bliss::Lexicon> l)
710  : Builder(m, l), iInitial_(OpenFst::InvalidStateId) {}
711 
712  virtual void prepare() {
713  stateMap_.clear();
714  iInitial_ = getStateId(Bliss::Phoneme::term, Bliss::Phoneme::term, 0, true);
715  c_->SetStart(iInitial_);
716  c_->SetFinal(iInitial_, OpenFst::Weight::One());
717  disambiguatorStates_.insert(iInitial_);
718  }
719 
720  virtual void buildAllophone(const Allophone* a) {
721  DBG(1) << allophones_->symbol(allophones_->index(a)) << std::endl;
722  Bliss::Phoneme::Id history =
723  (a->history().size() ? a->history()[0] : Bliss::Phoneme::term);
724  Bliss::Phoneme::Id center = a->central();
725  Bliss::Phoneme::Id future =
726  (a->future().size() ? a->future()[0] : Bliss::Phoneme::term);
727  const bool ciPhone = isCiPhone(center);
728  const bool finalPhone = a->boundary & Allophone::isFinalPhone;
729  const bool initialPhone = a->boundary & Allophone::isInitialPhone;
730  OpenFst::StateId iFrom = OpenFst::InvalidStateId, iTo = OpenFst::InvalidStateId;
731 
732  if (ciPhone) {
733  // for initial CI phones:
734  // (#,#,@i,1) -- si{#+#}@i : si@i --> (#,#,@i,0)
735  // (#,#,@i,0) -- si{#+#}@i : si@i --> (#,#,@i,0)
736  // for in-word CI phones:
737  // (#,#,1) -- si{#+#} : si --> (#,#,0)
738  // (#,#,0) -- si{#+#} : si --> (#,#,0)
739  iFrom = getStateId(Bliss::Phoneme::term, Bliss::Phoneme::term, 1, initialPhone);
740  iTo = getStateId(Bliss::Phoneme::term, Bliss::Phoneme::term, 0, initialPhone);
741  addArc(iFrom, iTo, a, center, initialPhone);
742  addArc(iTo, iTo, a, center, initialPhone);
743  }
744  else {
745  if (history == Bliss::Phoneme::term)
746  iFrom = getStateId(Bliss::Phoneme::term, Bliss::Phoneme::term, 0, initialPhone);
747  else
748  iFrom = getStateId(history, center, 0, initialPhone);
749 
750  if (future == Bliss::Phoneme::term)
751  iTo = getStateId(Bliss::Phoneme::term, Bliss::Phoneme::term, 1, finalPhone);
752  else
753  iTo = getStateId(center, future, 0, finalPhone);
754  addArc(iFrom, iTo, a, center, initialPhone);
755  }
756  if (finalPhone)
757  disambiguatorStates_.insert(iTo);
758  }
759 
760  virtual void finalize() {
761  // disambiguator loops at word boundary states
762  for (std::set<OpenFst::StateId>::const_iterator s = disambiguatorStates_.begin();
763  s != disambiguatorStates_.end(); ++s) {
764  addDisambiguatorArcs(*s, *s);
765  }
766  OpenFst::Label seqEnd = getSequenceEndSymbol();
767  // all boundary CI-states (#,#,@i) are final
768  for (StateSet::const_iterator s = stateMap_.begin(); s != stateMap_.end(); ++s) {
769  const TriphoneContextAndBoundary& tcb = s->first;
770  if (tcb.boundary_ && tcb.central_ == Bliss::Phoneme::term) {
771  // no "read at least one CI phone state"
772  c_->SetFinal(s->second, OpenFst::Weight::One());
773  if (seqEnd != OpenFst::Epsilon) {
774  addArc(s->second, s->second, OpenFst::Epsilon, seqEnd);
775  }
776  }
777  }
778  }
779 
780 private:
781  OpenFst::StateId iInitial_;
782  std::set<OpenFst::StateId> disambiguatorStates_;
783 };
