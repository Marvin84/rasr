60 void LexicalFstFactory::logOptions(const Options& options) const {
61  if (options.lookAhead & LabelLookAheadFlag)
62  log("using label look-ahead");
63  if (options.lookAhead & PushWeightsFlag) {
64  log("using weight pushing");
65  log("weight look-ahead accumulator: ")
66  << choiceAccumulatorType_[options.accumulatorType];
67  }
68  if (options.lookAhead & PushLabelsFlag)
69  log("using label pushing");
70  if (options.lookAhead & ArcLookAheadFlag)
71  log("using arc look-ahead");
72  if (options.lookAhead == NoLookAhead)
73  log("no look-ahead");

110 AbstractLexicalFst* LexicalFstFactory::read(const std::string& filename, const Options& options,
111  bool isMatcherFst, f32 scale, AbstractGrammarFst* g) const {
112  logOptions(options);
113  AbstractLexicalFst* l = create(options);
114  ensure(l);
115  if (isMatcherFst ||
116  !(options.lookAhead & (LabelLookAheadFlag | ArcLookAheadFlag))) {
117  log("assuming required fst type");
118  if (!l->load(filename))
119  criticalError("cannot load %s", filename.c_str());
120  }
121  else {
122  log("creating required fst type");
123  OpenFst::VectorFst* i = OpenFst::VectorFst::Read(filename);
124  convert(i, scale, l, g);
125  delete i;
126  }
127  return l;
128 }

132  verify(result);

134  log("re-scaling weights of L: %f", scale);

137  result->create(*base);

144 AbstractLexicalFst* LexicalFstFactory::create(const Options& options) {
145  switch (options.lookAhead) {
146  case ArcLookAhead:
147  return new ArcLookAheadFst();
148  break;
149  case PushLabels:
150  return createFst<PushLabelsLexicalFst>(options.accumulatorType);
151  break;
152  case PushWeights:
153  return createFst<PushWeightsLexicalFst>(options.accumulatorType);
154  break;
155  case PushLabelsOnly:
156  return new PushLabelsOnlyLexicalFst();
157  case LabelLookAhead:
158  return new LookAheadLexicalFst();
159  break;
160  case NoLookAhead:
161  return new StandardLexicalFst();
162  break;
163  default:
164  defect();
165  return 0;
166  break;
167  }
168 }
