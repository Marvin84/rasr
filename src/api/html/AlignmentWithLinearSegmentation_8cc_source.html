99 f64 LinearSegmenter::Delimiter::logLikelihood(u32 ib, u32 ie) const {
100  const f64 e = 0.5;
101  f64 yy = Core::Type<f64>::max;
102  f64 s1 = ib - 1 + nFeatures() - ie;
103  f64 m1 = M_[ib - 1] + M_[nFeatures()] - M_[ie];
104  f64 qq = Q_[ib - 1] + Q_[nFeatures()] - Q_[ie];
105  f64 xx = qq / s1 - ((m1 / s1) * (m1 / s1)) / (f_ > 0 ? f_ : 1);
106  f64 xsil = s1 * ::log(std::max(xx, e));
107  f64 s2 = ie - ib + 1;
108  f64 m2 = M_[ie] - M_[ib - 1];
109  qq = Q_[ie] - Q_[ib - 1];
110  xx = qq / s2 - ((m2 / s2) * (m2 / s2)) / (f_ > 0 ? f_ : 1);
111  f64 xspe = s2 * ::log(std::max(xx, e));
112  if (((m1 / s1) < (m2 / s2)) &&
113  ((ie - ib) >= (minimumSpeechProportion_ * nFeatures())))
114  yy = xsil + xspe;
115  return yy;
116 }

118 /*****************************************************************************/
119 /*
120  * taken from old standard system (aka delimit(*)):
121  *
122  * START/STOP-DETECTION USING TWO GAUSSIAN MODELS,
123  * ONE FOR SPEECH AND ONE FOR SILENCE;
124  * BRIDLE,SEDGWICK, PP.656-659, ICASSP'77.
125  *
126  * ITERATIVE APPROXIMATION BY VARYING IB AND IE SEPARATELY
127  *
128  * N
129  * F = N*log(var) = N * log( 1/N * SUM [x(n)-xx]**2 )
130  * n=1
131  *
132  * SUM [x(n)-xx]**2 = SUM [x(n)**2 - 2*xx*x(n) + xx**2]
133  * = SUM x(n)**2 - 2*xx*SUM x(n) + SUM xx**2
134  * = SUM x(n)**2 - 2*xx*N*xx + N *xx**2
135  * = SUM x(n)**2 - N*(xx**2)
136  *
137  * silence speech silence
138  * [1,...,IB-1] [IB,...,IE] [IE+1,...,NI]
139  */
140 /*****************************************************************************/
141 LinearSegmenter::Delimiter::Delimitation LinearSegmenter::Delimiter::getDelimitation() const {
142  if (minimumSpeechProportion_ == 1.0)
143  return std::make_pair(0, nFeatures() - 1);
144 
145  if (nFeatures() == 1)
146  return std::make_pair(0, 0);
147 
148  f64 x;
149  f64 x_min = Core::Type<f64>::max;
150  u32 ie, ib, iEnd = nFeatures() - 1, iBeg = 1;
151  for (u32 i = 0; i < numberOfIterations_; ++i) {
152  for (ib = 2; ib < iEnd - 1; ++ib) {
153  x = logLikelihood(ib, iEnd);
154  if (x < x_min) {
155  x_min = x;
156  iBeg = ib;
157  }
158  }
159  for (ie = iBeg + 1; ie < nFeatures() - 1; ie++) {
160  x = logLikelihood(iBeg, ie);
161  if (x < x_min) {
162  x_min = x;
163  iEnd = ie;
164  }
165  }
166  }
167 
168  if (statisticsChannel_.isOpen()) {
169  statisticsChannel_ << Core::XmlOpen("delimiter-statistics")
170  << Core::XmlFull("frames", nFeatures())
171  << Core::XmlFull("speech-begin", iBeg)
172  << Core::XmlFull("speech-end", iEnd)
173  << Core::XmlFull("score", x_min)
174  << Core::XmlClose("delimiter-statistics");
175  }
176 
177  return std::make_pair(iBeg - 1, iEnd - 1);
178 }

311 bool LinearSegmenter::getAlignment(Alignment& alignment) const {
312  if (nFeatures() < minimumSegmentLength_)
313  return false;
314  if (nFeatures() > maximumSegmentLength_)
315  return false;
316  TimeframeIndex time;
317  Delimiter::Delimitation delimitation = delimiter_->getDelimitation();
318 
319  if (silence_ == Fsa::InvalidLabelId)
320  respondToDelayedErrors();
321  for (time = 0; time < delimitation.first; ++time)
322  alignment.push_back(AlignmentItem(time, silence_));
323 
324  f32 slope = states_.size() - 1;
325  slope /= std::max((delimitation.second - delimitation.first), 1u);
326  u32 previousIndex = 0;
327  for (; time <= delimitation.second; ++time) {
328  u32 index = static_cast<u32>(round(slope * (time - delimitation.first)));
329  if (index - previousIndex > maximumJump_) {
330  alignment.clear();
331  return false;
332  }
333  alignment.push_back(AlignmentItem(time, states_[index]));
334  previousIndex = index;
335  }
336  if (previousIndex != (states_.size() - 1)) {
337  warning("Alignment problem, index mismatch: %u vs %zd", previousIndex, states_.size() - 1);
338  alignment.clear();
339  return false;
340  }
341 
342  for (; time < nFeatures(); ++time)
343  alignment.push_back(AlignmentItem(time, silence_));
344 
345  return true;
346 }
