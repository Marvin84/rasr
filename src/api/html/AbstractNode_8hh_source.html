102  /** Fetch data packet from the "from" port of link @param l.
103  * @param d data pointer to which retrieved packet is attached
104  * @return true on success, or false if the out-of-data
105  */
106  template<class T>
107  bool getData(Link* l, DataPtr<T>& d) {
108  if (l != 0) {
109  if (l->isDataAvailable())
110  return l->getData(d);
111  else {
112  if (l->getFromNode()->work(l->getFromPort()))
113  return l->getData(d);
114  else {
115  error("Node '%s' could not generate any output.",
116  l->getFromNode()->name().c_str());
117  }
118  }
119  }
120  d.reset();
121  return false;
122  }
