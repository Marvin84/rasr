29 /** harmonicSum
30  * s(n) = sum_{h = 1}^{h = H} (x_{(h * n) mod T) where
31  *
32  * The x is expected to be periodic (!) with a periodicity of T.
33  * Thus @param x represents one periode, i.e. it is of length T.
34  *
35  * Length of @param s determines the interval for the index n.
36  * @param H gives the maximum schrinkage.
37  */
38 
39 template<class Type>
40 void harmonicSum(const std::vector<Type>& x, std::vector<Type>& s, u32 H) {
41  ensure(x.size() >= s.size());
42 
43  u32 N = s.size();
44  u32 T = x.size();
45  H = H ? H : T;
46 
47  std::fill(s.begin(), s.end(), 0);
48 
49  for (u32 n = 0; n < N; n++) {
50  for (u32 h = 1; h <= H; h++)
51  s[n] += x[(n * h) % T];
52  }
53 }

55 /** harmonicProduct
56  * s(n) = prod_{h = 1}^{h = H} (x_{(h * n) mod T) where
57  *
58  * The x is expected to be periodic (!) with a periodicity of T.
59  * Thus @param x represents one periode, i.e. it is of length T.
60  *
61  * Length of @param s determines the interval for the index n.
62  * @param H gives the maximum schrinkage.
63  *
64  * normalization: to keep the product representable at high values of H
65  * - for each h the normalization factor is
66  * square root of the product of the energy of signal shrunken by h and
67  * the energy of the harmonic product at h -1
68  * - this normalization is motivated by the fourier transformed equivalent of product
69  * of symethric functions: cross-correlation.
70  */
71 
72 template<class Type>
73 void harmonicProduct(const std::vector<Type>& x, std::vector<Type>& s, u32 H) {
74  ensure(x.size() >= s.size());
75 
76  u32 N = s.size();
77  u32 T = x.size();
78  H = H ? H : T;
79 
80  s.resize(T);
81 
82  std::fill(s.begin(), s.end(), 1);
83 
84  for (u32 h = 1; h <= H; h++) {
85  Type energy = 0;
86 
87  for (u32 n = 0; n < T; n++)
88  energy += x[(n * h) % T] * x[(n * h) % T];
89 
90  Type harmonicProductEnergy = 0;
91 
92  for (u32 n = 0; n < T; n++)
93  harmonicProductEnergy += s[n] * s[n];
94 
95  Type normalize = sqrt(2 * energy * harmonicProductEnergy) / x.size();
96 
97  for (u32 n = 0; n < T; n++)
98  s[n] *= x[(n * h) % T] / normalize;
99  }
100 
101  s.resize(N);
102 }
