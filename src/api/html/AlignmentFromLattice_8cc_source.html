88 
89  Flow::DataPtr<Feature::FlowFeature> in;

91  SegmentwiseFeaturesRef features(new SegmentwiseFeatures);
92  while (getData(1, in)) {
93  Core::Ref<Feature> feature(new Feature(in));
94  features->feed(feature);

96  }
97  alignmentGenerator_->setFeatures(features);
98 
99  Flow::DataPtr<Flow::DataAdaptor<Flf::ConstLatticeRef>> lin;
100  Flf::ConstLatticeRef lattice;
101  while (getData(0, lin)) {
102  lattice = lin->data();

104  alignmentGenerator_->getAlignment(alignment->data(), lattice);
105 
106  if (writeAlphabet_)
107  alignment->data().setAlphabet(modelCombination_->acousticModel()->allophoneStateAlphabet());
108 
109  return putData(0, alignment) && putData(0, in.get());
