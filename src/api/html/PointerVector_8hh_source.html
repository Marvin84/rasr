23 /** Container of object references in the form of pointers.
24  * Supported functions:
25  * -Controlled adding of new object references.
26  * -Obtaining of constant/not-constant object references.
27  * -Deleting object referenced by elements of the vector.
28  */
29 template<class T>
30 class PointerVector : private std::vector<T*> {
31 public:

57  /** Adds the @param newObject to the position @param newIndex.
58  * Assertion is raised if the position @param newIndex is not zero.
59  * If @param newIndex is larger than the last index, zero elements are inserted.
60  * @return is index of the object.
61  */
62  size_t add(size_t newIndex, T* newObject) {
63  if (newIndex >= this->size())
64  Parent::resize(newIndex + 1, 0);
65  require(Parent::operator[](newIndex) == 0);
66  Parent::operator[](newIndex) = newObject;
67  return newIndex;
68  }
69 
70  T* release(size_t index) {
71  require(index < size());
72  T* result = Parent::operator[](index);
73  Parent:: operator[](index) = 0;
74  return result;
75  }
76 
77  /** New object is appended to the vector.
78  * @return is index of the object.
79  */
80  size_t pushBack(T* newObject) {
81  Parent::push_back(newObject);
82  return size() - 1;
83  }
84  /** Append with position verification
85  * New object is appended to the vector and assertion fails
86  * if @param newIndex is not equal to the position of the object.
87  */
88  size_t pushBack(size_t newIndex, T* newObject) {
89  require(newIndex == size());
90  return pushBack(newObject);
91  }
92 
93  /** @return type is T* to prevent the pointer value from changing at @param index.
94  */
95  T* operator[](size_t index) const {

117  void clear();
118  void resize(size_t _size) {
119  if (_size <= size()) {
120  for (size_t i = _size; i < this->size(); ++i)
121  release(i);
122  Parent::resize(_size);
123  }
124  else
125  Parent::resize(_size, 0);
126  }
