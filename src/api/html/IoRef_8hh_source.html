27 /**
28  * IoRef extends Ref by allowing reading and writing to binary streams.
29  *
30  * WARNING: This does NOT implement object serialisation (i.e. an object
31  * pointed to through two references that are both saved will be saved
32  * twice, and later restored as two separate copies.), the only goal of this
33  * class is to allow polymorphic containers to be saved and restored.
34  *
35  * A class participating in polymorpic binary stream operations must, in
36  * adition to the standard 'read' and 'write' methods, suport the virtual
37  * member function 'typeName', that provides an identifying 'std::string', that
38  * is unique among the different classes loaded through that particular
39  * 'IoRef' type. The member function 'typeName' must not access the state of
40  * the object in any way. (Think of it as a "static virtual".)
41  *
42  * All classes participating in polymorpic IO must be registered beforhand,
43  * using the 'registerClass' static member function. In addition to
44  * registring the IO tag, this also establishes the way to create "empty"
45  * instances of the class; the constructor of the registered class is then
46  * called using the arguments suplied to 'registerClass'.
47  *
48  * NOTE: At present only constructors with one or two arguments are
49  * suported. If needed it is easy to extend.
50  */
51 template<class Base>
52 class IoRef : public Ref<Base> {
53 public:
54  typedef Ref<Base> Precursor;
55  typedef Base* (*CreatorFunction)();

57 private:

61  }

114  arg1_<Derived, T1>() = arg1;

123  void read(BinaryInputStream& is) {
124  std::string tag;
125  is >> tag;
126  if (!(*this) || (*this)->typeName() != tag) {
127  CreatorFunction creator = creators_()[tag];
128  hope(creator);
129  (*this) = Core::ref(creator());
130  }
131  (*this)->read(is);
132  }
