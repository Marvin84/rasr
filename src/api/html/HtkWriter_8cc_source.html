26 void HtkWriter::write(
27  const std::string& id,
28  ConstWordLatticeRef lattice,
29  std::ostream& os) const {
30  lattice = normalize(lattice);
31  Fsa::ConstAutomatonRef acoustic = lattice->part(WordLattice::acousticFsa);
32  Fsa::ConstAutomatonRef lm = lattice->part(WordLattice::lmFsa);
33  require(acoustic->getInputAlphabet() == lpa_);
34 
35  os << "VERSION=1.1" << std::endl
36  << "UTTERANCE=" << id << std::endl;
37 
38  Fsa::AutomatonCounts counts = Fsa::count(acoustic);
39  os << "NODES=" << counts.nStates_ << '\t'
40  << "LINKS=" << counts.nArcs_ << std::endl;
41 
42  for (Fsa::StateId si = 0; si <= counts.maxStateId_; ++si) {
43  Fsa::ConstStateRef state = acoustic->getState(si);
44  if (!state)
45  continue;
46  os << "I=" << state->id() << '\t'
47  << "t=" << lattice->time(state->id())
48  << std::endl;
49  }
50 
51  u32 linkId = 0;
52  for (Fsa::StateId si = 0; si <= counts.maxStateId_; ++si) {
53  Fsa::ConstStateRef state = acoustic->getState(si);
54  if (!state)
55  continue;
56  for (Fsa::State::const_iterator arc = state->begin(); arc != state->end(); ++arc) {
57  // word and pronunciation variant
58  std::string word;
59  u32 variantNumber = 0;
60  const Bliss::LemmaPronunciation* lemmaPronunciation = lpa_->lemmaPronunciation(arc->input());
61  if (lemmaPronunciation) {
62  const Bliss::Lemma* lemma = lemmaPronunciation->lemma();
63  word = std::string(lemma->name());
64  Bliss::Lemma::PronunciationIterator lpi, lpi_end;
65  for (Core::tie(lpi, lpi_end) = lemma->pronunciations(); lpi != lpi_end; ++lpi) {
66  if (lpi == lemmaPronunciation)
67  break;
68  ++variantNumber;
69  }
70  }
71  else {
72  word = "@";
73  }
74  // acoustic likelihood
75  f32 acousticScore = arc->weight();
76  // language model likelihood
77  f32 lmScore = lm->getState(si)->begin()[arc - state->begin()].weight();
78  os << "J=" << linkId++ << '\t'
79  << "S=" << state->id() << '\t'
80  << "E=" << arc->target() << '\t'
81  << "W=" << '"' << word << '"' << '\t'
82  << "v=" << variantNumber << '\t'
83  << "a=" << -acousticScore << '\t'
84  << "l=" << -lmScore
85  << std::endl;
86  }
87  }
88  verify(linkId == counts.nArcs_);
89 }
