92 CombineLanguageModel::CombineLanguageModel(Core::Configuration const& c, Bliss::LexiconRef l)
93  : Core::Component(c), CombineLanguageModel::Precursor(c, l), lms_(), unscaled_lms_(), linear_combination_(paramLinearCombination(c)), lookahead_lm_(paramLookaheadLM(config)), recombination_lm_(paramRecombinationLM(config)) {
94  size_t num_lms = paramNumLms(c);

96  Core::Configuration sub_config = select(std::string("lm-") + std::to_string(i + 1));
97  lms_.push_back(Module::instance().createScaledLanguageModel(sub_config, l));

99  ssa_lms_.push_back(dynamic_cast<SearchSpaceAwareLanguageModel const*>(unscaled_lms_.back().get()));
100  skip_thresholds_.push_back(paramSkipThreshold(sub_config));
101  }
102  historyManager_ = new CombineHistoryManager(num_lms);
103 }

165 Score CombineLanguageModel::score(History const& history, Token w) const {
166  require(history.isManagedBy(historyManager_));
167  History const* hist = reinterpret_cast<History const*>(history.handle());
168  Score prev_score = 0.0;
169  bool override_score = false;
170  if (linear_combination_) {
171  Score s(std::numeric_limits<Score>::infinity());
172  for (size_t i = 0ul; i < lms_.size(); i++) {
173  Score raw_score = 0.0;
174  if (not override_score) {
175  raw_score = unscaled_lms_[i]->score(hist[i], w);
176  prev_score = raw_score;
177  override_score |= raw_score >= skip_thresholds_[i];
178  }
179  else {
180  raw_score = prev_score;
181  if (unscaled_lms_[i]->scoreCached(history, w)) {
182  raw_score = unscaled_lms_[i]->score(hist[i], w);
183  }
184  }
185  s = Math::scoreSum(s, raw_score - std::log(lms_[i]->scale()));
186  }
187  return s;
188  }
189  else {
190  Score s(0.0);
191  for (size_t i = 0ul; i < lms_.size(); i++) {
192  Score raw_score = unscaled_lms_[i]->score(hist[i], w);
193  if (not override_score) {
194  raw_score = unscaled_lms_[i]->score(hist[i], w);
195  prev_score = raw_score;
196  override_score |= raw_score >= skip_thresholds_[i];
197  }
198  else {
199  raw_score = prev_score;
200  if (unscaled_lms_[i]->scoreCached(history, w)) {
201  raw_score = unscaled_lms_[i]->score(hist[i], w);
202  }
203  }
204  s += raw_score * lms_[i]->scale();
205  }
206  return s;
207  }
208 }
