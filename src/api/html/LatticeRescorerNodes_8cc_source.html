56  }
57 
58  Flow::DataPtr<Flow::DataAdaptor<Flf::ConstLatticeRef>> in;
59  if (!getData(1, in)) {

61  return putData(0, in.get());

64  Flf::ConstLatticeRef l = in->data();

74  return putData(0, out) && putData(0, in.get());

144

146  if (!getData(1, inHyp)) {
147  error("could not read port hypotheses");
148  return putData(0, inHyp.get());
149  }
150 
151  Flow::DataPtr<Flow::DataAdaptor<AlignmentGeneratorRef>> inAli;
152  if (!getData(2, inAli)) {
153  error("could not read port alignments");
154  }
155  alignmentGenerator_ = inAli->data();
156 
157  Flow::DataPtr<Flow::DataAdaptor<Flf::ConstLatticeRef>> inRef;
158  if (!getData(3, inRef)) {
159  error("could not read port references");
160  return putData(0, inRef.get());
161  }
162 
163  Lattice::ConstWordLatticeRef ref = toWordLattice(inRef->data());
164  Lattice::ConstWordLatticeRef hyp = toWordLattice(inHyp->data());
165  Fsa::ConstAutomatonRef dist = getDistanceFsa(ref, hyp);
166  Flf::ConstLatticeRef l = inHyp->data();
167  l = Flf::fromFsa(dist, Flf::ConstSemiringRef(new Flf::TropicalSemiring(1)), 0);
168  l->setBoundaries(inHyp->data()->getBoundaries());
169  Flow::DataAdaptor<Flf::ConstLatticeRef>* out = new Flow::DataAdaptor<Flf::ConstLatticeRef>();
170  out->data() = l;
171  putData(0, out);
172 
173  require(!getData(1, inHyp));
174  require(!getData(2, inAli));
175  require(!getData(3, inRef));
176  return putData(0, inHyp.get());

410 Fsa::ConstAutomatonRef AcousticLatticeRescorerNode::getRescoredFsa(Lattice::ConstWordLatticeRef lattice) {
411  Fsa::ConstAutomatonRef fsa;
412  switch (rescoreMode_) {
413  case rescoreModeAlignment:
414  fsa = Fsa::ConstAutomatonRef(new AlignmentLatticeRescorerAutomaton(lattice, alignmentGenerator_));
415  break;
416  case rescoreModeCombined:
417  fsa = Fsa::ConstAutomatonRef(new CombinedAcousticLatticeRescorerAutomaton(lattice,
418  alignmentGenerator_,
419  acousticModel_,
420  alignmentGenerator_->features(), alignmentGenerator_->allophoneStateGraphBuilder()));
421  break;
422  case rescoreModeEm:
423  fsa = Fsa::ConstAutomatonRef(new EmissionLatticeRescorerAutomaton(lattice,
424  alignmentGenerator_,
425  alignmentGenerator_->features(),
426  acousticModel_));
427  break;
428  case rescoreModeTdp:
429  fsa = Fsa::ConstAutomatonRef(new TdpLatticeRescorerAutomaton(lattice,
430  alignmentGenerator_,
431  alignmentGenerator_->allophoneStateGraphBuilder(),
432  acousticModel_));
433  break;
434  default:
435  criticalError("Unknown mode in AcousticLatticeRescorerNode.");
436  }
437  return fsa;
438 }
