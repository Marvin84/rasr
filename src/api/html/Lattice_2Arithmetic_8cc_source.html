88 ConstWordLatticeRef linearCombination(
89  ConstWordLatticeRef lattice, const std::vector<Fsa::Weight>& scales) {
90  if (lattice->nParts() > 0) {
91  Fsa::ConstAutomatonRef total =
92  f32(scales[0]) != 1 ? Fsa::multiply(lattice->part(0), scales[0]) : lattice->part(0);
93  for (size_t i = 1; i < lattice->nParts(); ++i) {
94  total =
95  Fsa::extend(
96  total,
97  f32(scales[i]) != 1 ? Fsa::multiply(lattice->part(i), scales[i]) : lattice->part(i));
98  }
99  WordLattice* result = new WordLattice();
100  result->setWordBoundaries(lattice->wordBoundaries());
101  result->setFsa(Fsa::cache(total), WordLattice::totalFsa);
102  return ConstWordLatticeRef(result);
103  }
104  return ConstWordLatticeRef();
105 }
