24 /*
25  * This introduces a meta template list, implemented via variadic templates.
26  * Some reference:
27  * http://en.cppreference.com/w/cpp/language/parameter_pack
28  * C++ standard:
29  * http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2007/n2242.pdf
30  * See Boost.MPL for much more of this magic.
31  *
32  * Usage example:
33  *
34  * typedef TList<s32, f32> MyTypeList;
35  *
36  * struct MyHandler {
37  * template<typename T>
38  * void handle(int extra) {
39  * cout << T(extra) << " ";
40  * }
41  * };
42  *
43  * MyTypeList(MyHandler(), 42); // prints "42 42.0 "
44  *
45  * cout << MyTypeList::contains<s32>::value; // prints "1"
46  * cout << MyTypeList::contains<u8>::value; // prints "0"
47  *
48  */
49 
50 template<typename... Elems>
51 struct TList {
52  // This matches no template args.
53  static const size_t size = 0;
54 
55  template<typename Handler, typename... HandlerArgs>
56  static Handler&& forEach(Handler&& handler, HandlerArgs...) {
57  return std::forward<Handler>(handler);
58  }
59 
60  template<typename OtherElem>
61  struct contains : std::false_type {};
62 };

64 template<typename FirstElem, typename... OtherElems>
65 struct TList<FirstElem, OtherElems...> {
66  // This mathes one or more template args.
67  typedef FirstElem FirstElemT;
68  typedef TList<OtherElems...> TSubListT;
69  static const size_t size = 1 + sizeof...(OtherElems);
70 
71  template<typename Handler, typename... HandlerArgs>
72  static Handler&& forEach(Handler&& handler, HandlerArgs... args) {
73  handler.template handle<FirstElem>(args...);
74  return std::forward<Handler>(
75  TSubListT::forEach(handler, args...));
76  }
77 
78  template<typename OtherElem>
79  struct contains : std::integral_constant<bool, std::is_same<FirstElem, OtherElem>::value ||
80  TSubListT::template contains<OtherElem>::value> {};
81 };
