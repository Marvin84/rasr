111 CovarianceToDensitySetMap::CovarianceToDensitySetMap(const AbstractMixtureSetEstimator::MixtureEstimators& mixtureEstimators) {
112  MixtureEstimators::const_iterator m = mixtureEstimators.begin();
113  for (; m != mixtureEstimators.end(); ++m) {
114  Core::Ref<EbwDiscriminativeMixtureEstimator> mixtureEstimator(required_cast(EbwDiscriminativeMixtureEstimator*, m->get()));
115  for (DensityIndex dnsInMix = 0; dnsInMix < mixtureEstimator->nDensities(); ++dnsInMix) {

117  augmented.densityEstimator = Core::Ref<EbwDiscriminativeGaussDensityEstimator>(required_cast(EbwDiscriminativeGaussDensityEstimator*,

123  * Not all tying schemes are supported yet.
124  * More precisely, a density cannot be shared
125  * by more than one mixture.
126  */
127  require(set.find(augmented) == set.end());
128  set.insert(augmented);
129  }

180 IterationConstant Xi::denominator(Core::Ref<AbstractMixtureEstimator> mixtureEstimator) {
181  verify(mixtureEstimator);
182  IterationConstant maxAbsDtWeight = 0;
183  IterationConstant maxNumDenWeight = 0;
184  DensityEstimators::const_iterator d = mixtureEstimator->densityEstimators().begin();
185  for (; d != mixtureEstimator->densityEstimators().end(); ++d) {
186  Core::Ref<EbwDiscriminativeMeanEstimator> meanEstimator(required_cast(EbwDiscriminativeMeanEstimator*,
187  (*d)->mean().get()));
188  const IterationConstant absDtWeight = Core::abs(meanEstimator->dtWeight());
189  if (maxAbsDtWeight < absDtWeight) {
190  maxAbsDtWeight = absDtWeight;
191  maxNumDenWeight = std::max(meanEstimator->dtWeight() + meanEstimator->denWeight(), meanEstimator->denWeight());
192  }
193  }
194  if (maxAbsDtWeight < maxNumDenWeight and maxNumDenWeight > 0) {
195  return 1 + (maxAbsDtWeight - 1) * maxAbsDtWeight / maxNumDenWeight;
196  }
197  else {
198  return 1;
199  }
200 }

203  * iteration constants b
204  *
205  * t1 := -Gamma_{s,l}(x_d^2) + sigma_min·Gamma_{s,l}(1)
206  * t2 := 2·Gamma_{s,l}(x_d) - Gamma_{s,l}(1)·mu_{s,l,d}

212  *
213  * with sigma*^2 in {sigma_{s,l,d}^2, sigma_{s_d}^2, sigma_d^2}

295 /**
296  * calculate D_k_min
297  */
298 IterationConstant Icb::minimumDk(CovarianceToDensitySetMap::const_iterator covarianceAndDensitySet,
299  const CovarianceToMeanSetMap::MeanSet& meanSet) {
300  Core::Ref<EbwDiscriminativeCovarianceEstimator> covarianceEstimator(required_cast(EbwDiscriminativeCovarianceEstimator*,
301  covarianceAndDensitySet->first.get()));
302  const std::vector<Sum> dtSum2 = covarianceEstimator->dtSum(meanSet);
303  const std::vector<VarianceType>& previousCovariance = covarianceEstimator->previousCovariance();
304  ComponentIndex dimension = previousCovariance.size();
305  std::vector<IterationConstant> numerator(dimension, 0);
306  std::transform(numerator.begin(), numerator.end(), dtSum2.begin(),
307  numerator.begin(), std::minus<IterationConstant>());
308  std::vector<IterationConstant> denominator(dimension, 0);
309  const DensitySet& densitySet = covarianceAndDensitySet->second;
310  DensitySet::const_iterator d = densitySet.begin();
311  for (; d != densitySet.end(); ++d) {
312  Core::Ref<EbwDiscriminativeMeanEstimator> meanEstimator(required_cast(EbwDiscriminativeMeanEstimator*,
313  d->densityEstimator->mean().get()));
314  const std::vector<Sum> dtSum = meanEstimator->dtSum();
315  Weight dtWeight = meanEstimator->dtWeight();
316  const std::vector<MeanType>& previousMean = meanEstimator->previousMean();
317  for (ComponentIndex i = 0; i < dimension; ++i) {
318  const IterationConstant t2 = 2 * dtSum[i] - dtWeight * previousMean[i];
319  const IterationConstant t3 = dtSum[i] - dtWeight * previousMean[i];
320  numerator[i] += sigmaMinimum_ * dtWeight;
321  numerator[i] += t2 * previousMean[i];
322  numerator[i] += xi_[d->mixtureEstimator] * t3 * t3;
323  denominator[i] += (previousCovariance[i] - sigmaMinimum_) * d->previousMixtureWeight;
324  }
325  }
326  std::vector<IterationConstant> ratio(dimension);
327  std::transform(numerator.begin(), numerator.end(), denominator.begin(),
328  ratio.begin(), std::divides<IterationConstant>());
329  IterationConstant ratioMax = *std::max_element(ratio.begin(), ratio.end());
330  log("D-k-min-raw: ") << Core::form("%.2f", ratioMax);
331  IterationConstant D_k_min = ratioMax > 0 ? ratioMax : 0;
332  log("D-k-min: ") << Core::form("%.2f", D_k_min);
333  return D_k_min;
334 }

356 GlobalIcb::GlobalIcb(const Core::Configuration& c,
357  const MixtureEstimators& mixtureEstimators,
358  const CovarianceToDensitySetMap& cim,
359  IterationConstant stepSize,
360  const CovarianceToMeanSetMap& meanSetMap)
361  : Precursor(c, mixtureEstimators, cim, stepSize) {
362  CovarianceToDensitySetMap::const_iterator it = cim.begin();
363  for (; it != cim.end(); ++it) {
364  const CovarianceToMeanSetMap::MeanSet& meanSet = meanSetMap[Core::Ref<AbstractCovarianceEstimator>(it->first)];
365  const IterationConstant D_k_min = minimumDk(it, meanSet);
366  IterationConstant ic = default_;
367 
368  /**
369  * D_k = h · max_{s*,l*} { 1/c_{s,l} · (1/xi_s - Gamma_{s,l}(1)), D_k^min }
370  */
371  const DensitySet& densitySet = it->second;
372  DensitySet::const_iterator d = densitySet.begin();
373  for (; d != densitySet.end(); ++d) {
374  Core::Ref<EbwDiscriminativeMeanEstimator> meanEstimator(required_cast(EbwDiscriminativeMeanEstimator*,
375  d->densityEstimator->mean().get()));
376  IterationConstant D_k = 1 / xi_[d->mixtureEstimator];
377  D_k -= meanEstimator->dtWeight();
378  if (d->previousMixtureWeight > 0) {
379  D_k /= d->previousMixtureWeight;
380  }
381  ic = std::max(ic, std::max(D_k, D_k_min));
382  }
383  map_.insert(it->first, ic * stepSize_);
384  log("icb: ") << b(it->first);
385  }
386 }

408 LocalIcb::LocalIcb(const Core::Configuration& c,
409  const MixtureEstimators& mixtureEstimators,
410  const CovarianceToDensitySetMap& cim,
411  IterationConstant stepSize,
412  const CovarianceToMeanSetMap& meanSetMap)
413  : Precursor(c, mixtureEstimators, cim, stepSize) {
414  CovarianceToDensitySetMap::const_iterator it = cim.begin();
415  for (; it != cim.end(); ++it) {
416  const CovarianceToMeanSetMap::MeanSet& meanSet =
417  meanSetMap[Core::Ref<AbstractCovarianceEstimator>(it->first)];
418  const IterationConstant D_k_min = minimumDk(it, meanSet);
419 
420  /**
421  * D_k = h · max_{s*,l*} { 1/c_{s,l} · (1/xi_s - Gamma_{s,l}(1)), D_k^min }
422  */
423  const DensitySet& densitySet = it->second;
424  DensitySet::const_iterator d = densitySet.begin();
425  for (; d != densitySet.end(); ++d) {
426  if (map_.find(d->mixtureEstimator) == map_.end()) {
427  map_.insert(d->mixtureEstimator, stepSize_ * std::max(default_, D_k_min));
428  }
429  Core::Ref<EbwDiscriminativeMeanEstimator> meanEstimator(required_cast(EbwDiscriminativeMeanEstimator*,
430  d->densityEstimator->mean().get()));
431  IterationConstant D_k = 1 / xi_[d->mixtureEstimator];
432  D_k -= meanEstimator->dtWeight();
433  if (d->previousMixtureWeight > 0) {
434  D_k /= d->previousMixtureWeight;
435  }
436  map_[d->mixtureEstimator] = std::max(stepSize_ * D_k, map_[d->mixtureEstimator]);
437  }
438  }

591 Sum CambridgeIterationConstants::maximumRoot(Core::Ref<EbwDiscriminativeMeanEstimator> meanEstimator,
592  Core::Ref<EbwDiscriminativeCovarianceEstimator> covarianceEstimator) const {
593  verify(meanEstimator and covarianceEstimator);
594 
595  require(false); // beck: why is this here ????
596  const std::vector<Sum> Gamma_x2_sl;
597  const std::vector<Sum> Gamma_x_sl;
598  Weight Gamma_1_sl = 0;
599 
600  const std::vector<MeanType>& mu_sl = meanEstimator->previousMean();
601  const std::vector<VarianceType>& sigma_sl = covarianceEstimator->previousCovariance();
602 
603  const ComponentIndex dimension = mu_sl.size();
604  verify(Gamma_x2_sl.size() == dimension && Gamma_x_sl.size() == dimension);
605  verify(mu_sl.size() == dimension);
606  Sum rootMax = Core::Type<Sum>::min;
607  for (ComponentIndex i = 0; i < dimension; ++i) {
608  const Sum a = sigma_sl[i];
609  const Sum b = Gamma_x2_sl[i] + Gamma_1_sl * (sigma_sl[i] + mu_sl[i] * mu_sl[i]) - 2.0 * Gamma_x_sl[i] * mu_sl[i];
610  const Sum c = Gamma_1_sl * Gamma_x2_sl[i] - Gamma_x_sl[i] * Gamma_x_sl[i];
611 
612  Sum signB = b < 0 ? -1 : 1;
613  const Sum root2 = -0.5 * (b + signB * sqrt(b * b - 4.0 * a * c)) / a;
614  rootMax = std::max(rootMax, root2);
615  }
616 
617  int expo;
618  return ldexp(frexp(rootMax, &expo) + Core::Type<f32>::epsilon, expo);
619 }

621 void CambridgeIterationConstants::set(const InternalIc::AugmentedDensityEstimator& d) {
622  IterationConstant ic = default_;
623 
624  // do not take low occupied densities into account
625  Core::Ref<EbwDiscriminativeMeanEstimator> meanEstimator(required_cast(EbwDiscriminativeMeanEstimator*,
626  d.densityEstimator->mean().get()));
627  if (meanEstimator->weight() < minimumObservationWeight_) {
628  d.densityEstimator->setIterationConstant(ic);
629  return;
630  }
631 
632  Core::Ref<EbwDiscriminativeCovarianceEstimator> covarianceEstimator(required_cast(EbwDiscriminativeCovarianceEstimator*,
633  d.densityEstimator->covariance().get()));
634  const Sum rootMax = maximumRoot(meanEstimator, covarianceEstimator);
635  ic = ic > rootMax ? ic : rootMax;
636  if (meanEstimator->denWeight() == 0) {
637  d.densityEstimator->setIterationConstant(ic);
638  return;
639  }
640 
641  IterationConstant eHalfmax = 0.5 * stepSize_ * rootMax / meanEstimator->denWeight();
642  verify(eHalfmax >= 0);
643  IterationConstant e = (e_ == 0.0) ? eHalfmax : e_;
644  e *= meanEstimator->denWeight();
645  ic *= stepSize_;
646  ic = ic > e ? ic : e;
647  d.densityEstimator->setIterationConstant(ic);
648 }
