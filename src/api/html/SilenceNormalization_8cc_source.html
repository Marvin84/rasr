25 class GaussianDensity {
26 public:
27  GaussianDensity(f32 bias = 1.0)
28  : mean_(0), variance_(0), sigma_(0), offset_(0), energySum_(0), energySquareSum_(0), energyCount_(0), bias_(bias) {
29  }
30 
31  // Returns the score regarding the gaussian distribution
32  f64 score(f64 value) const {
33  f64 d = (value - mean_) / sigma_;
34  d = (d * d) * 0.5;
35  return (d + offset_) / bias_;
36  }
37 
38  void add(f64 energy) {
39  energySum_ += energy;
40  energyCount_ += 1;
41  energySquareSum_ += energy * energy;
42  }
43 
44  void estimate(bool estimateVariance = false) {
45  if (energyCount_) {
46  mean_ = energySum_ / energyCount_;
47  if (!estimateVariance) {
48  variance_ = (energySquareSum_ - 2 * mean_ * energySum_ + energyCount_ * mean_ * mean_) / energyCount_;
49  if (variance_ < 1)
50  variance_ = 1;
51  sigma_ = sqrt(variance_);
52  }
53  offset_ = sigma_ * sqrt(2 * M_PI);
54  }
55  energySum_ = 0;
56  energySquareSum_ = 0;
57  energyCount_ = 0;
58  }
59 
60  f64 mean() const {
61  return mean_;
62  }
63 
64  f64 sigma() const {
65  return sigma_;
66  }
67 
68 private:
69  f64 mean_;
70  f64 variance_;
71  f64 sigma_;
72  f64 offset_;
73 
74  f64 energySum_;
75  f64 energySquareSum_;
76  u32 energyCount_;
77 
78  f32 bias_;
79 };

195 bool SilenceNormalization::flushFromQueue(Vector<SilenceNormalization::Sample>& out) {
196  verify(out.empty());
197  if (flushQueue_.empty())
198  return false;
199  out = flushQueue_.front();

202  if (addNoise_) {
203  for (std::vector<Sample>::iterator it = out.begin(); it != out.end(); ++it)

209 void SilenceNormalization::startFlushingFromQueue(const std::vector<bool>& isSpeech) {
210  if (buffer_.empty())
211  return;
212  Time tolerance = 0.5 / sampleRate_;
213  Time samplelength = 1 / sampleRate_;
214  Vector<Sample> o;
215  u32 ofs = 0;
216  for (u32 i = 0; i < isSpeech.size(); ++i, ++ofs) {
217  if (isSpeech[i])
218  break;
219  }
220  std::cout << "offset: " << ofs << std::endl;
221  for (u32 sample = 0; sample < buffer_.size(); ++sample) {
222  if (!isSpeech.empty()) {
223  u32 block = sample / blockSizeSamples_;
224  verify(block < isSpeech.size());
225  if (!isSpeech[block])
226  continue;
227  }
228  if (!o.empty() && buffer_[sample].first > o.endTime() + tolerance && preserveTiming_) {
229  flushQueue_.push_back(o);
230  o = Vector<Sample>();
231  }
232  if (o.empty())
233  o.setStartTime(buffer_[sample].first);
234  o.push_back(buffer_[sample].second);
235  o.setEndTime(buffer_[sample].first + samplelength);
236  }
237 
238  if (!preserveTiming_) {
239  verify(flushQueue_.empty());
240  o.setStartTime(buffer_.front().first);
241  }
242 
243  o.setEndTime(o.startTime() + o.size() * samplelength);
244 
245  if (!o.empty())
246  flushQueue_.push_back(o);
247 
248  if (!flushQueue_.empty())
249  std::cout << "time range of first flushed item: " << flushQueue_.front().startTime() << " " << flushQueue_.front().endTime() << "(real start: " << buffer_.front().first << ")" << std::endl;
250 
251  buffer_.clear();
252 }

254 bool SilenceNormalization::flush(Vector<Sample>& out) {
255  if (needInit_)
256  init();
257 
258  out.clear();
259 
260  if (flushFromQueue(out))
261  return true;
262 
263  if (buffer_.empty())
264  return false;
265 
266  if (silenceFraction_ == 1.0 || buffer_.size() < blockSizeSamples_ * 6) {
267  if (silenceFraction_ < 1.0)
268  log() << "buffer too short:" << buffer_.size() << " min. " << blockSizeSamples_ * 6;
269  startFlushingFromQueue(std::vector<bool>());
270  return flushFromQueue(out);
271  }
272 
273  std::vector<f64> blocks;
274 
275  for (u32 b = 0; b < 1 + buffer_.size() / blockSizeSamples_; ++b) {
276  u32 start = b * blockSizeSamples_;
277  if (start >= buffer_.size())
278  break;
279  u32 end = (b + 1) * blockSizeSamples_;
280  if (end > buffer_.size())
281  end = buffer_.size();
282  f64 energySum = 0;
283  for (u32 s = start; s < end; ++s)
284  energySum += Core::abs(buffer_[s].second);
285  blocks.push_back(energySum / (end - start));
286  }
287  verify(blocks.size());
288 
289  std::vector<std::pair<f64, u32>> sortedBlocks;
290  for (u32 b = 0; b < blocks.size(); ++b)
291  sortedBlocks.push_back(std::make_pair(blocks[b], b));
292  std::sort(sortedBlocks.begin(),
293  sortedBlocks.end(),
294  Core::composeBinaryFunction(std::less<f64>(), Core::select1st<std::pair<f64, u32>>(), Core::select1st<std::pair<f64, u32>>()));
295 
296  bool classificationFailed = false;
297  u32 initOffset = std::max(sortedBlocks.size() * initializationFraction_, 3.0);
298 
299  GaussianDensity speech;
300  GaussianDensity silence;
301 
302  // True for speech, false for silence
303  std::vector<bool> isSpeech(blocks.size(), true);
304 
305  for (u32 it = 0; it < iterations_; ++it) {
306  if (it > 0) {
307  // Assign
308  for (u32 block = 0; block < blocks.size(); ++block) {
309  isSpeech[block] = (blocks[block] - silence.mean() >= (speech.mean() - silence.mean()) * silenceThreshold_) && blocks[block] > absoluteSilenceThreshold_;
310  }
311  }
312 
313  // Always re-assign the boundary sets, to make sure that we never flip speech and silence
314  for (u32 i = 0; i < initOffset; ++i) {
315  silence.add(sortedBlocks[i].first);
316  speech.add(sortedBlocks[sortedBlocks.size() - 1 - i].first);
317  }
318 
319  speech.estimate();
320  if (it == 0)
321  silence.estimate();
322  }
323 
324  {
325  u32 speechCount = 0, silenceCount = 0;
326  for (u32 block = 0; block < blocks.size(); ++block)
327  if (isSpeech[block])
328  speechCount += 1;
329  else
330  silenceCount += 1;
331 
332  if (speech.mean() <= silence.mean() || speechCount == 0 || silenceCount == 0) {
333  classificationFailed = true;
334  log() << "segment failed due to misclassification. Total speech: " << speechCount << " total silence: " << silenceCount;
335  isSpeech.assign(isSpeech.size(), !discardUnsure_); // Assign everything to speech, so that the speech-recognizer can decide
336  if (discardUnsure_)
337  log() << "discarded all";
338  else
339  log() << "accepted all";
340  }
341  }
342 
343  log() << "silence mean " << silence.mean() << " deviation " << silence.sigma();
344  log() << "speech mean " << speech.mean() << " deviation " << speech.sigma();
345 
346  // Apply consistency constraints on the blocks
347  u32 minSurroundingSilenceBlocks = std::max(minSurroundingSilence_ / blockSize_, 1.0);
348 
349  u32 speechCount = 0, silenceInSpeech = 0, silenceCount = 0;
350  for (u32 block = 0; block < blocks.size(); ++block)
351  if (isSpeech[block])
352  speechCount += 1;
353  else
354  silenceCount += 1;
355 
356  f32 oldRatio = silenceCount / (f32)(silenceCount + speechCount);
357  f32 ratio = silenceInSpeech / (f32)speechCount;
358 
359  u32 surrounding = 0;
360 
361  // Extend speech until the targeted ratio is achieved
362  while (surrounding < minSurroundingSilenceBlocks || ratio < silenceFraction_) {
363  ++surrounding;
364  std::vector<bool> isSpeechOld = isSpeech;
365  u32 oldCount = speechCount;
366  for (u32 block = 0; block < blocks.size(); ++block) {
367  if (!isSpeechOld[block] && !isSpeech[block]) {
368  if ((block > 0 && isSpeechOld[block - 1]) ||
369  (block + 1 < blocks.size() && isSpeechOld[block + 1])) {
370  isSpeech[block] = true;
371  silenceInSpeech += 1;
372  speechCount += 1;
373  ratio = silenceInSpeech / (f32)speechCount;
374  if (surrounding >= minSurroundingSilenceBlocks && ratio >= silenceFraction_)
375  break;
376  }
377  }
378  }
379 
380  if (oldCount == speechCount) {
381  log() << "not enough silence available (speech count " << speechCount << ")";
382  break; // Nothing can be done any more
383  }
384  }
385 
386  u32 oldBufferSize = buffer_.size();
387 
388  startFlushingFromQueue(isSpeech);
389 
390  if (fillUpSilence_ && !classificationFailed && flushQueue_.size()) {
391  Flow::Vector<Sample> silenceBlock;
392  while (ratio < silenceFraction_) {
393  u32 block = sortedBlocks[silenceBlock.size() % initOffset].second;
394  u32 firstSample = block * blockSizeSamples_;
395  u32 endSample = std::min((block + 1) * blockSizeSamples_, (u32)buffer_.size());
396  for (u32 sample = firstSample; sample < endSample; ++sample) {
397  silenceBlock.push_back(buffer_[sample].second);
398  }
399  silenceInSpeech += 1;
400  speechCount += 1;
401  ratio = silenceInSpeech / (f32)speechCount;
402  }
403  if (silenceBlock.size()) {
404  silenceBlock.setStartTime(flushQueue_.back().endTime());
405  silenceBlock.setEndTime(silenceBlock.startTime() + silenceBlock.size() / sampleRate_);
406  flushQueue_.push_back(silenceBlock);
407  log() << "added " << silenceBlock.size() << " additional silence samples to reach silence fraction";
408  }
409  }
410  u32 outputSize = 0;
411  for (std::list<Vector<Sample>>::iterator it = flushQueue_.begin(); it != flushQueue_.end(); ++it) {
412  outputSize += it->size();
413  log() << "speech " << it->startTime() << " " << it->endTime() << " " << it->size();
414  }
415  s32 difference = (oldBufferSize - outputSize);
416  log() << "accepted silence/speech samples: " << outputSize << " difference: " << difference << " difference fraction: " << difference / (f32)oldBufferSize << " new silence ratio: " << ratio << " old: " << oldRatio;
417  return flushFromQueue(out);
418 }
