161  /**
162  * Levenshtein alignment with fixed beam around main diagonal
163  **/
164  Score align(const ConfusionNetwork& ref, const ConfusionNetwork& hyp) const {
165  const Cost& costFcn = *costFcn_;
166  costFcn.init(ref, hyp);
167  const u32 lenRef = ref.size(), lenHyp = hyp.size();
168  // special case: empty hypothesis
169  if (lenHyp == 0) {
170  Score D = 0.0;
171  revAlignment_.clear();
172  revAlignment_.reserve(lenRef);
173  for (s32 refIndex = lenRef - 1; refIndex >= 0; --refIndex) {
174  D += costFcn.delCost(refIndex);
175  revAlignment_.push_back(std::make_pair(refIndex, Core::Type<u32>::max));
176  }
177  // return alignment cost
178  return D;
179  }
180  // special case: empty reference
181  if (lenRef == 0) {
182  Score D = 0.0;
183  revAlignment_.clear();
184  revAlignment_.reserve(lenHyp);
185  for (s32 hypIndex = lenHyp - 1; hypIndex >= 0; --hypIndex) {
186  D += costFcn.insCost(hypIndex);
187  revAlignment_.push_back(std::make_pair(Core::Type<u32>::max, hypIndex));
188  }
189  // return alignment cost
190  return D;
191  }
192  // common case:
193  /*
194  * The beam is assured to be wide enough to allow a successful alinment.
195  * Allocate the required memory.
196  */
197  allocate(lenRef, lenHyp);
198  verify((lenHyp <= lenRef + beamWidth_) && (lenRef <= lenHyp + beamWidth_));
199  /*
200  * Initialize alignment.
201  */
202  for (u32 beamIndex = 0; beamIndex < beamWidth_; ++beamIndex) {
203  D_[beamIndex] = MaxCost;
204  B_[beamIndex] = Invalid;
205  }
206  D_[beamWidth_] = 0.0;
207  B_[beamWidth_] = Start;
208  for (u32 refIndex = 0; refIndex < std::min(beamWidth_, lenRef); ++refIndex) {
209  verify(beamWidth_ + 1 + refIndex < beamDiameter_);
210  D_[beamWidth_ + 1 + refIndex] = D_[beamWidth_ + refIndex] + costFcn.delCost(refIndex);
211  verify(beamWidth_ + 1 + refIndex < beamDiameter_ * maxLen_);
212  B_[beamWidth_ + 1 + refIndex] = Deletion;
213  }
214  /*
215  * Perform alignment; iterate over hypothesis.
216  */
217  Core::ProgressIndicator pi(Core::form("align(#hyp=%d,beam=%d)", lenHyp, beamDiameter_));
218  pi.start(lenHyp);
219  for (u32 hypIndex = 0, bptrIndex = beamDiameter_; hypIndex < lenHyp; ++hypIndex, bptrIndex += beamDiameter_, pi.notify()) {
220  /*
221  * Adjust beam start.
222  * Insert insertion at bottom of beam, if still in initialization phase of beam.
223  */
224  u32 beamStart = 0, beamIndex = 0;
225  if (hypIndex < beamWidth_) {
226  beamStart = beamIndex = beamWidth_ - hypIndex;
227  verify(B_[bptrIndex + beamIndex - (beamDiameter_ - 1)] != Invalid);
228  verify(beamIndex < beamDiameter_);
229  D_[beamIndex - 1] = D_[beamIndex] + costFcn.insCost(hypIndex);
230  verify(bptrIndex + beamIndex - 1 < beamDiameter_ * maxLen_);
231  B_[bptrIndex + beamIndex - 1] = Insertion;
232  }
233  /*
234  * Adjust begin and end of reference index.
235  * Iterate over reference slice.
236  */
237  for (u32 refIndex = (hypIndex < beamWidth_) ? 0 : hypIndex - beamWidth_, refEnd = std::min(hypIndex + beamWidth_ + 1, lenRef);
238  refIndex < refEnd; ++refIndex, ++beamIndex) {
239  verify(B_[bptrIndex + beamIndex - beamDiameter_] != Invalid); // substitution
240  Score minCost = D_[beamIndex] + costFcn.subCost(refIndex, hypIndex);
241  OperationType op = Substitution;
242  if (beamIndex > beamStart) {
243  verify(B_[bptrIndex + beamIndex - 1] != Invalid); // deletion
244  Score delCost = D_[beamIndex - 1] + costFcn.delCost(refIndex);
245  if (delCost < minCost) {
246  minCost = delCost;
247  op = Deletion;
248  }
249  }
250  if (beamIndex + 1 < beamDiameter_) {
251  verify(B_[bptrIndex + beamIndex - (beamDiameter_ - 1)] != Invalid); // insertion
252  Score insCost = D_[beamIndex + 1] + costFcn.insCost(hypIndex);
253  if (insCost < minCost) {
254  minCost = insCost;
255  op = Insertion;
256  }
257  }
258  verify(beamIndex < beamDiameter_);
259  D_[beamIndex] = minCost;
260  verify(bptrIndex + beamIndex < beamDiameter_ * maxLen_);
261  B_[bptrIndex + beamIndex] = op;
262  }
263  /*
264  * Invalidate rest of beam.
265  */
266  for (; beamIndex < beamDiameter_; ++beamIndex) {
267  verify(beamIndex < beamDiameter_);
268  D_[beamIndex] = MaxCost;
269  verify(bptrIndex + beamIndex < beamDiameter_ * maxLen_);
270  B_[bptrIndex + beamIndex] = Invalid;
271  }
272  }
273  pi.finish(false);
274  costFcn.reset();
275  /*
276  * Trace
277  */
278  revAlignment_.clear();
279  u32 refIndex = lenRef - 1;
280  u32 hypIndex = lenHyp - 1;
281  u32 bptrIndex = beamDiameter_ * lenHyp + beamWidth_ + lenRef - lenHyp;
282  verify(bptrIndex < beamDiameter_ * maxLen_);
283  while (bptrIndex > beamWidth_) {
284  switch (B_[bptrIndex]) {
285  case Substitution:
286  revAlignment_.push_back(std::make_pair(refIndex, hypIndex));
287  --refIndex;
288  --hypIndex;
289  bptrIndex -= beamDiameter_;
290  break;
291  case Deletion:
292  revAlignment_.push_back(std::make_pair(refIndex, Core::Type<u32>::max));
293  --refIndex;
294  bptrIndex -= 1;
295  break;
296  case Insertion:
297  revAlignment_.push_back(std::make_pair(Core::Type<u32>::max, hypIndex));
298  --hypIndex;
299  bptrIndex -= (beamDiameter_ - 1);
300  break;
301  default:
302  defect();
303  }
304  }
305  verify(B_[bptrIndex] == Start);
306  // return alignment cost
307  verify(beamWidth_ + lenRef - lenHyp < beamDiameter_);
308  return D_[beamWidth_ + lenRef - lenHyp];
309  }

568  /**
569  * The probability that the slot's top hypothesis will be wrong
570  * 1.0 - max_{w} { w_{ref} * p_{ref}(w) + w_{hyp} * p_{hyp}(w) }
571  **/
572  class ExpectedError : public WeightedCost {
573  private:
574  Score delInsCost(const ConfusionNetwork::Slot& slot, ScoreId posteriorId, Score slotWeight, Score epsP) const {
575  Score p = 0.0, maxP = epsP;
576  if (slot.front().label == Fsa::Epsilon) {
577  maxP += slotWeight * slot.front().scores->get(posteriorId);
578  }
579  else {
580  p = slotWeight * slot.front().scores->get(posteriorId);
581  if (p > maxP)
582  maxP = p;
583  }
584  for (ConfusionNetwork::Slot::const_iterator itSlot = slot.begin() + 1, endSlot = slot.end(); itSlot != endSlot; ++itSlot) {
585  p = slotWeight * itSlot->scores->get(posteriorId);
586  if (p > maxP)
587  maxP = p;
588  }
589  verify((0.0 <= maxP) && (maxP <= 1.001));
590  return std::max(0.0, 1.0 - maxP) + 0.001; // prefer substitutions rather than del/ins
591  }

607  virtual Score subCost(u32 refId, u32 hypId) const {
608  const ConfusionNetwork::Slot & ref = refSlot(refId), &hyp = hypSlot(hypId);
609  ConfusionNetwork::Slot::const_iterator itRef = ref.begin();
610  ConfusionNetwork::Slot::const_iterator endRef = ref.end();
611  ConfusionNetwork::Slot::const_iterator itHyp = hyp.begin(), endHyp = hyp.end();
612  Score p = 0.0, maxP = 0.0;
613  for (Fsa::LabelId lastLabel = Core::Type<Fsa::LabelId>::max, nextLabel = itHyp->label;
614  nextLabel != Core::Type<Fsa::LabelId>::max;) {
615  for (; (itRef != endRef) && (itRef->label <= nextLabel); ++itRef) {
616  p = refNormedWeight * itRef->scores->get(refPosteriorId);
617  if (p > maxP)
618  maxP = p;
619  lastLabel = itRef->label;
620  }
621  if ((itHyp != endHyp) && (itHyp->label == lastLabel)) {
622  p += hypNormedWeight * itHyp->scores->get(hypPosteriorId);
623  if (p > maxP)
624  maxP = p;
625  }
626  nextLabel = (itRef != endRef) ? itRef->label : Core::Type<Fsa::LabelId>::max;
627  for (; (itHyp != endHyp) && (itHyp->label < nextLabel); ++itHyp) {
628  p = hypNormedWeight * itHyp->scores->get(hypPosteriorId);
629  if (p > maxP)
630  maxP = p;
631  }
632  nextLabel = (itHyp != endHyp) ? itHyp->label : Core::Type<Fsa::LabelId>::max;
633  }
634  verify((0.0 <= maxP) && (maxP <= 1.001));
635  return std::max(0.0, 1.0 - maxP);
636  }

681  void add(const ConfusionNetworkAlignment& alignment, ConfusionNetwork* combinedCn, ConstConfusionNetworkRef cn, u32 i) const {
682  Score refWeight = sumWeights_[i - 1];
683  Score hypWeight = weightedCombo_->weights()[i];
684  ScoreId posteriorId = cn->normalizedProperties->posteriorId;
685  const SemiringCombinationHelper& semiringCombo = *weightedCombo_->semiringCombination();
686  const Semiring& combinedSemiring = *semiringCombo.semiring();
687  ScoreId combinedPosteriorId = combinedCn->normalizedProperties->posteriorId;
688  ScoresRef subOne = combinedSemiring.one();
689  ScoresRef insOne = combinedSemiring.clone(combinedSemiring.one());
690  ScoresRef delOne = combinedSemiring.clone(combinedSemiring.one());
691  {
692  const ScoreIdList& posteriorIds = weightedCombo_->posteriorIds();
693  for (u32 j = 0; j < i; ++j)
694  semiringCombo.set(insOne, j, posteriorIds[j], 1.0);
695  semiringCombo.set(delOne, i, posteriorId, 1.0);
696  }
697  ConfusionNetwork::const_iterator itSlot = cn->begin();
698  ConfusionNetwork::iterator itCombinedSlot = combinedCn->begin();
699  for (const_iterator itPair = alignment.begin(), endPair = alignment.end(); itPair != endPair; ++itPair) {
700  if (itPair->first == Core::Type<u32>::max) {
701  // insertion
702  // insert slot
703 
704  itCombinedSlot = combinedCn->insert(itCombinedSlot, *itSlot);
705  if (itCombinedSlot->front().label != Fsa::Epsilon) {
706  ScoresRef combinedScores = combinedSemiring.clone(insOne);
707  combinedScores->set(combinedPosteriorId, refWeight);
708  itCombinedSlot->insert(itCombinedSlot->begin(), ConfusionNetwork::Arc(
709  Fsa::Epsilon, combinedScores,
710  itCombinedSlot->front().begin + itCombinedSlot->front().duration / 2, 0));
711  }
712  else {
713  ConfusionNetwork::Arc& combinedArc = itCombinedSlot->front();
714  ScoresRef subScores = combinedArc.scores;
715  ScoresRef combinedScores = combinedArc.scores = combinedSemiring.clone(insOne);
716  semiringCombo.set(combinedScores, i, subScores);
717  combinedScores->set(combinedPosteriorId, hypWeight * subScores->get(posteriorId) + refWeight);
718  combinedArc.from = combinedArc.to = Core::Type<u32>::max;
719  }
720  // other arcs
721  for (ConfusionNetwork::Slot::iterator itCombinedArc = itCombinedSlot->begin() + 1, endCombinedArc = itCombinedSlot->end();
722  itCombinedArc != endCombinedArc; ++itCombinedArc) {
723  ConfusionNetwork::Arc& combinedArc = *itCombinedArc;
724  ScoresRef subScores = combinedArc.scores;
725  ScoresRef combinedScores = combinedArc.scores = combinedSemiring.clone(subOne);
726  semiringCombo.set(combinedScores, i, subScores);
727  combinedScores->set(combinedPosteriorId, hypWeight * subScores->get(posteriorId));
728  combinedArc.from = combinedArc.to = Core::Type<u32>::max;
729  }
730  ++itSlot;
731  ++itCombinedSlot;
732  }
733  else if (itPair->second == Core::Type<u32>::max) {
734  // deletion
735  // add missing eps mass
736  if (itCombinedSlot->front().label != Fsa::Epsilon) {
737  ScoresRef combinedScores = combinedSemiring.clone(delOne);
738  combinedScores->set(combinedPosteriorId, hypWeight);
739  itCombinedSlot->insert(itCombinedSlot->begin(), ConfusionNetwork::Arc(
740  Fsa::Epsilon, combinedScores,
741  itCombinedSlot->front().begin + itCombinedSlot->front().duration / 2, 0));
742  }
743  else {
744  ScoresRef combinedScores = itCombinedSlot->front().scores;
745  semiringCombo.set(combinedScores, i, posteriorId, 1.0);
746  combinedScores->add(combinedPosteriorId, hypWeight);
747  }
748  ++itCombinedSlot;
749  }
750  else {
751  // substitution
752  ConfusionNetwork::Slot& combinedSlot = *itCombinedSlot;
753  ConfusionNetwork::Slot::iterator itCombinedArc = combinedSlot.begin();
754  for (ConfusionNetwork::Slot::const_iterator itArc = itSlot->begin(), endArc = itSlot->end(); itArc != endArc;) {
755  for (; (itCombinedArc != combinedSlot.end()) && (itCombinedArc->label < itArc->label); ++itCombinedArc)
756  ;
757  if ((itCombinedArc != combinedSlot.end()) && (itCombinedArc->label == itArc->label)) {
758  ScoresRef combinedScores = itCombinedArc->scores;
759  ScoresRef subScores = itArc->scores;
760  Score
761  refPosterior = combinedScores->get(combinedPosteriorId),
762  hypPosterior = hypWeight * subScores->get(posteriorId);
763  Score sumPosterior = refPosterior + hypPosterior;
764  itCombinedArc->begin = Time(
765  Core::round((refPosterior * Score(itCombinedArc->begin) + hypPosterior * Score(itArc->begin)) / sumPosterior));
766  itCombinedArc->duration = Time(
767  Core::round((refPosterior * Score(itCombinedArc->duration) + hypPosterior * Score(itArc->duration)) / sumPosterior));
768  semiringCombo.set(combinedScores, i, subScores);
769  combinedScores->set(combinedPosteriorId, sumPosterior);
770  ++itArc;
771  ++itCombinedArc;
772  }
773  for (; (itArc != endArc) && ((itCombinedArc == combinedSlot.end()) || (itArc->label < itCombinedArc->label)); ++itArc, ++itCombinedArc) {
774  const ConfusionNetwork::Arc& arc = *itArc;
775  ScoresRef combinedScores = combinedSemiring.clone(subOne);
776  semiringCombo.set(combinedScores, i, arc.scores);
777  combinedScores->set(combinedPosteriorId, hypWeight * arc.scores->get(posteriorId));
778  itCombinedArc = combinedSlot.insert(itCombinedArc, ConfusionNetwork::Arc(
779  arc.label,
780  combinedScores,
781  arc.begin, arc.duration));
782  }
783  }
784 
785  ++itSlot;
786  ++itCombinedSlot;
787  }
788  }
789  verify((itCombinedSlot == combinedCn->end()) && (itSlot == cn->end()));
790  }

818  ConstConfusionNetworkRef combine(const ConstConfusionNetworkRefList& cns, ConfusionNetworkAlignmentPtrList* alignments = 0) const {
819  verify(!cns.empty());
820 
821  if (alignments)
822  if (alignments->size() < cns.size() - 1)
823  alignments->resize(cns.size() - 1, 0);
824 
825  weightedCombo_->update(cns, true);
826  ConfusionNetwork* combinedCnPtr = new ConfusionNetwork;
827  if (!cns.front()->isNormalized())
828  Core::Application::us()->criticalError(
829  "Confusion network is not normalized. Confusion network combination requires normalized CNs.");
830  // add proportional posterior mass
831  init(combinedCnPtr, cns.front());
832 
833  verify(verifyCn(*combinedCnPtr, weightedCombo_->weights().front()));
834 
835  ScoreId combinedPosteriorId = combinedCnPtr->normalizedProperties->posteriorId;
836  sumWeights_.resize(cns.size());
837  sumWeights_.front() = weightedCombo_->weights().front();
838  for (u32 i = 1; i < cns.size(); ++i) {
839  ConstConfusionNetworkRef cn = cns[i];
840  if (!cn->isNormalized())
841  Core::Application::us()->criticalError(
842  "Confusion network is not normalized. Confusion network combination requires normalized CNs.");
843  Score weight = weightedCombo_->weights()[i];
844  Score sumWeight = sumWeights_[i] = sumWeights_[i - 1] + weight;
845  // set weights such the two aligned CNs make up a slot-wise posterior distribution,
846  // i.e. alignment score is normalized
847  weightedCostFcn_->setPosteriorIds(combinedPosteriorId, cn->normalizedProperties->posteriorId);
848  weightedCostFcn_->setNorms(sumWeights_[i - 1], 1.0);
849  weightedCostFcn_->setWeights(sumWeights_[i - 1] / sumWeight, weight / sumWeight);
850 
851  /*
852  * Looks not only a little hackish: it is!
853  */
854  const ConfusionNetworkAlignment* alignment = this;
855  if (alignments) {
856  if ((*alignments)[i - 1]) {
857  dbg("CNC: transfer alignment");
858 
859  alignment = (*alignments)[i - 1];
860  }
861  else {
862  align(*combinedCnPtr, *cn);
863  (*alignments)[i - 1] = new ConfusionNetworkAlignment(*this);
864  }
865  }
866  else
867  align(*combinedCnPtr, *cn);
868 
869  // add proportional posterior mass
870  add(*alignment, combinedCnPtr, cn, i);
871 
872  verify(verifyCn(*combinedCnPtr, sumWeight));
873  }
874  return ConstConfusionNetworkRef(combinedCnPtr);
875  }

1016 class RoverCombination : public ConfusionNetworkAlignment, public Core::ReferenceCounted {
1017  typedef ConfusionNetworkAlignment Precursor;
1018 
1019 public:
1020  /**
1021  * NIST/SCLITE Dynamic Programming string alignment:
1022  *
1023  * The DP string alignment algorithm performs a global minimization of a
1024  * Levenshtein distance function which weights the cost of correct words,
1025  * insertions, deletions and substitutions as 0, 3, 3 and 4 respectively.
1026  * The computational complexity of DP is 0(NN).
1027  *
1028  * When evaluating the output of speech recognition systems, the
1029  * precision of generated statistics is directly correlated to the
1030  * reference text accuracy. But uttered words can be coarticulated or
1031  * mumbled to where they have ambiguous transcriptions, (e.i., "what are"
1032  * or "what're"). In order to more accurately represent ambiguous
1033  * transcriptions, and not penalize recognition systems, the ARPA
1034  * community agreed upon a format for specifying alternative reference
1035  * transcriptions. The convention, when used on the case above, allows
1036  * the recognition system to output either transcripts, "what are" or
1037  * "what're", and still be correct.
1038  *
1039  * The case above handles ambiguously spoken words which are loud enough
1040  * for the transcriber to think something should be recognized. For
1041  * mumbled or quietly spoken words, the ARPA community agreed to neither
1042  * penalize systems which correctly recognized the word, nor penalize
1043  * systems which did not. To accommodate this, a NULL word, "@", can be
1044  * added to an alternative reference transcript. For example, "the" is
1045  * often spoken quickly with little acoustic evidence. If "the" and "@"
1046  * are alternates, the recognition system will be given credit for
1047  * outputting "the" but not penalized if it does not.
1048  *
1049  * The presence of alternate transcriptions represents added
1050  * computational complexity to the DP algorithm. Rather than align all
1051  * alternate reference texts to the hypothesis text, then choose the
1052  * lowest error rate alignment, this implementation of DP aligns two word
1053  * networks, thus reducing the computational complexity from 2^(ref_alts +
1054  * hyp_alts) * O(N_ref * N_hyp) to O((N_ref+ref_alts) *
1055  * (N_hyp+hyp_alts)).
1056  *
1057  * For a detailed explanation of DP alignment, see TIME WARPS, STRING
1058  * EDITS, AND MACROMOLECULES: THE THEORY AND PRACTICE OF SEQUENCE
1059  * COMPARISON, by Sankoff and Kruskal, ISBN 0-201-07809-0.
1060  *
1061  * As noted above, DP alignment minimizes a distance function that is applied
1062  * to word pairs. In addition to the "word" alignments which uses
1063  * a distance function defined by static weights, the sclite DP alignment module can
1064  * use two other distance functions. The first, called Time-Mediated alignment
1065  * and the second called Word-Weight-Mediated alignment.
1066  *
1067  *
1068  * Time-Mediated Alignment
1069  *
1070  * Time-Mediated alignment is a variation of DP alignment where
1071  * word-to-word distances are based on the time of occurence for
1072  * individual words. Time-mediated alignments are performed when the '-T'
1073  * option is exercised and the input formats for both the reference and hypothesis
1074  * files are in "ctm" format.
1075  *
1076  * Time-mediated alignments are computed by replacing the
1077  * standard word-to-word distance weights of 0, 3, 3, and 4 with measures
1078  * based on beginning and ending word times. The formulas for
1079  * time-mediated word-to-word distances are:
1080  *
1081  * D(correct) = | T1(ref) - T1(hyp) | + | T2(ref) - T2(hyp) |
1082  * D(insertion) = T2(hyp) - T1(hyp)
1083  * D(deletion) = T2(ref) - T1(ref)
1084  * D(substitution) = | T1(ref) - T1(hyp) | + | T2(ref) - T2(hyp) | + 0.001
1085  * Distance for an Insertion or Deletion of the NULL Token '@' = 0.001
1086  *
1087  * Where,
1088  * T1(x) is the beginning time mark of word x
1089  * T2(x) is the ending time mark of word x
1090  *
1091  **/
1092  class NistScliteCost;
1093  typedef Core::Ref<NistScliteCost> NistScliteCostRef;

1155  virtual Score subCost(u32 refId, u32 hypId) const {
1156  const ConfusionNetwork::Slot &ref = refSlot(refId), &hyp = hypSlot(hypId);
1157  verify(!ref.empty() && (hyp.size() == 1));
1158  Fsa::LabelId hypLabel = hyp.front().label;
1159  Score cost = Infinity;
1160  if (hypLabel == Fsa::Epsilon) {
1161  cost = delCost(refId) - 0.001;
1162  }
1163  else {
1164  Score tmpCost = 0.0;

1442  itArc->scores->set(posteriorId, *itWeight);
1443  }

1447  for (ConfusionNetwork::iterator itSlot = combinedCn->begin(), endSlot = combinedCn->end();
1448  itSlot != endSlot; ++itSlot, ++itConfidences) {

1450  ScoreList::const_iterator beginWeight = weightedCombo_->weights().begin(), itWeight;

1453  itConfidence = beginConfidence;
1454  for (; itConfidence != endConfidence; ++itConfidence, ++itWeight)
1455  norm += *itWeight * *itConfidence;

1462  }

1464  }
