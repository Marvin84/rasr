40  /** Input data f assumed to have N+1 elments:

78 /** Cosine Transform Node

131  needInit_ = true;
132  }
133  }
134  /** Calculates the time domain sample rate of the input.
135  * Motivation:
136  * Since discrete cosine transform does not handle sample rate,
137  * conversion of sample rate is based on the similarity of cosine transform
138  * to inverse Fourier transform. Thus, input is assumed to be in frequency domain and
139  * output will be transformed into time domain. Further assumption is that
140  * input is only the first half of a symmetric function (similar to amplitude spectrum.).
141  * Formulas:
142  * -"N+1" input type: time-sample-rate = 2 * (inputSize - 1) / frequency-sample-rate.
143  * -"even about N-1/2" input type: time-sample-rate = 2 * inputSize / frequency-sample-rate.
144  */
145  f64 timeDomainSampleRate(size_t inputSize);
146

150  shouldWarpDifferenctialUnit_ = should;
151  needInit_ = true;
152  }
153  }
154  void init(size_t inputSize);
155 
156 public:
