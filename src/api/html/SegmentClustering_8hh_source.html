570  /** compute the average (score?) for a given scaling factor alpha */
571  f32 computeAverage(const Math::Matrix<f32>& relativeFeature, const f32 alpha) {
572  u32 dim = relativeFeature.nRows();
573  f32 total = 0.0f;
574  f32 NormFac;
575  Math::Vector<f32> scaledFeature;
576  for (u32 i = 0; i < dim; i++) {
577  scaledFeature = relativeFeature[i] * alpha;
578  NormFac = scaledFeature.logSum();
579  total += exp(scaledFeature[i] - NormFac);
580  }
581  total /= (f32)dim;
582  return (total);
583  }

585  /** compute relative features to be used for alpha optimization */
586  void computeRelativeFeatures(const std::vector<Model>& models, Math::Matrix<f32>& relativeFeature) {

588  Core::WithProgressIndicator progressIndicator("computing relative likelihood features");

599  /** compute alpha scaling value for correlation-based distance scaling */
600  void computeAlpha(std::vector<Model>& models) {
601  f32 alpha = 0.0;
602  f32 step = 0.5f;
603  Math::Matrix<f32> relativeFeature(models.size());
604  f32 average = 0;
605 
606  // compute the relative features to be used for alpha optimization
607  computeRelativeFeatures(models, relativeFeature);
608 
609  // optimize the best alpha value
610  while (((average < alphaMin_) || (average > alphaMax_)) && !(Core::isAlmostEqualUlp(step, 0.0f, 100))) {
611  if (infoChannel_.isOpen()) {
612  infoChannel_ << Core::XmlEmpty("compute-alpha") + Core::XmlAttribute("alpha", alpha) + Core::XmlAttribute("average", average) + Core::XmlAttribute("alphaMin", alphaMin_) + Core::XmlAttribute("alphaMax", alphaMax_) + Core::XmlAttribute("step", step);
613  }
614  if (average < alphaMin_) {
615  alpha += step;
616  }
617  else {
618  alpha -= step;
619  }
620  step /= 2.0;
621  average = computeAverage(relativeFeature, alpha);
622  }
623  if ((average < alphaMin_) || (average > alphaMax_)) {
624  warning("Clustering goals with current alpha settings not reachable, computation of alpha failed: alpha=%lf, average=%lf, alphaMin_=%lf, alphaMax_=%lf, step=%lf",
625  alpha, average, alphaMin_, alphaMax_, step);
626  }
627  else {
628  if (infoChannel_.isOpen()) {
629  infoChannel_ << Core::XmlEmpty("compute-alpha") + Core::XmlAttribute("alpha", alpha) + Core::XmlAttribute("average", average) + Core::XmlAttribute("alphaMin", alphaMin_) + Core::XmlAttribute("alphaMax", alphaMax_) + Core::XmlAttribute("step", step);
630  }
631  }
632 
633  // update the scaling factor
634  for (u32 i = 0; i < models.size(); i++)
635  models[i].reset(alpha);
636  alpha_ = alpha;
637  }

766  /** find the best segment index w.r.t. the BIC scores */
767  const u32 findBestBIC() const {
768  s32 bestIdx = -1;
769  f32 bestScore = 0;
770  f32 currentScore = 0;
771  u32 stop = std::max(nrsegments_ - mincluster_, (u32)0);
772  if (nrsegments_ > maxcluster_)
773  bestIdx = nrsegments_ - maxcluster_ - 1;
774 
775  for (s32 i = bestIdx + 1; i < (s32)stop; i++) {
776  currentScore += mergeTracker_[i].score();
777  if (currentScore < bestScore) {
778  bestScore = currentScore;
779  bestIdx = i;
780  if (infoChannel_.isOpen()) {
781  infoChannel_ << Core::XmlEmpty("find-best-BIC") + Core::XmlAttribute("score", bestScore) + Core::XmlAttribute("cluster", bestIdx) + Core::XmlAttribute("min-cluster", mincluster_) + Core::XmlAttribute("max-cluster", maxcluster_) + Core::XmlAttribute("segments", nrsegments_) + Core::XmlAttribute("stop", stop);
782  }

784  }

790  stopIdx = findBestBIC();

792 
793  for (s32 i = 0; i <= stopIdx; i++) {
794  matrixTracker_[mergeTracker_[i].rightIdx()] = 0;
795  classid_[mergeTracker_[i].rightIdx()] = mergeTracker_[i].leftIdx();
796  }
797 
798  for (u32 i = nrsegments_ - 1; i > 0; i--) {
799  u32 j = i;
800  while (matrixTracker_[j] == 0)
801  j = classid_[j];
802  classid_[i] = j;

808  void clusterUpdate(std::vector<Model>& models) {
809  if (predefinedClusters_[parent_] >= 0)
810  for (child_ = parent_ + 1; child_ < predefinedClusters_.size(); child_++)
811  if (predefinedClusters_[child_] == predefinedClusters_[parent_]) {
812  predefinedClusters_[child_] = -1;
813  condenseDistMatrix(models);
814  }

818  u32 tmpMaxCluster = maxcluster_;
819  maxcluster_ = 1;
820  for (parent_ = 0; parent_ < predefinedClusters_.size(); parent_++)
821  clusterUpdate(models);
822  maxcluster_ = tmpMaxCluster;
823  }

825  std::vector<Model> filterModelsByFrames(const std::vector<Model>& models) {
826  std::vector<Model> newModels;
827  std::vector<std::string> newSegmentIds;
828  for (u32 i = 0; i < models.size(); ++i) {
829  if (models[i].nFrames() >= minframes_) {
830  newModels.push_back(models[i]);
831  newSegmentIds.push_back(segmentIds_[i]);
832  }
833  else {
834  defaultIds_.push_back(segmentIds_[i]);
835  }
836  }
837  segmentIds_ = newSegmentIds;
838  return newModels;
839  }
840 
841  /** clustering of the finalized models */
842  bool cluster(std::vector<Model>& models) {
843  models = filterModelsByFrames(models);
844 
845  initSegmentClustering(models);
846  if (predefinedClusters_.size() == models.size())
847  integratePredefinedClusters(models);
848  findBestPair();
849  while (nrcluster_ > 1) {
850  if (infoChannel_.isOpen()) {
851  infoChannel_ << Core::XmlEmpty("clustering ") + Core::XmlAttribute("number-of-clusters", nrcluster_);
852  }
853  condenseDistMatrix(models);
854  findBestPair();
855  }
856  returnClustering();
857  return (0);
858  }
