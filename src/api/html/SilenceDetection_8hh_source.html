204  bool send(Frame& out) {
205  if (!out)
206  return false;
207  if (out.silence() == unsure)
208  criticalError("Unsure frame");
209 
210  Flow::Vector<Float>* s = new Flow::Vector<Float>;
211  s->push_back((f32)(out.silence() == silence));
212  putData(1, s);
