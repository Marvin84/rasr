26 std::string Core::directoryName(const std::string& path) {
27  std::string result(path);
28  std::string::size_type i = result.find_last_of("/"), j;
29  if (i == std::string::npos) {
30  j = 0; // no slash at all -> no dir
31  }
32  else {
33  j = result.find_last_not_of("/", i);
34  if (j == std::string::npos) {
35  j = 1; // only leading slashed -> remove everything except for a signle leading slash
36  }
37  else {
38  j += 1; // remove all trailing slashes
39  }
40  }
41  result.erase(j);
42  return result;
43 }

67 std::string Core::normalizePath(const std::string& path) {
68  if (path == "")
69  return ".";
70 
71  size_t nInitialSlashes = 0;
72  while ((path.size() > nInitialSlashes) && (path[nInitialSlashes] == '/'))
73  ++nInitialSlashes;
74  if (nInitialSlashes > 2)
75  nInitialSlashes = 1;
76 
77  StringTokenizer components(path, "/");
78  std::vector<std::string> newComponents;
79  for (StringTokenizer::Iterator comp = components.begin(); comp != components.end(); ++comp) {
80  std::string component(*comp);
81  if (component.size() == 0 || component == ".")
82  continue;
83  else if ((component == "..") && !(newComponents.size() == 0 || newComponents.back() == ".."))
84  newComponents.pop_back();
85  else
86  newComponents.push_back(component);
87  }
88 
89  std::string result(nInitialSlashes, '/');
90  for (std::vector<std::string>::const_iterator c = newComponents.begin(); c != newComponents.end(); ++c) {
91  if (c != newComponents.begin())
92  result += "/";
93  result += *c;
94  }
95 
96  return result;
97 }

234 struct dirent* DirectoryFileIterator::nextEntry() {
235  struct dirent* result;
236  do {
237  result = readdir(directoryStack_.top().second);
238  // check wether d_name is one of the default directory names which
239  // we should avoid to trace due to recursion
240  } while (result != 0 &&
241  (strcmp(result->d_name, ".") == 0 ||
242  strcmp(result->d_name, "..") == 0));
243  if (result) {
244  path_ = joinPaths(directoryStack_.top().first, result->d_name);
245  if (stat64(joinPaths(base_, path_).c_str(), &state_) != 0)
246  perror(joinPaths(base_, path_).c_str());
247  }
248  return result;
