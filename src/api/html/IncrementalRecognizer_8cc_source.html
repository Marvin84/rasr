96  std::pair<std::deque<Word>, u32> parseForwardLattice(Flf::ConstLatticeRef lattice) {
97  std::deque<Word> ret;
98  u32 length = Core::Type<u32>::max;
99  Fsa::StateId state = lattice->initialStateId();
100  forwardScore_ = 0;
101  while (state != Fsa::InvalidStateId) {
102  Ftl::Automaton<Semiring, State>::ConstStateRef s = lattice->getState(state);
103  if (s->nArcs()) {
104  verify(s->nArcs() == 1); // Lattice-handling not supported ATM
105 
106  Word word;
107  word.amScore = s->getArc(0)->score(lattice->semiring()->id("am")) * lattice->semiring()->scale(lattice->semiring()->id("am"));
108  word.lmScore = s->getArc(0)->score(lattice->semiring()->id("lm")) * lattice->semiring()->scale(lattice->semiring()->id("lm"));
109 
110  const Bliss::LemmaPronunciation* pron = lpAlphabet_->lemmaPronunciation(s->getArc(0)->output());
111  if (verboseRefinement_) {
112  if (pron)
113  std::cout << "forward pron " << pron->lemma()->name() << " score " << word.amScore << " + " << word.lmScore << " = " << word.amScore + word.lmScore << std::endl;
114  else
115  std::cout << "forward anon score " << word.amScore << " + " << word.lmScore << " = " << word.amScore + word.lmScore << std::endl;
116  }
117  forwardScore_ += word.amScore + word.lmScore;
118  if (pron &&
119  (includeNoise_ || isRealWord(pron))) {
120  word.pron = pron;
121  word.start = lattice->boundary(state).time();
122  word.end = lattice->boundary(s->getArc(0)->target()).time();
123  if (word.end > word.start)
124  word.end -= 1;
125  word.originState = state;
126  ret.push_back(word);
127  }
128  state = s->getArc(0)->target();
129  length = lattice->boundary(state).time();
130  }
131  else {
132  state = Fsa::InvalidStateId;
133  }
134  }
135  if (verboseRefinement_)
136  std::cout << "forward total score " << forwardScore_ << std::endl;
137  verify(length != Core::Type<u32>::max);
138  if (length > 0)
139  length -= 1; // The sentence-end is always hypothesized one frame after the last hypothesis. That additional frame doesn't count.
140  return std::make_pair(ret, length);
141  }

151  backwardScore_ = 0;

153  Ftl::Automaton<Semiring, State>::ConstStateRef s = lattice->getState(state);
154  if (s->nArcs()) {
155  verify(s->nArcs() == 1); // Lattice-handling not supported ATM
156 
157  Word word;
158  word.amScore = s->getArc(0)->score(lattice->semiring()->id("am")) * lattice->semiring()->scale(lattice->semiring()->id("am"));
159  word.lmScore = s->getArc(0)->score(lattice->semiring()->id("lm")) * lattice->semiring()->scale(lattice->semiring()->id("lm"));
160 
161  const Bliss::LemmaPronunciation* pron = lpAlphabet_->lemmaPronunciation(s->getArc(0)->output());
162  if (pron && verboseRefinement_)
163  std::cout << "backward pron " << pron->lemma()->name() << " score " << word.amScore + word.lmScore << std::endl;
164 
165  backwardScore_ += word.amScore + word.lmScore;
166 
167  if (pron &&
168  (includeNoise_ || isRealWord(pron))) {
169  word.pron = pron;
170  word.end = segmentLength_ - 1 - lattice->boundary(state).time();
171  s32 start = ((s32)segmentLength_) - 1 - lattice->boundary(s->getArc(0)->target()).time();
172  if (start < (s32)word.end)
173  start += 1;
174  word.start = std::max(0, start);
175  word.originState = state;
176  ret.push_front(word);
177  }
178  state = s->getArc(0)->target();
179  }
180  else {
181  state = Fsa::InvalidStateId;
182  }
183  }
184  if (verboseRefinement_)
185  std::cout << "backward total score " << backwardScore_ << std::endl;
186  return ret;

222  void align() {
223  // Align forward/backward result
224  {
225  s32 currentForWord = 0, currentBackWord = 0;
226  while (currentForWord < (s32)forWords.size() && currentBackWord < (s32)backWords.size()) {
227  const Word& backWord(backWords[currentBackWord]);
228  const Word& forWord(forWords[currentForWord]);
229  if (backWord.pron == forWord.pron &&
230  backWord.intersects(forWord) &&
231  (!forceEqualAlignment_ || backWord.equals(forWord)) &&
232  (!correctForceEqualScore_ || std::abs(backWord.amScore - forWord.amScore) < 0.2 ||
233  ((backWord.lmScore == 0 || forWord.lmScore == 0) && backWord.lmScore != forWord.lmScore))) // ignore AM score if the LM score was 'overflown' into the AM
234  {
235  forWordAlignment.insert(std::make_pair(currentForWord, currentBackWord));
236  backWordAlignment.insert(std::make_pair(currentBackWord, currentForWord));
237  ++currentForWord;
238  ++currentBackWord;
239  }
240  else {
241  if (backWord.start < forWord.start) {
242  badBackWords.insert(currentBackWord);
243  ++currentBackWord;
244  }
245  else {
246  badForWords.insert(currentForWord);
247  ++currentForWord;
248  }
249  }
250  }
251 
252  while (currentForWord < (s32)forWords.size()) {
253  badForWords.insert(currentForWord);
254  ++currentForWord;
255  }
256 
257  while (currentBackWord < (s32)backWords.size()) {
258  badBackWords.insert(currentBackWord);
259  ++currentBackWord;
260  }
261  }
262  errorRate_ = (forWords.size() + backWords.size() - 2 * forWordAlignment.size()) / (f32)(forWords.size() + backWords.size());
263  }

265  void normalize() {
266  bool changed = true;
267  // Grow ranges for refinement: Extend the badness ranges where the boundary time of surrounding words is unequal
268  while (changed) {
269  changed = false;
270  for (std::set<s32>::const_iterator it = badForWords.begin(); it != badForWords.end(); ++it) {
271  if (forWordAlignment.count(*it - 1)) {
272  if (forWords[*it - 1].start != backWords[forWordAlignment[*it - 1]].start || !isRealWord(forWords[*it - 1].pron)) {
273  invalidateForWord(*it - 1);
274  changed = true;
275  break;
276  }
277  }
278  if (forWordAlignment.count(*it + 1)) {
279  if (forWords[*it + 1].end != backWords[forWordAlignment[*it + 1]].end || !isRealWord(forWords[*it + 1].pron)) {
280  invalidateForWord(*it + 1);
281  changed = true;
282  break;
283  }
284  }
285  }
286  for (std::set<s32>::const_iterator it = badBackWords.begin(); it != badBackWords.end(); ++it) {
287  if (backWordAlignment.count(*it - 1)) {
288  if (backWords[*it - 1].start != forWords[backWordAlignment[*it - 1]].start || !isRealWord(backWords[*it - 1].pron)) {
289  invalidateBackWord(*it - 1);
290  changed = true;
291  break;
292  }
293  }
294  if (backWordAlignment.count(*it + 1)) {
295  if (backWords[*it + 1].end != forWords[backWordAlignment[*it + 1]].end || !isRealWord(backWords[*it + 1].pron)) {
296  invalidateBackWord(*it + 1);
297  changed = true;
298  break;
299  }
300  }
301  }
302  // Remove intermediate alignment sequences which are shorter than minimumGoodLength, because the n-gram requires this (forwards)
303  {
304  s32 sequenceStart = -1;
305  for (s32 forWord = 0; forWord < (s32)forWords.size(); ++forWord) {
306  if (forWordAlignment.count(forWord)) {
307  if (sequenceStart == -1)
308  sequenceStart = forWord;
309  }
310  else {
311  // This intentionally doesn't trigger on a sequence which goes until the end or which starts at the beginning
312  if (sequenceStart > 0) {
313  u32 realWords = 0;
314  for (s32 q = sequenceStart; q < forWord; ++q)
315  if (isRealWord(forWords[q].pron))
316  ++realWords;
317  if (realWords < minimumGoodLength_) {
318  // The previous sequence is not the initial sequence, and not the last sequence. But it is too short.
319  for (s32 q = sequenceStart; q < forWord; ++q)
320  invalidateForWord(q);
321  changed = true;
322  }
323  }
324  sequenceStart = -1;
325  }
326  }
327  }
328  {
329  // Remove intermediate alignment sequences which are shorter than minimumGoodLength (backwards)
330  s32 sequenceStart = -1;
331  for (s32 backWord = 0; backWord < (s32)backWords.size(); ++backWord) {
332  if (backWordAlignment.count(backWord)) {
333  if (sequenceStart == -1)
334  sequenceStart = backWord;
335  }
336  else {
337  // This intentionally doesn't trigger a sequence which goes until the end
338  if (sequenceStart > 0 && backWord - sequenceStart < (s32)minimumGoodLength_) {
339  // The previous sequence is not the initial sequence, and not the last sequence. But it is too short.
340  for (s32 q = sequenceStart; q < backWord; ++q)
341  invalidateBackWord(q);
342  changed = true;
343  }
344  sequenceStart = -1;
345  }
346  }
347  }
348  }

686  /*
687  * In case of a valid label id:
688  * Before:
689  * am_score = emission_scale * emission + transition_scale * transition
690  * am_scale = 1.0
691  * lm_score = pronunciation_scale * pronunciation + lm_scale * lm
692  * lm_scale = 1.0
693  * Afterwards:
694  * am_score = emission_scale * emission + transition_scale * transition + pronunciation_scale * pronunciation
695  * am_scale = 1.0
696  * lm_score = lm
697  * lm_scale = lm_scale
698  */
699  ScoresRef buildScore(Fsa::LabelId label, Score amRecogScore, Score lmRecogScore) {
700  verify(amRecogScore != Semiring::Zero);
701  verify(lmRecogScore != Semiring::Zero);
702 
703  Score amScore, pronScore, lmScore;
704  if ((Fsa::FirstLabelId <= label) && (label <= Fsa::LastLabelId)) {
705  amScore = amRecogScore;
706  pronScore = lpAlphabet_->lemmaPronunciation(label)->pronunciationScore();
707  verify(pronScore != Semiring::Zero);
708  lmScore = (lmRecogScore - pronScale_ * pronScore) / lmScale_;
709  }
710  else {
711  amScore = amRecogScore;
712  pronScore = Semiring::One;
713  lmScore = lmRecogScore / lmScale_;
714  }
715 
716  ScoresRef scores = semiring_->create();
717  if (addPronunciationScores_) {
718  scores->set(amId_, amScore);
719  scores->set(pronunciationId_, pronScore);
720  }
721  else
722  scores->set(amId_, amScore + pronScale_ * pronScore);
723  scores->set(lmId_, lmScore);
724  if (addConfidenceScores_)
725  scores->set(confidenceId_, 0.0);
726  verify(semiring_->project(scores) != Semiring::Zero);
727 
728  return scores;
729  }

738  std::pair<Flf::ConstLatticeRef, Flf::ConstLatticeRef> recognizeFeatures(const std::deque<FeatureRef>& features) {
739  delayedRecognition_->reset();
740  recognizer_->restart();
741  for (std::deque<FeatureRef>::const_iterator it = features.begin(); it != features.end(); ++it)
742  delayedRecognition_->add(*it);
743  while (delayedRecognition_->flush())
744  ;
745  Core::Ref<const Search::LatticeAdaptor> recognizerLattice = recognizer_->getCurrentWordLattice();
746  Flf::ConstLatticeRef ret = buildLattice(recognizerLattice, true);
747  Flf::ConstLatticeRef meshLattice;
748  if (meshRescoring_) {
749  Search::SearchAlgorithm::RecognitionContext c = recognizer_->setContext(Search::SearchAlgorithm::RecognitionContext());
750  recognizer_->setContext(c);
751  meshLattice = mesh(ret);
752  if (meshCombination_)
753  ret = best(ret, BellmanFord);
754  else
755  ret = best(decodeRescoreLm(meshLattice,
756  mc_->languageModel(),
757  recognizer_->describePruning()->masterBeam(),
758  rescoreWordEndLimit_, c.prefix, c.suffix),
759  BellmanFord);
760  if (expandTransits_)
761  meshLattice = expandTransits(meshLattice, c.prePhon, c.sufPhon);
762  }
763  return std::make_pair(ret, meshLattice);
764  }

909  Lm::addLemmaScore(backwardRecognizer_->mc_->languageModel(),
910  it->pron->lemma(),
911  backwardRecognizer_->mc_->languageModel()->scale(),
912  forwardHypothesisBackwardHistory,
913  forwardHypothesisBackwardScore);

924  *it,

926  forwardHypothesisBackwardHistory,
927  forwardHypothesisBackwardScore);
928  }

931

933 
934  Score backwardHypothesisForwardScore = basicForwardScore;
935  Lm::History backwardHypothesisForwardHistory = basicForwardHistory;
936 
937  for (std::deque<ForwardBackwardAlignment::Word>::const_iterator it = alignment.backWords.begin(); it != alignment.backWords.end(); ++it) {

939  Lm::addLemmaScore(mc_->languageModel(),
940  it->pron->lemma(),
941  mc_->languageModel()->scale(),
942  backwardHypothesisForwardHistory,
943  backwardHypothesisForwardScore);
944

946  std::cout << "forward LM score component for "
947  << it->pron->lemma()->preferredOrthographicForm()
948  << ": "
949  << backwardHypothesisForwardScore - old
950  << std::endl;
951  }
952  }
953

955  Lm::addLemmaScore(mc_->languageModel(), *it, mc_->languageModel()->scale(), backwardHypothesisForwardHistory, backwardHypothesisForwardScore);
956 
957  backwardHypothesisForwardScore += mc_->languageModel()->sentenceEndScore(backwardHypothesisForwardHistory);
958 
959  // Step 4: Compute the backward score for the backward hypothesis
960 
961  Lm::History backwordHypothesisBackwardHistory = basicBackwardHistory;

964  for (std::deque<ForwardBackwardAlignment::Word>::const_reverse_iterator it = alignment.backWords.rbegin(); it != alignment.backWords.rend(); ++it) {

968  backwardRecognizer_->mc_->languageModel()->scale(),

995  *it,

997  backwordHypothesisBackwardHistory,
998  backwardHypothesisBackwardScore);

1008  std::cout << "backward hypothesis: forward LM score: " << backwardHypothesisForwardScore << " backward LM score: " << backwardHypothesisBackwardScore << std::endl;

1034  std::cout << "forward hypothesis: forward sentence end LM score " << mc_->languageModel()->sentenceEndScore(forwardHypothesisForwardHistory) << std::endl;
1035  std::cout << "forward hypothesis: backward sentence end LM score " << backwardRecognizer_->mc_->languageModel()->sentenceEndScore(forwardHypothesisBackwardHistory) << std::endl;
1036  std::cout << "backward hypothesis: forward LM score: " << backwardHypothesisForwardScore << " backward LM score: " << backwardHypothesisBackwardScore << std::endl;

1080  backwardRecognizer_->recognizer_->relaxPruning(adaptRelaxPruningFactor_, adaptRelaxPruningOffset_);
1081  }

1767  std::pair<ConstLatticeRef, ConstSegmentRef> buildLatticeAndSegment(Core::Ref<const Search::LatticeAdaptor> la) {
1768  verify(segment_);
1769 
1770  Speech::TimeframeIndex startTime;
1771  {
1772  Flf::LatticeHandler* handler = Module::instance().createLatticeHandler(config);
1773  ::Lattice::ConstWordLatticeRef lattice = la->wordLattice(handler);
1774  Core::Ref<const ::Lattice::WordBoundaries> boundaries = lattice->wordBoundaries();
1775  Fsa::ConstAutomatonRef amFsa = lattice->part(::Lattice::WordLattice::acousticFsa);
1776  startTime = (*boundaries)[amFsa->initialStateId()].time();
1777  }
1778 
1779  ConstLatticeRef partialLattice = buildLattice(la, true);
1780 
1781  Fsa::StateId endState = partialLattice->initialStateId();
1782  while (partialLattice->getState(endState)->nArcs())
1783  endState = partialLattice->getState(endState)->getArc(0)->target();
1784  Speech::TimeframeIndex endTime = partialLattice->boundary(endState).time() + startTime;
1785 
1786  log() << "got partial lattice for interval " << startTime << " -> " << endTime;
1787 
1788  verify(startTime < featureTimes_.size());
1789  if (endTime >= featureTimes_.size()) {
1790  log() << "end-time is too high: " << endTime << " max. " << featureTimes_.size() - 1 << ", truncated!";
1791  endTime = featureTimes_.size() - 1;
1792  }
1793 
1794  verify(startTime < endTime);
1795  verify(endTime < featureTimes_.size());
1796 
1797  SegmentRef newSegment(new Flf::Segment(segment_));
1798  newSegment->setOrthography("");
1799  newSegment->setStartTime(featureTimes_[startTime].startTime());
1800  newSegment->setEndTime(featureTimes_[endTime].endTime());
1801  verify(newSegment->segmentId().size());
1802  {
1803  std::string::size_type timeStart = newSegment->segmentId().rfind("_");
1804  std::string::size_type timeGap = newSegment->segmentId().rfind("-");
1805  std::ostringstream os;
1806  if (timeStart != std::string::npos && timeGap != std::string::npos && timeGap > timeStart) {
1807  // Create a new segment name with corrected time information in the identifier
1808  os << newSegment->segmentId().substr(0, timeStart + 1);
1809  os << std::setiosflags(std::ios::fixed) << std::setprecision(3) << newSegment->startTime() << "-" << newSegment->endTime();
1810  }
1811  else {
1812  // Create a new segment name by appending "_$subsegment"
1813  os << newSegment->segmentId() << "_" << subSegment_;
1814  }
1815  newSegment->setSegmentId(os.str());
1816  }
1817  log() << "created segment " << newSegment->segmentId();
1818  info(partialLattice, clog());
1819  subSegment_ += 1;
1820  return std::make_pair(partialLattice, newSegment);
1821  }

1829

1831  return ret;

1834  std::deque<FeatureRef> features;
1835  {
1836  FeatureRef feature;
1837  while (getData(feature))
1838  features.push_back(feature);
1839  }
1840

1843  if (preCacheAllFrames_) {

1845  PreCacher()
1846  : SearchAlgorithm(Core::Configuration()), Core::Component(Core::Configuration()) {}
1847  virtual void feed(const Mm::FeatureScorer::Scorer& scorer) {
1848  dynamic_cast<const Mm::CachedFeatureScorer::CachedContextScorerOverlay*>(scorer.get())->precache();
1849  }

1851  virtual Core::Ref<const Search::LatticeAdaptor> getCurrentWordLattice() const {}
1852  virtual void logStatistics() const {}

1856  virtual bool setModelCombination(const Speech::ModelCombination& modelCombination) {}

1859  Speech::RecognizerDelayHandler handler(&precacher, acousticModel_, contextScorerCache_);
1860  for (std::deque<FeatureRef>::iterator it = features.begin(); it != features.end(); ++it)
1861  handler.add(*it);
1862  while (handler.flush())
1863  ;
1864  }
1865 
1866  f32 preCachingTime = globalTimer_.user();
1867 
1868  segmentFeatureCount_ = features.size();
1869 
1870  Search::SearchAlgorithm::RecognitionContext refineContext;
1871  if (meshRescoring_)
1872  refineContext.latticeMode = Search::SearchAlgorithm::RecognitionContext::Yes;
1873  else
1874  refineContext.latticeMode = Search::SearchAlgorithm::RecognitionContext::No;
1875 
1876  Search::SearchAlgorithm::RecognitionContext oldForwardContext, oldBackwardContext;
1877  oldForwardContext = recognizer_->setContext(refineContext);
1878  oldBackwardContext = backwardRecognizer_->recognizer_->setContext(refineContext);
1879 
1880  bool separateLatticeRecognition = latticeRelaxPruningFactor_ > 1.0 || latticeRelaxPruningOffset_ > 0.0;
1881 
1882  if (separateLatticeRecognition) {
1883  // Due to word boundary crossings, skips can make the forward and backward models unequal,
1884  // which can make the forward-backward alignment fail forever. When we anyway add an addition
1885  // recognition pass to generate lattices, then we can completely disable skips during the forward/backward
1886  // search, as we do here. Otherwise, it is the users option, because the disabling of skips would change the models.
1887  recognizer_->setAllowHmmSkips(false);
1888  backwardRecognizer_->recognizer_->setAllowHmmSkips(false);
1889  }
1890 
1891  RecognizedSequence refined = refine(0, features, 0, 0, std::vector<const Bliss::Lemma*>(), std::vector<const Bliss::Lemma*>(), true);
1892 
1893  if (separateLatticeRecognition) {
1894  recognizer_->setAllowHmmSkips(true);
1895  backwardRecognizer_->recognizer_->setAllowHmmSkips(true);
1896  }
1897 
1898  ret = std::make_pair(refined.singleBestLattice, SegmentRef(new Flf::Segment(segment_)));
1899 
1900  verify(refined.pruning.get());
1901 
1902  std::cout << "used pruning (" << features.size() << " frames): " << refined.pruning->format() << std::endl;
1903  log() << "used pruning (" << features.size() << " frames): " << refined.pruning->format();
1904 
1905  f32 refinementTime = globalTimer_.user() - preCachingTime;
1906  f32 postProcessingStartTime = globalTimer_.user();
1907 
1908  if (separateLatticeRecognition) {
1909  refined.meshEntries.clear();
1910  Search::SearchAlgorithm::RecognitionContext latticeContext;
1911  latticeContext.latticeMode = Search::SearchAlgorithm::RecognitionContext::Yes;
1912  Search::SearchAlgorithm::PruningRef latticePruning = refined.pruning->clone();
1913  bool extended = latticePruning->extend(latticeRelaxPruningFactor_, latticeRelaxPruningOffset_, latticeRelaxPruningInterval_);
1914  verify(extended);
1915  std::cout << "generating lattice for whole segment with pruning: " << latticePruning->format() << std::endl;
1916  log() << "used extended pruning for lattice-generation (" << features.size() << " timeframes): " << latticePruning->format();
1917  Search::SearchAlgorithm::PruningRef oldForwardPruning = recognizer_->describePruning();
1918  recognizer_->resetPruning(latticePruning);
1919  recognizer_->setContext(latticeContext);
1920 
1921  ret.first = recognizeFeatures(features).first;
1922 
1923  if (meshRescoring_) {
1924  MeshEntry e;
1925  e.lattice = ret.first;
1926  refined.meshEntries.push_back(e);
1927  }
1928  else {
1929  u32 i = 0;
1930  while (true) {
1931  ++i;
1932  postProcessingStartTime = globalTimer_.user();
1933  ret.first = postProcess(ret.first);
1934  if (minArcsPerSecond_) {
1935  LatticeCounts counts = count(ret.first);
1936  f32 minArcs = (minArcsPerSecond_ * features.size()) / 100.0;
1937  if (counts.nArcs_ < minArcs) {
1938  if (i > maxLatticeRegenerations_) {
1939  log() << "not enough arcs: "
1940  << counts.nArcs_
1941  << " need at least "
1942  << minArcs
1943  << ", but NOT regenerating lattice because maximum number of regenerations is already reached (" << i << ")";
1944  break;
1945  }
1946  else {
1947  log() << "not enough arcs: " << counts.nArcs_ << " need at least " << minArcs;
1948  if (!recognizer_->relaxPruning(relaxPruningFactor_, relaxPruningOffset_)) {
1949  log() << "FAILED relaxing pruning for regeneration";
1950  break;
1951  }
1952  else {
1953  latticePruning = latticePruning->clone();
1954  latticePruning->extend(relaxPruningFactor_, relaxPruningOffset_, 0);
1955  recognizer_->resetPruning(latticePruning);
1956  log() << "regenerating lattice with extended pruning: " << latticePruning->format();
1957  ret.first = recognizeFeatures(features).first;
1958  }
1959  }
1960  }
1961  else {
1962  break;
1963  }
1964  }
1965  else {
1966  break;
1967  }
1968  }
1969  }
1970 
1971  recognizer_->resetPruning(oldForwardPruning);
1972 
1973  if (meshRescoring_ && forceForwardBackwardLattices_) {
1974  MeshEntry e;
1975  e.lattice = recognizeFeatures(std::deque<Flf::FeatureRef>(features.rbegin(), features.rend())).first;
1976  e.reverseOffset = features.size();
1977  refined.meshEntries.push_back(e);
1978  }
1979 
1980  info(ret.first, clog());
1981  }
1982 
1983  if (meshRescoring_ && !refined.meshEntries.empty()) {
1984  log() << "building mesh from " << refined.meshEntries.size() << " individual sub-meshes";
1985  ret.first = mesh(refined.meshEntries);
1986  log() << "rescoring with beam " << refined.pruning->maxMasterBeam();
1987  std::cout << "rescoring with beam " << refined.pruning->maxMasterBeam() << std::endl;
1988  ret.first = decodeRescoreLm(ret.first, mc_->languageModel(), refined.pruning->maxMasterBeam(), rescoreWordEndLimit_);
1989  ret.first = postProcess(ret.first);
1990  }
1991 
1992  recognizer_->setContext(oldForwardContext);
1993  backwardRecognizer_->recognizer_->setContext(oldBackwardContext);
1994 
1995  globalTimer_.stop();
1996 
1997  if (features.size()) {
1998  std::cout << "global needed time: " << globalTimer_.user()
1999  << " for frame-duration: " << (features.size() / 100.0)
2000  << " global RTF: " << globalRtf()
2001  << " (postprocessing RTF " << (globalTimer_.user() - postProcessingStartTime) / (features.size() / 100.0)
2002  << " forward-backward RTF " << refinementTime / (features.size() / 100.0)
2003  << ", precaching RTF " << preCachingTime / (features.size() / 100.0) << ")" << std::endl;
2004  log() << "global needed time: " << globalTimer_.user()
2005  << " for frame-duration: " << (features.size() / 100.0)
2006  << " global RTF: " << globalRtf()
2007  << " (postprocessing RTF " << (globalTimer_.user() - postProcessingStartTime) / (features.size() / 100.0)
2008  << " forward-backward RTF " << refinementTime / (features.size() / 100.0)
2009  << ", precaching RTF " << preCachingTime / (features.size() / 100.0) << ")";
2010  }
2011 
2012  finishRecognition();
2013  backwardRecognizer_->finishRecognition();
2014  return ret;
2015  }

2018  timer.start();

2030

2035  << " for frame-duration: " << (featureTimes_.size() / 100.0)
2036  << " global RTF: " << timer.user() / (featureTimes_.size() / 100.0) << std::endl;
2037 
2038  if (subSegment_) {
2039  ret = buildLatticeAndSegment(recognizer_->getCurrentWordLattice());
2040  finishRecognition();
2041  return ret;
2042  }
2043  else {
2044  Core::Ref<const Search::LatticeAdaptor> recognizerLattice = recognizer_->getCurrentWordLattice();

2046 
2047  info(ret.first, clog());
2048  processResult();
2049  finishRecognition();
