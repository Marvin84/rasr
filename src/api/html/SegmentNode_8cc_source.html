92 bool SegmentwiseFeaturesNode::work(Flow::PortId p) {
93  if (!Precursor::work(p)) {
94  return false;
95  }
96  Flow::DataPtr<Feature::FlowFeature> in;
97  Flow::DataAdaptor<ConstSegmentwiseFeaturesRef>* out = new Flow::DataAdaptor<ConstSegmentwiseFeaturesRef>();
98  out->invalidateTimestamp();
99  SegmentwiseFeaturesRef features(new SegmentwiseFeatures);
100  bool firstFeature = true;
101  while (getData(1, in)) {
102  Core::Ref<Feature> feature(new Feature(in));
103  if (firstFeature) {
104  if (!noDependencyCheck_)
105  checkFeatureDependencies(*feature);
106  firstFeature = false;
107  }
108  features->push_back(feature);
109  out->expandTimestamp(feature->timestamp());
110  }
111  if (features->empty()) {
112  out->invalidateTimestamp();
113  return false;
114  }
115  else {
116  out->data() = features;
117  return putData(0, out) && putData(0, in.get());
118  }
119 }
