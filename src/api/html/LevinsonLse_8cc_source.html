26 /* Changes compared to original Rabiner-Schafer version:
27  * -Initialisation changed due to Hermansky/Morgan:
28  * Instead of starting the loop with i = 1,
29  * avoid dividing by square energy in E_[1] = (1 + k(1)^2) * E(0) = (1 + R(1)^2 / E(0)^2) * E(0)
30  * yeilds in E_[1] = (R(0) + R(1) * k(1). Afterwords start loop with i = 2.

34 bool LevinsonLeastSquares::work(const std::vector<InputData>& R) {
35  require(!R.empty());
36 
37  setOrder(R.size() - 1);
38  size_t i, j;
39  size_t N = order();
40 
41  E_[0] = R[0];
42  if (Core::isAlmostEqual(E_[0], (Data)0.0))
43  return false;
44 
45  alpha_[1][1] = k_[1] = -R[1] / R[0];
46  E_[1] = R[0] + R[1] * k_[1];
47  for (i = 2; i <= N; i++) {
48  // k[i]
49  k_[i] = R[i];
50  for (j = 1; j <= (i - 1); j++)
51  k_[i] += alpha_[j][i - 1] * R[i - j];
52 
53  if (Core::isAlmostEqual(E_[i - 1], (Data)0.0))
54  return false;
55  k_[i] = -k_[i] / E_[i - 1];
56 
57  // a[j][i]
58  alpha_[i][i] = k_[i];
59  for (j = 1; j <= (i - 1); j++)
60  alpha_[j][i] = alpha_[j][i - 1] + k_[i] * alpha_[i - j][i - 1];
61 
62  // E[i]
63  E_[i] = (1.0 - k_[i] * k_[i]) * E_[i - 1];
64  }
65  return true;
66 }
