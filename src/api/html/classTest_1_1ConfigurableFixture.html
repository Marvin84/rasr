Detailed Description

Test fixture for classes derived from Core::Configurable / Core::Component, which require a configuration object.

Use like this:

class MyTest : public Test::ConfigurableFixture { public: void setUp() { setParameter("param-a", "1"); setParameter("param-x", "abc"); obj_ = new MyClass(config); }

void tearDown() { delete obj_; } protected: MyClass *obj_; };

TEST_F(Module, MyTest, Foo) { EXPECT_EQ(obj_->value(), 1); }
