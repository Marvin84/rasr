81  virtual Flow::PortId getOutput(const std::string& name) {
82  std::vector<s32> ins;
83  if (name.empty())
84  ins.push_back(-1);
85  else {
86  std::vector<bool> range;
87  if (Core::ParameterBitVector::parseRangeList(name, range))
88  for (u32 i = 0; i < range.size(); i++)
89  if (range[i])
90  ins.push_back(i);
91  }
92 
93  std::map<std::string, Flow::PortId>::iterator pos;
94  if ((pos = map_.find(name)) != map_.end())
95  return pos->second;
96 
97  PortId out = map_[name] = addOutput();
98  verify(out == (PortId)outputs_.size());

102  virtual bool work(Flow::PortId p) {
103  Flow::DataPtr<Flow::Vector<T>> in;
104  if (!getData(0, in)) {
105  bool ret = false;
106  for (u32 i = 0; i < outputs_.size(); i++)
107  if (putData(i, in.get()))
108  ret = true;
109  return ret;
110  }
111 
112  Time step = (in->getEndTime() - in->getStartTime()) / outputs_.size();
113 
114  for (u32 i = 0; i < outputs_.size(); i++) {
115  Flow::Vector<T>* out = new Flow::Vector<T>;
116  if ((outputs_[i].size() == 1) && (outputs_[i][0] == -1)) {
117  *out = *in;
118  }
119  else {
120  for (u32 j = 0; j < outputs_[i].size(); j++) {
121  size_t index = outputs_[i][j];
122  if (index < (size_t)in->size()) {
123  if (reverse_)
124  index = in->size() - 1 - index;
125  out->push_back((*in)[index]);
126  }
127  else
128  out->push_back(0);
129  }
130  }
131  if (timestamp_) {
132  out->setStartTime(in->getStartTime() + i * step);
133  out->setEndTime(in->getStartTime() + (i + 1) * step);
134  }
135  else {
136  out->setStartTime(in->getStartTime());
137  out->setEndTime(in->getEndTime());
138  }
139  if (!putData(i, out))
140  return false;
141  }
142  return true;
143  }
