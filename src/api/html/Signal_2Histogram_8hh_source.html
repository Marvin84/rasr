26 /**
27  * Histogram: supports
28  * estimation of histograms
29  * calculation of probability distribution function (pdf)
30  * calculation of cummulative probability distribution function (cdf)
31  * calculation of percentiles
32  */
33 template<class Value>
34 class Histogram : public LookupTable<f32, Value> {
35  typedef LookupTable<f32, Value> Precursor;
36 
37 public:
38  typedef typename Precursor::ValueType Weight;
39  typedef typename Precursor::ValueType Probability;
40 
41 public:
42  Histogram(const Value bucketSize = 0)
43  : Precursor(bucketSize) {}
44  Histogram(const Value bucketSize, const Value min, const Value max)
45  : Precursor(bucketSize, min, max) {}
46  void accumulate(const Value v, Weight weight = 1) {
47  *Precursor::insert(v, 0) += weight;
48  }
49  Value percentile(const Probability percent) const;
50  void getPdf(LookupTable<Probability, Value>& pdf) const;
51  void getCdf(LookupTable<Probability, Value>& cdf) const;
52 };

56 template<typename Value>
57 Value Histogram<Value>::percentile(const Probability percent) const {
58  Probability p = percent * this->sum();
59  typename Precursor::ConstantIterator b;
60 
61  for (b = this->begin(); b != this->end() && p > 0; ++b)
62  p -= *b;
63  return index(b - this->begin());
64 }

66 template<typename Value>
67 void Histogram<Value>::getPdf(LookupTable<Probability, Value>& pdf) const {
68  require(this->sum() != 0);
69  pdf = *this;
70  pdf.normalizeSurface();
71 }

73 template<typename Value>
74 void Histogram<Value>::getCdf(LookupTable<Probability, Value>& cdf) const {
75  require(this->sum() != 0);
76  cdf = *this;
77  std::partial_sum(cdf.begin(), cdf.end(), cdf.begin());
78  std::transform(cdf.begin(), cdf.end(), cdf.begin(),
79  std::bind2nd(std::divides<Probability>(), this->sum()));
80 }

109  verify_(v.size() == Precursor::size());

114 template<typename T>
115 T HistogramVector<T>::minimalBucketSize() const {
116  T result = Core::Type<T>::max;
117  for (typename Precursor::const_iterator i = this->begin(); i != this->end(); ++i)
118  result = std::min(result, i->bucketSize());
