81 template<typename T>
82 void FastMatrixMultiplicationNode<T>::initialize() {
83  if (needInit_) {
84  // use Math::Matrix for IO
85  Math::Matrix<T> naiveMatrix;
86  if (matrixFilename_.empty()) {
87  this->error() << "Matrix filename is empty.";
88  }
89  else {
90  if (!Math::Module::instance().formats().read(matrixFilename_, naiveMatrix)) {
91  error("Failed to read matrix from file '%s'.", matrixFilename_.c_str());
92  return;
93  }
94  }
95  nRows_ = naiveMatrix.nRows();
96  nCols_ = naiveMatrix.nColumns();
97  matrix_ = new T[nRows_ * nCols_];
98  // copy matrix into array, use row-major format
99  for (u32 i = 0; i < nRows_; i++) {
100  for (u32 j = 0; j < nCols_; j++) {
101  matrix_[i * nCols_ + j] = naiveMatrix[i][j];
102  }
103  }
104  }
105  needInit_ = false;
106 }

108 template<typename T>
109 bool FastMatrixMultiplicationNode<T>::work(Flow::PortId p) {
110  Flow::DataPtr<Flow::Vector<T>> in;
111  if (!getData(0, in))
112  return SleeveNode::putData(0, in.get());
113 
114  if (needInit_)
115  initialize();
116 
117  /* check the size of the two vectors */
118  if (in->size() != nCols_)
119  error() << "dimension missmatch: dimension of vector is " << in->size()
120  << ", expected " << nCols_;
121 
122  // create output object and reserve space
123  Flow::DataPtr<Flow::Vector<T>> out(new Flow::Vector<T>(nRows_));
124  out->setTimestamp(*in);
125 
126  // apply matrix, use Blas matrix-vector multiplication routine
127  T* vector = &(in->at(0));
128  Math::gemv<T>(CblasRowMajor, CblasNoTrans, nRows_, nCols_, 1.0, matrix_, nCols_, vector, 1, 0.0, &(out->at(0)), 1);
129 
130  return putData(0, out.get());
131 }
