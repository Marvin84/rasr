91  virtual void init(const std::vector<std::string>& arguments) {
92  u32 nMaxColumns = 0;
93  std::string filename = paramFile(config);
94  if (!filename.empty()) {
95  if (!Core::isValidPath(filename))
96  criticalError("BatchNode: Could not find \"%s\"", filename.c_str());
97  TextFileParser tf(filename, paramEncoding(config));
98  if (tf) {
99  log("BatchNode: Read from \"%s\"", filename.c_str());
100  for (;;) {
101  const TextFileParser::StringList& columns = tf.next();
102  if (tf) {
103  nMaxColumns = std::max(u32(columns.size()), nMaxColumns);
104  batches_.push_back(columns);
105  }
106  else
107  break;
108  }
109  }
110  else
111  criticalError("Could not open \"%s\"", filename.c_str());
112  }
113  else if (!arguments.empty()) {
114  log("BatchNode: Read from command line");
115  nMaxColumns = arguments.size();
116  batches_.push_back(arguments);
117  }
118  if (!batches_.empty())
119  log("BatchNode: Found %d batches with (up to) %d arguments",
120  (u32)batches_.size(), nMaxColumns);
121  else
122  error("BatchNode: No input");
123  itBatch_ = batches_.begin();
124  }
125 
126  virtual void sync() {
127  require(itBatch_ != batches_.end());
128  ++itBatch_;
129  }
130 
131  virtual bool good() {
132  return itBatch_ != batches_.end();
133  }
