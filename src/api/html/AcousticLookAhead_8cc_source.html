190 Score AcousticLookAhead::calculateDistance(const AcousticLookAhead::AcousticFeatureVector& mean1, const AcousticLookAhead::AcousticFeatureVector& mean2) {
191  int df, score = 0;
192  u32 cmp = 0;
193  u32 dimension = mean1.size();
194  verify(mean1.size() == mean2.size());
195  switch ((dimension - 1) % 8) {
196  while (cmp < dimension) {
197  case 7:
198  df = (mean1[cmp] - mean2[cmp]);
199  score += df * df;
200  ++cmp;
201  case 6:
202  df = (mean1[cmp] - mean2[cmp]);
203  score += df * df;
204  ++cmp;
205  case 5:
206  df = (mean1[cmp] - mean2[cmp]);
207  score += df * df;
208  ++cmp;
209  case 4:
210  df = (mean1[cmp] - mean2[cmp]);
211  score += df * df;
212  ++cmp;
213  case 3:
214  df = (mean1[cmp] - mean2[cmp]);
215  score += df * df;
216  ++cmp;
217  case 2:
218  df = (mean1[cmp] - mean2[cmp]);
219  score += df * df;
220  ++cmp;
221  case 1:
222  df = (mean1[cmp] - mean2[cmp]);
223  score += df * df;
224  ++cmp;
225  case 0:
226  df = (mean1[cmp] - mean2[cmp]);
227  score += df * df;
228  ++cmp;
229  }
230  }
231  verify_(cmp == dimension);
232  return score;
233 }

289 void AcousticLookAhead::AcousticLookAheadModel::estimate(const std::vector<u32>& assigned,
290  const std::vector<AcousticFeatureVector>& means,
291  u32 splits) {
292  if (assigned.empty())
293  return;
294 
295  means_.clear();
296 
297  const u32 dimension = means[assigned[0]].size();
298 
299  std::vector<f64> accumulator;
300  accumulator.resize(dimension, 0);
301  for (u32 a = 0; a < assigned.size(); ++a)
302  for (u32 d = 0; d < dimension; ++d)
303  accumulator[d] += means[assigned[a]][d];
304 
305  AcousticFeatureVector mean;
306  mean.resize(dimension, 0);
307 
308  for (u32 d = 0; d < dimension; ++d)
309  mean[d] = accumulator[d] / assigned.size();
310 
311  means_.push_back(mean);
312 
313  std::vector<u32> assignedToMean(assigned.size(), 0);
314 
315  for (int s = 0; s < splits; ++s) {
316  /// Step 1: Split all means
317  u32 splitMeans = means_.size();
318  for (u32 m = 0; m < splitMeans; ++m) {
319  means_.push_back(means_[m]);
320  for (u32 d = 0; d < dimension; ++d) {
321  means_[m][d] -= 0.000001;
322  means_.back()[d] += 0.000001;
323  }
324  }
325 
326  /// Step 2: Assign every observation to the closest mean
327  for (u32 a = 0; a < assigned.size(); ++a) {
328  Score bestDist = Core::Type<Score>::max;
329  for (u32 m = 0; m < means_.size(); ++m) {
330  Score dist = calculateDistance(means[assigned[a]], means_[m]);
331  if (dist < bestDist) {
332  bestDist = dist;
333  assignedToMean[a] = m;
334  }
335  }
336  }
337 
338  /// Step 3: Accumulate means
339 
340  std::vector<std::vector<f64>> accumulators(means_.size(), std::vector<f64>(dimension, 0));
341  std::vector<u32> countAssignedToMean(means_.size(), 0);
342 
343  for (u32 a = 0; a < assigned.size(); ++a) {
344  countAssignedToMean[assignedToMean[a]] += 1;
345  for (u32 d = 0; d < dimension; ++d)
346  accumulators[assignedToMean[a]][d] += means[assigned[a]][d];
347  }
348 
349  for (int m = means_.size() - 1; m >= 0; --m) {
350  if (countAssignedToMean[m] == 0) {
351  // Remove means that are not required
352  means_.erase(means_.begin() + m);
353  }
354  else {
355  for (u32 d = 0; d < dimension; ++d)
356  means_[m][d] = accumulators[m][d] / countAssignedToMean[m];
357  }
358  }
359  }
360 }

362 bool AcousticLookAhead::loadModels() {
363  std::string archive = paramAcousticLookAheadCacheArchive(config_);
364 
365  if (archive.empty())
366  return false;
367 
368  Core::MappedArchiveReader in = Core::Application::us()->getCacheArchiveReader(archive, archiveEntry());
369  if (!in.good())

371 
372  if (!in.check<u32>(checksum_, "network checksum") ||
373  !in.check<u32>(acousticLookaheadDepth_, "depth") ||
374  !in.check<u32>(lookaheadModelCount_, "model count") ||
375  !in.check<u32>(iterations_, "iterations") ||
376  !in.check<bool>(considerLabels_, "consider-labels") ||
377  !in.check<bool>(includeCurrentStateModel_, "include-current") ||
378  !in.check<bool>(multiplicity_, "consider-multiplicity") ||
379  !in.check<f32>(splittingThreshold_, "splitting-threshold") ||
380  !in.check<bool>(splitEmpty_, "split-empty"))
381  return false;
382 
383  u64 lookAheadModels = 0ul;
384  in >> lookAheadModels;
385  for (u32 a = 0; a < lookAheadModels; ++a)
386  acousticLookAheadModels_.push_back(AcousticLookAheadModel(in));
387  in >> means_ >> modelForIndex_;
388 
389  verify(in.good());
390  return true;

431 void AcousticLookAhead::initializeModels(EmissionSetCounter sets) {
432  if (acousticLookaheadDepth_ == 0)
433  return;
434  int splitsPerState = paramAcousticLookAheadSplitsPerState(config_);
435  int startSplittingAtIteration = paramAcousticLookAheadStartSplittingAtIteration(config_);
436 
437  Core::Application::us()->log() << "computing acoustic lookahead on " << sets.list.size() << " sets";
438  // Step 1: Load single-gaussian models
439 
440  verify(modelForIndex_.empty());
441  verify(acousticLookAheadModels_.empty());
442 
443  std::set<Mm::DensityIndex> hadDensity;
444 
445  std::vector<AcousticFeatureVector> means;
446  for (u32 mean = 0; mean < mixtureSet_->nMeans(); ++mean)
447  means.push_back(acousticLookAheadScorer_->multiplyAndQuantize(*mixtureSet_->mean(mean))[0]);
448 
449  // Actually the rand() is pseudo-random because we didn't seed the random number generator yet
450  // srand(0);
451 
452  // Step 2: Initialize each lookahead-model with a random single-gaussian distribution from the single-gaussian models
453  for (int a = 0; a < lookaheadModelCount_; ++a) {
454  Mm::DensityIndex pick = 0;
455  while (hadDensity.count(pick))
456  pick = rand() % mixtureSet_->nDensities();
457  hadDensity.insert(pick);
458  acousticLookAheadModels_.push_back(AcousticLookAheadModel(means[mixtureSet_->density(pick)->meanIndex()]));
459  }
460 
461  std::cout << "used " << lookaheadModelCount_ << " out of " << mixtureSet_->nDensities() << " densities" << std::endl;
462 
463  if (!multiplicity_)
464  for (u32 m = 0; m < sets.list.size(); ++m)
465  sets.list[m].second = 1;
466 
467  // Step 3: Iterate:
468  // - Assign each state to the best-matching look-ahead state
469  // - Re-compute a new single-gaussian distribution for the look-ahead state
470 
471  Core::Application::us()->log() << "shared look-ahead sets: " << sets.list.size();
472 
473  std::vector<u32> modelForSet(sets.list.size(), invalidLookAheadModel);
474 
475  for (int i = 0; i < iterations_; ++i) {
476  std::vector<u32> assignedSetsPerModel(acousticLookAheadModels_.size(), 0);
477  std::vector<u32> assignedWeightPerModel(acousticLookAheadModels_.size(), 0);
478 
479  f64 totalDistance = 0;
480 
481  // Maps each look-ahead model to all component mixtures that make up this model
482  // and the multiplicity of each component.
483  std::vector<std::map<Am::AcousticModel::EmissionIndex, u32>> assignments(acousticLookAheadModels_.size());
484 
485  Core::Application::us()->log() << "Acoustic lookahead iteration " << i + 1 << " of " << iterations_;
486 
487  u32 totalCount = 0;
488 
489  for (u32 m = 0; m < sets.list.size(); ++m) {
490  const std::set<Am::AcousticModel::EmissionIndex>& emissions(sets.list[m].first);
491  u32 mult = sets.list[m].second;
492  totalCount += mult;
493 
494  u32 bestLookahead = invalidLookAheadModel;
495  Score bestLookAheadScore = Core::Type<Score>::max;
496 
497  for (u32 lookahead = 0; lookahead < acousticLookAheadModels_.size(); ++lookahead) {
498  Score score = 0;
499  for (std::set<Am::AcousticModel::EmissionIndex>::const_iterator it = emissions.begin(); it != emissions.end(); ++it) {
500  verify(*it != StateTree::invalidAcousticModel);
501  u32 nDensities = mixtureSet_->mixture(*it)->nDensities();
502  verify(nDensities == 1);
503  Mm::GaussDensity* dens = mixtureSet_->density(mixtureSet_->mixture(*it)->densityIndex(0));
504  const AcousticFeatureVector& mean(means[dens->meanIndex()]);
505 
506  score += acousticLookAheadModels_.at(lookahead).distance(mean);
507  }
508 
509  if (score < bestLookAheadScore) {
510  bestLookahead = lookahead;
511  bestLookAheadScore = score;
512  }
513  }
514 
515  verify(bestLookahead != invalidLookAheadModel);
516  assignedSetsPerModel[bestLookahead] += 1;
517  assignedWeightPerModel[bestLookahead] += mult;
518  totalDistance += bestLookAheadScore * mult;
519  modelForSet[m] = bestLookahead;
520 
521  // Update the assignment
522  for (std::set<Am::AcousticModel::EmissionIndex>::const_iterator it = emissions.begin(); it != emissions.end(); ++it) {
523  if (assignments[bestLookahead].count(*it))
524  assignments[bestLookahead][*it] += mult;
525  else
526  assignments[bestLookahead][*it] = mult;
527  }
528  }
529 
530  Core::Application::us()->log() << "assignment distance: " << totalDistance / totalCount;
531 
532  Core::Application::us()->log() << "Estimating";
533 
534  for (u32 lookahead = 0; lookahead < acousticLookAheadModels_.size(); ++lookahead) {
535  std::vector<u32> assignedMeans;
536 
537  for (std::map<Mm::MixtureIndex, u32>::const_iterator assignIt = assignments[lookahead].begin(); assignIt != assignments[lookahead].end(); ++assignIt) {
538  Mm::MixtureIndex mix = (*assignIt).first;
539  Mm::GaussDensity* dens = mixtureSet_->density(mixtureSet_->mixture(mix)->densityIndex(0));
540  for (u32 m = 0; m < (*assignIt).second; ++m)
541  assignedMeans.push_back(dens->meanIndex());
542  }
543 
544  /// @todo Introduce weights for the splits, try out the better models again
545  acousticLookAheadModels_[lookahead].estimate(assignedMeans, means, i >= startSplittingAtIteration ? splitsPerState : 0);
546  }
547 
548  if ((splittingThreshold_ != 0.0 || splitEmpty_) && i + 1 != iterations_) {
549  std::vector<std::pair<u32, u32>> sorted;
550  for (u32 lookahead = 0; lookahead < acousticLookAheadModels_.size(); ++lookahead)
551  sorted.push_back(std::make_pair(lookahead, assignedWeightPerModel[lookahead]));
552 
553  std::sort(sorted.begin(), sorted.end(),
554  Core::composeBinaryFunction(std::less<u32>(),
555  Core::select2nd<std::pair<u32, u32>>(),
556  Core::select2nd<std::pair<u32, u32>>()));
557 
558  u32 splitted = 0;
559 
560  u32 splitUntil = sorted.size() * splittingThreshold_;
561  while (splitEmpty_ && splitUntil < sorted.size() && sorted[splitUntil].second == 0)
562  ++splitUntil;
563 
564  std::vector<u32> splitPotential = assignedSetsPerModel;
565 
566  for (u32 i = 0; i < splitUntil; ++i) {
567  u32 eliminate = sorted[i].first;
568 
569  u32 split = eliminate;
570  for (u32 s = sorted.size() - 1; s > i; --s) {
571  u32 n = sorted[s].first;
572  if (splitPotential[n] > 0) {
573  split = n;
574  splitPotential[split] -= 1;
575  break;
576  }
577  }
578 
579  if (split == eliminate)
580  break; // Found nothing to split
581 
582  ++splitted;
583  acousticLookAheadModels_[eliminate].split(acousticLookAheadModels_[split]);
584  }
585  Core::Application::us()->log() << "splitted " << splitted;
586  }
587 
588  {
589  Core::HistogramStatistics stats("emission-sets assigned per look-ahead node");
590  for (u32 lookahead = 0; lookahead < acousticLookAheadModels_.size(); ++lookahead)
591  stats += assignedSetsPerModel[lookahead];
592  stats.write(Core::Application::us()->log());
593  }
594 
595  {
596  Core::HistogramStatistics stats("weight assigned per look-ahead node");
597  for (u32 lookahead = 0; lookahead < acousticLookAheadModels_.size(); ++lookahead)
598  stats += assignedWeightPerModel[lookahead];
599  stats.write(Core::Application::us()->log());
600  }
601  }
602 
603  modelForIndex_.resize(sets.setForIndex.size(), invalidLookAheadModel);
604  for (u32 index = 0; index < sets.setForIndex.size(); ++index)
605  if (sets.setForIndex[index] != Core::Type<u32>::max)
606  modelForIndex_[index] = modelForSet[sets.setForIndex[index]];
607 
608  // More efficient representation in means_
609  ///@todo Make this the only representation
610  verify(means_.empty());
611  for (std::vector<AcousticLookAheadModel>::const_iterator it = acousticLookAheadModels_.begin(); it != acousticLookAheadModels_.end(); ++it) {
612  verify((*it).means_.size() == 1); // Splits are not supported (not useful)
613  means_.insert(means_.end(), (*it).means_.front().begin(), (*it).means_.front().end());
614  }
615 
616  saveModels();
617 }

631  bool includeCurrent) {
632  verify(state > 0 && state < network.structure.stateCount());
633  if (includeCurrent) {
634  Am::AcousticModel::EmissionIndex mixture = network.structure.state(state).stateDesc.acousticModel;
635  verify(mixture != StateTree::invalidAcousticModel);
636  target.insert(mixture);
637  }
638 
639  if (depth == 0)
640  return;
641 
642  for (HMMStateNetwork::SuccessorIterator successor = network.structure.successors(state); successor; ++successor) {
643  if (not successor.isLabel()) {
644  if (depth > 0) {
645  StateId succState = *successor;
646  verify(succState > 0 && succState < network.structure.stateCount());
647  Mm::MixtureIndex mixture = network.structure.state(succState).stateDesc.acousticModel;
648  if (mixture == StateTree::invalidAcousticModel) {
649  // Skip
650  getSuccessorMixtures(network, succState, target, depth);
651  }
652  else {
653  verify(mixture != StateTree::invalidAcousticModel);
654  target.insert(mixture);
655  getSuccessorMixtures(network, succState, target, depth - 1);
656  }
657  }
658  }
659  else {
660  // Epsilon-transition
661  verify(successor.label() < network.exits.size());
662  getSuccessorMixtures(network, network.exits[successor.label()].transitState, target, depth);
663  }
664  }

759 template<bool useAverage>
760 void AcousticLookAhead::computeAllLookAheadScoresInternal() {
761  if (acousticLookAhead_.empty()) {
762  preCachedLookAheadScores_.assign(preCachedLookAheadScores_.size(), 0);
763  return;
764  }
765 
766  uint len = acousticLookaheadDepth_;
767  if (acousticLookAhead_.size() < len)
768  len = acousticLookAhead_.size();
769 
770  float divFac = len != 0 ? (acousticLookAheadScale_ / len) : 0;
771 
772  if (useAverage) {
773  // Special-case for depth 1: Most efficient
774 
775  if (len == 1) {
776  u32 dimension = acousticLookAhead_[0].first.size();
777 
778  for (int lookaheadId = 0; lookaheadId < acousticLookAheadModels_.size(); ++lookaheadId)
779  preCachedLookAheadScores_[lookaheadId] = calculateDistance(means_.data() + (dimension * lookaheadId),
780  acousticLookAhead_[0].first.data(),
781  dimension) *
782  divFac;
783  return;
784  }
785 
786  // Quick computation with cache
787 
788  fillCacheForTimeframe(currentTimeFrame_);
789 
790  { // Initialize
791  const CacheForTimeframe& cache = cacheForTimeframe(currentTimeFrame_);
792  for (int lookaheadId = 0; lookaheadId < acousticLookAheadModels_.size(); ++lookaheadId)
793  preCachedLookAheadScores_[lookaheadId] = cache.simpleScoreCache_[lookaheadId];
794  }
795 
796  // Add intermediate timeframes
797  for (uint a = 1; a < len - 1; ++a) {
798  fillCacheForTimeframe(currentTimeFrame_ + a);
799  const CacheForTimeframe& cache = cacheForTimeframe(currentTimeFrame_ + a);
800  for (int lookaheadId = 0; lookaheadId < acousticLookAheadModels_.size(); ++lookaheadId)
801  preCachedLookAheadScores_[lookaheadId] += cache.simpleScoreCache_[lookaheadId];
802  }
803 
804  {
805  // Finalize: Add last timeframe and apply averaging-factor
806  fillCacheForTimeframe(currentTimeFrame_ + len - 1);
807  const CacheForTimeframe& cache = cacheForTimeframe(currentTimeFrame_ + len - 1);
808 
809  for (int lookaheadId = 0; lookaheadId < acousticLookAheadModels_.size(); ++lookaheadId)
810  preCachedLookAheadScores_[lookaheadId] = (preCachedLookAheadScores_[lookaheadId] + cache.simpleScoreCache_[lookaheadId]) * divFac;
811  }
812 
813  return;
814  }
815 
816  std::vector<Score>::iterator end = preCachedLookAheadScores_.end();
817  for (std::vector<Score>::iterator it = preCachedLookAheadScores_.begin(); it != end; ++it) {
818  u32 lookaheadId = it - preCachedLookAheadScores_.begin();
819  Score& score(*it);
820 
821  if (useAverage) {
822  verify(0); // Handled above
823  }
824  else {
825  score = Core::Type<Score>::max;
826 
827  uint len = acousticLookaheadDepth_;
828  if (acousticLookAhead_.size() < len)
829  len = acousticLookAhead_.size();
830 
831  for (uint a = 0; a < len; ++a) {
832  Score localScore = getCachedScaledScore(a + currentTimeFrame_, lookaheadId);
833  if (localScore < score)
834  score = localScore;
835  }
836  }
837  }
838 }
