87 bool CombinedGrammarFst::load(const std::string& root) {
88  log("loading root fst: %s", root.c_str());
89  rootFst_ = OpenFst::VectorFst::Read(root);
90  if (!rootFst_) {
91  error("error loading %s", root.c_str());
92  return false;
93  }
94  const std::vector<std::string> addOnFiles = paramAddOnFiles(config);
95  for (std::vector<std::string>::const_iterator f = addOnFiles.begin(); f != addOnFiles.end(); ++f) {
96  log("loading add on fst: %s", f->c_str());
97  OpenFst::VectorFst* fst = OpenFst::VectorFst::Read(*f);
98  if (!fst) {
99  error("error loading %s", f->c_str());
100  return false;
101  }
102  addOnFsts_.push_back(fst);
103  }
104  const std::vector<std::string> labels = paramReplaceLabels(config);
105  if (!labels.empty()) {
106  verify(rootFst_->OutputSymbols());
107  for (std::vector<std::string>::const_iterator l = labels.begin(); l != labels.end(); ++l) {
108  OpenFst::Label id = rootFst_->OutputSymbols()->Find(*l);
109  verify(id >= 0);
110  replaceLabels_.push_back(id);
111  log("using replace label: %s = %d", l->c_str(), id);
112  }
113  }
114  else {
115  const std::vector<int> ids = paramReplaceIds(config);
116  for (std::vector<int>::const_iterator i = ids.begin(); i != ids.end(); ++i) {
117  replaceLabels_.push_back(*i);
118  log("using replace label: %d", *i);
119  }
120  }
121  verify(replaceLabels_.size() == addOnFsts_.size());
122 
123  const std::vector<f64> scales = paramAddOnScales(config);
124  for (u32 i = 0; i < std::min(scales.size(), addOnFsts_.size()); ++i) {
125  f32 scale = scales[i];
126  log("applying scale to add on fst %d: %f", i, scale);
127  if (scale != 1.0) {
128  OpenFst::scaleWeights(addOnFsts_[i], scale);
129  }
130  }
131  replaceArcLabels();
132  return true;
133 }

135 void CombinedGrammarFst::replaceArcLabels() {
136  typedef std::pair<OpenFst::Label, OpenFst::Label> LabelPair;
137  std::vector<LabelPair> iLabelMap, oLabelMap;
138  for (u32 i = 0; i < addOnFsts_.size(); ++i) {
139  OpenFst::Label uniqLabel = -(i + 1);
140  log("add on %i: mapping label %d to %d", i, replaceLabels_[i], uniqLabel);
141  oLabelMap.push_back(std::make_pair(replaceLabels_[i], uniqLabel));
142  iLabelMap.push_back(std::make_pair(replaceLabels_[i], 0));
143  }
144  /* HACK: The 1.3.2 version of OpenFst contains an additional check in comparison to 1.2 trunk in
145  * src/include/fst/relabel.h that fails for some reason. It looks like commenting that check
146  * does not hurt, so why not to do it.
147  * [kozielski]
148  * */
149  log("If you get an error below you need to comment that check in OpenFST (src/include/fst/relabel.h) and link against the new version.");
150  FstLib::Relabel(rootFst_, iLabelMap, oLabelMap);
151 }

200 bool ComposedGrammarFst::load(const std::string& root) {
201  log("loading root fst: %s", root.c_str());
202  rootFst_ = OpenFst::VectorFst::Read(root);
203  if (!rootFst_)
204  error("error loading %s", root.c_str());
205  const std::string addOn = paramAddOnFile(config);
206  log("loading add on fst: %s", addOn.c_str());
207  addOnFst_ = OpenFst::VectorFst::Read(addOn);
208  if (!addOnFst_)
209  error("error loading %s", addOn.c_str());
210  if (!(rootFst_ && addOnFst_)) {
211  return false;
212  }
213  f32 scale = paramAddOnScale(config);
214  if (scale != 1.0) {
215  log("applying scale to add on fst: %f", scale);
216  OpenFst::scaleWeights(addOnFst_, scale);
217  }
218  rootFst_->SetOutputSymbols(0);
219  addOnFst_->SetInputSymbols(0);
220  FstLib::ArcSort(addOnFst_, FstLib::StdILabelCompare());
221  return true;
222 }

224 void ComposedGrammarFst::relabel(const GrammarRelabelerBase& relabeler) {
225  relabeler.getMap(&iLabelMap_);
226  log("relabeling map with %zd entries", iLabelMap_.size());
227  const OpenFst::SymbolTable* symbols = rootFst_->InputSymbols();
228  OpenFst::Label freeLabel = symbols->AvailableKey();
229  log("using dummy label: %d", freeLabel);
230  std::unordered_set<OpenFst::Label> mappedLabels(iLabelMap_.size());
231  for (GrammarRelabelerBase::LabelMap::const_iterator i = iLabelMap_.begin();
232  i != iLabelMap_.end(); ++i) {
233  mappedLabels.insert(i->first);
234  }
235  for (OpenFst::Label l = 1; l < freeLabel; ++l) {
236  if (!mappedLabels.count(l)) {
237  iLabelMap_.push_back(std::make_pair(l, freeLabel));
238  log("unmapped symbol: %d %s", l, symbols->Find(l).c_str());
239  }
240  }
241  log("updated relabeling map with %zd entries", iLabelMap_.size());
242  FstLib::ArcSort(rootFst_, FstLib::StdILabelCompare());
243  FstLib::ArcSort(addOnFst_, FstLib::StdILabelCompare());
244 }
