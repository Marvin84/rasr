77  * smoothing function for unsupervized
78  * Defined on the distance d=log(p/(1-p))

80  * The smoothing function is constant in the

95 UnsupervizedSmoothingFunction::UnsupervizedSmoothingFunction(const Core::Configuration& config)
96  : dA_(paramA(config)),
97  dB_(paramB(config)),
98  dS_((M_PI / (dB_ - dA_))),
99  xBn_(exp(-dB_) / (1 + exp(-dB_))),
100  xAn_(exp(-dA_) / (1 + exp(-dA_))),
101  xAp_(exp(dA_) / (1 + exp(dA_))),
102  xBp_(exp(dB_) / (1 + exp(dB_))),
103  nInfB_(0),
104  nBA_(0),
105  nAA_(0),
106  nAB_(0),
107  nBInf_(0) {
108  require(dA_ < dB_);
109  require(xBn_ > 0);
110  require(xBn_ < xAn_);
111  require(xAn_ < xAp_);
112  require(xAp_ < xBp_);
113  require(xBp_ < 1);
114 }

116 // f(x)
117 f64 UnsupervizedSmoothingFunction::f(f64 x) const {
118  if ((xAn_ < x) and (x < xAp_)) {
119  return 0;
120  }
121  else if ((x < xBn_) or (xBp_ < x)) {
122  return 1;
123  }
124  else {
125  const f64 d = std::log(x / (1 - x));
126  return (1 - cos((Core::abs(d) - dA_) * dS_)) / 2;
127  }
128 }

130 // f'(x)*x
131 f64 UnsupervizedSmoothingFunction::dfx(f64 x) const {
132  if ((xAn_ < x) and (x < xAp_)) {
133  return 0;
134  }
135  else if ((x < xBn_) or (xBp_ < x)) {
136  return 0;
137  }
138  else {
139  const f64 d = std::log(x / (1 - x));
140  return (dS_ / 2) * sin((Core::abs(d) - dA_) * dS_) / (1 - x);
141  }
142 }

144 void UnsupervizedSmoothingFunction::updateStatistics(f64 x) {
145  Precursor::updateStatistics(x);
146  if ((xAn_ < x) and (x < xAp_)) {
147  ++nAA_;
148  }
149  else if (x < xBn_) {
150  ++nInfB_;
151  }
152  else if (xBp_ < x) {
153  ++nBInf_;
154  }
155  else if ((xBn_ < x) and (x < xAn_)) {
156  ++nBA_;
157  }
158  else if ((xAp_ < x) and (x < xBp_)) {
159  ++nAB_;
160  }
161  else {
162  defect();
163  }
164 }
