145 template<typename T>
146 NeuralNetworkTrainer<T>* NeuralNetworkTrainer<T>::createSupervisedTrainer(const Core::Configuration& config) {
147  NeuralNetworkTrainer<T>* trainer = NULL;
148 
149  // get the type of the trainer
150  switch ((TrainerType)paramNetworkTrainer(config)) {
151  case feedForwardTrainer:
152  Core::Application::us()->log("Create trainer: feed-forward trainer");
153  trainer = new FeedForwardTrainer<T>(config);
154  break;
155  case frameClassificationErrorAccumulator:
156  Core::Application::us()->log("Create trainer: frame-classification-error");
157  trainer = new FrameErrorEvaluator<T>(config);
158  break;
159  case meanAndVarianceAccumulator:
160  Core::Application::us()->log("Create trainer: mean-and-variance-estimation");
161  trainer = new MeanAndVarianceTrainer<T>(config);
162  break;
163  case networkEvaluator:
164  Core::Application::us()->log("Create trainer: network-evaluator");
165  trainer = new NetworkEvaluator<T>(config);
166  break;
167  case autoencoderTrainer:
168  Core::Application::us()->log("Create trainer: autoencoder");
169  trainer = new FeedForwardAutoTrainer<T>(config);
170  break;
171  case pythonTrainer:
172 #ifdef MODULE_PYTHON
173  Core::Application::us()->log("Create trainer: Python trainer");
174  trainer = new PythonTrainer<T>(config);
175 #else
176  Core::Application::us()->criticalError("Python-trainer: Python support not compiled");
177 #endif
178  case pythonEvaluator:
179 #ifdef MODULE_PYTHON
180  Core::Application::us()->log("Create trainer: Python evaluator");
181  trainer = new PythonEvaluator<T>(config);
182 #else
183  Core::Application::us()->criticalError("Python-evaluator: Python support not compiled");
184 #endif
185 
186  break;
187  default: // dummy trainer
188  Core::Application::us()->warning("The given trainer is not a valid supervised trainer type. Create dummy trainer.");
189  trainer = new NeuralNetworkTrainer<T>(config);
190  Core::Application::us()->log("Create trainer: dummy");
191  break;
192  };
193 
194  return trainer;
195 }

280 template<typename T>
281 void FrameErrorEvaluator<T>::processBatch_finishWithAlignment(Math::CudaVector<u32>& alignment) {
282  alignment.initComputation();
283 
284  u32 nObservations = network().getLayerInput(0)[0]->nColumns();
285  verify_eq(nObservations, alignment.size());
286 
287  Precursor::criterion_->inputAlignment(alignment, network().getTopLayerOutput(), weights_);
288  if (Precursor::criterion_->discardCurrentInput()) {
289  Core::Component::log("discard current mini-batch");
290  return;
291  }
292 
293  u32 batchFrameClassificationErrors = network().getTopLayerOutput().nClassificationErrors(alignment);
294  T batchObjectiveFunction = 0;
295  Precursor::criterion_->getObjectiveFunction(batchObjectiveFunction);
296 
297  T batchEntropy_ = 0;
298  if (logFrameEntropy_) {
299  Math::FastVector<T> entropy(nObservations);
300  // not necessary once FastVector::columnEntropy() is implemented on GPU
301  network().getTopLayerOutput().finishComputation(true);
302  Math::FastMatrix<T>& output = network().getTopLayerOutput().asWritableCpuMatrix();
303  entropy.columnEntropy(output);
304  batchEntropy_ = entropy.sum();
305  network().getTopLayerOutput().initComputation(false);
306  }
307 
308  if (statisticsChannel_.isOpen()) {
309  statisticsChannel_ << Core::XmlOpen("batch-statistics")
310  << Core::XmlFull("frame-classification-error-rate-on-batch", (T)batchFrameClassificationErrors / nObservations)
311  << Core::XmlFull("objective-function-on-batch", batchObjectiveFunction / nObservations);
312  if (logFrameEntropy_)
313  statisticsChannel_ << Core::XmlFull("average-entropy-on-batch", batchEntropy_ / nObservations);
314  statisticsChannel_ << Core::XmlClose("batch-statistics");
315  }
316  nFrameClassificationErrors_ += batchFrameClassificationErrors;
317  nObservations_ += nObservations;
318  objectiveFunction_ += batchObjectiveFunction;
319  frameEntropy_ += batchEntropy_;
320 }

448 template<typename T>
449 void MeanAndVarianceTrainer<T>::writeMeanAndStandardDeviation(Statistics<T>& statistics) {
450  statistics.finalize(true);
451  statistics.finishComputation();
452  u32 dim = statistics.featureSum().size();
453  mean_.resize(dim);
454  standardDeviation_.resize(dim);
455  for (u32 i = 0; i < dim; i++) {
456  mean_.at(i) = statistics.featureSum().at(i);
457  standardDeviation_.at(i) = std::sqrt(statistics.squaredFeatureSum().at(i));
458  }
459  this->log("estimating mean and variance from ") << statistics.nObservations() << " observations";
460  this->log("write mean vector to file: ") << meanFile_;

462  this->log("write standard deviation vector to file: ") << standardDeviationFile_;

466 template<typename T>
467 void MeanAndVarianceTrainer<T>::processBatch_feedInput(std::vector<NnMatrix>& features, NnVector* weights, Bliss::Segment* segment) {
468  if (Precursor::weightedAccumulation_ && weights)
469  weights->initComputation();
470  features[0].initComputation();
471  if (features[0].nColumns() != tmp_.nColumns())
472  tmp_.resize(tmp_.nRows(), features[0].nColumns());
473 
474  tmp_.copy(features[0]);
475 
476  // weight features
477  if (Precursor::weightedAccumulation_ && weights)
478  features[0].multiplyColumnsByScalars(*weights);
479 
480  // accumulate sum
481  statistics_->featureSum().addSummedColumns(features[0]);
482 
483  // accumulate square sum
484  tmp_.elementwiseMultiplication(tmp_);
485  if (Precursor::weightedAccumulation_ && weights)
486  tmp_.multiplyColumnsByScalars(*weights);
487  statistics_->squaredFeatureSum().addSummedColumns(tmp_);
488 
489  // accumulate weight
490  if (Precursor::weightedAccumulation_ && weights)
491  statistics_->addToTotalWeight(weights->asum());
492  else
493  statistics_->addToTotalWeight(features[0].nColumns());
494  statistics_->incObservations(features[0].nColumns());
495 }

553 template<typename T>
554 void NetworkEvaluator<T>::processBatch_finishWithSpeechSegment(Bliss::SpeechSegment& segment) {
555  NnMatrix& networkOutput = network().getTopLayerOutput();
556  networkOutput.finishComputation(true);
557 
558  u32 frameCount = networkOutput.nColumns();
559 
560  if (dumpPosteriorsArchive_) {
561  Flow::ArchiveWriter<Math::Matrix<T>> writer(dumpPosteriorsArchive_.get());
562  networkOutput.convert(writer.data_->data());
563  writer.write(segment.fullName());
564  }
565 
566  if (dumpBestPosterioIndicesArchive_) {
567  Flow::ArchiveWriter<Math::Vector<u32>> writer(dumpBestPosterioIndicesArchive_.get());
568  Math::Vector<u32>& bestEmissions = writer.data_->data();
569  bestEmissions.resize(frameCount);
570  for (u32 t = 0; t < frameCount; ++t) {
571  u32 argMax = 0;
572  T maxValue = networkOutput.at(argMax, t);
573  for (u32 i = 1; i < networkOutput.nRows(); ++i) {
574  T value = networkOutput.at(i, t);
575  if (value > maxValue) {
576  maxValue = value;
577  argMax = i;
578  }
579  }
580  bestEmissions[t] = argMax;
581  }
582  writer.write(segment.fullName());
583  }
584 
585  networkOutput.initComputation(false);
586 }
