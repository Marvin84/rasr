278 bool LinearCombinationLatticeProcessorNode::checkNParts(u32 nParts) const {
279  bool success = true;
280  Core::StringHashMap<std::vector<Fsa::Weight>>::const_iterator it = scales_.begin();
281  for (; it != scales_.end(); ++it) {
282  if (it->second.size() != nParts) {
283  success = false;
284  error("mismatch in number of lattice parts and number of scales for output \"") << it->first << "\"";
285  }
286  }
287  return success;
288 }

290 void LinearCombinationLatticeProcessorNode::processWordLattice(
291  Lattice::ConstWordLatticeRef lattice, Bliss::SpeechSegment* s) {
292  if (!checkNParts(lattice->nParts())) {
293  return;
294  }
295  timeval start, end;
296  TIMER_START(start)
297  Lattice::ConstWordLatticeRef result = Lattice::linearCombination(lattice, scales_);
298  TIMER_STOP(start, end, timeProcessSegment_)
299  Precursor::processWordLattice(result, s);
