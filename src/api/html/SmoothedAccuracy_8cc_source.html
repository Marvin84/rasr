82 void SmoothedFrameStateAccuracyAutomaton::LatticeToActiveStates::discoverState(Fsa::ConstStateRef sp) {
83  const Speech::TimeframeIndex startTime = wordBoundaries_->time(sp->id());
84  for (Fsa::State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
85  const Bliss::LemmaPronunciation* pronunciation = parent_.alphabet_->lemmaPronunciation(a->input());
86  if (!pronunciation)
87  continue;
88  const Speech::TimeframeIndex endTime =
89  wordBoundaries_->time(fsa_->getState(a->target())->id());
90  Bliss::Coarticulated<Bliss::LemmaPronunciation> coarticulatedPronunciation(
91  *pronunciation, wordBoundaries_->transit(sp->id()).final,
92  wordBoundaries_->transit(fsa_->getState(a->target())->id()).initial);
93  const Speech::Alignment* alignment =
94  parent_.alignmentGenerator_->getAlignment(coarticulatedPronunciation, startTime, endTime);
95  for (std::vector<Speech::AlignmentItem>::const_iterator al = alignment->begin(); al != alignment->end(); ++al) {
96  parent_.stateIds_.grow(al->time, States());
97  States& stateIds = parent_.stateIds_[al->time];
98  if (stateIds.find(parent_.label(al->emission)) == stateIds.end()) {
99  stateIds[parent_.label(al->emission)] = f32(a->weight());
100  }
101  else {
102  stateIds[parent_.label(al->emission)] += f32(a->weight());
103  }
104  }
105  }
106 }

108 void SmoothedFrameStateAccuracyAutomaton::LatticeToActiveStates::finish() {
109  for (Speech::TimeframeIndex t = 0; t < parent_.stateIds_.size(); ++t) {
110  States& stateIds = parent_.stateIds_[t];
111  f64 sum = 0;
112  for (States::iterator it = stateIds.begin(); it != stateIds.end(); ++it) {
113  sum += it->second;
114  }
115  for (States::iterator it = stateIds.begin(); it != stateIds.end(); ++it) {
116  it->second = sum;
117  }
118  parent_.smoothing_.updateStatistics(sum);
119  }
120 }

135 void SmoothedFrameStateAccuracyAutomaton::modifyState(Fsa::State* sp) const {
136  const Speech::TimeframeIndex startTime = wordBoundaries_->time(sp->id());
137  for (Fsa::State::iterator a = sp->begin(); a != sp->end(); ++a) {
138  const Bliss::LemmaPronunciation* pronunciation = alphabet_->lemmaPronunciation(a->input());
139  f32 weight = 0;
140  if (pronunciation) {
141  const Speech::TimeframeIndex endTime =
142  wordBoundaries_->time(fsa_->getState(a->target())->id());
143  Bliss::Coarticulated<Bliss::LemmaPronunciation> coarticulatedPronunciation(
144  *pronunciation, wordBoundaries_->transit(sp->id()).final,
145  wordBoundaries_->transit(fsa_->getState(a->target())->id()).initial);
146  const Speech::Alignment* alignment =
147  alignmentGenerator_->getAlignment(coarticulatedPronunciation, startTime, endTime);
148  for (std::vector<Speech::AlignmentItem>::const_iterator al = alignment->begin(); al != alignment->end(); ++al) {
149  weight += accuracy(stateIds_[al->time], label(al->emission));
150  }
151  }
152  a->weight_ = Fsa::Weight(weight);
153  }
154 }
