64 void SimdGaussDiagonalMaximumFeatureScorer::init(const MixtureSet& mixtureSet) {
65  covarianceTable_.resize(mixtureSet.nCovariances());
66  for (size_t i = 0; i < covarianceTable_.size(); ++i)
67  covarianceTable_[i] = *mixtureSet.covariance(i);
68 
69  MeanType scaling = getScaling(mixtureSet);
70  log("Scaling factor: ") << scaling;
71 
72  scalingSquared_ = scaling * scaling;
73  inverseQuantizationFactor_ = 0.5 / scalingSquared_;
74 
75  for (u32 i = 0; i < covarianceTable_.size(); ++i)
76  covarianceTable_[i].scale(scaling);
77 
78  buildMixtureTable(mixtureSet);
79 }

81 void SimdGaussDiagonalMaximumFeatureScorer::buildMixtureTable(const MixtureSet& mixtureSet) {
82  dimension_ = mixtureSet.dimension();
83 
84  // create quantized mixture set
85  mixtureTable_.resize(mixtureSet.nMixtures());
86  for (MixtureIndex mix = 0; mix < mixtureTable_.size(); ++mix) {
87  const Mixture* mixture = mixtureSet.mixture(mix);
88  MixtureElement& quantizedMixture = mixtureTable_[mix];
89 
90  quantizedMixture.setNumberOfDensities(mixture->nDensities());
91  for (size_t dns = 0; dns < mixture->nDensities(); ++dns) {
92  const GaussDensity* density = mixtureSet.density(mixture->densityIndex(dns));
93 
94  quantizedMixture.density(dns).covarianceIndex_ = density->covarianceIndex();
95 
96  Weight scaledMinus2LogWeight = scalingSquared_ * -2 * mixture->logWeight(dns);
97 
98  FeatureScorerOptimization::createDensityElement(scaledMinus2LogWeight,
99  *mixtureSet.mean(density->meanIndex()),
100  covarianceTable_[density->covarianceIndex()],
101  quantizedMixture.density(dns));
102  }
103  }
104 }

106 MeanType SimdGaussDiagonalMaximumFeatureScorer::getScaling(const MixtureSet& mixtureSet) const {
107  MeanType minMean = Core::Type<MeanType>::max;
108  MeanType maxMean = Core::Type<MeanType>::min;
109 
110  for (DensityIndex i = 0; i < mixtureSet.nDensities(); ++i) {
111  const GaussDensity* density = mixtureSet.density(i);
112  const std::vector<MeanType>* mean = mixtureSet.mean(density->meanIndex());
113  const std::vector<VarianceType>& inverseSquareRootVariance =
114  covarianceTable_[density->covarianceIndex()].inverseSquareRootDiagonal();
115 
116  verify(mean->size() == inverseSquareRootVariance.size());
117 
118  for (ComponentIndex cmp = 0; cmp < mean->size(); ++cmp) {
119  MeanType dividedMean = (*mean)[cmp] * inverseSquareRootVariance[cmp];
120  minMean = std::min(minMean, dividedMean);
121  maxMean = std::max(maxMean, dividedMean);
122  }
123  }
124  log() << "Mean bounds: min= " << minMean << ", max= " << maxMean;
125  return quantizationScalingFactor(minMean, maxMean);
126 }

128 FeatureType SimdGaussDiagonalMaximumFeatureScorer::quantizationScalingFactor(FeatureType minValue, FeatureType maxValue) {
129  int quantizedIntervalSize =
130  (int)Core::Type<QuantizedType>::max - (int)Core::Type<QuantizedType>::min;
131  FeatureType intervalSize = 2 * std::max(Core::abs(minValue), Core::abs(maxValue));
132  return (FeatureType)quantizedIntervalSize / (1.25 * intervalSize);

148  * correctness:
149  * Do not use floating point operations after optimization_.distance(),
150  * but before calling optimization_.resetFloatingPointCalculation(), because
151  * this may result in incorrect behaviour.
152  *
153  * efficiency:
154  * For the sake of some processor types which can not switch quickly
155  * from optimized integer modus to floating point modus,
156  * use INTEGER OPERATIONS ONLY in this function!
157  */
158 std::pair<int, DensityIndex> SimdGaussDiagonalMaximumFeatureScorer::quantizedScore(const MixtureElement& mixture,
159  const std::vector<PreparedFeatureVector>& featuresPerCovariance) const {
160  int minScore = Core::Type<int>::max;
161  size_t bestDensity = Core::Type<size_t>::max;
162 
163  for (size_t dns = 0; dns < mixture.nDensities(); ++dns) {
164  const MixtureElement::Density& density = mixture.density(dns);
165 
166  int score = density.constantWeight_ +
167  optimization_.distance(density.preparedMean_, featuresPerCovariance[density.covarianceIndex_]);
168 
169  if (score < minScore) {
170  minScore = score;
171  bestDensity = dns;
172  }
173  }
174  optimization_.resetFloatingPointCalculation();
175  return std::make_pair(minScore, bestDensity);
176 }
