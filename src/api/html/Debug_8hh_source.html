38 /**
39  * execute debug command depending on debug level
40  *
41  * C++ example: DBGCMD(myLevel, callMyTimeConsumingDebugFunction());
42  *
43  * C example: DBGCMD(myLevel, fprintf(stdout, "(%d/%d) %s:%d:%s filename='%s'\n",
44  * myLevel, DBG_LEVEL,
45  * __FILE__, __LINE__, __PRETTY_FUNCTION__,
46  * filename));
47  *
48  */
49 #define DBGCMD(level, cmd) \
50  if (DBG_LEVEL >= level) \
51  cmd
