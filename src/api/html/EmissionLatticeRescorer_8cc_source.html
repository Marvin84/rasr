78 Lattice::ConstWordLatticeRef EmissionLatticeRescorer::work(Lattice::ConstWordLatticeRef lattice, Bliss::SpeechSegment* segment) {
79  EmissionLatticeRescorerAutomaton* f = 0;
80 // read alignment and forward network in parallel threads
81 // TODO better read emission indices directly from a file here
82 #pragma omp parallel sections
83  {
84 #pragma omp section // read alignment
85  {
86  timeval start, end;
87  TIMER_START(start)
88  alignmentGenerator_->setSpeechSegment(segment);
89  TIMER_STOP(start, end, timeIO_);
90  }
91 #pragma omp section // forward network
92  {
93  timeval start, end;
94  Speech::ConstSegmentwiseFeaturesRef features;
95  verify(segmentwiseFeatureExtractor_);
96  features = segmentwiseFeatureExtractor_->features(portId_);
97 
98  // always resize the activations to the sequence length
99  // TODO set a maximum in order to limit memory
100  TIMER_START(start);
101  network().resizeActivations(features->size());
102  TIMER_GPU_STOP(start, end, measureTime_, timeMemoryAllocation_)
103 
104  TIMER_START(start);
105  f = new EmissionLatticeRescorerAutomaton(
106  lattice, alignmentGenerator_, acousticModel_, features, checkValues_);
107  TIMER_GPU_STOP(start, end, measureTime_, timeForwarding_);
108  }
109  }
110  // implicit omp barrier
111  verify(f);
112  Lattice::WordLattice* result = new Lattice::WordLattice;
113  result->setWordBoundaries(lattice->wordBoundaries());
114  result->setFsa(Fsa::ConstAutomatonRef(f), Lattice::WordLattice::acousticFsa);
115  return Lattice::ConstWordLatticeRef(result);
116 }

162 bool EmissionLatticeRescorerAutomaton::forwardNetwork(bool checkValues) {
163  if (features_->size() == 0) {
164  Core::Application::us()->warning("no features in segment");
165  return true;
166  }
167  u32 nStreams = features_->at(0)->nStreams();
168  u32 nFrames = features_->size();
169  inputFeatures_.resize(nStreams, NnMatrix());
170 
171  for (u32 stream = 0; stream < nStreams; ++stream) {
172  u32 dim = (*features_->at(0))[stream]->size();
173  inputFeatures_.at(stream).resize(dim, nFrames);
174  for (u32 t = 0; t < nFrames; t++) {
175  Math::copy(dim, &((*features_->at(t))[stream]->at(0)), 1, &(inputFeatures_.at(stream).at(0, t)), 1);
176  }
177  }
178 
179  if (!network().isComputing()) {
180  network().initComputation();
181  }
182  bool result = network().forward(inputFeatures_);
183  network().getTopLayerOutput().finishComputation();
184  if (checkValues && !network().getTopLayerOutput().isFinite()) {
185  Core::Application::us()->error("non-finite scores, check whether model is valid (maybe learning rate too large in sequence training?)");
186  }
187 
188  return result;
189 }

208 Fsa::Weight EmissionLatticeRescorerAutomaton::_score(const Bliss::Coarticulated<Bliss::LemmaPronunciation>& coarticulatedPronunciation,
209  TimeframeIndex begtime, TimeframeIndex endtime) const {
210  verify(acousticModel_);
211  f32 score = fsa_->semiring()->one();
212  if (begtime < endtime) {
213  const Speech::Alignment* alignment = alignmentGenerator_->getAlignment(coarticulatedPronunciation, begtime, endtime);
214  if (labelType_ == Speech::Alignment::allophoneStateIds) {
215  for (std::vector<Speech::AlignmentItem>::const_iterator al = alignment->begin();
216  al != alignment->end(); ++al) {
217  score -= network().getTopLayerOutput()(labelWrapper().getOutputIndexFromClassIndex(acousticModel_->emissionIndex(al->emission)), al->time);
218  }
219  }
220  else {
221  for (std::vector<Speech::AlignmentItem>::const_iterator al = alignment->begin(); al != alignment->end(); ++al) {
222  score -= network().getTopLayerOutput()(labelWrapper().getOutputIndexFromClassIndex(al->emission), al->time);
223  }
224  }
225  }
226  else {
227  Core::Application::us()->warning("score 0 assigned to arc with begin time ")
228  << begtime << " , end time " << endtime << " and label id " << coarticulatedPronunciation.object().id();
229  }
230  return Fsa::Weight(score);
231 }
