393 /**
394  * SimpleFunctionNode for neural network activation function SOFTMAX
395  * y_i = EXP(x_i - MAX(x)) / SUM( EXP( x_i - MAX(i) ) )
396  */
397 template<class T>
398 class VectorSoftmaxFunction : public SimpleFunction<Vector<T>, T> {
399 public:
400  void apply(Vector<T>& v, T) {
401  /* MAX(v) */
402  T maxValue = *std::max_element(v.begin(), v.end());
403  /* v[i] = v[i] - MAX(v) */
404  std::transform(v.begin(), v.end(), v.begin(),
405  std::bind2nd(std::plus<T>(), -maxValue));
406  /* v[i] = exp( v[i] - MAX(v) )*/
407  std::transform(v.begin(), v.end(), v.begin(), static_cast<T (*)(T)>(std::exp));
408 
409  /* sum( exp( v[i] - MAX(v) ) ) */
410  T sumValue = std::accumulate(v.begin(), v.end(), (T)0);
411  /* exp( (v[i] - MAX(v)) ) / sum( exp( (v[i] - MAX(v)) ) ) */
412  std::transform(v.begin(), v.end(), v.begin(),
413  std::bind2nd(std::multiplies<T>(), (T)1 / sumValue));
414  }
415 
416  static std::string name() {
417  return Vector<T>::type()->name() + "-softmax";
418  }
419 };
