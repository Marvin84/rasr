72 template<class Map>
73 void XmlMapElement<Map>::newEntry(const XmlAttributes atts) {
74  const char* keyValue = atts[keyAttributeName_];
75  if (keyValue == 0) {
76  this->parser()->error("Map entry must have an attribute \"%s\"", keyAttributeName_);
77  return;
78  }
79 
80  std::stringstream keyStream(keyValue);
81  typename Map::key_type key;
82  keyStream >> key;
83 
84  const char* dataValue = atts[dataAttributeName_];
85  if (dataValue == 0) {
86  this->parser()->error("Map entry must have an attribute \"%s\"", dataAttributeName_);
87  return;
88  }
89 
90  std::stringstream dataStream(dataValue);
91  typename Map::mapped_type data;
92  dataStream >> data;
93 
94  (*Predecessor::product_)[key] = data;
95 }
