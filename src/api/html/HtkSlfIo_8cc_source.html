101 std::string HtkSlfContext::info() const {
102  std::ostringstream oss;
103  oss << "htk slf context" << std::endl
104  << " type: " << ((type_ == HtkSlfForward) ? "forward" : "backward") << std::endl
105  << " fps: " << fps_ << std::endl
106  << " capitalize: " << (isCapitalize_ ? "true" : "false") << std::endl
107  << " merge penalties: " << (mergePenalty_ ? "true" : "false") << std::endl;
108  oss << std::endl;
109  if (base_ == Core::Type<f32>::min)
110  oss << " base: n/a" << std::endl;
111  else if (base_ == -1.0)
112  oss << " base: e" << std::endl;
113  else
114  oss << " base: " << base_ << std::endl;
115  oss << " eps-symbol: " << epsSymbol_ << std::endl;
116  if (semiring_) {
117  oss << " semiring: " << semiring_->name() << std::endl
118  << " am : " << amId_ << std::endl
119  << " lm : " << lmId_ << std::endl;
120  if (!mergePenalty_)
121  oss << " penalty: " << penaltyId_ << std::endl;
122  }
123  else
124  oss << " semiring: n/a" << std::endl;
125  oss << " lm-name: " << lmName_ << std::endl;
126  if (wrdPenalty_ == Semiring::Invalid)
127  oss << " word penalty: n/a" << std::endl;
128  else
129  oss << " word penalty: " << wrdPenalty_ << std::endl;
130  if (silPenalty_ == Semiring::Invalid)
131  oss << " sil. penenalty: n/a" << std::endl;
132  else
133  oss << " sil. penenalty: " << silPenalty_ << std::endl;
134  oss << std::endl;
135  Lexicon::LemmaPronunciationAlphabetRef lemmaPronunciationAlphabet =
136  Lexicon::us()->lemmaPronunciationAlphabet();
137  if (unkId_ != Fsa::InvalidLabelId)
138  oss << " unknown: \"" << std::string(lemmaPronunciationAlphabet->symbol(unkId_)) << "\"" << std::endl;
139  else
140  oss << " unknown: n/a" << std::endl;
141  if (silId_ != Fsa::InvalidLabelId)
142  oss << " silence: \"" << std::string(lemmaPronunciationAlphabet->symbol(silId_)) << "\"" << std::endl;
143  else
144  oss << " silence: n/a" << std::endl;
145  return oss.str();
146 }

192 HtkSlfContextRef HtkSlfContext::create(const Core::Configuration& config) {
193  HtkSlfContextRef context = HtkSlfContextRef(new HtkSlfContext());
194  // set type
195  context->setType(HtkSlfType(paramType(config)));
196  // set fps
197  context->setFps(paramFps(config));
198  // set capitalize
199  context->setCapitalize(paramCapitalize(config));
200  // set penalties, if specified
201  Score wrdPenalty = paramWordPenalty(config, Semiring::Invalid);
202  Score silPenalty = paramSilPenalty(config, Semiring::Invalid);
203  if (wrdPenalty != Semiring::Invalid)
204  context->setPenalties(wrdPenalty, silPenalty);
205  else if (silPenalty != Semiring::Invalid)
206  Core::Application::us()->error("Cannot set silence penalty without setting word penalty");
207  // set merge penalties
208  context->setMergePenalties(paramMergePenalties(config));
209  // set eps-word
210  context->setEpsSymbol(paramEpsSymbol(config));
211  return context;
212 }

412 void HtkSlfBuilder::conformContext(const HtkSlfHeader& header) {
413  if (contextHandling_ & HtkSlfReader::ResetContext)
414  context_->clear();
415  if (contextHandling_ & HtkSlfReader::AdaptContext) {
416  adaptBase(header);
417  adaptSemiring(header);
418  adaptPenalties(header);
419  adaptOther(header);
420  }
421  if (!(context_->semiring())) {
422  status_ = htkNoSemiring;
423  return;
424  }
425  // do some pre-processing
426  if ((context_->base() == -1.0) || (context_->base() == 0.0))
427  scale_ = context_->base();
428  else
429  scale_ = -::log(context_->base());
430  wrdPenalty_ = (context_->wordPenalty() == Semiring::One) ? Semiring::One : convert(context_->wordPenalty());
431  silPenalty_ = (context_->silPenalty() == Semiring::One) ? Semiring::One : convert(context_->silPenalty());
432 }

434 HtkSlfBuilder::Status HtkSlfBuilder::header(const StringPairList& props, HtkSlfHeader& header) {
435  s32 required = 2;
436  header.reset();
437 
438  for (StringPairList::const_iterator it = props.begin();
439  it != props.end(); ++it) {
440  switch (it->first.at(0)) {
441  case 'V':
442  if (!Core::strconv(it->second, header.version))
443  error("Unable to parse \"%s\".", it->second.c_str());
444  break;
445  case 'U':
446  if (!Core::strconv(it->second, header.utterance))
447  error("Unable to parse \"%s\".", it->second.c_str());
448  break;
449  case 'S':
450  warning("Sub-Lattices are not supported.");
451  break;
452  case 'N':
453  if (!Core::strconv(it->second, header.nNodes))
454  error("Unable to parse #nodes \"%s\".", it->second.c_str());
455  else
456  --required;
457  break;
458  case 'L':
459  if (!Core::strconv(it->second, header.nLinks))
460  error("Unable to parse #links \"%s\".", it->second.c_str());
461  else
462  --required;
463  break;
464  default:
465  if (it->first == "base") {
466  if (!Core::strconv(it->second, header.base))
467  error("Unable to parse base \"%s\".", it->second.c_str());
468  }
469  else if (it->first == "lmname") {
470  if (!Core::strconv(it->second, header.lmName))
471  error("Unable to parse name \"%s\".", it->second.c_str());
472  }
473  else if (it->first == "lmscale") {
474  if (!Core::strconv(it->second, header.lmScale))
475  error("Unable to parse scale \"%s\".", it->second.c_str());
476  }
477  else if (it->first == "wdpenalty") {
478  if (!Core::strconv(it->second, header.wrdPenalty)) {
479  error("Unable to parse penalty \"%s\".", it->second.c_str());
480  header.wrdPenalty = Semiring::One;
481  }
482  }
483  else if (it->first == "silpenalty") {
484  if (!Core::strconv(it->second, header.silPenalty)) {
485  error("Unable to parse penalty \"%s\".", it->second.c_str());
486  header.silPenalty = Semiring::Invalid;
487  }
488  }
489  }
490  }
491  if (header.silPenalty == Semiring::Invalid)
492  header.silPenalty = header.wrdPenalty;
493  return (required) ? htkParseHeaderError : htkOk;
494 }

496 Fsa::LabelId HtkSlfBuilder::lemmaPronunciationId(std::string& symbol, s32 variant) const {
497  if (symbol == context_->epsSymbol()) {
498  if (variant != -1)
499  warning("Expect variant -1 for epsilon symbol \"%s\", found %d.",
500  symbol.c_str(), variant);
501  return Fsa::Epsilon;
502  }
503  Fsa::LabelId label = lpAlphabet_->specialIndex(symbol);
504  if (label != Fsa::InvalidLabelId) {
505  if (label != Fsa::Epsilon)
506  error("No special symbols besides epsilon are treated correctly; found special symbol \"%s\".",
507  symbol.c_str());
508  else if (variant != -1)
509  warning("Expected variant -1 for special symbol \"%s\", found %d.",
510  symbol.c_str(), variant);
511  return label;
512  }
513  if (context_->capitalize())
514  Core::convertToUpperCase(symbol);
515  if (variant == -1)
516  variant = 0;
517  label = lexicon_->lemmaPronunciationId(symbol, variant);
518  if (label == Fsa::InvalidLabelId) {
519  warning("Could not find any pronunciation for word \"%s\", substitute by unknown.",
520  symbol.c_str());
521  return context_->unknown();
522  }
523  return label;
524 }

530 HtkSlfBuilder::Status HtkSlfBuilder::node(const StringPairList& props) {
531  s32 required = 2;
532  Fsa::StateId id = Fsa::InvalidStateId;
533  Boundary boundary;
534  std::string label;
535  s32 variant = -1;
536 
537  for (StringPairList::const_iterator it = props.begin();
538  it != props.end(); ++it) {
539  switch (it->first.at(0)) {
540  case 'I':
541  if (!Core::strconv(it->second, id))
542  error("Unable to parse node id \"%s\".", it->second.c_str());
543  else
544  --required;
545  break;
546  case 't':
547  f32 time;
548  if (!Core::strconv(it->second, time))
549  error("Unable to parse time \"%s\".", it->second.c_str());
550  else {
551  --required;
552  boundary.setTime(timeframeIndex(time));
553  }
554  break;
555  case 'L':
556  warning("Sub-lattices are not supported.");
557  break;
558  case 'W':
559  if (!Core::strconv(it->second, label))
560  error("Unable to parse label \"%s\".", it->second.c_str());
561  break;
562  case 'v':
563  if (!Core::strconv(it->second, variant))
564  error("Unable to parse variant \"%s\".", it->second.c_str());
565  break;
566  case 'd':
567  if (id == 0) // We assume that the "I" property always occurs before the "d" property
568  warning("No support for the div field.");
569  break;
570  }
571  }
572  if (required)
573  return htkParseNodeError;
574 
575  if (id >= header_.nNodes)
576  return htkAddNodeError;
577  if (lat_->hasState(id))
578  return htkAddNodeError;
579  State* sp = lat_->createState(id);
580  lat_->setState(sp);
581  boundaries_->set(id, boundary);
582  nodes_[id].sp = sp;
583  if (!label.empty()) {
584  if ((nodes_[id].labelId = lemmaPronunciationId(label, variant)) == Fsa::InvalidLabelId) {
585  error("Could not find any pronunciation for orthography %s (node %d).",
586  label.c_str(), id);
587  return htkAddNodeError;
588  }
589  }
590  return htkOk;
591 }

593 HtkSlfBuilder::Status HtkSlfBuilder::link(const StringPairList& props) {
594  s32 required = 3;
595  Fsa::StateId linkId = Fsa::InvalidStateId;
596  Fsa::StateId sourceId, targetId;
597  Score amScore = Semiring::One;
598  Score lmScore = Semiring::One;
599  std::string label;
600  s32 variant = -1;
601  for (StringPairList::const_iterator it = props.begin();
602  it != props.end(); ++it) {
603  switch (it->first.at(0)) {
604  case 'J':
605  if (!Core::strconv(it->second, linkId))
606  error("Unable to parse id \"%s\".", it->second.c_str());
607  else
608  --required;
609  break;
610  case 'S':
611  if (!Core::strconv(it->second, sourceId))
612  error("Unable to parse id \"%s\".", it->second.c_str());
613  else
614  --required;
615  break;
616  case 'E':
617  if (!Core::strconv(it->second, targetId))
618  error("Unable to parse id \"%s\".", it->second.c_str());
619  else
620  --required;
621  break;
622  case 'a':
623  if (!Core::strconv(it->second, amScore))
624  error("Unable to parse score \"%s\".", it->second.c_str());
625  else
626  amScore = convert(amScore);
627  break;
628  case 'l':
629  if (!Core::strconv(it->second, lmScore))
630  error("Unable to parse score \"%s\".", it->second.c_str());
631  else
632  lmScore = convert(lmScore);
633  break;
634  case 'W':
635  if (!Core::strconv(it->second, label))
636  error("Unable to parse label \"%s\".", it->second.c_str());
637  break;
638  case 'v':
639  if (!Core::strconv(it->second, variant))
640  error("Unable to parse variant \"%s\".", it->second.c_str());
641  break;
642  case 'd':
643  if (linkId == 0) // We assume that the "J" field always occurs before the "d" field
644  warning("No support for the div field.");
645  break;
646  case 'n':
647  if (linkId == 0) // We assume that the "J" field always occurs before the "n" field
648  warning("No support for the ngram field.");
649  break;
650  }
651  }
652  if (required)
653  return htkParseLinkError;
654 
655  if (linkId >= header_.nLinks)
656  return htkAddLinkError;
657  if ((sourceId >= header_.nNodes) || (!(nodes_[sourceId].sp)))
658  return htkAddLinkError;
659  if ((targetId >= header_.nNodes) || (!(nodes_[targetId].sp)))
660  return htkAddLinkError;
661 
662  Fsa::LabelId labelId;
663  if (label.empty()) {
664  labelId = (context_->type() == HtkSlfForward) ? nodes_[targetId].labelId : nodes_[sourceId].labelId;
665  }
666  else {
667  labelId = lemmaPronunciationId(label, variant);
668  }
669  if (labelId == Fsa::InvalidLabelId) {
670  error("Could not find any pronunciation for orthography %s (link %d).",
671  label.c_str(), linkId);
672  return htkAddLinkError;
673  }
674  Score penalty = 0.0;
675  if ((Fsa::FirstLabelId <= labelId) && (labelId <= Fsa::LastLabelId))
676  penalty = (labelId == context_->silence()) ? silPenalty_ : wrdPenalty_;
677  ScoresRef scores = context_->semiring()->create();
678  if (context_->mergePenalty()) {
679  scores->set(context_->amId(), amScore + penalty);
680  }
681  else {
682  scores->set(context_->amId(), amScore);
683  scores->set(context_->penaltyId(), penalty);
684  }
685  scores->set(context_->lmId(), lmScore);
686  if (!nodes_[sourceId].sp->newArc(targetId, scores, Fsa::LabelId(labelId)))
687  return htkAddLinkError;
688  return htkOk;
689 }

691 HtkSlfBuilder::Status HtkSlfBuilder::setInitialAndFinalStates() {
692  if (nodes_.empty() || !nodes_[0].sp)
693  return htkNoInitialStateError;

695  bool hasFinalState = false;
696  for (NodeList::iterator it = nodes_.begin(); it != nodes_.end(); ++it)
697  if (it->sp && !it->sp->hasArcs()) {
698  it->sp->setFinal(lat_->semiring()->clone(lat_->semiring()->one()));
699  hasFinalState = true;

701  if (!hasFinalState)
702  return htkNoFinalStateError;

706 HtkSlfBuilder::Status HtkSlfBuilder::setCoarticulation() {
707  Lexicon::LemmaPronunciationAlphabetRef lpAlphabet = Lexicon::us()->lemmaPronunciationAlphabet();
708  Core::Ref<const Bliss::PhonemeInventory> phonemeInventory = Lexicon::us()->phonemeInventory();
709  Node& initial = nodes_[0];
710  initial.initialPhonemeId = initial.finalPhonemeId = Bliss::Phoneme::term;
711  Fsa::Stack<Fsa::StateId> stack;
712  stack.push(0);
713  while (!stack.isEmpty()) {
714  Fsa::StateId sourceId = stack.pop();
715  State* sp = nodes_[sourceId].sp;
716  Node& source = nodes_[sourceId];
717  if (!sp->hasArcs()) {
718  source.initialPhonemeId = Bliss::Phoneme::term;
719  continue;
720  }
721  Bliss::Phoneme::Id initialPhonemeId, finalPhonemeId;
722  for (State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
723  Fsa::StateId targetId = a->target();
724  Node& target = nodes_[targetId];
725  const Bliss::Pronunciation* p =
726  ((Fsa::FirstLabelId <= a->input()) && (a->input() <= Fsa::LastLabelId)) ? lpAlphabet->lemmaPronunciation(a->input())->pronunciation() : 0;
727  if (p && p->length()) {
728  initialPhonemeId = (*p)[0];
729  initialPhonemeId =
730  (phonemeInventory->phoneme(initialPhonemeId)->isContextDependent()) ? initialPhonemeId : Bliss::Phoneme::term;
731  finalPhonemeId = (*p)[p->length() - 1];
732  finalPhonemeId =
733  (phonemeInventory->phoneme(finalPhonemeId)->isContextDependent()) ? finalPhonemeId : Bliss::Phoneme::term;
734  }
735  else {
736  if (!target.sp->isFinal()) {
737  initialPhonemeId = source.finalPhonemeId;
738  finalPhonemeId = initialPhonemeId;
739  }
740  else {
741  // special treatment required for HTKs from FLF lattices
742  initialPhonemeId = finalPhonemeId = Bliss::Phoneme::term;
743  }
744  }
745  if (source.initialPhonemeId == InvalidPhonemeId)
746  source.initialPhonemeId = initialPhonemeId;
747  if ((initialPhonemeId == Bliss::Phoneme::term) || (initialPhonemeId != source.initialPhonemeId))
748  source.initialPhonemeId = source.finalPhonemeId = Bliss::Phoneme::term;
749  if (target.finalPhonemeId == InvalidPhonemeId)
750  target.finalPhonemeId = finalPhonemeId;
751  if ((finalPhonemeId == Bliss::Phoneme::term) || (finalPhonemeId != target.finalPhonemeId))
752  target.initialPhonemeId = target.finalPhonemeId = Bliss::Phoneme::term;
753  if (target.transit == TransitUnchecked) {
754  target.transit = TransitChecked;
755  stack.push(targetId);
756  }
757  }
758  }
759  for (NodeList::iterator it = nodes_.begin(); it != nodes_.end(); ++it)
760  if (it->transit != TransitUnchecked) {
761  Fsa::StateId sid = it->sp->id();
762  (*boundaries_)[sid].setTransit(
763  Boundary::Transit(it->finalPhonemeId, it->initialPhonemeId, AcrossWordBoundary));
764  }
765  lat_->addProperties(PropertyCrossWord);
766  return htkOk;
767 }

808 HtkSlfBuilder::Status HtkSlfBuilder::addProperty(const std::string& s, StringPairList& props) {
809  const char* c = s.c_str();
810  const char *keyStart, *keyEnd, *valueStart, *valueEnd;
811 
812  while (*c != '\0') {
813  for (; ::isspace(*c); ++c)
814  ;
815  if (*c == '\0')
816  break;
817  keyStart = c;
818 
819  for (; (*c != '=') && !::isspace(*c) && (*c != '\0'); ++c)
820  ;
821  if (*c == '\0')
822  return htkPropertyParseError;
823  keyEnd = c;
824 
825  for (; ::isspace(*c); ++c)
826  ;
827  if (*c != '=')
828  return htkPropertyParseError;
829  ++c;
830 
831  for (; ::isspace(*c); ++c)
832  ;
833  if (*c == '\0')
834  return htkPropertyParseError;
835 
836  if (*c == '"') {
837  valueStart = ++c;
838 
839  for (; (*c != '"') && (*c != '\0'); ++c)
840  ;
841  if (*c == '\0')
842  return htkPropertyParseError;
843  valueEnd = c;
844  ++c;
845  }
846  else {
847  valueStart = c;
848 
849  for (; !::isspace(*c) && (*c != '\0'); ++c)
850  ;
851  valueEnd = c;
852  }
853 
854  props.push_back(StringPair(
