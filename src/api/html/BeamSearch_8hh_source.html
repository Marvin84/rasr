39 void shortestPaths(const FstLib::Fst<A>& fst, typename A::Weight beam, FstLib::VectorFst<A>* lattice) {
40  typedef FstLib::Fst<A> Fst;
41  typedef typename A::StateId StateId;
42  typedef typename A::Weight Weight;
43  typedef typename A::Label Label;
44  typedef ShortestPathHyp<A> Hyp;
45  std::vector<Hyp> active, newActive;
46  verify(Weight::Properties() & FstLib::kPath);
47  FstLib::VectorFst<A> traceback;
48  typedef std::unordered_map<StateId, size_t> StateToHypMap;
49  StateToHypMap stateToHyp;
50  lattice->DeleteStates();
51  Hyp start;
52  start.state = fst.Start();
53  start.weight = Weight::One();
54  start.trace = traceback.AddState();
55  traceback.SetStart(start.trace);
56  active.push_back(start);
57  Hyp bestFinal;
58  bestFinal.state = FstLib::kNoStateId;
59  bestFinal.weight = Weight::Zero();
60  bestFinal.trace = traceback.AddState();
61  while (!active.empty() && bestFinal.state == FstLib::kNoStateId) {
62  // expand active hypotheses
63  Weight best = Weight::Zero();
64  for (typename std::vector<Hyp>::const_iterator hyp = active.begin(); hyp != active.end(); ++hyp) {
65  // determine best active final state if any
66  if (fst.Final(hyp->state) != Weight::Zero()) {
67  Weight finalWeight = FstLib::Times(hyp->weight, fst.Final(hyp->state));
68  if (bestFinal.weight != FstLib::Plus(bestFinal.weight, finalWeight)) {
69  bestFinal.weight = FstLib::Plus(bestFinal.weight, finalWeight);
70  bestFinal.state = hyp->state;
71  }
72  traceback.AddArc(bestFinal.trace, A(0, 0, fst.Final(hyp->state), hyp->trace));
73  }
74  // expand hypotheses
75  for (FstLib::ArcIterator<Fst> aiter(fst, hyp->state); !aiter.Done(); aiter.Next()) {
76  Hyp newHyp;
77  A arc = aiter.Value();
78  newHyp.state = arc.nextstate;
79  newHyp.weight = FstLib::Times(hyp->weight, arc.weight);
80  typename StateToHypMap::const_iterator i = stateToHyp.find(newHyp.state);
81  if (i != stateToHyp.end()) {
82  // recombine hypotheses
83  Hyp& h = newActive[i->second];
84  newHyp.trace = h.trace;
85  if (h.weight != FstLib::Plus(h.weight, newHyp.weight)) {
86  h = newHyp;
87  }
88  }
89  else {
90  stateToHyp.insert(typename StateToHypMap::value_type(newHyp.state, newActive.size()));
91  newHyp.trace = traceback.AddState();
92  newActive.push_back(newHyp);
93  }
94  // add backpointers
95  arc.nextstate = hyp->trace;
96  traceback.AddArc(newHyp.trace, arc);
97 
98  // keep track of the best score (for pruning)
99  if (best != FstLib::Plus(best, newHyp.weight))
100  best = FstLib::Plus(best, newHyp.weight);
101  } // for aiter
102  } // for hyp
103  // prune hypotheses
104  active.clear();
105  Weight threshold = FstLib::Times(best, beam);
106  for (typename std::vector<Hyp>::const_iterator hyp = newActive.begin(); hyp != newActive.end(); ++hyp) {
107  if (threshold != FstLib::Plus(hyp->weight, threshold))
108  active.push_back(*hyp);
109  }
110  /*! @todo garbage collection in the traceback: remove non-reachable states */
111  newActive.clear();
112  stateToHyp.clear();
113  } // while
114  if (bestFinal.state != FstLib::kNoStateId) {
115  // create lattice
116  traceback.SetFinal(traceback.Start(), Weight::One());
117  traceback.SetStart(bestFinal.trace);
118  FstLib::Connect(&traceback);
119  FstLib::Reverse(traceback, lattice);
120  }
121  else {
122  // no final state found
123  }
124 }

126 } // namespace OpenFst
