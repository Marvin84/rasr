49 void HMMStateNetwork::ChangePlan::apply() {
50  if (remove.empty() && add.empty())
51  return;
52 
53  SuccessorBatchId& list = structure->state(node).successors;
54 
55  std::set<StateId> targets;
56  std::set<u32> outputs;
57 
58  for (HMMStateNetwork::SuccessorIterator targetIt = structure->batchSuccessors(list); targetIt; ++targetIt) {
59  if (remove.find(*targetIt) != remove.end())
60  continue;
61  if (targetIt.isLabel()) {
62  outputs.insert(targetIt.label());
63  }
64  else {
65  targets.insert(*targetIt);
66  }
67  }
68 
69  for (std::set<int>::iterator it = add.begin(); it != add.end(); ++it) {
70  if (IS_LABEL(*it))
71  outputs.insert(LABEL_FROM_ID(*it));
72  else
73  targets.insert(*it);
74  }
75 
76  structure->clearOutputEdges(node);
77 
78  for (std::set<StateId>::const_iterator it = targets.begin(); it != targets.end(); ++it)
79  structure->addNodeToEdge(list, *it);
80 
81  for (std::set<u32>::iterator it = outputs.begin(); it != outputs.end(); ++it) {
82  verify(not IS_LABEL(*it));
83  structure->addOutputToEdge(list, ID_FROM_LABEL(*it));
84  }
85 }

141 u32 HMMStateNetwork::countReachableEnds(std::vector<u32>& counts, StateId node) const {
142  if (counts[node] == Core::Type<u32>::max) {
143  counts[node] = 0;
144 
145  for (HMMStateNetwork::SuccessorIterator targetIt = successors(node); targetIt; ++targetIt) {
146  if (targetIt.isLabel())
147  ++counts[node];
148  else
149  counts[node] += countReachableEnds(counts, *targetIt);
150  }
151  }
152 
153  return counts[node];

156 HMMStateNetwork::CleanupResult HMMStateNetwork::cleanup(std::list<Search::StateId> startNodes, Search::TreeIndex masterTree, bool clearDeadEnds, bool onlyBatches) {
157  if (clearDeadEnds && !onlyBatches) {
158  u32 deadEndNodes = 0;
159  std::vector<u32> reachableEnds(states_.size(), Core::Type<u32>::max);
160  for (StateId node = 1; node < states_.size(); ++node) {
161  countReachableEnds(reachableEnds, node);
162  if (reachableEnds[node] == 0) {
163  ++deadEndNodes;
164  clearOutputEdges(node);
165  }
166  }
167 
168  // Eventually remove dead nodes from predecessor batches
169  // Thereby they become unreachable and will be removed
170  u32 cleared = 0;
171  for (StateId node = 1; node < states_.size(); ++node) {
172  bool removed = false;
173  for (HMMStateNetwork::SuccessorIterator targetIt = successors(node); targetIt; ++targetIt) {
174  if (not targetIt.isLabel()) {
175  if (not successors(*targetIt)) {
176  ++cleared;
177  removeTargetFromNode(node, *targetIt);
178  removed = true;
179  break;
180  }
181  }
182  }
183  if (removed)
184  --node; //Process the same node again, as more targets may get removed
185  }
186 
187  Core::Application::us()->log() << "cleared " << cleared << " dead-end nodes";
188  }
189 
190  Core::Application::us()->log() << "total nodes before cleanup: " << states_.size();
191 
192  CleanupResult ret;
193  CountSizeTreeWalker counter(*this);
194  if (onlyBatches) {
195  for (TreeIndex tree = 1; tree < trees_.size(); ++tree)
196  counter.visitedTrees.insert(tree);
197  for (StateId node = 1; node < states_.size(); ++node)
198  counter.visited.insert(node);
199  }
200  else {
201  counter.visitedTrees.insert(masterTree);
202  counter.stopAtVisited = true;
203 
204  ///Mark all reachable trees and nodes
205  Core::Application::us()->log() << "calculating reachable nodes and trees";
206  for (std::list<StateId>::const_iterator it = startNodes.begin(); it != startNodes.end(); ++it)
207  counter.visit(*it, 1);
208  }
209 
210  {
211  std::vector<Tree> newTrees;
212  std::vector<StateId> newSubTreeListBatches;
213  std::vector<HMMState> newNodes;
214  std::vector<SuccessorBatchId> newEdgeTargetLists;
215 
216  //Must be created before adding the initial items, since it clears the lists
217  Tools::BatchManager<SubTreeListId, StateId, HMMState, true, InvalidBatchId> newSubTreeManager(newSubTreeListBatches, newNodes);
218 
219  newTrees.push_back(Tree());
220  newEdgeTargetLists.push_back(0);
221  newSubTreeListBatches.push_back(0);
222  newNodes.push_back(HMMState());
223 
224  std::vector<u32> orderBehind(states_.size(), 0);
225  std::vector<u32> follow(states_.size(), 0);
226 
227  std::vector<std::vector<StateId>> orderedPerTree;
228  orderedPerTree.push_back(std::vector<StateId>());
229 
230  for (u32 tree = 1; tree < trees_.size(); ++tree) {
231  /// @todo Build a topology and order the nodes in a stable way based on that
232  //Build the order so that the second-order batches are continuous
233  Tools::BatchManager<SubTreeListId, StateId, HMMState, true, Search::InvalidBatchId>::Iterator it = subTreeManager_.getIterator(trees_[tree].nodes);
234  for (; it; ++it) {
235  StateId node = *it;
236  if (counter.visited.count(node) == 0)
237  continue;
238  // 2nd order
239  u32 previousSkipTarget = 0;
240 
241  // first order
242  u32 previousTarget = 0;
243 
244  for (HMMStateNetwork::SuccessorIterator targetIt = successors(node); targetIt; ++targetIt) {
245  if (targetIt.isLabel())
246  break;
247 
248  StateId target = *targetIt;
249 
250  if (!orderBehind[target])
251  orderBehind[target] = previousTarget;
252  if (!follow[previousTarget])
253  follow[previousTarget] = target;
254  previousTarget = target;
255  verify(target < states_.size());
256 
257  for (HMMStateNetwork::SuccessorIterator skipTargetIt = successors(target); skipTargetIt; ++skipTargetIt) {
258  if (skipTargetIt.isLabel())
259  break;
260  StateId skipTarget = *skipTargetIt;
261  orderBehind[skipTarget] = previousSkipTarget;
262  follow[previousSkipTarget] = skipTarget;
263  previousSkipTarget = skipTarget;
264  }
265  }
266  }
267  std::vector<StateId> ordered;
268  std::unordered_set<StateId> had;
269 
270  {
271  Tools::BatchManager<SubTreeListId, StateId, HMMState, true, Search::InvalidBatchId>::Iterator it = subTreeManager_.getIterator(trees_[tree].nodes);
272  for (; it; ++it) {
273  StateId current = *it;
274 
275  if (counter.visited.count(current) == 0)
276  continue;
277 
278  if (onlyBatches) {
279  ordered.push_back(current);
280  }
281  else {
282  while (current) {
283  if (had.count(current))
284  break;
285 
286  ordered.push_back(current);
287  had.insert(current);
288  current = follow[current];
289  }
290  }
291  }
292  }
293 
294  orderedPerTree.push_back(ordered);
295  }
296 
297  for (u32 tree = 1; tree < trees_.size(); ++tree) {
298  if (counter.visitedTrees.find(tree) != counter.visitedTrees.end()) {
299  //Build the order so that the second-order batches are continuous
300  const std::vector<StateId>& ordered(orderedPerTree[tree]);
301 
302  //Transfer network into new list
303  ret.treeMap.insert(std::make_pair(tree, newTrees.size()));
304  newTrees.push_back(trees_[tree]);
305  newTrees.back().nodes = InvalidBatchId;
306 
307  //Transfer nodes into new batches
308 
309  for (u32 idx = 0; idx < ordered.size(); ++idx) {
310  StateId node = ordered[idx];
311  if (counter.visited.find(node) != counter.visited.end()) {
312  verify(newNodes.size() > 0);
313  StateId newNode = newSubTreeManager.appendOne(newTrees.back().nodes, states_[node]);
314  ret.nodeMap.insert(std::make_pair(node, newNode));
315  }
316  }
317  //No empty trees
318  verify(newTrees.back().nodes != InvalidBatchId);
319  }
320  else {
321  //This network is removed
322  }
323  }
324  Core::Application::us()->log() << "count of new nodes: " << newNodes.size();
325  verify(newNodes.size());
326  trees_.swap(newTrees);
327  states_.swap(newNodes);
328  subTreeListBatches_.swap(newSubTreeListBatches);
329  edgeTargetLists_.swap(newEdgeTargetLists);
330  }
331  verify(states_.size());
332 
333  std::vector<StateId> newEdgeTargetBatches;
334 
335  Tools::BatchManager<SuccessorBatchId, StateId, HMMState, false, InvalidBatchId, SingleSuccessorBatchMask> newEdgeTargetManager(newEdgeTargetBatches, states_);
336  //Must be created before adding the initial items, since it clears the lists
337  newEdgeTargetBatches.push_back(0);
338 
339  //Map the direct node members
340  for (u32 node = 1; node < states_.size(); ++node) {
341  std::unordered_map<TreeIndex, TreeIndex>::iterator it;
342 
343  //Map the edge-targets of single batches
344  SuccessorBatchId newBatch = InvalidBatchId;
345  SuccessorBatchId oldBatch = states_[node].successors;
346 
347  for (Tools::BatchManager<SuccessorBatchId, StateId, HMMState, false, Search::InvalidBatchId, SingleSuccessorBatchMask>::Iterator it = edgeTargetManager_.getIterator(oldBatch); it; ++it) {
348  if (IS_LABEL(*it)) {
349  //It's an label encoded as negative number
350  newEdgeTargetManager.appendToBatch(newBatch, *it, *it + 1);
351  }
352  else {
353  //It's a node
354  verify(counter.visited.find(*it) != counter.visited.end());
355  std::unordered_map<StateId, StateId>::const_iterator nodeMapIt = ret.nodeMap.find(*it);
356  verify(nodeMapIt != ret.nodeMap.end());
357  newEdgeTargetManager.appendToBatch(newBatch, (*nodeMapIt).second, (*nodeMapIt).second + 1);
358  }
359  }
360 
361  states_[node].successors = newBatch;
362  }
363 
364  for (u32 batchNum = 1; batchNum < edgeTargetLists_.size(); ++batchNum) {
365  if (edgeTargetLists_[batchNum]) {
366  SuccessorBatchId oldBatch = edgeTargetLists_[batchNum];
367  edgeTargetLists_[batchNum] = InvalidBatchId;
368 
369  for (Tools::BatchManager<SuccessorBatchId, StateId, HMMState, false, Search::InvalidBatchId, SingleSuccessorBatchMask>::Iterator it = edgeTargetManager_.getIterator(oldBatch); it; ++it) {
370  if (IS_LABEL(*it)) {
371  //It's an label encoded as negative number
372  newEdgeTargetManager.appendToBatch(edgeTargetLists_[batchNum], *it, *it + 1);
373  }
374  else {
375  //It's a node
376  verify(counter.visited.find(*it) != counter.visited.end());
377  std::unordered_map<StateId, StateId>::const_iterator nodeMapIt = ret.nodeMap.find(*it);
378  verify(nodeMapIt != ret.nodeMap.end());
379  newEdgeTargetManager.appendToBatch(edgeTargetLists_[batchNum], (*nodeMapIt).second, (*nodeMapIt).second + 1);
380  }
381  }
382  }
383  }
384 
385  edgeTargetBatches_.swap(newEdgeTargetBatches);
386  verify(states_.size());
387 
388  {
389  CountSizeTreeWalker counter2(*this);
390  counter2.stopAtVisited = true;
391 
392  ///Check again and make sure the same count of nodes is reachable
393  Core::Application::us()->log() << "re-calculating reachable nodes and trees";
394  for (std::list<StateId>::const_iterator it = startNodes.begin(); it != startNodes.end(); ++it) {
395  std::unordered_map<StateId, StateId>::const_iterator mapIt = ret.nodeMap.find(*it);
396  verify(mapIt != ret.nodeMap.end());
397 
398  StateId node = (*mapIt).second;
399 
400  counter2.visit(node, 1);
401  counter2.visitedTrees.insert(masterTree);
402  }
403  Core::Application::us()->log() << "previous reachable nodes: " << counter.visited.size() << " new reachable nodes: " << counter2.visited.size() << " new total nodes: " << states_.size();
404  Core::Application::us()->log() << "previous trees: " << counter.visitedTrees.size() << " new trees: " << counter2.visitedTrees.size();
405  Core::Application::us()->log() << "previous exits: " << counter.visitedFinalOutputs << " new exits: " << counter2.visitedFinalOutputs;
406  verify(counter2.visited.size() == counter.visited.size());
407  verify(counter2.visitedTrees.size() == counter.visitedTrees.size());
408  }
409 
410  return ret;
411 }
