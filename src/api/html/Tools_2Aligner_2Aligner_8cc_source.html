121 //extracts aligned words from a given linear automaton and stores them in a lexicon.
122 void Aligner::extract(Translation::ConditionalLexiconRef lexicon, ConstAutomatonRef f, int extractionOrder) {
123  Fsa::StateId initialState = f->initialStateId();
124  Fsa::ConstStateRef currentState(f->getState(initialState));
125 
126  std::vector<std::string> sourceVector;
127  std::vector<std::string> targetVector;
128  size_t delta;
129  std::vector<std::string> lexiconEntry(2 * (extractionOrder + 1), "<s>");
130 
131  while (!currentState->isFinal()) {
132  totalCost_ = totalCost_ + float(currentState->begin()->weight());
133 
134  //delta 0
135  if (currentState->begin()->output() == Epsilon) {
136  delta = 1;
137  sourceVector = Core::split(f->getInputAlphabet()->symbol(currentState->begin()->input()), "#");
138  targetVector = Core::split(f->getOutputAlphabet()->symbol(currentState->begin()->output()), "#");
139  }
140 
141  //delta 2
142  else if (currentState->begin()->input() == Epsilon) {
143  delta = 2;
144  sourceVector = Core::split(f->getInputAlphabet()->symbol(currentState->begin()->input()), "#");
145  targetVector = Core::split(f->getOutputAlphabet()->symbol(currentState->begin()->output()), "#");
146  }
147 
148  //delta 1
149  else {
150  delta = 0;
151  sourceVector = Core::split(f->getInputAlphabet()->symbol(currentState->begin()->input()), "#");
152  targetVector = Core::split(f->getOutputAlphabet()->symbol(currentState->begin()->output()), "#");
153  }
154 
155  lexiconEntry.erase(lexiconEntry.begin());
156  lexiconEntry.erase(lexiconEntry.begin());
157 
158  lexiconEntry.push_back(sourceVector[0]);
159  lexiconEntry.push_back(targetVector[0]);
160 
161  lexicon->addValue(delta, lexiconEntry, 1);
162  currentState = f->getState(currentState->begin()->target());
163  }
164 }

166 void Aligner::processFile(const std::string& sourceFilename,
167  const std::string& targetFilename) {
168  // initialize the lexicon used to generate the alignments
169  Translation::ConstConditionalLexiconRef conditionalLexicon;
170 
171  // this is confusing as it obvuscates the membership of the parameter lexicon.type
172  // however, it was doen for consistency, so that all lexicon parameters are given with
173  // the "lexicon" prefix
174  switch (paramLexiconType_(select("lexicon"))) {
175  case Translation::lexiconTypePlain:
176  conditionalLexicon = Translation::ConstConditionalLexiconRef(new Translation::ConditionalLexiconPlain(select("lexicon")));
177 
178  break;
179  case Translation::lexiconTypeSri:
180  conditionalLexicon = Translation::ConstConditionalLexiconRef(new Translation::ConditionalLexiconSri(select("lexicon")));
181  break;
182  default:
183  error("lexicon type unknown");
184  }
185 
186  Translation::ConditionalLexiconRef tmpLexicon;
187 
188  // iterate through the iterationsOrder vector, which is given as a parameter.
189  for (uint currentIteration = 0; currentIteration < iterationsOrder_.size() - 1; currentIteration = currentIteration + 2) {
190  // extraction the number of iterations per order out of the given iterationsOrder vector
191  uint numberOfIterations = iterationsOrder_[currentIteration];
192  uint extractionOrder = iterationsOrder_[currentIteration + 1];
193  uint nextExtractionOrder = 0;
194 
195  if (currentIteration == iterationsOrder_.size() - 2) {
196  nextExtractionOrder = extractionOrder;
197  }
198  else {
199  nextExtractionOrder = iterationsOrder_[currentIteration + 3];
200  }
201 
202  std::cerr << std::endl;
203 
204  // status information
205  std::cerr << "extraction order: " << extractionOrder << std::endl;
206  uint iteration = 0;
207  order_ = extractionOrder;
208 
209  // begin iteration
210  while (iteration < numberOfIterations) {
211  std::cerr << "iteration : " << iteration + 1;
212  std::cerr << "(" << numberOfIterations << ")" << std::endl;
213 
214  // instantiate the lexicon for storing the extracted entries
215  tmpLexicon = Translation::ConditionalLexiconRef(new Translation::ConditionalLexiconPlain(select("newlexicon")));
216 
217  Core::CompressedInputStream sourceStream(sourceFilename);
218  Core::CompressedInputStream targetStream(targetFilename);
219 
220  Translation::Reordering* reordering = new Translation::Reordering(select("reorder"));
221 
222  // main loop over the source and target corpora
223  while (sourceStream && targetStream && sentences_ <= segmentEnd_) {
224  // read sentences
225  std::string sourceSentence;
226  std::string targetSentence;
227  Core::getline(sourceStream, sourceSentence);
228  Core::getline(targetStream, targetSentence);
229 
230  // process only if in given sentence range and everything is ok with the sentences
231  if (sourceStream && targetStream && sentences_ >= segmentStart_ && sentences_ <= segmentEnd_ && sourceSentence.size() > 0 && targetSentence.size() > 0) {
232  // status information
233  if (sentences_ % 1000 == 0) {
234  std::cerr << "Sentence Number " << sentences_ << std::endl;
235  }
236 
237  /*
238  * create the Alignment Automaton, that is a composition of
239  * - the reordered source sentence
240  * - the general alignment automaton constructed from the sentences and lexicon
241  *
242  * the resulting automaton contains ALL POSSIBLE alignments between source and target
243  */
244  ConstAutomatonRef alignment(createAlignment(sourceSentence,
245  targetSentence,
246  conditionalLexicon,
247  reordering));
248 
249  // search for best alignment or substitute by empty automaton, if alignment is broken
250  // nbest output could be included here
251  if (!isEmpty(alignment)) {
252  alignment = best(cache(alignment));
253  }
254  else {
255  alignment = ConstAutomatonRef(staticCopy(std::string(" "), TropicalSemiring));
256  }
257 
258  // counting of alignments should happen here
259  if (iteration == numberOfIterations - 1) { //extraction with next order if at the end of the current iteration.
260  extract(tmpLexicon, alignment, nextExtractionOrder);
261  }
262  else {
263  extract(tmpLexicon, alignment, extractionOrder);
264  }
265 
266  // write best alignment to output
267  if ((currentIteration == iterationsOrder_.size() - 2) && (iteration == numberOfIterations - 1)) {
268  if (writeXml_)
269  writeXml(alignment, outputXml_);
270  if (writeBiLang_)
271  writeBiLang(alignment, outputBiLang_);
272  if (writeAachen_)
273  writeAachen(alignment, outputAachen_, sentences_);
274  }
275  }
276  sentences_++;
277  }
278 
279  sentences_ = segmentStart_;
280  std::cerr << "totalCost: " << totalCost_ << std::endl;
281  totalCost_ = 0;
282  iteration++;
283  // default normalization point 2
284  tmpLexicon->normalize(normalizePoint_);
285 
286  // switch lexica after every iteration
287  conditionalLexicon = tmpLexicon;
288  }
289 
290  totalCost_ = 0;
291  }
292 }

294 ConstAutomatonRef Aligner::createAlignment(std::string& sourceSentence,
295  std::string& targetSentence,
296  Translation::ConstConditionalLexiconRef conditionalLexicon,
297  Translation::Reordering* reordering) {
298  ConstAutomatonRef alignAutomaton(createAlignAutomaton(sourceSentence,
299  targetSentence,
300  conditionalLexicon));
301 
302  ConstAutomatonRef sourceAutomaton(staticCopy(numberTokens(sourceSentence), TropicalSemiring));
303 
304  sourceAutomaton = reordering->reorder(sourceAutomaton);
305  sourceAutomaton = multiply(sourceAutomaton, factorReorder_);
306 
307  if (useSourceLm_) {
308  sourceAutomaton = composeMatching(sourceAutomaton, sourceLmAutomaton_);
309  }
310 
311  ConstAutomatonRef alignment =
312  trim(
313  pruneSync(
314  composeMatching(
315  sourceAutomaton,
316  alignAutomaton),
317  beamThreshold_));
318 
319  return alignment;
320 }

405  useSourceLm_ = (sourceLmFilename_ != "");
