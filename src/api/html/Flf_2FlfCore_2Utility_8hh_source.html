58 /**
59  * Arc e with probability p and cost c:
60  * v := c * p
61  *
62  * Expectation semiring:
63  * (p1, v1) * (p2, v2) = (p1 * p2, p1 * v2 + p2 * v1)
64  * (p1, v1) + (p2, v2) = (p1 + p2, v1 + v2)
65  * One = (1, 0), Zero =(0, 0)
66  * => (p1, v1) * ... * (pN, vN) = (p1 * ... * pN, (p1 * ... * pN) * (c1 + ... + cN))
67  * => (p1, v1) + ... + (pN, vN) = (p1 + ... + pN, p1 * c1 + ... + cN * pN)
68  * Generalized Fwd/Bwd. probability (aka expected cost aka risk)
69  * for arc e with incoming (pIN, vIN) and outgoing (pOUT, vOUT)
70  * risk(e) = vIN + v + vOUT = (pIN * p * pOUT) * (cIN + c + cOUT)
71  *
72  * Cost semiring:
73  * (p1, c1) * (p2, c2) = (p1 * p2, c1 + c2)
74  * (p1, c1) + (p2, c2) = (p1 + p2, (p1 * c1 + p2 * c2) / (p1 + p2))
75  * One = (1, 0), Zero = (0, 0)
76  * => (p1, c1) * ... * (pN, cN) = (p1 * ... * pN, (c1 + ... + cN))
77  * => (p1, c1) + ... + (pN, cN) = (p1 + ... + pN, (p1 * c1 + ... + pN *cN) / (p1 + ... + pN))
78  * Generalized Fwd/Bwd. probability (aka expected cost aka risk)
79  * for arc e with incoming (pIN, cIN) and outgoing (pOUT, cOUT)
80  * risk(e) = (pIN * p * pOUT) * (cIN + c + cOUT)
81  *
82  **/
83 class CostCollector {
84 private:
85  typedef std::vector<std::pair<bool, f64>> CostList;
86  CostList expectedCosts_;
87 
88 public:
89  void reset();
90  // score -> negative logarithm of probability

92  void feed(f64 score, f64 cost);
93  // norm -> negative logarithm of probability
94  f64 get(f64 norm) const;
95  static CostCollector* create();
96 };
