78  void setHistory(InitItem& ii, const Token* newest, const Token* oldest) {
79  const Token *h, *t;
80  for (h = ii.history, t = newest; t != oldest && (*h == *t); ++h, ++t)
81  ;
82  if (*h == 0 && t == oldest)
83  return;
84  ii.history = histories.add0(newest, oldest);

88  setHistory(wsii, newest, oldest);

101 void ArpaLm::read() {
102  static const f64 ln10 = 2.30258509299404568402;
103  static const u32 maxNGramLength = 32;
104 
105  std::string filename(paramFilename(config));
106  log("reading ARPA language model from file \"%s\" ...", filename.c_str());
107 
108  std::string tempfile;
109 
110  if (paramReverseLm(config)) {
111  char* tmp = tempnam(0, "lm");
112  tempfile = tmp;
113  ::free(tmp);
114  log() << "reversing ARPA language model into temporary file '" << tempfile << "'";
115  Lm::reverseArpaLm(filename, tempfile);
116  log() << "successfully reversed";
117  filename = tempfile;
118  }
119 
120  Core::CompressedInputStream* cis = new Core::CompressedInputStream(filename.c_str());
121  Core::TextInputStream is(cis);
122  is.setEncoding(paramEncoding(config));
123 
124  if (!is) {
125  error("Failed to open language model file \"%s\".", filename.c_str());
126  return;
127  }
128  const bool includeZeroProb = !(paramSkipInfScore(config));
129 
130  InitData* data = new InitData;
131 
132  Core::MD5 md5;
133  std::string line;
134  u32 lineNumber = 0, totalNGrams = 0, expectedTotalNGrams = 0, maxTotalNGrams = 0;
135  enum { preamble,
136  sizes,
137  ngrams,
138  postamble,
139  unknown } state = preamble;
140  u32 nGram, n;
141  Token tokens[maxNGramLength];
142  Core::ProgressIndicator pi("reading ARPA lm", "n-grams");
143  pi.start();
144  while (!std::getline(is, line).eof()) {
145  ++lineNumber;
146  if (line.size() == 0)
147  continue;
148  else if (line[0] == '\\')
149  switch (state) { // handle section header
150  case preamble: {
151  if (line == "\\data\\")
152  state = sizes;
153  } break;
154  case sizes:
155  case ngrams:
156  case unknown:
157  if (line == "\\end\\")
158  state = postamble;
159  else if (sscanf(line.c_str(), "\\%u-gram:", &nGram) == 1) {
160  state = ngrams;
161  }
162  else {
163  warning("Unrecognized section starting in line %d", lineNumber);
164  state = unknown;
165  }
166  case postamble: break;
167  default: defect();
168  }
169  else
170  switch (state) { // handle other lines
171  case preamble: break;
172  case postamble: break;
173  case unknown: break;
174  case sizes: {
175  u32 nNGrams;
176  // u32 fraction = 3;
177  if (sscanf(line.c_str(), "ngram %u=%u", &nGram, &nNGrams) == 2) {
178  maxTotalNGrams += nNGrams;
179  /**
180  * Reserve memory only for a fraction of the LM,
181  * assume that some part of the LM is discarded because of a resricted vocabulary.
182  **/
183  // expectedTotalNGrams += nNGrams / fraction;
184  // data->reserve(expectedTotalNGrams, expectedTotalNGrams - nNGrams / fraction);
185  // fraction *= 3;
186  expectedTotalNGrams += nNGrams;
187  data->reserve(expectedTotalNGrams, expectedTotalNGrams - nNGrams);
188  }
189  pi.setTotal(expectedTotalNGrams);
190  } break;
191  case ngrams: {
192  md5.update(line);
193  std::istringstream lis(line);
194  f64 score;
195  std::string word;
196  if (!(lis >> score))
197  error("Expected float value in line %d", lineNumber);
198  hope(nGram <= maxNGramLength);
199  for (n = 0; n < nGram; ++n) {
200  if (lis >> word) {
201  Core::normalizeWhitespace(word);
202  Core::suppressTrailingBlank(word);
203  Token t = getToken(word);
204  if (t) {
205  tokens[n] = t;
206  }
207  else {
208  warning("unknown syntactic token '%s' in line %d", word.c_str(), lineNumber);
209  break;
210  }
211  }
212  }
213  if (n == nGram) {
214  std::reverse(tokens, tokens + n);
215  if (includeZeroProb || !Core::isAlmostEqual(score, InfScore, 0.1))
216  data->addScore(&tokens[1], &tokens[n], tokens[0], -ln10 * score);
217  if (lis >> score)
218  data->addBackOffScore(&tokens[0], &tokens[n], -ln10 * score);
219  ++totalNGrams;
220  }
221  pi.notify();
222  } break;
223  default: defect();
224  }
225  }
226  pi.finish();
227  if (state != postamble)
228  error("Premature end of language model file.");
229  dependency_.setValue(md5);
230  /*
231  log("%d/%d/%d loaded/expected/all n-grams", totalNGrams, expectedTotalNGrams, maxTotalNGrams);
232  */
233  initialize(&*data->items.begin(), &(*(data->items.end() - 1)) + 1);
234  delete data;
235 
236  if (tempfile.size())
237  std::remove(tempfile.c_str());
238 }
