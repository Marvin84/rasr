46 void StateSequence::createFromAllophone(Core::Ref<const Am::AcousticModel> model, const Am::Allophone* allophone) {
47  const Am::ClassicHmmTopology* hmmTopology = model->hmmTopology(allophone->central());
48  require(hmmTopology != 0);
49  int nPhoneStates = hmmTopology->nPhoneStates();
50  int nSubStates = hmmTopology->nSubStates();
51  int nStates = nPhoneStates * nSubStates;
52  flags_ = allophone->boundary;
53  states_.clear();
54  states_.reserve(nStates);
55  for (int s = 0; s < nStates; ++s) {
56  Am::AllophoneState as = model->allophoneStateAlphabet()->allophoneState(allophone, s / nSubStates);
57  Mm::MixtureIndex emission = model->emissionIndex(as);
58  Am::AcousticModel::StateTransitionIndex transition = model->stateTransitionIndex(as, s % nSubStates);
59  appendState(emission, transition);
60  }
61 }

81 bool StateSequenceList::read(const std::string& filename) {
82  Core::CompressedInputStream ifs(filename);
83  if (!ifs.good())
84  return false;
85  Core::BinaryInputStream bi(ifs);
86  char* buf = new char[strlen(MAGIC) + 1];
87  bi.read(buf, strlen(MAGIC) + 1);
88  if (strcmp(buf, MAGIC) != 0) {
89  Core::Application::us()->criticalError("cannot read header '%s', read '%s'", MAGIC, buf);
90  return false;
91  }
92  delete[] buf;
93  u32 s = 0, i = 0;
94  bi >> s;
95  resize(s);
96  for (iterator sequence = begin(); sequence != end(); ++sequence, ++i) {
97  if (!sequence->read(bi)) {
98  Core::Application::us()->criticalError("cannot read sequence %d", i);
99  return false;
100  }
101  }
102  return !ifs.fail();
103 }

113 void StateSequenceList::dump(Core::Ref<const Am::AcousticModel> am, const Bliss::LexiconRef lexicon,
114  Core::Channel& output) const {
115  const Am::ClassicAcousticModel* cam = dynamic_cast<const Am::ClassicAcousticModel*>(am.get());
116  verify(cam);
117  Am::ClassicStateTyingRef stateTying = cam->stateTying();
118  std::vector<std::list<Fsa::LabelId>> emissionToAllophoneState(stateTying->nClasses());
119  Am::ConstAllophoneStateAlphabetRef alloStateAlphabet = am->allophoneStateAlphabet();
120  for (std::pair<Am::AllophoneStateIterator, Am::AllophoneStateIterator> it = alloStateAlphabet->allophoneStates();
121  it.first != it.second; ++it.first) {
122  Mm::MixtureIndex emission = stateTying->classify(it.first.allophoneState());
123  emissionToAllophoneState[emission].push_back(it.first.id());
124  }
125  for (u32 s = 0; s < size(); ++s) {
126  const StateSequence& ss = (*this)[s];
127  output << s << " " << ss.isInitial() << " " << ss.isFinal() << " ";
128  for (u32 hmmState = 0; hmmState < ss.nStates(); ++hmmState) {
129  output << hmmState << "=(";
130  u32 t = ss.state(hmmState).transition_;
131  u32 m = ss.state(hmmState).emission_;
132  output << "t:" << t << " e:" << m << " ";
133  for (std::list<Fsa::LabelId>::const_iterator i = emissionToAllophoneState[m].begin();
134  i != emissionToAllophoneState[m].end(); ++i) {
135  output << alloStateAlphabet->symbol(*i) << " ";
136  }
137  output << ") ";
138  }
139  output << std::endl;
140  }
141 }

167 Fsa::LabelId AllophoneToAlloponeStateSequenceMap::stateSequenceIndex(Fsa::LabelId allophoneIndex) {
168  LabelTranslationMap::const_iterator i = labelMapping_.find(allophoneIndex);
169  if (i != labelMapping_.end())
170  return i->second;
171  Fsa::LabelId index;
172  if (allophoneAlphabet_->isDisambiguator(allophoneIndex)) {
173  if (removeDisambiguators_)
174  index = Fsa::Epsilon;
175  else {
176  index = getDisambiguator(nDisambiguators_);
177  nDisambiguators_++;
178  }
179  }
180  else {
181  const Am::Allophone* allophone = allophoneAlphabet_->allophone(allophoneIndex);
182  StateSequence stateSequence;
183  stateSequence.createFromAllophone(model_, allophone);
184  index = stateSequences_->index(stateSequence);
185  }
186  labelMapping_[allophoneIndex] = index;
187  return index;
188 }

222 void StateSequenceBuilder::build() {
223  typedef Am::AllophoneAlphabet::AllophoneList AllophoneList;
224  Core::Ref<const Am::AllophoneAlphabet> alphabet = am_->allophoneAlphabet();
225  allophoneToLabel_.resize(alphabet->nClasses());
226  if (addNonWords_) {
227  nonWordTokens_->createAllophones(alphabet);
228  allophoneToLabel_.resize(allophoneToLabel_.size() + nonWordTokens_->allophones().size());
229  }
230  const AllophoneList& allophoneList = alphabet->allophones();
231  for (AllophoneList::const_iterator ai = allophoneList.begin();
232  ai != allophoneList.end(); ++ai) {
233  Fsa::LabelId allophone = alphabet->index(*ai);
234  verify(!addNonWords_ || !nonWordTokens_->isNonWordPhone((*ai)->central()));
235  Fsa::LabelId label = map_->stateSequenceIndex(allophone);
236  addToMap(allophone, label);
237  }
238  for (u32 d = 0; d < nDisambiguators_; ++d) {
239  Fsa::LabelId amIndex = alphabet->disambiguator(d);
240  map_->stateSequenceIndex(amIndex);
241  }
242  if (addNonWords_) {
243  Fsa::LabelId label = map_->size();
244  const NonWordTokens::AllophoneMap& nonWordAllophones = nonWordTokens_->allophones();
245  for (NonWordTokens::AllophoneMap::const_iterator a = nonWordAllophones.begin();
246  a != nonWordAllophones.end(); ++a) {
247  Fsa::LabelId allophone = nonWordTokens_->allophoneId(a->second);
248  log("non word state sequences: %d -> label %d", allophone, label);
249  addToMap(allophone, label++);
250  }
251  }

271 void StateSequenceBuilder::addNonWordsToList(StateSequenceList* list) const {
272  Core::Ref<const Am::AllophoneAlphabet> alphabet = am_->allophoneAlphabet();
273  u32 nSequences = list->size();
274  const NonWordTokens::AllophoneMap& nonWordAllophones = nonWordTokens_->allophones();

314  OpenFst::SymbolTable* symbols = new OpenFst::SymbolTable("state-sequences");

350 StateSequenceList* HmmListConverter::creatStateSequenceList(const std::string& hmmListFile) const {
351  verify(stateSyms_ && hmmSyms_);
352  Core::CompressedInputStream cis(hmmListFile);
353  if (!cis) {
354  error("cannot read %s", hmmListFile.c_str());
355  return 0;
356  }
357  std::string line;
358  u32 nLine = 0;
359  StateSequenceList* list = new StateSequenceList();
360  while (cis) {
361  ++nLine;
362  std::getline(cis, line);
363  Core::stripWhitespace(line);
364  if (line.empty())
365  continue;
366  if (line == ".eps" || line == ".wb")
367  continue;
368  std::vector<std::string> fields = Core::split(line, " ");
369  if (fields.size() <= 1) {
370  error("wrong format in line %d: '%s'", nLine, line.c_str());
371  }
372  else {
373  if (!addHmm(list, fields)) {
374  error("parse error in line %d", nLine);
375  }
376  }
377  }
378  log("created %zd state sequences", list->size());
379  return list;
380 }

382 bool HmmListConverter::addHmm(StateSequenceList* list, const std::vector<std::string>& fields) const {
383  s32 hmm = hmmSyms_->Find(fields[0]);
384  if (hmm < 0) {
385  error("unknown hmm symbol: '%s'", fields[0].c_str());
386  return false;
387  }
388  hmm += hmmOffset;
389  if (hmm >= list->size()) {
390  list->resize(hmm + 1);
391  }
392  DBG(1) << VAR(fields[0]) << " " << VAR(hmm) << ENDDBG;
393  std::string phoneSym = Core::split(fields[0], "_")[0];
394  const bool isSilence = (phoneSym == silencePhone_);
395  if (isSilence) {
396  log("using silence transition model for hmm '%s'",
397  fields[0].c_str());
398  }
399  StateSequence& states = (*list)[hmm];
400  verify(states.nStates() == 0);
401  for (u32 i = 1; i < fields.size(); ++i) {
402  u32 s = stateSyms_->Find(fields[i]);
403  if (s < 0) {
404  error("unknown state symbol: '%s'",
405  fields[i].c_str());
406  return false;
407  }
408  Am::AcousticModel::StateTransitionIndex trans =
409  isSilence ? Am::TransitionModel::silence
410  : Am::TransitionModel::phone0;
411  states.appendState(s + hmmStateOffset, trans);
412  }
413  return true;
414 }
