30 class DeterminizeNode : public FilterNode {
31  friend class Network;
32 
33 public:
34  static const Core::ParameterBool paramToLogSemiring;
35  static const Core::ParameterFloat paramAlpha;
36 
37 private:
38  bool toLogSemiring_;
39  f32 alpha_;
40  ConstSemiringRef lastSemiring_;
41  ConstSemiringRef logSemiring_;
42 
43 protected:
44  virtual ConstLatticeRef filter(ConstLatticeRef l) {
45  if (!l)
46  return ConstLatticeRef();
47  if (toLogSemiring_) {
48  if (!lastSemiring_ || (lastSemiring_.get() != l->semiring().get())) {
49  lastSemiring_ = l->semiring();
50  logSemiring_ = toLogSemiring(lastSemiring_, alpha_);
51  }
52  l = changeSemiring(l, logSemiring_);
53  }
54  l = determinize(l);
55  if (toLogSemiring_) {
56  l = changeSemiring(l, lastSemiring_);
57  }
58  return l;
59  }
60 
61 public:

63  : FilterNode(name, config), toLogSemiring_(false) {}
64  ~DeterminizeNode() {}

66  toLogSemiring_ = paramToLogSemiring(config);

71  }
