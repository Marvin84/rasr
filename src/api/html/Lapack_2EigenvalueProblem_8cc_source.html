66 void EigenvalueProblem::supportEigenvectors(
67  const Matrix<int, RowByRowStorage>& isuppz, Matrix<double>& eigenvectors) const {
68  size_t n = eigenvectors.nColumns();
69  for (size_t i = 0; i < n; ++i) {
70  size_t firstNoneZero = isuppz(i, 0);
71  verify_(firstNoneZero < n);
72  size_t lastNoneZero = isuppz(i, 1);
73  verify_(firstNoneZero <= lastNoneZero && lastNoneZero < n);
74  for (size_t j = 1; j < firstNoneZero; ++j)
75  eigenvectors(j - 1, i) = 0;
76  for (size_t j = lastNoneZero + 1; j <= n; ++j)
77  eigenvectors(j - 1, i) = 0;
78  }
79 }

480 std::complex<Math::EigenvalueProblem::ValueType> GenEigenProblemWithSchurDecomposition::calculateEigenvalue(std::complex<ValueType> alpha,
481  ValueType alphaMinimum,
482  ValueType beta,
483  ValueType betaMinimum,
484  std::string& warningDescription) const {
485  ValueType absAlpha = Core::abs(alpha);
486  ValueType absBeta = Core::abs(beta);
487  if (absAlpha < alphaMinimum) {
488  warningDescription =
489  Core::form("eigenvalue set to zero due to abs(alpha)=%le smaller %le (abs(beta)=%le)",
490  (double)absAlpha, (double)alphaMinimum, (double)absBeta);
491  return std::complex<ValueType>(0, 0);
492  }
493  else if (absBeta < betaMinimum) {
494  warningDescription =
495  Core::form("eigenvalue set to zero due to abs(beta)=%le smaller %le (abs(alpha)=%le)",
496  (double)absBeta, (double)betaMinimum, (double)absAlpha);
497  return std::complex<ValueType>(0, 0);
498  }
499  return alpha / beta;
500 }

502 void GenEigenProblemWithSchurDecomposition::copyEigenvectors(const Math::Vector<std::complex<ValueType>>& alphas,
503  const Matrix<ValueType>& v,
504  Math::Matrix<std::complex<ValueType>>& eigenvectors) const {
505  /*
506  * The eigenvectors v(j) are stored one
507  * after another in the columns of VR, in the same order as
508  * their eigenvalues. If the j-th eigenvalue is real, then
509  * v(j) = VR(:,j), the j-th column of VR. If the j-th and
510  * (j+1)-th eigenvalues form a complex conjugate pair, then
511  * v(j) = VR(:,j)+i*VR(:,j+1) and v(j+1) = VR(:,j)-i*VR(:,j+1).
512  * Each eigenvector will be scaled so the largest component have
513  * abs(real part) + abs(imag. part) = 1.
514  * Not referenced if JOBVR = 'N'.
515  */
516  require(alphas.size() == v.nColumns());
517 
518  eigenvectors.resize(v.nRows(), v.nColumns());
519  for (size_t i = 0; i < eigenvectors.nColumns(); ++i) {
520  if (alphas[i].imag() == 0) {
521  for (u32 d = 0; d < v.nRows(); ++d)
522  eigenvectors[d][i] = std::complex<ValueType>(v(d, i), 0);
523  }
524  else {
525  verify((i + 1) < eigenvectors.nColumns());
526  for (u32 d = 0; d < v.nRows(); ++d) {
527  eigenvectors[d][i] = std::complex<ValueType>(v(d, i), v(d, i + 1));
528  eigenvectors[d][i + 1] = std::complex<ValueType>(v(d, i), -v(d, i + 1));
529  }
530  ++i;
531  }
532  }
533 }

535 void GenEigenProblemWithSchurDecomposition::writeConditionNumbers(Core::XmlWriter& os,
536  const Math::Vector<std::complex<ValueType>>& alphas,
537  const Vector<double>& betas,
538  const Math::Vector<std::complex<ValueType>>& eigenvalues,
539  const Vector<double>& eigenvalueReciprocalConditionNumbers,
540  const Vector<double>& eigenvectorReciprocalConditionNumbers,
541  double oneNormOfBalancedA, double oneNormOfBalancedB,
542  double& leftMaxPerMinScaling, double& rightMaxPerMinScaling) const {
543  require(alphas.size() == betas.size());
544  require(alphas.size() == eigenvalues.size());
545  require(alphas.size() == eigenvalueReciprocalConditionNumbers.size());
546  require(alphas.size() == eigenvectorReciprocalConditionNumbers.size());
547 
548  Math::Vector<std::complex<ValueType>>::const_iterator alpha = alphas.begin();
549  Vector<double>::ConstIterator beta = betas.begin();
550  Math::Vector<std::complex<ValueType>>::const_iterator eigenvalue = eigenvalues.begin();
551  Vector<double>::ConstIterator e = eigenvalueReciprocalConditionNumbers.begin();
552  Vector<double>::ConstIterator v = eigenvectorReciprocalConditionNumbers.begin();
553 
554  double perturbance = Core::Type<double>::epsilon * sqrt(oneNormOfBalancedA * oneNormOfBalancedA + oneNormOfBalancedB * oneNormOfBalancedB);
