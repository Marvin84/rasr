69 void RpropDiscriminativeCovarianceEstimator::initialize(const CovarianceToMeanSetMap& meanSetMap) {
70  const CovarianceToMeanSetMap::MeanSet& meanSet = meanSetMap[Core::Ref<AbstractCovarianceEstimator>(this)];
71  buffer_.resize(accumulator_.size());
72  for (ComponentIndex cmp = 0; cmp < buffer_.size(); ++cmp) {
73  buffer_[cmp] = sum(meanSet, cmp);
74  }
75  weight_ = 0;
76  for (CovarianceToMeanSetMap::MeanSet::const_iterator m = meanSet.begin(); m != meanSet.end(); ++m) {
77  const Core::Ref<RpropDiscriminativeMeanEstimator> meanEstimator(required_cast(RpropDiscriminativeMeanEstimator*, m->get()));
78  const Weight tmpWeight = meanEstimator->weight();
79  weight_ += tmpWeight;
80  for (ComponentIndex cmp = 0; cmp < buffer_.size(); ++cmp) {
81  const Sum tmp1 = 2 * meanEstimator->sum(cmp);
82  const Sum tmp2 = meanEstimator->previousMean(cmp) * tmpWeight;
83  buffer_[cmp] += meanEstimator->previousMean(cmp) * (tmp2 - tmp1);
84  }
85  }
86  verify(weight_ != 0);
87 }
