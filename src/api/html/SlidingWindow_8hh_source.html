24 /** SlidingWindow over a sequence of data
25  *
26  * Usage:
27  * -call add to update the sliding window;
28  * -call flush to update the sliding window if no more input is available;
29  *
30  * -call out to obtain the element at the output point (also called 'right' or 'present');
31  * -call removed to obtain the removed element after the last call to add or flush;
32  * (Remark: neither out nor removed change the status of the sliding window)
33  * -call get to obtain the element at a index relative to the present position;
34  *
35  * -use the iterators and operators to reach any arbitrary element in the sliding window.
36  */

81  template<typename V>
82  Flow::DataPtr<Flow::Vector<V>> setMean(Flow::Vector<V>& vec) const {
83  Flow::DataPtr<Flow::Data> data;
84  Flow::DataPtr<Flow::Vector<V>> currentData;
85 
86  // number of present past and future frames
87  size_t validFirst = this->slidingWindow_.pastSize();
88  size_t validLast = this->slidingWindow_.futureSize();
89 
90  // resize the vector and initialize it (with zeros)
91  this->slidingWindow_.getClosest(0, data);
92  if (vec.size() != ((Flow::DataPtr<Flow::Vector<V>>)data)->size()) {
93  vec.resize(((Flow::DataPtr<Flow::Vector<V>>)data)->size());
94  }
95  std::fill(vec.begin(), vec.end(), (V)0);
96 
97  // sum up all values
98  for (int i = -(int)validFirst; i <= (int)validLast; i++) {
99  this->slidingWindow_.get(i, data);
100  currentData = (Flow::DataPtr<Flow::Vector<V>>)data;
101 
102  std::transform(currentData.get()->begin(), currentData.get()->end(), vec.begin(), vec.begin(),
103  std::plus<V>());
104  } // End for sum up
105 
106  // ... normalize the sum to get the mean Values
107  std::transform(vec.begin(), vec.end(), vec.begin(),
108  std::bind2nd(std::divides<V>(), (validLast + validFirst + 1)));
109 
110  return (Flow::DataPtr<Flow::Vector<V>>)(&vec);
111  }
112 
113  virtual void get(int relativeIndex, T& out) const {
114  Flow::Vector<f32>* meanVec = new Flow::Vector<f32>();

400 bool SlidingWindow<T>::init(size_t maxSize, size_t right) {
401  // special case of collecting a whole segment
402  if (maxSize >= (size_t)Core::Type<s32>::max && right >= (size_t)Core::Type<s32>::max)
