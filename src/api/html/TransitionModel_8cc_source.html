155 /**
156  * Apply TransitionModel to a flat automaton.
157  *
158  * What is does: The purpose of this algorithm is to add loop and skip
159  * transitions to a "flat" automaton. (Flat meaning that it does not
160  * contain loops and skips.) The emission labels, i.e. the labels
161  * that are repeated or skipped are on the input side of the
162  * automaton, while the output labels will be unchanged.
163  *
164  * This can be viewed as a specialized compose algorithm for the
165  * time-distortion transducer (left) and a given automaton (right).
166  * If you read on, you will discover that considerable care must be
167  * taken to creating compact results.
168  *
169  * How it works: The state space is expanded so that we remember the
170  * most recent emission label, this is called "left state" in the
171  * following. "Right state" refers to the corresponding state in the
172  * original automaton. This expansion is necessary to provide the
173  * loop transitions. The representation of the left state is rather
174  * verbose. It consists of a mask stating which kinds of transition
175  * are possible, a reference to the state's transition model, and of
176  * course the most recent emission label. In fact only a small number
177  * of combinations of the possible values are actually used. (One
178  * could slim down the data structure to represent only the valid
179  * combinations. However priority was given to clarity and
180  * maintainability of the code, over the small increase in efficiency.)
181  * The function isStateLegitimate() specifies which potential states
182  * can be used. It is good to make these constraints as tight as
183  * possible in order to ensure the result automaton does not contain
184  * unnecessary states.
185  *
186  * The most recent emission label may be empty (Fsa::Epsilon). We
187  * call this a "discharged" state. This happens for three reasons: 1)
188  * At the word start no emission label has been consumed. 2) After
189  * processing an input epsilon arc. 3) In some situations we
190  * deliberately forget the emission label (see below).
191  *
192  * In the expanded state space, loop transitions are simple to
193  * implement. (In "discharged" states they are not allowed.)
194  * Concerning the other (forward, exit and skip) transitions, there is
195  * a little twist: When a right state has multiple incoming and
196  * outgoing arcs, we choose to first discharge the recent-most
197  * emission label by going to an appropriate left state via an epsilon
198  * transition. The alternative would be to avoid the epsilon
199  * transition and directly connect to all successor states. However,
200  * in practice this would dramatically increase the total number of
201  * arcs needed. So discharging is the preferable alternative. The
202  * discharged state can be thought of as the state when we have
203  * decided to leave the current state, but not yet chosen where to go
204  * to. As mentioned before, the appropriate set of transition weights
205  * is recorded, so that we know what to do when we forward or skip
206  * from the discharged state.
207  *
208  * Concerning skips: In general, a skip consists of two transitions:
209  * First an epsilon transition goes to an intermediate state that
210  * allows a forward only, and then another transition leads to the
211  * target state. In "favorable" situations this is optimized into a
212  * single transitions (skip optimization). If you have read so far,
213  * you are certainly able to figure out what these favorable
214  * conditions are.
215  *
216  * As you noticed, there is some freedom in designing the discharge
217  * transitions. It turns out that compact results can be obtained by
218  * combining the forward discharge with the intermediate skip states,
219  * and to combine exit and loop discharge states.
220  *
221  * Any disambiguator label is interpreted as a word boundary and is
222  * given the following special treatment: No loop transition, since
223  * the word boundary cannot be repeated. No skip transitions: The word
224  * boundary cannot be skipped and the final state before the boundary
225  * cannot be skipped. The latter is done for consistency with
226  * WordConditionedTreeSearch.
227  *
228  * Once the state space is constructed as describe above, it is
229  * relatively straight forward to figure out, which transition weight
230  * (aka time distortion penalty or TDP) must be applied to which arc.
231  * Unfortunately the current scheme is not able to distinguish phone-1
232  * from phone-2 states. This will require additional state space
233  * expansion by counting repetitive emission labels. Alternatively,
234  * and probably simpler, we change the labels to allow the distinction
235  * between phone-1 and phone-2.
236  *
237  * Todo:
238  * - An even better condition for discharging would be:
239  * (number of DIFFERENT incoming arc labels > 1) && (number of outgoing arcs > 1)
240  * Presently it is
241  * (number of incoming arcs > 1) && (number of outgoing arcs > 1).
242  * - Implement proper distinction between between phone-1 and phone-2 states.
243  * - Reconsider the transition model. (see discussion in Wiki)
244  * - It might be nice to convert this to lazy evaluation (should
245  * be straight forward).
246  **/

284  void add(Direction direction, Type type) {

373  bool isStateLegitimate(const State& s) const {
374  if (alphabet_->isDisambiguator(s.emission))
375  return false;
376  // word start state
377  if ((s.mask == (State::allowForward | State::allowSkip | State::allowExit | State::isFinal)) &&
378  (s.weights == TransitionModel::entryM1) &&
379  (s.emission == Fsa::Epsilon))
380  return true;
381  // normal emitting state
382  if ((s.mask == (State::allowForward | State::allowLoop | State::allowSkip | State::allowExit | State::isFinal)) &&
383  (s.weights >= TransitionModel::silence) &&
384  (s.emission != Fsa::Epsilon))
385  return true;
386  // discharged forward and intermediate skip state
387  if ((s.mask == (State::allowForward)) &&
388  (s.weights == State::noWeights) &&
389  (s.emission == Fsa::Epsilon))
390  return true;
391  // discharged skip and exit state
392  if ((s.mask == (State::allowSkip | State::allowExit)) &&
393  (s.weights != State::noWeights) &&
394  (s.emission == Fsa::Epsilon))
395  return true;
396  // post-epsilon state
397  if ((s.mask == (State::allowForward | State::allowSkip | State::allowExit | State::isFinal)) &&
398  (s.weights >= TransitionModel::silence) &&
399  (s.emission == Fsa::Epsilon))
400  return true;
401  return false;
402  }

404  Fsa::StateId getStateId(State::Mask m, Fsa::LabelId e, StateType t, Fsa::StateId r) {
405  State s(m, e, t, r);
406  StateMap::iterator i = states_.find(s);
407  if (i == states_.end()) {
408  verify(isStateLegitimate(s));
409  StackItem si(s, createState(s));

416  Core::Ref<Fsa::State> createState(const State& s) {
417  Core::Ref<Fsa::State> result = Core::ref(t_->newState());
418  Fsa::ConstStateRef sr;
419  bool isFinal = (s.mask & State::isFinal) && ((sr = in_->getState(s.right))->isFinal());
420  if (isFinal) {

465  void doForward(const StackItem& current, const Fsa::Arc* ra) {
466  require(!alphabet_->isDisambiguator(ra->input()));
467  require(ra->input() != Fsa::Epsilon);
468  current.result->newArc(getStateId(State::allowLoop | State::allowForward | State::allowSkip | State::allowExit | State::isFinal,
469  ra->input(),

477  require(current.emission != Fsa::Epsilon);
478  current.result->newArc(current.result->id(),

481  }

483  void doSkip(const StackItem& current, const Fsa::Arc* ra) {
484  require(!alphabet_->isDisambiguator(ra->input()));
485  require(ra->input() != Fsa::Epsilon);
486 
487  StateDegrees::Degree targetDegree = (*rightStateDegrees_)[ra->target()];
488  bool wouldSkipToDeadEnd = ((targetDegree(StateDegrees::outgoing, StateDegrees::emitting) +
489  targetDegree(StateDegrees::outgoing, StateDegrees::epsilon)) == 0);
490  if (wouldSkipToDeadEnd)
491  return;
492 
493  Fsa::Weight w = weight(current, StateTransitionModel::skip);
494  if (t_->semiring()->compare(w, t_->semiring()->max()) >= 0)
495  return;
496 
497  bool isEligbleForSkipOptimization = ((targetDegree(StateDegrees::outgoing, StateDegrees::disambiguating) == 0) &&
498  (targetDegree(StateDegrees::outgoing, StateDegrees::epsilon) == 0) &&
499  (targetDegree(StateDegrees::outgoing, StateDegrees::emitting) == 1));
500  Fsa::ConstStateRef rat;
501  const Fsa::Arc* ras = 0;
502  if (isEligbleForSkipOptimization) {
503  rat = in_->getState(ra->target());
504  verify(rat->nArcs() == 1);
505  ras = &*rat->begin();
506  if (ras->output() != Fsa::Epsilon)
507  isEligbleForSkipOptimization = false;
508  }
509 
510  Fsa::Arc* ca = current.result->newArc();
511  ca->weight_ = t_->semiring()->extend(ra->weight(), w);
512  if (isEligbleForSkipOptimization) {
513  verify(ras->input() != Fsa::Epsilon);
514  verify(!alphabet_->isDisambiguator(ras->input()));
515  ca->target_ = getStateId(State::allowLoop | State::allowForward | State::allowSkip | State::allowExit | State::isFinal,
516  ras->input(),
517  stateType(ras->input()),
518  ras->target());
519  ca->weight_ = t_->semiring()->extend(ca->weight_, ras->weight());
520  ca->input_ = ras->input();
521  }
522  else {
523  ca->target_ = getStateId(State::allowForward,
524  Fsa::Epsilon,
525  State::noWeights,
526  ra->target());
527  ca->input_ = Fsa::Epsilon;
528  }
529  ca->output_ = ra->output();
530  }
531 
532  void doExit(const StackItem& current, const Fsa::Arc* ra) {
533  require(alphabet_->isDisambiguator(ra->input()));
534  verify(!applyExitTransitionToFinalStates_);
535  current.result->newArc(getStateId(State::allowForward | State::allowSkip | State::allowExit | State::isFinal,
536  Fsa::Epsilon,
537  TransitionModel::entryM1,
538  ra->target()),
539  t_->semiring()->extend(ra->weight(), weight(current, StateTransitionModel::exit)),
540  ra->input(), ra->output());
541  }

543  void doDischarge(const StackItem& current) {
544  current.result->newArc(getStateId(State::allowForward,
545  Fsa::Epsilon,
546  State::noWeights,
547  current.right),
548  weight(current, StateTransitionModel::forward),
549  Fsa::Epsilon, Fsa::Epsilon);
550  current.result->newArc(getStateId(State::allowSkip | State::allowExit,
551  Fsa::Epsilon,
552  current.weights,
553  current.right),
554  t_->semiring()->one(),
555  Fsa::Epsilon, Fsa::Epsilon);
556  }

566 Fsa::ConstAutomatonRef TransitionModel::Applicator::apply(Fsa::ConstAutomatonRef input) {
567  require(alphabet_);
568  in_ = input;
569 
570  Core::ProgressIndicator pi("applying transition model", "states");
571  rightStateDegrees_ = new StateDegrees(in_, alphabet_);
572  rightStateDegrees_->dfs(&pi);
573 
574  t_ = Core::ref(new Fsa::StaticAutomaton);
575  t_->setType(input->type());
576  t_->setSemiring(input->semiring());
577  t_->setInputAlphabet(input->getInputAlphabet());
578  t_->setOutputAlphabet(input->getOutputAlphabet());
579 
580  Fsa::StateId initial = getStateId(State::allowForward | State::allowSkip | State::allowExit | State::isFinal,
581  Fsa::Epsilon,
582  TransitionModel::entryM1,
583  in_->initialStateId());
584  t_->setInitialStateId(initial);
585  pi.start();
586  while (!todo_.empty()) {
587  const StackItem current(todo_.top());
588  todo_.pop();
589  Fsa::ConstStateRef currentRight = in_->getState(current.right);
590 
591  StateDegrees::Degree degree = (*rightStateDegrees_)[current.right];
592  bool shouldDischarge = (degree(StateDegrees::incoming, StateDegrees::emitting) == StateDegrees::many) &&
593  ((degree(StateDegrees::outgoing, StateDegrees::emitting) == StateDegrees::many) ||
594  (degree(StateDegrees::outgoing, StateDegrees::disambiguating) == StateDegrees::many));
595 
596  // std::cerr << Core::form("expand: mask=%x\temission=%d\tweights=%d\tright=%zd\n", current.mask, current.emission, current.weights, current.right); // DEBUG
597 
598  if (current.mask & State::allowLoop)
599  doLoop(current);
600  if (current.emission != Fsa::Epsilon && shouldDischarge) {
601  doDischarge(current);
602  }
603  else {
604  for (Fsa::State::const_iterator ra = currentRight->begin(); ra != currentRight->end(); ++ra) {
605  if (ra->input() == Fsa::Epsilon) {
606  doEpsilon(current, &*ra);
607  }
608  else if (alphabet_->isDisambiguator(ra->input())) {
609  if (current.mask & State::allowExit)
610  doExit(current, &*ra);
611  }
612  else {
613  if (current.mask & State::allowForward)
614  doForward(current, &*ra);
615  if (current.mask & State::allowSkip)
616  doSkip(current, &*ra);
617  }
618  }
619  }
620  pi.notify(t_->size());
