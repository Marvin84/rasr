31 class ToWordBoundariesDfs : public Fsa::DfsState {
32 private:
33  Flf::ConstBoundariesRef boundaries_;
34  Core::Ref<Lattice::WordBoundaries> wordBoundaries_;
35 
36 public:
37  ToWordBoundariesDfs(Fsa::ConstAutomatonRef f, Flf::ConstBoundariesRef boundaries)
38  : Fsa::DfsState(f),
39  boundaries_(boundaries),
40  wordBoundaries_(new Lattice::WordBoundaries) {}
41  virtual void discoverState(Fsa::ConstStateRef sp) {
42  const Fsa::StateId s = sp->id();
43  const Flf::Boundary::Transit transit = boundaries_->transit(s);
44  Lattice::WordBoundary wordBoundary(
45  boundaries_->time(s),
46  std::make_pair(transit.final, transit.initial));
47  wordBoundaries_->set(s, wordBoundary);
48  }
49  Core::Ref<const Lattice::WordBoundaries> getWordBoundaries() {
50  dfs();
51  return wordBoundaries_;
52  }
53 };

55 Lattice::ConstWordLatticeRef toWordLattice(Flf::ConstLatticeRef l) {
56  Core::Ref<Lattice::WordLattice> result(new Lattice::WordLattice);
57  std::vector<Fsa::ConstAutomatonRef> fsas = Flf::toFsaVector(l);
58  require(!fsas.empty());
59  ToWordBoundariesDfs s(fsas.front(), l->getBoundaries());
60  result->setWordBoundaries(s.getWordBoundaries());
61  for (u32 i = 0; i < fsas.size(); ++i) {
62  result->setFsa(fsas[i], l->semiring()->key(i));
63  }
64  return result;
65 }

67 class FromWordBoundariesDfs : public Fsa::DfsState {
68 private:
69  Flf::StaticBoundaries* boundaries_;
70  Core::Ref<const Lattice::WordBoundaries> wordBoundaries_;
71 
72 public:
73  FromWordBoundariesDfs(Lattice::ConstWordLatticeRef lattice)
74  : Fsa::DfsState(lattice->part(0)),
75  boundaries_(new Flf::StaticBoundaries),
76  wordBoundaries_(lattice->wordBoundaries()) {}
77  virtual void discoverState(Fsa::ConstStateRef sp) {
78  const Fsa::StateId s = sp->id();
79  boundaries_->set(s, Flf::Boundary(wordBoundaries_->time(s),
80  Flf::Boundary::Transit(wordBoundaries_->transit(s).final,
81  wordBoundaries_->transit(s).initial)));
82  }
83  Flf::ConstBoundariesRef getBoundaries() {
84  dfs();
85  return Flf::ConstBoundariesRef(boundaries_);

89 Flf::ConstLatticeRef fromWordLattice(Lattice::ConstWordLatticeRef lattice) {
90  std::vector<Fsa::ConstAutomatonRef> fsas;
91  Flf::KeyList keys;
92  require(lattice->nParts() > 0);
93  for (size_t i = 0; i < lattice->nParts(); ++i) {
94  fsas.push_back(lattice->part(i));
95  keys.push_back(Flf::Key(lattice->name(i)));
96  }
97  Flf::ConstLatticeRef l = Flf::fromFsaVector(fsas,
98  Flf::Semiring::create(getSemiringType(fsas[0]->semiring()), //hack
99  fsas.size(),
100  Flf::ScoreList(fsas.size(), 1.0),
101  keys));
102  FromWordBoundariesDfs s(lattice);
103  l->setBoundaries(s.getBoundaries());
104  return l;
105 }

444 bool LatticeTransformNode::work(Flow::PortId p) {
445  Flow::DataPtr<Flow::DataAdaptor<Flf::ConstLatticeRef>> in;
446  if (!getData(0, in)) {
447  error("could not read port 0");
448  return putData(0, in.get());
449  }
450 
451  Flf::ConstLatticeRef l = in->data();
452  if (appendScore_) {
453  Flf::ScoreList scales = l->semiring()->scales();
454  scales.push_back(Flf::Score(1.0));
455  std::vector<Fsa::ConstAutomatonRef> fsas = Flf::toFsaVector(in->data());
456  fsas.push_back(transform(Flf::toFsa(l)));
457  Flf::ConstSemiringRef sr = Flf::cloneSemiring(l->semiring());
458  l = Flf::fromFsaVector(fsas, Flf::appendSemiring(sr));
459  }
460  else {
461  l = Flf::fromFsa(transform(toFsa(l)),
462  Flf::Semiring::create(l->semiring()->type(), 1),
463  0);
464  }
465  l->setBoundaries(in->data()->getBoundaries());
466 
467  Flow::DataAdaptor<Flf::ConstLatticeRef>* out = new Flow::DataAdaptor<Flf::ConstLatticeRef>();
468  out->data() = l;
469 
470  require(!getData(0, in));
471  return putData(0, out) && putData(0, in.get());
472 }

492 bool LatticeTransform2Node::work(Flow::PortId p) {
493  Flow::DataPtr<Flow::DataAdaptor<Flf::ConstLatticeRef>> in1;
494  if (!getData(0, in1)) {
495  error("could not read port 0");
496  return putData(0, in1.get());
497  }
498  Flow::DataPtr<Flow::DataAdaptor<Flf::ConstLatticeRef>> in2;
499  if (!getData(1, in2)) {
500  error("could not read port 1");
501  return putData(0, in2.get());
502  }
503  Flf::ConstLatticeRef l = in1->data();
504  if (appendScore_) {
505  Flf::ScoreList scales = l->semiring()->scales();
506  scales.push_back(Flf::Score(1.0));
507  std::vector<Fsa::ConstAutomatonRef> fsas = Flf::toFsaVector(in1->data());
508  fsas.push_back(transform(Flf::toFsa(in1->data()), Flf::toFsa(in2->data())));
509  Flf::ConstSemiringRef sr = Flf::cloneSemiring(l->semiring());
510  l = Flf::fromFsaVector(fsas, Flf::appendSemiring(sr));
511  }
512  else {
513  l = Flf::fromFsa(transform(Flf::toFsa(in1->data()), Flf::toFsa(in2->data())),
514  Flf::Semiring::create(l->semiring()->type(), 1),
515  0);
516  }
517  l->setBoundaries(in1->data()->getBoundaries());
518 
519  Flow::DataAdaptor<Flf::ConstLatticeRef>* out = new Flow::DataAdaptor<Flf::ConstLatticeRef>();
520  out->data() = l;
521  require(!getData(0, in1));
522  require(!getData(1, in2));
523  return putData(0, out) && putData(0, in1.get());
524 }
