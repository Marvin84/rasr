37  ZerogramAutomaton(Bliss::LexiconRef lexicon, Score score)
38  : LanguageModelAutomaton(lexicon) {
39  Fsa::State* state = new Fsa::State();
40  state->setFinal(Fsa::Weight(score));
41  Core::Ref<const Bliss::SyntacticTokenAlphabet> si = lexicon->syntacticTokenAlphabet();
42  for (Bliss::Lexicon::SyntacticTokenIterator i = lexicon_->syntacticTokens().first;
43  i < lexicon_->syntacticTokens().second; ++i)
44  state->newArc(0, Fsa::Weight(score), si->index(*i), si->index(*i));
45  state->sort(Fsa::byInput());
46  state_ = Fsa::ConstStateRef(state);
47  addProperties(Fsa::PropertySortedByInput);
48  setProperties(Fsa::PropertyLinear, Fsa::PropertyNone);
49  setProperties(Fsa::PropertyAcyclic, Fsa::PropertyNone);
50  }
51  virtual ~ZerogramAutomaton() {}
52  virtual Fsa::StateId initialStateId() const {
53  return 0;
54  }
55  virtual Fsa::ConstStateRef getState(Fsa::StateId s) const {
56  return state_;
57  }
58  virtual void releaseState(Fsa::StateId s) const {}
59 };
