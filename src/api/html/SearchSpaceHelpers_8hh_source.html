284  struct MeshHash {
285  inline size_t operator()(const WordEndHypothesisList::iterator weh) const {
286  size_t ret = MyStandardValueHash<u32>()(weh->transitState);
287 
288  if (meshHistoryPhones == 0)
289  return ret;
290  else if (meshHistoryPhones < 0)
291  return MyStandardValueHash<u32>()(ret + weh->pronunciation->id());
292 
293  int len = weh->pronunciation->pronunciation()->length();
294  int compareLen = std::min(len, meshHistoryPhones);
295  for (int a = 0; a < compareLen; ++a)
296  ret = MyStandardValueHash<u32>()(ret + weh->pronunciation->pronunciation()->phonemes()[len - compareLen + a]);
297  return ret;
298  }
299  };
300  struct MeshEquality {
301  inline bool operator()(const WordEndHypothesisList::iterator l,
302  const WordEndHypothesisList::iterator r) const {
303  if (l->transitState != r->transitState)
304  return false;
305  if (meshHistoryPhones == 0)
306  return true;
307  if (l->pronunciation == r->pronunciation)
308  return true;
309  if (meshHistoryPhones < 0)
310  return false;
311  int len1 = l->pronunciation->pronunciation()->length();
312  int len2 = r->pronunciation->pronunciation()->length();
313 
314  if ((len1 < meshHistoryPhones || len2 < meshHistoryPhones) && len1 != len2)
315  return false;
316 
317  int compareLen = std::min(len1, meshHistoryPhones);
318 
319  return memcmp(l->pronunciation->pronunciation()->phonemes() + len1 - compareLen,
320  r->pronunciation->pronunciation()->phonemes() + len2 - compareLen,
321  sizeof(Bliss::Phoneme::Id) * compareLen) == 0;
322  }
323  };
