22 DirectoryArchive::DirectoryArchive(const Core::Configuration& config, const std::string& path, AccessMode access)
23  : Archive(config, path, access) {
24  if (access & AccessModeWrite) {
25  if (!createDirectory(path))

27  }
28  else if (access & AccessModeRead) {
29  if (!isDirectory(path))
30  error("Directory \"%s\" does not exist", path.c_str());

35  error("Cannot access directory \"%s\": permission denied", path.c_str());
36  }

50 bool DirectoryArchive::probe(const std::string& file, const struct stat64& state, Sizes& sizes) const {
51  FILE* fp = fopen(joinPaths(path(), file).c_str(), "r");
52  if (!fp)
53  return false;
54 
55  u8 tmp[4];
56  sizes.setCompressed(0);
57  sizes.setUncompressed(0);
58  if (fread(tmp, 1, 2, fp) == 2) {
59  if ((tmp[0] == 0x1f) && (tmp[1] == 0x8b)) {
60  // assume gzip compressed file
61  fseek(fp, -4, SEEK_END);
62  if (fread(tmp, 1, 4, fp) == 4) {
63  sizes.setCompressed(state.st_size);
64  sizes.setUncompressed(tmp[0] | (u32(tmp[1]) << 8) | (u32(tmp[2]) << 16) | (u32(tmp[3]) << 24));
65  }
66  }
67  else
68  sizes.setUncompressed(state.st_size);
69  }
70  fclose(fp);
71  return true;
72 }

74 bool DirectoryArchive::discover(const std::string& file, Sizes& sizes) const {
75  struct stat64 state;
76  if (stat64(joinPaths(path(), file).c_str(), &state) != 0)
77  return false;
78  if (!S_ISREG(state.st_mode))

87 bool DirectoryArchive::read(const std::string& name, std::string& buffer) const {
88  require(hasAccess(AccessModeRead));
89  if (name.empty())
90  return false;
91  Sizes sizes;
92  if (!discover(name, sizes))
93  return false;
94  std::ifstream s(joinPaths(path(), name).c_str());
95  if (sizes.compressed())
96  return (bool)s.read(&buffer[0], sizes.compressed());
97  else
98  return (bool)s.read(&buffer[0], sizes.uncompressed());
99 }

101 bool DirectoryArchive::write(const std::string& name, const std::string& buffer, const Sizes& sizes) {
102  require(hasAccess(AccessModeWrite));
103  if (name.empty())
104  return false;
105  std::string file = joinPaths(path(), name);
106  std::string dir = directoryName(file);
107  if (!isDirectory(dir) && !createDirectory(dir))
108  error("Failed to create directory \"%s\".", dir.c_str());
109  std::ofstream s(file.c_str());
110  if (!s)
111  error("Failed to open file \"%s\" for writing.", file.c_str());
112  return (bool)s.write((const char*)&(buffer.c_str())[0], buffer.size());
