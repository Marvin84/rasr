102 bool AdaptedAcousticModel::setKey(const std::string& key) {
103  if (key == currentKey_)
104  return true;
105  currentKey_ = key;
106 
107  std::string mappedKey(key);
108  Core::StringHashMap<std::string>::const_iterator i = corpusKeyMap_.find(key);
109  if (i != corpusKeyMap_.end())
110  mappedKey = i->second;
111 
112  if (mappedKey == currentMappedKey_)
113  return true;
114  currentMappedKey_ = mappedKey;
115 
116  Core::Ref<Mm::MixtureSet> mixtureSet = Core::Ref<Mm::MixtureSet>(dynamic_cast<Mm::MixtureSet*>(Precursor::mixtureSet().get()));
117  if (!mixtureSet) {
118  error("Could not get mixture set from precursor.");
119  return false;
120  }
121 
122  // Do the adaptation
123  log("Adaptation key: ") << currentMappedKey_;
124 
125  if (!adaptorCache_.findForReadAccess(currentMappedKey_)) {
126  if (!adaptorEstimatorCache_.findForReadAccess(currentMappedKey_)) {
127  log("No adaptation data seen for key ") << currentMappedKey_ << "using unadapted references!\n";
128  }
129  else {
130  adaptorCache_.insert(currentMappedKey_, new Core::IoRef<Mm::Adaptor>(
131  (*adaptorEstimatorCache_.findForReadAccess(currentMappedKey_))->adaptor()));
132 
133  (*adaptorCache_.findForReadAccess(currentMappedKey_))->adaptMixtureSet(mixtureSet);
134  }
135  }
136  else {
137  (*adaptorCache_.findForReadAccess(currentMappedKey_))->adaptMixtureSet(mixtureSet);
138  }
139  mixtureSet_.reset();
140  adaptMixtureSet_ = mixtureSet;
141 
142  Core::Ref<Mm::ScaledFeatureScorer> featureScorer = Mm::Module::instance().createScaledFeatureScorer(select("mixture-set"),
143  Core::Ref<Mm::AbstractMixtureSet>(mixtureSet));
144  if (!featureScorer) {
145  error("Could not create feature scorer.");
146  return false;
147  }
148 
149  return setFeatureScorer(featureScorer);
150 }
