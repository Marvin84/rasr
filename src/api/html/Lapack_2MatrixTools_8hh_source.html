29 template<typename T, class P>
30 bool invert(Math::Matrix<T, P>& mat, bool critical = true) {
31  require(mat.nRows() == mat.nColumns());
32  int N = mat.nRows();
33  int* ipiv = new int[N];
34  T* contMatrix = new T[N * N];
35  int info;
36 
37  // Pack matrix in Fortran format
38  for (int i = 0; i < N; i++)
39  for (int j = 0; j < N; j++)
40  contMatrix[i + j * N] = mat[i][j];
41 
42  // LU decomposition
43  getrf(&N, &N, contMatrix, &N, ipiv, &info);
44  if (info) {
45  if (critical) {
46  Core::Application::us()->criticalError("LAPACK error: info=")
47  << info << " in xGETRF";
48  }
49  else {
50  Core::Application::us()->warning("LAPACK error: info=")
51  << info << " in xGETRF";
52  return false;
53  }
54  }
55 
56  // query and set buffer size
57  int lwork = -1;
58  T* work = new T[1];
59  getri(&N, contMatrix, &N, ipiv, work, &lwork, &info);
60  if (info) {
61  if (critical) {
62  Core::Application::us()->criticalError("LAPACK error: info=")
63  << info << " in xGETRI";
64  }
65  else {
66  Core::Application::us()->warning("LAPACK error: info=")
67  << info << " in xGETRI";
68  return false;
69  }
70  }
71  lwork = int(*work);
72 
73  delete[] work;
74  work = new T[lwork];
75 
76  // actual inverse calculation
77  getri(&N, contMatrix, &N, ipiv, work, &lwork, &info);
78  if (info) {
79  if (critical) {
80  Core::Application::us()->criticalError("LAPACK error: info=")
81  << info << " in xGETRI";
82  }
83  else {
84  Core::Application::us()->warning("LAPACK error: info=")
85  << info << " in xGETRI";
86  return false;
87  }
88  }
89 
90  // unpack into original matrix
91  for (int i = 0; i < N; i++)
92  for (int j = 0; j < N; j++)
93  mat[i][j] = contMatrix[i + j * N];
94 
95  delete[] ipiv;
96  delete[] contMatrix;
97  delete[] work;
98  return true;
99 }
