38  LevenshteinAutomaton(ConstAlphabetRef ref, ConstAlphabetRef test, f32 delCost, f32 insCost, f32 subCost, f32 corCost)
39  : inputAlphabet_(ref), outputAlphabet_(test) {
40  setProperties(PropertyStorage, PropertyStorage);
41 
42  AlphabetMapping mapping;
43  mapAlphabet(inputAlphabet_, outputAlphabet_, mapping, false);
44 
45  State* sp = new State(0, StateTagFinal, Weight(0.0));
46  for (Alphabet::const_iterator i = inputAlphabet_->begin(); i != inputAlphabet_->end(); ++i) {
47  if (mapping[LabelId(i)] != InvalidLabelId)
48  sp->newArc(0, Weight(corCost), LabelId(i), mapping[LabelId(i)]); // match
49  sp->newArc(0, Weight(delCost), LabelId(i), Epsilon); // deletion
50  for (Alphabet::const_iterator j = outputAlphabet_->begin(); j != outputAlphabet_->end(); ++j)
51  if (mapping[LabelId(i)] != j)
52  sp->newArc(0, Weight(subCost), LabelId(i), LabelId(j)); // substitution
53  }
54  for (Alphabet::const_iterator i = outputAlphabet_->begin(); i != outputAlphabet_->end(); ++i)
55  sp->newArc(0, Weight(insCost), Epsilon, LabelId(i)); // insertion
56  state_ = ConstStateRef(sp);
57  }

99  virtual void discoverState(ConstStateRef sp) {
100  for (State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
101  if (a->input() == Epsilon)
102  info_.del_++;
103  else {
104  if (a->output() == Epsilon)
105  info_.ins_++;
106  else if (a->input() != a->output())
107  info_.sub_++;
108  info_.total_++;
109  }
110  }
111  }
