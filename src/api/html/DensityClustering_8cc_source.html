45 void DensityClusteringBase::init(u32 dimension, u32 nDensities) {
46  dimension_ = dimension;
47  nDensities_ = nDensities;
48  clusterIndexForDensity_.resize(nDensities_, 0);
49  // verification to make sure that ClusterIndex can hold the cluster-indices
50  verify(nClusters_ - 1 <= Core::Type<ClusterIndex>::max);
51  if (nDensities_ < nClusters_) {
52  log() << "reducing number of clusters from " << nClusters_ << " to " << nDensities_ << " because there are too few densities";
53  nClusters_ = nDensities_;
54  }
55  verify(nClusters_ <= nDensities_);
56  verify(nSelected_ <= nClusters_);
57 }

59 bool DensityClusteringBase::load() {
60  Core::MappedArchiveReader is = Core::Application::us()->getCacheArchiveReader(paramCacheArchive(config), "density-clustering");
61  if (!is.good())
62  return false;
63 
64  if (!is.check<std::string>(FileMagic, "magic token") ||
65  !is.check<u32>(FileFormatVersion, "format version"))
66  return false;
67 
68  if (!readTypes(is)) {
69  warning("cannot read type information");
70  return false;
71  }
72 
73  if (!is.check<u32>(dimension_, "dimension") ||
74  !is.check<u32>(nClusters_, "number of clusters") ||
75  !is.check<u32>(nDensities_, "number of densities"))
76  return false;
77 
78  is >> clusterIndexForDensity_;
79  return readMeans(is);
80 }
