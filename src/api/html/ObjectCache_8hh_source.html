201 /**
202  * RandomObjectCacheList: list of ObjectCacheItems implementing the random strategy
203  *
204  * ObjectCacheItems are stored in a hash map by their keys.
205  * A deque is used to enable random access to keys and fast removal of keys.
206  *
207  * Each ObjectCacheItem points to its corresponding element in the deque.
208  * Erasing at the end of a deque is much faster then erasing somewhere
209  * in the middle of the deque:
210  * -Erasing an element swaps the element and the last element in the deque.
211  * -The corresponding ObjectCacheItem of the last element is updated to the
212  * -new position of its deque element. The new last element and the ObjectCacheItem
213  * -are erased.
214  * last() chooses a random element of the deque..
215  */
216 template<class _Key, class _Data, class _HashFunction, class _EqualKey>
217 class RandomObjectCacheList;

331 /** ObjectCache: implements the caching of larger objects.
332  *
333  * Datas are created by the ObjectCache.
334  * Datas can be accessed by their keys. Access is devided into readAccess
335  * and writeAccess to minimize the number of file operations.
336  * If more then maxSize_ object are referenced the last object in the CacheList
337  * is written back to its file if necesseary and it is removed from the memory.
338  * The last object is given by the cache strategy implemented by the CacheList. The
339  * files are stored in one directory. The file names are the keys of the Datas.
340  * Datas can be explicitly saved (writeBack) and explicitly removed (erase)
341  * from the memory.
342  * Store references to Datas with care! References are valid until the next access to
343  * the cache.
344  *
345  */

486 template<class List>
487 void ObjectCache<List>::writeBack(ListIterator item) {
488  if (item->second->dirty()) {
489  std::string filename = createFilename(item->first);
490 
491  std::string directory = directoryName(filename);
492  createDirectory(directory);
493  if (!isDirectory(directory)) {
494  error("Could not create directory '%s'.", directory.c_str());
495  return;
496  }
497 
498  BinaryOutputStream os(filename);
499  if (!os.isOpen() || !os.good()) {
500  error("Could not create file '%s'.", filename.c_str());
501  return;
502  }
503  item->second->write(os);
504  if (!os.good()) {
505  error("Could not save object into file '%s'.", filename.c_str());
506  return;
507  }
508 
509  if (dump_) {
510  XmlOutputStream xmlOs(filename + objectCacheDumpExtension);
511  xmlOs.setIndentation(4);
512  item->second->dump(xmlOs, item->first);
513  }
514  }
515  list_.erase(item);
516 }

526 template<class List>
527 typename ObjectCache<List>::ListIterator ObjectCache<List>::access(const Key& key) {
528  ListIterator item = list_.access(key);
529  if (item == list_.end()) {
530  BinaryInputStream is;
531  if (openInputStream(key, is)) {
532  while (list_.size() >= maxSize_)
533  writeBack(list_.last());
534 
535  item = list_.insert(key, new Data);
536  item->second->read(is);
537  if (!is.good())
538  error("Could not read object.");
539  }
540  }
541  return item;
542 }
