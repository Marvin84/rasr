77 void AbstractAcousticSegmentwiseTrainer::setSegmentwiseFeatureExtractor(Core::Ref<Speech::SegmentwiseFeatureExtractor> segmentwiseFeatureExtractor) {
78  require(segmentwiseFeatureExtractor);
79  segmentwiseFeatureExtractor_ = segmentwiseFeatureExtractor;
80  const std::string portName = paramPortName(config);
81  if (!portName.empty()) {
82  portId_ = segmentwiseFeatureExtractor_->addPort(portName);
83  if (portId_ == Flow::IllegalPortId) {
84  criticalError("Failed to retrieve output from flow network.");
85  }
86  }
87  const std::string accumulationPortName = paramAccumulationPortName(config);
88  if (!accumulationPortName.empty()) {
89  accumulationPortId_ = segmentwiseFeatureExtractor_->addPort(accumulationPortName);
90  if (accumulationPortId_ == Flow::IllegalPortId) {
91  criticalError("Failed to retrieve accumulation output from flow network.");
92  }
93  }
94  Precursor::setSegmentwiseFeatureExtractor(segmentwiseFeatureExtractor_);
95 }

120 AbstractAcousticSegmentwiseTrainer* AbstractAcousticSegmentwiseTrainer::createAbstractAcousticSegmentwiseTrainer(const Core::Configuration& config) {

122  case gaussianMixture:
