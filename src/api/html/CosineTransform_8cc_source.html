45 void CosineTransform::initNplusOneData(size_t inputSize,
46  size_t outputSize,
47  Math::UnaryAnalyticFunctionRef warpingFunction,
48  Math::UnaryAnalyticFunctionRef derivedWarpingFunction) {
49  transformation_.resize(outputSize, inputSize);
50  N_ = transformation_.nColumns() - 1;
51  for (size_t k = 0; k < transformation_.nRows(); ++k) {
52  transformation_[k][0] = 0.5;
53  transformation_[k][N_] = 0.5 * pow(-1, k);
54  for (size_t n = 1; n < N_; ++n) {
55  f64 omega = M_PI * n / N_;
56  transformation_[k][n] = cos(warpingFunction->value(omega) * k) * derivedWarpingFunction->value(omega);
57  }
58  }
59 }

61 void CosineTransform::initEvenAboutNminusHalf(size_t inputSize,
62  size_t outputSize,
63  Math::UnaryAnalyticFunctionRef warpingFunction,
64  Math::UnaryAnalyticFunctionRef derivedWarpingFunction) {
65  transformation_.resize(outputSize, inputSize);
66  N_ = transformation_.nColumns();
67  for (size_t k = 0; k < transformation_.nRows(); ++k) {
68  for (size_t n = 0; n < N_; ++n) {
69  f64 omega = M_PI * (n + 0.5) / N_;
70  transformation_[k][n] = cos(warpingFunction->value(omega) * k) * derivedWarpingFunction->value(omega);
71  }
72  }
73 }

75 void CosineTransform::apply(const std::vector<Value>& in, std::vector<Value>& out) const {
76  require_(in.size() == transformation_.nColumns());
77  out = transformation_ * in;
78  if (normalize_) {
79  verify_(N_ > 0);
80  std::transform(out.begin(), out.end(), out.begin(), std::bind2nd(std::divides<Value>(), N_));
81  }
82 }

122 void CosineTransformNode::init(size_t inputSize) {
123  if (inputSize == 0)
124  error("Empty input vector.");
125  if (outputSize_ == 0)
126  warning("Output size is set to zero.");
127  if (outputSize_ > inputSize)
128  /* One case where this can happen: In the cepstrum calculation of MFCCs,
129  * when we do the calculation for 8khz audio, with a filterbank width of 268.258,
130  * we get a 15-dim input vector here, and some people want to have a
131  * 16-dim output vector. This can cause trouble later in our LDA implementation
132  * or at other places because of the linear dependency.
133  * If there are cases where you legitimately want to have a smaller input size,
134  * maybe we could add an option to omit the warning.
135  */
136  warning("Output size (%zd) is bigger than input size (%zd).", outputSize_, inputSize);
137  f64 sampleRate = timeDomainSampleRate(inputSize);
138  respondToDelayedErrors();
139  Math::UnaryAnalyticFunctionRef warpingFunction = createWarpingFunction(sampleRate);
140  respondToDelayedErrors();
141 
142  CosineTransform::init(inputType_, inputSize, outputSize_, normalize_,
143  warpingFunction, shouldWarpDifferenctialUnit_);
144  needInit_ = false;
145 }

147 f64 CosineTransformNode::timeDomainSampleRate(size_t inputSize) {
148  f64 result = 0;
149  switch (inputType_) {

152  case evenAboutNminusHalf:
153  result = 2 * inputSize / frequencyDomainSampleRate_;

156  error("Sample rate (%f) is not set correctly.", result);
157  return result;

160 Math::UnaryAnalyticFunctionRef CosineTransformNode::createWarpingFunction(f64 sampleRate) {
161  require(sampleRate > 0);
162  Math::UnaryAnalyticFunctionRef result;
163 
164  Math::AnalyticFunctionFactory factory(select(paramWarpingFunction.name()));
165  factory.setSampleRate(sampleRate);
166  factory.setDomainType(Math::AnalyticFunctionFactory::normalizedOmegaDomain);
167  factory.setMaximalArgument(M_PI);
168 
169  result = factory.createIdentity();
170  std::string declaration = StringExpressionNode::value();
171  if (!declaration.empty()) {
172  result = factory.createUnaryFunction(declaration);
173  if (!result)
174  error("Failed to create warping function.");
175  else if (shouldWarpDifferenctialUnit_ && !result->derive()) {
176  result.reset();
177  error("Warping function not derivable.");
178  }
179  }
180  return result;
181 }
