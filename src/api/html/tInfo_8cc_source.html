23 template<class _Automaton>
24 class CountDfsState : public DfsState<_Automaton>, public Fsa::AutomatonCounts {
25  typedef DfsState<_Automaton> Precursor;
26 
27 public:
28  typedef typename _Automaton::State State;
29  typedef typename _Automaton::ConstStateRef ConstStateRef;
30  typedef typename _Automaton::ConstRef ConstAutomatonRef;
31 
32 public:
33  CountDfsState(ConstAutomatonRef f)
34  : Precursor(f) {
35  }
36  virtual void discoverState(ConstStateRef sp) {
37  if (sp->isFinal())
38  ++nFinals_;
39  ++nStates_;
40  if ((sp->id() > maxStateId_) || (maxStateId_ == Fsa::InvalidStateId))
41  maxStateId_ = sp->id();
42  for (typename State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
43  ++nArcs_;
44  if (a->input() == Fsa::Epsilon)
45  ++nIEps_;
46  if (a->output() == Fsa::Epsilon)
47  ++nOEps_;
48  if ((a->input() == Fsa::Epsilon) && (a->output() == Fsa::Epsilon))
49  ++nIoEps_;
50  if (a->input() == Fsa::Failure)
51  ++nIFail_;
52  if (a->output() == Fsa::Failure)
53  ++nOFail_;
54  if ((a->input() == Fsa::Failure) && (a->output() == Fsa::Failure))
55  ++nIoFail_;
56  }
57  }
58 };

235 template<class _Automaton>
236 void basicInfo(typename _Automaton::ConstRef f, Core::XmlWriter& o) {
237  o << Core::XmlFull("type", Fsa::TypeChoice[f->type()]) << Core::XmlFull("describe", f->describe());
238  std::string tmp;
239  for (Core::Choice::const_iterator i = Fsa::PropertyChoice.begin(); i != Fsa::PropertyChoice.end(); ++i) {
240  if (f->knowsProperty(i->value())) {
241  if (!tmp.empty())
242  tmp += " ";
243  if (!f->hasProperty(i->value()))
244  tmp += "!";
245  tmp += i->ident();
246  }
247  }
248  o << Core::XmlFull("properties", tmp);
249  o << Core::XmlFull("semiring", f->semiring()->name());
250  if (f->getInputAlphabet()) {
251  Fsa::AlphabetCounts c = Fsa::count(f->getInputAlphabet());
252  o << Core::XmlFull("input-labels", c.nLabels_);
253  }
254  if (f->type() == Fsa::TypeTransducer && f->getOutputAlphabet()) {
255  Fsa::AlphabetCounts c = Fsa::count(f->getOutputAlphabet());
256  o << Core::XmlFull("output-labels", c.nLabels_);
257  }
258  if (f->initialStateId() != Fsa::InvalidStateId)
259  o << Core::XmlFull("initial-state-id", f->initialStateId());
260 }

262 template<class _Automaton>
263 void info(typename _Automaton::ConstRef f, Core::XmlWriter& o, bool progress) {
264  o << Core::XmlOpen("fsa-info");
265  basicInfo<_Automaton>(f, o);
266  Fsa::AutomatonCounts c = count<_Automaton>(f, progress);
267  if (c.maxStateId_ != Fsa::InvalidStateId)
268  o << Core::XmlFull("max-state-id", c.maxStateId_);
269  o << Core::XmlFull("states", c.nStates_);
270  o << Core::XmlFull("arcs", c.nArcs_);
271  o << Core::XmlFull("final-states", c.nFinals_);
272  if (f->type() == Fsa::TypeAcceptor) {
273  o << Core::XmlFull("epsilon-arcs", c.nIEps_);
274  if (c.nIFail_ != 0)
275  o << Core::XmlFull("failure-arcs", c.nIFail_);
276  }
277  else if (f->type() == Fsa::TypeTransducer) {
278  o << Core::XmlFull("io-epsilon-arcs", c.nIoEps_);
279  o << Core::XmlFull("input-epsilon-arcs", c.nIEps_);
280  o << Core::XmlFull("output-epsilon-arcs", c.nOEps_);
281  if ((c.nIFail_ != 0) or (c.nOFail_ != 0)) {
282  o << Core::XmlFull("io-failure-arcs", c.nIoFail_);
283  o << Core::XmlFull("input-failure-arcs", c.nIFail_);
284  o << Core::XmlFull("output-failure-arcs", c.nOFail_);
285  }
286  }
287  size_t size = f->getMemoryUsed();
288  if (size)
289  o << Core::XmlFull("memory", size);
290  o << Core::XmlClose("fsa-info");
291 }
