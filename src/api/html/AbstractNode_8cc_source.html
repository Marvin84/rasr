46 bool AbstractNode::checkAndSetParameter(const std::string& name, const std::string& value) {
47  if (paramIgnoreUnknownParameters.match(name)) {
48  ignoreUnknownParameters_ = paramIgnoreUnknownParameters(value);
49  return true;
50  }
51  else {
52  std::string staticValue; /// It would be better to make sure that checkAndSetParameter is only called when parameters _changed_
53  if (config.get(name, staticValue) && staticValue.size()) {
54  if (value != staticValue)
55  warning() << "ignoring flow-parameter '" << name << "=" << value << "', because it was set through configuration: '" << staticValue << "'";
56  return true;
57  }
58  else {
59  return setParameter(name, value) || ignoreUnknownParameters_;
60  }
61  }
62 }

66 bool AbstractNode::addParameter(const std::string& name, const Core::StringExpression& value) {
67  std::string v;
68  if (value.isConstant()) {
69  if (!value.value(v))
70  defect();
71  return checkAndSetParameter(name, v);
72  }
73  std::pair<Parameters::iterator, bool> p =
74  parameters_.insert(Parameters::value_type(name, value));
75  if (p.second) {
76  if (p.first->second.value(v))
77  return checkAndSetParameter(p.first->first, v);
78  }
79  else {
80  error("Parameter \"%s\" already exists.", name.c_str());
81  }
82  return p.second;
83 }

87 bool AbstractNode::setNetworkParameter(const std::string& parameterName,
88  const std::string& networkParameterName,
89  const std::string& networkParameterValue) {
90  Parameters::iterator p = parameters_.find(parameterName);
91  verify(p != parameters_.end());
92 
93  if (!p->second.setVariable(networkParameterName, networkParameterValue))
94  defect();
95 
96  std::string v;
97  if (p->second.value(v)) {
98  if (checkAndSetParameter(parameterName, v)) {
99  // Erase output attributes (recursively) since the node has been changed
100  eraseOutputAttributes();
101  return true;
102  }
103  else {
104  error("Failed to set network parameter key=\"%s\" (value=\"%s\") "
105  "of parameter=\"%s\" in node=\"%s\" .",
106  networkParameterName.c_str(), networkParameterValue.c_str(),
107  parameterName.c_str(), name().c_str());
108  return false;
109  }
110  }
111  return true;
112 }
