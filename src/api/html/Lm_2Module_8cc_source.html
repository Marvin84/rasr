73 Core::Ref<LanguageModel> Module_::createLanguageModel(
74  const Core::Configuration& c,
75  Bliss::LexiconRef l) {
76  Core::Ref<LanguageModel> result;
77 
78  switch (lmTypeParam(c)) {
79 #ifdef MODULE_LM_ARPA
80  case lmTypeArpa: result = Core::ref(new ArpaLm(c, l)); break;
81  case lmTypeArpaWithClasses: result = Core::ref(new ArpaClassLm(c, l)); break;
82 #endif
83 #ifdef MODULE_LM_FSA
84  case lmTypeFsa: result = Core::ref(new FsaLm(c, l)); break;
85  case lmTypeCheatingSegment: result = Core::ref(new CheatingSegmentLm(c, l)); break;
86 #endif
87 #ifdef MODULE_LM_ZEROGRAM
88  case lmTypeZerogram: result = Core::ref(new Zerogram(c, l)); break;
89 #endif
90 #ifdef MODULE_LM_FFNN
91  case lmTypeFFNN: result = Core::ref(new FFNeuralNetworkLanguageModel(c, l)); break;
92 #endif
93  case lmTypeCombine: result = Core::ref(new CombineLanguageModel(c, l)); break;
94 #ifdef MODULE_LM_TFRNN
95  case lmTypeTFRNN: result = Core::ref(new TFRecurrentLanguageModel(c, l)); break;
96 #endif
97  default:
98  Core::Application::us()->criticalError("unknwon language model type: %d", lmTypeParam(c));
99  }
100  result->init();
101  if (result->hasFatalErrors())
102  result.reset();
103  return result;
104 }
