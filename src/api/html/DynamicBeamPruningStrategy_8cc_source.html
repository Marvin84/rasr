90 void MaximumDelayBeamPruningStrategy::frameFinished(TimeframeIndex time, f64 current_frame_time, f64 delay) {
91  if (time == 1) {
92  initialDelay_ = delay - current_frame_time;
93  }
94  delay -= initialDelay_ + std::min(initialDelay_, addInitialDelayPerFrameTime_ * time);
95  if (delay >= decrementBeamThreshold_) {
96  currentScale_ *= decrementBeamFactor_;
97  currentScale_ = std::max<Score>(minimumBeamScale_, currentScale_);
98  }
99  else if (delay <= incrementBeamThreshold_) {
100  currentScale_ *= incrementBeamFactor_;
101  currentScale_ = std::min<Score>(maximumBeamScale_, currentScale_);
102  }
103  std::cerr << "frame: " << time << " time: " << current_frame_time << " delay: " << delay << " initial delay: " << initialDelay_ << " scale: " << currentScale_ << std::endl;
104 }
