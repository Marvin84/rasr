29 template<class T>
30 class LinearFilter {
31 private:
32  std::vector<T> B_tilde_;
33  std::vector<T> A_tilde_;
34 
35  std::vector<T> u_;
36  std::vector<T> y_;
37 
38  T work(T u) {
39  T y = 0.0;
40 
41  for (s16 j = B_tilde_.size() - 1; j >= 0; j--) {
42  y += B_tilde_[j] * u_[j];
43  u_[j] = j ? u_[j - 1] : u;
44  }
45 
46  for (s16 i = A_tilde_.size() - 1; i >= 0; i--) {
47  y -= A_tilde_[i] * y_[i];
48  y_[i] = i ? y_[i - 1] : y;
49  }
50 
51  return y;
52  }
53 
54 public:
55  LinearFilter() {}
56  ~LinearFilter() {}

58  void setB(const std::vector<T>& B_tilde) {
59  B_tilde_ = B_tilde;
60  u_.resize(B_tilde.size());
61  for (u8 i = 0; i < u_.size(); i++)
62  u_[i] = 0;
63  }
64  void setA(const std::vector<T>& A_tilde) {
65  A_tilde_ = A_tilde;
66  y_.resize(A_tilde.size());
67  for (u8 j = 0; j < y_.size(); j++)
68  y_[j] = 0;
69  }
70  void setY0(const std::vector<T>& y0) {
71  verify(y_.size() == y0.size());
72  for (u8 i = 0; i < u_.size(); i++)
73  u_[i] = 0;
74  for (u8 j = 0; j < y_.size(); j++)
75  y_[j] = y0[j];
76  }

78  bool work(std::vector<T>& s) {
79  return work(&s, s);
80  }
81  bool work(const std::vector<T>* u, std::vector<T>& y) {
82  if (u) {
83  y.resize(u->size());
84  for (u32 t = 0; t < y.size(); t++)
85  y[t] = work((*u)[t]);
86  }
87  else {
88  for (u32 t = 0; t < y.size(); t++)
89  y[t] = work(0.0);
90  }
91 
92  return true;
93  }
94 
95  void reset() {
96  for (u8 i = 0; i < u_.size(); i++)
97  u_[i] = 0;
98  for (u8 j = 0; j < y_.size(); j++)
99  y_[j] = 0;
100  }
