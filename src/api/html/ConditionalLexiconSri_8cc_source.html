28 void ConditionalLexiconSri::read() {
29  if (lexiconFilename_ == "")

31  sriTupleVocabulary_.unkIsWord() = true;
32  File fpTupleLM(lexiconFilename_.c_str(), "r");

34 
35  if (userLmOrder_ > 0) {
36  lmOrder_ = userLmOrder_;
37  srilm_.setorder(lmOrder_);
38  }
39  else
40  lmOrder_ = srilm_.setorder();
41 
42  std::cerr << "finished reading lexicon" << std::endl;
43 
44  unknownIndex_ = sriTupleVocabulary_.unkIndex();

50  std::cerr << "sentence end index = " << sentenceEndIndex_ << std::endl;
51  std::cerr << "srilm_.setorder() = " << srilm_.setorder(lmOrder_) << std::endl;
52 
53  VocabIter vocabIterator(sriTupleVocabulary_);
54 
55  std::cerr << "extraction monolingual tokens from lm vocab" << std::endl;
56 
57  while (VocabString bilingualWord = vocabIterator.next()) {
58  // Now we extract the source and target parts
59  std::vector<std::string> fields = Core::split(bilingualWord, "|");
60  if (fields.size() == 2) { // If it is not a special word
61  Fsa::LabelId sourceLabelId;
62  Fsa::LabelId targetLabelId;
63  if (fields[0] != "$")
64  sourceLabelId = tokens_->addSymbol(fields[0]);
65  else
66  sourceLabelId = Fsa::Epsilon;
67 
68  if (fields[1] != "$")
69  targetLabelId = tokens_->addSymbol(fields[1]);
70  else
71  targetLabelId = Fsa::Epsilon;
72 
73  vocabMap[make_pair(sourceLabelId, targetLabelId)] = sriTupleVocabulary_.getIndex(bilingualWord);
74  }
75  }
76 }
