53 bool OpenSoundSystemDevice::setDeviceTracks(u8& _trackCount) {

69 bool OpenSoundSystemDevice::setDeviceFormat(u8& _sampleSize) {
70  require(isDeviceOpen());
71 
72  int format;
73  switch (_sampleSize) {
74  case 8:
75  format = AFMT_U8; // Shouldn't we use AFMT_S8 ?
76  break;
77  case 16:
78 #if __BYTE_ORDER == __LITTLE_ENDIAN
79  format = AFMT_S16_LE;
80 #elif __BYTE_ORDER == __BIG_ENDIAN
81  format = AFMT_S16_BE;
82 #else
83 #error "Machine has unsupported byte order!"
84 #endif
85  break;
86  default:
87  error("unsupported sample size: %d bit", _sampleSize);
88  return false;
89  }
90 
91  int actualFormat = format;
92  if (ioctl(fd_, SNDCTL_DSP_SETFMT, &actualFormat) == -1) {
93  error("during ioctl(SNDCTL_DSP_SETFMT)");
94  closeDevice();
95  return false;
96  }
97  if (actualFormat != format) {
98  warning("sound device does not support sample format %d", format);
99  warning("sound device uses format %d instead.", actualFormat);
100  }
101  return true;
102 }

152 template<typename T>
153 u32 OpenSoundSystemInputNode::readTyped(u32 nSamples, Flow::Timestamp*& d) {
154  require(fd_ >= 0);
155  Flow::Vector<T>* v = new Flow::Vector<T>(nSamples * trackCount_);
156  size_t bytesToRead = nSamples * trackCount_ * sizeof(T);
157  ssize_t bytesRead = ::read(fd_, (void*)(&*(v->begin())), bytesToRead);
158  if (bytesRead <= 0) {
159  if (bytesRead < 0)
160  error("read failed");
161  delete v;
162  d = 0;
163  return 0;
164  }
165  v->resize(bytesRead / sizeof(T));
166  d = v;
167  return v->size() / trackCount_;
168 }
