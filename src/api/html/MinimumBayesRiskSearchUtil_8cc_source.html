40 u32 levenshteinDistance(const std::vector<LabelId>& A, const std::vector<LabelId>& B) {
41  std::size_t M = A.size() + 1;
42  std::size_t N = B.size() + 1;
43  std::size_t m, n;
44  std::size_t a, b, c;
45 
46  u32** D;
47 
48  D = new u32*[M];
49 
50  for (m = 0; m < M; ++m) {
51  D[m] = new u32[N];
52  D[m][0] = m;
53  } //end for m
54  for (n = 0; n < N; ++n) {
55  D[0][n] = n;
56  } //end for n
57 
58  for (n = 1; n < N; ++n) {
59  for (m = 1; m < M; ++m) {
60  a = (D[m - 1][n] + 1);
61  b = (D[m][n - 1] + 1);
62  c = (D[m - 1][n - 1] + (A.at(m - 1) == B.at(n - 1) ? 0 : 1));
63  D[m][n] = (a <= b ? (a <= c ? a : (b <= c ? b : c)) : (b <= c ? b : c));
64  } //end for m
65  } //end for n
66 
67  u32 result = D[M - 1][N - 1];
68 
69  for (m = 0; m < M; ++m) {
70  delete[] D[m];
71  }
72  delete[] D;
73 
74  return result;
75 } //end levensthein

91 std::vector<Fsa::StateId> getDistances(ConstAutomatonRef fsa) {
92  std::vector<StateId> distances;
93 
94  std::set<StateId> contour;
95  std::set<StateId>::const_iterator state;
96  contour.insert(fsa->initialStateId());
97 
98  u32 dist = 0;
99  while (!contour.empty()) {
100  for (state = contour.begin(); state != contour.end(); ++state) {
101  if (distances.size() <= *state) {
102  distances.resize(*state + 1);
103  } //end if
104  distances.at(*state) = dist;
105  } //end for state
106  contour = getContour(contour, fsa);
107  ++dist;
108  } //end while
109  return distances;
110 } //end getDistances

150 ConstAutomatonRef normalizeNbestlist(ConstAutomatonRef nbest) {
151  Core::Ref<StaticAutomaton> nbestlist = staticCopy(nbest);
152 
153  Fsa::Accumulator* collector = Fsa::LogSemiring->getCollector(Fsa::LogSemiring->zero());
154  State* initialState = nbestlist->fastState(nbestlist->initialStateId());
155  for (State::iterator arc = initialState->begin(); arc != initialState->end(); ++arc) {
156  Weight score = arc->weight_;
157  // follow trace
158  State* s = nbestlist->fastState(arc->target());
159  while (!s->isFinal()) {
160  verify(s->nArcs() == 1);
161  Fsa::Arc& a = (*s->begin());
162  score = Fsa::LogSemiring->extend(score, a.weight());
163  a.weight_ = Fsa::LogSemiring->one();
164  s = nbestlist->fastState(a.target_);
165  }
166  score = Fsa::LogSemiring->extend(score, s->weight_);
167  arc->weight_ = score;
168  collector->feed(score);
169  }
170  Weight inverseNormalizationConstant = Fsa::LogSemiring->invert(collector->get());
171  delete collector;
172 
173  for (State::iterator arc = initialState->begin(); arc != initialState->end(); ++arc) {
174  arc->weight_ = Fsa::LogSemiring->extend(arc->weight(), inverseNormalizationConstant);
175  }
176 
177  return ConstAutomatonRef(nbestlist);
178 }
