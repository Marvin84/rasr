58 void FeatureAccumulator::processAlignedFeature(Core::Ref<const Feature> f, Am::AllophoneStateIndex id, Mm::Weight w) {
59  if (w == 0.0)
60  return;
61  if (acousticModel()->allophoneStateAlphabet()->isDisambiguator(id)) {
62  warning("Disambiguator %s found in alignment",
63  acousticModel()->allophoneStateAlphabet()->symbol(id).c_str());
64  return;
65  }
66  const Mm::FeatureVector& feature = *f->mainStream();
67  // Do I see my first feature? Yes, then initialize example size
68  if (nCols_)
69  verify(nCols_ == feature.size());
70  else {
71  nCols_ = feature.size();
72  require(nCols_ > 0);
73  }
74  nObs_ += w;
75  if ((size_t(id) >= examples_.size()) || !examples_[id]) {
76  Am::Properties props(map_, acousticModel()->allophoneStateAlphabet()->allophoneState(id));
77  examples_.set(id, new Cart::Example(new Cart::StoredProperties(props), new Cart::FloatBox(2, nCols_)));
78  }
79  Cart::Example& example = *examples_[id];
80  // example.write((Core::XmlWriter&)log());
81 
82  Mm::FeatureVector::const_iterator itFeature = feature.begin();
83  Cart::FloatBox::vector_iterator itSum = example.values->row(0).begin();
84  Cart::FloatBox::vector_iterator itSumOfSquares = example.values->row(1).begin();
85  for (; itFeature != feature.end(); ++itFeature, ++itSum, ++itSumOfSquares) {
86  *itSum += (*itFeature) * w;
87  *itSumOfSquares += ((*itFeature) * (*itFeature)) * w;
88  }
89  example.nObs += w;
90 }

124 /*
125  negative log-likelihood for a Gaussian with diagonal co-variance matrix
126 
127  for x_1^N with x_n = [ x_{n,1}, ... x_{n,D} ]
128  \mu_d = \frac{1}{N} \sum_{n=1}^{N} x_{n,d}
129  \sigma^2_d = \frac{1}{N} \sum_{n=1}^{N} (x_{n,d} - \mu_{n,d})^2
130  \theta = ([\mu]_1^D, [\sigma^2]_^D)
131  log N(x; \theta) = -\frac{1}{2} \left( \sum_1^D \log(2 \pi \sigma^2_d + \sum_1^D \frac{(x_d - \mu_d)^2}{\sigma^2_d} \right)
132  -LL(\theta| x_1^N) = \frac{1}{2} \left( N D + N \sum_{d=1}^{D} \log( 2 \pi \sigma^2_d) \right)
133 */
134 Cart::Score StateTyingDecisionTreeTrainer::LogLikelihoodGain::logLikelihood(
135  Cart::ExamplePtrList::const_iterator begin,
136  Cart::ExamplePtrList::const_iterator end) const {
137  if (begin == end)
138  return 0.0;
139 
140  size_t d = (*begin)->values->columns();
141  mu_.resize(d, 0.0);
142  sigmaSquare_.resize(d, 0.0);
143  f64 count = 0;
144  f64 ll = 0.0;
145 
146  if (parallel_) {
147 #pragma omp parallel
148  {
149  std::vector<f64> mu_private(d, 0.0); // to store temporary results
150  std::vector<f64> sigmaSquare_private(d, 0.0); // to store temporary results
151 
152 #pragma omp for nowait reduction(+ \
153  : count)
154  for (Cart::ExamplePtrList::const_iterator it = begin; it < end; ++it) { // supported by omp 3.0
155  const Cart::Example& example = **it;
156  require(example.values->rows() == 2);
157  require(example.values->columns() == d);
158 
159  count += example.nObs;
160 
161  std::vector<f64>::iterator itMu = mu_private.begin();
162  std::vector<f64>::iterator itSigmaSquare = sigmaSquare_private.begin();
163  Cart::FloatBox::const_vector_iterator itSum = example.values->row(0).begin();
164  Cart::FloatBox::const_vector_iterator itSumOfSquares = example.values->row(1).begin();
165  for (; itMu != mu_private.end(); ++itMu, ++itSigmaSquare, ++itSum, ++itSumOfSquares) {
166  *itMu += *itSum;
167  *itSigmaSquare += *itSumOfSquares;
168  }
169  }
170 
171 #pragma omp critical // if slow create e.g. std::vector<std::vector<f64> > mus_private(threadnum, std::vector<f64>(d, 0.0)) and merge them \
172  // with an omp for over dimension \
173  // and also merge with next for loop \
174  // order is not defined, non-deterministic sum-up!
175  {
176  std::vector<f64>::iterator itMu_private = mu_private.begin();
177  std::vector<f64>::iterator itSigmaSquare_private = sigmaSquare_private.begin();
178  std::vector<f64>::iterator itMu = mu_.begin();
179  std::vector<f64>::iterator itSigmaSquare = sigmaSquare_.begin();
180  for (; itMu != mu_.end(); ++itMu, ++itSigmaSquare, ++itMu_private, ++itSigmaSquare_private) {
181  *itMu += *itMu_private;
182  *itSigmaSquare += *itSigmaSquare_private;
183  }
184  }
185 
186 #pragma omp barrier // each thread should have access to count, mu, sigmsq
187 #pragma omp for reduction(+ \
188  : ll)
189  for (size_t dimCounter = 0; dimCounter < d; ++dimCounter) {
190  mu_[dimCounter] /= count;
191  sigmaSquare_[dimCounter] /= count;
192  sigmaSquare_[dimCounter] -= mu_[dimCounter] * mu_[dimCounter];
193  if (sigmaSquare_[dimCounter] < minSigmaSquare_)
194  sigmaSquare_[dimCounter] = minSigmaSquare_;
195 
196  ll += ::log(sigmaSquare_[dimCounter]);
197 
198  mu_[dimCounter] = sigmaSquare_[dimCounter] = 0.0;
199  }
200  }
201  ll = (0.5 * count) * (d + d * ::log(PI + PI) + ll);
202  return ll;
203  }
204  else { // backward compatibility
205  for (Cart::ExamplePtrList::const_iterator it = begin; it != end; ++it) {
206  const Cart::Example& example = **it;
207  require(example.values->rows() == 2);
208  require(example.values->columns() == d);
209  count += example.nObs;
210  std::vector<f64>::iterator itMu = mu_.begin();
211  std::vector<f64>::iterator itSigmaSquare = sigmaSquare_.begin();
212  Cart::FloatBox::const_vector_iterator itSum = example.values->row(0).begin();
213  Cart::FloatBox::const_vector_iterator itSumOfSquares = example.values->row(1).begin();
214  for (; itMu != mu_.end(); ++itMu, ++itSigmaSquare, ++itSum, ++itSumOfSquares) {
215  *itMu += *itSum;
216  *itSigmaSquare += *itSumOfSquares;
217  }
218  }
219  f64 n = count;
220  std::vector<f64>::iterator itMu = mu_.begin();
221  std::vector<f64>::iterator itSigmaSquare = sigmaSquare_.begin();
222  for (; itMu != mu_.end(); ++itMu, ++itSigmaSquare) {
223  *itMu /= n;
224  *itSigmaSquare /= n;
225  *itSigmaSquare -= (*itMu) * (*itMu);
226  if (*itSigmaSquare < minSigmaSquare_)
227  *itSigmaSquare = minSigmaSquare_;
228  ll += ::log(*itSigmaSquare);
229  *itMu = *itSigmaSquare = 0.0;
230  }
231  ll = (0.5 * n) * (d + d * ::log(PI + PI) + ll);
232  return ll;
233  }
234 }

242 Cart::Score StateTyingDecisionTreeTrainer::LogLikelihoodGain::operator()(
243  const Cart::ExamplePtrRange& leftExamples, const Cart::ExamplePtrRange& rightExamples,
244  const Cart::Score fatherLogLikelihood,
245  Cart::Score& leftChildLogLikelihood, Cart::Score& rightChildLogLikelihood) const {
