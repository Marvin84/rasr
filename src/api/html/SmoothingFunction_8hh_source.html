66 class LogSmoothingFunction : public SmoothingFunction {
67  typedef SmoothingFunction Precursor;
68 
69 private:
70  static const Core::ParameterFloat paramM;
71 
72 private:
73  f64 xM_;
74 
75 public:
76  LogSmoothingFunction(const Core::Configuration&);
77  virtual ~LogSmoothingFunction() {}
78  virtual const std::string name() const {
79  return Core::form("log(%f)", std::log(xM_));
80  }
81  // f(x)
82  virtual f64 f(f64 x) const {
83  x = std::max(x, Core::Type<f64>::epsilon);
84  return std::log(x / (x + xM_ * (1 - x)));
85  }
86  // f'(x)*x
87  virtual f64 dfx(f64 x) const {
88  x = std::max(x, (f64)Core::Type<f32>::epsilon);
89  return (1 / x) - (1 - xM_) / (x + xM_ * (1 - x));
90  }
91  virtual void dumpStatistics(Core::XmlWriter&) const;
92 };

94 class SigmoidSmoothingFunction : public SmoothingFunction {
95  typedef SmoothingFunction Precursor;
96 
97 private:
98  static const Core::ParameterFloat paramBeta;
99  static const Core::ParameterFloat paramM;
100 
101 private:
102  f64 beta_;
103  f64 xM_;
104 
105 public:
106  SigmoidSmoothingFunction(const Core::Configuration&);
107  virtual ~SigmoidSmoothingFunction() {}
108  virtual const std::string name() const {
109  return Core::form("sigmoid(%f,%f)", beta_, std::log(xM_));
110  }
111  // f(x)
112  virtual f64 f(f64 x) const {
113  return pow(x, beta_) / (pow(xM_ * (1 - x), beta_) + pow(x, beta_));
114  }
115  // f'(x)*x
116  virtual f64 dfx(f64 x) const {
117  x = std::max(x, (f64)Core::Type<f32>::epsilon);
118  return beta_ * f(x) * (1 - f(x)) / (1 - x);
119  }
120  virtual void dumpStatistics(Core::XmlWriter&) const;
121 };
