64 void GaussDiagonalMaximumFeatureScorer::init(const MixtureSet& mixtureSet) {
65  dimension_ = mixtureSet.dimension();
66 
67  mixtureTable_.resize(mixtureSet.nMixtures());
68  for (size_t i = 0; i < mixtureTable_.size(); ++i) {
69  mixtureTable_[i] = *mixtureSet.mixture(i);
70  mixtureTable_[i].scale(mixtureWeightScale_);
71  }
72 
73  densityTable_.resize(mixtureSet.nDensities());
74  for (size_t i = 0; i < densityTable_.size(); ++i)
75  densityTable_[i] = *mixtureSet.density(i);
76 
77  meanTable_.resize(mixtureSet.nMeans());
78  for (size_t i = 0; i < meanTable_.size(); ++i)
79  meanTable_[i] = *mixtureSet.mean(i);
80 
81  covarianceTable_.resize(mixtureSet.nCovariances());
82  for (size_t i = 0; i < covarianceTable_.size(); ++i) {
83  covarianceTable_[i] = *mixtureSet.covariance(i);
84  covarianceTable_[i].scale(gaussianScale_);
85  }
86 }

116 AssigningFeatureScorer::ScoreAndBestDensity GaussDiagonalMaximumFeatureScorer::calculateScoreAndDensity(
117  const CachedAssigningContextScorer* cs, MixtureIndex mixtureIndex) const {
118  const Context* c = required_cast(const Context*, cs);
119  Score bestScore = Core::Type<Score>::max;
120  size_t bestDensity = Core::Type<size_t>::max;
121  const MixtureFeatureScorerElement& mixture = mixtureTable_[mixtureIndex];
122  size_t nDensities = mixture.nDensities();
123  for (size_t dns = 0; dns < nDensities; ++dns) {
124  const GaussDensity& density = densityTable_[mixture.densityIndex(dns)];
125  const CovarianceFeatureScorerElement& covariance = covarianceTable_[density.covarianceIndex()];
126 
127  // code is architecture dependent without (f64)
128  // this can lead to unwanted behavior
129  f64 score = (f64)mixture.minus2LogWeights()[dns] +
130  (f64)covariance.logNormalizationFactor() +
131  (f64)distance(c->featureVector_, meanTable_[density.meanIndex()],
132  covariance.inverseSquareRootDiagonal());
133  if (bestScore > score) {
134  bestScore = score;
135  bestDensity = dns;
136  }
137  }
138  ScoreAndBestDensity result;
139  result.score = 0.5 * bestScore;
140  result.bestDensity = bestDensity;
141  return result;
142 }

144 Score GaussDiagonalMaximumFeatureScorer::distance(const std::vector<FeatureType>& feature,
145  const std::vector<MeanType>& mean,
146  const std::vector<VarianceType>& inverseSquareRootVar) const {
147  ComponentIndex cmp = 0;
148  ComponentIndex dimension = feature.size();
149  Score result = 0;
150  Score df;
151 #ifdef __SSE3__
152  __m128 sum = _mm_setzero_ps();
153 
154  ComponentIndex eff_dim = dimension & (~3u); // == dimension - dimension % 4
155  while (cmp < eff_dim) {
156  __m128 m = _mm_load_ps(mean.data() + cmp);
157  __m128 f = _mm_load_ps(feature.data() + cmp);
158  __m128 isrv = _mm_load_ps(inverseSquareRootVar.data() + cmp);
159  __m128 df = (m - f) * isrv;
160  sum += df * df;
161  cmp += 4u;
162  }
163  sum = _mm_hadd_ps(sum, sum);
164  __attribute__((aligned(16))) float buffer[4];
165  _mm_store_ps(buffer, sum);
166  result += buffer[0] + buffer[1];
167 
168  switch (dimension - eff_dim) {
169  case 3:
170  df = (mean[cmp] - feature[cmp]) * inverseSquareRootVar[cmp];
171  result += df * df;
172  ++cmp;
173  case 2:
174  df = (mean[cmp] - feature[cmp]) * inverseSquareRootVar[cmp];
175  result += df * df;
176  ++cmp;
177  case 1: df = (mean[cmp] - feature[cmp]) * inverseSquareRootVar[cmp]; result += df * df;
178  }
179 #else
180  switch ((dimension - 1) % 8) {
181  while (cmp < dimension) {
182  case 7:
183  df = (mean[cmp] - feature[cmp]) * inverseSquareRootVar[cmp];
184  result += df * df;
185  ++cmp;
186  case 6:
187  df = (mean[cmp] - feature[cmp]) * inverseSquareRootVar[cmp];
188  result += df * df;
189  ++cmp;
190  case 5:
191  df = (mean[cmp] - feature[cmp]) * inverseSquareRootVar[cmp];
192  result += df * df;
193  ++cmp;
194  case 4:
195  df = (mean[cmp] - feature[cmp]) * inverseSquareRootVar[cmp];
196  result += df * df;
197  ++cmp;
198  case 3:
199  df = (mean[cmp] - feature[cmp]) * inverseSquareRootVar[cmp];
200  result += df * df;
201  ++cmp;
202  case 2:
203  df = (mean[cmp] - feature[cmp]) * inverseSquareRootVar[cmp];
204  result += df * df;
205  ++cmp;
206  case 1:
207  df = (mean[cmp] - feature[cmp]) * inverseSquareRootVar[cmp];
208  result += df * df;
209  ++cmp;
210  case 0:
211  df = (mean[cmp] - feature[cmp]) * inverseSquareRootVar[cmp];
212  result += df * df;
213  ++cmp;
214  }
215  }
216 #endif
217  return result;
218 }

239 void GaussDiagonalSumFeatureScorer::calculateScoresAndNumberOfDensities(const CachedAssigningContextScorer* cs,
240  MixtureIndex mixtureIndex) const {
241  // check, if we must recompute scores_ and nDensities_
242  if (mixtureIndex == lastMixtureIndex_ && cs == lastContext_)
243  return;
244  lastContext_ = cs;
245  lastMixtureIndex_ = mixtureIndex;
246  const Context* c = required_cast(const Context*, cs);
247 
248  const MixtureFeatureScorerElement& mixture = mixtureTable_[mixtureIndex];
249  nDensities_ = mixture.nDensities();
250 
251  for (size_t dns = 0; dns < nDensities_; ++dns) {
252  const GaussDensity& density = densityTable_[mixture.densityIndex(dns)];
253  const CovarianceFeatureScorerElement& covariance = covarianceTable_[density.covarianceIndex()];
254 
255  Score score = mixture.minus2LogWeights()[dns] +
256  covariance.logNormalizationFactor() +
257  distance(c->featureVector_, meanTable_[density.meanIndex()], covariance.inverseSquareRootDiagonal());
258 
259  scores_[dns] = 0.5 * score;
260  }
261 }

263 AssigningFeatureScorer::ScoreAndBestDensity GaussDiagonalSumFeatureScorer::calculateScoreAndDensity(const CachedAssigningContextScorer* cs,
264  MixtureIndex mixtureIndex) const {
265  // 1.) calculate best (= viterbi) score
266  Score bestScore = Core::Type<Score>::max;
267  size_t bestDensity = Core::Type<size_t>::max;
268 
269  calculateScoresAndNumberOfDensities(cs, mixtureIndex);
270  for (size_t dns = 0; dns < nDensities_; ++dns) {
271  Score score = scores_[dns];
272  if (bestScore > score) {
273  bestScore = score;
274  bestDensity = dns;
275  }
276  }
277 
278  // 2.) compute sum of exponentiated truncated scores
279  Score sumExp = 0;
280  for (size_t dns = 0; dns < nDensities_; ++dns) {
281  sumExp += std::exp(bestScore - scores_[dns]);
282  }
283 
284  // 3.) subtract from viterbi score
285  ScoreAndBestDensity result;
286  result.score = bestScore - std::log(sumExp);
287  result.bestDensity = bestDensity;
288  return result;
289 }
