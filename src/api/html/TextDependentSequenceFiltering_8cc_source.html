89 void AlignmentToSequenceSelectionNode::PhonemeSet::setSymbols(
90  const std::vector<std::string>& symbols) {
91  delete phonemeMap_;
92  phonemeMap_ = new Bliss::PhonemeMap<char>(phonemeInventory_);
93  phonemeMap_->fill((char)false);
94  empty_ = true;
95  for (std::vector<std::string>::const_iterator i = symbols.begin(); i != symbols.end(); i++) {
96  const Bliss::Phoneme* ph = phonemeInventory_->phoneme(*i);
97  if (ph) {
98  (*phonemeMap_)[ph] = (char)true;
99  empty_ = false;
100  }
101  else
102  error("Phoneme '%s' not found in the phoneme inventory.", i->c_str());
103  }
104  if (!symbols.empty() && empty_) {
105  std::stringstream ss;
106  std::copy(symbols.begin(), symbols.end(), std::ostream_iterator<std::string>(ss, " "));
107  warning("The symbols '%s' yielded an empty phoneme set.", ss.str().c_str());
108  }
109 }

167 bool AlignmentToSequenceSelectionNode::work(Flow::PortId p) {
168  Flow::DataPtr<Flow::DataAdaptor<Alignment>> in;
169  if (getData(0, in)) {
170  const Alignment& alignment = in->data();
171  Flow::Vector<bool>* selection = new Flow::Vector<bool>(alignment.size());
172  selection->setTimestamp(*in);
173  Flow::Float32* ratio = new Flow::Float32();
174  ratio->setTimestamp(*in);
175  size_t filteredFeatureCount = 0;
176 
177  Flow::Vector<bool>::iterator s = selection->begin();
178  Alignment::const_iterator a = alignment.begin();
179  for (; a != alignment.end(); ++a, ++s) {
180  *s = true;
181  for (Sets::const_iterator set = sets_.begin(); set != sets_.end(); ++set)
182  *s = *s && (*set)->contains(a->emission);
183  if (*s)
184  ++filteredFeatureCount;
185  }
186  if (alignment.size() > 0)
187  ratio->data() = (f32)filteredFeatureCount / (f32)alignment.size();
188  else
189  ratio->data() = Core::Type<f32>::max + 1;
190 
191  putData(0, selection);
192  putData(1, ratio);
193  return true;
194  }
195  putData(0, in.get());
196  putData(1, in.get());
197  return true;
198 }
