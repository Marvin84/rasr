83 const Bliss::SyntacticToken* LanguageModel::getSpecialSyntacticToken(const std::string& name, bool required) const {
84  const Bliss::Lemma* lemma = lexicon()->specialLemma(name);
85 
86  if (!lemma) {
87  if (required) {
88  error("Cannot determine \"%s\" lm token. "
89  "No special lemma \"%s\" defined in lexicon.",
90  name.c_str(), name.c_str());
91  }
92  return 0;
93  }
94 
95  if (!lemma->hasSyntacticTokenSequence()) {
96  if (required) {
97  error("Cannot determine \"%s\" lm token. "
98  "Special lemma \"%s\" does not specify a syntactic token.",
99  name.c_str(), name.c_str());
100  }
101  return 0;
102  }
103 
104  const Bliss::SyntacticTokenSequence& tokenSequence(
105  lemma->syntacticTokenSequence());
106 
107  if (tokenSequence.length() < 1) {
108  if (required) {
109  error("Empty \"%s\" lm token. "
110  "Special lemma \"%s\" specifies a syntactic token sequence of length zero.",
111  name.c_str(), name.c_str());
112  }
113  return 0;
114  }
115 
116  const Bliss::SyntacticToken* token = tokenSequence[0];
117 
118  if (tokenSequence.length() > 1) {
119  warning("Ambiguous \"%s\" lm token. "
120  "Special lemma \"%s\" specifies a syntactic token sequence longer than one. "
121  "Using only \"%s\".",
122  name.c_str(), name.c_str(), token->symbol().str());
123  }
124 
125  return token;
126 }

175 void LanguageModel::getBatch(const History& history, const CompiledBatchRequest* cbr, std::vector<f32>& result) const {
176  const NonCompiledBatchRequest* ncbr = required_cast(const NonCompiledBatchRequest*, cbr);
177  const BatchRequest& request(ncbr->request);
178 
179  for (BatchRequest::const_iterator r = request.begin(); r != request.end(); ++r) {
180  Score sco = 0.0;
181  if (r->tokens.length() >= 1) {
182  sco += score(history, r->tokens[0]);
183  if (r->tokens.length() > 1) {
184  History h = extendedHistory(history, r->tokens[0]);
185  for (u32 ti = 1;; ++ti) {
186  Token st = r->tokens[ti];
187  sco += score(h, st);
188  if (ti + 1 >= r->tokens.length())
189  break;
190  h = extendedHistory(h, st);
191  }
192  }
193  }
194  sco *= ncbr->scale();
195  sco += r->offset;
196 
197  if (result[r->target] > sco)
198  result[r->target] = sco;
199  }
200 }
