164  virtual ConstStateRef getState(Fsa::StateId sid) const {
165  require(sid < states_.size());
166  InternalState state = states_[sid];
167  ConstStateRef sr = getCachedState(state.sid);
168  State* sp = clone(*semiring_, sr);
169  sp->setId(sid);
170  for (State::iterator a = sp->begin(); a != sp->end(); ++a) {
171  Fsa::LabelId labelId = a->input();
172  Lm::History history = state.history;
173  if ((Fsa::FirstLabelId <= labelId) && (labelId <= Fsa::LastLabelId)) {
174  const Bliss::Lemma* lemma = (lAlphabet_) ? lAlphabet_->lemma(labelId) : lpAlphabet_->lemmaPronunciation(labelId)->lemma();
175  require_(lemma);
176  Lm::Score lmScore = 0.0;
177  /*
178  * Attention:
179  * Max lm-rescorer uses
180  *
181  * Lm::addLemmaScore(lm_, <lm-scale>, lemma, 1.0, history, lmScore),
182  *
183  * but this is not feasible for the idea of Flf, because setting
184  * the scale for lm-composition directly or setting the scale for
185  * the lm-dimension might prodcue different results.
186  * The correct procedure would be
187  *
188  * Lm::addLemmaScore(lm_, scale_, lemma, 0.0, history, lmScore);
189  *
190  * and add the syntaxEmissionScore to another dimension.
191  * Soa is that the syntaxEmissionScore has no influence and is normally 0,
192  * thus introducing a new dimension and a new parameter seems to be overhead.
193  * Instead, the syntaxEmissionSale is set to lmScale.
194  */
195  Lm::addLemmaScore(lm_, scale_, lemma, scale_, history, lmScore);
196  a->weight_->add(id_, lmScore);
197  }
198  if (!forceSentenceEnd_ && !getCachedState(a->target())->hasArcs())
199  a->setTarget(insertState(a->target(), emptyHistory_));
200  else
201  a->setTarget(insertState(a->target(), history));
202  }
203  if (forceSentenceEnd_ && sp->isFinal())
204  sp->weight_->add(id_, scale_ * lm_->sentenceEndScore(state.history));
205  return ConstStateRef(sp);
206  }
