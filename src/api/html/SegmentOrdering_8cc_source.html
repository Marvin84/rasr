93 void SegmentOrderingVisitor::addSegment(Segment* segment) {
94  verify(!recordings_.empty());
95  segment->setRecording(recordings_.back());
96  if (segment->condition())
97  updateCondition(segment);
98  std::string name = _getName(segment);
99  if (segments_.find(name) != segments_.end()) {
100  Core::Application::us()->error("can not add segment, because it is already present in segment list: ") << name;
101  }
102  segments_.insert(SegmentMap::value_type(name, segment));
103  if (!predefinedOrder_)
104  segmentList_.push_back(name);
105 }

107 template<class T, class C>
108 const T* SegmentOrderingVisitor::updateSegmentData(Segment* segment, const T* entry, C& map) {
109  // Not needed for the root corpus.
110  if (entry->parent() != curCorpus_.front().first) {
111  typename C::const_iterator c = map.find(_getName(entry));
112  if (c != map.end()) {
113  entry = c->second;
114  }
115  else {
116  T* add = new T(*entry);
117  if (entry->parent() == curRecording_) {
118  add->setParent(recordings_.back());
119  }
120  else if (entry->parent() == curSegment_) {
121  add->setParent(segment);
122  }
123  else {
124  for (CorpusMap::const_reverse_iterator i = curCorpus_.rbegin();
125  i != curCorpus_.rend(); ++i) {
126  if (entry->parent() == i->first) {
127  add->setParent(i->second);
128  break;
129  }
130  }
131  }
132  map.insert(std::make_pair(_getName(entry), add));
133  entry = add;
134  }
135  }
136  return entry;
137 }

160 void SegmentOrderingVisitor::setSegmentList(const std::string& filename) {
161  verify(!autoShuffle_); // we will automatically fill segmentList_
162  verify(!filename.empty());
163  require(!predefinedOrder_); // not yet loaded
164 
165  Core::CompressedInputStream* cis = new Core::CompressedInputStream(filename.c_str());
166  Core::TextInputStream is(cis);
167  if (!is)
168  Core::Application::us()->criticalError("Failed to open segment list \"%s\".", filename.c_str());
169  std::string s;
170  while (Core::getline(is, s) != EOF) {
171  if ((s.size() == 0) || (s.at(0) == '#'))
172  continue;
173  Core::stripWhitespace(s);
174  segmentList_.push_back(s);
175  }
176 
177  predefinedOrder_ = true;
178 }

200 void SegmentOrderingVisitor::enterCorpus(Corpus* c) {
201  if (!curCorpus_.empty()) {

214 void SegmentOrderingVisitor::prepareSegmentLoop() {
215  if (autoShuffle_)
216  std::shuffle(segmentList_.begin(), segmentList_.end(), shuffleRandomEngine_);
217 
218  if (sortByTimeLength_) {
219  size_t n0 = 0;
220  while (n0 < segmentList_.size()) {
221  // Sort remaining segments - or chunk of it.
222  size_t n = segmentList_.size() - n0;
223  if (sortByTimeLengthChunkSize_ > 0)
224  n = std::min(n, (size_t)sortByTimeLengthChunkSize_);
225 
226  // Stable sort so that we keep the order deterministic.
227  std::stable_sort(
228  &segmentList_[n0], &segmentList_[n0] + n,
229  [this](const std::string& sn0, const std::string& sn1) {
230  Bliss::Segment* s0 = this->getSegmentByName(sn0);
231  Bliss::Segment* s1 = this->getSegmentByName(sn1);
232  require(s0 && s1);
233  Bliss::Time st0 = s0->end() - s0->start();
234  Bliss::Time st1 = s1->end() - s1->start();
235  return st0 < st1;
236  });
237 
238  n0 += n;
239  }
240  }
241 }

249 void SegmentOrderingVisitor::leaveCorpus(Corpus* corpus) {
250  curCorpus_.pop_back();
251  if (!curCorpus_.empty()) {
252  // not the root corpus
253  return;
254  }
255  // corpus is the root corpus. We don't have our own copy of this one.
256 
257  prepareSegmentLoop();
258 
259  CustomCorpusGuide corpusGuide(this, /* root */ corpus);
260  for (std::vector<std::string>::const_iterator name = segmentList_.begin(); name != segmentList_.end(); ++name)
261  corpusGuide.showSegmentByName(*name);
262 
263  finishSegmentLoop();
264 }

292 void SegmentOrderingVisitor::CustomCorpusGuide::showSegment(Segment* segment) {
293  if (segment->recording() != curRecording_) {
294  if (curRecording_)
295  parent_->visitor_->leaveRecording(curRecording_);
296  curRecording_ = segment->recording();
297  Corpus* recCorpus = static_cast<Corpus*>(curRecording_->parent());
298  if (recCorpus != curCorpus_) {
299  if (curCorpus_ != rootCorpus_) {
300  // Leave the sub-corpus.
301  parent_->visitor_->leaveCorpus(curCorpus_);
302  }
303  if (recCorpus != rootCorpus_) {
304  // This recCorpus is a sub-corpus.
305  parent_->visitor_->enterCorpus(recCorpus);
306  }
307  curCorpus_ = recCorpus;
308  }
309  parent_->visitor_->enterRecording(segment->recording());
310  }
311 
312  segment->accept(parent_->visitor_);
313 }
