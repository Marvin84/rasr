24 /** CorpusProcessorNode: base class for nodes implementing a whole pass over a corpus
25  *
26  * This class derives from Flow::Node and from an arbirtary CorpusProcessor
27  * The node generates output objects by passing processing corpus file: data is extracted from the
28  * dataSource (Flow::Network) and it is processed by the CorpusProcessor base class.
29  * The generated objects are sent the the next node in the network.
30  *
31  */
32 
33 template<class CorpusProcessorType>
34 class CorpusProcessorNode : public virtual Flow::Node,
35  public CorpusProcessorType {
36 private:
37  typedef CorpusProcessorType CorpusProcessorPrecursor;
38 
39 private:
40  const std::string dataSourceSelection;
41  const std::string corpusSelection;
42 
43  Flow::Network dataSource_;
44 
45 private:
46  bool dataSourceParameterName(std::string& name) {
47  if (name.find(dataSourceSelection + ".") == 0) {
48  name = name.substr(dataSourceSelection.size() + 1);
49  return true;
50  }
51  return false;
52  }

54  virtual bool work(Flow::PortId p) {
55  Speech::CorpusVisitor corpusVisitor(CorpusProcessorPrecursor::config, dataSource_, *this);
56  Bliss::CorpusDescription corpusDescription(CorpusProcessorPrecursor::select(corpusSelection));
57 
58  corpusDescription.accept(&corpusVisitor);
59 
60  return putData(p);
61  }
62 
63 protected:
64  /** putData: callback function to put the output into the network.
65  *
66  * It is called after the CorpusVisitor object has finished the corpus.
67  * Use the base functionality of Flow:Node to put the results of the processing into the network
68  *
69  * @param portId is the port on which a data object was requested
70  */
71 
72  virtual bool putData(Flow::PortId portId) = 0;

74 public:

79  dataSourceSelection("data-source"),

87  /** setParameter: passes parameters to the underlying dataSource network
88  *
89  * Network parameters of the main network are passed to the underlying dataSource network.
90  * E.g.: <node name="corpus-processor-1" filter="speech-corpus-processor"
91  * data-source.process-time="$(time)">
92  * The example node corpus-processor-1 hands on the value of its network parameter $(time) to the underlying
93  * dataSource network under the name "process-time".
94  *
95  * Parameters starting with "data-source." are only handed on to the underlying network.
96  *
97  * Remark:
98  * do not forget to call this function from the derived classes
99  */
100 
101  virtual bool setParameter(const std::string& name, const std::string& value) {
102  std::string n(name);
103 
104  if (dataSourceParameterName(n))
105  return dataSource_.setParameter(n, value);
106  else
107  return false;
108 
109  return true;
110  }
111 };
