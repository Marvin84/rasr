52  StateId mapState(Fsa::StateId fsaId) {
53  StateIdMap::const_iterator id = stateIdMap_.find(fsaId);
54  if (id == stateIdMap_.end()) {
55  StateId newId = fst_->AddState();
56  stateIdMap_.insert(std::make_pair(fsaId, newId));
57  return newId;
58  }
59  else {
60  return id->second;
61  }
62  }

64  Label mapLabel(SymbolTable* fstSymbols, Fsa::ConstAlphabetRef fsaSymbols, LabelMap& map, Fsa::LabelId fsaId) {
65  LabelMap::const_iterator it = map.find(fsaId);
66  if (it == map.end()) {
67  Label fstId;
68  std::string strLabel = fsaSymbols->symbol(fsaId);
69  if (!strLabel.empty())
70  fstId = fstSymbols->Find(strLabel.c_str());
71  else {
72  // std::cerr << "unknown label: " << fsaId << std::endl;
73  fstId = fstSymbols->AddSymbol(Core::form("fsa-label-%d", fsaId));
74  }
75  map.insert(std::make_pair(fsaId, fstId));
76  return fstId;
77  }
78  else {
79  return it->second;
80  }
81  }

83  Label mapInputLabel(Fsa::LabelId fsaId) {
84  return mapLabel(fst_->InputSymbols(), Precursor::fsa_->getInputAlphabet(), inputSymbolMap_, fsaId);
85  }

88  return mapLabel(fst_->OutputSymbols(), Precursor::fsa_->getOutputAlphabet(), outputSymbolMap_, fsaId);

91 public:
92  ConvertFsaDfsState(_ConstAutomatonRef f, VectorFst* fst)
93  : Precursor(f), fst_(fst) {
94  inputSymbolMap_.insert(std::make_pair(Fsa::Epsilon, Epsilon));

98  void discoverState(_ConstStateRef sp) {
99  Fsa::StateId fsaId = sp->id();
100  StateId id = mapState(fsaId);
101  if (fsaId == Precursor::fsa_->initialStateId())
102  fst_->SetStart(id);
103  if (sp->isFinal())
104  fst_->SetFinal(id, Weight(sp->weight_));
105  for (typename _State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
106  Label output = mapOutputLabel(Precursor::fsa_->type() == Fsa::TypeTransducer ? a->output() : a->input());
107  Label input = mapInputLabel(a->input());
108  fst_->AddArc(id, Arc(input, output, Weight(a->weight()), mapState(a->target())));
109  }
110  }
