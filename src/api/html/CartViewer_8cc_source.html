93 int CartViewer::main(const std::vector<std::string>& argv) {
94  {
95  std::string filename =
96  (!argv.empty()) ? argv[0] : DecisionTree::paramCartFilename(select("decision-tree"));
97  verify(!filename.empty());
98  tree_ = new DecisionTree(config);
99  if (!tree_->loadFromFile(filename)) {
100  error("Could not load decision tree from file \"%s\"", filename.c_str());
101  return 1;
102  }
103 
104  Core::Channel plainChannel(select("decision-tree"), "plain", Core::Channel::disabled);
105  if (plainChannel.isOpen())
106  tree_->write(plainChannel);
107  Core::XmlChannel xmlChannel(select("decision-tree"), "xml", Core::Channel::disabled);
108  if (xmlChannel.isOpen())
109  tree_->writeXml(xmlChannel);
110  Core::Channel dotChannel(select("decision-tree"), "dot", Core::Channel::disabled);
111  if (dotChannel.isOpen())
112  tree_->draw(dotChannel);
113  Core::Channel statChannel(select("decision-tree"), "statistics", Core::Channel::disabled);
114  if (statChannel.isOpen())
115  statChannel << "depth " << tree_->depth() << std::endl
116  << "nodes " << tree_->nNodes() << std::endl
117  << "leaves " << tree_->nLeaves() << std::endl;
118  }
119 
120  if (paramInteractive(config)) {
121  {
122  const PropertyMap& map = tree_->map();
123  keys_.resize(map.size());
124  for (PropertyMap::Index i = 0; i < PropertyMap::Index(map.size()); ++i) {
125  keys_[i] = map.key(i);
126  maxKeySize_ = std::max(maxKeySize_, keys_[i].size());
127  }
128  }
129  log("start interactive mode");
130  run();
131  }
132 
133  return 0;
134 }

160 bool CartViewer::readValue(u32 row, u32 col, PropertyMap::Index keyIndex, std::string& value) {
161  value.clear();
162  cout << xterm::move(row, col) << " " << flush;
163  std::getline(cin, value);
164  if (value.empty()) {
165  value = tree_->map().undefinedString;
166  return true;
167  }
168  const Core::Choice& choice = tree_->map()[keyIndex];
169  if (tree_->map().isDefined(choice[value]))
170  return true;
171  for (Core::Choice::const_iterator it = choice.begin();
172  it != choice.end(); ++it) {
173  if (it->ident().find(value) == 0) {
174  value = it->ident();
175  return true;
176  }
177  }
178  return false;
179 }

240 void CartViewer::run() {
241  StringList values(tree_->map().size(), tree_->map().undefinedString);
242  PropertyMap::Index numberOfKeys = PropertyMap::Index(keys_.size());
243 
244  u32 headerX =
245  1 + HEADER_X_OFFSET;
246  u32 headerY =
247  1 + HEADER_Y_OFFSET;
248  u32 bodyX =
249  1 + BODY_X_OFFSET;
250  u32 bodyY =
251  headerY + 1 + BODY_Y_OFFSET;
252  u32 footerX =
253  1 + FOOTER_X_OFFSET;
254  u32 footerY =
255  bodyY + u32(numberOfKeys) +
256  FOOTER_Y_OFFSET;
257  u32 inputY;
258  u32 inputX =
259  bodyX + u32(maxKeySize_) + 3;
260 
261  do {
262  cout << xterm::clear;
263  displayMask(bodyY, bodyX);
264  StringList::iterator valueIt = values.begin();
265  inputY = bodyY;
266  for (PropertyMap::Index i = 0; i < numberOfKeys; ++i, ++inputY, ++valueIt) {
267  displayString(headerY, headerX, "Please enter value");
268  displayValues(footerY, footerX, i);
269  while (!readValue(inputY, inputX, i, *valueIt)) {
270  displayError(headerY, headerX, "Could not find matching value, please try again");
271  cout << xterm::move(inputY, inputX) << xterm::kill;
272  }
273  displayString(inputY, inputX, *valueIt);
274  }
275  displayString(headerY, headerX, "Classify ...");
276  displayString(footerY, footerX, "");
277  cout << xterm::move(footerY, 0) << endl;
278 
279  Properties* props = new StoredProperties(tree_->getMap(), keys_, values);
280  if (mode_ == 0) {
281  DecisionTree::Path path = tree_->findPath(*props);
282  displayPath(path);
283  }
284  else {
285  DecisionTree::NodePtrList list = tree_->findAll(*props);
286  displayMultiple(list);
287  }
288  delete props;
289 
290  } while (readCommand());
291 }
