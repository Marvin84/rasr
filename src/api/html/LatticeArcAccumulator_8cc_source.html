41 void LemmaPronunciationId_hash_map::read(const std::string& filename) {
42  Core::TextInputStream stats(filename);
43  std::string line, orth, phons;
44  f32 score;
45 
46  while (std::getline(stats, line)) {
47  std::vector<std::string> instance = Core::split(line, "\t");
48 
49  verify(instance.size() == 3);
50 
51  orth = instance[0];
52  phons = instance[1];
53  score = atof(instance[2].c_str());
54 
55  Bliss::Lexicon::LemmaPronunciationIterator lpi, lpi_end;
56  std::string prefOrth;
57 
58  for (Core::tie(lpi, lpi_end) = lexicon->lemmaPronunciations(); lpi != lpi_end; ++lpi) {
59  const Bliss::LemmaPronunciation* lp = (*lpi);
60  prefOrth = std::string(lp->lemma()->preferredOrthographicForm());
61  if (prefOrth == orth) {
62  if (find(lp->id()) == this->end())
63  insert(std::pair<Fsa::LabelId, f32>(lp->id(), score));
64  else
65  (*this)[lp->id()] += score;
66 
67  break;
68  }
69  }
70  }

141 LatticeArcAccumulatorNode::~LatticeArcAccumulatorNode() {
142  //write statistics write-filename given
143  if (pronunciationStatsFilenameWrite_ != "")
144  arcAccumulator_->write(pronunciationStatsFilenameWrite_);
145 
146  //exit if lexicon-filename not given
147  if (lexiconFilename_ == "")
148  return;
149 
150  std::unordered_map<Fsa::LabelId, LemmaPronunciationId_hash_map> counts;
151  std::unordered_map<Fsa::LabelId, LemmaPronunciationId_hash_map> weights;
152 
153  // make a copy of the Lexicon, because it is modified
154  Core::Ref<const Bliss::Lexicon> lexicon = modelCombination_->lexicon();
155  Core::Ref<const Bliss::LemmaPronunciationAlphabet> lpa = lexicon->lemmaPronunciationAlphabet();
156 
157  Bliss::Lexicon::LemmaPronunciationIterator lpi, lpi_end;
158  for (Core::tie(lpi, lpi_end) = lexicon->lemmaPronunciations(); lpi != lpi_end; ++lpi) {
159  const Bliss::LemmaPronunciation* lp = (*lpi);
160  Fsa::LabelId lemmaId = lp->lemma()->id();
161  Fsa::LabelId lemmaPronId = lp->id();
162 
163  weights[lemmaId][lemmaPronId] = lp->pronunciationProbability();
164  counts[lemmaId][lemmaPronId] = 0.0;
165  }
166 
167  f32 totalCounts = 0.0;
168  std::unordered_map<Fsa::LabelId, f32>::const_iterator i;
169  for (i = (*arcAccumulator_).begin(); i != (*arcAccumulator_).end(); i++) {
170  const Bliss::LemmaPronunciation* lp = lpa->lemmaPronunciation(i->first);
171  if (lp) {
172  Fsa::LabelId lemmaId = lp->lemma()->id();
173  Fsa::LabelId lemmaPronId = lp->id();
174  verify(i->second >= 0.0);
175  counts[lemmaId][lemmaPronId] += i->second;
176  totalCounts += i->second;
177  }
178  }
179  log("total-counts: %f", totalCounts);
180 
181  std::unordered_map<Fsa::LabelId, LemmaPronunciationId_hash_map>::const_iterator l;
182  for (l = weights.begin(); l != weights.end(); l++) {
183  Fsa::LabelId lemmaId = l->first;
184  updateWeightsMap(counts[lemmaId], weights[lemmaId], tau_);
185  }
186 
187  for (Core::tie(lpi, lpi_end) = lexicon->lemmaPronunciations(); lpi != lpi_end; ++lpi) {
188  // a const_cast is not desirable, but the Lexicon stores only
189  // pointers to const LemmaPronunciations.
190  // We hope that the program terminates shortly afterwards.
191  /*! @todo: don't change the lexicon, change only the written file */
192  Bliss::LemmaPronunciation* lp = const_cast<Bliss::LemmaPronunciation*>(*lpi);
193  Fsa::LabelId lemmaId = lp->lemma()->id();
194  Fsa::LabelId lemmaPronId = lp->id();
195  f32 pronWeight = weights[lemmaId][lemmaPronId];
196  if (pronWeight < minimalPronWeight_)
197  pronWeight = minimalPronWeight_;
198  lp->setPronunciationProbability(pronWeight);
199  }
200  Core::TextOutputStream o(lexiconFilename_);
201  o.setEncoding(encoding_);
202  Core::XmlWriter xo(o);
203  o.setIndentation(2);
204  xo.generateFormattingHints();
205  lexicon->writeXml(xo);
206  o.close();
207  delete arcAccumulator_;
208 }

210 bool LatticeArcAccumulatorNode::updateWeightsMap(std::unordered_map<Fsa::LabelId, f32>& counts,
211  std::unordered_map<Fsa::LabelId, f32>& weights,
212  f32 tau) {
213  f32 totalCount = 0.0;
214  std::unordered_map<Fsa::LabelId, f32>::const_iterator pron_i;
215  for (pron_i = counts.begin(); pron_i != counts.end(); pron_i++)
216  totalCount += pron_i->second;
217 
218  if (totalCount > 0.0) {
219  f32 interpolationWeight = totalCount / (tau + totalCount);
220  for (pron_i = counts.begin(); pron_i != counts.end(); pron_i++) {
221  Fsa::LabelId pronId = pron_i->first;
222  weights[pronId] =
223  interpolationWeight * (pron_i->second / totalCount) + (1.0 - interpolationWeight) * weights[pronId];
224  }
225  }
226  return true;
227 }
