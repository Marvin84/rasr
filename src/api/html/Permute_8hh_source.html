31 class Bitvector {
32 private:
33  size_t size_;
34  std::vector<u32> bits_;
35 
36 public:
37  Bitvector(const Bitvector& v, size_t flipPos)
38  : size_(v.size_), bits_(v.bits_) {
39  flip(flipPos);
40  }
41  Bitvector(size_t size, bool value = false)
42  : size_(size), bits_((size + 31) / 32, value ? ~u32(0) : 0) {}
43  size_t size() const {
44  return size_;
45  }
46  bool operator[](size_t pos) const {
47  return (bits_[pos / 32] >> (pos % 32)) & 1;
48  }
49  bool operator==(const Bitvector& v) const {
50  return (bits_ == v.bits_);
51  }
52  bool operator<(const Bitvector& v) const {
53  return bits_ < v.bits_;
54  }
55 
56  void flip(size_t pos) {
57  bits_[pos / 32] ^= (1 << (pos % 32));
58  }
59  u32 hashKey() const {
60  u32 value = 0;
61  for (std::vector<u32>::const_iterator i = bits_.begin(); i != bits_.end(); ++i)
62  value |= *i;
63  return value;
64  }
65 
66  static void printBin(std::ostream& out, u32 x) {
67  // print u32 in readable form, usefull for debugging
68  for (int i = 0; i < 32; ++i)
69  out << ((x >> i) & 1 ? "1" : "0");
70  }
71 
72  bool isCovered(size_t A, size_t B) const {
73  // check if range [A,B) is covered
74 
75  size_t Abi = A / 32, Bbi = B / 32;
76  for (size_t i = Abi + 1; i < Bbi; ++i)
77  if (!(bits_[i] == ~u32(0)))
78  return 0;
79 
80  u32 maskA = (~u32(0) >> (A % 32)) << (A % 32);
81  u32 maskB = (~u32(0) << (32 - (B % 32))) >> (32 - (B % 32));
82 
83  if (Abi == Bbi)
84  return (bits_[Abi] & (maskA & maskB)) == (maskA & maskB);
85  else
86  return ((bits_[Abi] & maskA) == maskA) && ((bits_[Bbi] & maskB) == maskB);
87  }
88 
89  friend std::ostream& operator<<(std::ostream& o, const Bitvector& b) {
90  for (u32 i = 0; i < b.size_; ++i)
91  o << (b[i] ? "1" : "0");
92  return o;
93  }

103 struct LogWeighting {
104  bool isOrdered_;
105  f32 baseProbability_; // probability \alpha for "normal direction" (no reordering)
106  f32 logBaseProbability_;
107  ConstSemiringRef semiring_;
108  u32 nUncoveredPositionsInWindow_;
109  u32 windowSize_;
110  std::vector<f32> logProbabilitiesForReorderedPaths;
111  std::vector<f32> logProbabilitiesWitoutOrderedPath;
112  LogWeighting(ConstSemiringRef semiring, f32 baseProbability, u32 windowSize)
113  : baseProbability_(baseProbability), semiring_(semiring), windowSize_(windowSize), logProbabilitiesForReorderedPaths(windowSize_ + 1), logProbabilitiesWitoutOrderedPath(windowSize_ + 1) {
114  logBaseProbability_ = -log(baseProbability_);
115  for (u32 i = 1; i <= windowSize_; ++i) {
116  logProbabilitiesForReorderedPaths[i] = -log((1 - baseProbability_) / (i - 1));
117  logProbabilitiesWitoutOrderedPath[i] = -log(1.0 / i);
118  }
119  }
120  void processState(const Bitvector& used, const u32 lowerLimit, const u32 upperLimit) {
121  u32 b;
122  isOrdered_ = true;
123  for (b = 0; b < used.size() && used[b]; ++b)
124  ;
125  for (; b < used.size() && !used[b]; ++b)
126  ;
127  if (b < used.size())
128  isOrdered_ = false;
129  nUncoveredPositionsInWindow_ = 0;
130  for (u32 i = lowerLimit; i < upperLimit; ++i)
131  if (!used[i])
132  nUncoveredPositionsInWindow_++; // Here: ++ only if this position does not violate the distortion limit!
133  // (to be implemented! -> states_[s].depth_ must come in as a parameter!)
134  }
135  const void processArc(const bool isNextReordered, Arc* const a) {
136  Weight weight = semiring_->one();
137  if (nUncoveredPositionsInWindow_ == 1)
138  weight = Weight(0.0);
139  else if (isOrdered_) {
140  if (isNextReordered)
141  weight = Weight(logProbabilitiesForReorderedPaths[nUncoveredPositionsInWindow_]);
142  else
143  weight = Weight(logBaseProbability_);
144  }
145  else
146  weight = Weight(logProbabilitiesWitoutOrderedPath[nUncoveredPositionsInWindow_]);
147  a->weight_ = semiring_->extend(a->weight_, weight);
148  }
149  void setDistortionLimit(u16 distortionLimit) {}
150 };

162 
163 protected:
164  std::vector<Arc> arcs_;
165  Weight initialWeight_, finalWeight_;
166  Processing* processing_;
167  u32 windowSize_;
168 
169 public:
170  PermuteAutomatonBase(ConstAutomatonRef f, u32 windowSize, Processing* processing)
171  : DfsState(cache(f)), processing_(processing) {
172  if (!hasProperties(fsa_, PropertyLinear))
173  std::cerr << "permute is defined properly for linear automata only." << std::endl;
174  setProperties(PropertyAcyclic, PropertyAcyclic);
175  initialWeight_ = fsa_->semiring()->one();
176  finalWeight_ = fsa_->semiring()->one();
177  dfs();
178  if (arcs_.size() > 0)
179  // assumes commutative semiring
180  arcs_[0].weight_ = fsa_->semiring()->extend(initialWeight_, arcs_[0].weight_);
181  windowSize_ = std::min(size_t(windowSize), arcs_.size());
182  }
183  virtual ~PermuteAutomatonBase() {
184  delete processing_;
185  }
186  virtual void discoverState(ConstStateRef sp) {
187  for (State::const_iterator a = sp->begin(); a != sp->end(); ++a)
188  if ((a->input() == Epsilon) && (a->output() == Epsilon)) {
189  if (arcs_.size() > 0)
190  arcs_[arcs_.size() - 1].weight_ =
191  fsa_->semiring()->extend(arcs_[arcs_.size() - 1].weight(), a->weight());
192  else
193  initialWeight_ = fsa_->semiring()->extend(initialWeight_, a->weight());
194  }
195  else
196  arcs_.push_back(*a);
197  if (sp->isFinal())
198  finalWeight_ = sp->weight_;
199  }
200 
201  virtual Type type() const {
202  return fsa_->type();
203  }
204  virtual ConstSemiringRef semiring() const {
205  return fsa_->semiring();
206  }
207  virtual StateId initialStateId() const {
208  return 0;
209  }

212  }

260 
261  virtual ConstStateRef getState(StateId s) const {
262  if (s < states_.size()) {
263  State* sp = new State();
264  sp->setId(s);

281  }

284  states_.getMemoryUsed();

385 class DoubleLocalPermuteAutomaton : public PermuteAutomaton<Processing> {
386 private:
387  typedef PermuteAutomaton<Processing> Precursor;
388 
389 public:
390  DoubleLocalPermuteAutomaton(ConstAutomatonRef f, u32 windowSize, u16 distortionLimit, Processing* processing)
391  : PermuteAutomaton<Processing>(f, windowSize, distortionLimit, processing){};
392 
393  virtual void permuteArcs(State* sp) const {
394  // normal local window
395  StateId s = sp->id();
396  u32 lowerLimit = Precursor::states_[s].depth_;
397  u32 upperLimit = std::min(size_t(Precursor::states_[s].depth_ + Precursor::windowSize_), Precursor::states_[s].used_.size());
398  for (u32 i = lowerLimit; i < upperLimit; ++i)
399  if (!Precursor::states_[s].used_[i]) {
400  u32 depth;
401  if (i == Precursor::states_[s].depth_)
402  for (depth = Precursor::states_[s].depth_ + 1;
403  depth < Precursor::states_[s].used_.size() && Precursor::states_[s].used_[depth];
404  depth++)
405  ;
406  else
407  depth = Precursor::states_[s].depth_;
408  Arc* a = sp->newArc();
409  *a = Precursor::arcs_[i];
410  a->target_ = Precursor::insertState(depth, Precursor::states_[s].used_, i);
411  }
412 
413  // "inverse" local window
414  for (upperLimit = Precursor::states_[s].used_.size() - 1; Precursor::states_[s].used_[upperLimit] && upperLimit > 0; upperLimit--)
415  ;
416  lowerLimit = std::max(int(upperLimit - Precursor::windowSize_), int(Precursor::states_[s].depth_ + Precursor::windowSize_ - 1));
417  for (u32 i = upperLimit; i > lowerLimit; --i)
418  if (!Precursor::states_[s].used_[i]) {
419  u32 depth = Precursor::states_[s].depth_;
420  if (i == Precursor::states_[s].depth_)
421  depth++;
422  Arc* a = sp->newArc();
423  *a = Precursor::arcs_[i];
424  a->target_ = Precursor::insertState(depth, Precursor::states_[s].used_, i);
425  }
426  }
427  virtual std::string describe() const {
428  return Core::form("double-local-permute(%s,%d)", Precursor::fsa_->describe().c_str(), Precursor::windowSize_);
429  }

438  struct Spans : public std::vector<Span> {
439  typedef std::vector<Span> Parent;
440 
441  Spans()
442  : Parent() {}
443  Spans(const Spans& x)
444  : Parent(x) {}

454 
455  void reduce() {
456  while (reduce_())
457  ;
458  }
459  bool reduce_() {
460  if (Parent::size() < 2)
461  return false;
462  Span& a = Parent::operator[](Parent::size() - 2);
463  Span& b = Parent::back();
464  if (a.first == b.second) {
465  a.first = b.first;
466  Parent::pop_back();
467  return true;
468  }
469  if (a.second == b.first) {
470  a.second = b.second;
471  Parent::pop_back();
472  return true;
473  }
474  return false;
475  }

478  return (Parent::size() == 1);
479  }
480 
481  u32 hashKey() const {
482  return 59 * Parent::size();
483  }

486  for (size_t i = 0; i < x.size(); ++i)

488  return out;
489  }
490 
491  friend bool operator==(const Spans& a, const Spans& b) {

573  sp->weight_ = Precursor::fsa_->semiring()->one();
574

576  permuteArcs(sp);
577  }
578  else {
579  // check ITG constraints
580  if (states_[s].isValid()) {
581  sp->addTags(StateTagFinal);

590  virtual void permuteArcs(State* sp) const {
591  StateId s = sp->id();
592  u32 lowerLimit = 0;
593  u32 upperLimit = std::min(size_t(states_[s].depth_ + Precursor::windowSize_), states_[s].used_.size());
594

596  if (!states_[s].used_[i]) {

609  }
