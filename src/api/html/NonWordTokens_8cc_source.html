37 bool NonWordTokens::init() {
38  std::vector<std::string> lemmas = paramLemmas(config);
39  std::vector<std::string> phones = paramPhones(config);
40  if (paramUseSyntacticTokens(config)) {
41  log("using empty syntactic tokens");
42  setEmptySyntacticTokens();
43  }
44  else if (paramUseSilence(config)) {
45  log("using only silence");
46  setSilence();
47  }
48  else if (!lemmas.empty()) {
49  log("using %d lemmas", static_cast<int>(lemmas.size()));
50  setLemmas(lemmas);
51  }
52  else if (!phones.empty()) {
53  log("using %d phones", static_cast<int>(phones.size()));
54  setPhones(phones);
55  }
56  std::sort(lemmas.begin(), lemmas.end());
57  std::sort(phones.begin(), phones.end());
58  setNonWordPhones();
59  logSettings();
60  return true;
61 }

180  for (std::vector<std::string>::const_iterator symbol = lemmas.begin(); symbol != lemmas.end(); ++symbol) {
