140 SearchSpaceBase* ExpandingFsaSearch::createSearchSpace() {
141  NetworkType networkType = static_cast<NetworkType>(paramNetworkType_(config));
142  const bool allowSkips = paramAllowSkips(config);
143  log("HMM skips: %s", (allowSkips ? "true" : "false"));
144  Core::Timer timer;
145  timer.start();
146  SearchSpaceBase* result = SearchSpaceBase::create(networkType, allowSkips, config);
147  result->setPruningThreshold(paramAcousticPruningThreshold_(config));
148  log("using acoustic pruning threshold %0.2f", paramAcousticPruningThreshold_(config));
149  result->setPruningLimit(paramAcousticPruningLimit_(config));
150  result->setPruningBins(paramAcousticPruningBins_(config));
151  log("using acoustic pruning limit %d using %d bins",
152  paramAcousticPruningLimit_(config), paramAcousticPruningBins_(config));
153  result->setInitialEpsilonPruning(paramInitialEpsilonPruning_(config));
154  if (paramInitialEpsilonPruning_(config))
155  log("using initial epsilon pruning");
156  result->setEpsilonPruning(paramEpsilonArcPruning_(config));
157  if (paramEpsilonArcPruning_(config))
158  log("using epsilon arc pruning");
159  result->setProspectivePruning(paramProspectivePruning_(config));
160  if (paramProspectivePruning_(config))
161  log("using prospective pruning");
162  result->setLatticePruning(paramLatticePruning_(config));
163  log("using lattice pruning threshold %0.2f", paramLatticePruning_(config));
164  if (!Core::isAlmostEqual(paramWordEndPruning_(config), paramWordEndPruning_.defaultValue(), 0.1)) {
165  result->setWordEndPruning(true, paramWordEndPruning_(config));
166  log("using word end pruning. threshold %0.2f", paramWordEndPruning_(config));
167  }
168  result->setMergeSilenceLatticeArcs(paramMergeSilenceArcs_(config));
169  result->setMergeEpsilonPaths(paramMergeEpsilonPaths_(config));
170  result->setCreateLattice(paramCreateLattice(config),
171  static_cast<LatticeTraceRecorder::LatticeType>(paramLatticeType(config)));
172  result->setPurgeInterval(paramPurgeInterval(config));
173  result->setWeightScale(paramWeightScale(config));
174  log("arc weight scale: %0.2f", paramWeightScale(config));
175  result->setWordEndType(static_cast<WordEndDetector::WordEndType>(paramWordEndType_(config)));
176  result->setIgnoreLastOutput(paramIgnoreLastOutput(config));
177 
178  std::string statesFile = paramStateSequencesFile_(config);
179  if (statesFile.empty()) {
180  statesFile = paramEmissionSequencesFile_(config);
181  warning("using deprecated parameter %s", paramEmissionSequencesFile_.name().c_str());
182  }
183  log("reading state sequences from '%s'", statesFile.c_str());
184  stateSequences_ = new StateSequenceList();
185  if (!stateSequences_->read(statesFile))
186  error("cannot read state sequence file from '%s'", statesFile.c_str());
187  else
188  log("# state sequences: %d", u32(stateSequences_->size()));
189  result->setStateSequences(stateSequences_);
190 
191  result->setStatistics(paramDetailedStatistics(config));
192  timer.stop();
193  if (statisticsChannel_.isOpen()) {
194  statisticsChannel_ << Core::XmlOpen("search-space-preparation");
195  statisticsChannel_ << timer;
196  statisticsChannel_ << Core::XmlClose("search-space-preparation");
197  }
198  return result;
199 }

272 bool ExpandingFsaSearch::setModelCombination(const Speech::ModelCombination& modelCombination) {
273  lexicon_ = modelCombination.lexicon();
274  searchSpace_->setLexicon(lexicon_);
275  searchSpace_->setTransitionModel(modelCombination.acousticModel());
276  if (paramHasNonWords(config)) {
277  NonWordTokens nonWordTokens(select("non-word-tokens"), *lexicon_);
278  nonWordTokens.init();
279  u32 nNonWordModels = nonWordTokens.phones().size();
280  log("assuming last %d state sequences are non-word models", nNonWordModels);
281  searchSpace_->setUseNonWordModels(nNonWordModels);
282  }
283 
284  if (!paramNonWordOutput(config)) {
285  if (!searchSpace_->setNonWordPhones(modelCombination.acousticModel(), paramNonWordPhones(config))) {
286  error("cannot set non-word phones");
287  return false;
288  }
289  else {
290  log("%d non-word tokens without output", int(paramNonWordPhones(config).size()));
291  }
292  }
293  verify(stateSequences_);
294  StateSequenceResolver resolver(modelCombination.acousticModel(), *stateSequences_);
295  searchSpace_->setSilence(resolver.findSilence(lexicon_), silenceOutput());
296  return true;
297 }

308 s32 ExpandingFsaSearch::silenceOutput() const {
309  OpenFst::Label label = OpenFst::InvalidLabelId;
310  const Bliss::Lemma* lemma = lexicon_->specialLemma("silence");
311  if (outputType_ == OutputLemma) {
312  label = OpenFst::convertLabelFromFsa(lemma->id());
313  }
314  else if (outputType_ == OutputLemmaPronunciation) {
315  verify_eq(lemma->nPronunciations(), 1);
316  label = OpenFst::convertLabelFromFsa(lemma->pronunciations().first->id());
317  }
318  return label;
319 }
