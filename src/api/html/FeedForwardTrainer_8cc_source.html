138 template<typename T>
139 void FeedForwardTrainer<T>::initializeTrainer(u32 batchSize, std::vector<u32>& streamSizes) {
140  if (Precursor::needInit_) {
141  Precursor::initializeTrainer(batchSize, streamSizes);
142  if (this->hasNetwork()) {
143  lowestTrainableLayerIndex_ = -1;
144  }
145  // get trainable layer with lowest index
146  lowestTrainableLayerIndex_ = (s32)this->nLayers() - 1;
147  for (s32 layer = (s32)this->nLayers() - 1; layer >= 0; layer--) {
148  if (network().getLayer(layer).isTrainable()) {
149  lowestTrainableLayerIndex_ = layer;
150  }
151  }
152  // initialize error signal
153  errorSignal_.resize(this->nLayers());
154  for (u32 layer = 0; layer < this->nLayers(); layer++) {
155  errorSignal_[layer].resize(network().getLayer(layer).getOutputDimension(), batchSize);
156  }
157  errorSignalOut_.resize(this->nLayers());
158  for (s32 layer = (s32)this->nLayers() - 1; layer > lowestTrainableLayerIndex_; layer--) {
159  std::vector<NnMatrix*> errorSignalOut;
160  for (u32 i = 0; i < network().getLayer(layer).nPredecessors(); i++) {
161  errorSignalOut.push_back(&(errorSignal_[network().getLayer(layer).getPredecessor(i)]));
162  }
163  errorSignalOut_.at(layer) = errorSignalOut;
164  }
165  // initialize statistics
166  u32 statisticsType = estimator().requiredStatistics();
167  if (this->hasNetwork() && (statisticsChannel_.isOpen() || statisticsFilename_ != "")) // otherwise this information would be lost anyway
168  statisticsType |= Statistics<T>::BASE_STATISTICS;
169 
170  statistics_ = new Statistics<T>(this->nLayers(), statisticsType);
171  if (this->hasNetwork())
172  statistics_->initialize(network());
173  if (useDoublePrecisionAccumulator_)
174  initializeDoublePrecisionStatistics();
175  // initialize computation
176  statistics_->initComputation();
177  statistics_->reset();
178  for (u32 layer = 0; layer < errorSignal_.size(); layer++)
179  errorSignal_[layer].initComputation();
180  Precursor::needInit_ = false;
181  }
182 }

184 template<typename T>
185 void FeedForwardTrainer<T>::finalize() {
186  if (estimator().fullBatchMode() && statistics_) {
187  Prior<T> prior(this->config);
188  std::string priorFilename = prior.fileName();
189  if (priorFilename != "" && statistics().hasClassCounts()) {
190  require(this->classWeights_);
191  prior.setFromClassCounts(statistics(), *this->classWeights_);
192  prior.write();
193  }
194 
195  if (statistics().hasGradient() || statistics().hasBaseStatistics()) {
196  if (statisticsFilename_ != "") {
197  if (doublePrecisionStatistics_) {
198  statistics_->reset();
199  // convert double precision to single precision statistics
200  statistics_->add(*doublePrecisionStatistics_);
201  }
202  // write statistics
203  statistics().finishComputation();
204  statistics().write(statisticsFilename_);
205  }
206  else if (statistics().hasGradient())
207  this->warning("statistics-filename not set, do not write statistics");
208  }
209 
210  Core::Component::log("total-number-of-observations: ") << statistics_->nObservations();
211  if (statistics().hasBaseStatistics()) {
212  Core::Component::log("total-frame-classification-error: ") << (T)statistics_->classificationError();
213  Core::Component::log("total-objective-function: ") << statistics_->objectiveFunction();
214  }
215  }
216  if (!estimator().fullBatchMode() && estimator().accumulateMultipleBatches() > 1) {
217  u32 latestRecentBatchesCount = minibatchCount_ % estimator().accumulateMultipleBatches();
218  if (latestRecentBatchesCount > 0) {
219  // This means that we did not do a model-update with the latest recent batches.
220  Core::Component::log(
221  "The last %i batches were ignored because of accumulate-multiple-batches=%i",
222  latestRecentBatchesCount, estimator().accumulateMultipleBatches());
223  if (minibatchCount_ < estimator().accumulateMultipleBatches())
224  Core::Component::warning("We did not use any batches. (because accumulate-multiple-batches)");
225  }
226  }
227  else if (minibatchCount_ == 0)
228  Core::Component::warning("We did not use any batches.");
229  Precursor::finalize();
230  if (this->measureTime_) {
231  this->log() << Core::XmlOpen("time-feed-forward-nn-trainer")
232  << Core::XmlFull("forwarding", timeForwardPass_)
233  << Core::XmlFull("initial-error-signal", timeInitialErrorSignal_)
234  << Core::XmlFull("backward-pass", timeBackwardPass_)
235  << Core::XmlFull("gradient", timeGradient_)
236  << Core::XmlFull("base-statistics", timeBaseStatistics_)
237  << Core::XmlFull("regularization", timeRegularization_)
238  << Core::XmlFull("estimation", timeEstimation_)
239  << Core::XmlClose("time-feed-forward-nn-trainer");
240  }
241  {
242  this->log() << Core::XmlOpen("counts")
243  << Core::XmlFull("minibatches", minibatchCount_)
244  << Core::XmlFull("discarded-minibatches", discardedMinibatchCount_)
245  << Core::XmlClose("counts");
246  }

249 template<typename T>
250 void FeedForwardTrainer<T>::processBatch_feedInput(std::vector<NnMatrix>& features, NnVector* weights, Bliss::Segment* segment) {
251  minibatchCount_++;
252 
253  if (weightedAccumulation_) {
254  weights_ = weights;
255  if (!weights_)
256  Core::Component::warning("weightedAccumulation without weights.");
257  }
258  else {
259  weights_ = weights = NULL;
260  }
261 
262  // for profiling
263  resetBatchTimes();
264  timeval start, end;
265 
266  // initialization
267  if (doublePrecisionStatistics_)
268  // Always accumulate full-batch in double precision stats - reset single precision stats.
269  statistics_->reset();
270  else if (!estimator().fullBatchMode()) {
271  if ((minibatchCount_ - 1) % estimator().accumulateMultipleBatches() == 0)
272  statistics_->reset();
273  }
274 
275 #ifdef MODULE_PYTHON
276  pythonControl_.run_custom("init_segment", "{s:s}", "segment_name", segment ? segment->fullName().c_str() : NULL);
277 #endif
278 
279  u32 batchSize = features[0].nColumns();
280  setBatchSize(batchSize);
281  statistics_->incObservations(batchSize);
282  if (weightedAccumulation_ && weights) {
283  TIMER_START(start);
284  weights->initComputation();
285  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, timeSyncBatch_, timeSync_)
286  statistics_->addToTotalWeight(weights->asum());
287  }
288  else {
289  statistics_->addToTotalWeight(batchSize);
290  }
291 
292  // forward network
293  if (statistics_->hasBaseStatistics() || statistics_->hasGradient()) {
294  TIMER_START(start);
295  // sync required here only to include it in time measurement
296  for (u32 i = 0; i < features.size(); ++i)
297  features.at(i).initComputation();
298  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, timeSyncBatch_, timeSync_)
299 
300  TIMER_START(start);
301  network().forward(features);
302  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, timeForwardPassBatch_, timeForwardPass_)
303  }
304 }

306 template<typename T>
307 void FeedForwardTrainer<T>::processBatch_finishWithError_naturalPairing(T error, NnMatrix& errorSignal) {
308  timeval start, end;
309 
310 #ifdef MODULE_PYTHON
311  pythonControl_.run_custom(
312  "notify_segment_loss", "{s:s,s:f}",
313  "segment_name", NULL,
314  "loss", (float)error);
315 #endif
316 
317 start:
318 
319  // calculate number of classification errors and objective function
320  if (statistics_->hasBaseStatistics()) {
321  TIMER_START(start);
322  statistics_->addToObjectiveFunction(error);
323  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, timeBaseStatisticsBatch_, timeBaseStatistics_)
324 
325  // apply regularization only when not in batch mode
326  if (!estimator().fullBatchMode()) {
327  TIMER_START(start);
328  u32 batchSize = network().getLayerInput(0)[0]->nColumns();
329  T regularizerError = regularizer().objectiveFunction(network(), T(batchSize));
330  statistics_->addToObjectiveFunction(regularizerError);
331  error += regularizerError;
332 
333  if (logFrameEntropy_) {
334  Math::FastVector<T> entropy(batchSize);
335  // not necessary once FastVector::columnEntropy() is implemented on GPU
336  network().getTopLayerOutput().finishComputation(true);
337  Math::FastMatrix<T>& output = network().getTopLayerOutput().asWritableCpuMatrix();
338  entropy.columnEntropy(output);
339  network().getTopLayerOutput().initComputation(false);
340  statistics_->addToEntropy(entropy.sum());
341  }
342  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, timeRegularizationBatch_, timeRegularization_)
343  }
344  }
345 
346  if (statistics_->hasGradient()) {
347  // reset error signals
348  if (!errorSignal_.empty()) {
349  for (u32 layer = 0; layer < errorSignal_.size() - 1; layer++)
350  errorSignal_.at(layer).setToZero();
351 
352  // Special case: we have filled the matrix already inplace.
353  // In that case, nothing needs to be done. Otherwise, copy it over.
354  NnMatrix& topErrorSignal = errorSignal_[errorSignal_.size() - 1];
355  if (&errorSignal != &topErrorSignal) {
356  topErrorSignal.copy(errorSignal);
357  }
358  }
359 
360  // error backprop
361  errorBackpropagation();
362 
363  // collect gradient
364  collectGradient();
365 
366  // apply regularization only when not in batch mode
367  if (!estimator().fullBatchMode()) {
368  TIMER_START(start);
369  regularizer().addGradient(network(), statistics(), T(statistics().nObservations()));
370  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, timeRegularizationBatch_, timeRegularization_)
371  }
372  }
373 
374  if (gradientCheck_) {
375  // Check every gradient component by a simple numerical
376  // gradient approximation into that direction.
377  gradientCheck();
378  }
379 
380  // update (only if has gradient)
381  if (statistics_->hasGradient() && !estimator().fullBatchMode()) {
382  if (minibatchCount_ % estimator().accumulateMultipleBatches() == 0) {
383  // maybe normalize statistics by batch size
384  statistics_->finalize(normalizeByNOfObservations_);
385  // update model
386  TIMER_START(start);
387  estimator().estimate(network(), statistics());
388  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, timeEstimationBatch_, timeEstimation_);
389  }
390  }
391 
392  if (simpleGradientCheck_ || convergenceCheck_) {
393  // Forward again with new parameters, after the estimate.
394  // Use already set features.
395  network().forwardLayers();
396  }
397 
398  if (simpleGradientCheck_) {
399  // Make a simple numerical gradient approximation
400  // into the direction of the estimator step.
401  simpleGradientCheck(error);
402  }
403 
404  if (convergenceCheck_) {
405  require(!normalizeByNOfObservations_); // otherwise the stat.finalize-call above breaks it
406  if (convergenceCheckRepeat(error, errorSignal))
407  goto start;
408  }
409 
410  if (doublePrecisionStatistics_ && estimator().fullBatchMode())
411  doublePrecisionStatistics_->add(*statistics_);
412 
413  // logging
414  if (statisticsChannel_.isOpen() && statistics_->hasBaseStatistics() && !(estimator().fullBatchMode())) {
415  if (minibatchCount_ % estimator().accumulateMultipleBatches() == 0) {
416  // Note that we called statistics_->finalize() above, which might have normalized these.
417  statisticsChannel_ << Core::XmlOpen("batch-statistics")
418  << Core::XmlFull("mini-batch-number", minibatchCount_)
419  << Core::XmlFull("frame-classification-error-rate-on-batch", statistics_->classificationError())
420  << Core::XmlFull("objective-function-on-batch", statistics_->objectiveFunction());
421  if (estimator().accumulateMultipleBatches() > 1)
422  statisticsChannel_ << Core::XmlFull("batch-total-time-frames", statistics_->nObservations());
423  if (logFrameEntropy_)
424  statisticsChannel_ << Core::XmlFull("batch-average-entropy", statistics_->entropy());
425  statisticsChannel_ << Core::XmlClose("batch-statistics");
426  }
427  else {
428  statisticsChannel_ << Core::XmlOpen("batch-statistics-accumulated-so-far")
429  << Core::XmlFull("mini-batch-number", minibatchCount_)
430  << Core::XmlFull("accumulated-frame-classification-error-rate", statistics_->classificationError())
431  << Core::XmlFull("accumulated-objective-function", statistics_->objectiveFunction());
432  if (estimator().accumulateMultipleBatches() > 1)
433  statisticsChannel_ << Core::XmlFull("accumulated-time-frames", statistics_->nObservations());
434  statisticsChannel_ << Core::XmlClose("batch-statistics-accumulated-so-far");
435  }
436  }
437 
438  weights_ = NULL;
439 }

459 template<typename T>
460 void FeedForwardTrainer<T>::processBatch_finishWithAlignment(Math::CudaVector<u32>& alignment) {
461  timeval start, end;
462 
463  // count classes
464  if (statistics_->hasClassCounts())
465  updateClassCounts(alignment, statistics());
466 
467  {
468  TIMER_START(start);
469  alignment.initComputation();
470  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, this->timeSyncBatch_, this->timeSync_);
471  }
472 
473  // calculate objective function and classification errors
474  T error = 0;
475  bool discard = false;
476  if (statistics_->hasBaseStatistics() || statistics_->hasGradient()) {
477  TIMER_START(start);
478  Precursor::criterion_->inputAlignment(alignment, network().getTopLayerOutput(), weights_);
479  discard = Precursor::criterion_->discardCurrentInput();
480  if (!discard) {
481  Precursor::criterion_->getObjectiveFunction(error);
482  statistics_->incClassificationErrors(network().getTopLayerOutput().nClassificationErrors(alignment));
483  }
484  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, this->timeBaseStatisticsBatch_, this->timeBaseStatistics_);
485  }
486 
487  // compute gradient
488  NnMatrix dummyErrorSignal; // might be needed if no gradient is calculated
489  NnMatrix& errorSignal = errorSignal_.empty() ? dummyErrorSignal : errorSignal_[errorSignal_.size() - 1];
490 
491  if (!discard && statistics_->hasGradient()) {
492  // set error signal of top layer
493  TIMER_START(start);
494  Precursor::criterion_->getErrorSignal_naturalPairing(errorSignal, network().getTopLayer());
495  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, this->timeInitialErrorSignalBatch_, this->timeInitialErrorSignal_);
496  }
497 
498  if (!discard)
499  this->processBatch_finishWithError_naturalPairing(error, errorSignal);
500  else
501  processBatch_finishDiscard();
502 }

537 template<typename T>
538 void FeedForwardTrainer<T>::errorBackpropagation() {
539  timeval start, end;
540  TIMER_START(start);
541 
542  // error backpropagation
543  for (s32 layer = (s32)network().nLayers() - 1; layer > lowestTrainableLayerIndex_; layer--) {
544  if (errorSignalClip_ < Core::Type<T>::max)
545  errorSignal_.at(layer).clip(errorSignalClip_);
546 
547  network().getLayer(layer).backpropagateWeights(
548  errorSignal_.at(layer), errorSignalOut_.at(layer));
549  network().getLayer(layer - 1).backpropagateActivations(
550  errorSignal_.at(layer - 1),
551  errorSignal_.at(layer - 1),
552  network().getLayerOutput(layer - 1));
553  }
554 
555  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, timeBackwardPassBatch_, timeBackwardPass_);
556 }

558 template<typename T>
559 void FeedForwardTrainer<T>::collectGradient() {
560  timeval start, end;
561  TIMER_START(start);
562 
563  // gradient computation
564  for (s32 layer = (s32)network().nLayers() - 1; layer >= lowestTrainableLayerIndex_; layer--) {
565  /* update the gradient, if layer is trainable */
566  if (network().getLayer(layer).isTrainable()) {
567  for (u32 stream = 0; stream < statistics_->gradientWeights(layer).size(); stream++) {
568  NnMatrix& layerInputStream = *(network().getLayerInput(layer)[stream]);
569  NnMatrix& gradientWeights = statistics_->gradientWeights(layer)[stream];
570  NnVector& gradientBias = statistics_->gradientBias(layer);
571 
572  // let every layer update the gradients
573  network().getLayer(layer).addToWeightsGradient(layerInputStream,
574  errorSignal_.at(layer), stream, gradientWeights);
575  network().getLayer(layer).addToBiasGradient(layerInputStream,
576  errorSignal_.at(layer), stream, gradientBias);
577  }
578  }
579  }
580  TIMER_GPU_STOP_SUM2(start, end, this->measureTime_, timeGradientBatch_, timeGradient_);
581 }

652 template<typename T>
653 void FeedForwardAutoTrainer<T>::processBatch_finishWithAlignment(Math::CudaVector<u32>& alignment) {
654  NnMatrix dummyErrorSignal; // might be needed if no gradient is calculated
655  NnMatrix& errorSignal = errorSignal_.empty() ? dummyErrorSignal : errorSignal_[errorSignal_.size() - 1];
656 
657  // calculate number of classification errors and objective function
658  if (statistics_->hasBaseStatistics()) {
659  // apply regularization only when not in batch mode
660  if (!estimator().fullBatchMode()) {
661  u32 batchSize = network().getLayerInput(0)[0]->nColumns();
662  T regularizerError = regularizer().objectiveFunction(network(), T(batchSize));
663  statistics_->addToObjectiveFunction(regularizerError);
664  }
665  }
666 
667  if (statistics_->hasGradient()) {
668  // reset error signals
669  if (!errorSignal_.empty()) {
670  for (u32 layer = 0; layer < errorSignal_.size(); layer++) {
671  errorSignal_.at(layer).setToZero();
672  }
673 
674  // Special case: we have filled the matrix already inplace.
675  // In that case, nothing needs to be done. Otherwise, copy it over.
676  NnMatrix& topErrorSignal = errorSignal_[errorSignal_.size() - 1];
677  if (&errorSignal != &topErrorSignal) {
678  topErrorSignal.copy(errorSignal);
679  }
680 
681  u32 layerId = network().getLayerIdByName(referenceInputLayer_);
682  NnMatrix& netInput = *network().getLayerInput(layerId)[referenceInputLayerPort_];
683  NnMatrix& netOutput = network().getTopLayerOutput();
684 
685  topErrorSignal.add(netOutput);
686  topErrorSignal.add(netInput, (T)-1.0);
687 
688  T err = topErrorSignal.sumOfSquares() / netOutput.size();
689  statistics_->addToObjectiveFunction(err * netOutput.nColumns());
690  }
691 
692  // error backprop
693  Precursor::errorBackpropagation();
694 
695  // collect gradient
696  Precursor::collectGradient();
697 
698  // apply regularization only when not in batch mode
699  if (!estimator().fullBatchMode()) {
700  regularizer().addGradient(network(), statistics(), T(statistics().nObservations()));
701  }
702  }
703 
704  // update (only if has gradient)
705  if (statistics_->hasGradient() && !estimator().fullBatchMode()) {
706  if (Precursor::minibatchCount_ % estimator().accumulateMultipleBatches() == 0) {
707  // maybe normalize statistics by batch size
708  statistics_->finalize(Precursor::normalizeByNOfObservations_);
709  // update model
710  estimator().estimate(network(), statistics());
711  }
712  }
713 
714  if (Precursor::doublePrecisionStatistics_ && estimator().fullBatchMode())
715  Precursor::doublePrecisionStatistics_->add(*statistics_);
716 
717  // logging
718  if (statisticsChannel_.isOpen() && statistics_->hasBaseStatistics() && !(estimator().fullBatchMode())) {
719  if (Precursor::minibatchCount_ % estimator().accumulateMultipleBatches() == 0) {
720  // Note that we called statistics_->finalize() above, which might have normalized these.
721  statisticsChannel_ << Core::XmlOpen("batch-statistics")
722  << Core::XmlFull("mini-batch-number", Precursor::minibatchCount_)
723  << Core::XmlFull("objective-function-on-batch", statistics_->objectiveFunction());
724  if (estimator().accumulateMultipleBatches() > 1)
725  statisticsChannel_ << Core::XmlFull("batch-total-time-frames", statistics_->nObservations());
726  statisticsChannel_ << Core::XmlClose("batch-statistics");
727  }
728  else {
729  statisticsChannel_ << Core::XmlOpen("batch-statistics-accumulated-so-far")
730  << Core::XmlFull("mini-batch-number", Precursor::minibatchCount_)
731  << Core::XmlFull("accumulated-objective-function", statistics_->objectiveFunction());
732  if (estimator().accumulateMultipleBatches() > 1)
733  statisticsChannel_ << Core::XmlFull("accumulated-time-frames", statistics_->nObservations());
734  statisticsChannel_ << Core::XmlClose("batch-statistics-accumulated-so-far");
735  }
736  }
737 }
