80  void buildString(Fsa::State* source, Fsa::State* target, const std::string& orth) {
81  Fsa::Weight one = Fsa::Weight(product_->semiring()->one());
82  const char *cc, *nc;
83  Fsa::State* state = source;
84  for (cc = nc = orth.c_str(); *cc; cc = nc) {
85  do
86  ++nc;
87  while (*nc && utf8::byteType(*nc) == utf8::multiByteTail);
88  std::string letter(cc, nc);
89  Fsa::State* next = (*nc) ? product_->newState() : target;
90  state->newArc(next->id(), one, parser_->lexicon()->letter(letter)->id());
91  state = next;
92  }
93  if (state != target) // when orth has length zero
94  source->newArc(target->id(), one, Fsa::Epsilon);
95  }

97  virtual void newEdge(
98  OrthographicParser::Node from,
99  OrthographicParser::Node to,
100  const Lemma* lemma) {
101  Fsa::State* source = static_cast<Fsa::State*>(from);
102  Fsa::State* target = static_cast<Fsa::State*>(to);
103  if (lemma) {
104 #if 1
105  buildString(source, target, lemma->preferredOrthographicForm());
106 #else
107  const OrthographicFormList& forms(lemma->orthographicForms());
108  for (OrthographicFormList::Iterator orth = forms.begin(); orth != forms.end(); ++orth)
109  buildString(source, target, *orth);
110 #endif
111  }
112  else {
113  source->newArc(target->id(), product_->semiring()->one(), Fsa::Epsilon);
114  }
115  }

134 Evaluator::Evaluator(
135  const Core::Configuration& c,
136  Core::Ref<const Lexicon> l)
137  : Core::Component(c),
138  lexicon_(l),
139  orthParser_(),
140  letterAcceptorBuilder_(0),
141  editDistance_(0),
142  graphStatisticsChannel_(config, "graph-statistics"),
143  graphDumpChannel_(config, "dump-graphs") {
144  shallComputeWordErrors_ = paramWordErrors(config);
145  shallComputeLetterErrors_ = paramLetterErrors(config);
146  shallFilterLettersByEvalTokens_ = paramFilterLettersByEvalTokens(config);
147  shallComputePhonemeErrors_ = paramPhonemeErrors(config);
148 
149  orthParser_ = Core::ref(new Bliss::OrthographicParser(config, lexicon_));
150  lemmaPronToLemma_ = Fsa::multiply(lexicon_->createLemmaPronunciationToLemmaTransducer(), Fsa::Weight(0.0));
151  lemmaToSynt_ = Fsa::multiply(lexicon_->createLemmaToSyntacticTokenTransducer(), Fsa::Weight(0.0));
152  lemmaToEval_ = Fsa::multiply(lexicon_->createLemmaToEvaluationTokenTransducer(), Fsa::Weight(0.0));
153  // lemmaToPreferredEval_ = Fsa::multiply(lexicon_->createLemmaToPreferredEvaluationTokenSequenceTransducer(), Fsa::Weight(0.0));
154 
155  if (shallComputeLetterErrors_ || shallComputePhonemeErrors_) {
156  lemmaToLemmaConfusion_ = Fsa::composeMatching(lemmaToEval_, Fsa::invert(lemmaToEval_));
157  lemmaToLemmaConfusion_ = Fsa::cache(lemmaToLemmaConfusion_);
158  }
159 
160  if (shallComputeLetterErrors_) {
161  letterAcceptorBuilder_ = new LetterAcceptorBuilder;
162  lemmaToLetter_ = lexicon_->createLemmaToOrthographyTransducer(false, shallFilterLettersByEvalTokens_);
163  }
164 
165  if (shallComputePhonemeErrors_) {
166  Fsa::ConstAutomatonRef phonToPron =
167  lexicon_->createPhonemeToLemmaPronunciationTransducer();
168  lemmaPronToPhoneme_ = Fsa::cache(Fsa::invert(phonToPron));
169  lemmaToPhoneme_ = Fsa::invert(
170  Fsa::removeDisambiguationSymbols(
171  Fsa::fuse(
172  Fsa::composeSequencing(
173  phonToPron,
174  lemmaPronToLemma_),
175  createAnyPhonemeToUnknownTransducer(lexicon_))));
176  lemmaToPhoneme_ = Fsa::cache(lemmaToPhoneme_);
177  }
178 
179  editDistance_ = new EditDistance(select("edit-distance"));
180 }

278 /**
279  * Compute and report density of the lattice in terms of arcs per
280  * reference word.
281  *
282  * Warning: The word lattice density as defined here is a bit
283  * "skewed": The denominator counts evaluation tokens, while the
284  * numerator counts lemma pronuncations. (Actually it is not even
285  * guatanteed that @c nWordArcs corresponds to lemma pronuncations,
286  * since the interface of this function is rather fexible.) However,
287  * this may be justified since eval tokens corrspond to "words that
288  * matter" and the number of arcs in the lattice to "search and
289  * storage effort. Therefore the resulting quotient is a figure of
290  * efficiency of the lattice. Another issue is that the number of
291  * eval tokens can be ambiguous (depending on the lexicon). This is
292  * addressed by using the sequence which yields the fewest errors.
293  */
294 void Evaluator::reportLatticeDensity(Fsa::ConstAutomatonRef lattice, const ErrorStatistic& latticeWordErrors) {
295  Core::XmlWriter& os(clog());
296 
297  u32 nCorrectEvalTokens = latticeWordErrors.nLeftTokens();
298  os << Core::XmlFull("correct-words", nCorrectEvalTokens);
299 
300  Fsa::AutomatonCounts countsLattice = Fsa::count(lattice);
301  u32 nArcs = countsLattice.nArcs_ - countsLattice.nIEps_;
302  os << Core::XmlFull("word-arcs", nArcs);
303 
304  f32 wordLatticeDensity = f32(nArcs) / f32(nCorrectEvalTokens);
305  os << Core::XmlFull("word-lattice-density", wordLatticeDensity);
306 }

308 u32 Evaluator::evaluateWords(Fsa::ConstAutomatonRef lemmaOrPronOrSyntOrEval, const std::string& name) {
309  require(!Fsa::isEmpty(lemmaOrPronOrSyntOrEval));
310  require(lemmaOrPronOrSyntOrEval->getOutputAlphabet() == lexicon_->lemmaPronunciationAlphabet() ||
311  lemmaOrPronOrSyntOrEval->getOutputAlphabet() == lexicon_->lemmaAlphabet() ||
312  lemmaOrPronOrSyntOrEval->getOutputAlphabet() == lexicon_->syntacticTokenAlphabet() ||
313  lemmaOrPronOrSyntOrEval->getOutputAlphabet() == lexicon_->evaluationTokenAlphabet());
314 
315  Fsa::ConstAutomatonRef lemma, lemmaPron;
316  if (lemmaOrPronOrSyntOrEval->getOutputAlphabet() == lexicon_->lemmaPronunciationAlphabet()) {
317  lemmaPron = lemmaOrPronOrSyntOrEval;
318  lemma = Fsa::cache(Fsa::composeMatching(lemmaPron, lemmaPronToLemma_));
319  }
320  else if (lemmaOrPronOrSyntOrEval->getOutputAlphabet() == lexicon_->lemmaAlphabet()) {
321  lemma = lemmaOrPronOrSyntOrEval;
322  }
323  else if (lemmaOrPronOrSyntOrEval->getOutputAlphabet() == lexicon_->syntacticTokenAlphabet()) {
324  lemma = Fsa::cache(Fsa::composeMatching(lemmaOrPronOrSyntOrEval, Fsa::invert(lemmaToSynt_)));
325  }
326  else if (lemmaOrPronOrSyntOrEval->getOutputAlphabet() == lexicon_->evaluationTokenAlphabet()) {
327  lemma = Fsa::cache(Fsa::composeMatching(lemmaOrPronOrSyntOrEval, Fsa::invert(lemmaToEval_)));
328  }
329 
330  u32 result = 0;
331 
332  if (shallComputeWordErrors_) {
333  Fsa::ConstAutomatonRef eval = Fsa::projectOutput(Fsa::composeMatching(lemma, lemmaToEval_));
334  eval = Fsa::staticCopy(Fsa::trim(eval));
335  ErrorStatistic errors = evaluateMetric(correct_.eval, eval, "eval", name);
336  reportLatticeDensity(Fsa::projectOutput(lemma), errors);
337  result = errors.nErrors();
338  }
339 
340  if (shallComputeLetterErrors_) {
341  Fsa::ConstAutomatonRef orth = Fsa::projectOutput(Fsa::composeMatching(lemma, lemmaToLetter_));
342  orth = Fsa::staticCopy(Fsa::trim(orth));
343  evaluateMetric(correct_.orth, orth, "orth", name);
344  }
345 
346  if (shallComputePhonemeErrors_ && correct_.phon && lemmaPron) {
347  Fsa::ConstAutomatonRef phon = Fsa::removeDisambiguationSymbols(Fsa::projectOutput(Fsa::composeMatching(lemmaPron, lemmaPronToPhoneme_)));
348  phon = Fsa::staticCopy(Fsa::trim(phon));
349  evaluateMetric(correct_.phon, phon, "phon", name);
350  }
351 
352  return result;
353 }

372 u32 Evaluator::evaluate(Fsa::ConstAutomatonRef aa, const std::string& name) {
373  require(!Fsa::isEmpty(aa));
374  if (aa->getOutputAlphabet() == lexicon_->lemmaPronunciationAlphabet())
375  return evaluateWords(aa, name);
376  if (aa->getOutputAlphabet() == lexicon_->lemmaAlphabet())
377  return evaluateWords(aa, name);
378  if (aa->getOutputAlphabet() == lexicon_->phonemeInventory()->phonemeAlphabet())
379  return evaluatePhonemes(aa, name);
380  if (aa->getOutputAlphabet() == lexicon_->syntacticTokenAlphabet())
381  return evaluateWords(aa, name);
382  if (aa->getOutputAlphabet() == lexicon_->evaluationTokenAlphabet())
383  return evaluateWords(aa, name);
384  require(aa->getOutputAlphabet() != aa->getOutputAlphabet());
385  return 0;
386 }

411  << Core::XmlFull("cost", al.cost)
412  << Core::XmlFull("edit-operations", errors.nErrors())
413  << (Core::XmlFull("score", al.score) + Core::XmlAttribute("source", "best"))
414  << (Core::XmlFull("count", errors.nRightTokens()) + Core::XmlAttribute("event", "token") + Core::XmlAttribute("source", "best"))
415  << (Core::XmlFull("count", errors.nLeftTokens()) + Core::XmlAttribute("event", "token") + Core::XmlAttribute("source", "reference"));
416  os << Core::XmlClose("statistic");
417 
418  // report best sequence
419  os << Core::XmlOpen(type) + Core::XmlAttribute("source", "best sequence");
420  for (EditDistance::Alignment::const_iterator ai = al.begin(); ai != al.end(); ++ai)
421  if (ai->b)
422  os << ai->b->symbol() << Core::XmlBlank();
423  os << Core::XmlClose(type);
424 
425  // report number of errors
426  errors.write(os);
427 
428  os << Core::XmlClose("evaluation");
429 
430  return errors;
