83  * Calculate 1 - p_t(w_n|x_1,...,x_T) for each time frame and word index.

85 class TimeframeErrorDfsState : public DfsState {
86 private:
87  class AccumulatorMap : public std::unordered_map<Fsa::LabelId, Fsa::Accumulator*> {
88  public:
89  ~AccumulatorMap() {
90  for (iterator it = begin(); it != end(); ++it) {
91  delete it->second;
92  }
93  }
94  };
95  typedef std::vector<AccumulatorMap> Accumulators;
96 
97 public:
98  class ScoreMap : public std::unordered_map<Fsa::LabelId, f32> {
99  public:
100  f32 score(Fsa::LabelId id) const {
101  ScoreMap::const_iterator it = find(id);
102  verify(it != end());
103  return it->second;
104  }
105  };
106  typedef Core::Vector<ScoreMap> Scores;
107 
108 private:
109  Accumulators accumulators_;
110  Scores* scores_;
111  const Mapping& mapping_;
112  f32 tolerance_;
113 
114 public:
115  TimeframeErrorDfsState(ConstWordLatticeRef lattice, const Mapping& mapping)
116  : DfsState(lattice),
117  accumulators_(lattice->maximumTime()),
118  scores_(0),
119  mapping_(mapping),
120  tolerance_(-0.001) {}
121  virtual ~TimeframeErrorDfsState() {}
122 
123  void discoverState(Fsa::ConstStateRef sp) {
124  const Speech::TimeframeIndex start = wordBoundaries_->time(sp->id());
125  verify(start != Speech::InvalidTimeframeIndex);
126  for (Fsa::State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
127  const Speech::TimeframeIndex end = wordBoundaries_->time(fsa_->getState(a->target())->id());
128  verify(end != Speech::InvalidTimeframeIndex);
129  const Fsa::LabelId id = mapping_.map(a->input());
130  verify(end <= accumulators_.size());
131  for (Speech::TimeframeIndex time = start; time < end; ++time) {
132  AccumulatorMap& acc = accumulators_[time];
133  if (acc.find(id) == acc.end()) {
134  acc[id] = fsa_->semiring()->getCollector(a->weight());
135  }
136  else {
137  acc[id]->feed(a->weight());
138  }
139  }
140  }
141  }
142 
143  void finish() {
144  verify(scores_);
145  scores_->resize(accumulators_.size());
146  for (Speech::TimeframeIndex time = 0; time < accumulators_.size(); ++time) {
147  AccumulatorMap& acc = accumulators_[time];
148  ScoreMap& sco = (*scores_)[time];
149  sco.clear();
150  for (AccumulatorMap::const_iterator a = acc.begin(); a != acc.end(); ++a) {
151  f32 p = f32(a->second->get());
152  verify(p > tolerance_);
153  sco[a->first] = Fsa::Weight(1 - exp(-p));
154  }
155  }
156  }
157 
158  void getScores(Scores* scores) {
159  require(scores);
160  scores_ = scores;
161  dfs();
162  scores_ = 0;
163  }
164 };

214 class SumTimeframeErrorAutomaton : public TimeframeErrorAutomaton {

219 
220 protected:
221  virtual Fsa::Weight score(Speech::TimeframeIndex start, Speech::TimeframeIndex end, Fsa::LabelId id) const {
222  f32 sum = 0;
223  for (Speech::TimeframeIndex time = start; time < end; ++time) {
224  sum += scores_[time].score(id);
