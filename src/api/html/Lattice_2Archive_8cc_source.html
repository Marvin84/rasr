209 void FsaArchiveWriter::store(
210  const std::string& id, ConstWordLatticeRef wordLattice) {
211  // when the first lattice needs to be saved, we create a Fsa::ArchiveWriter
212  if (!fsaArchiveWriter_ and wordLattice->nParts() > 0) {
213  Fsa::ConstAutomatonRef fsa = wordLattice->part(0);
214  if (fsa->type() == Fsa::TypeAcceptor) {
215  fsaArchiveWriter_ =
216  new Fsa::ArchiveWriter(
217  config,
218  pathname_,
219  fsa->getInputAlphabet());
220  }
221  else {
222  fsaArchiveWriter_ =
223  new Fsa::ArchiveWriter(
224  config,
225  pathname_,
226  fsa->getInputAlphabet(),
227  fsa->getOutputAlphabet());
228  }
229  }
230 
231  for (size_t i = 0; i < wordLattice->nParts(); ++i) {
232  Fsa::ConstAutomatonRef part = wordLattice->part(i);
233  if (part)
234  fsaArchiveWriter_->store(
235  filename(id, wordLattice->name(i)),
236  part);
237  }
238 
239  if (wordLattice->wordBoundaries()) {
240  Core::ArchiveWriter o(*fsaArchiveWriter_->archive(), id + wordBoundarySuffix, true);
241  if (!o) {
242  error("Failed to write word boundaries \"%s\"", id.c_str());
243  }
244  Core::BinaryOutputStream bo(o);
245  if (!writeWordBoundariesBinary(wordLattice->wordBoundaries(), bo)) {
246  error("Failed to write word boundaries \"%s\"", id.c_str());
247  }
248  }
249 }

290 FsaArchiveReader::FsaArchiveReader(
291  const Core::Configuration& config,
292  const std::string& pathname,
293  Bliss::LexiconRef lexicon)
294  : Core::Component(config),
295  Precursor(config, pathname, lexicon),

297  fsaArchiveReader_ = new Fsa::ArchiveReader(config, pathname);
298  if (!fsaArchiveReader_->archive()) {
299  error("Failed to open FSA archive '%s'.", pathname.c_str());
300  return;
301  }
302  switch (paramAlphabet(config)) {
303  case lemmaPronunciationAlphabetAcceptor:
304  fsaArchiveReader_->mapOutput(lexicon_->lemmaPronunciationAlphabet());
305  case lemmaPronunciationAlphabet:
306  fsaArchiveReader_->mapInput(lexicon_->lemmaPronunciationAlphabet());
307  break;
308  case lemmaAlphabet:
309  fsaArchiveReader_->mapInput(lexicon_->lemmaAlphabet());
310  break;
311  case syntacticTokenAlphabet:
312  fsaArchiveReader_->mapInput(lexicon_->syntacticTokenAlphabet());
313  break;
314  case evaluationTokenAlphabet:
315  fsaArchiveReader_->mapInput(lexicon_->evaluationTokenAlphabet());
316  break;
317  case noLexiconCheck:
318  // do nothing
319  break;
320  default:
321  criticalError("Unknown alphabet.");
322  }
323 
324  loadLatticeConfiguration();
325  if (amScale() != 1.0)
326  error("cannot set am-scale to %f; am-scale for fsa lattices is always 1.0",
327  amScale());
328  if (penaltyScale() != amScale())
329  error("cannot set penalty-scale to %f; penalty-scale for fsa lattices is always equal to am-scale, which is %f",
330  penaltyScale(), amScale());
331  if (lmScale() != 0.0)
332  error("cannot set lm-scale to %f; lm-scale for fsa lattices is always 0.0",

363 ConstWordLatticeRef FsaArchiveReader::get(
364  const std::string& id, const std::vector<std::string>& names) {
365  Core::Ref<WordLattice> wordLattice;
366  Core::Ref<WordBoundaries> wordBoundaries;
367  std::string wordBoundariesName(id + wordBoundarySuffix);
368  if (fsaArchiveReader_->archive()->hasFile(wordBoundariesName)) {
369  Core::ArchiveReader i(*fsaArchiveReader_->archive(), wordBoundariesName);
370  if (i) {
371  Core::BinaryInputStream bi(i);
372  wordBoundaries = Core::ref(new WordBoundaries);
373  if (!readWordBoundariesBinary(wordBoundaries, bi)) {
374  error("Failed to read word boundaries \"%s\"", id.c_str());
375  wordBoundaries.reset();
376  }
377  }
378  else {
379  error("Failed to read word boundaries \"%s\"", id.c_str());
380  }
381  }
382  else {
383  error("Could not find word boundaries \"%s\"", id.c_str());
384  }
385 
386  wordLattice = Core::ref(new WordLattice);
387  wordLattice->setWordBoundaries(wordBoundaries);
388  for (size_t i = 0; i < names.size(); ++i) {
389  wordLattice->setFsa(fsaArchiveReader_->get(filename(id, names[i])), names[i]);
390  if (!wordLattice->part(names[i])) {
391  error("Failed to read part '%s'.", names[i].c_str());
392  wordLattice.reset();
393  return wordLattice;
394  }
395  }
396  return wordLattice;
397 }

511 Fsa::ConstAutomatonRef createEpsilonToSentenceBoundariesTransducer(
512  Bliss::LexiconRef lexicon,
513  Fsa::ConstSemiringRef semiring) {
514  Fsa::StaticAutomaton* epsilonToSentenceBoundaries = new Fsa::StaticAutomaton(Fsa::TypeTransducer);
515  epsilonToSentenceBoundaries->setSemiring(semiring);
516  epsilonToSentenceBoundaries->setInputAlphabet(lexicon->lemmaPronunciationAlphabet());
517  epsilonToSentenceBoundaries->setOutputAlphabet(lexicon->lemmaAlphabet());
518  Fsa::State* initialState = new Fsa::State(0);
519  epsilonToSentenceBoundaries->setState(initialState);
520  epsilonToSentenceBoundaries->setInitialStateId(0);
521  Fsa::State* finalState = new Fsa::State(1);
522  finalState->setFinal(Fsa::Weight(0.0));
523  epsilonToSentenceBoundaries->setState(finalState);
524  const Bliss::Lemma* lemma;
525  if ((lemma = lexicon->specialLemma("sentence-begin")))
526  initialState->newArc(1, Fsa::Weight(0.0), Fsa::Epsilon, lexicon->lemmaAlphabet()->index(lemma));
527  if ((lemma = lexicon->specialLemma("sentence-end")))
528  initialState->newArc(1, Fsa::Weight(0.0), Fsa::Epsilon, lexicon->lemmaAlphabet()->index(lemma));
529  if ((lemma = lexicon->specialLemma("sentence-boundary")))
530  initialState->newArc(1, Fsa::Weight(0.0), Fsa::Epsilon, lexicon->lemmaAlphabet()->index(lemma));
531  if (!initialState->hasArcs()) {
532  epsilonToSentenceBoundaries->clear();
533  initialState = new Fsa::State(0);
534  initialState->setFinal(Fsa::Weight(0.0));
535  epsilonToSentenceBoundaries->setState(initialState);
536  epsilonToSentenceBoundaries->setInitialStateId(0);
537  }
538  return Fsa::ConstAutomatonRef(epsilonToSentenceBoundaries);
539 }
