120 template<typename T>
121 void SegmentwiseNnTrainer<T>::processWordLattice(Lattice::ConstWordLatticeRef lattice, Bliss::SpeechSegment* segment) {
122  SpeechPrecursor::processWordLattice(lattice, segment);
123  timeval start_all, end_all;
124  timeval start, end;
125  TIMER_START(start_all)
126 
127  // initialization
128  initSegment(lattice, segment);
129 #ifdef MODULE_PYTHON
130  pythonControl_.run_custom("init_segment", "{s:s}", "segment_name", segment->fullName().c_str());
131 #endif
132 
133  // create numerator lattice (required for alignment vector).
134  // this is a lattice containing only one word seq, the reference orthography.
135  TIMER_START(start)
136  Lattice::ConstWordLatticeRef numeratorLattice = Speech::AbstractSegmentwiseTrainer::extractNumerator(segment->orth(), lattice);
137  TIMER_GPU_STOP(start, end, measureTime_, timeNumeratorExtraction_)
138 
139  // extract alignment vector from numerator lattice.
140  // we need it for basic prior/FER statistics and for the CE-smoothing.
141  // if it fails, skip segment.
142  bool alignmentOK = getAlignmentVector(numeratorLattice);
143 
144  if (!alignmentOK) {
145  this->warning("Computing alignment vector failed."
146  "\nThis is probably caused by inf-scores in the lattice - check whether the learning rate is set too high!\n"
147  "Skipping segment.");
148  errorSignal_.back().finishComputation(false);
149  network().getTopLayerOutput().initComputation(false);
150  LatticeSetProcessor::processWordLattice(lattice, segment);
151  return;
152  }
153 
154  // compute initial error signal and sync it to GPU
155  // the error signal is defined by the criterion, e.g. MMI or MPE
156  // assume network is already forwarded
157  TIMER_START(start)
158  bool errorSignalOk = true;
159  if (ceSmoothingWeight_ < 1)
160  errorSignalOk = computeInitialErrorSignal(lattice, numeratorLattice, segment, localObjectiveFunction_, !statistics_->hasGradient());
161  else
162  errorSignal_.back().setToZero();
163  TIMER_GPU_STOP(start, end, measureTime_, timeErrorSignal_)
164 
165  if (!errorSignalOk) {
166  this->warning("could not compute error signal (bad lattice?), skipping segment");
167  errorSignal_.back().finishComputation(false);
168  network().getTopLayerOutput().initComputation(false);
169  LatticeSetProcessor::processWordLattice(lattice, segment);
170  return;
171  }
172 
173  if (maxoutLayer_) { // if mixture expand the errorsignal
174  TIMER_START(start)
175  errorSignal_.back().initComputation(); // move data to GPU
176  TIMER_GPU_STOP(start, end, measureTime_, timeSync_)
177  errorSignal_[maxoutLayer_->getPredecessor(0)].maxoutErrorExpand(maxoutLayer_->getMixture(), maxoutLayer_->getOffset(), maxoutLayer_->getMaxindex(), errorSignal_.back()); // expand
178  }
179 
180  // sync to GPU
181  TIMER_START(start)
182  alignment_.initComputation(); // move to GPU
183  if (NeuralNetworkTrainer<f32>::weightedAccumulation_)
184  weights_.initComputation(); // move to GPU
185  network().getTopLayerOutput().initComputation(); // EmissionLatticeRescorerAutomaton::forwardNetwork() sets isComputing_ to false
186  prior_.initComputation();
187  TIMER_GPU_STOP(start, end, measureTime_, timeSync_)
188 
189  // apply Cross-entropy smoothing
190  // side effects: log-priors are added to scores, softmax is applied
191  localCeObjectiveFunction_ = smoothErrorSignalWithCE();
192 
193 #ifdef MODULE_PYTHON
194  pythonControl_.run_custom(
195  "notify_segment_loss", "{s:s,s:f}",
196  "segment_name", segment->fullName().c_str(),
197  "loss", (float)localObjectiveFunction_);
198 #endif
199 
200  // compute gradient
201  if (statistics_->hasGradient()) {
202  // sync error signal to GPU
203  TIMER_START(start)
204  errorSignal_.back().initComputation(); // move to GPU if not there
205  TIMER_GPU_STOP(start, end, measureTime_, timeSync_)
206 
207  // weight error signal
208  if (NeuralNetworkTrainer<f32>::weightedAccumulation_) {
209  TIMER_START(start)
210  if (!maxoutLayer_) {
211  errorSignal_.back().multiplyColumnsByScalars(weights_); // weight error signal
212  }
213  else {
214  errorSignal_[maxoutLayer_->getPredecessor(0)].multiplyColumnsByScalars(weights_); // weight error signal
215  }
216  TIMER_GPU_STOP(start, end, measureTime_, timeErrorSignal_)
217  }
218 
219  // backpropate error and collect gradient
220  backpropagateError();
221  collectGradient();
222  }
223 
224  // update base statistics
225  accumulateBaseStatistics();
226  accumulatePrior();
227 
228  if (!estimator().fullBatchMode())
229  statistics_->finalize(false);
230 
231  // apply regularizer
232  // TODO this doesn't work with double precision yet
233  statistics_->addToObjectiveFunction(regularizer().objectiveFunction(network(), 1));
234  regularizer().addGradient(network(), *statistics_, 1.0);
235 
236  // update model (different code for single and double precision)
237  if (!estimator().fullBatchMode())
238  updateModel();
239 
240  // relase lock on CPU memory for error signal of top layer
241  errorSignal_.back().finishComputation(false);
242 
243  // we are done ..
244  TIMER_STOP(start_all, end_all, this->timeProcessSegment_)
245 
246  logSegmentStatistics();
247 
248  segmentNeedsInit_ = true;
249  LatticeSetProcessor::processWordLattice(lattice, segment);
250 }

284 template<typename T>
285 void SegmentwiseNnTrainer<T>::initializeTrainer() {
286  if (needInit_) {
287  // do some checks
288 
289  // network must have at least one layer
290  require_gt(network().nLayers(), 0);
291  // training only makes sense, if there is at least one trainable layer
292  require_gt(network().nTrainableLayers(), 0);
293 
294  if (ceSmoothingWeight_ > 0.0) {
295  // require a linear+softmax layer (softmax is not evaluated) optionally followed by maxoutvar
296  topLayer_ = dynamic_cast<LinearAndSoftmaxLayer<f32>*>(&network().getTopLayer());
297  if (!topLayer_) { // if not LinearAndSoftmaxLayer maybe mixture
298  maxoutLayer_ = dynamic_cast<MaxoutVarLayer<f32>*>(&network().getTopLayer());
299  require(maxoutLayer_);
300  topLayer_ = dynamic_cast<LinearAndSoftmaxLayer<f32>*>(&network().getLayer(maxoutLayer_->getPredecessor(0)));
301  }
302  require(topLayer_);
303  require(!topLayer_->evaluatesSoftmax());
304  }
305 
306  // initialize statistics
307  u32 statisticsType = estimator().requiredStatistics() | Statistics<T>::BASE_STATISTICS;
308  statistics_ = new Statistics<T>(network().nLayers(), statisticsType);
309  statistics_->initialize(network());
310  statistics_->initComputation(false);
311  statistics_->reset();
312  if (accumulatePrior_) {
313  u32 priorStatisticsType = Statistics<T>::CLASS_COUNTS;
314  priorStatistics_ = new Statistics<f32>(network().nLayers(), priorStatisticsType);
315  priorStatistics_->initialize(network());
316  priorStatistics_->initComputation(false);
317  priorStatistics_->reset();
318  }
319 
320  // initialize error signal (sequence length unknown yet, therefore size of matrices not set)
321  errorSignal_.resize(network().nLayers(), NnMatrixf32());
322  for (u32 layer = 0; layer < errorSignal_.size() - 1; layer++)
323  errorSignal_.at(layer).initComputation(false);
324 
325  // create ErrorSignalAccumulator
326  accumulator_ = new ErrorSignalAccumulator<f32>(&(errorSignal_.back()), &labelWrapper());
327 
328  // initialize class weights
329  // need to set weightedAccumulation when using frame rejection heuristic
330  NeuralNetworkTrainer<f32>::weightedAccumulation_ |= frameRejectionThreshold_ > 0;
331  setClassWeights();
332 
333  // get prior if cross-entropy smoothing is used
334  if (ceSmoothingWeight_ > 0) {
335  Prior<f32> prior(config);
336  if (prior.fileName() != "") {
337  prior.read();
338  prior_.resize(prior.size());
339  prior.getVector(prior_);
340  prior_.initComputation();
341  priorScale_ = prior.scale();
342  if (priorScale_ > 0)
343  this->log("prior is subtracted from scores in cross-entropy smoothing");
344  }
345  }
346 
347  needInit_ = false;
348  }
349 }

353  // logging
354  logTrainingStatistics();
355 
356  // stochastic mode : write network parameters
357  if (!estimator().fullBatchMode()) {

360  }
361

363  if (statisticsFilename_ != "" && statistics_) {

390 template<typename T>
391 void SegmentwiseNnTrainer<T>::setClassWeights() {
392  classWeights_.resize(0);
393  classWeights_.resize(labelWrapper().nClassesToAccumulate(), 1.0);
394 
395  std::string classWeightsFilename(paramClassWeightsFile(config));
396  f32 silenceWeight = paramSilenceWeight(config);
397 
398  if (classWeightsFilename != "" && silenceWeight != -1.0)
399  this->error("Can not use both silence weight and class weights file");
400  else if (classWeightsFilename != "") {
401  this->log("reading class weights file ") << classWeightsFilename;
402  Math::Module::instance().formats().read(classWeightsFilename, classWeights_);
403  if (classWeights_.size() != labelWrapper().nClassesToAccumulate())
404  this->error("dimension mismatch: class weights vs number of classes to accumulate")
405  << classWeights_.size() << " != " << labelWrapper().nClassesToAccumulate();
406  NeuralNetworkTrainer<f32>::weightedAccumulation_ = true;
407  }
408  else if (silenceWeight != -1.0) {
409  Am::Allophone silenceAllophone(this->acousticModel()->silence(), Am::Allophone::isInitialPhone | Am::Allophone::isFinalPhone);
410  u32 silence = this->acousticModel()->emissionIndex(this->acousticModel()->allophoneStateAlphabet()->index(&silenceAllophone, 0));
411  this->log("silence index is ") << silence;
412  if (labelWrapper().isClassToAccumulate(silence)) {
413  classWeights_.at(labelWrapper().getOutputIndexFromClassIndex(silence)) = silenceWeight;
414  this->log("using silence weight ") << silenceWeight;
415  }
416  else
417  this->warning("silence weight has no effect, because silence is not accumulated");
418  NeuralNetworkTrainer<f32>::weightedAccumulation_ = true;
419  }
420 }

432 template<typename T>
433 void SegmentwiseNnTrainer<T>::backpropagateError() {
434  timeval start, end;
435  TIMER_START(start)
436 
437  u32 lastLayer;
438  if (maxoutLayer_) {
439  require_eq(network().nLayers() - 2, maxoutLayer_->getPredecessor(0)); // ugly, TODO
440  lastLayer = network().nLayers() - 2;
441  }
442  else {
443  lastLayer = network().nLayers() - 1;
444  }
445 
446  // reset all error signals except of the last one
447  for (u32 layer = 0; layer < lastLayer; layer++)
448  errorSignal_.at(layer).setToZero();
449 
450  // error backpropagation
451  for (s32 layer = (s32)lastLayer; layer > (s32)network().lowestTrainableLayerIndex(); layer--) {
452  std::vector<NnMatrixf32*> errorSignalOut;
453  for (u32 i = 0; i < network().getLayer(layer).nPredecessors(); i++)
454  errorSignalOut.push_back(&(errorSignal_[network().getLayer(layer).getPredecessor(i)]));
455  network().getLayer(layer).backpropagateWeights(errorSignal_.at(layer), errorSignalOut);
456  network().getLayer(layer - 1).backpropagateActivations(
457  errorSignal_.at(layer - 1),
458  errorSignal_.at(layer - 1),
459  network().getLayerOutput(layer - 1));
460  }
461 
462  TIMER_GPU_STOP(start, end, measureTime_, timeBackpropagation_)
463 }

465 template<typename T>
466 void SegmentwiseNnTrainer<T>::collectGradient() {
467  // just like FeedForwardTrainer<T>::collectGradient()
468  timeval start, end;
469  TIMER_START(start)
470 
471  // compute gradient from error signal and activations
472  for (s32 layer = (s32)network().nLayers() - 1; layer >= (s32)network().lowestTrainableLayerIndex(); layer--) {
473  /* update the gradient, if layer has weights */
474  if (network().getLayer(layer).isTrainable()) {
475  for (u32 stream = 0; stream < statistics_->gradientWeights(layer).size(); stream++) {
476  NnMatrixf32& layerInputStream = *(network().getLayerInput(layer)[stream]);
477  NnMatrixf32& gradientWeights = statistics_->gradientWeights(layer)[stream];
478  NnVectorf32& gradientBias = statistics_->gradientBias(layer);
479 
480  // let every layer update the gradients
481  network().getLayer(layer).addToWeightsGradient(layerInputStream, errorSignal_.at(layer), stream, gradientWeights);
482  network().getLayer(layer).addToBiasGradient(layerInputStream, errorSignal_.at(layer), stream, gradientBias);
483  }
484  }
485  }
486  TIMER_GPU_STOP(start, end, measureTime_, timeGradient_)
487 }

489 template<typename T>
490 void SegmentwiseNnTrainer<T>::initSegment(Lattice::ConstWordLatticeRef lattice, Bliss::SpeechSegment* segment) {
491  if (!lattice) {
492  this->log("no lattice found, skipping segment");
493  return;
494  }
495  if (needInit_)
496  initializeTrainer();
497  if (!estimator().fullBatchMode()) {
498  statistics_->reset();
499  ceObjectiveFunction_ = 0;
500  }
501  // objective function of segment
502  localObjectiveFunction_ = 0;
503  localCeObjectiveFunction_ = 0;
504  localClassificationErrors_ = 0;
505 
506  // resizing
507  // activations: assume that resizing of has already happened in rescoring step
508  sequenceLength_ = this->features()->size();
509  require_eq(network().activationsSize(), sequenceLength_);
510  timeval start, end;
511  TIMER_START(start)
512  if (statistics_->hasGradient())
513  resizeErrorSignal();
514  TIMER_GPU_STOP(start, end, measureTime_, timeMemoryAllocation_)
515  segmentNeedsInit_ = false;
516 }

596 template<typename T>
597 void SegmentwiseNnTrainer<T>::accumulateBaseStatistics() {
598  verify(!segmentNeedsInit_);
599  timeval start, end;
600  TIMER_START(start)
601  if (statistics_->hasBaseStatistics()) {
602  localClassificationErrors_ = network().getTopLayerOutput().nClassificationErrors(alignment_);
603  statistics_->incClassificationErrors(localClassificationErrors_);
604  statistics_->incObservations(sequenceLength_);
605  if (this->weightedAccumulation_)
606  statistics_->addToTotalWeight(weights_.asum());
607  else
608  statistics_->addToTotalWeight(sequenceLength_);

628 template<typename T>
629 T SegmentwiseNnTrainer<T>::smoothErrorSignalWithCE() {
630  require(prior_.isComputing());
631  verify(!segmentNeedsInit_);
632  verify_ge(ceSmoothingWeight_, 0.0);
633  verify_le(ceSmoothingWeight_, 1.0);
634  timeval start, end;
635  TIMER_START(start)
636 
637  T ceObjectiveFunction = 0;
638  if (ceSmoothingWeight_ > 0.0) {
639  if (!maxoutLayer_) { // simple linear+softmax
640  require(topLayer_);
641  require(network().getTopLayerOutput().isComputing());
642 
643  if (priorScale_ > 0)
644  network().getTopLayerOutput().addToAllColumns(prior_, priorScale_); // add prior, previously assumed: priors was merged in bias, output was ~likelihood
645  topLayer_->applySoftmax(network().getTopLayerOutput()); // and apply softmax
646 
647  if (statistics_->hasGradient()) {
648  errorSignal_.back().initComputation(); // move data to GPU
649  errorSignal_.back().scale(1.0 - ceSmoothingWeight_);
650  // softmax - kronecker delta (minimization problem)
651  errorSignal_.back().add(network().getTopLayerOutput(), ceSmoothingWeight_); // add the CE error signal
652  errorSignal_.back().addKroneckerDelta(alignment_, -ceSmoothingWeight_);
653  }
654  }
655  else { // linear+softmax followed by maxoutvar = mixture layer
656  require(topLayer_);
657  require(maxoutLayer_);
658  require(network().getTopLayerOutput().isComputing());
659  require(network().getLayerOutput(maxoutLayer_->getPredecessor(0)).isComputing());
660  require(maxoutLayer_->getOffset().isComputing());
661  require(maxoutLayer_->getMixture().isComputing());
662  require(maxoutLayer_->getMaxindex().isComputing());
663 
664  if (priorScale_ > 0)
665  network().getLayerOutput(maxoutLayer_->getPredecessor(0)).expandAddToAllColumns(maxoutLayer_->getMixture(), maxoutLayer_->getOffset(), prior_,
666  priorScale_); // add state prior to all hidden variable per state (maximum approx)
667  topLayer_->applySoftmax(network().getLayerOutput(maxoutLayer_->getPredecessor(0))); // and apply softmax
668  network().getTopLayerOutput().maxoutvar(maxoutLayer_->getMixture(),
669  maxoutLayer_->getOffset(),
670  network().getLayerOutput(maxoutLayer_->getPredecessor(0)), maxoutLayer_->getMaxindex()); // and redo the maxoutvar (but index should remain the same)
671 
672  if (statistics_->hasGradient()) {
673  errorSignal_[maxoutLayer_->getPredecessor(0)].scale(1.0 - ceSmoothingWeight_); // rescale the expanded seq.err.signal
674  // softmax with maxout - kronecker delta (minimization problem)
675  errorSignal_[maxoutLayer_->getPredecessor(0)].add(network().getLayerOutput(maxoutLayer_->getPredecessor(0)),
676  ceSmoothingWeight_); // add the CE error signal
677  errorSignal_[maxoutLayer_->getPredecessor(0)].addKroneckerDelta(alignment_,
678  maxoutLayer_->getOffset(),
679  maxoutLayer_->getMaxindex(),
680  -ceSmoothingWeight_); // hard target is the (maximal) hidden variable of the target state, maximum approximation
681  }
682  }
683 
684  if (this->weightedAccumulation_)
685  ceObjectiveFunction = network().getTopLayerOutput().weightedCrossEntropyObjectiveFunction(alignment_, weights_);
686  else
687  ceObjectiveFunction = network().getTopLayerOutput().crossEntropyObjectiveFunction(alignment_);
688  }
689  TIMER_GPU_STOP(start, end, measureTime_, timeCESmoothing_)
690  return ceObjectiveFunction;
691 }
