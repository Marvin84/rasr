56 bool StringExpressionNode::update(const Timestamp& timestamp) {
57  Core::StringHashMap<InputPort>::iterator i;
58  for (i = inputPorts_.begin(); i != inputPorts_.end(); ++i) {
59  while (!i->second.timestamp_.contains(timestamp)) {
60  DataPtr<String> in;

62  update(i, *in);
