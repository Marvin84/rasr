35  virtual void startElement(const char* name, const Core::XmlAttributes atts) {
36  if (std::string(name) == n_->type_->name()) {

39 
40  const char* start = atts["start"];

44  const char* end = atts["end"];

47 
48  buffer_.clear();

52  virtual void endElement(const char* name) {
53  if (std::string(name) == n_->type_->name()) {
54  verify(n_->out_);
55 
56  n_->setElements(buffer_);
57 
58  if (n_->out_->startTime() == 0.0)
59  n_->out_->setStartTime(n_->timeInS_);
60 
61  if (n_->out_->endTime() == 0.0)
62  n_->out_->setEndTime(n_->timeInS_ + n_->lengthInS_);
63 
64  n_->timeInS_ += n_->shiftInS_;
65 
66  n_->issue();
67  }
