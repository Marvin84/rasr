42 MinimumBayesRiskSearchNode::MinimumBayesRiskSearchNode(const Core::Configuration& config)
43  : Component(config),
44  LatticeSetProcessor(config),
45  search_(0),
46  nBestListExtractor_(0) {
47  switch (paramSearchMethod(config)) {
48  case searchDryRun:
49  search_ = 0;
50  break;
51  case searchMap:
52  search_ = new Search::MinimumBayesRiskMapSearch(select("search"));
53  break;
54  case searchNBestListNaive:
55  search_ = new Search::MinimumBayesRiskNBestListNaiveSearch(select("search"));
56  nBestListExtractor_ = new NBestListExtractor(select("n-best-list-extraction"));
57  break;
58  case searchAStar:
59  search_ = new Search::MinimumBayesRiskAStarSearch(select("search"));
60  break;
61  default:
62  criticalError("search method does not exist!");
63  }
64 }

91 void MinimumBayesRiskSearchNode::processWordLattice(Lattice::ConstWordLatticeRef lattice, Bliss::SpeechSegment* s) {
92  // case dry run
93  if (!search_) {
94  return;
95  }
96 
97  if (lattice->nParts() != 1) {
98  error("Lattice must consist of a single part.");
99  }
100 
101  Fsa::ConstAutomatonRef fsa = lattice->mainPart();
102  if (nBestListExtractor_) {
103  fsa = nBestListExtractor_->getNBestList(lattice)->mainPart();
104  }
105 
106  /**
107  * Assumption: fsa is an n-best list or lattice with evaluation tokens
108  * as input labels.
109  */
110  if (fsa->getInputAlphabet() != lexicon_->evaluationTokenAlphabet()) {
111  error("Lattice must have the evaluation alphabet as input alphabet.");
112  }
113  search_->performSearch(fsa);
114 
115  Core::Ref<Lattice::WordLattice> result(new Lattice::WordLattice);
116  result->setFsa(search_->getBestAutomaton(), Lattice::WordLattice::totalFsa);
117  Precursor::processWordLattice(result, s);
118 }
