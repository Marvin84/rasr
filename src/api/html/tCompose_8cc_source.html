101  void stringPotential(_ConstAutomatonRef f, Fsa::StateId s, Core::Vector<Fsa::LabelIdStrings::Id>& potentials, bool input) const {
102  // cache potentials
103  if (s >= potentials.size())
104  potentials.grow(s, Fsa::LabelIdStrings::Invalid);
105  if (potentials[s] != Fsa::LabelIdStrings::Invalid)
106  return;
107 
108  // start with s being the first state to explore
109  Fsa::Stack<Fsa::StateId> states_;
110  states_.push(s);
111  typedef Core::Vector<std::pair<Fsa::StateId, Fsa::LabelIdStrings::Id>> StatePotentials;
112  StatePotentials potentials_;
113  Fsa::LabelIdStrings::Id start = stringPotentials_.start();
114 
115  // we may encounter a trap (no final state) => don't loop infinitely
116  while (!states_.empty()) {
117  Fsa::Stack<Fsa::StateId> newStates_;
118  // as long as we have only a single state to inspect we'll also store its potential
119  if (states_.size() == 1)
120  potentials_.push_back(std::make_pair(states_.top(), stringPotentials_.start() - start));
121  Fsa::LabelId label = Fsa::InvalidLabelId;
122 
123  // check wether all arcs leaving states have the same label (stack of states is empty if finished)
124  while (!states_.empty()) {
125  _ConstStateRef sp = f->getState(states_.pop());
126  if (sp->isFinal()) {
127  FinalOrNoMatchingLabels:
128  start = stringPotentials_.stop(start);
129  Fsa::StateId maxStateId = 0;
130  for (StatePotentials::const_iterator p = potentials_.begin(); p != potentials_.end(); ++p)
131  maxStateId = std::max(maxStateId, p->first);
132  if (maxStateId >= potentials.size())
133  potentials.grow(maxStateId, Fsa::LabelIdStrings::Invalid);
134  for (StatePotentials::const_iterator p = potentials_.begin(); p != potentials_.end(); ++p) {
135  potentials[p->first] = stringPotentials_.insert(start + p->second);
136  }
137  return;
138  }
139  for (typename _State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
140  Fsa::LabelId newLabel = input ? a->input() : a->output();
141  if (newLabel != Fsa::Epsilon) {
142  newStates_.push(a->target()); // does non-sense in goto-case
143  if (label == newLabel)
144  newStates_.push(a->target());
145  else if (label == Fsa::InvalidLabelId)
146  label = newLabel;
147  else
148  goto FinalOrNoMatchingLabels;
149  }
150  else
151  states_.push(a->target());
152  }
153  }
154  stringPotentials_.append(label);
155  std::swap(states_, newStates_);
156  }
157 
158  // in case of a trap we set potentials to empty strings
159  stringPotentials_.discard(start);
160  Fsa::StateId maxStateId = 0;
161  for (StatePotentials::const_iterator p = potentials_.begin(); p != potentials_.end(); ++p)
162  maxStateId = std::max(maxStateId, p->first);
163  if (maxStateId >= potentials.size())
164  potentials.grow(maxStateId, Fsa::LabelIdStrings::Invalid);
165  for (StatePotentials::const_iterator p = potentials_.begin(); p != potentials_.end(); ++p) {
166  potentials[p->first] = emptyStringPotential_;
167  }
168  }
169  Fsa::LabelIdStrings::Id stringPotentialLeft(Fsa::StateId s) const {
170  stringPotential(fl_, s, stringPotentialsLeft_, false);
171  return stringPotentialsLeft_[s];
172  }
173  Fsa::LabelIdStrings::Id stringPotentialRight(Fsa::StateId s) const {
174  stringPotential(fr_, s, stringPotentialsRight_, true);

177  bool areStringPotentialsPrefixes(Fsa::StateId sl, Fsa::StateId sr) const {

179  Fsa::LabelIdStrings::Id ll = stringPotentialLeft(sl), lr = stringPotentialRight(sr);

183  for (; *il == *ir; ++il, ++ir)
184  return ((*il == Fsa::InvalidLabelId) || (*ir == Fsa::InvalidLabelId));
185  }

247  void composeArcs(_State* sp, _ConstStateRef sl, _ConstStateRef sr,
248  typename _State::const_iterator al, typename _State::const_iterator ar) const {
249  // the following factor of 2 comparison is heuristic, but tested on a larger set of automata
250  if (((sl->nArcs() << 2) < sr->nArcs()) || (sl->nArcs() > (sr->nArcs() << 2))) {
251  for (; (al < sl->end()) && (ar < sr->end());) {
252  if ((al->output() > Fsa::LastLabelId) || (ar->input() > Fsa::LastLabelId))
253  break;
254  if (al->output() == ar->input()) {
255  for (typename _State::const_iterator a = ar; ((a < sr->end()) && (al->output() == a->input())); ++a)
256 #ifdef STRING_POTENTIALS
257  if (areStringPotentialsPrefixes(al->target(), a->target()))
258 #endif
259  sp->newArc(insertState(al->target(), 0, a->target()),
260  semiring()->extend(al->weight(), a->weight()), al->input(), a->output());
261  ++al;
262  }
263  else if (al->output() > ar->input()) {
264  u32 left = 0, right = sr->end() - ar;
265  while (right - left > 1) {
266  u32 i = (left + right) >> 1;
267  if (al->output() > (ar + i)->input())
268  left = i;
269  else
270  right = i;
271  }
272  ar += right;
273  }
274  else {
275  u32 left = 0, right = sl->end() - al;
276  while (right - left > 1) {
277  u32 i = (left + right) >> 1;
278  if ((al + i)->output() < ar->input())
279  left = i;
280  else
281  right = i;
282  }
283  al += right;
284  }
285  }
286  }
287  else {
288  for (; (al < sl->end()) && (ar < sr->end()); ++al) {
289  if ((al->output() > Fsa::LastLabelId) || (ar->input() > Fsa::LastLabelId))
290  break;
291  if (al->output() >= ar->input()) {
292  for (; ((ar < sr->end()) && (al->output() > ar->input())); ++ar)
293  ;
294  for (typename _State::const_iterator a = ar; ((a < sr->end()) && (al->output() == a->input())); ++a)
295 #ifdef STRING_POTENTIALS
296  if (areStringPotentialsPrefixes(al->target(), a->target()))
297 #endif
298  sp->newArc(insertState(al->target(), 0, a->target()),
299  semiring()->extend(al->weight(), a->weight()), al->input(), a->output());
300  }
301  }
302  }
303  Fsa::StateTag tags = sl->tags() & sr->tags();
304  if (tags & Fsa::StateTagFinal)
305  sp->weight_ = semiring()->extend(sl->weight_, sr->weight_);
306  sp->setTags(tags);
307  }

309  void composeSpecialArcs(_State* sp, _ConstStateRef sl, _ConstStateRef sr) const {
310  Fsa::LabelId label;
311  typename _State::const_iterator al, ar;
312 
313  for (typename _State::const_reverse_iterator sar = sr->rbegin();
314  (sar != sr->rend()) && (sar->input() > Fsa::LastLabelId); ++sar) {
315  label = sar->output();
316  switch (sar->input()) {
317  case Fsa::Any:
318  for (al = sl->begin(); al < sl->end(); ++al) {
319  if (fr_->type() == Fsa::TypeAcceptor)
320  label = al->output();
321  sp->newArc(insertState(al->target(), 0, sar->target()),
322  semiring()->extend(al->weight(), sar->weight()), al->input(), label);
323  }
324  break;
325  case Fsa::Failure:
326  for (al = sl->begin(), ar = sr->begin(); (al < sl->end()) && (ar < sr->end()); ++ar) {
327  if (al->output() > Fsa::LastLabelId)
328  break;
329  if (ar->input() >= al->output())
330  for (; ((al < sl->end()) && (ar->input() == al->output())); ++al)
331  ;
332  if (ar->input() > al->output()) {
333  if (type() == Fsa::TypeAcceptor)
334  label = Fsa::Failure;
335  sp->newArc(insertState(sl->id(), 0, sar->target()),
336  semiring()->extend(al->weight(), sar->weight()), Fsa::Failure, label);
337  }
338  }
339  break;
340  case Fsa::Else:
341  for (al = sl->begin(), ar = sr->begin(); (al < sl->end()) && (ar < sr->end()); ++ar) {
342  if (al->output() == ar->input())
343  for (; ((al < sl->end()) && (al->output() == ar->input())); ++al)
344  ;
345  if ((al < sl->end()) && (al->output() < ar->input())) {
346  if (type() == Fsa::TypeAcceptor)
347  label = Fsa::Else;
348  sp->newArc(insertState(sl->id(), 0, sar->target()),
349  semiring()->extend(al->weight(), sar->weight()), Fsa::Failure, label);
350  break;
351  }
352  if ((al->output() > Fsa::LastLabelId) || (ar->input() > Fsa::LastLabelId))
353  break;
354  }
355  break;
356  default:
357  break;
358  }
359  }
360  }

390 class ComposeMatchingAutomaton : public GeneralComposeAutomaton<_Automaton> {
391 private:
392  typedef GeneralComposeAutomaton<_Automaton> Precursor;
393 
394 public:
395  typedef typename _Automaton::State _State;
396  typedef typename _Automaton::ConstStateRef _ConstStateRef;
397  typedef typename _Automaton::ConstRef _ConstAutomatonRef;
398 
399 public:
400  ComposeMatchingAutomaton(_ConstAutomatonRef fl, _ConstAutomatonRef fr, bool reportUnknowns = true)
401  : Precursor(fl, fr, reportUnknowns) {}
402  virtual ~ComposeMatchingAutomaton() {}
403 
404  virtual _ConstStateRef getState(Fsa::StateId s) const {
405  if (s < Precursor::states_.size()) {
406  Fsa::StateId l = Precursor::states_[s].l(), r = Precursor::states_[s].r();
407  _ConstStateRef sl = Precursor::fl_->getState(l), sr = Precursor::fr_->getState(r);
408  typename _State::const_iterator al = sl->begin(), ar = sr->begin();
409 
410  _State* sp = new _State(s);
411  switch (Precursor::states_[s].f()) {
412  case 0:
413  if ((al != sl->end()) && (al->output() == Fsa::Epsilon)) {
414  for (; (ar != sr->end()) && (ar->input() == Fsa::Epsilon); ++ar)
415  ;
416  if ((ar != sr->end()) || (sr->isFinal()))
417  for (; (al != sl->end()) && (al->output() == Fsa::Epsilon); ++al)
418  sp->newArc(Precursor::insertState(al->target(), 1, r),
419  this->semiring()->extend(al->weight(), Precursor::fr_->semiring()->one()),
420  al->input(), Fsa::Epsilon);
421  al = sl->begin();
422  ar = sr->begin();
423  }
424  if ((ar != sr->end()) && (ar->input() == Fsa::Epsilon)) {
425  for (; (al != sl->end()) && (al->output() == Fsa::Epsilon); ++al)
426  ;
427  if ((al != sl->end()) || (sl->isFinal()))
428  for (; (ar != sr->end()) && (ar->input() == Fsa::Epsilon); ++ar)
429  sp->newArc(Precursor::insertState(l, 2, ar->target()),
430  this->semiring()->extend(Precursor::fl_->semiring()->one(), ar->weight()),
431  Fsa::Epsilon, ar->output());
432  al = sl->begin();
433  ar = sr->begin();
434  }
435  break;
436  case 1:
437  for (; (al != sl->end()) && (al->output() == Fsa::Epsilon); ++al)
438  sp->newArc(Precursor::insertState(al->target(), 1, r),
439  this->semiring()->extend(al->weight(), Precursor::fr_->semiring()->one()),
440  al->input(), Fsa::Epsilon);
441  break;
442  case 2:
443  for (; (ar != sr->end()) && (ar->input() == Fsa::Epsilon); ++ar)
444  sp->newArc(Precursor::insertState(l, 2, ar->target()),
445  this->semiring()->extend(Precursor::fl_->semiring()->one(), ar->weight()),
446  Fsa::Epsilon, ar->output());
447  break;
448  default:
449  break;
450  }
451  Precursor::composeArcs(sp, sl, sr, al, ar);
452  Precursor::composeSpecialArcs(sp, sl, sr);
453  return _ConstStateRef(sp);
454  }
455  return _ConstStateRef();
456  }
457  virtual std::string describe() const {
458  return Core::form("composeMatching(%s,%s)", Precursor::fl_->describe().c_str(), Precursor::fr_->describe().c_str());
459  }
