23 /*
24  * This code has to be very careful when changing the mapped files, because there
25  * may be other processes which try to write at the same time, or which are
26  * using some part of the data through memory-mapping.
27  *
28  * Therefore we use a copy-on-write mechanism: Whenever we do a change to the file,
29  * we create a new modified version of the file, and then move it
30  * over the old version. This way each new version gets a new identity (eg. inode number), and thus
31  * processed which are reading the old version are not affected.
32  *
33  * The temporary files are called $filename.temp.$hostname.$pid in the original location. Modern copy-on-write
34  * file systems like ZFS will be able to handle the copying and moving very efficiently.
35  *
36  * The changed files are moved back when the process finishes. The downside of this
37  * approach is that crashed processes may leave behind dangling temporary files.
38  * */

40 std::string hostAndProcessId() {
41  enum { Len = 1000 };
42  char buf[Len];
43  gethostname(buf, Len);
44  u64 pid = getpid();
45  std::ostringstream os;
46  os << buf << "." << pid;
47  return os.str();
48 }

61 bool MMappedFile::load(std::string file) {
62  unload();
63 
64  desc_ = open(file.c_str(), O_RDONLY);
65  if (desc_ == -1)
66  return false;
67  struct stat buf;
68  fstat(desc_, &buf);
69  size_ = buf.st_size;
70  data_ = reinterpret_cast<char*>(mmap(0, size_, PROT_READ, MAP_SHARED, desc_, 0));
71  if (data_ == MAP_FAILED) {
72  data_ = 0;
73  size_ = 0;
74  }
75  return (bool)data_;
76 }

128 MappedArchiveWriter MappedArchive::getWriter(std::string name) {
129  initializeTempFile();
130 
131  if (tempFile_.empty())
132  return MappedArchiveWriter();
133 
134  verify(!currentWriter_);
135  currentWriter_ = new Writer(name, new std::ofstream(tempFile_.c_str(), std::ofstream::app), this);
136 
137  // Skip the header, which we will write later
138  currentWriter_->itemOffset_ = currentWriter_->out_->tellp();
139  if (!currentWriter_->out_->good() || currentWriter_->itemOffset_ == -1) {
140  delete currentWriter_;
141  currentWriter_ = 0;
142  Core::Application::us()->log() << tempFile_ << ": creating writer failed " << name;
143  return MappedArchiveWriter();
144  }
145  else {
146  Core::Application::us()->log() << tempFile_ << ": created writer " << name;
147  }
148 
149  verify(currentWriter_->itemOffset_ != 0); // Must be at least behind the header
150 
151  // Add some zero-filled header space, which will be filled in releaseWriter
152  std::vector<char> dummy(sizeof(u32) + sizeof(u64) + name.length(), 0);
153  currentWriter_->out_->write(dummy.data(), dummy.size());
154 
155  currentWriter_->dataOffset_ = currentWriter_->out_->tellp();
156 
157  wroteTempFileItems_.insert(name);
158 
159  return MappedArchiveWriter(Core::Ref<Writer>(currentWriter_));
160 }

162 void MappedArchive::releaseWriter(MappedArchive::Writer* writer) {
163  verify(currentWriter_ == writer);
164 
165  bool failed = !writer->out_->good();
166 
167  // Write the ItemInfo header
168  size_t writePos = writer->out_->tellp();
169 
170  delete writer->out_;
171  currentWriter_ = 0;
172 
173  if (failed) {
174  Core::Application::us()->log() << tempFile_ << ": Writing temp-file failed, making archive read-only";
175  readOnly_ = true;
176  tempFile_.clear();
177  wroteTempFileItems_.clear();
178  return;
179  }
180 
181  // We have to use fopen instead of ofstream to update the header, because
182  // only its "r+" mode allows selectively overwriting some bytes of a file.
183  FILE* f = fopen(tempFile_.c_str(), "r+");
184  fseek(f, writer->itemOffset_, SEEK_SET);
185  verify(writer->dataOffset_ == writer->itemOffset_ + sizeof(u32) + sizeof(u64) + writer->name_.length());
186 
187  u32 nameLength = writer->name_.length();
188  u64 dataSize = writePos - writer->dataOffset_;
189 
190  fwrite((char*)&nameLength, sizeof(u32), 1, f);
191  fwrite((char*)&dataSize, sizeof(u64), 1, f);
192 
193  verify(ftell(f) == writer->itemOffset_ + sizeof(u32) + sizeof(u64));
194 
195  fwrite(writer->name_.data(), sizeof(char), writer->name_.length(), f);
196 
197  verify(ftell(f) == writer->itemOffset_ + sizeof(u32) + sizeof(u64) + nameLength);
198  verify(writer->dataOffset_ == writer->itemOffset_ + sizeof(u32) + sizeof(u64) + nameLength);
199 
200  verify((size_t)ftell(f) == writer->dataOffset_);
201  fclose(f);
202 
203  Core::Application::us()->log() << tempFile_ << ": wrote " << writer->name_ << ", data size: " << dataSize;
204 }

215 void MappedArchive::finalize() {
216  if (wroteTempFileItems_.empty())
217  return;
218 
219  // Unfortunately we cannot use Core::Application::us()->log() here, because the system may already be shutting down
220 
221  std::map<std::string, MappedItem> addItems;
222  bool failed = false;
223  for (std::vector<MappedItem>::reverse_iterator it = items_.rbegin(); it != items_.rend(); ++it) {
224  if (wroteTempFileItems_.count(it->name) == 0) {
225  wroteTempFileItems_.insert(it->name);
226  std::cout << tempFile_ << ": Keeping " << it->name << " of size " << it->size << std::endl;
227  std::ofstream of(tempFile_.c_str(), std::ofstream::app);
228  u32 nameLength = it->name.length();
229  u64 dataSize = it->size;
230  of.write((char*)&nameLength, sizeof(u32));
231  of.write((char*)&dataSize, sizeof(u64));
232  of.write(it->name.data(), it->name.length());
233  of.write(it->data, it->size);
234  if (!of.good()) {
235  failed = true;
236  std::cout << tempFile_.data() << ": Writing failed, discarding written data" << std::endl;
237  break;
238  }
239  }
240  }
241 
242  if (!failed && rename(tempFile_.c_str(), archivePath_.c_str()) == 0) {
243  std::cout << tempFile_.data() << ": Renamed to " << archivePath_.data() << std::endl;
244  }
245  else {
246  if (!failed)
247  std::cout << tempFile_.data() << ": Failed renaming to " << archivePath_.data() << std::endl;
248  ;
249  remove(tempFile_.c_str());
250  }
251 }

253 bool MappedArchive::loadData() {
254  items_.clear();
255  bool loaded = loadFile();
256  if (loaded && get<u32>() == Version) {
257  Core::Application::us()->log() << archivePath_ << ": Loading";
258 
259  while (readOffset_ < mapped_->size()) {
260  if (!available(sizeof(u32) + sizeof(u64)))
261  return false;
262 
263  u32 nameLength = get<u32>();
264  u64 dataSize = get<u64>();
265 
266  if (nameLength == 0 || !available(nameLength + dataSize))
267  return false;
268 
269  std::string name;
270  for (u32 i = 0; i < nameLength; ++i)
271  name.push_back(get<char>());
272 
273  MappedItem i;
274  i.name = name;
275  i.data = mapped_->data<char>(readOffset_);
276  i.size = dataSize;
277 
278  skip(dataSize);
279 
280  Core::Application::us()->log() << archivePath_ << ": Item " << i.name << " size " << i.size;
281 
282  items_.push_back(i);
283  }
284 
285  return true;
286  }
287 
288  return false;
289 }
