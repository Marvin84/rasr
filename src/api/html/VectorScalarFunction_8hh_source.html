37 /** NormFunction: parameter-th norm of input vector
38  * norm = (sum_i |x_i|^parameter )^{1/parameter}
39  */
40 template<class T>
41 class NormFunction : public virtual Core::Component,
42  public VectorScalarFunction<T, f64> {
43 public:
44  static std::string name() {
45  return "norm";
46  }
47  NormFunction(const Core::Configuration& c)
48  : Component(c) {}
49 
50  T apply(const std::vector<T>& v, f64 parameter) {
51  if (!v.empty()) {
52  typename std::vector<T>::const_iterator begin = v.begin();
53  typename std::vector<T>::const_iterator end = v.end();
54  T result = 0;
55  if (parameter == (f64)1) {
56  for (; begin != end; ++begin)
57  result += Core::abs(*begin);
58  }
59  else if (parameter == (f64)2) {
60  result = sqrt(std::inner_product(begin, end, begin, 0.0));
61  }
62  else if (parameter >= Core::Type<f64>::max) {
63  for (; begin != end; ++begin) {
64  T tmp = Core::abs(*begin);
65  if (result < tmp)
66  result = tmp;
67  }
68  }
69  else {
70  for (; begin != end; ++begin)
71  result += pow(Core::abs(*begin), parameter);
72  result = pow(result, 1.0 / parameter);
73  }
74  return result;
75  }
76  criticalError("Input vector is empty.");
77  return 0;
78  }
79 };
