155  Lattice::ConstWordLatticeRef denominator,
156  Bliss::SpeechSegment* s) {
157  if (denominator and denominator->nParts() == 1) {
158  if (needsMerging(denominator, s)) {
159  Lattice::ConstWordLatticeRef numerator;
160  if (numeratorArchiveReader_) {

195 void SpokenAndCompetingListProcessor::processWordLattice(
196  Lattice::ConstWordLatticeRef l, Bliss::SpeechSegment* s) {
197  if (l) {
198  Core::Vector<Lattice::ConstWordLatticeRef> hypotheses;
199 
200  // add spoken hypothesis from archive
201  require(numeratorArchiveReader_);
202  hypotheses.push_back(
203  numeratorArchiveReader_->get(
204  s->fullName(),
205  prefix()));
206  if (!Fsa::isLinear(hypotheses.back()->mainPart())) {
207  criticalError("numerator must be linear");
208  }
209 
210  // add competing hypotheses from @input l
211  Fsa::ConstStateRef lSp =
212  l->mainPart()->getState(l->mainPart()->initialStateId());
213  for (Fsa::State::const_iterator hIt = lSp->begin();
214  (hypotheses.size() < numberOfHypotheses_) and (hIt != lSp->end());
215  ++hIt) {
216  Lattice::ConstWordLatticeRef h = Lattice::partial(l, hIt->target(), hIt->weight());
217  if (needsMerging(h, s)) {
218  hypotheses.push_back(h);
219  }
220  }
221 
222  // combine hypotheses in single lattice
223  for (u32 n = 0; n < hypotheses.size(); ++n) {
224  hypotheses[n] =
225  Lattice::staticCopy(
226  Lattice::normalize(
227  hypotheses[n]));
228  }
229  Lattice::ConstWordLatticeRef result = Lattice::unite(hypotheses);
230 
231  // pass result to next node
232  LatticeSetProcessor::processWordLattice(result, s);
233  }
234  else {
235  warning("no move for '%s' because of empty n-best list", s->fullName().c_str());
236  }
237 }
