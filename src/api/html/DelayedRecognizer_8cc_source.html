72 Core::Ref<const Feature> RecognizerDelayHandler::add(Core::Ref<const Feature> f) {
73  if (featureBuffer_.size() < bufferSize_) {
74  featureBuffer_.push_back(f);
75  return Core::Ref<const Feature>();
76  }
77  else {
78  Core::Ref<const Mm::ScaledFeatureScorer> scorer = acousticModel_->featureScorer();
79  if (scorer->isBuffered() && initializeScorer_)
80  initializeBatchScorer();
81  featureBuffer_.push_back(f);
82  Core::Ref<const Feature> currentFeature = featureBuffer_.front();
83  if (cache_.get()) {
84  Mm::CachedContextScorer cachedScorer = cache_->cacheScorer(currentFeature, scorer->nMixtures());
85  if (!cachedScorer->precached())
86  cachedScorer->setScorer(scorer->getScorer(featureBuffer_[featureScorerOffset_]));
87  featureBuffer_.pop_front();
88  setLookAhead();
89  recognizer_->feed(cachedScorer);
90  cache_->uncacheScorer(currentFeature, cachedScorer);
91  }
92  else {
93  Mm::FeatureScorer::Scorer currentScorer = scorer->getScorer(featureBuffer_[featureScorerOffset_]);
94  featureBuffer_.pop_front();
95  setLookAhead();
96  recognizer_->feed(currentScorer);
97  }
98  return currentFeature;
99  }
100 }

102 Core::Ref<const Feature> RecognizerDelayHandler::flush() {
103  if (featureBuffer_.empty())
104  return Core::Ref<const Feature>();
105  Core::Ref<const Mm::ScaledFeatureScorer> scorer = acousticModel_->featureScorer();
106  if (scorer->isBuffered() && initializeScorer_)
107  initializeBatchScorer();
108 
109  bool allPreCached = (bool)cache_;
110  if (cache_) {
111  for (std::deque<Core::Ref<const Feature>>::iterator it = featureBuffer_.begin(); it != featureBuffer_.end(); ++it) {
112  Mm::CachedContextScorer cachedScorer = cache_->cacheScorer(*it, scorer->nMixtures());
113  if (!cachedScorer->precached())
114  allPreCached = false;
115  cache_->uncacheScorer(*it, cachedScorer);
116  }
117  }
118 
119  Mm::FeatureScorer::Scorer currentScorer;
120  if (!allPreCached) {
121  if (featureBuffer_.size() > featureScorerOffset_) {
122  currentScorer = scorer->getScorer(featureBuffer_[featureScorerOffset_]);
123  }
124  else {
125  verify_(scorer->isBuffered());
126  currentScorer = scorer->flush();
127  }
128  }
129  Core::Ref<const Feature> currentFeature = featureBuffer_.front();
130  featureBuffer_.pop_front();
131  if (cache_) {
132  Mm::CachedContextScorer cachedScorer = cache_->cacheScorer(currentFeature, scorer->nMixtures());
133  cachedScorer->setScorer(currentScorer);
134  setLookAhead();
135  recognizer_->feed(cachedScorer);
136  cache_->uncacheScorer(currentFeature, cachedScorer);
137  }
138  else {
139  setLookAhead();
140  recognizer_->feed(currentScorer);
141  }
142  return currentFeature;
143 }

153 void RecognizerDelayHandler::initializeBatchScorer() {
154  Core::Ref<const Mm::ScaledFeatureScorer> scorer = acousticModel_->featureScorer();
155  u32 pos = 0;
156  while (!scorer->bufferFilled() && pos < featureBuffer_.size()) {
157  scorer->addFeature(featureBuffer_[pos]);
158  ++pos;
159  }
160  featureScorerOffset_ = pos;
161  // No check whether the scorer buffer is filled.
162  // The scorer buffer size might be bigger than our feature buffer size.
163  initializeScorer_ = false;
