35  void visit(StateId node, Token token) {
36  bool hadToken = token;
37 
38  if (token)
39  token = this->visitNode(node, token);
40 
41  if (token) {
42  for (HMMStateNetwork::SuccessorIterator target = tree.successors(node); target; ++target) {
43  if (target.isLabel()) {
44  this->visitExit(target.label(), token);
45  }
46  else {
47  visit(*target, token);
48  }
49  }
50  }
51

54  }
