33 struct hash_LabelIdVector {
34  size_t operator()(const LabelIdVector* x) const {
35  size_t return_value = 1;
36  hash<unsigned int> f;
37  for (LabelIdVector::const_iterator i = x->begin(); i != x->end(); ++i) {
38  size_t hash_value = f(*i);
39  return_value ^= hash_value;
40  return_value += (hash_value << (return_value & 7));
41  return_value ^= ((*i) << ((return_value >> 3) & 15));
42  }
43  return return_value;
44  }
45 };

47 struct equal_LabelIdVector {
48  size_t operator()(const LabelIdVector* x, const LabelIdVector* y) const {
49  bool return_value = 1;
50  if (x->size() != y->size())
51  return false;
52  LabelIdVector::const_iterator i = x->begin();
53  LabelIdVector::const_iterator j = y->begin();
54  while (i != x->end() && return_value) {
55  return_value *= (*i) == (*j);
56  ++i;
57  ++j;
58  }
59  return return_value;
60  }
61 };
