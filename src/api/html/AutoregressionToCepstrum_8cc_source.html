21 void Signal::autoregressionToCepstrum(f32 gain, const std::vector<f32>& a, std::vector<f32>& c) {
22  require(c.size() >= 2);
23  require(c.size() <= (a.size() + 1));
24 
25  size_t nCepstrum = c.size();
26 
27  c[0] = 2 * log(gain);
28  c[1] = -a[0];
29 
30  for (size_t n = 2; n < nCepstrum; ++n) {
31  c[n] = n * a[n - 1];
32  for (size_t k = 1; k < n; ++k)
33  c[n] += (n - k) * c[n - k] * a[k - 1];
34  c[n] /= (-(f32)n);
35  }
36 }
