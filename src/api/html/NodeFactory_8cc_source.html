22 void dumpFormatted(std::ostream& o, const std::string& s, u32 indent = 0) {
23  std::string::size_type i, j;
24  for (i = 0; (j = s.find_first_of("\n", i)) != std::string::npos; i = j + 1)
25  o << std::setw(indent) << " " << s.substr(i, j - i) << std::endl;
