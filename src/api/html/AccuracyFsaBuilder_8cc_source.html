69 void TimeAlignmentBasedMetricLatticeBuilder::initializeShortPauses(Bliss::LexiconRef lexicon) {
70  verify(shortPauses_.empty());
71  verify(tokenType_ != noneType);
72  std::vector<std::string> shortPausesLemmata = paramShortPausesLemmata(config);
73  for (std::vector<std::string>::const_iterator it = shortPausesLemmata.begin();
74  it != shortPausesLemmata.end(); ++it) {
75  std::string name(*it);
76  Core::normalizeWhitespace(name);
77  log("Append short pause lemma \"") << name << "\"";
78  const Bliss::Lemma* lemma = lexicon->lemma(name);
79  if (!lemma) {
80  error("Unknown lemma \"") << name << "\"";
81  continue;
82  }
83  switch (tokenType_) {
84  case lemmaType: {
85  shortPauses_.insert(lemma->id());
86  } break;
87  case lemmaPronunciationType:
88  case phoneType: {
89  if (lemma->nPronunciations() == 0) {
90  error("Short-pauses lemma \"") << lemma->name() << "\" does not have any pronunciation.";
91  }
92  Bliss::Lemma::PronunciationIterator p, p_end;
93  for (Core::tie(p, p_end) = lemma->pronunciations(); p != p_end; ++p) {
94  if (tokenType_ == lemmaPronunciationType) {
95  shortPauses_.insert(p->id());
96  }
97  else { // phoneType
98  const Bliss::Pronunciation& pron = *(p->pronunciation());
99  for (u32 i = 0; i < pron.length(); ++i) {
100  shortPauses_.insert(pron[i]);
101  }
102  }
103  }
104  } break;
105  default: {
106  criticalError("unknown token type");
107  } break;
108  }
109  }
110 }
