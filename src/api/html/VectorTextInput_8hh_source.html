63 /**
64  * VectorTextInputNode: data source using XML input
65  *
66  * Node reads an arbitrary XML file filtering out only tags produced by
67  * Flow::Vector::dump(...). All other tags are ignored.
68  * Node goes though the whole file and puts the read Vector<T> objects into
69  * its output link.
70  *
71  */
72 
73 template<class T>
74 class VectorTextInputNode : public VectorTextInputNodeBase {
75 private:
76  size_t givenSize_;
77 
78 protected:
79  virtual void create(size_t dim) {
80  givenSize_ = dim;
81  out_ = dataPtr(new Vector<T>());
82  }
83  virtual void setElements(const std::string& buffer) {
84  DataPtr<Vector<T>> out(out_);
85  ensure(out);
86 
87  out->clear();
88  std::istringstream stream(buffer);
89  std::copy(std::istream_iterator<T>(stream), std::istream_iterator<T>(),
90  std::back_inserter(*out));
91  if (givenSize_ > 0 && givenSize_ != out->size())
92  error("Vector dimension mismatch while processing \"%s\": %zd given and %zd read.", buffer.c_str(), givenSize_, out->size());
93  }
94 
95 public:
96  static std::string filterName() {
97  return std::string("generic-vector-") + Core::Type<T>::name + "-text-input";
98  }
99  VectorTextInputNode(const Core::Configuration& c)
100  : Core::Component(c), VectorTextInputNodeBase(c, Vector<T>::type()), givenSize_(0) {}
101  virtual ~VectorTextInputNode() {}
102 };
