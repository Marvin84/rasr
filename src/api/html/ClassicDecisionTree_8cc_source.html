58 void mapContext(PropertyMap::StringList& keys,
59  PropertyMap::ListOfIndexedStringList& values,
60  const Phonology& phonology) {
61  const Bliss::PhonemeInventory& inventory = *(phonology.getPhonemeInventory());
62  PropertyMap::IndexedStringList phonemeMap(inventory.nPhonemes() + 1);
63  phonemeMap[0].first = "#";
64  phonemeMap[0].second = PropertyMap::Index(Bliss::Phoneme::term);
65  Bliss::PhonemeInventory::PhonemeIterator it = inventory.phonemes().first;
66  for (size_t i = 1; i <= inventory.nPhonemes(); ++i, ++it) {
67  phonemeMap[i].first = (const std::string)(*it)->symbol();
68  phonemeMap[i].second = PropertyMap::Index((*it)->id());
69  }
70 
71  for (s32 i = phonology.maximumHistoryLength() - 1; 0 <= i; --i) {
72  keys.push_back("");
73  std::stringstream ss;
74  ss << "history[" << i << "]";
75  ss >> keys.back();
76  values.push_back(phonemeMap);
77  }
78  keys.push_back("central");
79  values.push_back(phonemeMap);
80  for (s32 i = 0; i < phonology.maximumFutureLength(); ++i) {
81  keys.push_back("");
82  std::stringstream ss;
83  ss << "future[" << i << "]";
84  ss >> keys.back();
85  values.push_back(phonemeMap);
86  }

170 PropertyMapDiff::PropertyMapDiff(const Core::Configuration& config,
171  const PropertyMap& map1,
172  const PropertyMap& map2)
173  : Core::Component(config),
174  map1_(map1),
175  map2_(map2),
176  differences_(0) {
177  Core::Component::Message msg = log();
178  Core::XmlWriter& xml = (Core::XmlWriter&)msg;
179 
180  size_t l = 0, r = 0;
181  if (!(map1_[l] == map2_[r])) {
182  logDiff(xml, l, r);
183  differences_ |= 1;
184  }
185  ++l;
186  ++r;
187  if (!(map1_[l] == map2_[r])) {
188  logDiff(xml, l, r);
189  differences_ |= 2;
190  }
191  ++l;
192  ++r;
193  if (map1.historySize() != map2.historySize()) {
194  xml << Core::XmlOpen("diff")
195  << Core::XmlOpen("left")
196  << Core::XmlFull("history-size", map1.historySize())
197  << Core::XmlClose("left")
198  << Core::XmlOpen("right")
199  << Core::XmlFull("history-size", map2.historySize())
200  << Core::XmlClose("right")
201  << Core::XmlClose("diff");
202  differences_ |= 8;
203  }
204  l += map1.historySize();
205  r += map2.historySize();
206  if (!(map1_[l] == map2_[r])) {
207  logDiff(xml, l, r);
208  differences_ |= 4;
209  }
210  ++l;
211  ++r;
212  if (map1.futureSize() != map2.futureSize()) {
213  xml << Core::XmlOpen("diff")
214  << Core::XmlOpen("left")
215  << Core::XmlFull("future-size", map1.futureSize())
216  << Core::XmlClose("left")
217  << Core::XmlOpen("right")
218  << Core::XmlFull("future-size", map2.futureSize())
219  << Core::XmlClose("right")
220  << Core::XmlClose("diff");
221  differences_ |= 16;
222  }
223  l += map1.futureSize();
224  r += map2.futureSize();
225  for (; l < map1.size() && r < map2.size(); ++l, ++r) {
226  if (map1.key(l) != map2.key(r)) {
227  xml << Core::XmlOpen("diff")
228  << Core::XmlOpen("left")
229  << Core::XmlFull("condtion", map1.key(l))
230  << Core::XmlClose("left")
231  << Core::XmlOpen("right")
232  << Core::XmlFull("condtion", map2.key(r))
233  << Core::XmlClose("right")
234  << Core::XmlClose("diff");
235  differences_ |= 32;
236  }
237  }
238  if ((l != map1.size()) || (r != map2.size())) {
239  xml << Core::XmlOpen("diff")
240  << Core::XmlOpen("left")
241  << Core::XmlFull("number-of-conditions",
242  map1.size() - map1.historySize() - map1.futureSize() - 3)
243  << Core::XmlClose("left")
244  << Core::XmlOpen("right")
245  << Core::XmlFull("number-of-conditions",
246  map2.size() - map2.historySize() - map2.futureSize() - 3)
247  << Core::XmlClose("right")
248  << Core::XmlClose("diff");
249  differences_ |= 32;
250  }
251 
252  if (!hasDifferences())
253  xml << Core::XmlFull("diff", "no differences");
254 }
