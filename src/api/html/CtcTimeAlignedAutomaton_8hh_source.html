33 /* It's already the composed automaton of the allophone state automaton
34  * and the posterior scores (with the time index as an additional state).
35  * Creating an automaton representing the time-state where there are
36  * arcs for all the input alphabet would be theoretically possible,
37  * but not practically because the alphabet is too big.
38  * We don't need to go through the whole alphabet but just through
39  * the allophone state automaton states.
40  */

95  void maybeInit() {
96  if (initialized_)
97  return;
98  initialized_ = true;
99  if (dead_)
100  return; // No need to explore.
101 
102  require_lt(allophoneStateId_, parent_->hypothesesAllophoneStateFsa_->size());
103  auto* allophoneState = parent_->hypothesesAllophoneStateFsa_->fastState(allophoneStateId_);
104  require(allophoneState);
105  if (timeIdx_ == parent_->nTimeFrames_ && allophoneState->isFinal()) {
106  this->addTags(Fsa::StateTagFinal);
107  }
108 
109  this->setWeight(allophoneState->weight());
110 
111  if (timeIdx_ < parent_->nTimeFrames_) {
112  // If we did not reached the final time index, we have exactly
113  // those outgoing arcs as the underlying allophone state.
114  // All of these outgoing arcs are increasing the time index by one.
115  this->arcs_.reserve(allophoneState->nArcs());
116  for (size_t i = 0; i < allophoneState->nArcs(); ++i) {
117  const Arc& allophoneArc = *(*allophoneState)[i];
118  verify_ne(allophoneArc.input(), Fsa::Epsilon);
119  Fsa::StateId targetStateId = parent_->getStateId(timeIdx_ + 1, allophoneArc.target(), false);
120  // If we visited this frame in a backward search before,
121  // it can happen that we know that this arc leads nowhere.
122  if (targetStateId == Fsa::InvalidStateId)
123  continue;
124 
125  Arc& arc = *this->newArc();
126  arc.setInput(allophoneArc.input());
127  arc.setOutput(allophoneArc.output());
128  arc.setTarget(targetStateId);
129  FloatT weight = (FloatT)allophoneArc.weight();
130  weight += parent_->getAllophoneAcousticFeatureWeight(timeIdx_, allophoneArc.input());
131  arc.setWeight((Weight)weight);
132  }
133  }
134  }

136  template<int TimeIdxDiff>
137  void addDirScore(FloatT sourceScore, const Arc& allophoneArc) {
138  TimeIndex t = timeIdx_;
139  if (TimeIdxDiff > 0) {
140  require_gt(t, 0);
141  t--;
142  }
143  require_lt(t, parent_->nTimeFrames_);
144  FloatT weight = sourceScore;
145  weight += (FloatT)allophoneArc.weight();
146  weight += parent_->getAllophoneAcousticFeatureWeight(t, allophoneArc.input());
147  dirScore<TimeIdxDiff>() = (FloatT)Fsa::LogSemiring->collect(Fsa::Weight(dirScore<TimeIdxDiff>()), Fsa::Weight(weight));
148  }

209  void initStartState() {
210  Fsa::StateId allophoneInitialStateId = hypothesesAllophoneStateFsa_->initialStateId();

219  void initFinalStates() {
220  if (!hypothesesAllophoneStateFsaTransposed_) {
221  auto transposedFsa = Fsa::transpose(hypothesesAllophoneStateFsa_);
222  // It should be a static automaton.
223  const auto* transposedFsaStatic = dynamic_cast<const Fsa::StaticAutomaton*>(transposedFsa.get());
224  require(transposedFsaStatic);
225  hypothesesAllophoneStateFsaTransposed_ = Core::ref(transposedFsaStatic);
226  verify_ne(hypothesesAllophoneStateFsaTransposed_->initialStateId(), Fsa::InvalidStateId);
227  }
228  ConstStateRef transposedInitial = hypothesesAllophoneStateFsaTransposed_->getState(hypothesesAllophoneStateFsaTransposed_->initialStateId());
229  verify(transposedInitial);
230  // We expect that the transpose algo introduced a new initial state with only eps arcs
231  // to the original final states. Otherwise, no other eps arcs should have been introduced.
232  statesStartEndIdxs_[nTimeFrames_].first = states_.size();
233  for (const Arc& arc : *transposedInitial) {
234  verify_eq(arc.input(), Fsa::Epsilon);
235  Fsa::StateId allophoneStateId = arc.target();
236  Fsa::StateId ownStateId = getStateId(nTimeFrames_, allophoneStateId, true); // setup final state
237  verify_ne(ownStateId, Fsa::InvalidStateId);
238  states_[ownStateId].foundBackward_ = true;
239  states_[ownStateId].bwdScore_ = Fsa::LogSemiring->one();
240  }
241  statesSearchCompleted_[nTimeFrames_] = true;
242  statesStartEndIdxs_[nTimeFrames_].second = states_.size();
243  }

245  template<int TimeIdxDiff>
246  void _search(TimeIndex timeIdx, const Fsa::StaticAutomaton& fsa) {
247  bool searchAlreadyCompleted = statesSearchCompleted_[timeIdx + TimeIdxDiff];
248  if (!searchAlreadyCompleted)
249  statesStartEndIdxs_[timeIdx + TimeIdxDiff].first = states_.size();
250  auto ownStateIdStartEnd = statesStartEndIdxs_[timeIdx];
251  require_le(ownStateIdStartEnd.second, states_.size());
252  for (Fsa::StateId ownStateId = ownStateIdStartEnd.first; ownStateId < ownStateIdStartEnd.second; ++ownStateId) {
253  if (states_[ownStateId].dead_)
254  continue;
255  auto* allophoneState = fsa.fastState(states_[ownStateId].allophoneStateId_);
256  verify(allophoneState);
257  for (const Arc& arc : *allophoneState) {
258  verify_ne(arc.input(), Fsa::Epsilon);
259  Fsa::StateId targetAllophoneStateId = arc.target();
260  // Maybe setup new state.
261  Fsa::StateId targetOwnStateId = getStateId(timeIdx + TimeIdxDiff, targetAllophoneStateId, true);
262  if (!searchAlreadyCompleted)
263  verify_ne(targetOwnStateId, Fsa::InvalidStateId);
264  if (targetOwnStateId != Fsa::InvalidStateId) {
265  verify_lt(targetOwnStateId, states_.size());
266  State& targetState = states_[targetOwnStateId];
267  if (TimeIdxDiff > 0)
268  targetState.foundForward_ = true;
269  else if (TimeIdxDiff < 0)
270  targetState.foundBackward_ = true;
271  targetState.template addDirScore<TimeIdxDiff>(states_[ownStateId].template dirScore<TimeIdxDiff>(), arc);
272  }
273  }
274  }
275  if (!searchAlreadyCompleted)
276  statesStartEndIdxs_[timeIdx + TimeIdxDiff].second = states_.size();
277  statesSearchCompleted_[timeIdx + TimeIdxDiff] = true;
278  }

280  template<int TimeIdxDiff>
281  FloatT getMinDirScore(TimeIndex timeIdx) {
282  FloatT minScore = Core::Type<FloatT>::max;
283  auto ownStateIdStartEnd = statesStartEndIdxs_[timeIdx];
284  require_le(ownStateIdStartEnd.second, states_.size());
285  for (Fsa::StateId ownStateId = ownStateIdStartEnd.first; ownStateId < ownStateIdStartEnd.second; ++ownStateId) {
286  State& state = states_[ownStateId];
287  if (state.dead_)
288  continue;
289  if (minScore > state.template dirScore<TimeIdxDiff>())
290  minScore = state.template dirScore<TimeIdxDiff>();
291  }
292  return minScore;
293  }

295  template<int TimeIdxDiff>
296  void _prune(TimeIndex timeIdx, FloatT threshold) {
297  timeIdx += TimeIdxDiff;
298  threshold += getMinDirScore<TimeIdxDiff>(timeIdx);
299  auto ownStateIdStartEnd = statesStartEndIdxs_[timeIdx];
300  require_le(ownStateIdStartEnd.second, states_.size());
301  for (Fsa::StateId ownStateId = ownStateIdStartEnd.first; ownStateId < ownStateIdStartEnd.second; ++ownStateId) {
302  State& state = states_[ownStateId];
303  if (state.template dirScore<TimeIdxDiff>() > threshold)
304  state.dead_ = true;
305  }
306  }

327  void markDeadStates(TimeIndex timeIdx) {
328  // This assumes that the time frame was both visited by the
329  // forward and backward search.
330  // In that case, states which have not been found by both
331  // are dead states.
332  auto ownStateIdStartEnd = statesStartEndIdxs_[timeIdx];
333  require_le(ownStateIdStartEnd.second, states_.size());
334  for (Fsa::StateId ownStateId = ownStateIdStartEnd.first; ownStateId < ownStateIdStartEnd.second; ++ownStateId) {
335  State& state = states_[ownStateId];
336  if (!state.foundForward_ || !state.foundBackward_)
337  state.dead_ = true;
338  }
339  }

341  bool haveAnyStates(TimeIndex timeIdx) {
342  auto ownStateIdStartEnd = statesStartEndIdxs_[timeIdx];
343  require_le(ownStateIdStartEnd.second, states_.size());
344  for (Fsa::StateId ownStateId = ownStateIdStartEnd.first; ownStateId < ownStateIdStartEnd.second; ++ownStateId) {
345  State& state = states_[ownStateId];
346  if (!state.dead_)

352  template<int TimeIdxDiff>
353  void normalizeScores(TimeIndex timeIdx) {
354  // Basically we want: x_s /= sum(x)
355  // In -log-space, that is: x_s -= collect(x)
356  auto* collector = Fsa::LogSemiring->getCollector();
357  auto ownStateIdStartEnd = statesStartEndIdxs_[timeIdx];
358  require_le(ownStateIdStartEnd.second, states_.size());
359  for (Fsa::StateId ownStateId = ownStateIdStartEnd.first; ownStateId < ownStateIdStartEnd.second; ++ownStateId) {
360  State& state = states_[ownStateId];
361  if (state.dead_)
362  continue;
363  collector->feed(Fsa::Weight((FloatT)state.template dirScore<TimeIdxDiff>()));
364  }
365  FloatT score_sum = collector->get();
366  delete collector;
367  for (Fsa::StateId ownStateId = ownStateIdStartEnd.first; ownStateId < ownStateIdStartEnd.second; ++ownStateId) {
368  State& state = states_[ownStateId];
369  if (state.dead_)
370  continue;
371  state.template dirScore<TimeIdxDiff>() -= score_sum;
372  }
373  }

375  void fullSearch(FloatT pruneThreshold) {
376  require(isEmpty_);
377  require(!statesSearchCompleted_[0]);
378  require(!statesSearchCompleted_[nTimeFrames_]);
379  initStartState();
380  initFinalStates();
381 
382  // Search forward up to the middle frame - and the same for backward.
383  // This is exclusive the middle frame - but both searches will the middle frame.
384  TimeIndex middleTimeFrame = nTimeFrames_ / 2;
385  for (TimeIndex timeIdx = 0; timeIdx < middleTimeFrame; ++timeIdx) {
386  forwardSearch(timeIdx);
387  fwdPrune(timeIdx, pruneThreshold);
388  }
389  for (TimeIndex timeIdx = nTimeFrames_; timeIdx > middleTimeFrame + 1; --timeIdx) {
390  backwardSearch(timeIdx);
391  bwdPrune(timeIdx, pruneThreshold);
392  }
393  backwardSearch(middleTimeFrame + 1);
394 
395  // Both searches reached the middle frame, so any state which is not
396  // both found by forward + backward is a dead state.
397  markDeadStates(middleTimeFrame);
398 
399  if (!haveAnyStates(middleTimeFrame))
400  return;
401  isEmpty_ = false;
402 
403  // Complete the search of non-dead states both forward and backward.
404  // Now we can mark all the remaining dead states.
405  for (TimeIndex timeIdx = middleTimeFrame; timeIdx < nTimeFrames_; ++timeIdx) {
406  forwardSearch(timeIdx);
407  markDeadStates(timeIdx + 1);
408  }
409  for (TimeIndex timeIdx = middleTimeFrame; timeIdx > 0; --timeIdx) {
410  backwardSearch(timeIdx);
411  markDeadStates(timeIdx - 1);
412  }
413  }

415  void fullSearchFwdOnly(FloatT pruneThreshold) {
416  require_gt(nTimeFrames_, 0);
417  require(isEmpty_);
418  require(!statesSearchCompleted_[0]);
419  require(!statesSearchCompleted_[nTimeFrames_]);

434  void fullSearchAutoIncrease(FloatT minPruneThreshold, FloatT maxPruneThreshold) {
435  require_le(minPruneThreshold, maxPruneThreshold);

457  void extractAlignment(Speech::Alignment& out, FloatT minProbGT = 0, FloatT gamma = 1) {
458  out.clear();
459  for (TimeIndex timeIdx = 0; timeIdx < nTimeFrames_; ++timeIdx) {
460  auto ownStateIdStartEnd = statesStartEndIdxs_[timeIdx];
461  require_le(ownStateIdStartEnd.second, states_.size());
462  for (Fsa::StateId ownStateId = ownStateIdStartEnd.first; ownStateId < ownStateIdStartEnd.second; ++ownStateId) {
463  State& state = states_[ownStateId];
464  if (state.dead_)
465  continue;
466  auto* allophoneState = hypothesesAllophoneStateFsa_->fastState(states_[ownStateId].allophoneStateId_);
467  verify(allophoneState);
468  for (const Arc& arc : *allophoneState) {
469  verify_ne(arc.input(), Fsa::Epsilon);
470  Fsa::StateId targetAllophoneStateId = arc.target();
471  Fsa::StateId targetOwnStateId = getStateId(timeIdx + 1, targetAllophoneStateId, false);
472  if (targetOwnStateId != Fsa::InvalidStateId) {
473  State& targetState = states_[targetOwnStateId];
474  Am::AcousticModel::EmissionIndex emissionIdx = acousticModel_->emissionIndex(arc.input());
475  out.push_back(Speech::AlignmentItem(timeIdx,
476  arc.input(),
477  _totalScoreForArc(timeIdx, state, targetState, (FloatT)arc.weight(), emissionIdx)));
478  }
479  }
480  }
481  }
482  if (out.empty())
483  return; // unlikely but would happen if FSA was empty
484  out.combineItems(Fsa::LogSemiring);
485  require(!out.empty());
486  out.sortItems(false); // smallest -log-score means highest scores
487  out.clipWeights(0, Core::Type<Mm::Weight>::max);
488  out.multiplyWeights(gamma);
489  out.shiftMinToZeroWeights(); // more stable expm, is equivalent with normalizeWeights()
490  out.expm(); // to std space
491  out.normalizeWeights();
492  out.filterWeightsGT(minProbGT);
493  }

495  template<typename MatrixT>
496  void extractAlignmentMatrix(MatrixT& out, u32 nClasses, bool initMatrix) {
497  FloatT logZero = (FloatT)Fsa::LogSemiring->zero();
498  if (initMatrix) {
499  out.resize(nClasses, nTimeFrames_);
500  for (FloatT& v : out)
501  v = logZero;
502  }
503  else {
504  require_eq(out.nRows(), nClasses);
505  require_eq(out.nColumns(), nTimeFrames_);
506  }
507  for (TimeIndex timeIdx = 0; timeIdx < nTimeFrames_; ++timeIdx) {
508  auto ownStateIdStartEnd = statesStartEndIdxs_[timeIdx];
509  require_le(ownStateIdStartEnd.second, states_.size());
510  for (Fsa::StateId ownStateId = ownStateIdStartEnd.first; ownStateId < ownStateIdStartEnd.second; ++ownStateId) {
511  State& state = states_[ownStateId];
512  if (state.dead_)
513  continue;
514  auto* allophoneState = hypothesesAllophoneStateFsa_->fastState(states_[ownStateId].allophoneStateId_);
515  verify(allophoneState);
516  for (const Arc& arc : *allophoneState) {
517  verify_ne(arc.input(), Fsa::Epsilon);
518  Fsa::StateId targetAllophoneStateId = arc.target();
519  Fsa::StateId targetOwnStateId = getStateId(timeIdx + 1, targetAllophoneStateId, false);
520  if (targetOwnStateId != Fsa::InvalidStateId) {
521  State& targetState = states_[targetOwnStateId];
522  Am::AcousticModel::EmissionIndex emissionIdx = acousticModel_->emissionIndex(arc.input());
523  require_lt(emissionIdx, nClasses);
524  FloatT weight = _totalScoreForArc(timeIdx, state, targetState, (FloatT)arc.weight(), emissionIdx);
525 
526  // Check for absolut limits.
527  if (weight > logZero || std::isinf(weight) || Math::isnan(weight))
528  weight = logZero;
529 
530  if (out.at(emissionIdx, timeIdx) >= logZero)
531  out.at(emissionIdx, timeIdx) = weight;
532  else
533  out.at(emissionIdx, timeIdx) = Fsa::LogSemiring->collect(Weight(out.at(emissionIdx, timeIdx)), Weight(weight));
534  }
535  }
536  }
537  }

540  Fsa::StateId getStateId(TimeIndex timeIdx, Fsa::StateId allophoneStateId, bool autoCreateNew) {
541  verify_le(timeIdx, nTimeFrames_);
542  {
543  // Search if it exists in cache.
544  require_lt(allophoneStateId, statesByAllo_.size());
545  auto& stateIdsByTime = statesByAllo_[allophoneStateId];
546  auto s = stateIdsByTime.find(timeIdx);
547  if (s != stateIdsByTime.end()) {
548  Fsa::StateId stateId = s->second;
549  State& state = states_[stateId];
550  if (state.dead_)
551  return Fsa::InvalidStateId;
552  return stateId;
553  }
554  }
555  // If we already have exhausted the search on this timeframe,
556  // any new state would be a dead-end.
557  if (statesSearchCompleted_[timeIdx])
558  return Fsa::InvalidStateId;
559  // Create new one.
560  require(autoCreateNew);
561  Fsa::StateId stateId = states_.size();
562  states_.push_back(State(this, stateId, timeIdx, allophoneStateId));
563  statesByAllo_[allophoneStateId][timeIdx] = stateId;
564  if (stateId == 0)
565  verify_eq(timeIdx, 0);
566  if (timeIdx == 0)
567  verify_eq(stateId, 0); // there is only a single start state
568  return stateId;
