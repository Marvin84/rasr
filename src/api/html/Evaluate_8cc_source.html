131 Bliss::Evaluator* Evaluator::blissEvaluator() {
132  if (!blissEvaluator_ || (lastLemmaUpdates_ != Lexicon::us()->nLemmaUpdates())) {
133  if (blissEvaluator_) {
134  delete blissEvaluator_;
135  log("%d lemmas were updated; create new evaluator.",

137  }
138  blissEvaluator_ = new Bliss::Evaluator(config, Lexicon::us());
139  lastLemmaUpdates_ = Lexicon::us()->nLemmaUpdates();
140  }
141  return blissEvaluator_;

144 void Evaluator::logEvaluation(ConstLatticeRef l, const std::string& name) {
145  openLayer(name);

176 void Evaluator::evaluate(ConstLatticeRef l) {
177  if (l)
178  l = projectInput(mapInput(l, MapToLemma));
179  if (logSingleBest_) {
180  ConstLatticeRef eval = (l && !l->hasProperty(Fsa::PropertyLinear)) ? best(l, ssspAlgorithm_) : l;
181  logEvaluation(eval, "single best");
182  }
183  if (logBestInLattice_) {
184  ConstLatticeRef eval = l;
185  logEvaluation(eval, "best-in-lattice");
186  }
187 }

201 class EvaluatorNode : public FilterNode {
202  typedef FilterNode Precursor;
203 
204 private:
205  Evaluator* evaluator_;
206  ConstSemiringRef evalSemiring_;
207  bool validReference_;
208  Lexicon::SymbolMap lemmaSymbolMap_;
209  std::vector<Fsa::LabelId> refLabels_;
210 
211 protected:
212  void setReference() {
213  if (validReference_)
214  return;
215  if (connected(1)) {
216  ConstSegmentRef segment = requestSegment(1);
217  if (segment->hasOrthography()) {
218  lemmaSymbolMap_.indices(segment->orthography(), refLabels_);
219  evaluator_->setReference(segment->orthography());
220  }
221  else {
222  error("EvaluatorNode: Missing reference.");
223  evaluator_->setReference("");
224  }
225  }
226  else {
227  std::string refStr = requestString(2);
228  lemmaSymbolMap_.indices(refStr, refLabels_);
229  evaluator_->setReference(refStr);
230  }
231  refLabels_.clear();
232  validReference_ = true;
233  }
234  virtual ConstLatticeRef filter(ConstLatticeRef l) {
235  setReference();
236  ConstLatticeRef eval = l;
237  if (evalSemiring_)
238  eval = changeSemiring(eval, evalSemiring_);
239  evaluator_->evaluate(eval);
240  return l;
241  }
242 
243 public:
244  EvaluatorNode(const std::string& name, const Core::Configuration& config)
245  : Precursor(name, config), evaluator_(0) {
246  validReference_ = false;
247  }
248  virtual ~EvaluatorNode() {
249  delete evaluator_;
250  }
251  virtual void init(const std::vector<std::string>& arguments) {
252  if (!(connected(1) || connected(2)))
253  criticalError("EvaluatorNode: Need a data source either at port 1 or port 2");
254  evaluator_ = new Evaluator(config, name);
255  evalSemiring_ = Semiring::create(select("semiring"));
256  if (evalSemiring_)
257  log("Use semiring %s for evaluation.", evalSemiring_->name().c_str());
258  lemmaSymbolMap_ = Lexicon::us()->symbolMap(Lexicon::LemmaAlphabetId);
259  }
260  void sync() {
261  validReference_ = false;
262  }
263 };
