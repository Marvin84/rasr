75 class LmCache {
76  ///@todo Make sure that the handles of the cached histories are kept alive
77 public:
78  ///Should be called regularly to clean up the cache
79  ///All items that were not requested since the last call to clean()

81  ///@return the count of items remaining in the cache
82  u32 clean() {
83  oldCache_.swap(newCache_);
84  newCache_.clear();
85  return oldCache_.size();

88  ///@return Core::Type<Lm::Score>::max if the item needs to be calculated
89  LmCacheItem& retrieve(const LmCacheKey& key) {
90  Hash::iterator it = newCache_.find(key);
91  if (it != newCache_.end())
92  return (*it).second;
93 
94  it = oldCache_.find(key);
95 
96  LmCacheItem item = LmCacheItem(Core::Type<Lm::Score>::max);
97 
98  if (it != oldCache_.end()) {
99  item = (*it).second;
100  oldCache_.erase(it);
101  }
102 
103  return newCache_.insert(std::make_pair(key, item)).first->second;
104  }
105 
106 private:
107  typedef std::unordered_map<LmCacheKey, LmCacheItem, LmCacheHash, LmCacheEqual> Hash;
108  Hash oldCache_;
109  Hash newCache_;
