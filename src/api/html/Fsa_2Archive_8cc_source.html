126 ConstAutomatonRef ArchiveReader::get(const std::string& id) {
127  require(archive_);
128  Core::Ref<StorageAutomaton> storage(new StaticAutomaton);
129 
130  // these will be overwritten if they are contained in the file
131  storage->setType(TypeTransducer);
132  storage->setSemiring(semiring_);
133  if (inputAlphabet_) {
134  storage->setInputAlphabet(inputAlphabet_);
135  }
136  if (outputAlphabet_) {
137  storage->setOutputAlphabet(outputAlphabet_);
138  }
139 
140  if (!archive_->hasFile(id)) {
141  error("Could not find fsa \"%s\"", id.c_str());
142  return ConstAutomatonRef();
143  }
144  log("Reading fsa \"%s\"", id.c_str());
145  Core::ArchiveReader is(*archive_, id);
146  if (!readBinary(storage.get(), is)) {
147  error("Failed to read fsa");
148  return ConstAutomatonRef();
149  }
150 
151  ConstAutomatonRef result = storage;
152  if (result->type() == TypeTransducer) {
153  result = Fsa::mapInput(result, inputMapping_);
154  result = Fsa::mapOutput(result, outputMapping_);
155  }
156  else if (result->type() == TypeAcceptor) {
157  if (inputMapping_.to()) {
158  result = mapInputOutput(result, inputMapping_);
159  }
160  }
161 
162  return result;
163 }

199 void ArchiveWriter::store(const std::string& id, ConstAutomatonRef f) {
200  require(archive_);
201  StoredComponents what = storeAll;
202  if (f->inputAlphabet() == inputAlphabet_)
203  what &= ~storeInputAlphabet;
204  if (what & storeInputAlphabet)
205  warning("store input alphabet");
206  if (f->outputAlphabet() == outputAlphabet_)
207  what &= ~storeOutputAlphabet;
208  if (what & storeOutputAlphabet && outputAlphabet_)
209  warning("store output alphabet");
210 
211  info(f, log("Writing fsa \"%s\"", id.c_str()));
212  Core::ArchiveWriter os(*archive_, id, true);
213  writeBinary(f, os, what);
214  if (!os)
215  error("Failed to write fsa \"%s\"", id.c_str());
216 }
