28 void FastFourierTransform::createBitReversalReordering(u32 size) {
29  size_ = size;
30  u32 length = size / 2;
31  reording_.resize(size_);
32  std::iota(reording_.begin(), reording_.end(), 0);
33 
34  u32 j = 1;
35  for (u32 i = 1; i < size; i += 2) {
36  if (j > i) {
37  reording_[i - 1] = j - 1;
38  reording_[i] = j;
39  }
40  u32 m = length;
41  while (m >= 2 && j > m) {
42  j -= m;
43  m >>= 1;
44  }
45  j += m;
46  }
47 }

59 void FastFourierTransform::transform(std::vector<f32>& v, bool inverse) {
60  const u32 size = v.size();
61  const f64 thetaBase = DPi * (inverse ? -1.0 : 1.0);
62 
63  bitReversalReordering(v);
64  u32 curLength = 2;
65  // estimate DFFT using Danielson and Lanczos formula
66  while (curLength < size) {
67  u32 step = curLength << 1;
68  // initialization of trigonometric recurrence
69  f64 theta = thetaBase / curLength;
70  f64 sinHTheta = std::sin(0.5 * theta);
71  f64 wpR = -2.0 * sinHTheta * sinHTheta;
72  f64 wpI = std::sin(theta);
73  f64 wR = 1.0;
74  f64 wI = 0.0;
75  for (u32 m = 1; m < curLength; m += 2) {
76  for (u32 i = m; i <= size; i += step) {
77  // Danielson & Lanczos formula
78  u32 j = i + curLength;
79  f32 tmpR = wR * v[j - 1] - wI * v[j];
80  f32 tmpI = wR * v[j] + wI * v[j - 1];
81  v[j - 1] = v[i - 1] - tmpR;
82  v[j] = v[i] - tmpI;
83  v[i - 1] += tmpR;
84  v[i] += tmpI;
85  }
86  // trigonometric recurrence
87  f64 oldWR = wR;
88  wR = wR * wpR - wI * wpI + wR;
89  wI = wI * wpR + oldWR * wpI + wI;
90  }
91  curLength = step;
92  }
93 }

95 void FastFourierTransform::transformReal(std::vector<float>& v, bool inverse) {
96  // the real data is treated as complex array, transformed by transform(..)
97  // and then separated to get the transform of the real data
98  const u32 size = v.size();
99  const u32 sizeD4 = size >> 2;
100  const f64 theta = (Pi * (inverse ? -1 : 1)) / (size >> 1);
101  const f32 c = (inverse ? 0.5 : -0.5);
102  if (!inverse) {
103  // forward transform data
104  transform(v, false);
105  }
106  // initialization trigonometric recurrence
107  f64 sinHTheta = std::sin(0.5 * theta);
108  f64 wpR = -2.0 * sinHTheta * sinHTheta;
109  f64 wpI = std::sin(theta);
110  f64 wR = wpR + 1;
111  f64 wI = wpI;
112 
113  for (u32 i = 1; i < sizeD4; ++i) {
114  u32 i1 = i + i;
115  u32 i2 = i1 + 1;
116  u32 i3 = size - i1;
117  u32 i4 = i3 + 1;
118  // separate the two transforms ...
119  f64 h1R = 0.5 * (v[i1] + v[i3]);
120  f64 h1I = 0.5 * (v[i2] - v[i4]);
121  f64 h2R = -c * (v[i2] + v[i4]);
122  f64 h2I = c * (v[i1] - v[i3]);
123  // ... and combine them to the true
124  // transform of the original real data
125  v[i1] = h1R + wR * h2R - wI * h2I;
126  v[i2] = h1I + wR * h2I + wI * h2R;
127  v[i3] = h1R - wR * h2R + wI * h2I;
128  v[i4] = -h1I + wR * h2I + wI * h2R;
129  // trigonometric recurrence
130  f64 oldWR = wR;
131  wR = wR * wpR - wI * wpI + wR;
132  wI = wI * wpR + oldWR * wpI + wI;
133  }
134  if (inverse) {
135  f32 h = v[0];
136  v[0] = 0.5 * (h + v[1]);
137  v[1] = 0.5 * (h - v[1]);
138  // inverse transform
139  transform(v, true);
140  }
141  else {
142  f32 h = v[0];
143  v[0] = h + v[1];
144  v[1] = h - v[1];
145  }
146 }
