113 inline void addSyntacticTokenScore(

115  Score lmScale,
116  const Bliss::SyntacticToken* st,
117  Score syntaxEmissionScale,

136 /** Language model score convenience function for lemmas. */
137 inline void addLemmaScore(
138  Core::Ref<const LanguageModel> lm,
139  Score lmScale,
140  const Bliss::Lemma* lemma,
141  Score syntaxEmissionScale,
142  History& history, Score& score) {
143  require(lemma);
144  const Bliss::SyntacticTokenSequence tokenSequence(lemma->syntacticTokenSequence());
145  for (u32 ti = 0; ti < tokenSequence.length(); ++ti) {
146  const Bliss::SyntacticToken* st = tokenSequence[ti];
147  addSyntacticTokenScore(lm, lmScale, st, syntaxEmissionScale, history, score);
148  history = lm->extendedHistory(history, st);
149  }
150 }

152 /** Language model score convenience function for lemmas. */
153 inline void addLemmaScoreOmitExtension(
154  Core::Ref<const LanguageModel> lm,
155  Score lmScale,
156  const Bliss::Lemma* lemma,
157  Score syntaxEmissionScale,
158  const History& history, Score& score) {
159  require(lemma);
160  const Bliss::SyntacticTokenSequence tokenSequence(lemma->syntacticTokenSequence());
161  if (tokenSequence.length() == 0) {
162  return;
163  }
164  else if (tokenSequence.length() == 1) {
165  addSyntacticTokenScore(lm, lmScale, tokenSequence[0], syntaxEmissionScale, history, score);
166  }
167  else {
168  Lm::History h = history;
169  u32 last = tokenSequence.length() - 1;
170  for (u32 ti = 0; ti < tokenSequence.length(); ++ti) {
171  const Bliss::SyntacticToken* st = tokenSequence[ti];
172  addSyntacticTokenScore(lm, lmScale, st, syntaxEmissionScale, h, score);
173  if (ti != last)
174  h = lm->extendedHistory(h, st);
175  }
176  }
177 }
