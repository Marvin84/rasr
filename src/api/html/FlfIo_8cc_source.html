95 std::string FlfDescriptor::info() const {
96  std::ostringstream oss;
97  oss << "lattice descriptor:" << std::endl
98  << " semiring type: " << getSemiringTypeName(semiringType) << std::endl
99  << " lattice root: " << latticeRoot << std::endl
100  << " structure file: " << structureFile(latticeRoot) << std::endl;
101  if (hasInputAlphabet) {
102  oss << " input-alphabet name: " << inputAlphabetName << std::endl;
103  if (inputAlphabetFile)
104  oss << " input-alphabet file: " << inputAlphabetFile(latticeRoot) << std::endl;
105  else
106  oss << " no input-alphabet file" << std::endl;
107  }
108  else
109  oss << " no input-alphabet" << std::endl;
110  if (hasOutputAlphabet) {
111  oss << " output-alphabet name: " << outputAlphabetName << std::endl;
112  if (outputAlphabetFile)
113  oss << " output-alphabet file: " << outputAlphabetFile(latticeRoot) << std::endl;
114  else
115  oss << " no output-alphabet file" << std::endl;
116  }
117  else
118  oss << " no output-alphabet" << std::endl;
119  if (hasBoundaries)
120  oss << " boundaries file: " << boundariesFile(latticeRoot) << std::endl;
121  else
122  oss << " no boundaries" << std::endl;
123  oss << " scores:" << std::endl;
124  oss << " " << std::setw(4) << " "
125  << " "
126  << std::setw(16) << std::left << "key"
127  << " "
128  << std::setw(5) << std::left << "scale"
129  << " "
130  << "fsa" << std::endl;
131  KeyList::const_iterator itK = keys.begin();
132  ScoreList::const_iterator itS = scales.begin();
133  FsaDescriptorList::const_iterator itF = scoreFiles.begin();
134  for (size_t id = 1; id <= scoreFiles.size(); ++id, ++itK, ++itS, ++itF) {
135  oss << " " << std::setw(4) << std::right << id << ". ";
136  oss << std::setw(16) << std::left << *itK << " ";
137  if (*itS == Semiring::UndefinedScale)

139  else
140  oss << std::setw(5) << std::right << *itS << " ";
141  if (*itF)

143  oss << std::endl;

153 void partial(FlfDescriptorRef desc, const KeySet& keys, bool onlyScores) {
154  FsaDescriptor alternativeStructureFile;
155  bool useAlternativeStructureFile = false;
156  KeyList::iterator itKey = desc->keys.begin();
157  ScoreList::iterator itScale = desc->scales.begin();
158  FsaDescriptorList::iterator itScoreFile = desc->scoreFiles.begin();
159  while (itKey != desc->keys.end())
160  if (keys.find(*itKey) == keys.end()) {
161  if (*itScoreFile == desc->structureFile)
162  useAlternativeStructureFile = true;
163  itKey = desc->keys.erase(itKey);
164  itScale = desc->scales.erase(itScale);
165  itScoreFile = desc->scoreFiles.erase(itScoreFile);
166  }
167  else {
168  if (!alternativeStructureFile)
169  alternativeStructureFile = *itScoreFile;
170  ++itKey;
171  ++itScale;
172  ++itScoreFile;
173  }
174  if (onlyScores) {
175  desc->structureFile.clear();
176  desc->boundariesFile.clear();
177  desc->hasBoundaries = false;
178  desc->inputAlphabetFile.clear();
179  desc->hasInputAlphabet = false;
180  desc->outputAlphabetFile.clear();
181  desc->hasOutputAlphabet = false;
182  }
183  else if (useAlternativeStructureFile && alternativeStructureFile)
184  desc->structureFile = alternativeStructureFile;
185 }

187 void append(FlfDescriptorRef desc, const KeyList& keys, const ScoreList& scales) {
188  verify(keys.size() == scales.size());
189  KeyList::const_iterator itKey = keys.begin();
190  ScoreList::const_iterator itScale = scales.begin();
191  for (; itKey != keys.end(); ++itKey, ++itScale) {
192  desc->keys.push_back(*itKey);
193  desc->scales.push_back(*itScale);
194  desc->scoreFiles.push_back(FsaDescriptor());
195  }

234 std::pair<Fsa::ConstAlphabetRef, Fsa::ConstAlphabetRef> loadAlphabets(const FsaDescriptor& qf, Core::Archive* archive) {
235  if (!qf)
236  return std::make_pair(Fsa::ConstAlphabetRef(), Fsa::ConstAlphabetRef());
237  Fsa::StaticAutomaton s;
238  s.setSemiring(Fsa::TropicalSemiring);
239  InputStream is(qf.file, archive);
240  if (!is || !Fsa::read(&s, qf.format, is)) {
241  Core::Application::us()->criticalError(
242  "Failed to read fsa \"%s\"",
243  qf.file.c_str());
244  return std::make_pair(Fsa::ConstAlphabetRef(), Fsa::ConstAlphabetRef());
245  }
246  return std::make_pair(s.getInputAlphabet(), s.getOutputAlphabet());
247 }

258  return Lexicon::us()->alphabetMap(alphabet, alphabetId);

298 void FlfContext::overhaul(FlfDescriptorRef desc, const Fsa::StaticAutomaton& structureFsa) const {
299  if (!semiring_)
300  Core::Application::us()->criticalError(
301  "FlfContext: No semiring set");
302  if (desc->scoreFiles.size() != semiring_->size())
303  Core::Application::us()->criticalError(
304  "FlfContext: Semiring dimension mismatch, expected %zu, got %zu",
305  desc->scoreFiles.size(),
306  semiring_->size());
307  if (fsaType_ == Fsa::TypeUnknown)
308  Core::Application::us()->criticalError(
309  "FlfContext: No fsa type set");
310  if (!inputAlphabetMap_)
311  Core::Application::us()->criticalError(
312  "FlfContext: Missing input alphabet");
313  if ((fsaType_ != Fsa::TypeAcceptor) && (!outputAlphabetMap_))
314  Core::Application::us()->criticalError(
315  "FlfContext: Missing output alphabet in not-acceptor automaton");
316 }

318 void FlfContext::adapt(FlfDescriptorRef desc, const Fsa::StaticAutomaton& structureFsa, Core::Archive* archive) {
319  // adapt semiring
320  if (!semiring_) {
321  Fsa::SemiringType type = desc->semiringType;
322  if (type == Fsa::SemiringTypeUnknown)
323  type = Fsa::getSemiringType(structureFsa.semiring());
324  if ((type != Fsa::SemiringTypeTropical) && (type != Fsa::SemiringTypeLog)) {
325  Core::Application::us()->warning(
326  "FlfContext: No or unknown semiring type specified, assume tropical semiring");
327  type = Fsa::SemiringTypeTropical;
328  }
329  setSemiring(Semiring::create(
330  type,
331  desc->scoreFiles.size(),
332  desc->scales,
333  desc->keys));
334  desc->keys = semiring_->keys();
335  }
336  // adapt alphabets => fsa type
337  if (fsaType_ == Fsa::TypeUnknown) {
338  require(!inputAlphabetMap_ && !outputAlphabetMap_);
339  if ((bool)desc->inputAlphabetFile)
340  loadInputAlphabet(
341  FsaDescriptor::joinPaths(desc->inputAlphabetFile, desc->latticeRoot),
342  archive,
343  desc->inputAlphabetName);
344  else
345  setInputAlphabet(
346  structureFsa.getInputAlphabet(),
347  desc->inputAlphabetName);
348  if (structureFsa.type() != Fsa::TypeAcceptor) {
349  if ((bool)desc->outputAlphabetFile) {
350  loadOutputAlphabet(
351  FsaDescriptor::joinPaths(desc->outputAlphabetFile, desc->latticeRoot),
352  archive,
353  desc->outputAlphabetName);
354  }
355  else
356  setOutputAlphabet(
357  structureFsa.getOutputAlphabet(),
358  desc->outputAlphabetName);
359  }
360  }
361  overhaul(desc, structureFsa);
362 }

499  void dim(const Core::XmlAttributes atts) {
500  const char* c = atts["id"];
501  if (hasId_ == -1)
502  hasId_ = bool(c);
503  if (c) {
504  if (!(hasId_))
505  parser()->error("do not mix dim-tags with and w/o id-attribute");
506  if (!Core::strconv(std::string(c), id_))
507  parser()->error("could not parse id: %s", cdata_.c_str());
508  }
509  else
510  ++id_;
511  if (n_ == 0) {
512  if (desc_->keys.size() <= id_) {
513  desc_->keys.resize(id_ + 1);
514  desc_->scales.resize(id_ + 1, Semiring::UndefinedScale);
515  desc_->scoreFiles.resize(id_ + 1);
516  }
517  }
518  else if (size_t(id_) >= n_)
519  parser()->error("id %zu found, but dimensionality is set to %zu", id_, n_);
520  }

638 void FlfReader::init() {
639  KeyList partialKeys = paramKeys(select("partial"));
640  for (KeyList::const_iterator it = partialKeys.begin(); it != partialKeys.end(); ++it)
641  partialKeys_.insert(*it);
642  const Core::Configuration appendConfig(config, "append");

699 void FlfReader::addStructure(StaticLattice& l, const Fsa::StaticAutomaton& f) {
700  if (f.maxStateId() == Fsa::InvalidStateId)
701  return;
702  ConstSemiringRef semiring = l.semiring();
703  Lexicon::AlphabetMap& mapInputLabel = *context_->inputAlphabetMap();
704  Lexicon::AlphabetMap& mapOutputLabel = (context_->fsaType() == Fsa::TypeAcceptor) ? *context_->inputAlphabetMap() : *context_->outputAlphabetMap();
705  for (Fsa::StateId fsaSid = 0; fsaSid <= f.maxStateId(); ++fsaSid) {
706  const Fsa::State* fsaSp = f.fastState(fsaSid);
707  if (fsaSp) {
708  State* sp = new State(fsaSp->id(), fsaSp->tags(),
709  fsaSp->isFinal() ? semiring->clone(semiring->defaultWeight()) : semiring->one());
710  for (Fsa::State::const_iterator fsaA = fsaSp->begin(); fsaA != fsaSp->end(); ++fsaA) {
711  sp->newArc(fsaA->target(),
712  semiring->clone(semiring->defaultWeight()),
713  mapInputLabel[fsaA->input()], mapOutputLabel[fsaA->output()]);
714  }
715  sp->minimize();
716  l.setState(sp);
717  }
718  }
719  l.setInitialStateId(f.initialStateId());
720 }

722 void FlfReader::addScores(StaticLattice& l, ScoreId id, const Fsa::StaticAutomaton& f) {
723  require(l.size() == f.size());
724  for (Fsa::StateId fsaSid = 0; fsaSid < f.size(); ++fsaSid) {
725  const Fsa::State* fsaSp(f.fastState(fsaSid));
726  if (fsaSp) {
727  const Fsa::State* fsaSp(f.fastState(fsaSid));
728  State* sp(l.fastState(fsaSid));
729  require(sp);
730  if (fsaSp->isFinal()) {
731  require(sp->isFinal());
732  sp->weight_->set(id, Flf::Score(fsaSp->weight()));
733  }
734  require(sp->nArcs() == fsaSp->nArcs());
735  State::const_iterator a = sp->begin();
736  for (Fsa::State::const_iterator fsaA = fsaSp->begin(); fsaA != fsaSp->end(); ++fsaA, ++a) {
737  require(a->target() == fsaA->target());
738  a->setScore(id, Flf::Score(fsaA->weight()));
739  }
740  }
741  }

744 ConstLatticeRef FlfReader::read(FlfDescriptorRef desc, Core::Archive* archive) {
745  require(desc);
746  if (!partialKeys_.empty())
747  partial(desc, partialKeys_, false);
748  if (!appendKeys_.empty())
749  append(desc, appendKeys_, appendScales_);
750 
751  if (!desc->structureFile) {
752  warning("No structure file specified");
753  return ConstLatticeRef();
754  }
755  Fsa::StaticAutomaton structureFsa;
756  structureFsa.setSemiring(Fsa::TropicalSemiring);
757  structureFsa.setType(context_->fsaType());
758  if (context_->fsaType() == Fsa::TypeAcceptor)
759  structureFsa.setInputAlphabet(context_->inputAlphabetMap()->from());
760  if (context_->fsaType() == Fsa::TypeTransducer)
761  structureFsa.setOutputAlphabet(context_->outputAlphabetMap()->from());
762  {
763  std::string filename = Core::joinPaths(desc->latticeRoot, desc->structureFile.file);
764  InputStream is(filename, archive);
765  if (!is || !Fsa::read(&structureFsa, desc->structureFile.format, is)) {
766  warning("Could not read structure from fsa \"%s\"", filename.c_str());
767  return ConstLatticeRef();
768  }
769  if (!structureFsa.hasProperty(Fsa::PropertyAcyclic))
770  warning("Assume \"%s\" to be acyclic",
771  filename.c_str());
772  }
773  conform(desc, structureFsa, archive);
774 
775  log() << desc->info() << context()->info();
776 
777  StaticLattice* lat = new StaticLattice("static-lattice", context_->fsaType());
778  ConstLatticeRef l(lat);
779  if (!desc->id.empty())
780  lat->setDescription(desc->id);
781  lat->setSemiring(context_->semiring());
782  lat->setInputAlphabet(context_->inputAlphabetMap()->to());
783  if (context_->fsaType() != Fsa::TypeAcceptor)
784  lat->setOutputAlphabet(context_->outputAlphabetMap()->to());
785  lat->addProperties(Fsa::PropertyAcyclic | PropertyCrossWord);
786  addStructure(*lat, structureFsa);
787  {
788  Fsa::StaticAutomaton scoresFsa;
789  scoresFsa.setSemiring(Fsa::TropicalSemiring);
790  scoresFsa.setType(context_->fsaType());
791  if (context_->fsaType() == Fsa::TypeAcceptor)
792  scoresFsa.setInputAlphabet(context_->inputAlphabetMap()->from());
793  if (context_->fsaType() == Fsa::TypeTransducer)
794  scoresFsa.setOutputAlphabet(context_->outputAlphabetMap()->from());
795  for (ScoreId id = 0; id < ScoreId(desc->scoreFiles.size()); ++id)
796  if ((bool)desc->scoreFiles[id]) {
797  if (desc->scoreFiles[id] == desc->structureFile)
798  addScores(*lat, id, structureFsa);
799  else {
800  scoresFsa.setSemiring(structureFsa.semiring());
801  std::string filename = Core::joinPaths(desc->latticeRoot, desc->scoreFiles[id].file);
802  InputStream is(filename, archive);
803  if (!is || !Fsa::read(&scoresFsa, desc->scoreFiles[id].format, is))
804  warning("Could not read scores from fsa \"%s\"", filename.c_str());
805  else
806  addScores(*lat, id, scoresFsa);
807  scoresFsa.clear();
808  }
809  }
810  }
811  if ((bool)desc->boundariesFile) {
812  std::string filename = desc->boundariesFile(desc->latticeRoot);
813  InputStream is(filename, archive);
814  if (!is || !readBoundariesBinary(l, is)) {
815  warning("Could not read boundaries from \"%s\"", filename.c_str());
816  l->setBoundaries(InvalidBoundaries);
817  }
818  }
819  return l;
820 }

856 FlfDescriptorRef FlfWriter::buildDescriptor(
857  ConstLatticeRef f,
858  const std::string& descFilename) const {
859  FlfDescriptor* desc = new FlfDescriptor;
860  std::string basename = Core::baseName(descFilename);
861  std::string suffix = "";
862  if (basename == "-") {
863  warning("you try to write lattice in fsa-based format to standard out");
864  basename = "stdout";
865  }
866  else {
867  if (basename.find(".gz", basename.size() - 3) != std::string::npos) {
868  basename.erase(basename.size() - 3);
869  suffix = ".gz";
870  }
871  }
872  basename = Core::stripExtension(basename);
873  desc->latticeRoot = Core::directoryName(descFilename);
874  desc->semiringType = f->semiring()->type();
875  desc->hasInputAlphabet = true;
876  desc->inputAlphabetName = Lexicon::us()->alphabetName(
877  Lexicon::us()->alphabetId(f->getInputAlphabet()));
878  desc->inputAlphabetFile = FsaDescriptor(
879  "bin", Core::form("%s-input-alphabet.binfsa%s", basename.c_str(), suffix.c_str()));
880  if (f->type() != Fsa::TypeAcceptor) {
881  desc->hasOutputAlphabet = true;
882  desc->outputAlphabetName = Lexicon::us()->alphabetName(
883  Lexicon::us()->alphabetId(f->getOutputAlphabet()));
884  desc->outputAlphabetFile = FsaDescriptor(
885  "bin", Core::form("%s-output-alphabet.binfsa%s", basename.c_str(), suffix.c_str()));
886  }
887  else
888  desc->hasOutputAlphabet = false;
889  if (f->getBoundaries()->valid()) {
890  desc->hasBoundaries = true;
891  desc->boundariesFile = BoundariesDescriptor(Core::form("%s.binwb%s", basename.c_str(), suffix.c_str()));
892  }
893  else
894  desc->hasBoundaries = false;
895  desc->keys = f->semiring()->keys();
896  desc->scales = f->semiring()->scales();
897  desc->scoreFiles.resize(f->semiring()->size());
898  if (desc->scoreFiles.empty()) {
899  desc->structureFile.format = "bin";
900  desc->structureFile.file = Core::form("%s.binfsa%s", basename.c_str(), suffix.c_str());
901  }
902  else {
903  KeyList::const_iterator itKey = desc->keys.begin();
904  for (u32 i = 0; i < desc->scoreFiles.size(); ++i, ++itKey) {
905  desc->scoreFiles[i].format = "bin";
906  if (itKey->empty())
907  desc->scoreFiles[i].file = Core::form(
908  "%s-dim%d.binfsa%s", basename.c_str(), i, suffix.c_str());
909  else
910  desc->scoreFiles[i].file = Core::form(
911  "%s-%s.binfsa%s", basename.c_str(), itKey->c_str(), suffix.c_str());
912  }
913  desc->structureFile = desc->scoreFiles[0];
914  }
915  return FlfDescriptorRef(desc);
916 }

918 bool FlfWriter::writeDescriptor(FlfDescriptorRef desc,
919  const std::string& descFilename,
920  Core::Archive* archive) const {
921  require(!descFilename.empty() && (bool)desc->structureFile);
922  OutputStream os(descFilename, archive);
923  Core::XmlOutputStream xml(os.steal());
924  xml.generateFormattingHints(true);
925  xml.setIndentation(2);
926  {
927  Core::XmlOpen openLattice("lattice");
928  openLattice + Core::XmlAttribute("semiring", getSemiringTypeName(desc->semiringType));
929  if (!desc->id.empty())
930  openLattice + Core::XmlAttribute("id", desc->id);
931  xml << openLattice;
932  }
933  std::ostringstream oss;
934  oss << Core::timestamp;
935  xml << Core::XmlOpen("head")
936  << Core::XmlFull("creator", "RWTH-i6")
937  << Core::XmlFull("timestamp", oss.str())
938  << Core::XmlClose("head");
939  xml << Core::XmlEmpty("structure") + Core::XmlAttribute("format", desc->structureFile.format) + Core::XmlAttribute("file", desc->structureFile.file);
940  if (desc->hasInputAlphabet)
941  xml << Core::XmlEmpty((desc->hasOutputAlphabet ? "input-alphabet" : "alphabet")) + Core::XmlAttribute("name", desc->inputAlphabetName) + Core::XmlAttribute("format", desc->inputAlphabetFile.format) + Core::XmlAttribute("file", desc->inputAlphabetFile.file);
942  if (desc->hasOutputAlphabet)
943  xml << Core::XmlEmpty("output-alphabet") + Core::XmlAttribute("name", desc->outputAlphabetName) + Core::XmlAttribute("format", desc->outputAlphabetFile.format) + Core::XmlAttribute("file", desc->outputAlphabetFile.file);
944  if (desc->hasBoundaries)
945  xml << Core::XmlEmpty("boundaries") + Core::XmlAttribute("format", "bin") + Core::XmlAttribute("file", desc->boundariesFile.file);
946  xml << Core::XmlOpen("scores") + Core::XmlAttribute("n", desc->scoreFiles.size());
947  KeyList::const_iterator keyIt = desc->keys.begin();
948  ScoreList::const_iterator scaleIt = desc->scales.begin();
949  FsaDescriptorList::const_iterator fsaIt = desc->scoreFiles.begin();
950  for (ScoreId id = 0; id < desc->scoreFiles.size(); ++id, ++keyIt, ++scaleIt, ++fsaIt) {
951  xml << Core::XmlOpen("dim") + Core::XmlAttribute("id", id);
952  if (!keyIt->empty())
953  xml << Core::XmlFull("name", *keyIt);
954  if ((*scaleIt != Semiring::UndefinedScale) && (*scaleIt != Semiring::DefaultScale))
955  xml << Core::XmlFull("scale", *scaleIt);
956  if (*fsaIt)
957  xml << Core::XmlEmpty("fsa") + Core::XmlAttribute("format", fsaIt->format) + Core::XmlAttribute("file", fsaIt->file);
958  xml << Core::XmlClose("dim");
959  }
960  xml << Core::XmlClose("scores")
961  << Core::XmlClose("lattice");
962  return (bool)xml;
963 }

965 bool FlfWriter::write(ConstLatticeRef f,
966  FlfDescriptorRef desc,
967  const std::string& descFilename,
968  bool storeAlphabet,
969  Core::Archive* archive) const {
970  verify(desc);
971  ConstSemiringRef semiring = f->semiring();
972  if (partialAdd_) {
973  if (!writeDescriptor(desc, descFilename, archive))
974  return false;
975  partial(desc, partialKeys_, true);
976  }
977  else {
978  if (partial_)
979  partial(desc, partialKeys_, false);
980  if (!writeDescriptor(desc, descFilename, archive))
981  return false;
982  }
983  f = persistent(f);
984  if (desc->hasInputAlphabet && storeAlphabet) {
985  OutputStream os(Core::joinPaths(desc->latticeRoot, desc->inputAlphabetFile.file), archive);
986  if (!Fsa::write(toUnweightedFsa(f, Fsa::Weight(Semiring::One)), desc->inputAlphabetFile.format, os, Fsa::storeInputAlphabet))
987  return false;
988  }
989  if (desc->hasOutputAlphabet && storeAlphabet) {
990  OutputStream os(Core::joinPaths(desc->latticeRoot, desc->outputAlphabetFile.file), archive);
991  if (!Fsa::write(toUnweightedFsa(f, Fsa::Weight(Semiring::One)), desc->outputAlphabetFile.format, os, Fsa::storeOutputAlphabet))
992  return false;
993  }
994  if (desc->hasBoundaries && (bool)desc->boundariesFile) {
995  OutputStream os(Core::joinPaths(desc->latticeRoot, desc->boundariesFile.file), archive);
996  if (!writeBoundariesBinary(f, os))
997  return false;
998  }
999  bool embeddedStructure = false;
1000  FsaDescriptorList::const_iterator itScoreFile = desc->scoreFiles.begin();
1001  for (KeyList::const_iterator itKey = desc->keys.begin(), endKey = desc->keys.end();
1002  itKey != endKey; ++itKey, ++itScoreFile)
1003  if (*itScoreFile) {
1004  ScoreId dim = semiring->id(*itKey);
1005  verify(semiring->hasId(dim));
1006  OutputStream os(Core::joinPaths(desc->latticeRoot, itScoreFile->file), archive);
1007  if (!Fsa::write(toFsa(f, dim), itScoreFile->format, os, Fsa::storeStates))
1008  return false;
1009  if (*itScoreFile == desc->structureFile)
1010  embeddedStructure = true;
1011  }
1012  if (!embeddedStructure && (bool)desc->structureFile) {
1013  OutputStream os(Core::joinPaths(desc->latticeRoot, desc->structureFile.file), archive);
1014  if (!Fsa::write(toUnweightedFsa(f, Fsa::Weight(Semiring::One)), desc->structureFile.format, os, Fsa::storeStates))
1015  return false;
1016  }
1017  return true;
1018 }

1077 void FlfArchiveReader::setDescriptor(const Core::Configuration& config, FlfContextRef context) {
1078  ConstSemiringRef _semiring = semiring();
1079  require(_semiring);
1080 
1081  // get suffixes for boundaries, structure fsa, and score fsas
1082  boundariesSuffix_ = paramSuffix(select("boundaries"));
1083  suffixes_.resize(_semiring->size());
1084  const Core::Configuration structureConfig(config, "structure");
1085  structureSuffix_ = paramSuffix(structureConfig, "");
1086  std::string structureFormat = FsaDescriptor::paramFormat(structureConfig, "bin");
1087  StringList formats(_semiring->size());
1088  StringList::iterator itSuffix = suffixes_.begin();
1089  StringList::iterator itFormat = formats.begin();
1090  for (KeyList::const_iterator itKey = _semiring->keys().begin();
1091  itKey != _semiring->keys().end(); ++itKey, ++itSuffix, ++itFormat) {
1092  const Core::Configuration keyConfig(config, *itKey);
1093  *itSuffix = paramSuffix(keyConfig, "");
1094  *itFormat = FsaDescriptor::paramFormat(keyConfig, "bin");
1095  if (structureSuffix_.empty() && !itSuffix->empty()) {
1096  structureSuffix_ = *itSuffix;
1097  structureFormat = *itFormat;
1098  }
1099  }
1100  if (structureSuffix_.empty())
1101  criticalError("Discard flf-header, but no suffix for structure file given.");
1102 
1103  // create flf descriptor
1104  desc_ = FlfDescriptorRef(new FlfDescriptor());
1105  desc_->semiringType = _semiring->type();
1106  desc_->scales.resize(_semiring->size());
1107  ScoreList::iterator itDescScale = desc_->scales.begin();
1108  for (ScoreList::const_iterator itScale = semiring()->scales().begin();
1109  itScale != semiring()->scales().end(); ++itScale, ++itDescScale)
1110  *itDescScale = *itScale;
1111  desc_->keys.resize(_semiring->size());
1112  KeyList::iterator itDescKey = desc_->keys.begin();
1113  for (KeyList::const_iterator itKey = semiring()->keys().begin();
1114  itKey != semiring()->keys().end(); ++itKey, ++itDescKey)
1115  *itDescKey = *itKey;
1116  desc_->latticeRoot = "";
1117  desc_->structureFile = FsaDescriptor(structureFormat, "");
1118  desc_->hasInputAlphabet = false;
1119  desc_->hasOutputAlphabet = false;
1120  desc_->boundariesFile = BoundariesDescriptor();
1121  desc_->hasBoundaries = !boundariesSuffix_.empty();
1122  desc_->scoreFiles.resize(_semiring->size());
1123  FsaDescriptorList::iterator itFsa = desc_->scoreFiles.begin();
1124  for (StringList::const_iterator itFormat = formats.begin();
1125  itFormat != formats.end(); ++itFormat, ++itFsa)
1126  itFsa->format = *itFormat;
1127 
1128  // load alphabets
1129  // input alphabet, required -> acceptor
1130  const Core::Configuration inputAlphabetConfig(config, "input-alphabet");
1131  std::string inputAlphabetName = paramName(inputAlphabetConfig);
1132  FsaDescriptor inputAlphabetQf(inputAlphabetConfig);
1133  if (!inputAlphabetQf) {
1134  if (hasFile("input-alphabet.binfsa.gz"))
1135  inputAlphabetQf = FsaDescriptor("bin:input-alphabet.binfsa.gz");
1136  else if (hasFile("alphabets.binfsa.gz"))
1137  inputAlphabetQf = FsaDescriptor("bin:alphabets.binfsa.gz");
1138  else if (hasFile("alphabets.binfsa"))
1139  inputAlphabetQf = FsaDescriptor("bin:alphabets.binfsa");
1140  }
1141  if (inputAlphabetName.empty())
1142  inputAlphabetName = "lemma-pronunciation";
1143  if (inputAlphabetQf)
1144  context->loadInputAlphabet(inputAlphabetQf, archive, inputAlphabetName);
1145  else
1146  context->setInputAlphabet(Fsa::ConstAlphabetRef(), inputAlphabetName);
1147 
1148  //output alphabet, optional -> transducer
1149  const Core::Configuration outputAlphabetConfig(config, "output-alphabet");
1150  std::string outputAlphabetName = paramName(outputAlphabetConfig);
1151  FsaDescriptor outputAlphabetQf(outputAlphabetConfig);
1152  if (!outputAlphabetQf) {
1153  if (hasFile("output-alphabet.binfsa.gz"))
1154  outputAlphabetQf = FsaDescriptor("bin:output-alphabet.binfsa.gz");
1155  }
1156  if (outputAlphabetName.empty() && outputAlphabetQf)
1157  outputAlphabetName = "lemma-pronunciation";
1158  if (!outputAlphabetName.empty()) {
1159  if (outputAlphabetQf)
1160  context->loadOutputAlphabet(outputAlphabetQf, archive, outputAlphabetName);
1161  else
1162  context->setOutputAlphabet(Fsa::ConstAlphabetRef(), outputAlphabetName);
1163  }
1164 }
