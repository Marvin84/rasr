42 Score CombinedFeatureScorer::CombinedContextScorer::score(EmissionIndex e) const {
43  Score result = 0;
44  std::vector<Scorer>::const_iterator scorer = contextScorers_.begin();
45  std::vector<Scorer>::const_iterator lastScorer = contextScorers_.end();
46  MixtureIndexTableRow::const_iterator mixtureIndex = mixtureIndexTable_[e].begin();
47 
48  // negativ logarithms of probabilities are added
49  // -> we calculate the joined probability of the models
50  // assuming statistical independence
51  if (scorer != lastScorer) {
52  do {
53  result += (*scorer)->score(*mixtureIndex);
54  ++scorer;
55  ++mixtureIndex;
56  } while (scorer != lastScorer);
57  }
58  return result;
59 }

85 bool CombinedFeatureScorer::verifyMixtureIndexTable() {
86  for (EmissionIndex e = 0; e < nEmissions(); ++e) {
87  if (mixtureIndexTable_[e].size() != nModels())
88  return false;
89  }
90 
91  for (size_t i = 0; i < nModels(); ++i) {
92  MixtureIndex nMixtures = featureScorers_[i]->nMixtures();
93  for (EmissionIndex e = 0; e < nEmissions(); ++e) {
94  const MixtureIndexTableRow& l(mixtureIndexTable_[e]);
95  if ((l[i] < 0) || (l[i] >= nMixtures))
96  return false;
97  }
98  }
99  return true;
100 }
