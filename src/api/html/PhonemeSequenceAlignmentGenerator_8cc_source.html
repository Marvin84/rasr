86 bool PhonemeSequenceAlignmentGenerator::Cache::insert(const Key& key, Alignment* alignment) {
87  AlignmentMap::iterator itAlignment = alignments_.find(key);
88  if (itAlignment == alignments_.end()) {
89  alignments_[key] = alignment;
90  setDirty();
91  return true;
92  }
93  else if (itAlignment->second->score() != alignment->score()) {
94  delete itAlignment->second;
95  itAlignment->second = alignment;
96  setDirty();
97  return true;
98  }
99  return false;

102 void PhonemeSequenceAlignmentGenerator::Cache::initializeArchive(const ModelCombination& modelCombination) {
103  verify(!archive_);
104  Core::DependencySet dependencies;
105  modelCombination.getDependencies(dependencies);
106  /*! \todo missing dependency: pruning parameters */
107  acousticModel_ = modelCombination.acousticModel();
108 
109  if (!paramCache(config).empty()) {
110  if (labelType_ == Speech::Alignment::allophoneStateIds)
111  log("expecting an alignment archive with allophone-state-id labels");
112  else
113  log("expecting an alignment archive with emission-id labels");
114 
115  const bool readOnly = paramReadOnly(config);
116  Core::Archive::AccessMode accessMode = Core::Archive::AccessModeReadWrite;
117  if (readOnly) {

122  Core::ArchiveReader ar(*archive_, "DEPENDENCIES");
123  if (ar.isOpen()) {

125  if (!archiveDependencies.read(config, ar)) {

129  else if (archiveDependencies == dependencies) {
130  if (!readOnly) {
131  warning("Archive \"%s\" exists already and will be overwritten.",
132  archive_->path().c_str());
133  }
134  }
135  }

150 void PhonemeSequenceAlignmentGenerator::Cache::read(const std::string& segmentId) {
151  verify(archive_);
152  verify(!dirty());
153  Core::ArchiveReader reader(*archive_, segmentId);
154  if (reader.isOpen()) {
155  Core::BinaryInputStream i(reader);
156  u32 size;
157  Key key;
158  for (i >> size; i && alignments_.size() < size;) {
159  i >> key;
160  alignments_[key] = new Alignment();
161  alignments_[key]->setLabelType(labelType_);
162  i >> *alignments_[key];
163  }
164  if (!i)
165  criticalError("Failed to read observation-to-state alignments \"%s\" from cache \"%s\"",
166  segmentId.c_str(), archive_->path().c_str());
167 
168  u32 nScores = 0, nMatchingScores = 0;
169  for (i >> size; i && nScores < size; ++nScores) {
170  i >> key;
171  f32 score;
172  i >> score;
173  AlignmentMap::iterator itAlignment = alignments_.find(key);
174  if (itAlignment != alignments_.end()) {
175  itAlignment->second->setScore(score);
176  ++nMatchingScores;
177  }
178  }
179  if (nScores != nMatchingScores)
180  warning("Found %d scores without matching alignment; discard scores.", (nScores - nMatchingScores));
181  if (!i)
182  criticalError("Failed to read observation-to-state alignments \"%s\" from cache \"%s\"",
183  segmentId.c_str(), archive_->path().c_str());
184  }
185 }

187 void PhonemeSequenceAlignmentGenerator::Cache::write(const std::string& segmentId) {
188  verify(archive_);
189  verify(dirty());
190  if (archive_->hasAccess(Core::Archive::AccessModeWrite)) {
191  Core::ArchiveWriter writer(*archive_, segmentId, false);
192  if (writer.isOpen()) {
193  Core::BinaryOutputStream o(writer);
194  o << (u32)alignments_.size();
195  if (labelType_ == Alignment::allophoneStateIds) {
196  for (AlignmentMap::const_iterator it = alignments_.begin(); o && it != alignments_.end(); ++it)
197  o << it->first << *it->second;
198  }
199  else {
200  for (AlignmentMap::const_iterator it = alignments_.begin(); o && it != alignments_.end(); ++it) {
201  Alignment tmpAlignment(*it->second);
202  tmpAlignment.mapToEmissionIdLabels(acousticModel_);
203  o << it->first << tmpAlignment;
204  }
205  }
206  if (!o)
207  error("could not write state lattice \"%s\" to cache \"%s\"", segmentId.c_str(),
208  archive_->path().c_str());
209 
210  o << (u32)alignments_.size();
211  for (AlignmentMap::const_iterator it = alignments_.begin(); o && it != alignments_.end(); ++it)
212  o << it->first << (f32)it->second->score();
213  if (!o)
214  error("could not write state lattice \"%s\" to cache \"%s\"", segmentId.c_str(),
215  archive_->path().c_str());
216  }
217  else {
218  error("could not open file \"%s\" for writing in cache \"%s\"",
219  segmentId.c_str(), archive_->path().c_str());
220  }
221  }
222  else {
223  warning("could not write state lattice because archive is read only");
224  }
225  resetDirty();
226 }

280 const Alignment* PhonemeSequenceAlignmentGenerator::getAlignment(
281  const Bliss::Coarticulated<Bliss::LemmaPronunciation>& lemmaPronunciation,
282  TimeframeIndex tbeg, TimeframeIndex tend) {
283  timeval start, end;
284  timeval start2, end2;
285  TIMER_START(start);
286  const Key key(lemmaPronunciation, tbeg, tend);
287  const Alignment* result = 0;
288  if (!useAlignmentCache_ or !alignmentCache_->findForReadAccess(key, &result)) {
289  TIMER_START(start2);
290  aligner_.setModel(
291  modelAcceptorCache_->get(allophoneStateGraphBuilder_->createFunctor(
292  Bliss::Coarticulated<Bliss::Pronunciation>(
293  *lemmaPronunciation.object().pronunciation(),
294  lemmaPronunciation.leftContext(),
295  lemmaPronunciation.rightContext()))),
296  acousticModel());
297  std::vector<Mm::FeatureScorer::Scorer> scorers;
298  getScorers(tbeg, tend, scorers);
299  aligner_.feed(scorers);
300  Alignment* alignment = new Alignment;
301  aligner_.getAlignment(*alignment);
302 
303  verify(alignment->empty() or (alignment->score() != Core::Type<Score>::max));
304 
305  if (addEmissionScores_) {
306  addEmissionScores(*alignment, scorers);
307  }
308  TIMER_STOP(start2, end2, timeComputeAlignment_);
309  TIMER_START(start2);
310  alignment->addTimeOffset(tbeg);
311  if (useAlignmentCache_ and !alignmentCache_->insert(key, alignment)) {
312  defect();
313  }
314  result = alignment;
315  TIMER_STOP(start2, end2, timeWriteAlignment_);
316  }
317  /// TODO Make skipping optional (?)
318  // Eventually perform mapping of cached alignments if allophone state ids are stored (default)
319  if (result && (alignmentCache_->labelType() == Alignment::allophoneStateIds))
320  const_cast<Alignment*>(result)->setAlphabet(acousticModel()->allophoneStateAlphabet(), true);
321  TIMER_STOP(start, end, timeGetAlignment_);
322  return result;
323 }

360 Fsa::State* LatticeToAlignmentFsa::boundaryState(Fsa::StateId s) {
361  if (mapping_.find(s) == mapping_.end()) {

373 void LatticeToAlignmentFsa::insertAlignment(
374  const Alignment* alignment,
375  Fsa::StateId from,
376  const Fsa::State::const_iterator& arc) {
377  // assumption: Viterbi alignment
378  if (alignment->size() == 1) {
379  boundaryState(from)->newArc(boundaryState(arc->target())->id(), arc->weight(), alignment->front().emission);
380  }
381  else if (alignment->size() > 1) {
382  Fsa::State* sp = boundaryState(from);
383  Fsa::State* _sp = f_->newState();
384  Alignment::const_iterator a = alignment->begin();
385  sp->newArc(_sp->id(), arc->weight(), a->emission);
386  sp = _sp;
387  ++a;
388  for (; a != alignment->end() - 1; ++a) {
389  _sp = f_->newState();
390  sp->newArc(_sp->id(), arc->weight(), a->emission);
391  sp = _sp;
392  }
393  sp->newArc(boundaryState(arc->target())->id(), arc->weight(), a->emission);
394  }

397 void LatticeToAlignmentFsa::discoverState(Fsa::ConstStateRef sp) {
398  const TimeframeIndex begtime = wordBoundaries_->time(sp->id());
399  require(begtime != InvalidTimeframeIndex);
400  const Bliss::LemmaPronunciationAlphabet* alphabet =
401  required_cast(const Bliss::LemmaPronunciationAlphabet*, fsa_->getInputAlphabet().get());
402  for (Fsa::State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
403  const Alignment* alignment = 0;
404  const Bliss::LemmaPronunciation* pronunciation = alphabet->lemmaPronunciation(a->input());
405  const TimeframeIndex endtime = wordBoundaries_->time(a->target());
406  if (pronunciation) {
407  Bliss::Coarticulated<Bliss::LemmaPronunciation> coarticulatedPronunciation(
408  *pronunciation, wordBoundaries_->transit(sp->id()).final,
409  wordBoundaries_->transit(fsa_->getState(a->target())->id()).initial);
410  alignment = alignmentGenerator_->getAlignment(coarticulatedPronunciation, begtime, endtime);
411  }
412  else {
413  dummyAlignment_.resize(endtime - begtime, AlignmentItem(InvalidTimeframeIndex, Fsa::Epsilon));
414  alignment = &dummyAlignment_;
415  }
416  insertAlignment(alignment, sp->id(), a);
417  }
418 }

420 std::pair<Fsa::ConstAutomatonRef, Fsa::Weight> LatticeToAlignmentFsa::convert(
421  Lattice::ConstWordLatticeRef lattice) {
422  Fsa::Weight totalInv;
423  fsa_ = Fsa::cache(Fsa::posterior64(Fsa::changeSemiring(lattice->mainPart(), Fsa::LogSemiring),
424  totalInv));
425  if (fsa_->semiring()->compare(totalInv, fsa_->semiring()->zero()) == 0) {
426  Core::Application::us()->warning("no valid path in fsa");
427  }
428  wordBoundaries_ = lattice->wordBoundaries();
429  f_ = new Fsa::StaticAutomaton;
430  f_->setType(Fsa::TypeAcceptor);
431  f_->setSemiring(Fsa::TropicalSemiring);
432  f_->setInputAlphabet(alignmentGenerator_->allophoneStateAlphabet());
433  f_->setProperties(Fsa::PropertyAcyclic, Fsa::hasProperties(fsa_, Fsa::PropertyAcyclic) ? Fsa::PropertyAcyclic : 0);
434  f_->setDescription(std::string("alignment-fsa(") + fsa_->describe() + ")");
435  mapping_.clear();
436  dfs();
437  return std::make_pair(Fsa::ConstAutomatonRef(f_), totalInv);
438 }

459 void LatticeToAlignment::discoverState(Flf::ConstStateRef sp) {
460  const TimeframeIndex begtime = fsa_->boundary(sp->id()).time();
461  if (begtime == InvalidTimeframeIndex) {
462  return;
463  }
464  const Bliss::LemmaPronunciationAlphabet* alphabet = required_cast(const Bliss::LemmaPronunciationAlphabet*, fsa_->getInputAlphabet().get());
465  for (Flf::State::const_iterator arc = sp->begin(); arc != sp->end(); ++arc) {
466  const Bliss::LemmaPronunciation* pronunciation = alphabet->lemmaPronunciation(arc->input());
467  if (pronunciation) {
468  const TimeframeIndex endtime = fsa_->boundary(arc->target()).time();
469  Bliss::Coarticulated<Bliss::LemmaPronunciation> coarticulatedPronunciation(*pronunciation, fsa_->boundary(sp->id()).transit().final,
470  fsa_->boundary(arc->target()).transit().initial);
471  const Alignment* arcAlignment = alignmentGenerator_->getAlignment(coarticulatedPronunciation, begtime, endtime);
472  require(arcAlignment);
473  const Mm::Weight arcWeight = f32(fsa_->semiring()->project(arc->weight()));
474  for (Alignment::const_iterator in = arcAlignment->begin(); in != arcAlignment->end(); ++in) {
475  latticeAlignment_.push_back(AlignmentItem((*in).time, (*in).emission, arcWeight));
476  }
477  }
478  }
