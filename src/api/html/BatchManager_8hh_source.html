23 /**
24  * For performance-reasons this iterator should be used in a different way than standard iterators.
25  * It's underlying Data-Structure is much more flexible than the one used with the old iterators,
26  * but when used the right way these iterators provide about the same performance as the old ones.
27  * They only allow forward-iteration.
28  *
29  * usage-example:
30  * BatchIndexIterator <...,...> it;
31  *
32  * for(it = ...; it; ++it)
33  * {
34  * //use the iterator-value *it
35  * }
36  *
37  * WARNING: Unlike index iterators, pointer-iterators become invalid and may lead to crashes
38  * whenever the Array they iterate through is reallocated, so there should be no changes in the
39  * Structure for their lifetime.
40  */
41 
42 template<class BatchIdType, class NodeIdType, class NodeType, const BatchIdType invalidBatch_, const BatchIdType singleBatchMask_, const bool debug_ = false>

54  return batches_[batchId_ + 2];
55  }
56 
57  inline BatchIdType getNextBatch() {
58  if (batchId_ & singleBatchMask_)
59  return invalidBatch_;
60 
61  return (BatchIdType)batches_[batchId_ + 1];
62  }
63 
64  inline NodeIdType getBatchStartingNode() {
65  return batches_[batchId_];
66  }
67 
68  inline NodeType* getBorderPointer() {
69  return &(nodes_[getBorderNode()]);
70  }

76 public:
77  BatchPointerIterator(BatchIdType batchId, std::vector<NodeIdType>& batches, std::vector<NodeType>& values)
78  : batches_(batches), nodes_(values), batchId_(batchId), valid_(true) {
79  if (batchId_ == invalidBatch_) {
80  valid_ = false;
81  return;
82  };
83 
84  if (batchId & singleBatchMask_) {
85  current_ = batchId & (~singleBatchMask_);
86  currentBorder_ = current_ + 1;
87  return;
88  }
89 
90  currentBorder_ = getBorderPointer();
91  current_ = getBatchStartingPointer();
92  if (current_ == currentBorder_)
93  NextBatch();
94  }
95 
96  void NextBatch() {
97  batchId_ = getNextBatch();
98  if (batchId_ == invalidBatch_) {
99  current_ = NULL;
100  valid_ = false;
101  }
102  else {
103  current_ = getBatchStartingPointer();
104  currentBorder_ = getBorderPointer();
105  if (current_ == currentBorder_)
106  NextBatch();
107  }
108  }
109 
110  bool operator==(const BatchPointerIterator<BatchIdType, NodeIdType, NodeType, invalidBatch_, singleBatchMask_>& rhs) const {
111  return current_ == rhs.current_;
112  }

115  if (debug_)
116  verify(valid_);
117  ++current_;
118  if (current_ == currentBorder_) {
119  NextBatch();
120  }
121  return *this;
122  }
123 
124  inline NodeType* operator*() const {
125  if (debug_)
126  verify(valid_);
127  return current_;
128  }
129 
130  /** Returns the index of the current position in the array
131  */
132 
133  inline NodeIdType index() const {

138  return valid_;
139  }

142  ***Warning: Since all links have to be followed, this has a bad worst-case-performance. */
143  NodeIdType countToEnd() {
144  BatchIdType cid = batchId_;
145  NodeIdType count = currentBorder_ - current_;
146 
147  while ((cid = (BatchIdType)batches_[cid + 1]) != invalidBatch_) {
148  count += batches_[cid + 2] - batches_[cid];
149  }
150

159  BatchIdType batchId_; //The ID of the current Batch
160  NodeIdType currentBorder_;
161  NodeIdType current_;

174 public:
175  inline BatchIndexIterator(BatchIdType batchId, const std::vector<NodeIdType>& batches)
176  : batches_(&batches),
177  batchId_(batchId),
178  valid_(true) {
179  if (batchId & singleBatchMask_) {
180  current_ = batchId & (~singleBatchMask_);
181  currentBorder_ = current_ + 1;
182  return;
183  }
184 
185  if (batchId_ == invalidBatch_) {
186  valid_ = false;
187  return;
188  }
189 
190  currentBorder_ = getBorderNode();
191  current_ = getBatchStartingNode();
192  if (current_ == currentBorder_)
193  NextBatch();
194  }
195 
196  bool isLastBatch() const {
197  if (batchId_ & singleBatchMask_ || batchId_ == invalidBatch_)
198  return true;
199  return getNextBatch() == invalidBatch_;
200  }
201 
202  BatchIdType batchId() const {
203  return batchId_;
204  }
205 
206  inline void NextBatch() {
207  if (batchId_ & singleBatchMask_) {
208  valid_ = false;
209  return;
210  }
211 
212  batchId_ = getNextBatch();
213  if (batchId_ == invalidBatch_) {
214  valid_ = false;
215  }
216  else {
217  current_ = getBatchStartingNode();
218  currentBorder_ = getBorderNode();
219  if (current_ == currentBorder_)
220  NextBatch();
221  }
222  }
223  bool operator==(const BatchIndexIterator<BatchIdType, NodeIdType, invalidBatch_, singleBatchMask_>& rhs) const {
224  return current_ == rhs.current_;
225  }
226  inline BatchIndexIterator<BatchIdType, NodeIdType, invalidBatch_, singleBatchMask_>& operator++() {
227  if (debug_)
228  verify(valid_);
229  ++current_;
230  if (current_ == currentBorder_) {
231  NextBatch();
232  }
233  return *this;
234  }
235 
236  BatchIndexIterator<BatchIdType, NodeIdType, invalidBatch_, singleBatchMask_>& operator+=(u32 steps) {
237  while (steps) {
238  if (debug_)
239  verify(valid_);
240  u32 currentStep = steps;
241  if (currentBorder_ - current_ < currentStep)
242  currentStep = currentBorder_ - current_;
243  steps -= currentStep;
244  current_ += currentStep;
245  if (current_ == currentBorder_) {
246  NextBatch();
247  }
248  }
249  return *this;
250  }
251 
252  NodeIdType operator*() const {
253  if (debug_)
254  verify(valid_);
255  return current_;
256  }
257 
258  inline operator bool() const {
259  return valid_;
260  }
261 
262  inline bool ready() const {
263  return !valid_;
264  }

266  /**Counts the the steps from the current position until the end of the sequence is reached.
267  ***Warning: Since all links have to be followed, this has a bad worst-case-performance. */
268  NodeIdType countToEnd() {
269  if (!valid_)
270  return 0;
271  BatchIdType cid = batchId_;
272  NodeIdType count = currentBorder_ - current_;
273 
274  while ((cid & singleBatchMask_) == 0 && (cid = (BatchIdType)(*batches_)[cid + 1]) != invalidBatch_) {
275  count += (*batches_)[cid + 2] - (*batches_)[cid];
276  }
277 
278  return count;
279  }
280 
281  ///Returns the count of steps needed until the given node is reached. The node must be part of this batch.
282  u32 countUntil(NodeIdType until) {
283  verify(until >= current_);
284 
285  if (until < currentBorder_)
286  return until - current_; //Standard case
287 
288  BatchIdType cid = batchId_;
289 
290  NodeIdType count = currentBorder_ - current_;
291 
292  while ((cid = (BatchIdType)(*batches_)[cid + 1]) != invalidBatch_) {
293  if (until >= (*batches_)[cid] && until < (*batches_)[cid + 2])
294  return count + until - (*batches_)[cid]; //the node is contained by the current batch
295 
296  count += (*batches_)[cid + 2] - (*batches_)[cid];
297  }
298  verify(0); //Should not happen, as it must be in the batch
299 
300  return count;
301  }

483 /**
484  * invalidBatch_ is the index that is interpreted as "no batch".
485  * If mergeBatches_ is true, space is saved by using the end-index of one batch as the start-index of the next.
486  * In an ideal structure this saves about 1/3 of the batches-arrays size, but with mergeBatches_ batches can only grow,
487  * not shrink.
488  *
489  * Possible optimizations:
490  * The batch-chains have no pointer to their end, so the append-operation has to walk through all batches whenever a new batch is appended.
491  * For cases where the order doesn't count, it might be useful to add another operation that inserts a new batch into the chain at the second
492  * position(behind the first, so the ID of the first does not have to be changed, but without the need to walk through the whole chain)
493  *
494  * Explanation of the data-structure:
495  * Generally this can be used to represent sets of Values(of type NodeType) that belong together(batches), that are stored compactly and can be
496  * iterated efficiently.
497  */
498 
499 ///@param BatchIdType Type of batch indices (usually u32)
500 ///@param NodeIdType Type of node indices (usually u32)
501 ///@param NodeType Type of the contained nodes.
502 ///@param manageNodes Whether this manager should also manage the node vector.
503 /// If this is true, functions that change the node vector are allowed to be called.
504 /// If it is false, only functions that change the batches are allowed to be called.
505 ///@param invalidBatch_ Value of the "invalid" batch. Usually a special value like Core::Type<BatchIdType>::max()
506 ///@param singleBatchMask_ An optional mask to be used for marking "single" batches containing only one single item.
507 /// The batches batch-id will be directly the id of the node, marked with this mask. This saves
508 /// a lot of space when there is many batches consisting of only one item.
509 ///@param mergeBatches_ Whether batches should be merged. If this is true space is saved by using the end-index
510 /// of one batch as the start-index of the next. In an ideal structure this saves about 1/3
511 /// of the batches-arrays size, but then batches can only grow, not shrink.
512 template<class BatchIdType, class NodeIdType, class NodeType,
513  bool manageNodes, const BatchIdType invalidBatch_, const BatchIdType singleBatchMask_ = 0, const bool mergeBatches_ = true, const bool debug_ = false>
514 class BatchManager : private BatchBuilder<BatchIdType, NodeIdType, NodeType, invalidBatch_, singleBatchMask_> {
515 public:
516  typedef BatchManager<BatchIdType, NodeIdType, NodeType, invalidBatch_, mergeBatches_, debug_> mytype;
517  typedef mytype Self;

519  typedef typename SelfBuilder::Iterator Iterator;

523 private:
524  inline BatchIdType createNewBatch(NodeIdType from, NodeIdType to, BatchIdType append = invalidBatch_, BatchIdType appendTo = invalidBatch_, bool forceNormalBatch = false) {
525  if (singleBatchMask_ && !forceNormalBatch) {
526  if (appendTo == invalidBatch_ && append == invalidBatch_) {
527  //Eventually Create a single-batch
528  if (to == from + 1)
529  return ((BatchIdType)from) | singleBatchMask_;
530  }
531  }
532 
533  BatchIdType appendToLast = invalidBatch_;
534  if (appendTo != invalidBatch_)
535  appendToLast = this->getLastBatch(appendTo);
536 
537  BatchIdType ret = (BatchIdType)SelfBuilder::batches_.size();
538 
539  if (appendTo != invalidBatch_ && this->SelfBuilder::batches_[appendToLast + 2] == from) {
540  //The last value of the previous batch is the one before the first value of this batch(checked above),
541  //and the last batch is the one that this one should be appendet to.
542  //We can simply expand the previous batch.
543  SelfBuilder::batches_[appendToLast + 2] = to;
544  return appendTo;
545  }
546 
547  if (!mergeBatches_ || ret == 0 || SelfBuilder::batches_[ret - 1] != from) {
548  SelfBuilder::batches_.push_back(from);
549  }
550  else {
551  --ret; //The Node of "from" can be found at SelfBuilder::batches_[--ret]
552  }
553 
554  SelfBuilder::batches_.push_back(append);
555  SelfBuilder::batches_.push_back(to);
556 
557  if (debug_)
558  verify(append != ret);
559 
560  if (appendTo != invalidBatch_) {
561  SelfBuilder::appendToBatchPrivate(appendToLast, ret);
562  return appendTo;
563  }
564 
565  return ret;
566  }

673  /// @warning Only when this manager does not manage the nodes, or if history-recording is not used!
674  /// Appends the nodes in the range from @p from to @p to to the batch with id @p id.
675  /// Changes @p id to reflect the new content.
676  ///
677  /// Append/prepend-operations should always be used with ranges as big as possible, for performance-reasons.
