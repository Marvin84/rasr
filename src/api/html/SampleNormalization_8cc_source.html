87 bool SampleNormalization::update(const Sample* in) {
88  if (in) {
89  slidingWindow_.add(*in);
90 
91  Sample removed;
92  updateStatistics(in, (slidingWindow_.removed(removed) ? &removed : 0));
93  }
94  else
95  slidingWindow_.flushOut();
96 
97  Sample out;
98  if (slidingWindow_.out(out)) {
99  normalize(out);
100  out_.push_back(out);
101  return true;
102  }
103  return false;
104 }

106 void SampleNormalization::updateStatistics(const Sample* add, const Sample* remove) {
107  if (add) {
108  sum_ += *add;
109 
110  sumWeight_++;
111  changed_ = true;
112  }
113  if (remove) {
114  sum_ -= (*remove);
115 
116  sumWeight_--;
117  changed_ = true;
118  }
119 }

133 bool SampleNormalization::put(const Vector<Sample>& in) {
134  if (needInit_)
135  init();
136 
137  size_t nPendingSamples = out_.size() + slidingWindow_.futureSize();
138  Time bufferEndTime = outputStartTime_ + (Time)nPendingSamples / sampleRate_;
139  if (!in.equalsToStartTime(bufferEndTime)) {
140  if (nPendingSamples > 0)
141  return false;
142  outputStartTime_ = in.startTime();
143  }
144  for (u32 t = 0; t < in.size(); t++)
145  update(&in[t]);
146  return true;

161 bool SampleNormalization::flush(Vector<Sample>& out) {
162  if (needInit_)
163  init();
164 
165  while (update(0)) {
166  if (get(out))
167  return true;
