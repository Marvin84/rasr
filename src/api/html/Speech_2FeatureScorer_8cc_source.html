78 FeatureScorer::Weight FeatureScorer::featureScoreWeight(const Flow::Timestamp& featureTimestamp) {
79  if (weightScores_) {
80  Flow::DataPtr<Flow::DataAdaptor<Weight>> weight;
81  if (weightSource_->getData(weightPort_, weight)) {
82  if (featureTimestamp.contains(*weight)) {
83  if ((*weight)() >= 0)
84  return (*weight)();
85  else
86  criticalError("Weight (%f) smaller then zero.", (*weight)());
87  }
88 
89  criticalError("Timestamp of feature (%f-%f) does not contain the one of score-weight (%f-%f)",
90  featureTimestamp.startTime(), featureTimestamp.endTime(),
91  weight->startTime(), weight->endTime());
92  }
93  criticalError("Score-weight stream is shorter than the feature weight");
94  }
95  return 1;
96 }

111 void FeatureScorer::write() {
112  if (outputChannel_.isOpen()) {
113  outputChannel_.precision(paramPrecision(config));
114 
115  outputChannel_ << Core::XmlOpen("corpus-key-to-score-accumulator-map");
116  CorpusKeyToScoreMap::const_iterator i;
117  for (i = corpusKeyToScoreMap_.begin(); i != corpusKeyToScoreMap_.end(); ++i) {
118  outputChannel_ << Core::XmlOpen("score-accumulator") +
119  Core::XmlAttribute("corpus-key", i->first);
120  outputChannel_ << Core::XmlOpen("weighted-sum-of-scores");
121  outputChannel_ << std::setiosflags(std::ios::scientific) << i->second.weightedSum();
122  outputChannel_ << Core::XmlClose("weighted-sum-of-scores");
123  outputChannel_ << Core::XmlOpen("sum-of-weights");
124  outputChannel_ << std::setiosflags(std::ios::scientific) << i->second.sumOfWeights();
125  outputChannel_ << Core::XmlClose("sum-of-weights");
126  outputChannel_ << Core::XmlClose("score-accumulator");
127  }
128  outputChannel_ << Core::XmlClose("corpus-key-to-score-accumulator-map");
129  }
130  else {
131  criticalError("Could not dump corpus-key-to-score-map since channel \"output\" is not open.");
132  }
133 }
