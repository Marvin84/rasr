37 const WordBoundary WordBoundaries::getFinalWordBoundary() const {
38  Core::Vector<WordBoundary>::const_iterator it = internal_.begin();
39  for (; it != internal_.end(); ++it) {
40  if (it->valid()) {
41  break;
42  }
43  }
44  if (it == internal_.end()) {
45  return WordBoundary();
46  }
47  WordBoundary result = *it;
48  ++it;
49  for (; it != internal_.end(); ++it) {
50  if (it->valid() and (it->time() > result.time())) {
51  result = *it;
52  }
53  }
54  // final word boundary must be non-coarticulated
55  require(result.transit() == WordBoundary::Transit());
56  return result;
57 }

78 bool WordBoundaries::readBinary(Core::BinaryInputStream& bi) {
79  clear();
80 
81  char magic[Header::magicSize];
82  u32 version;
83  bi.read(magic, Header::magicSize);
84  if (!bi)
85  return false;
86  if (strncmp(magic, Header::magic, Header::magicSize) != 0) {
87  // backward compatibility: old format without any header
88 #if defined(BISANI)
89  version = 1;
90 #else
91  version = 2;
92 #endif
93  bi.seek(-Header::magicSize, std::ios::cur);
94  }
95  else {
96  if (!(bi >> version))
97  return false;
98  }
99 
100  u32 _size;
101  if (!(bi >> _size))
102  return false;
103  for (size_t i = 0; i < _size; ++i) {
104  u32 time;
105  if (!(bi >> time))
106  return false;
107  WordBoundary::Transit transit;
108  if (version >= 2) {
109  if (!transit.readBinary(bi))
110  return false;
111  }
112  else {
113  u32 transitId;
114  bi >> transitId;
115  }
116  push_back(WordBoundary(time, transit));
117  }
118 
119  return bi;
120 }

328  virtual Fsa::ConstStateRef getState(Fsa::StateId s) const {
329  Fsa::State* sp = new Fsa::State(s);
330  // merge arcs
331  const MergedStates& states = states_[s].mergedStates();
332  for (MergedStates::const_iterator it = states.begin(); it != states.end(); ++it) {
333  Fsa::ConstStateRef _sp = Fsa::SlaveAutomaton::fsa_->getState(*it);
334  sp->addTags(_sp->tags());
335  if (_sp->isFinal())
336  sp->weight_ = _sp->weight();
337  for (Fsa::State::const_iterator a = _sp->begin(); a != _sp->end(); ++a) {
338  Fsa::Arc* _a = sp->newArc();
339  *_a = *a;
340  _a->target_ = stateMap_[_a->target()];
341  }
342  }
343 
344  // remove duplicate entries
345  sp->sort(ByInputAndTarget());
346  if (sp->nArcs() > 1) {
347  Fsa::State::iterator ai = sp->begin() + 1, ao = sp->begin();
348  for (; ai != sp->end(); ++ai) {
349  if ((ai->input() != ao->input()) || (ai->target() != ao->target()))
350  *(++ao) = *ai;
351 #if 0
352  else
353  hope(semiring()->compare(ai->weight(), ao->weight()) == 0);
354 #endif
355  }

357  }
