84 bool SourceNode::configure() {
85  if (!(isFileOpen() || openFile()))
86  return false;
87 
88  SampleCount startSample = getStartSample();
89  if (!seek(startSample)) {
90  closeFile();
91  return false;
92  }
93  verify(sampleCount_ == startSample);
94 
95  Core::Ref<Flow::Attributes> a(new Flow::Attributes());
96  a->set("file", filename_);
97  a->set("sample-rate", sampleRate_);
98  a->set("sample-size", sampleSize_);
99  a->set("track-count", trackCount_);
100  switch (sampleSize_) {
101  case 8:
102  a->set("datatype", Flow::Vector<s8>::type()->name());
103  break;
104  case 16:
105  a->set("datatype", Flow::Vector<s16>::type()->name());
106  break;
107  case 32:
108  a->set("datatype", Flow::Vector<f32>::type()->name());
109  break;
110  default:
111  error("unsupported sample size: %d bit", sampleSize_);
112  return false;
113  }
114  if (totalSampleCount_ != invalidSampleCount) {
115  a->set("total-duration", Flow::Time(totalSampleCount_) / sampleRate_);
116  }
117  return putOutputAttributes(0, a);
118 }

159 bool SourceNode::work(Flow::PortId out) {
160  if (!isFileOpen())
161  return false;
162 
163  SampleCount endSample = getEndSample();
164  if (sampleCount_ >= endSample)
165  return putEos(0);
166 
167  u32 samplesToRead = (sampleCount_ + blockSize_ <= endSample) ? blockSize_ : endSample - sampleCount_;
168 
169  Flow::Timestamp* v = 0;
170  u32 samplesRead = read(samplesToRead, v);
171 
172  if (!v)
173  return putEos(0);
174 
175  v->setStartTime(Flow::Time(sampleCount_) / Flow::Time(sampleRate_));
176  sampleCount_ += samplesRead;
177  v->setEndTime(Flow::Time(sampleCount_) / Flow::Time(sampleRate_));
178 
179  ensure(v->startTime() >= startTime_ || Core::isAlmostEqual(v->startTime(), startTime_));
180  ensure(v->endTime() <= endTime_ || Core::isAlmostEqual(v->endTime(), endTime_));
181  return putData(0, v);
182 }
