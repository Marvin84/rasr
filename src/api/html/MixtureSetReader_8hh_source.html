105 template<class MixtureType>
106 Core::Ref<MixtureType> MixtureSetReader::read(const std::string& filename) const {
107  MixtureSetRef result;
108  Reader* reader = reader_->getObject(Core::filenameExtension(filename), config);
109  if (!reader) {
110  // create default reader
111  reader = new MixtureSetEstimatorReader(config);
112  }
113  if (!reader->read(filename, result))
114  error("Cannot read mixture set from file '%s'", filename.c_str());
115  delete reader;
116  return Core::Ref<MixtureType>(dynamic_cast<MixtureType*>(result.get()));
117 }
