135 inline void Aligner::SearchSpace::activateOrUpdateStateHypothesisViterbi(StateId state,
136  Score score,
137  Trace trace,
138  Am::AllophoneStateIndex emission) {
139  require(state < stateHypothesisForState_.size());
140  StateHypothesisIndex shi = stateHypothesisForState_[state];
141  if (shi >= StateHypothesisIndex(stateHypotheses_.size()) ||
142  shi < firstHypCurrentFrame_ ||
143  stateHypotheses_[shi].state != state) {
144  stateHypothesisForState_[state] = stateHypotheses_.size();
145  stateHypotheses_.push_back(StateHypothesis(state, score, trace, emission));
146  }
147  else {
148  require(shi < (StateHypothesisIndex)stateHypotheses_.size());
149  StateHypothesis& sh(stateHypotheses_[shi]);
150  if (sh.score >= score) {
151  sh.score = score;
152  require(sh.trace < (StateHypothesisIndex)stateHypotheses_.size());
153  require(trace < (StateHypothesisIndex)stateHypotheses_.size());
154  sh.trace = trace;
155  sh.emission = emission;
156  }
157  }
158 }

160 inline void Aligner::SearchSpace::activateOrUpdateStateHypothesisBaumWelch(StateId toState,
161  Score toStatePotential,
162  StateHypothesisIndex fromShi,
163  Am::AllophoneStateIndex emission,
164  Fsa::Weight arcScore) {
165  StateHypothesisIndex toShi = stateHypothesisForState_[toState];
166  Fsa::State* fromShs = traceback_->fastState(fromShi);
167  Fsa::State* toShs;
168  StateHypothesisIndex nHypotheses = stateHypotheses_.size();
169  if (toShi >= nHypotheses /* invalid */ ||
170  toShi < firstHypCurrentFrame_ /* not same time-frame */ ||
171  stateHypotheses_[toShi].state != toState /* invalid, likely toShi == 0, i.e. not yet set */) {
172  // We have the first hypothesis for this state in this time-frame.
173  toShs = new Fsa::State(nHypotheses);
174  traceback_->setState(toShs);
175  stateHypothesisForState_[toState] = nHypotheses;
176  stateHypotheses_.push_back(StateHypothesis(toState, toStatePotential, fromShi, emission)); // trace and emission unused
177  }
178  else {
179  StateHypothesis& sh(stateHypotheses_[toShi]);
180  sh.score = (Score)traceback_->semiring()->collect((Fsa::Weight)sh.score, (Fsa::Weight)toStatePotential);
181  sh.trace = fromShi; // unused in baum-welch training
182  sh.emission = emission; // also unused
183  toShs = traceback_->fastState(toShi);
184  }
185  ++stateHypotheses_[fromShi].nRefs;
186  toShs->newArc(fromShs->id(), arcScore, emission);
187 }

189 void Aligner::SearchSpace::addStartupHypothesis() {
190  require_(model_);
191  require_eq(stateHypotheses_.size(), 0);
192  stateHypotheses_.push_back(StateHypothesis(model_->initialStateId(), 0.0, invalidTrace, invalidTrace));
193  stateHypotheses_[0].nRefs = 1; // never delete the entry state hypothesis!
194  stateHypothesisForState_[model_->initialStateId()] = 0;
195  firstHypCurrentFrame_ = 0;
196  if (!viterbi_) {
197  traceback_->setType(Fsa::TypeAcceptor);
198  traceback_->setSemiring(Fsa::LogSemiring);
199  traceback_->setInputAlphabet(model_->getInputAlphabet());
200  traceback_->addProperties(Fsa::PropertyAcyclic);
201  Fsa::State* final = traceback_->newFinalState(traceback_->semiring()->one());
202  verify(final->id() == 0);
203  }
204 }

206 void Aligner::SearchSpace::expand(
207  Core::Ref<const Am::AcousticModel> acousticModel,
208  Mm::FeatureScorer::Scorer& emissionScores) {
209  require_(model_);
210  StateHypothesisIndex firstHypPreviousFrame = firstHypCurrentFrame_;
211  firstHypCurrentFrame_ = stateHypotheses_.size();
212  Core::Statistics<f32> variability("emission score variability");
213  for (StateHypothesisIndex shi = firstHypPreviousFrame; shi < firstHypCurrentFrame_; ++shi) {
214  const StateHypothesis& sh(stateHypotheses_[shi]);
215  const Fsa::State* s = model_->fastState(sh.state);
216  const Score score = sh.score;
217  Score minScore = Core::Type<Score>::max, maxScore = Core::Type<Score>::min;
218  for (Fsa::State::const_iterator a = s->begin(); a != s->end(); ++a) {
219  // all the scores in -log space
220  Score emissionScore = emissionScores->score(acousticModel->emissionIndex(a->input_));
221  Score transitionScore = Score(a->weight_);
222  minScore = std::min(minScore, emissionScore);
223  maxScore = std::max(maxScore, emissionScore);
224  if (viterbi_) {
225  Score sco = score + transitionScore + emissionScore;
226  activateOrUpdateStateHypothesisViterbi(a->target_, sco, shi, a->input_);
227  }
228  else { // baum-welch
229  Score arcScore = transitionScore + emissionScore;
230  activateOrUpdateStateHypothesisBaumWelch(a->target_, score + arcScore, shi, a->input_, (Fsa::Weight)arcScore);
231  }
232  }
233  if (logVariability_)
234  variability += (maxScore - minScore);
235  }
236  if (logVariability_)
237  log() << variability;
238  if (viterbi_) {
239  for (StateHypothesisIndex shi = firstHypCurrentFrame_; shi < (StateHypothesisIndex)stateHypotheses_.size(); ++shi)
240  ++stateHypotheses_[stateHypotheses_[shi].trace].nRefs;
241  }
242  ++time_;
243 }

254 void Aligner::SearchSpace::prune(Score threshold, u32 nBest) {
255  // maybe delete all except nBest first
256  if (nBest < nStateHypothesesCurFrame()) {
257  // Store all idx because we cannot simply reorder stateHypotheses_ in-place
258  // because of traceback handling in Baum-Welch.
259  std::vector<StateHypothesisIndex> nBestList(nStateHypothesesCurFrame());
260  for (size_t i = 0; i < nBestList.size(); ++i)
261  nBestList[i] = StateHypothesisIndex(i + firstHypCurrentFrame_);
262 
263  // This resorts the elements such that nBestList[0,nBest) have score <=
264  // the remaining elements. Because we are in -log space, these
265  // are the n-best elements then.
266  std::nth_element(
267  nBestList.begin(), nBestList.begin() + nBest, nBestList.end(),
268  [this](StateHypothesisIndex x, StateHypothesisIndex y) {
269  return stateHypotheses_[x].score < stateHypotheses_[y].score;
270  });
271 
272  // Set score of all remaining above threshold so that they get deleted in the next step.
273  for (auto it = nBestList.begin() + nBest; it != nBestList.end(); ++it) {
274  stateHypotheses_[*it].score = threshold + 1;
275  }
276  }
277 
278  StateHypothesisList::iterator in, out, in_end;
279  StateHypothesisIndex shi;
280  in = out = stateHypotheses_.begin() + firstHypCurrentFrame_;
281  in_end = stateHypotheses_.end();
282  shi = firstHypCurrentFrame_;
283 
284  if (viterbi_) {
285  for (; in != in_end; ++in, ++shi) {
286  if (in->score <= threshold) {
287  *(out++) = *in;
288  }
289  else {
290  deleteStateHypothesisViterbi(shi);
291  }
292  }
293  }
294  else { // baum-welch
295  for (; in != in_end; ++in, ++shi) {
296  if (in->score <= threshold) {
297  Fsa::StateId out_shi = out - stateHypotheses_.begin();
298  Fsa::State* s = traceback_->fastState(shi);
299  s->setId(out_shi);
300  traceback_->setState(s);
301  *(out++) = *in;
302  }
303  else {
304  deleteStateHypothesisBaumWelch(shi);
305  }
306  }
307  }
308  stateHypotheses_.erase(out, stateHypotheses_.end());
309 }

317 inline void Aligner::SearchSpace::deleteStateHypothesisBaumWelch(StateHypothesisIndex shi) {
318  std::queue<StateHypothesisIndex> hyps;
319  hyps.push(shi);
320  while (!hyps.empty()) {
321  shi = hyps.front();
322  hyps.pop();
323  if (--(stateHypotheses_[shi].nRefs) <= 0) {
324  Fsa::State* s = traceback_->fastState(shi);
325  for (Fsa::State::const_iterator a = s->begin(); a != s->end(); ++a) {
326  hyps.push(a->target());
327  }
328  }
329  }

332 void Aligner::SearchSpace::collectGarbage() const {
333  StateHypothesisList::iterator in, out, in_end;
334  StateHypothesisIndex shi_in, shi_out;
335  in = out = stateHypotheses_.begin();
336  in_end = stateHypotheses_.end();
337  shi_in = shi_out = 0;
338  if (viterbi_) {
339  StateHypothesisIndex* map = new StateHypothesisIndex[stateHypotheses_.size()];
340  for (; in != in_end; ++in, ++shi_in) {
341  if (in->nRefs > 0 || shi_in >= firstHypCurrentFrame_) {
342  *(out++) = *in;
343  map[shi_in] = shi_out;
344  ++shi_out;
345  }
346  }
347  firstHypCurrentFrame_ = (out - stateHypotheses_.begin()) - nStateHypothesesCurFrame();
348  stateHypotheses_.erase(out, stateHypotheses_.end());
349  in = stateHypotheses_.begin() + 1;
350  in_end = stateHypotheses_.end();
351  for (; in != in_end; ++in)
352  in->trace = map[in->trace];
353  delete[] map;
354  }
355  else { // baum-welch
356  for (StateHypothesisIndex shi = (StateHypothesisIndex)stateHypotheses_.size();
357  shi <= (StateHypothesisIndex)traceback_->maxStateId(); ++shi) {
358  traceback_->deleteState(shi);
359  }
360 
361  for (; in != in_end; ++in, ++shi_in) {
362  if (in->nRefs > 0 || shi_in >= firstHypCurrentFrame_) {
363  *(out++) = *in;
364  }
365  else {
366  traceback_->deleteState(shi_in);
367  }
368  }
369  firstHypCurrentFrame_ = (out - stateHypotheses_.begin()) - nStateHypothesesCurFrame();
370  stateHypotheses_.erase(out, stateHypotheses_.end());
371  traceback_->normalize();
372  ensure(traceback_->maxStateId() == stateHypotheses_.size() - 1);
373  }
374 }

376 Score Aligner::SearchSpace::finalStatePotential() const {
377  Score result = Core::Type<Score>::max;
378  for (StateHypothesisIndex shi = firstHypCurrentFrame_; shi < (StateHypothesisIndex)stateHypotheses_.size(); ++shi) {
379  const Fsa::State* s = model_->fastState(stateHypotheses_[shi].state);
380  if (s->isFinal()) {
381  const Score score = (Score)s->weight_ + stateHypotheses_[shi].score;
382  if (viterbi_) {
383  result = std::min(result, score);
384  }
385  else { // baum-welch
386  if (result == Core::Type<Score>::max) {
387  result = score;
388  }
389  else {
390  result = (Score)traceback_->semiring()->collect((Fsa::Weight)result, (Fsa::Weight)score);
391  }
392  }
393  }
394  }
395  return result;
396 }

398 Fsa::ConstAutomatonRef Aligner::SearchSpace::getAlignmentFsaViterbi() const {
399  StateHypothesisIndex bestHyp = invalidTrace;
400  Score bestScore = Core::Type<Score>::max;
401  for (StateHypothesisIndex shi = firstHypCurrentFrame_; shi < (StateHypothesisIndex)stateHypotheses_.size(); ++shi) {
402  const Fsa::State* s = model_->fastState(stateHypotheses_[shi].state);
403  if (s->isFinal()) {
404  const Score score = (Score)s->weight_ + stateHypotheses_[shi].score;
405  if (score < bestScore) {
406  bestScore = score;
407  bestHyp = shi;
408  }
409  }
410  }
411  Fsa::StaticAutomaton* result = new Fsa::StaticAutomaton();
412  result->setType(Fsa::TypeAcceptor);
413  result->setSemiring(Fsa::TropicalSemiring);
414  result->setInputAlphabet(model_->getInputAlphabet());
415  result->addProperties(Fsa::PropertyLinear | Fsa::PropertyAcyclic | Fsa::PropertySausages);
416  if (bestScore == Core::Type<Score>::max) {
417  //return empty automaton if alignment does not reach final state
418  return Core::ref(result);
419  }
420 
421  Fsa::State* s = new Fsa::State(time_);
422  s->setFinal(result->semiring()->one());
423  result->setState(s);
424  StateHypothesisIndex shi = bestHyp;
425  StateHypothesisIndex trace = stateHypotheses_[bestHyp].trace;
426  for (TimeframeIndex time = time_; time > 0; --time) {
427  ensure(trace != invalidTrace);
428  s = new Fsa::State(time - 1);
429  s->newArc(time, Fsa::Weight(stateHypotheses_[shi].score - stateHypotheses_[trace].score), stateHypotheses_[shi].emission);
430  result->setState(s);
431  shi = trace;
432  trace = stateHypotheses_[shi].trace;
433  }
434  result->setInitialStateId(0);
435  return Fsa::ConstAutomatonRef(result);
436 }

438 Fsa::ConstAutomatonRef Aligner::SearchSpace::getAlignmentFsaBaumWelch() const {
439  Fsa::State* final = traceback_->newState();
440  traceback_->setInitialStateId(final->id());
441  for (StateHypothesisIndex shi = firstHypCurrentFrame_; shi < (StateHypothesisIndex)stateHypotheses_.size(); ++shi) {
442  const Fsa::State* s = model_->fastState(stateHypotheses_[shi].state);
443  if (s->isFinal()) {
444  final->newArc(shi, s->weight_, Fsa::Epsilon);
445  }
446  }
447  Fsa::ConstAutomatonRef result = Fsa::transpose(tracebackRef_);
448  hope(result->hasProperty(Fsa::PropertyStorage | Fsa::PropertyAcyclic));
449  // otherwise wrap "staticCopy" around transpose or use a non-lazy implementation of transpose
450  traceback_->deleteState(final->id()); // undo changes in the traceback fsa to make this method 'const'
451  return result;
452 }

578 void Aligner::restart() {
579  ss_->clear();

594 void Aligner::feed(const std::vector<Mm::FeatureScorer::Scorer>& emissionScorers) {
595  Score previousScore = Core::Type<Score>::max;
596  // make a copy of the feature scorers HERE to reuse the cache over all pruning iterations
597  std::vector<Mm::FeatureScorer::Scorer> scorers = emissionScorers;
598  nIterations_ = 0;
599  for (Score t = minAcousticPruningThreshold_;
600  t <= maxAcousticPruningThreshold_;
601  t *= acousticPruningThresholdIncrementFactor_) {
602  nIterations_++;
603  restart();
604  acousticPruningThreshold_ = t;
605  std::vector<Mm::FeatureScorer::Scorer>::iterator s;
606  for (s = scorers.begin(); s != scorers.end(); ++s)
607  feed(*s);
608  if (acousticPruningThresholdIncrementFactor_ <= 1)
609  break;
610  Score score = alignmentScore();
611  if (reachedFinalState() and nStateHypotheses_.average() >= minAverageNumberOfStateHypotheses_) {
612  if (!increasePruningUntilNoScoreDifference_)
613  break;
614  if (t > minAcousticPruningThreshold_ and Core::isAlmostEqual(score, previousScore))
615  break;
616  }
617  previousScore = score;
618  }
619  if (logIterations_)
620  log("required number of alignment iterations: ") << nIterations_;
621 }

680 void Search::extractAlignment(Speech::Alignment& out, Fsa::ConstAutomatonRef alignmentPosteriorFsa, Mm::Weight minProbGT, Mm::Weight gamma) {
681  AlignmentExtractor ae(alignmentPosteriorFsa);
682  ae.extract(out); // we get scores in -log space
683  if (out.empty())
684  return; // unlikely but would happen if FSA was empty
685  out.combineItems(Fsa::LogSemiring);
686  require(!out.empty());
687  out.sortItems(false); // smallest -log-score means highest scores
688  out.clipWeights(0, Core::Type<Mm::Weight>::max);
689  out.multiplyWeights(gamma);
690  out.shiftMinToZeroWeights(); // more stable expm, is equivalent with normalizeWeights()
691  out.expm(); // to std space
692  out.normalizeWeights();
693  out.filterWeightsGT(minProbGT);
694 }

696 void Aligner::getAlignment(Alignment& result,
697  std::pair<Fsa::ConstAutomatonRef, Fsa::Weight> alignmentPosteriorFsa) const {
698  // The weights of resulting alignment are probabilities in std space.
699  // In case of Viterbi, they are normally all 1.0.
700  AlignmentExtractor ae(alignmentPosteriorFsa.first);
701  ae.extract(result); // we get scores in -log space
702  if (!viterbi_) {
703  result.combineItems(Fsa::LogSemiring);
704  result.sortItems(false);
705  result.clipWeights(0, Core::Type<Mm::Weight>::max);
706  result.filterWeights(0, getLogWeightThreshold());
707  if (!usePartialSums_) { // default case is use=false.
708  result.shiftMinToZeroWeights(); // more stable expm, is equivalent with normalizeWeights()
709  result.expm(); // to std space
710  result.normalizeWeights();
711  result.filterWeightsGT(0.0);
712  }
713  else { // weigths of alignment items are not normalized over a timeframe
714  result.addWeight(-f32(alignmentPosteriorFsa.second));
715  result.expm();
716  }
717  }
718  else {
719  // This addition is in std space, so it only make sense if we had 0 before. This should be the case.
720  result.addWeight(1);
721  }
722  for (Alignment::const_iterator it = result.begin(); it != result.end(); ++it)
723  statePosteriorStats_ += it->weight;
724  result.setScore(alignmentScore());
725  result.setAlphabet(acousticModel_->allophoneStateAlphabet());

757 std::pair<Fsa::ConstAutomatonRef, Fsa::Weight> Aligner::getAlignmentPosteriorFsa(
758  Fsa::ConstAutomatonRef alignmentFsa) const {

777  }

783 class Aligner::WordLatticeBuilder::AddWordBoundaryDisambiguatorsAutomaton : public Fsa::SlaveAutomaton {
784  typedef Fsa::SlaveAutomaton Precursor;
785 
786 protected:
787  mutable ConstStateRef lastState_;
788  Fsa::LabelId disambiguator_;
789  Fsa::Weight one_;
790  Fsa::StateId mapStateId(Fsa::StateId id) const {
791  if (id == Fsa::InvalidStateId)
792  return id;
793  return ((id & Fsa::StateIdMask) * 2) | (id & ~Fsa::StateIdMask);
794  }
795  Fsa::StateId mapStateIdInverse(Fsa::StateId id) const {
796  if (id == Fsa::InvalidStateId)
797  return id;
798  return ((id & Fsa::StateIdMask) / 2) | (id & ~Fsa::StateIdMask);
799  }
800  bool isDisambiguatorState(Fsa::StateId id) const {
801  return (id % 2 == 1);
802  }
803 
804 public:
805  AddWordBoundaryDisambiguatorsAutomaton(Fsa::ConstAutomatonRef f, Fsa::LabelId disambiguator)
806  : Precursor(f), disambiguator_(disambiguator), one_(f->semiring()->one()) {}
807 
808  virtual Fsa::StateId initialStateId() const {
809  return mapStateId(fsa_->initialStateId());
810  }
811 
812  virtual ConstStateRef getState(Fsa::StateId s) const {
813  Fsa::State* sp;
814  if (lastState_->refCount() == 1) {
815  sp = const_cast<State*>(lastState_.get());
816  *sp = *Precursor::fsa_->getState(mapStateIdInverse(s));
817  }
818  else {
819  sp = new State(*Precursor::fsa_->getState(mapStateIdInverse(s)));
820  lastState_ = ConstStateRef(sp);
821  }
822  sp->setId(s);
823  for (Fsa::State::iterator a = sp->begin(); a != sp->end(); ++a) {
824  a->target_ = mapStateId(a->target());
825  }
826  if (!isDisambiguatorState(s)) {
827  sp->newArc(s + 1, one_, disambiguator_);
828  }
829  return lastState_;
830  }

832  virtual std::string describe() const {
833  return std::string("add-word-boundary-disambiguators(") + fsa_->describe() + ")";
834  }

838  typedef Fsa::DfsState Precursor;
839 
840 private:

842  Lattice::WordBoundaries* wordBoundaries_; /**< word boundaries to build */
843  Am::ConstAllophoneStateAlphabetRef allophoneStateAlphabet_;

845  struct StateInfo {
846  TimeframeIndex time_; /**< current timeframe */
847  Fsa::LabelId wordLabelId_; /**< label of the current word */
848  Fsa::LabelId lastAllophone_; /**< last allophone found on the path */

850  void reset(TimeframeIndex time) {
851  time_ = time;
852  wordLabelId_ = Fsa::InvalidLabelId;
853  lastAllophone_ = Fsa::InvalidLabelId;
854  }
855  };
856  typedef Core::Vector<StateInfo> StateInfos;
857  StateInfos stateInfos_;
858 
859 protected:
860  Fsa::State* getState(Fsa::StateId id) {
861  require(id >= 0);
862  Fsa::StateRef sp = t_->state(id);
863  if (sp) {
864  return sp.get();
865  }
866  else {
867  Fsa::State* s = new Fsa::State(id);
868  t_->setState(s);
869  Fsa::ConstStateRef orig = fsa_->getState(id);
870  if (orig->isFinal()) {
871  s->setFinal(orig->weight());
872  }
873  return s;
874  }
875  }

877  void setCoarticulatedWordBoundary(Fsa::StateId s, TimeframeIndex time, Fsa::LabelId lastAllophone) {
878  Lattice::WordBoundary::Transit transit;
879  if (lastAllophone != Fsa::InvalidLabelId) {
880  Am::Phonology::SemiContext future = allophoneStateAlphabet_->allophoneState(lastAllophone).allophone()->future();
881  if (future.size()) {
882  transit.initial = allophoneStateAlphabet_->allophoneState(lastAllophone).allophone()->central();
883  transit.final = future[0];
884  }
885  }
886  wordBoundaries_->set(s, Lattice::WordBoundary(time, transit));
887  }

889  void exploreArc(Fsa::ConstStateRef fromState, const Fsa::Arc& a) {
890  Fsa::StateId from = fromState->id();
891  Fsa::StateId to = a.target();
892  stateInfos_.grow(to);
893  StateInfo& fromInfo = stateInfos_[from];
894  StateInfo& toInfo = stateInfos_[to];
895  Fsa::State* s = getState(from);
896  getState(to);
897 
898  if (a.output_ != Fsa::Epsilon) {
899  toInfo.wordLabelId_ = a.output_;
900  }
901  else {
902  toInfo.wordLabelId_ = fromInfo.wordLabelId_;
903  }
904 
905  if (a.input() == Fsa::Epsilon) {
906  toInfo.lastAllophone_ = fromInfo.lastAllophone_;
907  toInfo.time_ = fromInfo.time_;
908  s->newArc(to, a.weight(), Fsa::Epsilon);
909  }
910  else if (allophoneStateAlphabet_->isDisambiguator(a.input())) {
911  s->newArc(to, a.weight(), toInfo.wordLabelId_);
912  setCoarticulatedWordBoundary(a.target(), fromInfo.time_, fromInfo.lastAllophone_);
913  toInfo.reset(fromInfo.time_);
914  }
915  else {
916  toInfo.lastAllophone_ = a.input_;
917  toInfo.time_ = fromInfo.time_ + 1;
918  s->newArc(to, a.weight(), Fsa::Epsilon);
919  }
920  }

947
