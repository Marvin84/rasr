93 bool TensorflowForwardNode::work(Flow::PortId p) {
94  // check if computation is needed
95  require_lt(static_cast<size_t>(p), current_output_frame_.size());
96  if (current_output_frame_[p] >= outputs_[p].size() and not eos_) {
97  auto timer_start = std::chrono::steady_clock::now();
98  // gather inputs, for all input ports we gather all data and feed it to the tensorflow runtime in one go
99  size_t start_frame = timestamps_.size();
100  std::vector<std::deque<Flow::DataPtr<Flow::Timestamp>>> data(input_port_names_.size());
101  for (size_t i = 0ul; i < input_port_names_.size(); i++) {
102  bool success = true;
103  while (success) {
104  Flow::DataPtr<Flow::Timestamp> d;
105  success = getData(i, d);
106  if (success and Flow::Data::isNotSentinel(&(*d))) {
107  data[i].push_back(d);
108  }
109  if (i == 0ul) {
110  timestamps_.push_back(*d.get());
111  }
112  }
113  }
114  eos_ = true;
115 
116  // check if there is a non-empty stream
117  bool all_empty = true;
118  for (auto const& stream : data) {
119  all_empty = all_empty and stream.empty();
120  }
121  if (all_empty) {
122  return putData(p, Flow::Data::eos());
123  }
124 
125  std::vector<std::pair<std::string, Tensor>> inputs;
126  for (size_t i = 0ul; i < input_port_names_.size(); i++) {
127  auto const& input_name = input_port_names_[i];
128  auto const& tensor_info = tensor_input_map_.get_info(input_name);
129  inputs.push_back(std::make_pair(tensor_info.tensor_name(), toTensor(data[i])));
130  if (not tensor_info.seq_length_tensor_name().empty()) {
131  inputs.push_back(std::make_pair(tensor_info.seq_length_tensor_name(), Tensor::create(std::vector<s32>{static_cast<s32>(data[i].size())})));
132  }
133  }
134 
135  std::vector<Tensor> tf_output;
136  session_.run(inputs, output_tensor_names_, {}, tf_output);
137 
138  for (size_t i = 0ul; i < tf_output.size(); i++) {
139  appendToOutput(tf_output[i], start_frame, outputs_[i]);
140  }
141  auto timer_end = std::chrono::steady_clock::now();
142  log("flow fwd time: ") << std::chrono::duration<double, std::milli>(timer_end - timer_start).count();
143  }
144 
145  // the tensorflow graph is not required to have outputs of the same length, thus we have to check again here
146  if (current_output_frame_[p] >= outputs_[p].size()) {
147  return putData(p, Flow::Data::eos());
148  }
149  else {
150  return putData(p, outputs_[p][current_output_frame_[p]++]);
151  }
152 }

154 Tensor TensorflowForwardNode::toTensor(std::deque<Flow::DataPtr<Flow::Timestamp>> const& data) const {
155  if (data.front()->datatype() == Flow::Vector<f32>::type()) {
156  return vectorToTensor<f32>(data);
157  }
158  else if (data.front()->datatype() == Flow::Vector<f64>::type()) {
159  return vectorToTensor<f64>(data);
160  }
161  else if (data.front()->datatype() == Flow::Vector<s64>::type()) {
162  return vectorToTensor<s64>(data);
163  }
164  else if (data.front()->datatype() == Flow::Vector<u64>::type()) {
165  return vectorToTensor<u64>(data);
166  }
167  else if (data.front()->datatype() == Flow::Vector<s32>::type()) {
168  return vectorToTensor<s32>(data);
169  }
170  else if (data.front()->datatype() == Flow::Vector<u32>::type()) {
171  return vectorToTensor<u32>(data);
172  }
173  else {
174  criticalError("Unsupported input datatype: ") << *data.front()->datatype();
175  }
176 
177  return Tensor();
178 }

180 template<typename T>
181 Tensor TensorflowForwardNode::vectorToTensor(std::deque<Flow::DataPtr<Flow::Timestamp>> const& data) const {
182  Flow::Vector<T>* first = dynamic_cast<Flow::Vector<T>*>(data.front().get());
183  require(first != nullptr);
184 
185  std::vector<Math::FastMatrix<T>> batches;
186  batches.emplace_back(first->size(), data.size());
187  auto& matrix = batches.front();
188  for (size_t i = 0ul; i < data.size(); i++) {
189  Flow::Vector<T>* vec = dynamic_cast<Flow::Vector<T>*>(data[i].get());
190  require(first != nullptr);
191  require_eq(vec->size(), matrix.nRows());
192 
193  std::copy(vec->begin(), vec->end(), &matrix.at(0, i));
194  }
195 
196  Tensor tensor;
197  tensor.set(batches, true);
198  return tensor;
199 }

201 void TensorflowForwardNode::appendToOutput(Tensor const& tensor, size_t start_frame, std::deque<Flow::Data*>& data, size_t drop_left, size_t drop_right) const {
202  int num_dims = tensor.numDims();
203  std::string dt_name = tensor.dataTypeName();
204  require_eq(num_dims, 3); // TODO: support for one dimensional outputs
205  if (dt_name == "DT_FLOAT") {
206  appendVectorsToOutput<f32>(tensor, start_frame, data, drop_left, drop_right);
207  }
208  else if (dt_name == "DT_DOUBLE") {
209  appendVectorsToOutput<f64>(tensor, start_frame, data, drop_left, drop_right);
210  }
211  else if (dt_name == "DT_INT64") {
212  appendVectorsToOutput<s64>(tensor, start_frame, data, drop_left, drop_right);
213  }
214  else if (dt_name == "DT_UINT64") {
215  appendVectorsToOutput<u64>(tensor, start_frame, data, drop_left, drop_right);
216  }
217  else if (dt_name == "DT_INT32") {
218  appendVectorsToOutput<s32>(tensor, start_frame, data, drop_left, drop_right);
219  }
220  else if (dt_name == "DT_UINT32") {
221  appendVectorsToOutput<u32>(tensor, start_frame, data, drop_left, drop_right);
222  }
223  else {
224  criticalError("Unsupported output datatype: ") << dt_name;
225  }
226 }

228 template<typename T>
229 void TensorflowForwardNode::appendVectorsToOutput(Tensor const& tensor, size_t start_frame, std::deque<Flow::Data*>& data, size_t drop_left, size_t drop_right) const {
230  require_ge(tensor.dimSize(2), 0);
231  for (size_t t = drop_left; t + drop_right < static_cast<size_t>(tensor.dimSize(1)); t++) {
232  Flow::Vector<T>* vec = new Flow::Vector<T>(static_cast<size_t>(tensor.dimSize(2)));
233  tensor.get<T>(0ul, t, *vec);
234  vec->setTimestamp(timestamps_[std::min(start_frame + t - drop_left, timestamps_.size() - 1ul)]);
235  data.push_back(vec);
236  }
237 }

272 bool TensorflowOverlappingForwardNode::work(Flow::PortId p) {
273  // check if computation is needed
274  require_lt(static_cast<size_t>(p), current_output_frame_.size());
275 
276  if (current_output_frame_[p] >= outputs_[p].size() and not eos_) {
277  auto timer_start = std::chrono::steady_clock::now();
278  // gather inputs for all input ports, assume they are of same length
279  require_ge(timestamps_.size(), leftContextSize_);
280  size_t start_frame = timestamps_.size() - leftContextSize_;
281  while (featureBuffer_[0].size() < maxBufferSize_ and not eos_) {
282  Flow::Timestamp ts;
283  std::vector<Flow::DataPtr<Flow::Timestamp>> data;
284  for (size_t i = 0ul; i < input_port_names_.size(); i++) {
285  Flow::DataPtr<Flow::Timestamp> d;
286  bool success = getData(i, d);
287  if (success and Flow::Data::isNotSentinel(&(*d))) {
288  data.push_back(d);
289  }
290  if (i == 0ul) {
291  ts = *d.get();
292  }
293  }
294  if (data.size() == input_port_names_.size()) {
295  for (size_t i = 0ul; i < featureBuffer_.size(); i++) {
296  featureBuffer_[i].emplace_back(std::move(data[i]));
297  }
298  timestamps_.push_back(ts);
299  }
300  else {
301  eos_ = true;
302  }
303  }
304 
305  if (eos_) {
306  rightContextSize_ = 0u;
307  }
308  else {
309  rightContextSize_ = contextSize_;
310  }
311 
312  // check if there is data to process
313  if (featureBuffer_[0].empty()) {
314  return putData(p, Flow::Data::eos());
315  }
316 
317  std::vector<std::pair<std::string, Tensor>> inputs;
318  for (size_t i = 0ul; i < input_port_names_.size(); i++) {
319  auto const& input_name = input_port_names_[i];
320  auto const& tensor_info = tensor_input_map_.get_info(input_name);
321  inputs.push_back(std::make_pair(tensor_info.tensor_name(), toTensor(featureBuffer_[i])));
322  if (not tensor_info.seq_length_tensor_name().empty()) {
323  inputs.push_back(std::make_pair(tensor_info.seq_length_tensor_name(), Tensor::create(std::vector<s32>{static_cast<s32>(featureBuffer_[i].size())})));
324  }
325  }
326 
327  std::vector<Tensor> tf_output;
328  session_.run(inputs, output_tensor_names_, {}, tf_output);
329 
330  for (size_t i = 0ul; i < tf_output.size(); i++) {
331  appendToOutput(tf_output[i], start_frame, outputs_[i], leftContextSize_, rightContextSize_);
332  }
333 
334  leftContextSize_ = rightContextSize_;
335  for (size_t i = 0ul; i < featureBuffer_.size(); i++) {
336  while (featureBuffer_[i].size() > 2 * leftContextSize_) {
337  featureBuffer_[i].pop_front();
338  }
339  }
340 
341  auto timer_end = std::chrono::steady_clock::now();
342  log("flow fwd time: ") << std::chrono::duration<double, std::milli>(timer_end - timer_start).count();
343  }
344 
345  // the tensorflow graph is not required to have outputs of the same length, thus we have to check again here
346  if (current_output_frame_[p] >= outputs_[p].size()) {
347  return putData(p, Flow::Data::eos());
348  }
349  else {
350  return putData(p, outputs_[p][current_output_frame_[p]++]);
351  }
352 }
