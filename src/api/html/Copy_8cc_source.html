61 // deep copy by value, i.e. state and weights are copied
62 struct DeepCopyStateByValue {
63  ConstSemiringRef semiring;
64  DeepCopyStateByValue(ConstSemiringRef semiring)
65  : semiring(semiring) {}
66  State* operator()(ConstStateRef sr) const {
67  State* sp = new State(*sr);
68  if (sr->isFinal())
69  sp->setWeight(semiring->clone(sr->weight()));
70  for (State::iterator a = sp->begin(); a != sp->end(); ++a)
71  a->setWeight(semiring->clone(a->weight()));
72  return sp;
73  }
74  std::string describe(ConstLatticeRef l) const {
75  return Core::form("deepCopy(%s)", l->describe().c_str());
76  }

195 template<class CopyState>
196 std::pair<bool, bool> normalizeCopy_(ConstLatticeRef l, StaticLattice* staticLattice, StaticBoundaries* staticBoundaries) {
197  require(staticLattice);
198  if (!l || (l->initialStateId() == Fsa::InvalidStateId))
199  return std::make_pair(false, false);
200  staticLattice->clear();
201  if (staticBoundaries)
202  staticBoundaries->clear();
203  ConstBoundariesRef boundaries = l->getBoundaries();
204  if (!boundaries->valid())
205  staticBoundaries = 0;
206  CopyState copyState(l->semiring());
207  CachingMapBuilder stateMapping(l);
208  StateIdList& mapping = stateMapping.mapping();
209  copyStub(l, staticLattice, staticBoundaries, "normalized-" + copyState.describe(l));
210  staticLattice->setInitialStateId(mapping[l->initialStateId()]);
211  for (ConstStateRefList::reverse_iterator itSr = stateMapping.states().rbegin();
212  itSr != stateMapping.states().rend(); ++itSr) {
213  Fsa::StateId oldSid = (*itSr)->id();
214  State* sp = copyState(*itSr);
215  sp->setId(mapping[oldSid]);
216  for (State::iterator a = sp->begin(); a != sp->end(); ++a)
217  a->setTarget(mapping[a->target()]);
218  staticLattice->setState(sp);
219  if (staticBoundaries)
220  staticBoundaries->set(sp->id(), boundaries->get(oldSid));
221  }
222  return staticBoundaries ? std::make_pair(true, true) : std::make_pair(true, false);
223 }

298 class CopyNode : public FilterNode {
299 public:
300  static const Core::ParameterBool paramDeepCopy;
301  static const Core::ParameterBool paramTrim;
302  static const Core::ParameterBool paramNormalize;
303 
304 private:
305  bool deepCopy_;
306  bool trim_;
307  bool normalize_;
308 
309 protected:
310  virtual ConstLatticeRef filter(ConstLatticeRef l) {
311  if (!l || (l->initialStateId() == Fsa::InvalidStateId))
312  return l;
313  if (trim_ && normalize_) {
314  StaticLatticeRef trimmedLattice = StaticLatticeRef(new StaticLattice);
315  persistent(l, trimmedLattice.get(), 0);
316  trimInPlace(trimmedLattice);
317  trimmedLattice->setBoundaries(l->getBoundaries());
318  l = deepCopy_ ? normalizeDeepCopy(trimmedLattice) : normalizeCopy(trimmedLattice);
319  }
320  else if (trim_) {
321  StaticLatticeRef trimmedLattice = StaticLatticeRef(new StaticLattice);
322  if (deepCopy_)
323  deepCopy(l, trimmedLattice.get(), 0);
324  else
325  persistent(l, trimmedLattice.get(), 0);
326  trimInPlace(trimmedLattice);
327  trimmedLattice->setBoundaries(l->getBoundaries());
328  l = copyBoundaries(trimmedLattice);
329  }
330  else if (normalize_) {
331  l = deepCopy_ ? normalizeDeepCopy(l) : normalizeCopy(l);
332  }
333  else {
334  l = deepCopy_ ? deepCopy(l) : persistent(l);
335  }
336  return l;
337  }
338 
339 public:
340  CopyNode(const std::string& name, const Core::Configuration& config)
341  : FilterNode(name, config) {
342  deepCopy_ = paramDeepCopy(config);
343  trim_ = paramTrim(config);
344  normalize_ = paramNormalize(config);
345  }
346  virtual ~CopyNode() {}
