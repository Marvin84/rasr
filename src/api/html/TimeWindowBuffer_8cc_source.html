97 template<typename T>
98 bool TimeWindowBuffer<T>::flush(Flow::Vector<T>& out) {
99  if (needInit_) {
100  init();
101  verify(!needInit_);
102  }
103 
104  verify(buffer_.size() < 2 * std::max(shift_, length_));
105 
106  if (buffer_.empty())
107  return false;
108 
109  if (flushAll())
110  flushed_ = (shift_ >= buffer_.size());
111  else
112  flushed_ = (std::max(shift_, length_) >= buffer_.size());
113  copy(out, std::min(length_, u32(buffer_.size())));
114 
115  if (flushed())
116  needInit_ = true;
117  else {
118  buffer_.erase(buffer_.begin(), buffer_.begin() + shift_);
119 
120  bufferStartTime_ += (Time)shift_ / (Time)sampleRate_;
121  }
122 
123  return true;
124 }
