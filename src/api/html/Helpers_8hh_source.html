127 inline std::vector<float> parsePythonArray(const std::string& dp) {
128  std::vector<float> ret;
129  verify(dp[0] == '[');
130  int pos = 1;
131  while (pos < dp.length()) {
132  size_t next = dp.find(",", pos);
133  if (next == std::string::npos)
134  next = dp.find("]", pos);
135  verify(next != std::string::npos);
136 
137  if (pos == next)
138  break;
139 
140  double val = strtod(dp.c_str() + pos, 0);
141  ret.push_back(val);
142 
143  pos = next + 1;
144  }
145  return ret;

309  void match() {
310  rematch:
311  if (currentA_.start > currentA_.end || currentB_.start > currentB_.end)
312  pop();
313 
314  if (ready_ || a_[currentA_.start] == b_[currentB_.start])
315  return; // Match
316 
317  bool matched;
318  int splitStartA, splitStartB;
319 
320  /// @todo Think out some optimized inverted logic when lengthA > lengthB
321 
322  // Split and match from a
323  splitStartA = (currentA_.end - currentA_.start) / 2 + currentA_.start;
324 
325  typename std::vector<T>::const_iterator it = std::lower_bound(b_.begin() + currentB_.start, b_.begin() + currentB_.end + 1, a_[splitStartA]);
326  if (it == b_.end()) {
327  currentA_.end = splitStartA - 1;
328  goto rematch;
329  }
330 
331  matched = *it == a_[splitStartA];
332 
333  splitStartB = it - b_.begin();
334 
335  if (splitStartA == currentA_.start || splitStartB == currentB_.start) {
336  // The left split is empty, throw it away
337  if (matched)
338  currentA_.start = splitStartA; // Next round will be a match
339  else
340  currentA_.start = splitStartA + 1;
341  currentB_.start = splitStartB;
342  }
343  else if (splitStartA >= currentA_.end || (matched && splitStartB >= currentB_.end)) {
344  // The right split is empty, throw it away
345  if (matched) {
346  currentA_.end = splitStartA;
347  currentB_.end = splitStartB;
348  }
349  else {
350  // Discard last element, since we had a mismatch
351  currentA_.end = splitStartA - 1;
352  currentB_.end = splitStartB - 1;
353  }
354  }
355  else {
356  stack_.push_back(std::make_pair(currentA_, currentB_));
357  if (matched)
358  stack_.back().first.start = splitStartA;
359  else
360  stack_.back().first.start = splitStartA + 1;
361  stack_.back().second.start = splitStartB;
362  currentA_.end = splitStartA - 1;
363  currentB_.end = splitStartB - 1;
364  }
365 
366  goto rematch;
367  }
