168 void CorpusDescriptionParser::initSchema() {
169  XmlElement* conditionDesc = collect(new ConditionDescriptionElement(
170  this, ConditionDescriptionElement::handler(&Self::defineCondition)));
171  XmlElement* segmentConditionDesc = collect(new ConditionDescriptionElement(
172  this, ConditionDescriptionElement::handler(&Self::defineSegmentCondition)));
173 
174  XmlElement* condition = collect(new Core::XmlEmptyElementRelay(
175  "condition", this, startHandler(&Self::selectDefaultCondition)));
176  XmlElement* segmentCondition = collect(new Core::XmlEmptyElementRelay(
177  "condition", this, startHandler(&Self::selectSegmentCondition)));
178 
179  XmlElement* speakerDesc = collect(new SpeakerDescriptionElement(
180  this, SpeakerDescriptionElement::handler(&Self::defineSpeaker)));
181  XmlElement* segmentSpeakerDesc = collect(new SpeakerDescriptionElement(
182  this, SpeakerDescriptionElement::handler(&Self::defineSegmentSpeaker)));
183 
184  XmlElement* speaker = collect(new Core::XmlEmptyElementRelay(
185  "speaker", this, startHandler(&Self::selectDefaultSpeaker)));
186  XmlElement* segmentSpeaker = collect(new Core::XmlEmptyElementRelay(
187  "speaker", this, startHandler(&Self::selectSegmentSpeaker)));
188 
189  XmlElement* orth = collect(new OrthographyElement(
190  "orth", this, OrthographyElement::handler(&Self::setOrth)));
191 
192  XmlElement* leftContextOrth = collect(new OrthographyElement(
193  "left-context-orth", this, OrthographyElement::handler(&Self::setLeftContextOrth)));
194 
195  XmlElement* rightContextOrth = collect(new OrthographyElement(
196  "right-context-orth", this, OrthographyElement::handler(&Self::setRightContextOrth)));
197 
198  XmlElement* description_decl = collect(new XmlIgnoreElement("description", this)); // FFE
199 
200  XmlRegularElement* segment = new XmlRegularElementRelay(
201  "segment", this,
202  startHandler(&Self::startSegment),
203  endHandler(&Self::endSegment));
204  collect(segment);
205 
206  // The automaton we want to build is as follow:
207  // Condition/Speaker must apear before Orth, left and right Context may appear anywhere, but at most once
208  // The original automaton (before left/right context) consisted of 5 states, now we need to replicate it
209  // 4 times to cover the condition that neither left/right, only left/right and both left/right have been parsed
210  for (XmlRegularElement::State offset = 0; offset <= 15; offset += 5) {
211  segment->addTransition(offset + 0, offset + 0, description_decl);
212  segment->addTransition(offset + 0, offset + 1, segmentConditionDesc);
213  segment->addTransition(offset + 0, offset + 1, segmentCondition);
214  segment->addTransition(offset + 2, offset + 3, segmentConditionDesc);
215  segment->addTransition(offset + 2, offset + 3, segmentCondition);
216  segment->addTransition(offset + 0, offset + 2, segmentSpeakerDesc);
217  segment->addTransition(offset + 0, offset + 2, segmentSpeaker);
218  segment->addTransition(offset + 1, offset + 3, segmentSpeakerDesc);
219  segment->addTransition(offset + 1, offset + 3, segmentSpeaker);
220  segment->addTransition(offset + 0, offset + 4, orth);
221  segment->addTransition(offset + 1, offset + 4, orth);
222  segment->addTransition(offset + 2, offset + 4, orth);
223  segment->addTransition(offset + 3, offset + 4, orth);
224  }
225 
226  // now we connect the different states
227  for (XmlRegularElement::State s = 0; s < 5; s++) {
228  segment->addTransition(s, s + 5, leftContextOrth);
229  segment->addTransition(s, s + 10, rightContextOrth);
230  segment->addTransition(s + 5, s + 15, rightContextOrth);
231  segment->addTransition(s + 10, s + 15, leftContextOrth);
232  }
233 
234  // all states are final
235  for (XmlRegularElement::State s = 0; s < 20; s++) {
236  segment->addFinalState(s);
237  }
238 
239  segment->ignoreUnknownElements();
240 
241  XmlRegularElement* recording = new XmlRegularElementRelay(
242  "recording", this,
243  startHandler(&Self::startRecording),
244  endHandler(&Self::endRecording));
245  collect(recording);
246  recording->addTransition(0, 0, description_decl);
247  recording->addTransition(0, 0, conditionDesc);
248  recording->addTransition(0, 0, condition);
249  recording->addTransition(0, 0, speakerDesc);
250  recording->addTransition(0, 0, speaker);
251  recording->addTransition(0, 0, segment);
252  recording->addFinalState(0);
253  recording->ignoreUnknownElements();
254 
255  XmlElement* include_decl = collect(new XmlEmptyElementRelay(
256  "include", this,
257  startHandler(&Self::include)));
258 
259  XmlMixedElement* subcorpus = new XmlMixedElementRelay(
260  // should use XmlRegularElement, but cannot due to nesting restriction
261  "subcorpus", this,
262  startHandler(&Self::startSubcorpus),
263  endHandler(&Self::endSubcorpus),
264  0,
265  XML_NO_MORE_CHILDREN);
266  collect(subcorpus);
267  subcorpus->addChild(description_decl);
268  subcorpus->addChild(subcorpus);
269  subcorpus->addChild(include_decl);
270  subcorpus->addChild(conditionDesc);
271  subcorpus->addChild(condition);
272  subcorpus->addChild(speakerDesc);
273  subcorpus->addChild(speaker);
274  subcorpus->addChild(recording);
275 
276  XmlRegularElement* corpus = new XmlRegularElementRelay(
277  "corpus", this,
278  startHandler(&Self::startCorpus),
279  endHandler(&Self::endCorpus));
280  collect(corpus);
281  corpus->addTransition(0, 0, description_decl);
282  corpus->addTransition(0, 0, subcorpus);
283  corpus->addTransition(0, 0, include_decl);
284  corpus->addTransition(0, 0, conditionDesc);
285  corpus->addTransition(0, 0, condition);
286  corpus->addTransition(0, 0, speakerDesc);
287  corpus->addTransition(0, 0, speaker);
288  corpus->addTransition(0, 0, recording);
289  corpus->addFinalState(0);
290 
291  setRoot(corpus);
292 }

396 void CorpusDescriptionParser::startRecording(const XmlAttributes atts) {
397  const char* name = atts["name"];
398  const char* audio = atts["audio"];
399  const char* video = atts["video"];
400 
401  recording_ = new Recording(corpus_);
402 
403  if (!name) {
404  error("attribute 'name' missing in 'recording' tag");
405  }
406  else {
407  if (corpus_->isNameReserved(name)) {
408  error("recording \"%s\" already defined in the section", name);
409  }
410  else {
411  corpus_->reserveName(name);
412  recording_->setName(name);
413  }
414  }
415 
416  if (!audio && !video) {
417  error("At least one of attribute 'audio' or 'video' has "
418  "to be specified in 'recording' tag.");
419  }
420  else {
421  if (audio) {
422  std::string audioFilename(audio);
423  audioFilename = config.resolve(audioFilename);
424  audioFilename = joinPaths(audioDir_, audioFilename);
425  recording_->setAudio(audioFilename);
426  }
427  if (video) {
428  std::string videoFilename(video);
429  videoFilename = config.resolve(videoFilename);
430  videoFilename = joinPaths(videoDir_, videoFilename);
431  recording_->setVideo(videoFilename);
432  }
433  }
434 
435  recording_->duration_ = Core::Type<Time>::max;
436  currentSection_ = recording_;
437  segmentNum_ = 0;
438 
439  if (progressIndicator_) {
440  progressIndicator_->setTask(std::string("CorpusDescriptionParser processing recording '") + name + "'");
441  progressIndicator_->start();
442  }
443 
444  if (corpusVisitor_) {
445  corpusVisitor_->enterRecording(recording_);
446  }

449 void CorpusDescriptionParser::startSegment(const XmlAttributes atts) {
450  verify(recording_);
451 
452  const char* name = atts["name"];
453  const char* start = atts["start"];
454  const char* end = atts["end"];
455  const char* track = atts["track"];
456 
457  segment_ = new SpeechSegment(recording_);
458  ++segmentNum_;
459 
460  if (progressIndicator_) {
461  progressIndicator_->notify(segmentNum_);
462  }
463 
464  if (name) {
465  if (recording_->isNameReserved(name)) {
466  error("segment \"%s\" already defined in the section", name);
467  }
468  else {
469  recording_->reserveName(name);
470  segment_->setName(name);
471  }
472  }
473  else {
474  std::string autoName = form("%d", segmentNum_);
475  while (recording_->isNameReserved(autoName)) {
476  autoName += "+";
477  }
478  recording_->reserveName(autoName);
479  segment_->setName(autoName);
480  }
481  segment_->start_ = (start) ? atof(start) : 0.0;
482  segment_->end_ = (end) ? atof(end) : recording_->duration();
483  if (segment_->start() > segment_->end()) {
484  error("illegal segment boundary times");
485  }
486  segment_->track_ = (track) ? atoi(track) : 0;
487 }
