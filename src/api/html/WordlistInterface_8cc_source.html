102 HistoryHash WordlistInterfaceLm::hashKey(HistoryHandle hd) const {
103  const HistoryDescriptor* cd = (const HistoryDescriptor*)hd;
104 
105  u32 s = 0;
106  for (unsigned int i = 0; i < cd->length; i++) {
107  s = (s << 8) + cd->history[i];
108  u32 o = s & 0xff000000;
109  if (o) {
110  s = s ^ (o >> 24);
111  s = s ^ o;
112  }
113  }
114  return s;
