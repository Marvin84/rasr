37  if (!pruning) {
38  std::cout << "pruning missing in path-trace" << std::endl;

41 
42  if (pron && pron->lemma() && pron->lemma()->symbol()) {
43  component.log() << "Word identity:" << pron->lemma()->symbol();
44  component.log() << "Word pron length:" << pron->pronunciation()->length();
45  if (!pron->lemma()->hasEvaluationTokenSequence())
46  return; // Don't log tokens which are not evaluated
47  }
48 
49  for (std::map<std::pair<char const*, int>, Score>::iterator it = pruning->offsets.begin(); it != pruning->offsets.end(); ++it) {
50  if (it->first.second == -1)
51  component.log() << "Word " << it->first.first << ":" << it->second;
52  else
53  component.log() << "Word " << it->first.first << ": [" << it->first.second << "] " << it->second;
54  }
55  }

57  void maximizeOffset(char const* desc, Score offset, int index = -1) {
58  if (!pruning)
59  makeUnique();
60 
61  std::map<std::pair<char const*, int>, Score>::iterator it = pruning->offsets.find(std::make_pair(desc, index));
62  if (it == pruning->offsets.end()) {
63  makeUnique();
64  pruning->offsets.insert(std::make_pair(std::make_pair(desc, index), offset));
65  return;
66  }
67 
68  if (offset > it->second) {
69  makeUnique();
70  pruning->offsets[std::make_pair(desc, index)] = offset;
71  }
72  }
