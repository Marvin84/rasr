151 
152  Index operator[](const Index keyIndex) const {
153  require(alloState_.allophone());
154  const Allophone& allo(*alloState_.allophone());
155  switch (keyIndex) {
156  case 0:
157  return Index(alloState_.state());
158  case 1:
159  return Index(allo.boundary);
160  default:
161  Index contextIndex = keyIndex - 2;
162  if (contextIndex < Index(map_.historySize())) {
163  return (contextIndex < maxHistorySize_) ? Index(allo.history()[Index(map_.historySize()) - contextIndex - 1]) : map_.termIndex;
164  // map_.undefinedIndex;
165  }
166  contextIndex -= Index(map_.historySize());
167  if (contextIndex == 0) {
168  return Index(allo.central());
169  }
170  --contextIndex;
171  if (contextIndex < Index(map_.futureSize())) {
172  return (contextIndex < maxFutureSize_) ? Index(allo.future()[contextIndex]) : map_.termIndex;
173  // map_.undefinedIndex;
174  }
175  if (keyIndex < Index(map_.size()))
176  return (cond_) ? Index(cond_->hasCondition(map_.key(keyIndex))) : Index(false);
177  return map_.undefinedIndex;
178  }
179  }
