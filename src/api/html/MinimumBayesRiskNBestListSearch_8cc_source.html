65 void MinimumBayesRiskNBestListNaiveSearch::performSearch(Fsa::ConstAutomatonRef nBestList) {
66  nBestList = normalizeNbestlist(nBestList);
67 
68  evaluationSpaceSize_ = paramNumberHypothesesEvaluation(config);
69  summationSpaceSize_ = paramNumberHypothesesSummation(config);
70 
71  Fsa::ConstStateRef initialState = nBestList->getState(nBestList->initialStateId());
72  evaluationSpaceSize_ = std::min(evaluationSpaceSize_, initialState->nArcs());
73  summationSpaceSize_ = std::min(summationSpaceSize_, initialState->nArcs());
74  u32 nBestListSize = std::max(evaluationSpaceSize_, summationSpaceSize_);
75 
76  require(initialState->nArcs() > 0);
77  require(summationSpaceSize_ > 0);
78  require(evaluationSpaceSize_ > 0);
79 
80  // generate vector of string hypotheses
81  std::vector<StringHypothesis> hypotheses;
82  u32 n = 0;
83  Fsa::State::const_iterator arc = initialState->begin();
84  while (n < nBestListSize) {
85  Fsa::ConstAutomatonRef sentenceFsa = Fsa::partial(nBestList, arc->target());
86  std::vector<Fsa::LabelId> sentence;
87  Fsa::getLinearInput(sentenceFsa, sentence);
88  hypotheses.push_back(StringHypothesis(sentence, arc->weight()));
89  ++arc;
90  ++n;
91  }
92 
93  mapSentence_ = hypotheses.front().sentence_;
94  mapProbability_ = hypotheses.front().probability_;
95  mbrSentence_ = mapSentence_;
96  mbrProbability_ = mapProbability_;
97 
98  Fsa::Weight distanceOne;
99  Fsa::Weight maxDistanceOne;
100  mapRisk_ = posteriorRiskNBestList(mapSentence_, hypotheses, distanceOne, maxDistanceOne);
101  setDistanceOneCriterion(distanceOne, maxDistanceOne);
102  mbrRisk_ = mapRisk_;
103 
104  clog() << Core::XmlFull("map-probability", exp(-f32(mapProbability_)));
105  clog() << Core::XmlFull("map-risk", exp(-f32(mapRisk_)));
106 
107  Fsa::Weight risk = Fsa::LogSemiring->zero();
108  u32 mbrPosition = 0;
109 
110  Core::ProgressIndicator p("hypotheses", "");
111  p.start(evaluationSpaceSize_);
112 
113  /**
114  * If one of the criteria is fulfilled, then the searchcomputation does not have to be
115  * carried out, but the MAP sentence is guaranteed to minimize the Bayes risk w.r.t.
116  * Levenshtein loss. Only if neither of both criteria is fulfilled, the search will
117  * be carried out.
118  */
119  if (oneHalfCriterion()) {
120  clog() << Core::XmlFull("one-half-criterion", true);
121  clog() << Core::XmlFull("distance-one-criterion", distanceOneCriterion());
122  }
123  else if (distanceOneCriterion()) {
124  clog() << Core::XmlFull("one-half-criterion", false);
125  clog() << Core::XmlFull("distance-one-criterion", true);
126  }
127  else {
128  clog() << Core::XmlFull("one-half-criterion", false);
129  clog() << Core::XmlFull("distance-one-criterion", false);
130 
131  // MAP hypothesis has been accomplished, so we start with the second hypothesis
132  for (u32 n = 1; n < hypotheses.size(); ++n) {
133  StringHypothesis& hypothesis = hypotheses[n];
134  p.notify();
135  Fsa::Weight risk = posteriorRiskNBestList(hypothesis.sentence_, hypotheses, mbrRisk_);
136  if (Fsa::LogSemiring->compare(risk, mbrRisk_) > 0) {
137  mbrSentence_ = hypothesis.sentence_;
138  mbrProbability_ = hypothesis.probability_;
139  mbrRisk_ = risk;
140  mbrPosition = n;
141  }
142  }
143  }
144 
145  p.finish();
146 
147  bestAutomaton_ = Fsa::partial(nBestList,
148  (nBestList->getState(nBestList->initialStateId())->begin() + mbrPosition)->target());
149 
150  clog() << Core::XmlFull("mbr-risk", exp(-f32(mbrRisk_)));
151  clog() << Core::XmlFull("mbr-position", mbrPosition);
152 
153  MinimumBayesRiskSearch::performSearch(nBestList);
154 }

161 Fsa::Weight posteriorRiskNBestList(const std::vector<Fsa::LabelId>& trueSentence,
162  const std::vector<StringHypothesis>& hypotheses,
163  Fsa::Weight& distanceOne, Fsa::Weight& maxDistanceOne) {
164  Fsa::Weight result;
165 
166  maxDistanceOne = Fsa::LogSemiring->zero();
167 
168  // to be deleted!!!
169  Fsa::Accumulator* collectorResult = Fsa::LogSemiring->getCollector(Fsa::LogSemiring->zero());
170  // to be deleted!!!
171  Fsa::Accumulator* collectorDistanceOne = Fsa::LogSemiring->getCollector(Fsa::LogSemiring->zero());
172 
173  std::vector<StringHypothesis>::const_iterator hypothesis = hypotheses.begin();
174  for (; hypothesis != hypotheses.end(); ++hypothesis) {
175  u32 lDistance = levenshteinDistance(trueSentence, hypothesis->sentence_);
176  if (lDistance > 0) {
177  collectorResult->feed(
178  Fsa::LogSemiring->extend(
179  hypothesis->probability_,
180  Fsa::Weight(-std::log(f32(lDistance)))));
181  }
182  if (lDistance == 1) {
183  collectorDistanceOne->feed(hypothesis->probability_);
184  if (Fsa::LogSemiring->compare(hypothesis->probability_, maxDistanceOne) < 0) {
185  maxDistanceOne = hypothesis->probability_;
186  }
187  }
188  }
189 
190  result = collectorResult->get();
191  distanceOne = collectorDistanceOne->get();
192 
193  delete collectorResult;
194  delete collectorDistanceOne;
195 
196  return result;
197 }

199 Fsa::Weight posteriorRiskNBestList(const std::vector<Fsa::LabelId>& trueSentence,
200  const std::vector<StringHypothesis>& hypotheses,
201  const Fsa::Weight& pruningThreshold) {
202  Fsa::Weight result;
203 
204  // To be deleted!!!
205  Fsa::Accumulator* collectorResult = Fsa::LogSemiring->getCollector(Fsa::LogSemiring->zero());
206 
207  std::vector<StringHypothesis>::const_iterator hypothesis = hypotheses.begin();
208  for (; hypothesis != hypotheses.end(); ++hypothesis) {
209  u32 lDistance = levenshteinDistance(trueSentence, hypothesis->sentence_);
210  if (lDistance > 0) {
211  collectorResult->feed(Fsa::LogSemiring->extend(hypothesis->probability_,
212  Fsa::Weight(-std::log(f32(lDistance)))));
213  }
214  if (Fsa::LogSemiring->compare(collectorResult->get(), pruningThreshold) < 0) {
215  delete collectorResult;
216  return Fsa::Weight(Core::Type<f32>::min);
217  }
218  }
219 
220  result = collectorResult->get();
221  delete collectorResult;
222  return result;
223 }
