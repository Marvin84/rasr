95 bool FeatureScorerNode::putData(Mm::FeatureScorer::Scorer scorer) {
96  Flow::Vector<FeatureType>* out = NULL;
97  out = new Flow::Vector<FeatureType>(scorer->nEmissions());
98  for (u32 i = 0; i < scorer->nEmissions(); ++i) {
99  // A FeatureScorer returns the scores in -log space.
100  // This Flow node is expected to return the scores in +log space.
101  out->at(i) = -scorer->score(i);
102  }
103 
104  // It's always related to the *front* of timeStamps_, in case there was buffering, which refers to the oldest frame.
105  require(!timeStamps_.empty());
106  out->setTimestamp(timeStamps_.front());
107  timeStamps_.pop_front();
108 
109  // putData() will overtake out.
110  return Precursor::putData(0, out);
111 }

113 template<class T> // T is Flow::Vector<FeatureType> or Flow::TypedAggregate<Flow::Vector<FeatureType> >
114 bool FeatureScorerNode::work() {
115  // The FeatureScorer interface, without buffering (!isBuffered()):
116  // For every input, get a scorer via getScorer().
117  // With buffering (isBuffered()):
118  // Fill buffer until full (bufferFilled()). When we get more input, use getScorer().
119  // (The TrainerFeatureScorer has an infinite buffer, i.e. it's never full.)
120  // Then, while not bufferEmpty(), call flush() to get a scorer for each remaining frame.
121  // Thus, in every case, we can read as much input as we can here.
122 
123  while (true) {
124  // pull feature from incoming connections
125  Flow::DataPtr<T> ptrFeatures; // features from the flow network (single feature stream or aggregated features)
126  if (!getData(0, ptrFeatures)) {
127  require(ptrFeatures == Flow::Data::eos()); // or what else can it be?
128  break;
129  }
130  timeStamps_.push_back((Flow::Timestamp)*ptrFeatures.get());
131 
132  if (fs_->isBuffered() && !fs_->bufferFilled()) {
133  // fill buffer
134  fs_->addFeature(convertFeature(*ptrFeatures.get()));
135  }
136  else {
137  Mm::FeatureScorer::Scorer scorer = fs_->getScorer(convertFeature(*ptrFeatures.get()));
138  if (!putData(scorer))
139  return false;
140  }
141  }
142  // We get out of the loop only when there is no more input data.
143 
144  // In case of buffering, wait until we get all the input data,
145  // and only then get out the scores. That is how the FeatureScores expects it.
146  // In case of bidir-RNNs (e.g. via TrainerFeatureScorer via PythonTrainer),
147  // it's important that way, because the NN forward will happen at the first flush() call.
148  if (fs_->isBuffered()) {
149  while (!fs_->bufferEmpty()) {
150  Mm::FeatureScorer::Scorer scorer = fs_->flush();
151  if (!putData(scorer))
152  return false;
153  }
154  }
155 
156  require(timeStamps_.empty());
157  // There might be different behavior for different FeatureScorer's.
158  fs_->finalize(); // finalize this segment
159  fs_->reset(); // for the next round
160 
161  return Precursor::putData(0, Flow::Data::eos());
162 }
