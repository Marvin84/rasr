167 bool Node::putData(PortId out, Data* d) {
168  require(validOutputPort(out));
169  require(d != 0);
170 
171  if (dataChannel_.isOpen()) {
172  dataChannel_ << Core::XmlOpen("dump-data") + Core::XmlAttribute("node", fullName());
173  if (nOutputLinks(out) > 0)
174  d->dump(dataChannel_);
175  else
176  dataChannel_ << Core::XmlEmpty("dropped");
177  dataChannel_ << Core::XmlClose("dump-data");
178  }
179 
180  if (nOutputLinks(out) == 0) {
181  d->lock();
182  if (d->refCount() == 0)
183  d->free();
184  else
185  d->release();
186  return false;
187  }
188 
189  for (size_t i = 0; i < nOutputLinks(out); i++) {
190  if (!outputs_[out][i]->putData(d))
191  return false;
192  }
193  return true;
194 }

198 Core::Ref<const Attributes> Node::getInputAttributes(PortId in) {
199  Core::Ref<const Attributes> result;
200  Link* inputLink = inputs_[in];
201  if (inputLink != 0) {
202  if (inputLink->areAttributesAvailable()) {
203  result = inputLink->attributes();
204  }
205  else {
206  AbstractNode* predecessorNode = inputLink->getFromNode();
207  ensure(predecessorNode != 0);
208  if (predecessorNode->configure()) {
209  result = inputLink->attributes();
210  // If fails, from-node of the input link did not generate output attributes
211  // although configuration did not fail.
212  ensure(result);
213  }
214  else {
215  error() << "Configuration of node '" << predecessorNode->name() << "' failed.";
216  result = Core::ref(new Attributes());
217  }
218  }
219  }
220  else {
221  warning("Dead input port: %d.", in);
222  result = Core::ref(new Attributes());
223  }
224  return result;
225 }

237 bool Node::putOutputAttributes(PortId out, Core::Ref<const Attributes> a) {
238  require(validOutputPort(out));
239  require(a);
240 
241  if (attributesChannel_.isOpen()) {
242  attributesChannel_ << Core::XmlOpen("dump-attributes") + Core::XmlAttribute("node", fullName());
243  if (nOutputLinks(out) > 0)
244  attributesChannel_ << *a;
245  else
246  attributesChannel_ << Core::XmlEmpty("dropped");
247  attributesChannel_ << Core::XmlClose("dump-attributes");
248  }
249 
250  // link attributes to each outgoing link
251  for (std::vector<Link*>::iterator it = outputs_[out].begin();
252  it != outputs_[out].end(); it++) {
253  (*it)->setAttributes(a);
254  (*it)->configure();
255  (*it)->clear();
256  }
257  return true;
258 }

262 void Node::eraseOutputAttributes() {
263  for (std::vector<std::vector<Link*>>::iterator output = outputs_.begin();
264  output != outputs_.end(); ++output) {
265  for (std::vector<Link*>::iterator link = output->begin();
266  link != output->end(); ++link) {
267  // Recursion must not stop even if the link attribute is erased.
268  // Erased link does not obviously means that all the successor node output links are erased.
269  // E.g. using a cache node.
270  (*link)->eraseAttributes();
271  AbstractNode* successorNode = (*link)->getToNode();
272  if (successorNode != 0)
273  successorNode->eraseOutputAttributes();
274  }
275  }
276 }
