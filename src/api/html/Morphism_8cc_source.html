110 Fsa::ConstStateRef NaryMorphismAutomaton::getState(Fsa::StateId s) const {
111  if ((s >= mappedWordBoundaries_->size()) or !(*mappedWordBoundaries_)[s].valid()) {
112  if (s == fsa_->initialStateId()) {
113  mappedWordBoundaries_->set(s, WordBoundary(0, WordBoundary::Transit()));
114  }
115  else {
116  Fsa::StateId subS = Fsa::InvalidStateId;
117  u32 n = 0;
118  while (n < morphisms_.size()) {
119  subS = morphisms_[n]->map(s);
120  if (subS != Fsa::InvalidStateId) {
121  break;
122  }
123  ++n;
124  }
125  verify(subS != Fsa::InvalidStateId);
126  mappedWordBoundaries_->set(s, (*wordBoundaries_[n])[subS]);
127  }
128  }
129  return fsa_->getState(s);
130 }
