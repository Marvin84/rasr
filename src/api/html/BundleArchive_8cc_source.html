152 bool Core::BundleArchive::readIndex(const std::string& filename) {
153  CompressedInputStream fin(filename);
154  if (!fin)
155  return false;
156  u32 tmp;
157  fin >> tmp;
158  if (archiveFiles_.size() != tmp)
159  return false;
160  for (std::vector<std::string>::const_iterator f = archiveFiles_.begin(); f != archiveFiles_.end(); ++f) {
161  std::string buffer;
162  fin >> buffer;
163  if (*f != buffer)
164  return false;
165  }
166  fin >> tmp;
167  for (u32 i = 0; i < tmp; ++i) {
168  std::string buffer;
169  u32 n;
170  fin >> buffer >> n;
171  fileMap_[buffer] = n;
172  }
173  return true;
174 }

176 BundleArchive::ArchiveRef Core::BundleArchive::getArchive(const std::string& file) const {
177  StringHashMap<u32>::const_iterator idx = fileMap_.find(file);
178  if (idx == fileMap_.end()) {
179  return ArchiveRef();
180  }
181  ArchiveCache::const_iterator archive = archiveCache_.find(idx->second);
182  if (archive == archiveCache_.end()) {
183  ArchiveRef archive = Archive::create(config, archiveFiles_[idx->second], AccessModeRead);
184  verify(archive);
185  archiveCache_[idx->second] = ArchiveRef(archive);
186 
187  // Keep track of open archives and destroy the least recently used
188  // if maximal allowed number is exceeded.
189  lruQueue_.push(idx->second);
190  if (lruQueue_.size() > maxOpenFiles_) {
191  delete archiveCache_[lruQueue_.front()];
192  archiveCache_.erase(lruQueue_.front());
193  lruQueue_.pop();
194  }
195  }
196  return archiveCache_[idx->second];
197 }
