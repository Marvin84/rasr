80 void OrthographicParser::parse(const std::string& orth, Handler& handler) const {
81  require(Core::isWhitespaceNormalized(orth, Core::requireTrailingBlank));
82 
83  handler.initialize(this);
84 
85  std::string::size_type i, j, right_most;
86  std::string::size_type length = orth.length();
87  Bliss::OrthographicPrefixTree::ItemRange matchingLemmas;
88 
89  std::vector<Node> b_nodes(length + 1, 0);
90 
91  b_nodes[right_most = 0] = handler.newNode();
92  for (i = 0; i <= length; ++i) {
93  verify(i <= right_most);
94 
95  if (!b_nodes[i]) {
96  verify(i < right_most);
97  continue;
98  }
99 
100  for (int l = length - i; l >= 0; --l) {
101  matchingLemmas = prefixTree_.lookup(orth.c_str() + i, l);
102  if (l < 0)
103  continue;
104  j = i + l;
105  if (!b_nodes[j])
106  b_nodes[j] = handler.newNode();
107  if (right_most < j)
108  right_most = j;
109  Bliss::OrthographicPrefixTree::ItemLocation lemma;
110  for (lemma = matchingLemmas.first; lemma != matchingLemmas.second; ++lemma) {
111  const bool isSilenceLemma = (lemma->second == lexicon_->specialLemma("silence"));
112  // Note: the silence lemma will match for the empty orthography, i.e. l == 0, j == i.
113  if (allowForSilenceRepetitions_ || !isSilenceLemma || (j > i)) {
114  handler.newEdge(b_nodes[i], b_nodes[j], lemma->second);
115  }
116  else {
117  handler.newSilenceEdge(b_nodes[i], handler.newNode(), lemma->second, i == length);
118  }
119  }
120  }
121 
122  if (i == right_most && i < length) {
123  // forward the j
124  for (j = i + 1; j < length && orth[j] != utf8::blank; ++j) {
125  }
126  ++j; // one after the blank
127  verify(j <= length);
128  verify(!b_nodes[j]);
129  b_nodes[j] = handler.newNode();
130  verify(right_most < j);
131  right_most = j;
132  handler.newUnmatchableEdge(b_nodes[i], b_nodes[j], orth.substr(i, j - i));
133  }
134  }
135 
136  handler.finalize(b_nodes[0], b_nodes[length]);
137 }

141  Core::enforceTrailingBlank(orth);

173 void LemmaAcceptorBuilder::newSilenceEdge(
174  OrthographicParser::Node from,
175  OrthographicParser::Node to,
176  const Lemma* silenceLemma,
177  bool isFinal) {
178  Fsa::State* source = static_cast<Fsa::State*>(from);
179  Fsa::State* target = static_cast<Fsa::State*>(to);
180  require(target->nArcs() == 0);
181  for (Fsa::State::const_iterator a = source->begin(); a != source->end(); ++a) {
182  target->insert(target->end(), *a);
183  }
184  if (isFinal) {
185  target->setFinal(result_->semiring()->one());
186  }
187  source->newArc(target->id(), result_->semiring()->one(), (silenceLemma) ? silenceLemma->id() : Fsa::Epsilon);

200 Core::Ref<LemmaAcceptor> OrthographicParser::createLemmaAcceptor(const std::string& orth) const {
201  Core::Ref<LemmaAcceptor> result;
202  {
203  LemmaAcceptorBuilder lab;
204  parse(orth, lab);
205  result = lab.product();
206  }
207 
208  if (paramNormalizeLemmaSequenceScores(getConfiguration())) {
209  // if a word in the training corpus is a OOV and no unknown lemma is defined the resulting automaton will have
210  // an espilon arc and one unproductive state (for each OOV). We need to remove those to ensure the following
211  // operations work.
212  //Fsa::ConstAutomatonRef temp = Fsa::normalize(Fsa::removeEpsilons(Fsa::trim(result)));
213  //std::cerr << Fsa::staticCopy(temp)->size() << " " << result->size() << " " << std::endl;
214  Fsa::ConstAutomatonRef temp = Fsa::staticCompactCopy(Fsa::removeEpsilons(Fsa::trim(result)));
215  result->clear();
216  Fsa::copy(result._get(), temp);
217  Fsa::ConstSemiringRef semiring = Fsa::LogSemiring;
218  result->setSemiring(semiring);
219  if (allowForSilenceRepetitions_) {
220  error("not implemented at the moment: lemma acceptor with silence loop. set allow-for-silence-repetitions=false. you will anyway have tdp.silence.loop.");
221  return result;
222  }
223  // For each state, sum up all outgoing arc scores (and maybe final state score) and then normalize.
224  for (Fsa::StateId sid = 0; sid < result->size(); ++sid) {
225  auto& state = *result->fastState(sid);
226  // First some check if it is like we expect it to be.
227  if (state.nArcs() == 0)
228  verify(state.isFinal()); // Dead end not really expected.
229  for (auto& arc : state)
230  verify_ne(arc.target(), sid); // Loop?
231  // Now collect all scores.
232  auto* collector = semiring->getCollector();
233  if (state.isFinal())
234  collector->feed(state.weight());
235  for (auto& arc : state)
236  collector->feed(arc.weight());
237  Fsa::Weight collected = collector->get();
238  delete collector;
239  // Now normalize.
240  Fsa::Weight normFactor = semiring->invert(collected);
241  if (state.isFinal())
242  state.setWeight(semiring->extend(state.weight(), normFactor));
243  for (auto& arc : state)
244  arc.setWeight(semiring->extend(arc.weight(), normFactor));
245  }
246  }
247 
248  return result;
249 }
