62 void BatchFeatureScorerBase::initialize(const MixtureSet& mixtureSet) {
63  require(paddedDimension_ > 0);
64  nMixtures_ = mixtureSet.nMixtures();
65  offsets_.resize(nMixtures_ + 1, 0);
66  nDensities_ = 0;
67  for (size_t m = 0; m < nMixtures_; ++m) {
68  const Mixture& mixture = *mixtureSet.mixture(m);
69  offsets_[m] = nDensities_;
70  nDensities_ += mixture.nDensities();
71  }
72  offsets_[nMixtures_] = nDensities_;
73  Core::allocateAlignedVector<f32>(&scores_, nMixtures_ * bufferSize_, 0.0, 16);
74  cached_.resize(nMixtures_ * bufferSize_, false);
75 }

145 void BatchFloatFeatureScorer::init(const MixtureSet& mixtureSet) {
146  if (mixtureSet.nCovariances() != 1) {
147  criticalError("feature scorer supports only globally pooled covariance");
148  }
149  dimension_ = mixtureSet.dimension();
150  paddedDimension_ = ((dimension_ + BlockSize - 1) / BlockSize) * BlockSize;
151  initialize(mixtureSet);
152 
153  Core::allocateAlignedVector<f32>(&variance_, paddedDimension_, 0.0, 16);
154  CovarianceFeatureScorerElement covariance;
155  covariance = *mixtureSet.covariance(0);
156  std::copy(covariance.inverseSquareRootDiagonal().begin(),
157  covariance.inverseSquareRootDiagonal().end(),
158  variance_);
159  const float logNormFactor = covariance.logNormalizationFactor();
160 
161  Core::allocateAlignedVector<f32>(&features_, bufferSize_ * paddedDimension_, 0.0, 16);
162  Core::allocateAlignedVector<f32>(&means_, nDensities_ * paddedDimension_, 0.0, 16);
163  Core::allocateAlignedVector<f32>(&constants_, nDensities_, 0.0, 16);
164  for (size_t m = 0; m < nMixtures_; ++m) {
165  const Mixture& mixture = *mixtureSet.mixture(m);
166  f32* mean = means_ + offsets_[m] * paddedDimension_;
167  f32* c = constants_ + offsets_[m];
168  for (size_t dns = 0; dns < mixture.nDensities(); ++dns) {
169  const GaussDensity& density = *mixtureSet.density(mixture.densityIndex(dns));
170  verify(density.covarianceIndex() == 0);
171  const Mean& dnsMean = *mixtureSet.mean(density.meanIndex());
172  std::transform(dnsMean.begin(), dnsMean.end(), variance_, mean, std::multiplies<float>());
173  mean += paddedDimension_;
174  *c = logNormFactor - 2 * mixture.logWeight(dns);
175  ++c;
176  }
177  }
178 }

184 /**
185  * Get the min score for all densities of mixture e, using the
186  * given feature vectors.
187  */
188 template<class DensitySelector>
189 void BatchFloatFeatureScorer::fillScoreCacheTpl(EmissionIndex e, u32 featureIndex, u32 length, const DensitySelector& selector) const {
190  const size_t startIdx = featureIndex;
191  const size_t endIdx = featureIndex + length;
192  const size_t posOffset = e * bufferSize_;
193  for (size_t t = startIdx; t < endIdx; ++t) {
194  size_t pos = posOffset + (t % bufferSize_);
195  scores_[pos] = Core::Type<f32>::max;
196  cached_[pos] = true;
197  }
198  __m128 x1, x2, s1, s2, c;
199  const size_t endDns = offsets_[e + 1];
200  f32* scoreBase = scores_ + posOffset;
201  for (size_t dns = offsets_[e]; dns < endDns; ++dns) {
202  const f32* mean = means_ + dns * paddedDimension_;
203  const f32* dnsConst = constants_ + dns;
204  c = _mm_load_ss(dnsConst);
205  for (u32 t = startIdx; t < endIdx; ++t) {
206  const size_t rp = (t % bufferSize_);
207  if (!selector(rp, dns))
208  continue;
209  const f32* feature = features_ + (rp * paddedDimension_);
210  s1 = c;
211  s2 = _mm_setzero_ps();
212  for (int d = 0; d < paddedDimension_; d += BlockSize) {
213  x1 = _mm_sub_ps(_mm_load_ps(mean + d), _mm_load_ps(feature + d));
214  s1 = _mm_add_ps(s1, _mm_mul_ps(x1, x1));
215  x2 = _mm_sub_ps(_mm_load_ps(mean + d + 4), _mm_load_ps(feature + d + 4));
216  s2 = _mm_add_ps(s2, _mm_mul_ps(x2, x2));
217  }
218  s1 = _mm_add_ps(s1, s2);
219  s2 = s1;
220  s1 = _mm_shuffle_ps(s1, s2, _MM_SHUFFLE(1, 0, 3, 2));
221  s1 = _mm_add_ps(s1, s2);
222  s2 = s1;
223  s1 = _mm_shuffle_ps(s1, s2, _MM_SHUFFLE(2, 3, 0, 1));
224  s1 = _mm_add_ps(s1, s2);
225  f32* score = scoreBase + rp;
226  _mm_store_ss(score, _mm_min_ps(_mm_load_ss(score), s1));
227  }
228  }
229  for (size_t t = startIdx; t < endIdx; ++t) {
230  f32& s = *(scoreBase + (t % bufferSize_));
231  if (s < Core::Type<f32>::max)
232  s *= 0.5;
233  }
234 }

319 f32 BatchIntFeatureScorer::quantizationScale(const MixtureSet& mixtureSet) const {
320  require(variance_);
321  MeanType minMean = Core::Type<MeanType>::max;
322  MeanType maxMean = Core::Type<MeanType>::min;
323 
324  for (DensityIndex i = 0; i < mixtureSet.nDensities(); ++i) {
325  const GaussDensity* density = mixtureSet.density(i);
326  const std::vector<MeanType>* mean = mixtureSet.mean(density->meanIndex());
327 
328  for (ComponentIndex cmp = 0; cmp < mean->size(); ++cmp) {
329  MeanType dividedMean = (*mean)[cmp] * variance_[cmp];
330  minMean = std::min(minMean, dividedMean);
331  maxMean = std::max(maxMean, dividedMean);
332  }
333  }
334  int quantizedIntervalSize = (int)Core::Type<QuantizedType>::max - (int)Core::Type<QuantizedType>::min;
335  f32 intervalSize = 2 * std::max(Core::abs(minMean), Core::abs(maxMean));
336  return static_cast<f32>(quantizedIntervalSize) / (1.25 * intervalSize);
337 }

339 void BatchIntFeatureScorer::init(const MixtureSet& mixtureSet) {
340  if (mixtureSet.nCovariances() != 1) {
341  criticalError("feature scorer supports only globally pooled variance");
342  }
343  dimension_ = mixtureSet.dimension();
344  paddedDimension_ = ((dimension_ + BlockSize - 1) / BlockSize) * BlockSize;
345  initialize(mixtureSet);
346 
347  Core::allocateAlignedVector<f32>(&variance_, paddedDimension_, 0.0, 16);
348  CovarianceFeatureScorerElement covariance;
349  covariance = *mixtureSet.covariance(0);
350  std::copy(covariance.inverseSquareRootDiagonal().begin(),
351  covariance.inverseSquareRootDiagonal().end(),
352  variance_);
353 
354  f32 scale = quantizationScale(mixtureSet);
355  f32 scaleSquared = scale * scale;
356  scale_ = 2.0 * scaleSquared;
357  std::transform(variance_, variance_ + paddedDimension_, variance_,
358  std::bind2nd(std::multiplies<f32>(), scale));
359  const float logNormFactor = covariance.logNormalizationFactor() * scaleSquared;
360 
361  // allocate feature buffer and means in the same memory area.
362  Core::allocateAlignedVector<QuantizedType>(&features_, bufferSize_ * paddedDimension_ + nDensities_ * paddedDimension_, 0, 16);
363  means_ = features_ + bufferSize_ * paddedDimension_;
364  Core::allocateAlignedVector<s32>(&constants_, nDensities_, 0, 16);
365  for (size_t m = 0; m < nMixtures_; ++m) {
366  const Mixture& mixture = *mixtureSet.mixture(m);
367  QuantizedType* mean = means_ + offsets_[m] * paddedDimension_;
368  s32* c = constants_ + offsets_[m];
369  for (size_t dns = 0; dns < mixture.nDensities(); ++dns) {
370  const GaussDensity& density = *mixtureSet.density(mixture.densityIndex(dns));
371  verify(density.covarianceIndex() == 0);
372  const Mean& dnsMean = *mixtureSet.mean(density.meanIndex());
373  std::transform(dnsMean.begin(), dnsMean.end(), variance_, mean,
374  MultiplyAndQuantize());
375  mean += paddedDimension_;
376  *c = static_cast<s32>(logNormFactor - scale_ * mixture.logWeight(dns));
377  ++c;
378  }
379  }
380 }

391 static inline void addDistance(__m128i& x, __m128i& m, __m128i& sum) {
392  x = _mm_or_si128(_mm_subs_epu8(m, x), _mm_subs_epu8(x, m));
393  // x = | m - x | = (d15 d14 ... d0)
394  m = x;
395  x = _mm_unpackhi_epi8(x, _mm_setzero_si128());
396  // x = (d15 0 d14 0 d13 0 .. d8 0)
397  m = _mm_unpacklo_epi8(m, _mm_setzero_si128());
398  // m = (d7 0 d6 0 d5 0 .. d0 0)
399  x = _mm_madd_epi16(x, x);
400  // x = (d15 * d15 + d14 * d14, .. , d9 * d9 + d8 * d8)
401  m = _mm_madd_epi16(m, m);
402  // m = (d7 * d7 + d6 * d6, .. , d1 * d1 + d0 * d0)
403  // accumulate
404  sum = _mm_add_epi32(sum, x);
405  sum = _mm_add_epi32(sum, m);
406 }

408 static inline s32 horizontalAdd(__m128i& sum) {
409  // sum = (s3 s2 s1 s0)
410  __m128i s = _mm_add_epi32(
411  // (s1 s0 s1 s0)
412  _mm_shuffle_epi32(sum, _MM_SHUFFLE(3, 2, 3, 2)),
413  // (s3 s2 s3 s2)
414  _mm_shuffle_epi32(sum, _MM_SHUFFLE(1, 0, 1, 0)));
415  // s = (s1+s3 s0+s2 s1+s3 s0+s2)
416  return _mm_cvtsi128_si32(_mm_add_epi32(s,
417  // (s0+s3 s1+s2 s0+s2 s1+s3)
418  _mm_shuffle_epi32(s, _MM_SHUFFLE(2, 3, 0, 1))));
419  // = (s0+s1+s2+s3)
420 }
421 } // namespace

423 template<class DensitySelector>
424 void BatchIntFeatureScorer::fillScoreCacheTpl(EmissionIndex e, u32 featureIndex, u32 length, const DensitySelector& selector) const {
425  const size_t startIdx = featureIndex;
426  const size_t endIdx = featureIndex + length;
427  const size_t posOffset = e * bufferSize_;
428  for (size_t t = startIdx, i = 0; t < endIdx; ++t, ++i) {
429  cached_[posOffset + (t % bufferSize_)] = true;
430  }
431  f32* scoreBase = scores_ + posOffset;
432  const size_t startDns = offsets_[e];
433  const size_t endDns = offsets_[e + 1];
434  const QuantizedType* meanStart = means_ + startDns * paddedDimension_;
435  const s32* dnsConstStart = constants_ + startDns;
436  for (u32 t = startIdx; t < endIdx; ++t) {
437  const size_t rp = (t % bufferSize_);
438  selector.seek(rp, startDns);
439  const QuantizedType* feature = features_ + (rp * paddedDimension_);
440  const QuantizedType* mean = meanStart;
441  _mm_prefetch(const_cast<char*>(reinterpret_cast<const char*>(mean)), _MM_HINT_T1);
442  const s32* dnsConst = dnsConstStart;
443  s32 best = 2147483647;
444  for (size_t dns = startDns; dns < endDns; ++dns) {
445  if (selector.value()) {
446  __m128i sum = _mm_setzero_si128();
447  for (int d = 0; d < paddedDimension_; d += BlockSize) {
448  __m128i m, x;
449  m = _mm_load_si128(reinterpret_cast<const __m128i*>(mean + d));
450  // m = mean[d .. d+15]
451  x = _mm_load_si128(reinterpret_cast<const __m128i*>(feature + d));
452  // x = feature[d .. d+15]
453  addDistance(x, m, sum);
454  }
455  {
456  s32 tmp = horizontalAdd(sum);
457  tmp += *dnsConst;
458  if (tmp < best)
459  best = tmp;
460  }
461  }
462  mean += paddedDimension_;
463  ++dnsConst;
464  selector.next();
465  }
466  f32* result = scoreBase + rp;
467  _mm_prefetch(const_cast<char*>(reinterpret_cast<const char*>(result)), _MM_HINT_NTA);
468  *result = static_cast<f32>(best) / scale_;
469  }
470 }

558 void BatchUnrolledIntFeatureScorer::fillScoreCache(EmissionIndex e, u32 featureIndex, u32 length) const {
559  const size_t startIdx = featureIndex;
560  const size_t endIdx = featureIndex + length;
561  const size_t posOffset = e * bufferSize_;
562  for (size_t t = startIdx, i = 0; t < endIdx; ++t, ++i) {
563  cached_[posOffset + (t % bufferSize_)] = true;
564  }
565  f32* scoreBase = scores_ + posOffset;
566  const size_t startDns = offsets_[e];
567  const size_t endDns = offsets_[e + 1];
568  const QuantizedType* meanStart = means_ + startDns * paddedDimension_;
569  const s32* dnsConstStart = constants_ + startDns;
570  for (u32 t = startIdx; t < endIdx; ++t) {
571  const size_t rp = (t % bufferSize_);
572  const QuantizedType* feature = features_ + (rp * paddedDimension_);
573  const QuantizedType* mean = meanStart;
574  _mm_prefetch(const_cast<char*>(reinterpret_cast<const char*>(mean)), _MM_HINT_T1);
575  const s32* dnsConst = dnsConstStart;
576  s32 best = 2147483647;
577  for (size_t dns = startDns; dns < endDns; ++dns) {
578  __m128i sum = _mm_setzero_si128();
579  {
580  __m128i m1, m2, m3, x1, x2, x3;
581  m1 = _mm_load_si128(reinterpret_cast<const __m128i*>(mean));
582  x1 = _mm_load_si128(reinterpret_cast<const __m128i*>(feature));
583  m2 = _mm_load_si128(reinterpret_cast<const __m128i*>(mean + BlockSize));
584  x2 = _mm_load_si128(reinterpret_cast<const __m128i*>(feature + BlockSize));
585  addDistance(m1, x1, sum);
586  m3 = _mm_load_si128(reinterpret_cast<const __m128i*>(mean + 2 * BlockSize));
587  x3 = _mm_load_si128(reinterpret_cast<const __m128i*>(feature + 2 * BlockSize));
588  addDistance(m2, x2, sum);
589  addDistance(m3, x3, sum);
590  }
591  mean += Dimension;
592  {
593  s32 tmp = horizontalAdd(sum);
594  tmp += *dnsConst;
595  if (tmp < best)
596  best = tmp;
597  }
598  ++dnsConst;
599  }
600  f32* result = scoreBase + rp;
601  _mm_prefetch(const_cast<char*>(reinterpret_cast<const char*>(result)), _MM_HINT_NTA);
602  *result = static_cast<f32>(best) / scale_;
603  }
