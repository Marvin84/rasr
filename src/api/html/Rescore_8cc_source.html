31 /**
32  * reduce many scores to a single score;
33  * the sum of the scalar product remains unchanged
34  **/
35 class ReduceScoresLattice : public RescoreLattice {
36  typedef RescoreLattice Precursor;
37 
38 private:
39  ScoreIdList fromIds_;
40  ScoreId toId_;
41  ScoreList scales_;
42 
43 protected:
44  void reduce(ScoresRef sr) const {
45  Score score = sr->project(scales_);
46  for (ScoreIdList::const_iterator it = fromIds_.begin(); it != fromIds_.end(); ++it)
47  sr->set(*it, Semiring::One);
48  sr->set(toId_, score);
49  }
50 
51 public:
52  ReduceScoresLattice(ConstLatticeRef l, const ScoreIdList& fromIds, ScoreId toId, RescoreMode rescoreMode)
53  : Precursor(l, rescoreMode), fromIds_(fromIds), toId_(toId) {
54  const ScoreList& srScales(l->semiring()->scales());
55  scales_.resize(srScales.size(), 0.0);
56  for (ScoreIdList::const_iterator it = fromIds_.begin(); it != fromIds_.end(); ++it)
57  scales_[*it] = srScales[*it] / srScales[toId];
58  }
59  virtual ~ReduceScoresLattice() {}
60 
61  virtual void rescore(State* sp) const {
62  if (sp->isFinal())
63  reduce(sp->weight_);
64  for (State::iterator a = sp->begin(); a != sp->end(); ++a)
65  reduce(a->weight_);
66  }
67 
68  virtual std::string describe() const {
69  return Core::form("reduce(%s)", fsa_->describe().c_str());
70  }
71 };
72 class ReduceScoresNode : public RescoreNode {

75 public:
76  static const Core::ParameterStringVector paramKeys;
77 
78 private:
79  KeyList keys_;
80  ScoreIdList ids_;
81  ConstSemiringRef lastSemiring_;
82  ConstSemiringRef lastReducedSemiring_;
83 
84 protected:
85  virtual ConstLatticeRef rescore(ConstLatticeRef l) {
86  ConstSemiringRef semiring = l->semiring();
87  if (!lastReducedSemiring_ || (lastSemiring_.get() != semiring.get())) {
88  lastSemiring_ = semiring;
89  lastReducedSemiring_ = cloneSemiring(semiring);
90  ids_.resize(keys_.size());
91  ScoreIdList::iterator itId = ids_.begin();
92  for (KeyList::const_iterator itKey = keys_.begin(); itKey != keys_.end(); ++itKey, ++itId) {
93  *itId = semiring->id(*itKey);
94  if (!semiring->hasId(*itId))
95  criticalError("Semiring \"%s\" has no key \"%s\"",
96  semiring->name().c_str(), itKey->c_str());
97  lastReducedSemiring_->setScale(*itId, 0.0);
98  }
99  lastReducedSemiring_->setScale(ids_.front(), semiring->scale(ids_.front()));
100  }
101  l = ConstLatticeRef(new ReduceScoresLattice(l, ids_, ids_[0], rescoreMode));
102  l = changeSemiring(l, lastReducedSemiring_);
103  return l;
104  }

354 PenaltyMapRef buildPenaltyMap(const StringListList& orthClasses, const ScoreList& orthClassPenalties) {
355  require(orthClasses.size() == orthClassPenalties.size());
356  Lexicon* lexicon = Lexicon::us().get();
357  PenaltyMap* penalties = new PenaltyMap;
358  ScoreList::const_iterator itPenalty = orthClassPenalties.begin();
359  for (StringListList::const_iterator itSS = orthClasses.begin();
360  itSS != orthClasses.end(); ++itSS, ++itPenalty)
361  for (StringList::const_iterator itS = itSS->begin();
362  itS != itSS->end(); ++itS) {
363  Fsa::LabelId label = lexicon->lemmaId(*itS);
364  if (label != Fsa::InvalidLabelId)
365  penalties->insert(std::make_pair(label, *itPenalty));
366  else
367  Core::Application::us()->error(
368  "PenaltyMap: Could not find a lemma for orthography \"%s\".",
369  itS->c_str());
370  }
371  return PenaltyMapRef(penalties);
372 }

374 class ExtendByLemmaSpecificPenaltyLattice : public RescoreLattice {
375  typedef RescoreLattice Precursor;
376 
377 private:
378  Score defaultPenalty_;
379  PenaltyMapRef penalties_;
380  ScoreId id_;
381  Score scale_;
382  Core::Ref<const Bliss::LemmaAlphabet> lAlphabet_;
383  Core::Ref<const Bliss::LemmaPronunciationAlphabet> lpAlphabet_;
384 
385 protected:
386  Fsa::LabelId getLemmaId(Fsa::LabelId labelId) const {
387  if (lAlphabet_)
388  return labelId;
389  else
390  return lpAlphabet_->lemmaPronunciation(labelId)->lemma()->id();
391  }
392 
393 public:
394  ExtendByLemmaSpecificPenaltyLattice(ConstLatticeRef l, Score defaultPenalty, PenaltyMapRef penalties, ScoreId id, Score scale, RescoreMode rescoreMode)
395  : Precursor(l, rescoreMode), defaultPenalty_(defaultPenalty), penalties_(penalties), id_(id), scale_(scale) {
396  require(penalties);
397  lAlphabet_ = Lexicon::us()->lemmaAlphabet();
398  lpAlphabet_ = Lexicon::us()->lemmaPronunciationAlphabet();
399  if (l->getOutputAlphabet() == lAlphabet_)
400  lpAlphabet_.reset();
401  else if (l->getOutputAlphabet() == lpAlphabet_)
402  lAlphabet_.reset();
403  else
404  defect();
405  }
406  virtual ~ExtendByLemmaSpecificPenaltyLattice() {}
407 
408  virtual void rescore(State* sp) const {
409  for (State::iterator a = sp->begin(); a != sp->end(); ++a) {
410  Fsa::LabelId labelId = a->output();
411  if ((Fsa::FirstLabelId <= labelId) && (labelId <= Fsa::LastLabelId)) {
412  const Bliss::Lemma* lemma = (lAlphabet_) ? lAlphabet_->lemma(labelId) : lpAlphabet_->lemmaPronunciation(labelId)->lemma();
413  require_(lemma);
414  PenaltyMap::const_iterator it = penalties_->find(lemma->id());
415  if (it != penalties_->end())
416  a->weight_->add(id_, scale_ * it->second);
417  else
418  a->weight_->add(id_, scale_ * defaultPenalty_);
419  }
420  }
421  }
422 
423  ConstBoundariesRef getBoundaries() const {
424  return fsa_->getBoundaries();
425  }

429  }
