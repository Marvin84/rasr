78 template<class I, class O>
79 bool MergerNode<I, O>::work(Flow::PortId p) {
80  if (Precursor::needInit_)
81  Precursor::init();
82 
83  Precursor::inputData_.resize(this->nInputs());
84 
85  if (this->nInputs() == 0) {
86  this->error("No input connected.");
87  return Precursor::putNullData();
88  }
89 
90  Time startTime, endTime;
91  if (!Precursor::getData(startTime, endTime))
92  return Precursor::putNullData();
93 
94  OutputData* out = merge(Precursor::inputData_);
95  ensure(out != 0);
96 
97  out->setStartTime(startTime);
98  out->setEndTime(endTime);
99  return this->putData(0, out);
100 }
