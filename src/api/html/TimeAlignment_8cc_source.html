93  void distributeTimeOverUnalignedWords(StaticBoundaries* b, const std::vector<std::pair<Fsa::StateId, Fsa::LabelId>>& wordStarts) const {
94  if (wordStarts.size() < 3)
95  return;
96  Time startTime = (*b)[wordStarts.back().first].time();
97  Time duration = ((*b)[wordStarts.front().first].time() - startTime) / (wordStarts.size() - 1);
98  Core::Component::Message msg(log("Could not align:\n"));
99  msg << "\t" << startTime << "\t" << lAlphabet_->symbol(wordStarts.back().second) << "\n";
100  startTime += duration;
101  for (u32 i = wordStarts.size() - 2; 0 < i; --i) {
102  msg << "\t" << startTime << "\t" << lAlphabet_->symbol(wordStarts[i].second) << "\n";
103  (*b)[wordStarts[i].first].setTime(startTime);
104  startTime += duration;
105  }
106  }

138  ConstLatticeRef align(ConstLatticeRef lHyp, ConstPosteriorCnRef cnRef) {
139  verify((lHyp->type() == Fsa::TypeAcceptor) && (Lexicon::us()->alphabetId(lHyp->getInputAlphabet()) == Lexicon::LemmaAlphabetId));
140  ConstSemiringRef hypSemiring = lHyp->semiring();
141  ScoresRef One = hypSemiring->one();
142  Core::Vector<std::pair<Fsa::LabelId, ScoresRef>> hyp(1, std::make_pair(Fsa::Epsilon, One));
143  for (ConstStateRef sr = lHyp->getState(lHyp->initialStateId()); sr->hasArcs();
144  sr = lHyp->getState(sr->begin()->target())) {
145  if (sr->begin()->input() != Fsa::Epsilon) {
146  hyp.push_back(std::make_pair(sr->begin()->input(), sr->begin()->weight()));
147  hyp.push_back(std::make_pair(Fsa::Epsilon, One));
148  }
149  else {
150  hyp.back().second = hypSemiring->extend(hyp.back().second, sr->begin()->weight());
151  }
152  }
153 
154  verify(hyp.size() <= BptrMask);
155  verify(hyp.size() / 2 <= cnRef->size());
156 
157  const Score unkScore = (Semiring::Max - 1.0) / Score(cnRef->size());
158  Core::Vector<Score> scores1(hyp.size() + 1, Semiring::Max), scores2(hyp.size() + 1, Semiring::Max);
159  Core::Vector<Score>*lastScoresPtr = &scores1, *currentScoresPtr = &scores2;
160  (*lastScoresPtr)[0] = 0.0;
161  Core::Vector<u16> bptr(hyp.size() * cnRef->size());
162 
163  for (u32 t = 0; t < cnRef->size(); ++t) {
164  const u32 offset = t * hyp.size();
165  const PosteriorCn::Slot& pdf = (*cnRef)[t];
166  Core::Vector<Score>& lastScores = *lastScoresPtr, &currentScores = *currentScoresPtr;
167  currentScores[0] = Semiring::Max;
168  for (s32 i = 0; size_t(i) < hyp.size(); ++i) {
169  currentScores[i + 1] = Semiring::Max;
170  const PosteriorCn::Arc cnArc(hyp[i].first, 0);
171  PosteriorCn::Slot::const_iterator itPdf = std::lower_bound(pdf.begin(), pdf.end(), cnArc);
172  Score localScore = unkScore;
173  u16 flag = 0;
174  if ((itPdf != pdf.end()) && (itPdf->label == cnArc.label)) {
175  localScore = -::log(itPdf->score);
176  flag = AlignedFlag;
177  }
178  else if (cnArc.label == Fsa::Epsilon)
179  continue;
180  s32 maxJ = (cnArc.label == Fsa::Epsilon) ? 1 : 2;
181  for (s32 j = 0; j <= maxJ; ++j) {
182  if (lastScores[i - j + 1] == Semiring::Max)
183  continue;
184  Score score = lastScores[i - j + 1] + localScore;
185  if (score < currentScores[i + 1]) {
186  currentScores[i + 1] = score;
187  bptr[offset + i] = (i - j) | flag;
188  }
189  }
190  }
191  std::swap(lastScoresPtr, currentScoresPtr);
192  }
193  verify(((*lastScoresPtr)[hyp.size() - 1] < Semiring::Max) || ((*lastScoresPtr)[hyp.size()] < Semiring::Max));
194 
195  std::vector<std::pair<Fsa::StateId, Fsa::LabelId>> wordStarts;
196  StaticBoundaries* b = new StaticBoundaries;
197  StaticLattice* s = new StaticLattice;
198  s->setDescription(Core::form("aligned(%s)", lHyp->describe().c_str()));
199  s->setType(Fsa::TypeAcceptor);
200  s->setProperties(Fsa::PropertyAcyclic | Fsa::PropertyLinear, Fsa::PropertyAll);
201  s->setInputAlphabet(lHyp->getInputAlphabet());
202  s->setSemiring(lHyp->semiring());
203  s->setBoundaries(ConstBoundariesRef(b));
204  State* lastSp = s->newState();
205  b->set(lastSp->id(), Boundary(cnRef->size()));
206  lastSp->setFinal(One);
207  wordStarts.push_back(std::make_pair(lastSp->id(), Fsa::InvalidLabelId));
208  u32 j = ((*lastScoresPtr)[hyp.size() - 1] < (*lastScoresPtr)[hyp.size()]) ? hyp.size() - 2 : hyp.size() - 1;
209  u32 lastJ = j;
210  std::pair<Fsa::LabelId, ScoresRef>
211  currentHyp = std::make_pair(Fsa::InvalidLabelId, One),
212  lastHyp = hyp[lastJ];
213  bool isLastAligned = false, hasUnaligned = false;
214  for (s32 t = cnRef->size() - 1; 0 <= t; --t) {
215  const u32 offset(t * hyp.size());
216  currentHyp = hyp[j];
217  if (j < lastJ) {
218  State* sp = s->newState();
219  sp->newArc(lastSp->id(), lastHyp.second, lastHyp.first);
220  b->set(sp->id(), Boundary(t + 1));
221  if (isLastAligned) {
222  if (lastHyp.first != Fsa::Epsilon) {
223  if (hasUnaligned)
224  distributeTimeOverUnalignedWords(b, wordStarts);
225  wordStarts.clear();
226  hasUnaligned = false;
227  }
228  }
229  else
230  hasUnaligned = true;
231  wordStarts.push_back(std::make_pair(sp->id(), lastHyp.first));
232  lastJ = j;
233  lastSp = sp;
234  lastHyp = currentHyp;
235  isLastAligned = false;
236  }
237  else
238  verify(j == lastJ);
239  if (bptr[offset + j] & AlignedFlag)
240  isLastAligned = true;
241  j = bptr[offset + j] & BptrMask;
242  }
243  verify(currentHyp.first != Fsa::InvalidLabelId);
244  State* sp = s->newState();
245  sp->newArc(lastSp->id(), lastHyp.second, lastHyp.first);
246  b->set(sp->id(), Boundary(0));
247  if (isLastAligned && hasUnaligned)
248  distributeTimeOverUnalignedWords(b, wordStarts);
249  else if (!isLastAligned) {
250  wordStarts.push_back(std::make_pair(sp->id(), lastHyp.first));
251  distributeTimeOverUnalignedWords(b, wordStarts);
252  }
253  s->setInitialStateId(sp->id());
254 
255  verify(s->getBoundaries()->valid());
256 
257  return ConstLatticeRef(s);
258  }

260  ConstLatticeRef align(ConstLatticeRef lHyp, ConstLatticeRef lRef) {
261  ensure(Lexicon::us()->alphabetId(lRef->getInputAlphabet()) == Lexicon::LemmaAlphabetId);
262  ConstLatticeRef lMeshedRef = persistent(staticScore(mesh(lRef, MeshTypeTimeBoundary), lHyp->semiring(), lHyp->semiring()->one()));
263  ConstLatticeRef lAlignedHyp = best(composeSequencing(lMeshedRef, lHyp), BellmanFord);
264 
265  if (!lAlignedHyp || (lAlignedHyp->initialStateId() == Fsa::InvalidStateId))
266  return ConstLatticeRef();
267  lAlignedHyp->addProperties(Fsa::PropertyLinear);
268  return lAlignedHyp;
269  }

272  ConstLatticeRef align(ConstLatticeRef lHyp, ConstLatticeRef lRef, ConstPosteriorCnRef cnRef) {
273  ++statistics_.n;
274  if (!lHyp || (lHyp->initialStateId() == Fsa::InvalidStateId))
275  return ConstLatticeRef();
276  if (!lHyp->hasProperty(Fsa::PropertyLinear)) {
277  for (ConstStateRef sr = lHyp->getState(lHyp->initialStateId()); sr->hasArcs();
278  sr = lHyp->getState(sr->begin()->target()))
279  if (sr->nArcs() != 1)
280  criticalError(
281  "Cannot align \"%s\" is not linear; lattice is not linear.",
282  lHyp->describe().c_str());
283  lHyp->addProperties(Fsa::PropertyLinear);
284  }
285  lHyp->setBoundaries(InvalidBoundaries);
286  ConstLatticeRef l;
287  if (lRef && tryIntersection_) {
288  l = align(lHyp, lRef);
289  if (l)
290  ++statistics_.nIntersections;
291  }
292  if (!l && (lRef || cnRef)) {
293  if (!cnRef) {
294  std::pair<ConstLatticeRef, ConstFwdBwdRef> fbResult = fbBuilder_->build(lRef);
295  lRef = fbResult.first;
296  cnRef = buildFramePosteriorCn(lRef, fbResult.second);
297  }
298  l = align(lHyp, cnRef);
299  if (l)
300  ++statistics_.nFrameAlignments;
301  }
302  return l;
303  }
