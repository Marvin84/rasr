73 class ArcRemovalLattice : public SlaveLattice {
74 private:
75  typedef typename ArcFilter::Filter Filter;
76  typedef typename ArcFilter::WeakOrder WeakOrder;
77  ConstSemiringRef semiring_;
78  const WeakOrder weakOrder_;
79  mutable TopologicalOrderQueueRef topQueue_;
80  mutable Core::Vector<ScoresRef> closureScores_;
81 
82 public:
83  ArcRemovalLattice(ConstLatticeRef l)
84  : SlaveLattice(cache(ConstLatticeRef(new ArcSortLattice<WeakOrder>(l)))),
85  semiring_(l->semiring()),
86  weakOrder_(l) {
87  ConstStateMapRef topologicalOrderMap = findTopologicalOrder(l);
88  verify(topologicalOrderMap && (topologicalOrderMap->maxSid != Fsa::InvalidStateId));
89  topQueue_ = createTopologicalOrderQueue(l, topologicalOrderMap);
90  closureScores_.grow(topologicalOrderMap->maxSid);
91  }
92  virtual ~ArcRemovalLattice() {}
93 
94  virtual ConstStateRef getState(Fsa::StateId sid) const {
95  ConstStateRef sr = fsa_->getState(sid);
96  Filter filter(fsa_, sr);
97 
98  // Check in O(1), if matching arc(s) exist
99  if (!sr->hasArcs() || !filter(*sr->begin()))
100  return sr;
101 
102  // Initialize closure
103  verify(sid < closureScores_.size());
104  TopologicalOrderQueue& topQ = *topQueue_;
105  State* sp = new State(sr->id());
106  State::const_iterator a = sr->begin();
107  for (; (a != sr->end()) && filter(*a); ++a)
108  if (!closureScores_[a->target()]) {
109  closureScores_[a->target()] = a->weight();
110  topQ.insert(a->target());
111  }
112  else
113  closureScores_[a->target()] =
114  semiring_->collect(closureScores_[a->target()], a->weight());
115  for (; a != sr->end(); ++a)
116  *sp->newArc() = *a;
117 
118  // Process closure
119  ScoresRef finalWeight;
120  if (sr->isFinal()) {
121  sp->addTags(Fsa::StateTagFinal);
122  finalWeight = sr->weight();
123  }
124  else
125  finalWeight = semiring_->zero();
126  while (!topQ.empty()) {
127  Fsa::StateId epsSid = topQ.top();
128  topQ.pop();
129  ConstStateRef epsSr = fsa_->getState(epsSid);
130  ScoresRef score = closureScores_[epsSid];
131  closureScores_[epsSid].reset();
132  if (epsSr->isFinal()) {
133  sp->addTags(Fsa::StateTagFinal);
134  finalWeight = semiring_->collect(
135  finalWeight, semiring_->extend(score, epsSr->weight()));
136  }
137  State::const_iterator a = epsSr->begin(), a_end = epsSr->end();
138  for (; (a != a_end) && (a->input() == Fsa::Epsilon) && (a->output() == Fsa::Epsilon); ++a)
139  if (!closureScores_[a->target()]) {
140  closureScores_[a->target()] = semiring_->extend(score, a->weight());
141  topQ.insert(a->target());
142  }
143  else
144  closureScores_[a->target()] =
145  semiring_->collect(
146  closureScores_[a->target()], semiring_->extend(
147  score, a->weight()));
148  for (; a != a_end; ++a) {
149  State::iterator pos = sp->lower_bound(*a, weakOrder_);
150  if ((pos == sp->end()) || (a->target() != pos->target()) || (a->input() != pos->input()) || (a->output() != pos->output())) {
151  sp->insert(pos, *a)->weight_ =
152  semiring_->extend(score, a->weight());
153  }
154  else
155  pos->weight_ = semiring_->collect(
156  pos->weight(), semiring_->extend(score, a->weight()));
157  }
158  }
159  if (sp->isFinal())
160  sp->setWeight(finalWeight);
161  return ConstStateRef(sp);
162  }
163  virtual std::string describe() const {
164  return Core::form("%s(%s)", ArcFilter::describe().c_str(), fsa_->describe().c_str());
165  }
166 };
