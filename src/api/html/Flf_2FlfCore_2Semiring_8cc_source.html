218 int Semiring::compare(const ScoresRef& a, const ScoresRef& b) const {
219  if (a.get() == b.get())
220  return 0;
221  Score pa = 0.0, pb = 0.0;
222  Scores::const_iterator ita = begin(a), itb = begin(b);
223  for (ScoreList::const_iterator its = scales_.begin(), send = scales_.end();
224  its != send; ++its, ++ita, ++itb) {
225  pa += (*its) * (*ita);
226  pb += (*its) * (*itb);
227  }
228  return Core::isAlmostEqualUlp(pa, pb, tolerance_) ? 0 : (pa < pb) ? -1 : 1;
229 }

266 std::string Semiring::describe(const ScoresRef& a, Fsa::Hint hint) const {
267  if (n_ == 0)
268  return "[]";
269  std::ostringstream oss;
270  if (hint & Fsa::HintShowDetails) {
271  Scores::const_iterator ita = begin(a), enda = end(a);
272  ScoreList::const_iterator its = scales_.begin();
273  oss << "[";
274  if (hint & Fsa::HintAsProbability) {
275  if (hint & HintUnscaled) {
276  oss << ::exp(-(*ita));
277  if (*its != DefaultScale)
278  oss << "^" << *its;
279  for (++ita, ++its; ita != enda; ++ita, ++its) {
280  oss << " ";
281  oss << ::exp(-(*ita));
282  if (*its != DefaultScale)
283  oss << "^" << *its;
284  }
285  }
286  else {
287  oss << ::exp(-(*its * *ita));
288  for (++ita, ++its; ita != enda; ++ita, ++its)
289  oss << " " << exp(-(*its * *ita));
290  }
291  }
292  else {
293  if (hint & HintUnscaled) {
294  if (*its != DefaultScale)
295  oss << *its << "*";
296  oss << *ita;
297  for (++ita, ++its; ita != enda; ++ita, ++its) {
298  oss << " ";
299  if (*its != DefaultScale)
300  oss << *its << "*";
301  oss << *ita;
302  }
303  }
304  else {
305  oss << (*its * *ita);
306  for (++ita, ++its; ita != enda; ++ita, ++its)
307  oss << " " << (*its * *ita);
308  }
309  }
310  oss << "]";
311  }
312  else {
313  if (hint & Fsa::HintAsProbability)
314  oss << ::exp(-(a->project(scales_)));
315  else
316  oss << a->project(scales_);
317  }
318  return oss.str();
319 }

336 void Semiring::compress(ByteVector& v, const ScoresRef& a) const {
337  if (a == one_)
338  v.push_back(1);
339  else if (a == zero_)
340  v.push_back(0);
341  else if (a == invalid_)
342  v.push_back(2);
343  else {
344  v.push_back(3);
345  for (Scores::const_iterator it = begin(a), itEnd = end(a); it != itEnd; ++it) {
346  u32 x(f32_u32_reinterpret_cast(f32(*it)));
347  v.push_back((x >> 24) & 0xff);
348  v.push_back((x >> 16) & 0xff);
349  v.push_back((x >> 8) & 0xff);
350  v.push_back(x & 0xff);
351  }
352  }
353 }

355 ScoresRef Semiring::uncompress(ByteVector::const_iterator& vIt) const {
356  u8 header(*vIt++);
357  switch (header) {
358  case 0:
359  return zero_;
360  case 1:
361  return one_;
362  case 2:
363  return invalid_;
364  case 3: {
365  ScoresRef a(create());
366  for (Scores::iterator it = begin(a), itEnd = end(a); it != itEnd; ++it) {
367  u32 x(0);
368  x |= *(vIt++);
369  x <<= 8;
370  x |= *(vIt++);
371  x <<= 8;
372  x |= *(vIt++);
373  x <<= 8;
374  x |= *(vIt++);
375  *it = f32_u32_reinterpret_cast(x);
376  }
377  return a;
378  }
379  default:
380  defect();
381  return invalid_;
382  }
383 }

626 ScoresRef LogSemiring::collect(const ScoresRef& a, const ScoresRef& b) const {
627  if (b.get() == zero_.get())
628  return a;
629  if (a.get() == zero_.get())
630  return b;
631  Scores* cPtr = alloc();
632  ScoresRef c(cPtr);
633  Score x = 0.0, y = 0.0, norm = 0.0;
634  Scores::iterator itc = begin(cPtr);
635  Scores::const_iterator ita = begin(a), itb = begin(b);
636  for (ScoreList::const_iterator its = scales_.begin(), ends = scales_.end();
637  its != ends; ++its, ++itc, ++ita, ++itb) {
638  const Score scale = *its;
639  if (scale != 0.0) {
640  if ((*ita == Core::Type<Score>::max) || (*itb == Core::Type<Score>::max))
641  return max();
642  const Score sa = scale * *ita, sb = scale * *itb;
643  x += sa;
644  y += sb;
645  norm += * itc = logAdd(sa, sb);
646  }
647  else
648  *itc = 0.0;
649  }
650  const Score z = logAdd(x, y) / norm;
651  itc = begin(cPtr);
652  for (ScoreList::const_iterator its = scales_.begin(), ends = scales_.end();
653  its != ends; ++its, ++itc) {
654  const Score scale = *its;
655  if (scale != 0.0)
656  *itc *= z / scale;
657  }
658  return c;
659 }
