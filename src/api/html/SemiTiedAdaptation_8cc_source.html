57 void SemiTiedEstimator::estimateWMatrices() {
58 #ifdef __semi_tied_one_base__
59  log("semi tied one base");
60 #endif
61 
62  FullAdaptorViterbiEstimator::estimateWMatrices();
63 
64  NodeId id, qId;
65  NodeIdToIdSetMap tyingScheme;
66 
67  for (Core::BinaryTree::LeafList::const_iterator p = tree_->leafList().begin();
68  p != tree_->leafList().end(); ++p) {
69  id = tree_->id(p);
70 
71  while (id != tree_->root() && count_[id] <= minSemiTiedAdaptationObservations_) {
72  id = tree_->previous(id);
73  }
74  //get corresponding matrix
75 
76  qId = id;
77 
78  MatrixConstIterator pit = w_.find(qId);
79  while (pit == w_.end() && qId != tree_->root()) {
80  qId = tree_->previous(qId);
81  pit = w_.find(qId);
82  }
83 #ifdef __semi_tied_one_base__
84  qId = tree_->root();
85 #endif
86  if (qId == tree_->root() && (pit == w_.end())) {
87  //not enough observations in complete tree
88  log("too few observations for base adaptation\n") << minAdaptationObservations_ << " observations needed, " << count_[qId] << " seen.\n"
89  << "resetting matrix to unity";
90  w_[qId] = adaptationUnitMatrix(dimension_);
91  return;
92  }
93  if (qId != id)
94  tyingScheme[qId].insert(id); //else: qId is already leaf
95  }
96 
97  for (NodeIdToIdSetMap::iterator p = tyingScheme.begin();
98  p != tyingScheme.end(); ++p) {
99  if (p->second.size() == 1) {
100  qId = p->first;
101  id = p->second.begin()->id;
102  if (tree_->left(qId) == id) {
103  p->second.insert(IdSetEntry(tree_->right(qId), false));
104  }
105  else {
106  p->second.insert(IdSetEntry(tree_->left(qId), false));
107  }
108  }
109  }
110 
111  if (adaptationDumpChannel_.isOpen()) {
112  adaptationDumpChannel_ << Core::XmlOpen("semi-tied-tying-scheme");
113  for (NodeIdToIdSetMap::const_iterator p = tyingScheme.begin();
114  p != tyingScheme.end(); ++p) {
115  const IdSet& s = p->second;
116  adaptationDumpChannel_ << "\n"
117  << p->first << ": ";
118  for (IdSetConstIterator q = s.begin(); q != s.end(); ++q) {
119  adaptationDumpChannel_ << q->id << "(" << q->isActive << ") ";
120  }
121  }
122  adaptationDumpChannel_ << Core::XmlClose("semi-tied-tying-scheme");
123  }
124 
125  Math::Vector<Matrix> zMatrices;
126  Math::Vector<Matrix> gMatrices;
127  double maxValue = Core::Type<double>::min;
128 
129  {
130  Math::Vector<Precursor::ZAccumulator> z;
131  Math::Vector<Precursor::GAccumulator> g;
132  propagate(leafZAccumulators_, z, tree_->root());
133  propagate(leafGAccumulators_, g, tree_->root());
134  zMatrices.resize(z.size());
135  gMatrices.resize(g.size());
136 
137  for (u32 i = 0; i < z.size(); ++i) {
138  zMatrices[i] = z[i].squareMatrix();
139  gMatrices[i] = g[i].squareMatrix();
140  double temp = zMatrices[i].maxElement();
141  if (temp > maxValue)
142  maxValue = temp;
143  temp = gMatrices[i].maxElement();
144  if (temp > maxValue)
145  maxValue = temp;
146  }
147  }
148 
149  //normalize all matrices by overall largest element
150  double weight = 1.0 / maxValue;
151  weight = 1.0;
152 
153  ensure(zMatrices.size() == gMatrices.size());
154  for (u32 i = 0; i < zMatrices.size(); ++i) {
155  zMatrices[i] *= weight;
156  gMatrices[i] *= weight;
157  }
158 
159  Core::ProgressIndicator pi("estimating MLLR matrices");
160  pi.start(tyingScheme.size());
161  for (NodeIdToIdSetMap::const_iterator p = tyingScheme.begin();
162  p != tyingScheme.end(); ++p) {
163  qId = p->first;
164  solveEstimationEquations(p->second, qId, zMatrices, gMatrices);
165  pi.notify();
166  }
167  pi.finish();
168 }

170 void SemiTiedEstimator::solveEstimationEquations(const IdSet& idSet,
171  NodeId qId,
172  const Math::Vector<Matrix>& z,
173  const Math::Vector<Matrix>& g) {
174  std::map<NodeId, Math::Vector<double>> lambda;
175  std::map<NodeId, Matrix> previousMatrices;
176  Math::Vector<double> bias = w_[qId].column(0);
177  w_[qId].removeColumn(0);
178 
179 #ifdef __dumpMatrix__
180  string fileName = "w_init.m";
181 
182  std::ofstream file(fileName.c_str(), std::ios::binary);
183  w_[qId].print_raw(file, 25);
184  file.close();
185 
186  for (IdSetConstIterator c = idSet.begin(); c != idSet.end(); ++c) {
187  std::ostringstream os;
188  os << "g_" << c->id << ".m" << ends;
189  file.open(os.str().c_str());
190  g[c->id].print_raw(file, 25);
191  file.close();
192  os.str("");
193  os << "z_" << c->id << ".m" << ends;
194  file.open(os.str().c_str());
195  z[c->id].print_raw(file, 25);
196  file.close();
197  }
198 #endif
199 
200  Matrix U = w_[qId], V(dimension_);
201  Math::Vector<double> wInit(dimension_, 0.0);
202 
203  //get initial values;
204  svd(U, wInit, V, w_[qId]);
205 
206 #ifndef __lambda_only__
207  for (IdSetConstIterator c = idSet.begin(); c != idSet.end(); ++c) {
208  lambda[c->id] = wInit;
209  }
210 
211  SemiTiedOptimizationFunction function(dimension_, g, z, lambda);
212  SemiTiedOptimizationGradient gradient(dimension_, g, z, lambda);
213  int iterations;
214  double returnValue;
215  Math::Vector<double> x = convert2Vector(lambda, U, V);
216 
217  frprmn(x, stopCriterion_, iterations, returnValue, function, gradient); //numerical recipes conjugate gradient
218  log("number of conjugate gradient iterations for semi-tied MLLR estimation: ") << iterations;
219 
220  convert2Matrices(x, lambda, U, V, dimension_);
221 #else
222  log("lambda only estimation");
223  Matrix UT = U.transpose();
224  Matrix UT_U = UT * U;
225  Matrix VT = V.transpose();
226 
227  for (IdSetConstIterator c = idSet.begin(); c != idSet.end(); ++c) {
228  Matrix VT_Gc_V = VT * g[c->id] * V;
229  Math::Vector<double> diag_UT_Zc_V = Math::diagonal(UT * z[c->id] * V);
230  Matrix Ac = Math::multiplyElementwise(UT_U, VT_Gc_V);
231  Math::Vector<double> x(diag_UT_Zc_V.size());
232  s32 status = solveLinearLeastSquares(x, Ac, diag_UT_Zc_V);
233  if (status != 0)
234  error("Lapack routine solveLinearLeastSquares failed! status = ") << status;
235  lambda[c->id] = x;
236  }
237 
238 #endif
239 
240  for (IdSetConstIterator c = idSet.begin(); c != idSet.end(); ++c) {
241  if (c->isActive) {
242  w_[c->id] = U * Math::makeDiagonalMatrix(lambda[c->id]) * V.transpose();
243  w_[c->id].insertColumn(0, bias);
244  }
245  }
246  w_[qId].insertColumn(0, bias);
247 }

276 SemiTiedOptimizationFunction::ResultType SemiTiedOptimizationFunction::operator()(const ArgumentType& x) const {
277  convert2Matrices(x, lambda_, U_, V_, dimension_);
278  UT_ = U_.transpose();
279  f1_.fill(0.0);
280  f2_.fill(0.0);
281 
282  for (std::map<NodeId, Math::Vector<double>>::const_iterator p = lambda_.begin(); p != lambda_.end(); ++p) {
283  NodeId c = p->first;
284  AT_ = V_ * Math::makeDiagonalMatrix(lambda_.find(c)->second) * UT_;
285  f1_ = f1_ + AT_ * AT_.transpose() * G_[c];
286  f2_ = f2_ + AT_ * Z_[c];
287  }
288  return (f1_.trace() - 2 * f2_.trace());
289 }

305 void SemiTiedOptimizationGradient::operator()(const InputType& x, ResultType& f) const {
306  f.clear();
307  convert2Matrices(x, lambda_, U_, V_, dimension_);
308 
309  UT_ = U_.transpose();
310  VT_ = V_.transpose();
311 
312  f1_.fill(0.0);
313  f2_.fill(0.0);
314  UT_U_ = UT_ * U_;
315 
316  for (std::map<NodeId, Math::Vector<double>>::const_iterator p = lambda_.begin(); p != lambda_.end(); ++p) {
317  NodeId c = p->first;
318  lc_ = Math::makeDiagonalMatrix(lambda_.find(c)->second);
319  const Math::Matrix<double>& Gc = G_[c];
320  const Math::Matrix<double>& Zc = Z_[c];
321  VT_Gc_V_ = VT_ * Gc * V_;
322 
323  f0_[c] = 2.0 * Math::diagonal(UT_U_ * lc_ * VT_Gc_V_ - UT_ * Zc * V_);
324  f1_ = f1_ + U_ * lc_ * VT_Gc_V_ * lc_ - Zc * V_ * lc_;
325  f2_ = f2_ + Gc * V_ * lc_ * UT_U_ * lc_ - Zc.transpose() * U_ * lc_;
326  }
327  f = convert2Vector(f0_, 2.0 * f1_, 2.0 * f2_);
328 }

330 void Mm::convert2Matrices(const Math::Vector<double>& x, std::map<NodeId, Math::Vector<double>>& l,
331  Math::Matrix<double>& u, Math::Matrix<double>& v, ComponentIndex dim) {
332  div_t s1 = div((int)x.size(), (int)dim);
333  require(s1.rem == 0);
334 
335  size_t nClasses = s1.quot - 2 * dim;
336  require(nClasses == l.size());
337 
338  u32 c = 0;
339  for (std::map<NodeId, Math::Vector<double>>::iterator p = l.begin(); p != l.end(); ++p) {
340  p->second = Math::Vector<double>(x.begin() + c * dim, x.begin() + c * dim + dim);
341  ++c;
342  }
343 
344  size_t i = c * dim;
345  if (u.nRows() != dim)
346  u.resize(dim);
347 
348  for (size_t col = 0; col < u.nColumns(); ++col) {
349  for (size_t row = 0; row < u.nRows(); row++) {
350  u[row][col] = x[i];
351  ++i;
352  }
353  }
354  ensure(i - 1 < x.size());
355 
356  if (v.nRows() != dim)
357  v.resize(dim);
358  for (size_t col = 0; col < v.nColumns(); ++col) {
359  for (size_t row = 0; row < v.nRows(); row++) {
360  v[row][col] = x[i];
361  ++i;
362  }
363  }
364  ensure(i - 1 < x.size());
365 }
