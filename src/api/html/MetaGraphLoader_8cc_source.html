34 std::unique_ptr<Graph> MetaGraphLoader::load_graph() {
35  auto timer_start = std::chrono::steady_clock::now();
36  if (meta_graph_file_.empty()) {
37  criticalError("no graph-def-path set");
38  }
39 
40  tf::Env* env = tf::Env::Default();
41  tf::MetaGraphDef meta_graph_def;
42  tf::Status status = ReadBinaryProto(env, meta_graph_file_, &meta_graph_def);
43  if (not status.ok()) {
44  criticalError("error reading graph def %s", status.ToString().c_str());
45  }
46 
47  if (not meta_graph_def.has_graph_def()) {
48  criticalError("meta-graph has not graph def");
49  }
50  if (not meta_graph_def.has_saver_def()) {
51  criticalError("meta-graph has not saver def");
52  }
53 
54  restore_op_name_ = meta_graph_def.saver_def().restore_op_name();
55  restore_filename_tensor_name_ = meta_graph_def.saver_def().filename_tensor_name();
56 
57  tf::GraphDef const& graph_def = meta_graph_def.graph_def();
58  std::unique_ptr<Graph> result(new Graph());
59  setGraphDef(*result, graph_def);
60 
61  for (std::string const& lib : required_libraries_) {
62  result->addLibrary(lib);
63  }
64 
65  std::unordered_map<std::string, Variable> vars;
66  for (int i = 0; i < graph_def.node_size(); i++) {
67  auto& node = graph_def.node(i);
68  if (node.op() == "Placeholder" or node.op() == "PlaceholderV2") {
69  result->addInput(node.name());
70  }
71  else if (node.op() == "Variable" or node.op() == "VariableV2") {
72  DataType dt = node.attr().find("dtype")->second.type();
73  std::string var_name = node.name() + ":0"; // to use the same name as is used inside collections we append ":0" here
74  auto const& shape = node.attr().find("_output_shapes")->second.list().shape(0);
75  std::vector<s64> dims(static_cast<size_t>(shape.dim_size()));
76  for (int i = 0; i < shape.dim_size(); i++) {
77  dims[i] = shape.dim(i).size();
78  }
79  vars[var_name] = {var_name, "", "", "", dt, dims};
80  }
81  }
82 
83  variable_initializers_.clear();
84  auto const& collections = meta_graph_def.collection_def();
85  for (auto const& keyval : collections) {
86  if (keyval.first == "variables") {
87  if (not keyval.second.has_bytes_list()) {
88  error("variables collection is not a byte-list");
89  break;
90  }
91  auto const& list = keyval.second.bytes_list();
92  for (int i = 0; i < list.value_size(); i++) {
93  std::string const& var_def_string = list.value(i);
94  tf::VariableDef var_def;
95  var_def.ParseFromString(var_def_string);
96  auto iter = vars.find(var_def.variable_name());
97  if (iter != vars.end()) {
98  iter->second.initial_value_name = var_def.initial_value_name();
99  iter->second.initializer_name = var_def.initializer_name();
100  iter->second.snapshot_name = var_def.snapshot_name();
101  result->addVariable(iter->second);
102  variable_initializers_.push_back(var_def.initializer_name());
103  }
104  }
105  }
106  else if (keyval.first == "update_ops") {
107  if (not keyval.second.has_node_list()) {
108  error("update_ops collection is not a node-list");
109  break;
110  }
111  auto const& list = keyval.second.node_list();
112  for (int i = 0; i < list.value_size(); i++) {
113  result->addUpdateOp(list.value(i));
114  }
115  }
116  else if (keyval.first == "_RETURNN_state_vars") {
117  if (not keyval.second.has_node_list()) {
118  error("_RETURNN_state_vars collection is not a node-list");
119  break;
120  }
121  auto const& list = keyval.second.node_list();
122  for (int i = 0; i < list.value_size(); i++) {
123  result->addStateVar(list.value(i));
124  }
125  }
126  }
127 
128  auto timer_end = std::chrono::steady_clock::now();
129  log("Session::loadGraph: ") << std::chrono::duration<double, std::milli>(timer_end - timer_start).count() << "ms";
130 
131  return result;
132 }

134 void MetaGraphLoader::initialize(Session& session) {
135  auto timer_start = std::chrono::steady_clock::now();
136 
137  if (saved_model_file_.empty()) {
138  criticalError("no saved-model-file set");
139  }
140 
141  std::vector<Tensor> outputs;
142  Tensor filename_tensor;
143  filename_tensor.set<Tensorflow::tstring>(saved_model_file_);
144  session.run({std::make_pair<>(restore_filename_tensor_name_, filename_tensor)}, {}, {restore_op_name_}, outputs);
145 
146  auto timer_end = std::chrono::steady_clock::now();
147  log("Session::initialize: ") << std::chrono::duration<double, std::milli>(timer_end - timer_start).count() << "ms " << saved_model_file_;
148 }
