72 void LabelingFeatureExtractor::processFeature(Core::Ref<const Feature> feature) {
73  while (!currentTimestamp_.contains(feature->timestamp())) {
74  Flow::DataPtr<Flow::String> in;
75  if (dataSource()->getData(labelsPort_, in)) {
76  Core::StringHashMap<LabelIndex>::iterator index = labelToIndexMap_.find((*in)());
77  if (index != labelToIndexMap_.end()) {
78  currentLabelIndex_ = index->second;
79  currentTimestamp_ = *in;
80  }
81  else
82  criticalError("Unknown label recivied '%s'.", (*in)().c_str());
83  }
84  else {
85  criticalError("No label found for interval [%f..%f].",
86  feature->timestamp().startTime(), feature->timestamp().endTime());
87  }
88  }
89  labeledFeatureProcessor_.processLabeledFeature(feature, currentLabelIndex_);
90 }
