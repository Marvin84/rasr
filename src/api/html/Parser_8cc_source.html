252 void XmlQuestionRefListBuilderElement::forEachValue(const std::string& key) {
253  if (isForEachValue_) {
254  if (valueList_.empty()) {
255  const Core::Choice& values = (*mapRef_)[key];
256  if (values.nChoices() == 2)
257  addQuestion(key, values.begin()->ident());
258  else
259  for (Core::Choice::const_iterator it = values.begin();
260  it != values.end(); ++it)
261  addQuestion(key, it->ident());
262  }
263  else
264  for (std::vector<std::string>::const_iterator it = valueList_.begin();
265  it != valueList_.end(); ++it)
266  addQuestion(key, *it);
267  }
268  else {
269  if (values_.empty()) {
270  parser()->warning("In element \"%s\": no range for question for key \"%s\"; assume predicate",
271  name(), key.c_str());
272  values_.push_back("true");
273  }
274  addQuestion(key, values_);
275  }
276 }

278 void XmlQuestionRefListBuilderElement::forEachKey() {
279  if (isForEachKey_) {
280  if (keyList_.empty())
281  for (size_t i = 0; i < mapRef_->size(); ++i)
282  forEachValue(mapRef_->key(i));
283  else
284  for (std::vector<std::string>::const_iterator it = keyList_.begin();
285  it != keyList_.end(); ++it)
286  forEachValue(*it);
287  }
288  else {
289  if (mapRef_->isDefined(key_))
290  forEachValue(key_);
291  else
292  parser()->criticalError("In element \"%s\": missing key", name());
293  }

440 void XmlBinaryTreeRootBuilderElement::start(const XmlAttributes attr) {
441  Precursor::start(attr);
442  Node::Id id;
443  const char* c = attr["id"];
444  verify(c);
445  if (!Core::strconv(std::string(c), id))
446  parser()->error(
447  "In element \"%s\": non-interpretable value \"%s\"",
448  name(), c);
449  if (!node_) {
450  node_ = new BinaryTree::Node(0, id);
451  }
452  else if (!node_->leftChild_) {
453  node_->leftChild_ = new Node(node_, id);
454  node_ = node_->leftChild_;
455  }
456  else if (!node_->rightChild_) {
457  node_->rightChild_ = new Node(node_, id);
458  node_ = node_->rightChild_;
459  }
460  else {
461  defect();
462  }
463 }
464 
465 void XmlBinaryTreeRootBuilderElement::end() {
466  Precursor::end();
467  require((!node_->leftChild_ && !node_->rightChild_) ||
468  (node_->leftChild_ && node_->rightChild_));
469  if (node_->father_ == 0) {
470  delegateOrDelete(node_);
471  node_ = 0;
472  }
473  else {
474  node_ = node_->father_;
475  }
476 }

590 bool convert(const std::string& buffer, FloatBox& values) {
591  const char* start = buffer.c_str();
592  char* end;
593  for (FloatBox::vector_iterator it = values.begin();
594  it != values.end(); ++it) {
595  *it = ::strtod(start, &end);
596  verify_(start != end);
597  start = end;
598  }
599  while (::isspace(*start)) {
600  ++start;
601  }
602  return (*start == '\0');
603 }

717 struct XmlExampleListMergerElement::ExampleListMerger {
718  XmlExampleListMergerElement* father;
719  typedef std::unordered_map<Properties*, Example*, Properties::PtrHashFcn, Properties::PtrEqual> ExampleMap;
720  ExampleMap exampleMap;
721 
722  ExampleListMerger(XmlExampleListMergerElement* father)
723  : father(father) {
724  for (ExampleList::iterator itExample = father->examples_->begin(), endExample = father->examples_->end();
725  itExample != endExample; ++itExample)
726  if (!exampleMap.insert(std::make_pair((*itExample)->properties, &**itExample)).second)
727  father->parser()->criticalError("Initial example list contains examples eith equal properties.");
728  }
729 
730  void merge(Example* example) {
731  std::pair<ExampleMap::iterator, bool> result = exampleMap.insert(std::make_pair(example->properties, example));
732  if (result.second) {
733  father->examples_->add(example);
734  }
735  else {
736  Example& trgExample = *result.first->second;
737  trgExample.nObs += example->nObs;
738  *trgExample.values += *example->values;
739  delete example;
740  }
741  }
742 };
