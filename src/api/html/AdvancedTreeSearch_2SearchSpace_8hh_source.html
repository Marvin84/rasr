566  // endTime is _inclusive_
567  virtual bool merge(const Core::Ref<Pruning>& rhs, TimeframeIndex ownLength, TimeframeIndex startTime, TimeframeIndex endTime) {
568  PruningDesc* rhsDesc(dynamic_cast<PruningDesc*>(rhs.get()));
569  if (rhsDesc->timeBeamMap.empty() && startTime == 0 && endTime + 1 >= ownLength && timeBeamMap.empty()) {
570  beam = rhsDesc->beam;
571  return true;
572  }
573  verify(rhsDesc);
574  verify(startTime <= endTime);
575  if (endTime >= timeBeamMap.size())
576  timeBeamMap.resize(endTime + 1, Core::Type<Score>::max);
577  for (TimeframeIndex t = startTime; t <= endTime; ++t)
578  timeBeamMap[t] = rhsDesc->beamForTime(t - startTime);
579  return true;
580  }

582  /// Extend the pruning-thresholds by a score offset, a score factor, and by a specific number of timeframes
583  virtual bool extend(Score scoreFactor, Score scoreOffset, TimeframeIndex timeOffset) {
584  if (timeBeamMap.size()) {
585  std::vector<Score> oldTimePruningMap;
586  oldTimePruningMap.swap(timeBeamMap);
587  timeBeamMap.resize(oldTimePruningMap.size() + timeOffset, Core::Type<Score>::max);
588  for (TimeframeIndex t = 0; t < timeBeamMap.size(); ++t) {
589  int start = std::max((int)t - (int)timeOffset, 0);
590  int end = std::min((int)t + (int)timeOffset + 1, (int)oldTimePruningMap.size());
591  for (TimeframeIndex tOld = start; tOld < end; ++tOld) {
592  if (oldTimePruningMap[tOld] != Core::Type<Score>::max &&
593  (oldTimePruningMap[tOld] > timeBeamMap[t] || timeBeamMap[t] == Core::Type<Score>::max))
594  timeBeamMap[t] = oldTimePruningMap[tOld];
595  }
596  }
597  }
598 
599  if (beam != Core::Type<Score>::max)
600  beam = beam * scoreFactor + scoreOffset;
601 
602  for (TimeframeIndex t = 0; t < timeBeamMap.size(); ++t)
603  if (timeBeamMap[t] != Core::Type<Score>::max)
604  timeBeamMap[t] = timeBeamMap[t] * scoreFactor + scoreOffset;
605 
606  return true;
607  }
