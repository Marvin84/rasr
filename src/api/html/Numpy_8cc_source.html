59 void initNumpy() {
60  // Init NumPy C-API.
61  // Note that it is important that we do the _import_array() call in the same C unit
62  // where we use the Numpy C-API. This is because it will init a static variable
63  // which lives in this unit.
64  // Other units which would want to call the Numpy C-API directly would thus
65  // also need to call _import_array() somewhere.
66  // However, we should anyway wrap all Numpy calls in this unit.
67  {
68  // _import_array() will overwrite the original exception in case any happens.
69  // We want to see it, thus do the import right here, and in case it fails, print the original exception.
70  PyObject* numpy_mod = PyImport_ImportModule("numpy.core.multiarray");
71  if (!numpy_mod) {
72  Core::printLog("initNumpy: `import numpy.core.multiarray` failed.");
73  goto error;
74  }
75  Py_DECREF(numpy_mod);
76  }
77 
78  if (_import_array() < 0) {
79  Core::printLog("initNumpy: _import_array() failed.");
80  goto error;
81  }
82 
83  return;
84 
85 error:
86  handlePythonError();
87  dumpModulesEnv();
88  criticalError("NumPy init failed");
89 }

99 template<typename NumpyT, typename ContainerT>
100 void numpy2rawMat(PyArrayObject* nparr, ContainerT& nnmat) {
101  require_eq(PyArray_TYPE(nparr), NumpyType<NumpyT>::type());
102  require_eq(PyArray_NDIM(nparr), 2);
103  if (std::is_same<decltype(nnmat.at(0, 0)), NumpyT&>::value &&
104  PyArray_STRIDES(nparr)[0] == sizeof(NumpyT) &&
105  PyArray_STRIDES(nparr)[1] == sizeof(NumpyT) * nnmat.nRows()) {
106  // fast version. we can just use memcpy
107  memcpy(&nnmat.at(0, 0), PyArray_BYTES(nparr), nnmat.nRows() * nnmat.nColumns() * sizeof(NumpyT));
108  }
109  else {
110  for (u32 i = 0; i < nnmat.nRows(); ++i)
111  for (u32 j = 0; j < nnmat.nColumns(); ++j) {
112  NumpyT v = *(NumpyT*)PyArray_GETPTR2(nparr, i, j);
113  nnmat.at(i, j) = (typename ContainerT::value_type)v;
114  }
115  }
116 }

169 template<typename FastContainerType>
170 bool arraySanityChecks_base(CriticalErrorFunc criticalErrorFunc,
171  PyObject* nparr,
172  const FastContainerType& nnContainer,
173  int ndim,
174  const char* func,
175  int numtype = -1) {
176  if (!PyArray_Check(nparr)) {
177  criticalErrorFunc() << Core::form("%s: expected PyArrayObject but got %s", func, nparr->ob_type->tp_name);
178  return false;
179  }
180 
181  if (PyArray_NDIM((PyArrayObject*)nparr) != ndim) {
182  criticalErrorFunc() << Core::form("%s: expect %iD array, got %i dims", func, ndim, PyArray_NDIM((PyArrayObject*)nparr));
183  return false;
184  }
185 
186  // Need this to get well-behaved access to the internal data below.
187  // See docs for PyArray_GETPTR2().
188  if (!PyArray_ISBEHAVED_RO((PyArrayObject*)nparr)) {
189  criticalErrorFunc() << Core::form("%s: expect behaved/standard Numpy array", func);
190  return false;
191  }
192 
193  int npTypeNum = PyArray_TYPE((PyArrayObject*)nparr);
194  if (numtype == -1) {
195  switch (npTypeNum) {
196  case NPY_FLOAT:
197  case NPY_DOUBLE:
198  // ok
199  break;
200  default:
201  criticalErrorFunc() << Core::form("%s: expect float/double Numpy array. got type %i", func, npTypeNum);
202  return false;
203  }
204  }
205  else if (numtype != npTypeNum) {
206  criticalErrorFunc() << Core::form("%s: expect type %i Numpy array. got type %i", func, numtype, npTypeNum);
207  return false;
208  }
209 
210  return true;
211 }

225 template<typename T>
226 bool numpy2nnMatrix(CriticalErrorFunc criticalErrorFunc, PyObject* nparr, Math::CudaMatrix<T>& nnmat) {
227  if (!arraySanityChecks(criticalErrorFunc, nparr, nnmat, 2, "numpy2nnMatrix"))
228  return false;
229 
230  size_t s1 = PyArray_DIM((PyArrayObject*)nparr, 0);
231  size_t s2 = PyArray_DIM((PyArrayObject*)nparr, 1);
232  nnmat.resize(s1, s2);
233 
234  Python::ObjRef _nparr;
235  if (PyArray_TYPE((PyArrayObject*)nparr) != NumpyType<T>::type() ||
236  PyArray_STRIDES((PyArrayObject*)nparr)[0] != sizeof(T) ||
237  PyArray_STRIDES((PyArrayObject*)nparr)[1] != sizeof(T) * nnmat.nRows()) {
238  // numpy2rawMat has a slow fallback which works in all cases.
239  // However, we can speed it up by converting it to our prefered format which we can handle much faster.
240  // So we ultimatively hope that Numpy can do that faster than our slow fallback would be.
241  // Sprint has the matrix data in the format of a Fortran array.
242  _nparr.takeOver(PyArray_FROM_OTF(nparr, NumpyType<T>::type(), NPY_ARRAY_FORCECAST | NPY_ARRAY_F_CONTIGUOUS));
243  }
244  if (!_nparr)
245  _nparr.copyRef(nparr);
246  // Just handle both cases without warning about casts.
247  int npTypeNum = PyArray_TYPE((PyArrayObject*)_nparr.obj);
248  if (npTypeNum == NPY_FLOAT)
249  numpy2rawMat<float, Math::CudaMatrix<T>>((PyArrayObject*)_nparr.obj, nnmat);
250  else if (npTypeNum == NPY_DOUBLE)
251  numpy2rawMat<double, Math::CudaMatrix<T>>((PyArrayObject*)_nparr.obj, nnmat);
252  else
253  assert(false);
254 
255  return true;
256 }
