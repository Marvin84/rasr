44 Fsa::ConstAutomatonRef SearchAlgorithm::Traceback::lemmaAcceptor(Core::Ref<const Bliss::Lexicon> lexicon) const {
45  Bliss::LemmaAcceptor* result = new Bliss::LemmaAcceptor(lexicon);
46  Fsa::State * s1, *s2;
47  s1 = result->newState();
48  result->setInitialStateId(s1->id());
49  for (u32 i = 0; i < size(); ++i)
50  if ((*this)[i].pronunciation) {
51  s2 = result->newState();
52  s1->newArc(s2->id(), result->semiring()->one(), (*this)[i].pronunciation->lemma()->id());
53  s1 = s2;
54  }
55  result->setStateFinal(s1);
56  return Fsa::ConstAutomatonRef(result);
