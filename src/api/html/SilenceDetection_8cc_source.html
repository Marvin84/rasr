24 SilenceDetection::Float SilenceDetection::Frame::getEnergy(const Float energyScalingFactor) const {
25  if (!operator bool())
26  return 0;
27 
28  SilenceDetection::Float energy = 0;
29  for (u32 i = 0; i < get()->size(); i++)
30  energy += (*get())[i] * (*get())[i];
31 
32  energy /= get()->size();
33 
34  if (energy == 0.0)
35  return -10000.0;
36 
37  return 10 * log10(energy) * energyScalingFactor;
38 }

58 void SilenceDetection::updateThreshold() {
59  // if histogram is not representative, move
60  f32 sparseEventRatio = sparseEventRatio_ *
61  (f32)histogramSlidingWindow_.size() / (f32)histogramSlidingWindow_.maxSize();
62 
63  Float noiseFractile = fractile(sparseEventRatio);
64  Float speechFractile = fractile(1.0 - sparseEventRatio);
65 
66  if (speechFractile - noiseFractile > minSnr_) {
67  // histogram has two clear maximum
68  threshold_ = (1.0 - thresholdInterpolationFactor_) * noiseFractile +
69  thresholdInterpolationFactor_ * speechFractile;
70  // for continuoity
71  threshold_ = std::max(threshold_, noiseFractile + minSnr_);
72  }
73  else {
74  // histogram has one maximum, probably just noise
75  threshold_ = noiseFractile + minSnr_;
76  }
77 }

106 SilenceDetection::SilenceType SilenceDetection::isSilence(const SilenceDetection::Float energy) {
107  if (lastDecision_ == silence) {
108  if (energy < threshold_)
109  return silence;
110  else if (nUnsure_ + 1 < minSpeechLength_) {
111  nUnsure_++;
112  return unsure;
113  }
114  else
115  return speech;
116  }
117  else {
118  if (energy >= threshold_)
119  return speech;
120  else if (nUnsure_ + 1 < minSilenceLength_) {
121  nUnsure_++;
122  return unsure;
123  }
124  else
125  return silence;
126  }
127 }

129 bool SilenceDetection::updateDecision(Frame& in, Frame& out) {
130  SilenceType currentDecision = (!in ? lastDecision_ : isSilence(accumulateBlockEnergy_ / blockSlidingWindow_.size()));
131  if (currentDecision != unsure) {
132  u32 i = 1;
133  // set begining of a silence interval to speech
134  if (lastDecision_ == speech && currentDecision == silence)
135  for (; i <= endDelay_; i++)
136  decisionSlidingWindow_[nUnsure_ - i].silence() = speech;
137 
138  //set unsures to the last decision
139  for (; i <= nUnsure_; i++)
140  decisionSlidingWindow_[nUnsure_ - i].silence() = currentDecision;
141 
142  lastDecision_ = currentDecision;
143  nUnsure_ = 0;
144  }
145 
146  in.silence() = currentDecision;
147  if (in)
148  decisionSlidingWindow_.add(in);
149  else
150  decisionSlidingWindow_.flushOut();
151  return decisionSlidingWindow_.out(out);
152 }

168 bool SilenceDetection::init() {
169  for (u32 i = 0; i < energyHistogram_.size(); i++)
170  energyHistogram_[i] = 0;
171  if (!histogramSlidingWindow_.init(histogramSlidingWindowSize_, histogramSlidingWindowRight_))
172  return false;
173 
174  threshold_ = 0;
175 
176  accumulateBlockEnergy_ = 0;
177  if (blockSize_ < 1 || !blockSlidingWindow_.init(blockSize_, blockSize_ / 2))
178  return false;
179 
180  lastDecision_ = silence;
181  nUnsure_ = 0;
182  if (std::max(minSpeechLength_, minSilenceLength_) < 1 ||
183  !decisionSlidingWindow_.init(std::max(minSpeechLength_, minSilenceLength_),
184  std::max(minSpeechLength_, minSilenceLength_) - 1))
185  return false;
186 
187  if (!delaySlidingWindow_.init(std::max(beginDelay_, (u32)1), std::max(beginDelay_, (u32)1) - 1))
188  return false;
189 
190  if (beginDelay_ + endDelay_ > minSilenceLength_)
191  return false;
192 
193  return !(need_init_ = false);
194 }
