60 Flow::Time DataSource::segmentDuration(Bliss::Segment* s) {
61  std::istringstream durationAttr(getAttribute(mainPortId_, "total-duration"));
62  Flow::Time duration;
63  if (durationAttr >> duration)

76 void DataSource::updateProgressStatus(Flow::PortId portId, const Flow::DataPtr<Flow::Data> d) {
77  if (portId == mainPortId_) {
78  Flow::DataPtr<Flow::Timestamp> t(d);
79  if (t) {
80  if (startTime_ > t->startTime())
81  startTime_ = t->startTime();
82  if (endTime_ < t->endTime())
83  endTime_ = t->endTime();
84  if (!noProgressIndication_) {
85  progressIndicator_.notify(int(roundf((endTime_ - startTime_) * 1000.0)));
86  }
87  }
88  }
89  if (portId >= (Flow::PortId)nFrames_.size())
90  nFrames_.resize(portId + 1, 0);
91  ++nFrames_[portId];
92 }
