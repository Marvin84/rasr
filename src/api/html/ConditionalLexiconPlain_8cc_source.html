32 void ConditionalLexiconPlain::read(std::istream& is) {
33  std::string line;
34  unsigned nEntries = 0;
35 
36  while (std::getline(is, line)) {
37  std::vector<std::string> splitLine = Core::split(line, " ");
38  size_t lexiconIndex;
39  Core::strconv(splitLine[0], lexiconIndex);
40 
41  if (lexica_.size() < lexiconIndex + 1) {
42  std::cerr << "resizing lexicon to " << lexiconIndex + 1 << std::endl;
43 
44  //initialize new lexica
45  size_t oldSize = lexica_.size();
46  lexica_.resize(lexiconIndex + 1);
47  for (size_t i = oldSize; i <= lexiconIndex; ++i) {
48  lexica_[i] = LexiconRef(new Lexicon);
49  }
50  std::cerr << "lexica_.size()=" << lexica_.size() << std::endl;
51  }
52 
53  Cost value;
54  Core::strconv(splitLine[1], value);
55 
56  std::vector<Fsa::LabelId> key;
57  for (size_t i = 2; i < splitLine.size(); ++i) {
58  Fsa::LabelId symbol;
59  if (splitLine[i] != "NULL")
60  symbol = tokens_->addSymbol(splitLine[i]);
61  else
62  symbol = Fsa::Epsilon;
63  key.push_back(symbol);
64  }
65 
66  lexica_[lexiconIndex]->store(key, value);
67  ++nEntries;
68  }
69  std::cerr << "read " << nEntries << " entries" << std::endl;
70 }
