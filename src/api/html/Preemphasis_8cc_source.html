51 void Preemphasis::apply(Flow::Vector<f32>& v) {
52  ensure(sampleRate_);
53 
54  if (needInit_ || !v.equalsToStartTime(previousEndTime_))
55  init(v.empty() ? 0 : v[0]);
56 
57  f32 current;
58  if (alpha_ != 1.0) {
59  for (u32 i = 0; i < v.size(); i++) {
60  current = v[i];
61  v[i] -= alpha_ * previous_;
62  previous_ = current;
63  }
64  }
65  else {
66  current = previous_;
67  previous_ = v[v.size() - 1];
68  for (u32 i = v.size() - 1; i > 0; i--)
69  v[i] -= v[i - 1];
70  v[0] -= current;
71  }
72 
73  previousEndTime_ = v.endTime();
74 }
