42 struct hash_LabelIdPair {
43  size_t operator()(const LabelIdPair& x) const {
44  size_t return_value = 23;
45  hash<unsigned int> f;
46 
47  size_t hash_value = f(x.first);
48  return_value ^= hash_value;
49  return_value += (hash_value << (return_value & 7));
50  return_value ^= ((x.first) << ((return_value >> 3) & 15));
51  hash_value = f(x.second);
52  return_value ^= hash_value;
53  return_value += (hash_value << (return_value & 7));
54  return_value ^= ((x.second) << ((return_value >> 3) & 15));
55 
56  return return_value;
57  }
58 };

98  virtual Translation::Cost getCost(const size_t index, const std::vector<Fsa::LabelId>& key) const {
99  VocabIndex tupleIndex(mapToTupleIndex_(key[0], key[1]));
100  VocabIndexHistory tupleHistory(lmOrder_ - 1);
101 
102  if (key.size() > 2) {
103  std::vector<VocabIndex> sriKey;
104  size_t keyIndex(2);
105  while (keyIndex < key.size()) {
106  if (key[keyIndex] == fsaSentenceBeginSymbol_ || key[keyIndex + 1] == fsaSentenceBeginSymbol_) {
107  sriKey.push_back(sentenceBeginIndex_);
108  break;
109  }
110  else {
111  sriKey.push_back(mapToTupleIndex_(key[keyIndex], key[keyIndex + 1]));
112  }
113  keyIndex += 2;
114  }
115  tupleHistory.expand(sriKey);
116  }
117  else {
118  tupleHistory.expandEmpty();
119  }
120 
121  return -srilm_.wordProb(tupleIndex, tupleHistory.rbegin());
122  }
