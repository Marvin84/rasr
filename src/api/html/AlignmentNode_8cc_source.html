160 void AlignmentNode::createModel() {
161  verify(modelCache_);
162  verify(allophoneStateGraphBuilder_);
163 
164  Fsa::ConstAutomatonRef transducer; // allophone state to lemma-pronunciation transducer
165  if (wordLatticeBuilder_) {
166  if (phonemeSequenceSet_ != orthography) {
167  criticalError("model type must be forced-alignment");
168  }
169  if (tracebackArchiveReader_) {
170  warning("transducer will be overwritten");
171  }
172  verify(transducerCache_);
173  transducer = transducerCache_->get(allophoneStateGraphBuilder_->createTransducerFunctor(segmentId_, orthography_, leftContextOrthography_, rightContextOrthography_));
174  wordLatticeBuilder_->setModelTransducer(transducer);
175  }
176 
177  Fsa::ConstAutomatonRef model = modelCache_->find(segmentId_);
178  if (!model) {
179  if (tracebackArchiveReader_) {
180  Lattice::ConstWordLatticeRef lattice = tracebackArchiveReader_->get(segmentId_);
181  transducer = allophoneStateGraphBuilder_->buildTransducer(
182  Fsa::projectInput(
183  Fsa::composeMatching(Fsa::invert(lemmaPronunciationToLemma_),
184  lattice->part(Lattice::WordLattice::acousticFsa))));
185  }
186  if (transducer) {
187  model = modelCache_->get(allophoneStateGraphBuilder_->createFinalizationFunctor(segmentId_, transducer));
188  }
189  else if (phonemeSequenceSet_ == orthography) {
190  model = modelCache_->get(allophoneStateGraphBuilder_->createFunctor(segmentId_, orthography_, leftContextOrthography_, rightContextOrthography_));
191  }
192  else if (phonemeSequenceSet_ == lemmaLoop || phonemeSequenceSet_ == phoneLoop) {
193  model = aligner_.getModel();
194  if (!model) {
195  if (phonemeSequenceSet_ == lemmaLoop) {
196  model = modelCache_->get(allophoneStateGraphBuilder_->createFunctor(segmentId_, AllophoneStateGraphBuilder::lemma));
197  }
198  else {
199  model = modelCache_->get(allophoneStateGraphBuilder_->createFunctor(segmentId_, AllophoneStateGraphBuilder::phone));
200  }
201  }
202  }
203  else {
204  criticalError("cannot create/set model");
205  }
206  }
207  else if (tracebackArchiveReader_) {
208  warning("Using model acceptor from cache. Dependencies on Traceback Archives are not handled.");
209  }
210  acousticModel_->setKey(segmentId_);
211  aligner_.setModel(model, acousticModel_);
212 }

226 
227  verify(!modelCache_);

229  modelCache_->setDependencies(dependencies);
230  modelCache_->respondToDelayedErrors();
231 
232  verify(!transducerCache_);
233  verify(!latticeArchiveWriter_);
234  verify(!tracebackArchiveReader_);
235  if (tracebackChannel_.isOpen() || paramStoreLattices(config)) {
236  wordLatticeBuilder_ = new Aligner::WordLatticeBuilder(select("word-lattice-builder"),
237  modelCombination.lexicon(),

239 
240  if (paramStoreLattices(config)) {

243  if (latticeArchiveWriter_->hasFatalErrors()) {
244  delete latticeArchiveWriter_;
245  latticeArchiveWriter_ = 0;
246  }
247  }
248  Core::DependencySet dependencies;
249  modelCombination.getDependencies(dependencies);
250  transducerCache_ = new FsaCache(select("transducer-cache"), Fsa::storeStates);
251  transducerCache_->setDependencies(dependencies);
252  transducerCache_->forceInputAlphabet(acousticModel_->allophoneStateAlphabet());
253  transducerCache_->respondToDelayedErrors();
254  }
255 
256  if (paramUseTracebacks(config)) {
257  log("opening traceback archive");
258  tracebackArchiveReader_ = Lattice::Archive::openForReading(select("traceback-archive"),
259  modelCombination.lexicon());
260  if (tracebackArchiveReader_->hasFatalErrors()) {
261  delete tracebackArchiveReader_;
262  tracebackArchiveReader_ = 0;
263  }
264  lemmaPronunciationToLemma_ = modelCombination.lexicon()->createLemmaPronunciationToLemmaTransducer();
265  }
266 
267  writeAlphabet_ = paramWriteAlphabet(config);
268 
269  needInit_ = false;
270 }

272 bool AlignmentNode::work(Flow::PortId p) {
273  Flow::DataAdaptor<Alignment>* alignment = new Flow::DataAdaptor<Alignment>();
274  alignment->invalidateTimestamp();
275 
276  if (writeAlphabet_)
277  alignment->data().setAlphabet(acousticModel_->allophoneStateAlphabet());
278 
279  bool firstFeature = true;
280  Flow::DataPtr<Feature::FlowFeature> in;
281  std::vector<Mm::FeatureScorer::Scorer> scorers;
282  // reset feature scorer for usage with embedded flow files
283  acousticModel_->featureScorer()->reset();
284  auto featureScorer = acousticModel_->featureScorer();
285  while (getData(0, in)) {
286  Core::Ref<const Feature> feature(new Feature(in));
287  if (firstFeature) {
288  checkFeatureDependencies(*feature);
289  firstFeature = false;
290  }
291  if (featureScorer->isBuffered() && !featureScorer->bufferFilled()) {
292  featureScorer->addFeature(feature);
293  }
294  else {
295  scorers.push_back(featureScorer->getScorer(feature));
296  }
297  alignment->expandTimestamp(feature->timestamp());
298  if (tracebackChannel_.isOpen())
299  featureTimes_.push_back(feature->timestamp());
300  }
301 
302  if (featureScorer->isBuffered()) {
303  while (!featureScorer->bufferEmpty()) {
304  scorers.push_back(featureScorer->flush());
305  }
306  }
307 
308  // finalize embedded network if applicable i.e. EOS
309  if (firstFeature) {
310  acousticModel_->featureScorer()->finalize();
311  }
312 
313  aligner_.feed(scorers);
314  if (!aligner_.reachedFinalState())
315  warning("Alignment did not reach any final state in segment '%s'", segmentId_.c_str());
316 
317  Fsa::ConstAutomatonRef alignmentFsa = aligner_.getAlignmentFsa();
318  if (aligner_.reachedFinalState() && (paramStoreLattices(config) || tracebackChannel_.isOpen()))
319  createWordLattice(alignmentFsa);
320 
321  std::pair<Fsa::ConstAutomatonRef, Fsa::Weight> alignmentPosteriorFsa = aligner_.getAlignmentPosteriorFsa(alignmentFsa);
322  aligner_.getAlignment(alignment->data(), alignmentPosteriorFsa);
323 
324  featureTimes_.clear();
325 
326  return putData(0, alignment) && putData(0, in.get());
327 }

353 void AlignmentNode::logTraceback(Lattice::ConstWordLatticeRef wordLattice) const {
354  Fsa::ConstAutomatonRef fn = wordLattice->part(Lattice::WordLattice::acousticFsa);
355  Fsa::ConstStateRef state = fn->getState(fn->initialStateId());
356  Fsa::ConstAlphabetRef ia = fn->getInputAlphabet();
357 
358  u32 previousIndex = wordLattice->time(state->id());
359  tracebackChannel_ << Core::XmlOpen("traceback") + Core::XmlAttribute("type", "xml") + Core::XmlAttribute("segment", segmentId_);
360  while (!state->isFinal()) {
361  u32 index = wordLattice->time(state->begin()->target());
362  tracebackChannel_ << Core::XmlOpen("item") + Core::XmlAttribute("type", "lemma-pronunciation")
363  << Core::XmlFull("lemma-pronunciation", ia->symbol(state->begin()->input()))
364  << Core::XmlFull("score", f32(state->begin()->weight())) + Core::XmlAttribute("type", "acoustic")
365  << Core::XmlEmpty("samples") + Core::XmlAttribute("start", f32(featureTimes_[previousIndex].startTime())) +
366  Core::XmlAttribute("end", f32(featureTimes_[index - 1].endTime()))
367  << Core::XmlEmpty("features") + Core::XmlAttribute("start", previousIndex) +
368  Core::XmlAttribute("end", index - 1)
369  << Core::XmlClose("item");
370  previousIndex = wordLattice->time(state->begin()->target());
371  state = fn->getState(state->begin()->target());
372  }
373  tracebackChannel_ << Core::XmlClose("traceback");
374 }

497 bool AlignmentDumpNode::createContext(const std::string& id) {
498  reader_ = 0;
499 
500  // Open archive first only for reading. independent if node has an input.
501  if (!hasAccess(Core::Archive::AccessModeRead)) {
502  if (!open(Core::Archive::AccessModeRead)) {
503  // Report error only if node has no input.
504  if (!inputConnected(0)) {
505  error("Failed to open archive '%s' for reading and "
506  "node has no input.",
507  filename_.c_str());
508  return false;
509  }
510  }
511  }
512  archiveExists_ = false;
513  if (hasAccess(Core::Archive::AccessModeRead))
514  archiveExists_ = archive_->hasFile(segmentId_ = id);
515  if (archiveExists_) {
516  return true;
517  }
518  else if (inputConnected(0)) {
519  // Open archive for writing only on demand.
520  if (!hasAccess(Core::Archive::AccessModeWrite)) {
521  if (!open(Core::Archive::AccessModeRead | Core::Archive::AccessModeWrite)) {
522  error("Failed to open archive '%s'.", filename_.c_str());
523  return false;
524  }
525  }
526  verify(hasAccess(Core::Archive::AccessModeWrite));
527  writer_ = newWriter(id);
528  if (!writer_) {
529  error("Failed to prepare writing to cache for id \"%s\".",
530  id.c_str());
531  }
532  return true;
533  }
534  error("Cannot provide data for id \"%s\": "
535  "Neither archive entry nor input available.",
536  id.c_str());
537  return false;
538 }

551 bool AlignmentDumpNode::work(Flow::PortId p) {
552  // converts standard alignment to plain text alignment
553  if (alignmentType_ == standard) {
554  Flow::DataPtr<Flow::DataAdaptor<Alignment>> in;
555 
556  if (!writer_) {
557  if (segmentId_.empty())
558  criticalError("File not given.");
559  }
560  Core::TextOutputStream o(writer_);
561  require(o && o.good());
562  o.setEncoding("UTF-8");
563 
564  while (getData(0, in)) {
565  Alignment& a = in->data();
566 
567  // dump segment information
568  std::string plainData = "";
569 
570  TimeframeIndex time = 0;
571  Alignment::const_iterator i, j;
572 
573  // dump alignment
574  std::vector<Alignment::Frame> frames;
575  a.getFrames(frames);
576 
577  for (std::vector<Alignment::Frame>::const_iterator i = frames.begin(); i != frames.end(); ++i) {
578  // print feature and feature times if the input link is connected
579  if (inputConnected(1)) {
580  Flow::DataPtr<Feature::FlowFeature> f;
581  if (getData(1, f)) {
582  plainData = Core::form("%i %.3f %.3f ", time, f->startTime(), f->endTime());
583  if (writer_->isOpen()) {
584  o << plainData;
585  if (!o)
586  error("could not write data \"%s\" to archive \"%s\"", segmentId_.c_str(),
587  archive_->path().c_str());
588  }
589  else {
590  error("could not open file \"%s\" for writing in archive \"%s\"",
591  segmentId_.c_str(), archive_->path().c_str());
592  }
593  }
594  else {
595  error("Feature stream and alignment not synchronized at time %d.", time);
596  }
597  }
598 
599  // print alignment items
600  Alignment::iterator j, j_end;
601  for (Core::tie(j, j_end) = *i; j != j_end; ++j) {
602  plainData = Core::form("%s %g\n",
603  acousticModel_->allophoneStateAlphabet()->symbol(j->emission).c_str(),
604  j->weight);
605  if (writer_->isOpen()) {
606  o << plainData;
607  if (!o)
608  error("could not write data \"%s\" to archive \"%s\"", segmentId_.c_str(),
609  archive_->path().c_str());
610  }
611  else {
612  error("could not open file \"%s\" for writing in archive \"%s\"",
613  segmentId_.c_str(), archive_->path().c_str());
614  }
615  }
616  ++time;
617  }
618  putData(0, in.get());
619  }
620  o.close();
621 
622  if (inputConnected(1)) {
623  Flow::DataPtr<Feature::FlowFeature> f;
624  if (getData(1, f))
625  warning("Feature stream and alignment not synchronized: There are features remaining.");
626  }
627  return putData(0, in.get());
628 
629  //converts plain Text alignments to standard alignments
630  }
631  else if (alignmentType_ == plainText) {
632  if (!reader_)
633  reader_ = newReader(segmentId_);
634 
635  if (!reader_) {
636  if (segmentId_.empty())
637  criticalError("File not given.");
638  else
639  return false;
640  }
641 
642  Core::TextInputStream i(reader_);
643 
644  require(i);
645  require(i.good());
646  i.setEncoding("UTF-8");
647 
648  Flow::DataPtr<Flow::DataAdaptor<std::string>> in;
649  Alignment a;
650 
651  TimeframeIndex timeFrame = 0;
652  f64 weight = 0;
653  f64 startTime = 0.0;
654  f64 endTime = 0.0;
655  std::string allophoneString = "";
656 
657  f64 previousStartTime = 0.0;
658  f64 previousEndTime = 0.0;
659  f64 timeStampStart = 0.0;
660  f64 timeStampEnd = 0.0;
661  bool firstSegment = true;
662  bool singleSegment = true;
663 
664  std::string line;
665 
666  while (Core::getline(i, line) != EOF) {
667  previousEndTime = endTime;
668
