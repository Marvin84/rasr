35  * checks for each state if it is reachable by an imbalanced silence arc,
36  * i.e. an arc with silence output label but input label != silence.

38 template<class A>
39 class SilenceLabelVisitor {
40 public:
41  typedef A Arc;
42  typedef typename Arc::StateId StateId;
43  typedef typename Arc::Label Label;
44 
45  SilenceLabelVisitor(Label ilabel, Label olabel,
46  std::vector<bool>* silence_states)
47  : states_(silence_states), ilabel_(ilabel), olabel_(olabel) {}
48 
49  void InitVisit(const FstLib::Fst<Arc>& fst) {}
50  bool InitState(StateId s, StateId root) {
51  return true;
52  }
53  bool TreeArc(StateId s, const Arc& a) {
54  SetArc(a);
55  return true;
56  }
57  bool BackArc(StateId s, const Arc& a) {
58  SetArc(a);
59  return true;
60  }
61  bool ForwardOrCrossArc(StateId s, const Arc& a) {
62  SetArc(a);
63  return true;
64  }
65  void FinishState(StateId s, StateId parent, const Arc* parent_arc) {}
66  void FinishVisit() {}
67 
68 private:
69  void SetArc(const Arc& a) {
70  if (a.ilabel != ilabel_ && a.olabel == olabel_) {
71  if (a.nextstate >= states_->size())
72  states_->resize(a.nextstate + 1, false);
73  states_->at(a.nextstate) = true;
74  }
75  }
76  std::vector<bool>* states_;
77  Label ilabel_, olabel_;
78 };

198 void LatticeTraceRecorder::shortestPath(BestPath* path) const {
199  path->clear();
200  Lattice best;
201  FstLib::ShortestPath(*lattice_, &best);
202  StateId s = best.Start();
203  Weight sum = Weight::One();
204  while (s != OpenFst::InvalidStateId) {
205  verify_le(best.NumArcs(s), 1);
206  ArcIterator aiter(best, s);
207  if (aiter.Done()) {
208  s = OpenFst::InvalidStateId;
209  }
210  else {
211  sum = Times(sum, aiter.Value().weight);
212  if (aiter.Value().olabel)
213  path->append(aiter.Value().olabel, 0,
214  BestPath::ScoreVector(sum.AmScore(), sum.LmScore()));
215  s = aiter.Value().nextstate;
216  }
217  }
218  path->append(0, 0, BestPath::ScoreVector(sum.AmScore(), sum.LmScore()));
219 }

281  * Similar to FstLib::Connect, but keeps the StateInfo.
282  * Generates an intermediate copy of lattice_

284 void HmmLatticeTraceRecorder::trimLattice() {
285  std::vector<bool> access, coaccess;
286  uint64 props;
287  FstLib::SccVisitor<Arc> scc_visitor(0, &access, &coaccess, &props);
288  FstLib::DfsVisit(*lattice_, &scc_visitor);
289  std::vector<StateId> newid(lattice_->NumStates(), 0);
290  Lattice newLattice;
291  for (StateIterator siter(*lattice_); !siter.Done(); siter.Next()) {
292  if (!access[siter.Value()] || !coaccess[siter.Value()]) {
293  newid[siter.Value()] = OpenFst::InvalidStateId;
294  }
295  else {
296  StateId ns = newLattice.AddState();
297  newid[siter.Value()] = ns;
298  verify_(ns <= siter.Value());
299  states_[ns] = states_[siter.Value()];
300  }
301  }
302  for (StateIterator siter(*lattice_); !siter.Done(); siter.Next()) {
303  StateId s = newid[siter.Value()];
304  if (s != OpenFst::InvalidStateId) {
305  newLattice.SetFinal(s, lattice_->Final(siter.Value()));
306  newLattice.ReserveArcs(s, lattice_->NumArcs(siter.Value()));
307  for (ArcIterator aiter(*lattice_, siter.Value()); !aiter.Done(); aiter.Next()) {
308  Arc arc = aiter.Value();
309  arc.nextstate = newid[aiter.Value().nextstate];
310  if (arc.nextstate != OpenFst::InvalidStateId)
311  newLattice.AddArc(s, arc);
312  }
313  }
314  }
315  newLattice.SetStart(newid[lattice_->Start()]);
316  verify_ne(newid[lattice_->Start()], OpenFst::InvalidStateId);
317  *lattice_ = newLattice;
318 }

320 TraceRecorder::TraceRef HmmLatticeTraceRecorder::addTrace(
321  TraceRef sibling, TraceRef predecessor, OpenFst::Label output,
322  const StateSequence* hmm, TimeframeIndex time,
323  Score score, Score arcScore, bool wordEnd) {
324  StateId state = (sibling == InvalidTraceRef ? getState(time) : sibling);
325  verify_(states_[state].time == time);
326  StateId prev = (predecessor == InvalidTraceRef ? lattice_->Start() : predecessor);
327  const Score threshold = states_[state].score + pruningThreshold_;
328  if (score > threshold) {
329  // anticipated pruning of arcs
330  verify_(sibling != InvalidTraceRef);
331  return state;
332  }
333  // create reverse arc
334  Label input = getInputLabel(hmm);
335  const Score relScore = score - states_[prev].score;
336  Arc newArc(input, output, Weight(relScore - arcScore, arcScore), prev);
337  // try to merge epsilon arcs as soon as possible
338  if ((newArc.ilabel == OpenFst::Epsilon)) {
339  if (lattice_->NumArcs(prev) == 1)
340  mergeEpsilonArc(&newArc);
341  if (newArc.ilabel == OpenFst::Epsilon)
342  hasEps_[state] = true;
343  }
344  else if (hasEps_[newArc.nextstate] && lattice_->NumArcs(newArc.nextstate) == 1) {
345  mergeEpsilonArc(&newArc);
346  }
347  bool add = true;
348  // merge silence arcs
349  if (mergeSilence_ && newArc.ilabel == silence_)
350  add = !mergePredecessorArcs(state, newArc, score);
351  if (add)
352  addArc(state, newArc, score);
353  return state;
354 }

356 void HmmLatticeTraceRecorder::mergeEpsilonArc(Arc* arc) const {
357  ArcIterator aiter(*lattice_, arc->nextstate);
358  verify_(!aiter.Done());
359  const Arc& prevArc = aiter.Value();
360  if (!(arc->olabel && prevArc.olabel)) {
361  if (!arc->olabel)
362  arc->olabel = prevArc.olabel;
363  if (!arc->ilabel)
364  arc->ilabel = prevArc.ilabel;
365  arc->nextstate = prevArc.nextstate;
366  arc->weight = Times(prevArc.weight, arc->weight);
367  }
368 }

370 bool HmmLatticeTraceRecorder::mergePredecessorArcs(StateId state, const Arc& arc, Score score) {
371  bool merged = false;
372  for (ArcIterator aiter(*lattice_, arc.nextstate); !aiter.Done(); aiter.Next()) {
373  const Arc& prevArc = aiter.Value();
374  if (prevArc.ilabel == arc.ilabel) {
375  Arc newArc = arc;
376  newArc.nextstate = prevArc.nextstate;
377  newArc.weight = Times(prevArc.weight, arc.weight);
378  if (!(prevArc.olabel && arc.olabel) || prevArc.olabel == arc.olabel) {
379  if (!arc.olabel || arc.olabel == silenceOutput_)
380  newArc.olabel = prevArc.olabel;
381  addArc(state, newArc);
382  merged = true;
383  }
384  }
385  }
386  return merged;
387 }

397 void HmmLatticeTraceRecorder::createBestPath(const WordEndDetector& wordEnds, bool ignoreLast,
398  TraceRef end, BestPath* path) {
399  verify(!finished_);
400  TraceRef trace = end;
401  path->clear();
402  path->append(OpenFst::Epsilon, states_[end].time,
403  BestPath::ScoreVector(0, states_[end].score));
404  std::deque<Label> outputs;
405  std::deque<BestPath::Item> timeAndScore;
406  bool ignoreOutput = ignoreLast;
407  while (trace != lattice_->Start()) {
408  ArcIterator aiter(*lattice_, trace);
409  aiter.Seek(states_[trace].bestArc);
410  verify(!aiter.Done());
411  const Arc& arc = aiter.Value();
412  const StateSequence* hmm = (arc.ilabel ? getHmm(arc.ilabel) : 0);
413  const Score score = states_[trace].score + arc.weight.Combined().Value();
414  if (hmm && wordEnds.isNonWord(hmm)) {
415  path->append(OpenFst::Epsilon, states_[trace].time, score);
416  }
417  else if (hmm && wordEnds.isWordEnd(*hmm, arc.olabel))
418  timeAndScore.push_back(BestPath::Item(OpenFst::Epsilon, states_[trace].time, score));
419  if (arc.olabel != OpenFst::Epsilon) {
420  if (!ignoreOutput)
421  outputs.push_back(arc.olabel);
422  else
423  ignoreOutput = false;
424  }
425  if (!outputs.empty() && !timeAndScore.empty()) {
426  BestPath::Item item = timeAndScore.front();
427  /* TODO: The loop below is a bugfix, maybe not the best one.
428  * There are multiple arcs (in case of character LM) for one word end.
429  * Without the fix some of the arcs are not printed and the beginning of the sentence is truncated.
430  * [kozielski]
431  */
432  while (outputs.size()) {
433  item.word = outputs.front();
434  path->append(item);
435  outputs.pop_front();
436  }
437  timeAndScore.pop_front();
438  }
439  trace = arc.nextstate;
440  }
441  std::reverse(path->begin(), path->end());
442 }

522 /**
523  * calculate for each states the difference in score between the best path
524  * through the state and the overall best path (w.r.t the given final states)
525  */
526 void HmmLatticeTraceRecorder::calculatePruningScores(const std::vector<TraceRef>& finalStates) {
527  std::vector<bool> enqueued(active_.size(), false);
528  std::vector<bool>& visited = active_;
529  std::fill(visited.begin(), visited.end(), false);
530  for (std::vector<TraceRef>::const_iterator i = finalStates.begin(); i != finalStates.end(); ++i) {
531  visited[*i] = true;
532  enqueued[*i] = true;
533  }
534  std::deque<StateId> queue(finalStates.begin(), finalStates.end());
535  while (!queue.empty()) {
536  StateId s = queue.front();
537  queue.pop_front();
538  verify_(s < enqueued.size());
539  enqueued[s] = false;
540  verify_(visited[s]);
541  for (ArcIterator aiter(*lattice_, s); !aiter.Done(); aiter.Next()) {
542  const Arc& arc = aiter.Value();
543  StateId ns = arc.nextstate;
544  Score diff = arc.weight.Combined().Value() + states_[ns].score + states_[s].diff - states_[s].score;
545  if (diff < 0.0)
546  diff = 0;
547  if (visited.size() <= ns) {
548  visited.resize(ns + 1, false);
549  enqueued.resize(ns + 1, false);
550  }
551  if (!visited[ns] || diff < states_[ns].diff) {
552  // initialize or update score for ns
553  states_[ns].diff = diff;
554  visited[ns] = true;
555  if (!enqueued[ns]) {
556  enqueued[ns] = true;
557  queue.push_back(ns);
558  }
559  }
560  }
561  }
562 }

564 void HmmLatticeTraceRecorder::prune(const std::vector<TraceRef>& finalStates) {
565  require(!finalStates.empty());
566  calculatePruningScores(finalStates);
567  std::deque<StateId> queue(finalStates.begin(), finalStates.end());
568  std::vector<bool>& visited = active_;
569  std::fill(visited.begin(), visited.end(), false);
570  while (!queue.empty()) {
571  StateId s = queue.front();
572  queue.pop_front();
573  MutableArcIterator oiter(lattice_, s);
574  for (ArcIterator aiter(*lattice_, s); !aiter.Done(); aiter.Next()) {
575  const Arc& arc = aiter.Value();
576  StateId ns = arc.nextstate;
577  verify_(ns < visited.size());
578  Score diff = arc.weight.Combined().Value() + states_[ns].score + states_[s].diff - states_[s].score;
579  if (diff >= pruningThreshold_) {
580  // prune arc
581  if (oiter.Position() < states_[s].bestArc) {
582  // adjust pointer to best arc
583  --states_[s].bestArc;
584  }
585  continue;
586  }
587  if (oiter.Position() != aiter.Position())
588  oiter.SetValue(aiter.Value());
589  oiter.Next();
590  if (!visited[ns]) {
591  visited[ns] = true;
592  queue.push_back(ns);
593  }
594  }
595  size_t deletedArcs = lattice_->NumArcs(s) - oiter.Position();
596  if (deletedArcs)
597  lattice_->DeleteArcs(s, deletedArcs);
598  }
599  // non-reachable states will be removed by the next purge() call
600 }
