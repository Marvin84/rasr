293  /** Creates an object implementing the Mel warping function
294  * f_{Mel} = 2595 * log10 (1 + f / 700)
295  *
296  * Discrete input:
297  * i_{Mel} = 2595 * log10 (1 + i / frequencySampleRate / 700) * frequencySampleRate
298  *
299  * if @param discretizeArgument is true then floorf is applied to the frequency:
300  * -'i / frequencySampleRate' replaced by 'floorf(i / frequencySampleRate)'
301  * -floorf gives less skipped warped indices
302  * -Acctually applied for combatibility with the old standard system.
303  *
304  * Note: sampleRate has to be initialized by sampling rate on frequency axis:
305  * i.e.: #FFT-points / time-domain-sampling-rate.
306  *
307  * Normalized omega input:
308  * omega_{Mel} = 2595 * log10 (1 + omega * sampleRate / (2 * pi) / 700) * (2 * pi) / sampleRate
309  * Note: sampleRate is the one in time domain.
310  */
311  UnaryAnalyticFunctionRef createMelWarpingFunction(bool discretizeArgument = false) const;
312  /** Creates an object implementing the Bark warping function
313  * Continuous input:
314  * f_{Bark} = 6 * ln(f / 600 + sqrt((f / 600)^2 + 1)) = 6 * asinh(f / 600);
315  * Remark: @see ArchSinh
316  * Discrete input:
317  * i_{Bark} = 6 * asinh(i / frequencySampleRate / 600) * frequencySampleRate
318  * Normalized omega input:
319  * omega_{Mel} = 6 * asinh(omega * sampleRate / (2 * pi) / 600) * (2 * pi) / sampleRate
320  * Note: sampleRate is the one in time domain.
321  */
322  UnaryAnalyticFunctionRef createBarkWarpingFunction() const;
323  /** Two piece linear warping function
324  *
325  * a <= 1:
326  * f < limit : f_warped = a * f
327  * f >= limit : f_warped = (1.0 - a * limit) / (1.0 - limit) * f
328  *
329  * a > 1
330  * f < limit / a : f_warped = a * f
331  * f >= limit / a : (a - a * limit) / (a - limit) * f
332  * Note: this case is equal to the mathematical inverse of the warping function set up
333  * with (limit, 1/a).
334  */
335  UnaryAnalyticFunctionRef createTwoPieceLinearFunction(UnaryAnalyticFunction::Argument warpingFactor,
336  UnaryAnalyticFunction::Argument limit) const;
337  /** Bilinear Warping Function.
338  * x_{B} = x - T / pi * arctan(a * sin(2 * pi / T * x) / (1 + a * cos(2 * pi / T * x))),
339  * where a = 1 - warping-factor, and T/2 is the point which is mapped on itself.
340  */
341  UnaryAnalyticFunctionRef createBilinearWarpingFunction(UnaryAnalyticFunction::Argument warpingFactor) const {
342  return UnaryAnalyticFunctionRef(
343  new BilinearTransform(1 - warpingFactor, 2 * maximalArgument_));
344  }
345  /** Three piece affine warping function
346  *
347  * a <= 1:
348  * f < lowerLimit : f_warped = ((a * lowerLimit + aShift * (a - 1))/lowerLimit) * f
349  * lowerLimit <= f < upperLimit : f_warped = a * f + aShift * (a-1)
350  * f >= upperLimit : f_warped = m * f + c
351  * where m = (1 - (a * upperLimit + aShift * (a - 1)))(1 - upperLimit)
352  *
353  * a > 1
354  * Note: this case is equal to the mathematical inverse of the warping function set up
355  * with (lowerLimit, upper_limit, 1/a).
356  */
357  UnaryAnalyticFunctionRef createThreePieceAffineFunction(UnaryAnalyticFunction::Argument warpingFactor,
358  UnaryAnalyticFunction::Argument aShift,
359  UnaryAnalyticFunction::Argument lowerLimit,
360  UnaryAnalyticFunction::Argument upperLimit) const;
