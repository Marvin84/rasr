22 /**
23  * Interface for performing a search for the best string in an Fsa w.r.t. minimum Bayes risk.
24  * This kind of search takes only an Fsa and a configuration as an argument.
25  * For statistics purposes, the number of evaluated hypotheses and the number of computations
26  * performed during the search can be optionally computed by the implementing classes.
27  *
28  */

130 /**
131  * The standard approach to speech recognition is based on a 0-1-loss Bayes decision rule.
132  * This always chooses the hypothesis with the maximum a-posteriori probability and hence
133  * the best hypothesis will always be the path in the Fsa with the highest or lowest score
134  * (depending on how the scores i.e. the semiring is defined).
135  */
136 class MinimumBayesRiskMapSearch : public MinimumBayesRiskSearch {
137 public:
138  /**
139  * The only arguments a minimum Bayes risk search takes are a configuration.
140  * The automaton being searched will be a parameter of the execution of the search.
141  *
142  * @param config configuration of the search.
143  */
144  MinimumBayesRiskMapSearch(const Core::Configuration&);
145 
146  /**
147  * As the optimal hypothesis is the hypothesis with the highest/lowest score,
148  * only Fsa::best will be performed.
149  *
150  * @param fsa the automaton to be searched.
151  */
152  virtual void performSearch(Fsa::ConstAutomatonRef fsa);
153 };
