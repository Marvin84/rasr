63 class SparseVector {
64 public:
65  T& operator[](size_t idx) {
66  require_lt(idx, items_.size());
67  require(used_[idx]);
68  return items_[idx];
69  }

73  require(used_[idx]);

77  size_t insert(T const& item) {
78  if (freeList_.empty()) {
79  size_t idx = items_.size();
80  items_.push_back(item);
81  used_.push_back(true);
82  return idx;
83  }
84  else {
85  size_t idx = freeList_.top();
86  require(not used_[idx]);
87  items_[idx] = item;
88  used_[idx] = true;
89  freeList_.pop();
90  return idx;
91  }
92  }
93 
94  void erase(size_t idx) {
95  require_lt(idx, items_.size());
96  if (used_[idx]) {
97  items_[idx] = T();
98  used_[idx] = false;
99  freeList_.push(idx);
100  }
101  }
102 
103  void clear() {
104  items_.clear();
105  used_.clear();
106  while (not freeList_.empty()) {
107  freeList_.pop();
108  }
109  }
110 
111  void filter(std::vector<bool> const& keep) {
112  require_le(keep.size(), items_.size());
113  size_t i = 0ul;
114  for (; i < keep.size(); i++) {
115  if (not keep[i]) {
116  erase(i);
117  }
118  }
119  for (; i < items_.size(); i++) {
120  erase(i);
121  }
122  }
123 
124  size_t pos(T const* item) const {
125  return std::distance(items_.data(), item);
126  }
127 
128  size_t size() const {
129  return items_.size() - freeList_.size();
130  }
131 
132  size_t storageSize() const {
133  return items_.size();
134  }
135 
136 private:
137  std::vector<T> items_;
138  std::vector<bool> used_;

235  /// Modifies the given trace-id with a specific value. The value can later be retrieved through
236  /// getModification(traceid), where traceid is the returned value.
237  TraceId modify(TraceId trace, u32 value, u32 value2 = 0, u32 value3 = 0) {
238  verify_(trace != invalidTraceId);
239  verify_(!isModified(trace));
240  TraceId ret;
241 
242  value += 1; // Offset by 1, so we can also modify with 0
243 
244 #ifdef FAST_TRACE_MODIFICATION
245  if (value < 255 && value2 == 0) {
246  ret = (value << 24) | trace;
247  }
248  else {
249 #endif
250  ret = modifications_.insert(std::make_pair<u32, Modification>((u32)trace, Modification(value, value2, value3)));
251  ret |= ModifyMask;
252 #ifdef FAST_TRACE_MODIFICATION
253  }
254 #endif
255 
256  verify_(getModification(ret) == Modification(value - 1, value2, value3));
257  return ret;
258  }
