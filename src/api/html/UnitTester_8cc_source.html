98 CppUnit::Test* UnitTester::findTest(CppUnit::Test* root, const std::string& name) {
99  std::deque<CppUnit::Test*> to_visit;
100  to_visit.push_back(root);
101  while (!to_visit.empty()) {
102  CppUnit::Test* t = to_visit.front();
103  to_visit.pop_front();
104  if (t->getName() == name)
105  return t;
106  for (int i = 0; i < t->getChildTestCount(); ++i) {
107  to_visit.push_back(t->getChildTestAt(i));
108  }
109  }
110  return 0;
111 }

127  CppUnit::Test* test = findTest(root, *t);
128  if (!test)
