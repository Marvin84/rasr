35 /* These are the four functions used in the four steps of the MD5 algorithm
36  and defined in the RFC 1321. The first function is a little bit optimized
37  (as found in Colin Plumbs public domain implementation). */
38 #define FF(b, c, d) (d ^ (b & (c ^ d)))
39 #define FG(b, c, d) FF(d, b, c)
40 #define FH(b, c, d) (b ^ c ^ d)
41 #define FI(b, c, d) (c ^ (b | ~d))
42 #define rol(value, bits) (((value) << (bits)) | ((value) >> (32 - (bits))))

45  u32 correct_words[16];
46  u32 A = A_;

50  u32* cwp = correct_words;
51 
52 #if __BYTE_ORDER == __BIG_ENDIAN
53  {
54  int i;
55  u8 *p2, *p1;
56  for (i = 0, p1 = (u8*)data, p2 = (u8*)correct_words; i < 16; i++, p2 += 4) {
57  p2[3] = *p1++;
58  p2[2] = *p1++;
59  p2[1] = *p1++;
60  p2[0] = *p1++;
61  }
62  }
63 #elif __BYTE_ORDER == __LITTLE_ENDIAN
64  memcpy(correct_words, data, 64);
65 #else
66 #error "Machine has unsupported byte order!"

163 void MD5::update(const char* v, u32 n) const {
164  u32 pos = 0;
165 
166  require_(!isFinalized_);
167 
168  // flush the buffer
169  if (count == 64) {
170  transform(buf);
171  count = 0;
172  nblocks++;
173  }
174  if (!v)
175  return;
176  if (count) {
177  for (; n && count < 64; n--)
178  buf[count++] = v[pos++];
179  update(NULL, 0);
180  if (!n)
181  return;
182  }
183 
184  while (n >= 64) {
185  transform(v);
186  count = 0;
187  nblocks++;
188  n -= 64;
189  v += 64;
190  }
191  for (; n && count < 64; n--)
192  buf[count++] = v[pos++];
193 }

213 void MD5::finalize(void) const {
214  u32 t, msb, lsb;
215 
216  require(!isFinalized_);
217 
218  // flush
219  update(NULL, 0);
220  msb = 0;
221  t = nblocks;
222 
223  // multiply by 64 to make a byte count
224  if ((lsb = t << 6) < t)
225  msb++;
226  msb += t >> 26;
227  t = lsb;
228 
229  // add the count
230  if ((lsb = t + count) < t)
231  msb++;
232  t = lsb;
233 
234  // multiply by 8 to make a bit count
235  if ((lsb = t << 3) < t)
236  msb++;
237  msb += t >> 29;
238 
239  // enough room
240  if (count < 56) {
241  /* pad */
242  buf[count++] = 0x80;
243  while (count < 56)
244  buf[count++] = 0;
245  }
246  else {
247  // need one extra block
248  // pad character
249  buf[count++] = 0x80;
250  while (count < 64)
251  buf[count++] = 0;
252 
253  // flush
254  update(NULL, 0);
255 
256  // fill next block with zeroes
257  memset(buf, 0, 56);
258  }
259 
260  // append the 64 bit count
261  buf[56] = lsb;
262  buf[57] = lsb >> 8;
263  buf[58] = lsb >> 16;
264  buf[59] = lsb >> 24;
265  buf[60] = msb;
266  buf[61] = msb >> 8;
267  buf[62] = msb >> 16;
268  buf[63] = msb >> 24;
269  transform(buf);
270 
271  u32* buf32 = (u32*)buf;
272  buf32[0] = A_;
273  buf32[1] = B_;
274  buf32[2] = C_;
275  buf32[3] = D_;
276 #if __BYTE_ORDER == __BIG_ENDIAN
277  swapEndianess<4>(buf32, 4);
278 #elif __BYTE_ORDER == __LITTLE_ENDIAN
279 #else
280 #error "Machine has unsupported byte order!"
281 #endif
282 
283  isFinalized_ = true;
284 }
