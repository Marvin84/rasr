124 /**
125  * Finds best match using dynamic programming:
126  * Runtime is at most |Component| * |Resource|
127  **/
128 s32 Configuration::Resource::match(const std::vector<std::string>& components) const {
129  std::vector<s16> nMatches(components.size() + 1, Core::Type<s16>::min);
130  nMatches.front() = 0;
131  bool hasMatch = true;
132  StringTokenizer tokenizer(getName(), resource_separation_string);
133  for (StringTokenizer::Iterator itResource = tokenizer.begin(), endResource = tokenizer.end(); hasMatch && (itResource != endResource); ++itResource) {
134  hasMatch = false;
135  if (*itResource == resource_wildcard_string) {
136  for (std::vector<s16>::iterator itMatches = nMatches.begin() + 1; itMatches != nMatches.end(); ++itMatches)
137  if (*(itMatches - 1) > *itMatches)
138  *itMatches = *(itMatches - 1);
139  hasMatch = true;
140  }
141  else {
142  std::vector<std::string>::const_reverse_iterator itComponent = components.rbegin();
143  for (std::vector<s16>::reverse_iterator itMatches = nMatches.rbegin(); itMatches != nMatches.rend() - 1; ++itMatches, ++itComponent)
144  if ((*(itMatches + 1) >= 0) && (*itComponent == *itResource)) {
145  *itMatches = *(itMatches + 1) + 1;
146  hasMatch = true;
147  }
148  else
149  *itMatches = Core::Type<s16>::min;
150  nMatches.front() = Core::Type<s16>::min;
151  }
152  }
153  return (nMatches.back() >= 0) ? nMatches.back() : -1;
154 }

251 const Configuration::Resource* Configuration::ResourceDataBase::find(
252  const std::string& parameter) const {
253  require(isWellFormedParameterName(parameter));
254 
255  // split parameter string into components
256  std::vector<std::string> components;
257  StringTokenizer tokenizer(parameter, resource_separation_string);
258  for (StringTokenizer::Iterator token = tokenizer.begin(); token != tokenizer.end(); ++token) {
259  components.push_back(*token);
260  }
261 
262  s32 specific = 0;
263  u32 ties = 0;
264  const Resource* result = 0;
265 
266  // try all resources to find best (most specific) match
267  std::set<Resource>::const_iterator it;
268 
269  // debug
270  // std::cerr << "match " << parameter << "(" << resources.size() << "):" << std::endl;
271 
272  for (it = resources.begin(); it != resources.end(); it++) {
273  s32 m = it->match(components);
274 
275  // debug
276  // if (m > 0) std::cerr << it->getName() << " matches with score " << m << std::endl;
277 
278  if (m > specific) {
279  specific = m;
280  ties = 0;
281  result = &(*it);
282  }
283  else if (m == specific) {
284  ++ties;
285  }
286  }
287 
288  if (ties > 0) {
289  std::cerr << "configuration warning: \""
290  << parameter << "\" is matched by "
291  << (ties + 1) << " equally specific resources:" << std::endl;
292  for (it = resources.begin(); it != resources.end(); it++) {
293  s32 m = it->match(components);
294  if (m == specific)
295  std::cerr << " - \"" << it->getName() << "\"" << std::endl;
296  }
297 
298  std::cerr << "using: \""
299  << result->getName() << "\"."
300  << std::endl;
301  }
302 
303  if (isLogging_) {
304  std::cerr << "configuration: \"";
305  if (result)
306  std::cerr << parameter << "\" is matched by \""
307  << result->getName() << "\""
308  << std::endl;
309  else
310  std::cerr << parameter << "\" is not matched."
311  << std::endl;
312  }
313 
314  return result;
315 }

347 bool Configuration::isWellFormedName(const std::string& s, bool allowWildcards) {
348  enum { require_component,
349  component,
350  require_separator,
351  error } state = require_component;
352  for (std::string::size_type i = 0; i < s.size() && state != error; ++i) {
353  switch (state) {
354  case require_component: {
355  if (s[i] == resource_separation_char)
356  state = error;
357  else if (s[i] == resource_wildcard_char) {
358  state = (allowWildcards) ? require_separator : error;
359  }
360  else
361  state = component;
362  } break;
363  case component: {
364  if (s[i] == resource_separation_char)
365  state = require_component;
366  else if (s[i] == resource_wildcard_char)
367  state = error;
368  else
369  state = component;
370  } break;
371  case require_separator: {
372  if (s[i] == resource_separation_char)
373  state = require_component;
374  else
375  state = error;
376  } break;
377  default: break;
378  }
379  }
380 
381  switch (state) {
382  case component:
383  case require_separator:
384  return true;
385  break;
386  case error:
387  case require_component:
388  default:
389  return false;
390  break;
391  }
392 }

460 std::string Configuration::prepareResourceName(
461  const std::string& selection,
462  const std::string& parameter) {
463  if (!(selection.empty() || isWellFormedResourceName(selection))) {
464  std::cerr << "selection is not valid: '" << selection << std::endl;
465  }
466  require(selection.empty() || isWellFormedResourceName(selection));
467  if (!isWellFormedResourceName(parameter)) {
468  std::cerr << "parameter name is not valid: '" << parameter << "'" << std::endl;
469  }
470  require(isWellFormedResourceName(parameter));
471 
472  std::string result;
473 
474  if (selection.empty()) {
475  result = parameter;
476  }
477  else {
478  result = selection;
479  if ((result[result.length() - 1] == resource_wildcard_char) &&
480  (parameter[0] == resource_wildcard_char)) {
481  result += std::string(parameter.begin() + 1, parameter.end());
482  }
483  else {
484  result += resource_separation_string;
485  result += parameter;
486  }
487  }
488 
489  ensure(isWellFormedResourceName(result));
490  return result;
491 }

524 bool Configuration::setFromEnvironment(const std::string& variable) {
525  // parse user supplied variable
526  const char* env = getenv(variable.c_str());
527  if (!env)
528  return false;
529 
530  std::string token;
531  std::istringstream iss(env);
532  std::vector<std::string> arguments;
533 
534  arguments.push_back("-");
535  while (wsgetline(iss, token, " ") != EOF) {
536  arguments.push_back(token);
537  iss >> std::ws;
538  if (iss.get() == '"') {
539  if ((wsgetline(iss, token, "\"")) != EOF) {
540  arguments.push_back(token);
541  }
542  }
543  else
544  iss.unget();
545  }
546 
547  // parse options exactly like an additional command line
548  setFromCommandline(arguments, db_->addSource("environment", variable));
549 
550  return true;
551 }

557 bool Configuration::setFromFile(const std::string& filename,
558  const std::string& currentSelection) {
559  int linenr;
560  std::ifstream ifs(filename.c_str());
561  std::string line, warningDescription;
562  std::string includeFilename;
563  std::string selection(currentSelection), parameter, value;
564  // check if input stream has been succesfully opened
565  if (!ifs) {
566  std::cerr << "configuration error: failed to open file \""
567  << filename << "\" for reading.";
568  if (errno) {
569  std::cerr << " (" << strerror(errno) << ")";
570  errno = 0;
571  }
572  std::cerr << std::endl;
573  return false;
574  }
575 
576  SourceDescriptor* sd = db_->addSource("file", filename);
577 
578  // read configuration file from input stream
579  linenr = 0;
580  while (getline(ifs, line) != EOF) {
581  ++linenr;
582  warningDescription = "";
583  if (line.size() == 0)
584  continue;
585 
586  if (parseInclude(line, includeFilename, warningDescription)) {
587  if (!warningDescription.empty())
588  warning(filename, linenr, warningDescription);
589  std::string f(resolve(includeFilename));
590  if (!isRegularFile(f))
591  f = joinPaths(directoryName(filename), f);
592  if (isRegularFile(f))
593  setFromFile(f, selection);
594  else
595  error(filename, linenr, "Include file '" + f + "' does not exist.");
596  }
597  else if (parseSelection(line, selection, warningDescription)) {
598  if (!warningDescription.empty())
599  warning(filename, linenr, warningDescription);
600  }
601  else if (parseComment(line, warningDescription)) {
602  if (!warningDescription.empty())
603  warning(filename, linenr, warningDescription);
604  }
605  else if (parseResource(line, parameter, value, warningDescription)) {
606  if (!warningDescription.empty())
607  warning(filename, linenr, warningDescription);
608  if (!isWellFormedResourceName(parameter))
609  warning(filename, linenr, "bad resource name: '" + parameter + "'");
610  else {
611  std::string resource = prepareResourceName(selection, parameter);
612  set(resource, value, sd);
613  }
614  }
615  else {
616  error(filename, linenr, "syntax error");
617  }
618  }
619  return true;
620 }

622 bool Configuration::parseInclude(const std::string& line,
623  std::string& includeFilename,
624  std::string& warningDescription) {
625  std::istringstream iss(line);
626  std::string token;
627  iss >> token;
628  if (token == "include") {
629  token.clear();
630  iss >> token;
631  if (!token.empty() && token[0] != '=') {
632  includeFilename = token;
633  token.clear();
634  iss >> token;
635  if (!token.empty() &&
636  Application::us()->getCommentCharacters().find(token[0]) == std::string::npos) {
637  warningDescription = "unknown include argument '" + token + "'";
638  }
639  return true;
640  }
641  }
642  return false;
643 }

678 bool Configuration::parseResource(const std::string& line,
679  std::string& parameter, std::string& value,
680  std::string& warningDescription) {
681  std::istringstream iss(line);
682  int status = wsgetline(iss, parameter, "= \t\n" + Application::us()->getCommentCharacters());
683  if ((status != EOF) && (!parameter.empty())) {
684  parameter = resolve(parameter);
685  // check wether we have an argument
686  if (status != 0) {
687  // Comment, whitespace or '=' found and parameter is not empty...
688  iss.unget(); // put the last character back in order to parse the remainder correctly.
689  iss >> std::ws; // Seek to a non-whitespace character.
690  int peek = iss.get();
691  if (peek == '=') {
692  // ... It was '=', so try to parse the value until a comment or new line.
693  if (wsgetline(iss, value, "\n\"" + Application::us()->getCommentCharacters()) != EOF) {
694  // Value found, let's strip it.
695  std::string::size_type i = value.find_last_not_of(" \t");
696  if (i != std::string::npos)
697  value.erase(i + 1);
698  }
699  else {
700  // Line was empty after the '='. It is an unfinished line.
701  return false;
702  }
703  }
704  else if (Application::us()->getCommentCharacters().find(peek) == std::string::npos &&
705  peek != EOF) {
706  // ... It was, whitespace followed by something but not a comment.
707  return false;
708  }
709  else {
710  // ... It was comment or trailing whitespace.
711  // Let's handle the parameter as flag type, and simply set it to 'true'.
712  value = "true";
713  }
714  }
715  else {
716  // Neither comment nor whitespace nor '=' found but parameter is not empty.
717  // Thus, it must be a flag type. Simply set the parameter to 'true'.
718  value = "true";
719  }
720  }
721  else {
722  // Line was empty or comment, whitespace or '=' found but parameter was empty.
723  return false;
724  }
725  return true;
726 }

759 std::vector<std::string> Configuration::setFromCommandline(
760  const std::vector<std::string>& arguments,
761  const SourceDescriptor* source) {
762  std::string option;
763  std::vector<std::string> unparsed;
764  enum { Option,
765  Argument } state = Option;
766 
767  if (arguments.size() == 0)
768  return unparsed;
769  for (std::vector<std::string>::const_iterator arg = arguments.begin() + 1; arg != arguments.end(); ++arg) {
770  if (((*arg)[0] == '-') && ((*arg)[1] == '-') && (state == Argument)) {
771  tryToSet(option, "true", source);
772  state = Option;
773  }
774  switch (state) {
775  case Option:
776  if (((*arg)[0] == '-') && ((*arg)[1] == '-')) {
777  if ((*arg)[1] == '\0')
778  unparsed.push_back(*arg);
779  else {
780  std::string::size_type equalSignPos = arg->find("=");
781  if (equalSignPos == std::string::npos)
782  equalSignPos = arg->size();
783  option = arg->substr(2, equalSignPos - 2);
784  if (option[0] == resource_separation_char)
785  option = option.substr(1);
786  else
787  option = Application::us()->name() + resource_separation_string + option;
788  if (equalSignPos != arg->size()) {
789  std::string argument;
790  argument = arg->substr(equalSignPos + 1, std::string::npos);
791  tryToSet(option, argument, source);
792  }
793  else
794  state = Argument;
795  }
796  }
797  else
798  unparsed.push_back(*arg);
799  break;
800  case Argument:
801  tryToSet(option, *arg, source);
802  state = Option;
803  break;
804  }
805  }
806 
807  if (state == Argument) {
808  // assume open last option to be a flag
809  tryToSet(option, "true", source);
810  }
811 
812  return unparsed;
813 }

846 std::string Configuration::resolveReferences(const std::string& value) const {
847  std::string result;
848  std::string::size_type begin, pos = 0;
849 
850  while ((begin = value.find("$(", pos)) != std::string::npos) {
851  result += value.substr(pos, begin - pos);
852  pos = begin + 2;
853  std::string::size_type end = value.find(")", pos);
854  if (end != std::string::npos) {
855  const Resource* resource = 0;
856  std::string query(value.substr(pos, end - pos));
857 
858  if (isWellFormedParameterName(query)) {
859  // search for reference local to global
860  for (std::string scope = getSelection();;) {
861  resource = find(prepareResourceName(scope, query));
862  if (resource)
863  break;
864  if (scope.size() == 0)
865  break;
866  std::string::size_type sp = scope.find_last_of(resource_separation_char);
867  if (sp == std::string::npos)
868  sp = 0;
869  scope.erase(sp);
870  }
871  }
872  else {
873  std::cerr << "configuration error: malformed reference \"" << query << "\"" << std::endl;
874  }
875 
876  if (resource) {
877  result += getResolvedValue(resource);
878  }
879  else {
880  // if not still not resolved -> warn
881  std::cerr << "configuration error: could not resolve reference \""
882  << query << "\"" << std::endl;
883  result += "(" + query + ")";
884  }
885 
886  pos = end + 1;
887  }
888  else {
889  std::cerr << "configuration error: unclosed bracket in value \""
890  << value << "\"" << std::endl;
891  return std::string("");
892  }
893  }
894  result += value.substr(pos, std::string::npos);
895 
896  return result;

899 std::string Configuration::resolveArithmeticExpressions(const std::string& value) {
900  std::string result;
901  std::string::size_type begin, pos = 0;
902  ArithmeticExpressionParserDriver parser;
903 
904  while ((begin = value.find("$[", pos)) != std::string::npos) {
905  bool error = false;
906  std::string errMsg = "";
907  result += value.substr(pos, begin - pos);
908  pos = begin + 2;
909  std::string::size_type end = value.find("]", pos);
910  if (end != std::string::npos) {
911  std::string query(value.substr(pos, end - pos));
912  std::string format = "float";
913  if ((begin = query.find(",")) != std::string::npos) {
914  format = query.substr(begin + 1);
915  stripWhitespace(format);
916  query = query.substr(0, begin);
917  }
918  double dValue = 0;
919  if (!parser.parse(query, dValue)) {
920  error = true;
921  errMsg = "parse error";
922  errMsg += parser.getLastError();
923  }
924  if (format == "int" || format == "i")
925  result += form("%d", int(dValue));
926  else {
927  if (format != "float" && format != "f") {
928  error = true;
929  errMsg = "unknown format: \"";
930  errMsg += format;
931  errMsg += "\"";
932  }
933  result += form("%f", dValue);
934  }
935  pos = end + 1;
936  }
937  else {
938  error = true;
939  errMsg = "unclosed bracket";
940  }
941  if (error) {
942  std::cerr << "configuration error: " << errMsg << " in value \""
943  << value << "\"" << std::endl;
944  return std::string("");
945  }
946  }
947  result += value.substr(pos, std::string::npos);
948 
949  return result;
