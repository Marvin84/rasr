75  * returns true, if the quadratic equation has a real solution

77 template<typename T>
78 bool solveQuadraticEquation(T p, T q, T& xplus, T& xminus) {
79  T discriminant = 0.25 * p * p - q;
80  if (discriminant < 0)
81  return false;
82  else {
83  T delta = std::sqrt(discriminant);
84  xminus = -0.5 * p - delta;
85  xplus = -0.5 * p + delta;
86  return true;
87  }
88 }

90 /**
91  * isnan(val) returns true if val is nan.
92  * We cannot rely on std::isnan or x!=x, because GCC may wrongly optimize it
93  * away when compiling with -ffast-math (default in RASR), even with volatile,
94  * because -ffast-math implies -ffinite-math-only.
95  * Earlier, we checked the binary representation here, but that is of course
96  * architecture dependent.
97  * Most platforms have __isnan, which is not handled by the compiler, but
98  * implemented in libm (C stdlib).
99  * Some references:
100  * http://refspecs.linuxbase.org/LSB_3.1.1/LSB-Core-generic/LSB-Core-generic/baselib---isnan.html
101  * http://stackoverflow.com/questions/22931147/stdisinf-does-not-work-with-ffast-math-how-to-check-for-infinity
102  **/

104 bool isnan(T val);

172 /**
173  * compute the sum of two scores in probability space and convert back to score space
174  * (e.g. -log(exp(-a) + exp(-b)) ) in a numerically stable way
175  */
176 template<typename T>
177 T scoreSum(T a, T b) {
178  T diff = a - b;
179  if (isnan(diff)) { // a = b = inf => both probabilities are 0 => sum should be zero aswell
180  return ::std::numeric_limits<T>::infinity();
181  }
182  return ::std::min(a, b) - ::log1p(::std::exp(-::std::abs(diff)));
183 }
