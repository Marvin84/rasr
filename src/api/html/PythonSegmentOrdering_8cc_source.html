23 /**
24  * Python interface:
25  *
26  * def getSegmentList(corpusName, segmentList, segmentsInfo, config)
27  *
28  * corpusName is the name of the root corpus.
29  * segmentList is a list of strings with the segment-names of the corpus.
30  * segmentsInfo is a dict segmentName -> info, where info is a dict with info such as
31  * nframes (only if use-data-source = true, but that's the default).
32  * getSegmentList() can be any generator. It should yield segment-name strings.
33  *
34  * Dummy example:
35  *
36  * def getSegmentList(corpusName, segmentList, segmentsInfo, config): return segmentList
37  *
38  * Usually, there is only one root corpus.
39  * However, the function still might get called multiple times for the same root corpus
40  * because e.g. for ProgressIndicationVisitorAdaptor, we count the total amount of segments.
41  *
42  * Note that there are usually other CorpusVisitors behind the SegmentOrderingVisitor
43  * which might further filter out the segments, such as the SegmentPartitionVisitorAdaptor.
44  **/

63 PythonSegmentOrderingVisitor::PythonSegmentOrderingVisitor(const std::string& pyModPath, const std::string& pyModName, const std::string& pyConfig, Core::Component& owner) {
64  pythonInitializer.init();
65  pyConfig_ = pyConfig;
66  allowCopy_ = paramPythonSegmentOrderAllowCopy(owner.getConfiguration());
67  withInfo_ = paramWithSegmentInfo(owner.getConfiguration());
68 
69  {
70  Python::ScopedGIL gil;
71 
72  if (!pyModPath.empty())
73  Python::addSysPath(pyModPath);
74 
75  pyMod_.takeOver(PyImport_ImportModule(pyModName.c_str()));
76  if (!pyMod_) {
77  Python::criticalError(
78  "python-segment-order: cannot import module '%s'",
79  pyModName.c_str());
80  return;
81  }
82  }
83 
84  if (paramUseDataSource(owner.getConfiguration())) {
85  if (!withInfo_)
86  owner.error("python-segment-order: python-segment-order-with-segment-info must be enabled for use-data-source");
87  // See Speech::DataExtractor.
88  dataSource_ = Core::ref(Speech::Module::instance().createDataSource(owner.select("feature-extraction"), /*loadFromFile*/ true));
89  require(dataSource_);
90  dataSource_->respondToDelayedErrors();
91  dataSource_->setProgressIndication(false);
92  }

149 PyObject* PythonSegmentOrderingVisitor::getSegmentsInfo() {
150  // We expect to have the Python GIL.
151 
152  // dict[segmentName,info].
153  Python::ObjRef segmentsDict;
154  segmentsDict.takeOver(PyDict_New());
155 
156  for (const std::string& segmentName : segmentList_) {
157  Segment* segment = getSegmentByName(segmentName);
158  if (!segment) {
159  Core::Application::us()->error("segment '%s' not found", segmentName.c_str());
160  continue;
161  }
162 
163  // dict[attrib,value] for segment.
164  Python::ObjRef infoDict;
165  infoDict.takeOver(PyDict_New());
166 
167  // So far, only the length.
168  if (dataSource_) {
169  size_t nFrames = getSegmentNumFrames(segment);
170  Python::dict_SetItemString(infoDict, "nframes", nFrames);
171  }
172 
173  Python::dict_SetItemString(infoDict, "time", segment->end() - segment->start());
174 
175  PyDict_SetItemString(segmentsDict, segmentName.c_str(), infoDict);
176  infoDict.clear();
177  }
178 
179  if (PyErr_Occurred())
180  Python::criticalError("error collecting segments info");
181 
182  return segmentsDict.release();
183 }

185 void PythonSegmentOrderingVisitor::leaveCorpus(Bliss::Corpus* corpus) {
186  curCorpus_.pop_back();
187  if (!curCorpus_.empty()) {
188  // not the root corpus
189  return;
190  }
191  // corpus is the root corpus. We don't have our own copy of this one.
192 
193  prepareSegmentLoop();
194 
195  {
196  CustomCorpusGuide corpusGuide(this, /* root */ corpus);
197  Python::ScopedGIL gil;
198 
199  Python::ObjRef pySegmentList;
200  Python::ObjRef iterator, item;
201 
202  // Now, pass the current segmentList_ to Python.
203  // We expect to get some iterable segment-name-list returned.
204  pySegmentList.takeOver(Python::PyCallKw(
205  pyMod_,
206  "getSegmentList", "{s:s,s:N,s:N,s:s}",
207  "corpusName", corpus->name().c_str(),
208  "segmentList", Python::newList(segmentList_),
209  "segmentsInfo", withInfo_ ? getSegmentsInfo() : Python::incremented(Py_None),
210  "config", pyConfig_.c_str()));
211  if (!pySegmentList)
212  Python::criticalError("python-segment-order: getSegmentList() failed");
213 
214  iterator.takeOver(PyObject_GetIter(pySegmentList));
215  if (!iterator)
216  Python::criticalError("python-segment-order: getSegmentList() did not return an iterable object");
217 
218  while (item.takeOver(PyIter_Next(iterator))) {
219  std::string segmentName;
220  if (!Python::pyObjToStr(item, segmentName)) {
221  Python::criticalError("python-segment-order: segment-name is not a string but a %s",
222  item->ob_type->tp_name);
223  break;
224  }
225 
226  // Pass on the segment to the rest of Sprint.
227  // Release the Python GIL meanwhile.
228  Python::ScopedAllowThreads allowThreads;
229  corpusGuide.showSegmentByName(segmentName);
230  }
231 
232  if (PyErr_Occurred())
233  Python::criticalError("python-segment-order: failed to get next segment-name");
234 
235  iterator.clear();
236  }
237 
238  finishSegmentLoop();
239 }
