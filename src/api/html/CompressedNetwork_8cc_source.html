71  void createArcs(typename Automaton::StateId stateId, Fsa::Stack<typename Automaton::StateId>& statesToExplore) {
72  /**! @todo check: do not expand any epsilon arc */
73  Fsa::Stack<EpsilonArc> arcsToExplore;
74 
75  StateIndex stateIndex = getStateIndex(stateId);
76  states_[stateIndex].begin = curArcs_;
77  states_[stateIndex].epsilonArcsBegin = curEpsArcs_;
78 
79  for (typename Automaton::ArcIterator a = fsa_.arcs(stateId); !a.done(); a.next()) {
80  const typename Automaton::Arc& arc = a.value();
81  verify(fsa_.arcInput(arc) <= Core::Type<InternalLabel>::max);
82  verify(fsa_.arcOutput(arc) <= Core::Type<InternalLabel>::max);
83  if (fsa_.arcInput(arc) == OpenFst::Epsilon) {
84  if (removeEpsArcs_ && (fsa_.arcOutput(arc) == OpenFst::Epsilon && !fsa_.isFinal(fsa_.arcTarget(arc)))) {
85  arcsToExplore.push(EpsilonArc(fsa_.arcTarget(arc), fsa_.arcOutput(arc), fsa_.arcWeightValue(arc)));
86  }
87  else {
88  new (epsilonArcs_ + curEpsArcs_) EpsilonArc(getStateIndex(fsa_.arcTarget(arc)), fsa_.arcOutput(arc), fsa_.arcWeightValue(arc));
89  ++curEpsArcs_;
90  statesToExplore.push(fsa_.arcTarget(arc));
91  }
92  }
93  else {
94  new (arcs_ + curArcs_) Arc(getStateIndex(fsa_.arcTarget(arc)), fsa_.arcInput(arc), fsa_.arcOutput(arc), fsa_.arcWeightValue(arc));
95  ++curArcs_;
96  statesToExplore.push(fsa_.arcTarget(arc));
97  }
98  }
99  while (!arcsToExplore.empty()) {
100  EpsilonArc epsArc = arcsToExplore.pop();
101  for (typename Automaton::ArcIterator a = fsa_.arcs(epsArc.nextstate); !a.done(); a.next()) {
102  const typename Automaton::Arc& arc = a.value();
103  Score weight = epsArc.weight + Score(fsa_.arcWeightValue(arc));
104  if (fsa_.arcInput(arc) == OpenFst::Epsilon) {
105  if (fsa_.arcOutput(arc) == OpenFst::Epsilon && !fsa_.isFinal(fsa_.arcTarget(arc)))
106  arcsToExplore.push(EpsilonArc(fsa_.arcTarget(arc), OpenFst::Epsilon, weight));
107  else {
108  new (epsilonArcs_ + curEpsArcs_) EpsilonArc(getStateIndex(fsa_.arcTarget(arc)), fsa_.arcOutput(arc), weight);
109  ++curEpsArcs_;
110  statesToExplore.push(fsa_.arcTarget(arc));
111  }
112  }
113  else {
114  typename Automaton::Label output = epsArc.olabel;
115  if (fsa_.arcOutput(arc) != OpenFst::Epsilon) {
116  output = fsa_.arcOutput(arc);
117  }
118  new (arcs_ + curArcs_) Arc(getStateIndex(fsa_.arcTarget(arc)), fsa_.arcInput(arc), output, weight);
119  ++curArcs_;
120  statesToExplore.push(fsa_.arcTarget(arc));
121  }
122  }
123  }
124  u32 nArcs = curArcs_ - states_[stateIndex].begin;
125  u32 nEpsArcs = curEpsArcs_ - states_[stateIndex].epsilonArcsBegin;
126  verify(nArcs <= Core::Type<ArcCount>::max);
127  verify(nEpsArcs <= Core::Type<EpsArcCount>::max);
128  states_[stateIndex].nArcs = nArcs;
129  states_[stateIndex].nEpsilonArcs = nEpsArcs;
130  }

132 public:
133  void createNetwork() {
134  stateIndex_.resize(fsa_.nStates(), InvalidStateIndex);
135  network_->nStates_ = fsa_.nStates();
136  network_->nArcs_ = fsa_.nArcs();
137  network_->nEpsilonArcs_ = fsa_.nEpsilonArcs();
138  states_ = static_cast<State*>(malloc(network_->nStates_ * sizeof(State)));
139  arcs_ = static_cast<Arc*>(malloc(network_->nArcs_ * sizeof(Arc)));
140  epsilonArcs_ = static_cast<EpsilonArc*>(malloc(network_->nEpsilonArcs_ * sizeof(EpsilonArc)));
141  curStates_ = curArcs_ = curEpsArcs_ = 0;
142  typename Automaton::StateId initial = fsa_.initialStateId();
143  Fsa::Stack<typename Automaton::StateId> statesToExplore;
144  statesToExplore.push(initial);
145  while (!statesToExplore.empty()) {
146  typename Automaton::StateId s = statesToExplore.pop();
147  State& state = getState(s);
148  if (state.begin != InvalidArcIndex) {
149  // state already expanded
150  continue;
151  }
152  createArcs(s, statesToExplore);
153  }
154  verify(curStates_ == network_->nStates());
155  verify(curArcs_ == network_->nArcs());
156  verify(curEpsArcs_ = network_->nEpsilonArcs());
157  network_->initialStateIndex_ = stateIndex_[initial];
158  }

269  off_t pos, pad;

284 u32 CompressedNetwork::writeData(int fd) const {
285  off_t pos;
286  ssize_t nBytes;
287  ImageHeader header;
288 
289  // write header
290  memcpy(header.magic, ImageHeader::magicWord, 8);
291  header.version = ImageHeader::formatVersion;
292  header.initialStateIndex = initialStateIndex_;
293  header.nStates = nStates();
294  header.nArcs = nArcs();
295  header.nEpsilonArcs = nEpsilonArcs();
296  header.statesOffset = 0;
297  header.arcsOffset = 0;
298  header.epsArcsOffset = 0;
299  header.end = 0;
300  nBytes = sizeof(ImageHeader);
301  if (::write(fd, &header, nBytes) != nBytes)
302  return 1;
303 
304  // write arrays
305  if (!writeArray(fd, header.statesOffset, states_, nStates()))
306  return 2;
307  if (!writeArray(fd, header.arcsOffset, arcs_, nArcs()))
308  return 3;
309  if (!writeArray(fd, header.epsArcsOffset, epsilonArcs_, nArcs()))
310  return 4;
311 
312  // determine file size
313  if ((pos = lseek(fd, 0, SEEK_CUR)) == (off_t)-1)
314  return 5;
315  header.end = pos;
316 
317  // write header with offsets
318  if ((pos = lseek(fd, 0, SEEK_SET)) == (off_t)-1)
319  return 6;
320  nBytes = sizeof(ImageHeader);
321  if (::write(fd, &header, nBytes) != nBytes)
322  return 7;
323  return 0;
324 }

342 bool CompressedNetwork::readData(int fd) {
343  ssize_t nBytes;
344  ImageHeader header;
345  nBytes = sizeof(ImageHeader);
346  if (::read(fd, &header, nBytes) != nBytes) {
347  warning("cannot read header");
348  return false;
349  }
350  if (memcmp(header.magic, ImageHeader::magicWord, 8)) {
351  warning("bad magic word in file header");
352  return false;
353  }
354  if (header.version != ImageHeader::formatVersion) {
355  warning("file format is in version %d, expected %d", header.version, ImageHeader::formatVersion);
356  return false;
357  }
358  nStates_ = header.nStates;
359  nArcs_ = header.nArcs;
360  nEpsilonArcs_ = header.nEpsilonArcs;
361  initialStateIndex_ = header.initialStateIndex;
362 
363  mmap_ = (char*)mmap(0, mmapSize_ = header.end, PROT_READ, MAP_SHARED, fd, 0);
364  if (mmap_ == MAP_FAILED) {
365  error("mmap failed");
366  return false;
367  }
368  states_ = reinterpret_cast<State*>(mmap_ + header.statesOffset);
369  arcs_ = reinterpret_cast<Arc*>(mmap_ + header.arcsOffset);
370  epsilonArcs_ = reinterpret_cast<EpsilonArc*>(mmap_ + header.epsArcsOffset);
371  return true;
