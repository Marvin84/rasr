31  Math::AnalyticFunctionFactory factory(select(paramWarpingFunction.name()));
32  factory.setSampleRate(sampleRate_);
33  factory.setDomainType(Math::AnalyticFunctionFactory::discreteDomain);
34  factory.setMaximalArgument(inputSize_ - 1);
35  Math::UnaryAnalyticFunctionRef warpingFunction = factory.createUnaryFunction(Flow::StringExpressionNode::value());
36  if (!warpingFunction)
37  criticalError("Could not create warping function.");

39  if (interpolateOverWarpedAxis_) {
40  warping_.setWarpingFunction(warpingFunction, inputSize_, mergeType_, interpolationType_);
