56 static bool cmp_edges(Edge const& a, Edge const& b) {
57  Fsa::StateId adiff = (a.to - a.from);
58  Fsa::StateId bdiff = (b.to - b.from);
59  if (adiff == bdiff) {
60  if (a.to == b.to) {
61  return a.from < b.from;
62  }
63  return a.to < b.to;
64  }
65  else {
66  return adiff < bdiff;
67  }
68 }

70 static void toposort(std::vector<Fsa::StateId>& states,
71  std::vector<std::pair<Fsa::StateId, float>>& final_states,
72  std::vector<Edge>& edges) {
73  std::vector<Fsa::StateId> state_map(states.size());
74 
75  std::vector<size_t> in_count(states.size(), 0ul);
76  for (Edge const& e : edges) {
77  if (e.to != e.from) {
78  in_count[e.to] += 1ul;
79  }
80  }
81 
82  Fsa::StateId next_id = 0u;
83  std::vector<size_t>::const_iterator it;
84  while ((it = std::find(in_count.begin(), in_count.end(), 0ul)) != in_count.end()) {
85  Fsa::StateId state = static_cast<Fsa::StateId>(it - in_count.begin());
86  state_map[state] = next_id++;
87  // this could be more efficient, but I am too lazy now
88  for (Edge const& e : edges) {
89  if (e.from == state and e.to != e.from) {
90  in_count[e.to]--;
91  }
92  }
93  in_count[state] = std::numeric_limits<size_t>::max();
94  }
95 
96  apply_state_map(state_map, states, final_states, edges);
97 
98  std::sort(edges.begin(), edges.end(), &cmp_edges);
99 }

101 static void filter_edges(std::vector<Edge>& edges) {
102  size_t cur = 0ul;
103  for (size_t i = 1ul; i < edges.size(); i++) {
104  if (edges[cur].from == edges[i].from && edges[cur].to == edges[i].to && edges[cur].emission_idx == edges[i].emission_idx) {
105  edges[cur].weight = Math::scoreSum(edges[cur].weight, edges[i].weight);
106  }
107  else {
108  edges[++cur] = edges[i];
109  }
110  }
111  edges.resize(cur + 1);
112 }

114 static void make_single_final_state(std::vector<Fsa::StateId>& states, std::vector<std::pair<Fsa::StateId, float>>& final_states, std::vector<Edge>& edges) {
115  if (final_states.size() == 1 && final_states.front().first == states.back() && final_states.front().second == 0) {
116  return; // nothing to do
117  }
118 
119  Fsa::StateId new_final = static_cast<Fsa::StateId>(states.size());
120  states.push_back(new_final);
121  size_t old_edges_size = edges.size();
122  for (size_t e = 0ul; e < old_edges_size; e++) {
123  for (std::pair<Fsa::StateId, float>& final_state : final_states) {
124  if (final_state.first != edges[e].to)
125  continue;
126  Edge new_edge = edges[e];
127  new_edge.to = new_final;
128  new_edge.weight += final_state.second;
129  edges.push_back(new_edge);
130  break;
131  }
132  }
133 
134  final_states.resize(1ul);
135  final_states.front() = std::make_pair(new_final, 0.0f);
136 }

142 AllophoneStateFsaExporter::ExportedAutomaton AllophoneStateFsaExporter::exportFsaForOrthography(std::string const& orthography) const {
143  Core::Ref<Am::AcousticModel> am = mc_.acousticModel();
144  Speech::AllophoneStateGraphRef graph = allophone_state_graph_builder_->build(orthography);
145  graph = Fsa::projectInput(graph);
146  graph = Fsa::removeDisambiguationSymbols(graph);
147  graph = Fsa::removeEpsilons(graph);
148  graph = Fsa::normalize(graph);
149  // TODO: use Fsa::topologicalSort here, remove toposort below, cleanup
150  Core::Ref<Fsa::StaticAutomaton> automaton = Fsa::staticCopy(graph);
151  require_eq(automaton->initialStateId(), 0);
152 
153  std::vector<Fsa::StateId> states;
154  std::vector<std::pair<Fsa::StateId, float>> final_states;
155  std::vector<Edge> edges;
156 
157  for (Fsa::StateId s = 0ul; s <= automaton->maxStateId(); s++) {
158  if (automaton->hasState(s)) {
159  states.push_back(s);
160  Fsa::State const* state = automaton->fastState(s);
161  for (Fsa::State::const_iterator a = state->begin(); a != state->end(); ++a) {
162  verify(automaton->hasState(a->target_));
163  if (Speech::Score(a->weight_) >= Core::Type<Speech::Score>::max)
164  continue;
165  edges.push_back(Edge(s, a->target_, am->emissionIndex(a->input_), Speech::Score(a->weight_)));
166  }
167  if (state->isFinal()) {
168  final_states.push_back(std::make_pair(s, Speech::Score(state->weight())));
169  }
170  }
171  }
172 
173  toposort(states, final_states, edges);
174  filter_edges(edges);
175  make_single_final_state(states, final_states, edges);
176 
177  ExportedAutomaton result;
178  result.num_states = states.size();
179  result.num_edges = edges.size();
180  result.edges = std::vector<u32>(edges.size() * 3ul);
181  result.weights = std::vector<f32>(edges.size());
182 
183  for (size_t e = 0ul; e < edges.size(); e++) {
184  result.edges[e] = edges[e].from;
185  result.edges[e + edges.size()] = edges[e].to;
186  result.edges[e + 2 * edges.size()] = edges[e].emission_idx;
187  result.weights[e] = edges[e].weight;
188  }
189 
190  return result;
191 }
