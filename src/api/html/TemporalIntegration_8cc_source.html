70 void TemporalIntegration::transform(Vector<Sample>& out) {
71  u32 channels = out[0].size();
72  windowFunction_->setLength(out.size());
73  for (u32 ch = 0; ch < channels; ch++) {
74  out[0][ch] *= (windowFunction_->getWindow())[0]; // WARNING: probably fabs of out is missing, with Hanning window[0] was 0, no problem
75  for (u32 i = 1; i < out.size(); i++) {
76  out[0][ch] += fabs(out[i][ch]) * (windowFunction_->getWindow())[i];
77  }
78  }
79  Vector<Sample>::iterator iter = out.begin() + 1;
80  out.erase(iter, out.end());
81 }
