160 template<class T>
161 class Directory : private std::map<std::string, T*> {
162  typedef std::map<std::string, T*> Base;
163 
164 public:
165  typedef Directory<T> Self;
166  typedef T Entry;
167 
168  Directory() {}
169 
170  Directory(const Directory& o) {
171  for (typename Base::const_iterator i = o.begin(); i != o.end(); ++i) {
172  Base::insert(std::make_pair(i->first, new T(*i->second)));
173  }
174  }
175 
176  ~Directory() {
177  for (typename Base::iterator i = this->begin(); i != this->end(); ++i) {
178  delete i->second;
179  }
180  }
181 
182  bool hasKey(const std::string& k) const {
183  return Base::find(k) != this->end();
184  }
185 
186  /**
187  * Add an entry.
188  * The directory will take over ownership of the entry and delete
189  * it when the directory is destroyed.
190  */
191  void add(Entry* e) {
192  require(!hasKey(e->name()));
193  Base::insert(std::make_pair(e->name(), e));
194  }
195 
196  Entry* lookup(const std::string& k) const {
197  typename Base::const_iterator i = Base::find(k);
198  return (i != this->end()) ? i->second : 0;
199  }
200 };
