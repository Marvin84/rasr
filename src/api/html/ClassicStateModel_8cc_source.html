74 AllophoneAlphabet::AllophoneAlphabet(const Core::Configuration& config, ConstPhonologyRef phonology, Bliss::LexiconRef lexicon)
75  : Fsa::Alphabet(), Core::Component(config), phonology_(phonology), pi_(phonology->getPhonemeInventory()), isCrossWord_(phonology->isCrossWord()) {
76  nDisambiguators_ = 0;
77  std::string filename = paramAddFromFile(config);
78  if (!filename.empty()) {
79  load(filename);
80  log("%d allophones after adding allophones from file \"%s\"",
81  (int)allophones_.size(), filename.c_str());
82  }
83  if (paramAddFromLexicon(config)) {
84  add(lexicon);
85  log("%d allophones after adding allophones from lexicon",
86  (int)allophones_.size());
87  }
88  if (paramAddAll(config)) {
89  add(pi_);
90  log("%d allophones after adding all allophones",
91  (int)allophones_.size());
92  }
93  for (s32 p = 0; p < pi_->nPhonemes(); ++p) {
94  std::string phoneme(pi_->phoneme(p + 1)->symbol());
95  /* names of phonemes must not conflict with allophone specification
96  syntax, otherwise allophone-mapping may fail */
97  if (!(phoneme.find("{") == std::string::npos &&
98  phoneme.find("}") == std::string::npos &&
99  phoneme.find(".") == std::string::npos &&
100  phoneme.find("+") == std::string::npos))
101  error() << "bad phoneme name: '" << phoneme
102  << "', must not contain any of the following characters: { } + .";
103  }
104 
105  setSilence(lexicon);
106 }

141 void AllophoneAlphabet::add(Core::Ref<const Bliss::PhonemeInventory> pi) {
142  typedef std::vector<Bliss::Phoneme::Id> PhonemeIdList;
143  PhonemeIdList phonemeIds, ciPhonemeIds;
144  Bliss::PhonemeInventory::PhonemeIterator itPhoneme, endPhoneme;
145  for (Core::tie(itPhoneme, endPhoneme) = pi->phonemes(); itPhoneme != endPhoneme; ++itPhoneme) {
146  if (*itPhoneme == (Bliss::Phoneme*)/* WARNING: This cast is probably not intended that way... */ (long)Bliss::Phoneme::term)
147  continue;
148  if ((*itPhoneme)->isContextDependent())
149  phonemeIds.push_back((*itPhoneme)->id());
150  else
151  ciPhonemeIds.push_back((*itPhoneme)->id());
152  }
153 
154  typedef std::vector<PhonemeIdList::const_iterator> PhonemeIdIteratorList;
155  PhonemeIdIteratorList historyIts, futureIts;
156 
157  Core::ProgressIndicator* progress = new Core::ProgressIndicator("index all allophones");
158  progress->start();
159  for (s32 i = 0; i <= phonology_->maximumFutureLength(); ++i) {
160  for (s32 j = 0; j <= phonology_->maximumHistoryLength(); ++j) {
161  while (true) {
162  Phonology::SemiContext future, history;
163  PhonemeIdIteratorList::iterator itIt;
164  for (itIt = historyIts.begin(); itIt != historyIts.end(); ++itIt)
165  history.append(1, **itIt);
166  for (itIt = futureIts.begin(); itIt != futureIts.end(); ++itIt)
167  future.append(1, **itIt);
168 
169  for (PhonemeIdList::iterator itP = phonemeIds.begin(); itP != phonemeIds.end(); ++itP) {
170  Allophone* allo = new Allophone(*itP, Allophone::isWithinPhone);
171  allo->setHistory(history);
172  allo->setFuture(future);
173  insert(allo);
174  progress->notify();
175  }
176  if ((j == 0) || (isCrossWord_ && (j <= 1))) {
177  for (PhonemeIdList::iterator itP = phonemeIds.begin(); itP != phonemeIds.end(); ++itP) {
178  Allophone* allo = new Allophone(*itP, Allophone::isInitialPhone);
179  allo->setHistory(history);
180  allo->setFuture(future);
181  insert(allo);
182  progress->notify();
183  }
184  }
185  if ((i == 0) || (isCrossWord_ && (i <= 1))) {
186  for (PhonemeIdList::iterator itP = phonemeIds.begin(); itP != phonemeIds.end(); ++itP) {
187  Allophone* allo = new Allophone(*itP, Allophone::isFinalPhone);
188  allo->setHistory(history);
189  allo->setFuture(future);
190  insert(allo);
191  progress->notify();
192  }
193  }
194  if (((i == 0) && (j == 0)) || (isCrossWord_ && ((i <= 1) && (j <= 1)))) {
195  for (PhonemeIdList::iterator itP = phonemeIds.begin(); itP != phonemeIds.end(); ++itP) {
196  Allophone* allo = new Allophone(*itP, Allophone::isInitialPhone | Allophone::isFinalPhone);
197  allo->setHistory(history);
198  allo->setFuture(future);
199  insert(allo);
200  progress->notify();
201  }
202  }
203 
204  for (itIt = historyIts.begin(); itIt != historyIts.end(); ++itIt)
205  if (++(*itIt) == phonemeIds.end())
206  *itIt = phonemeIds.begin();
207  else
208  break;
209  if (itIt == historyIts.end()) {
210  for (itIt = futureIts.begin(); itIt != futureIts.end(); ++itIt)
211  if (++(*itIt) == phonemeIds.end())
212  *itIt = phonemeIds.begin();
213  else
214  break;
215  if (itIt == futureIts.end())
216  break;
217  }
218  }
219  historyIts.push_back(phonemeIds.begin());
220  }
221  historyIts.clear();
222  futureIts.push_back(phonemeIds.begin());
223  }
224  for (PhonemeIdList::iterator itP = ciPhonemeIds.begin(); itP != ciPhonemeIds.end(); ++itP) {
225  insert(new Allophone(*itP, Allophone::isWithinPhone));
226  progress->notify();
227  insert(new Allophone(*itP, Allophone::isInitialPhone));
228  progress->notify();
229  insert(new Allophone(*itP, Allophone::isFinalPhone));
230  progress->notify();
231  insert(new Allophone(*itP, Allophone::isInitialPhone | Allophone::isFinalPhone));
232  progress->notify();
233  }
234  progress->finish();
235  delete progress;
236 }

241 std::pair<AllophoneAlphabet::AllophoneMap::iterator, bool> AllophoneAlphabet::createAndInsert(const std::vector<bool>& cd,
242  const Bliss::Pronunciation& p, s32 b, s32 c, s32 e, s16 boundary) const {
243  Allophone* allo = new Allophone(p[c], boundary);
244  for (s32 i = c - 1; (i >= b); --i)
245  phonology_->appendHistory(*allo, p[i]);
246  for (s32 i = c + 1; (i <= e); ++i)
247  phonology_->appendFuture(*allo, p[i]);
248  return insert(allo);
249 }

251 void AllophoneAlphabet::add(Bliss::LexiconRef lexicon) {
252  require(pi_);
253  const s32 maxHistory = phonology_->maximumHistoryLength();
254  const s32 maxFuture = phonology_->maximumFutureLength();
255 
256  std::vector<bool> cd(pi_->nPhonemes() + 1);
257  Bliss::PhonemeInventory::PhonemeIterator itPhoneme, endPhoneme;
258  for (Core::tie(itPhoneme, endPhoneme) = pi_->phonemes(); itPhoneme != endPhoneme; ++itPhoneme)
259  cd[(*itPhoneme)->id()] = (*itPhoneme)->isContextDependent();
260 
261  AllophoneList leftCwAllophones;
262  AllophoneList rightCwAllophones;
263  AllophoneList leftRightCwAllophones;
264  PhonemeIdSet initialPhonemeSet;
265  PhonemeIdSet finalPhonemeSet;
266 
267  Core::ProgressIndicator* progress;
268  progress = new Core::ProgressIndicator("index allophones from lexicon: within word");
269  progress->start(lexicon->nPronunciations());
270  std::pair<AllophoneMap::iterator, bool> insAllo;
271  Bliss::Lexicon::PronunciationIterator lexicon_pron, lexicon_pron_end;
272  for (Core::tie(lexicon_pron, lexicon_pron_end) = lexicon->pronunciations(); lexicon_pron != lexicon_pron_end; ++lexicon_pron) {
273  const Bliss::Pronunciation& p(**lexicon_pron);
274  s32 maxPronId = p.length() - 1;
275  if (maxPronId < 0)
276  continue;
277  if (cd[p[0]])
278  initialPhonemeSet.insert(p[0]);
279  if (cd[p[maxPronId]])
280  finalPhonemeSet.insert(p[maxPronId]);
281 
282  // split by cd/id
283  std::vector<s32> contextChanges;
284  for (s32 i = 1; i <= maxPronId; ++i)
285  if (cd[p[i - 1]] != cd[p[i]])
286  contextChanges.push_back(i - 1);
287  contextChanges.push_back(maxPronId);
288 
289  s32 firstPronId, lastPronId = -1;
290  for (std::vector<s32>::const_iterator itLastPronId = contextChanges.begin();
291  itLastPronId != contextChanges.end(); ++itLastPronId) {
292  firstPronId = lastPronId + 1;
293  lastPronId = *itLastPronId;
294 
295  // configure next segment
296  s32 historyLength, futureLength;
297  bool isLeftCw, isRightCw;
298  if (cd[p[firstPronId]]) {
299  historyLength = maxHistory;
300  futureLength = maxFuture;
301  isLeftCw = (isCrossWord_ && (firstPronId == 0));
302  isRightCw = (isCrossWord_ && (lastPronId == maxPronId));
303  }
304  else {
305  historyLength = 0;
306  futureLength = 0;
307  isLeftCw = false;
308  isRightCw = false;
309  }
310  s16 initialFlag = (firstPronId == 0) ? Allophone::isInitialPhone : Allophone::isWithinPhone;
311  s16 finalFlag = (lastPronId == maxPronId) ? Allophone::isFinalPhone : Allophone::isWithinPhone;
312 
313  // special treatment for segments of length one
314  if (firstPronId == lastPronId) {
315  insAllo = createAndInsert(cd, p, firstPronId, firstPronId, firstPronId, initialFlag | finalFlag);
316  if (insAllo.second) {
317  if (isLeftCw && (historyLength > 0))
318  leftCwAllophones.push_back(insAllo.first->first);
319  if (isRightCw && (futureLength > 0))
320  rightCwAllophones.push_back(insAllo.first->first);
321  if (isLeftCw && isRightCw && (historyLength > 0) && (futureLength > 0))
322  leftRightCwAllophones.push_back(insAllo.first->first);
323  }
324  continue;
325  }
326 
327  // let's rock
328  s32 b = firstPronId, c = firstPronId, e = firstPronId + futureLength;
329  if (historyLength == 0) {
330  // initial allophone
331  insAllo = createAndInsert(cd, p, firstPronId, firstPronId, std::min(e, lastPronId), initialFlag);
332  if (insAllo.second && isRightCw && (e > lastPronId))
333  rightCwAllophones.push_back(insAllo.first->first);
334  ++b;
335  ++c;
336  ++e;
337  }
338  else {
339  // initial cw allophone
340  insAllo = createAndInsert(cd, p, firstPronId, firstPronId, std::min(e, lastPronId), initialFlag);
341  if (insAllo.second) {
342  if (isLeftCw)
343  leftCwAllophones.push_back(insAllo.first->first);
344  if ((e > lastPronId) && isRightCw) {
345  if (isLeftCw)
346  leftRightCwAllophones.push_back(insAllo.first->first);
347  rightCwAllophones.push_back(insAllo.first->first);
348  }
349  }
350  // left and left/right cw allophones
351  for (++c, ++e; c < std::min(firstPronId + historyLength, lastPronId); ++c, ++e) {
352  insAllo = createAndInsert(cd, p, firstPronId, c, std::min(e, lastPronId), Allophone::isWithinPhone);
353  if (insAllo.second) {
354  if (isLeftCw)
355  leftCwAllophones.push_back(insAllo.first->first);
356  if ((e > lastPronId) && isRightCw) {
357  if (isLeftCw)
358  leftRightCwAllophones.push_back(insAllo.first->first);
359  rightCwAllophones.push_back(insAllo.first->first);
360  }
361  }
362  }
363  }
364  if (futureLength == 0) {
365  // within allophones
366  for (; c < lastPronId; ++b, ++c)
367  createAndInsert(cd, p, b, c, c, Allophone::isWithinPhone);
368  // final allophone
369  createAndInsert(cd, p, b, lastPronId, lastPronId, finalFlag);
370  }
371  else {
372  // within allophones
373  for (; e <= lastPronId; ++b, ++c, ++e)
374  createAndInsert(cd, p, b, c, e, Allophone::isWithinPhone);
375  // right cw allophones
376  for (; c < lastPronId; ++b, ++c) {
377  insAllo = createAndInsert(cd, p, b, c, lastPronId, Allophone::isWithinPhone);
378  if (insAllo.second && isRightCw)
379  rightCwAllophones.push_back(insAllo.first->first);
380  }
381  // final cw allophone
382  insAllo = createAndInsert(cd, p, b, lastPronId, lastPronId, finalFlag);
383  if (insAllo.second) {
384  if ((historyLength > lastPronId) && isLeftCw) {
385  leftCwAllophones.push_back(insAllo.first->first);
386  if (isRightCw)
387  leftRightCwAllophones.push_back(insAllo.first->first);
388  }
389  if (isRightCw)
390  rightCwAllophones.push_back(insAllo.first->first);
391  }
392  }
393  }
394  progress->notify();
395  }
396  progress->finish();
397  delete progress;
398  if (isCrossWord_) {
399  typedef std::vector<Bliss::Phoneme::Id> PhonemeIdList;
400  PhonemeIdList initialPhonemes(initialPhonemeSet.begin(), initialPhonemeSet.end());
401  PhonemeIdList finalPhonemes(finalPhonemeSet.begin(), finalPhonemeSet.end());
402 
403  progress = new Core::ProgressIndicator("index allophones from lexicon: across word");
404  progress->start();
405  // left cw allophones
406  for (AllophoneList::const_iterator itAllo = leftCwAllophones.begin(); itAllo != leftCwAllophones.end(); ++itAllo) {
407  const Allophone& allo(**itAllo);
408  for (PhonemeIdList::const_iterator itFinal = finalPhonemes.begin(); itFinal != finalPhonemes.end(); ++itFinal) {
409  Allophone* cwAllo = new Allophone(allo);
410  phonology_->appendHistory(*cwAllo, *itFinal);
411  insert(cwAllo);
412  progress->notify();
413  }
414  }
415  // right cw allophones
416  for (AllophoneList::const_iterator itAllo = rightCwAllophones.begin(); itAllo != rightCwAllophones.end(); ++itAllo) {
417  const Allophone& allo(**itAllo);
418  for (PhonemeIdList::const_iterator itInitial = initialPhonemes.begin(); itInitial != initialPhonemes.end(); ++itInitial) {
419  Allophone* cwAllo = new Allophone(allo);
420  phonology_->appendFuture(*cwAllo, *itInitial);
421  insert(cwAllo);
422  progress->notify();
423  }
424  }
425  // left/right cw allophones
426  for (AllophoneList::const_iterator itAllo = leftRightCwAllophones.begin(); itAllo != leftRightCwAllophones.end(); ++itAllo) {
427  const Allophone& allo(**itAllo);
428  for (PhonemeIdList::const_iterator itFinal = finalPhonemes.begin(); itFinal != finalPhonemes.end(); ++itFinal)
429  for (PhonemeIdList::const_iterator itInitial = initialPhonemes.begin();
430  itInitial != initialPhonemes.end(); ++itInitial) {
431  Allophone* cwAllo = new Allophone(allo);
432  phonology_->appendHistory(*cwAllo, *itFinal);
433  phonology_->appendFuture(*cwAllo, *itInitial);
434  insert(cwAllo);
435  progress->notify();
436  }
437  }
438  progress->finish();
439  delete progress;
440  }
441 }

443 void AllophoneAlphabet::load(const std::string& filename) {
444  verify(!filename.empty());
445  if (!allophones_.empty())
446  warning("Indices of allophones in file \"%s\" seems NOT to match indices in current allophone alphabet", filename.c_str());
447  Core::CompressedInputStream cis(filename);
448  if (!cis.isOpen())
449  error() << "Failed opening allophone file '" << filename << "' for reading";
450  std::string line;
451  Core::ProgressIndicator* progress = new Core::ProgressIndicator("index allophones from file");
452  progress->start();
453  while (cis) {
454  std::getline(cis, line);
455  Core::stripWhitespace(line);
456  if (!line.empty() && (*line.c_str() != '#')) {
457  if (!insert(fromString(line)).second)
458  warning("Indices of allophones in file \"%s\" seems NOT to match indices in current allophone alphabet; allophone %s does already exist",
459  filename.c_str(), line.c_str());
460  progress->notify();
461  }
462  }
463  progress->finish();
464  delete progress;
465 }

535 Allophone AllophoneAlphabet::fromString(const std::string& s) const {
536  const Bliss::Phoneme* phoneme;
537  const char * c, *d, *e, *f;
538  c = d = s.c_str();
539  for (; (*d != '{') && (*d != '\0'); ++d)
540  ;
541  verify(*d != '\0');
542  phoneme = pi_->phoneme(std::string(c, std::string::size_type(d - c)));
543  if (!phoneme)
544  criticalError("unknown phoneme: '%s'", std::string(c, std::string::size_type(d - c)).c_str());
545  Allophone allo(phoneme->id(), Allophone::isWithinPhone);
546  c = d;
547  for (; ((*d != '+') && (*d != '\0')); ++d)
548  ;
549  verify(*d != '\0');
550  e = f = d;
551  do {
552  f = --e;
553  for (; *e != '{'; --e)
554  ;
555  std::string p(e + 1, std::string::size_type(f - e));
556  if (p == "#") {
557  verify(allo.history().empty() && (*e == '{'));
558  break;
559  }
560  phoneme = pi_->phoneme(p);
561  if (!phoneme)
562  criticalError() << "unknown phoneme: " << p << " taken from allophone " << s;
563  phonology_->appendHistory(allo, phoneme->id());
564  } while (*e != '{');
565  c = e = f = d;
566  for (; (*d != '}') && (*d != '\0'); ++d)
567  ;
568  verify(*d != '\0');
569  do {
570  f = ++e;
571  for (; *e != '}'; ++e)
572  ;
573  std::string p(f, std::string::size_type(e - f));
574  if (p == "#") {
575  verify(allo.future().empty() && (*e == '}'));
576  break;
577  }
578  phoneme = pi_->phoneme(p);
579  if (!phoneme)
580  criticalError() << "unknown phoneme: " << p << " taken from allophone " << s;
581  phonology_->appendFuture(allo, phoneme->id());
582  } while (*e != '}');
583  c = ++d;
584  for (; (*d != '\0'); ++d)
585  ;
586  std::string b(c, std::string::size_type(d - c));
587  if (b.empty()) {
588  // pass
589  }
590  else if (b == "@i") {
591  allo.boundary = Allophone::isInitialPhone;
592  }
593  else if (b == "@f") {
594  allo.boundary = Allophone::isFinalPhone;
595  }
596  else if (b == "@i@f") {
597  allo.boundary = Allophone::isInitialPhone | Allophone::isFinalPhone;
598  }
599  else
600  defect();
601  return allo;
602 }

730 Fsa::LabelId AllophoneStateAlphabet::index(const std::string& symbol, bool create) const {
731  //
732  // @todo: avoid hard coding lengths and special delimiter symbols
733  // what about using a delimiter string or class instead of hard coding '.'?
734  //
735  // Example:
736  // a{b+c}@i@f.10
737  // <allo-center><allo-context-delimiter [{]><allo-left><allo-neighbor-delimiter [+]><allo-right><allo-context-delimiter [}]>[@i][@f]<state-delimiter [.]><state-number>
738 
739  // we search for last occurrence of '.' in order to allow more than 9 HMM states
740  const size_t found = symbol.find_last_of(".");
741  //
742  // there must be at least one state-delimiter -
743  // larger than 5 in order to use the '.' also as a phoneme symbol
744  // e.g. .{.+.}.10
745  //
746  // @todo: would be better to check if a closing context-delimiter has been
747  // found before the state-delimiter, as '5' depends again on other delimiter
748  // lengths!
749  ensure((found != std::string::npos) && (found > 5)); // '5' depends again on other delimiter lengths
750  DBG(DEBUG_ALLOPHONE_INFO) << VAR(symbol.substr(0, found)) << " " << VAR(symbol.substr(found + 1)) << ENDDBG;
751  Allophone allo = allophoneAlphabet_->fromString(symbol.substr(0, found));
752  s16 state;
753  if (!Core::strconv(symbol.substr(found + 1), state))
754  defect();
755  return index(&allo, state, create);
756 }
