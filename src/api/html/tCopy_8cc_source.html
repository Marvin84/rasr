61 template<class _Automaton>
62 void copy(StorageAutomaton<_Automaton>* f,
63  const std::string& str) {
64  typedef typename _Automaton::Arc _Arc;
65  typedef typename _Automaton::State _State;
66 
67  f->clear();
68  f->setType(Fsa::TypeAcceptor);
69 
70  f->addProperties(Fsa::PropertySorted);
71  f->addProperties(Fsa::PropertyLinear | Fsa::PropertyAcyclic);
72 
73  Fsa::StateId s = 0;
74  _State* sp = new _State(s++);
75  f->setInitialStateId(sp->id());
76  sp->weight_ = f->semiring()->one();
77 
78  Fsa::StaticAlphabet* sinput(0);
79  Fsa::ConstAlphabetRef input;
80  if (f->getInputAlphabet())
81  input = f->getInputAlphabet();
82  else {
83  sinput = new Fsa::StaticAlphabet();
84  f->setInputAlphabet(Fsa::ConstAlphabetRef(sinput));
85  }
86 
87  Core::StringTokenizer tokenizer(str);
88  for (Core::StringTokenizer::Iterator token = tokenizer.begin(); token != tokenizer.end(); ++token) {
89  std::string tmp = *token;
90  if (!tmp.empty()) {
91  _Arc* a = sp->newArc();
92  a->input_ = a->output_ = (sinput ? sinput->addSymbol(tmp)
93  : input->index(tmp));
94  a->weight_ = f->semiring()->one();
95  f->setState(sp);
96  sp = new _State(s++);
97  a->target_ = sp->id();
98  }
99  }
100  sp->setTags(Fsa::StateTagFinal);
101  sp->weight_ = f->semiring()->one();
102  f->setState(sp);
103  f->normalize();
104 }

106 template<class _Automaton>
107 void copy(StorageAutomaton<_Automaton>* f,
108  const Fsa::Hash<std::string, Core::StringHash>& tokens,
109  u32 sausage) {
110  typedef typename _Automaton::Arc _Arc;
111  typedef typename _Automaton::State _State;
112 
113  f->setType(Fsa::TypeAcceptor);
114  f->addProperties(Fsa::PropertySorted);
115  f->addProperties(Fsa::PropertyLinear | Fsa::PropertyAcyclic);
116 
117  _State* sp = new _State(0);
118  f->setInitialStateId(sp->id());
119  sp->weight_ = f->semiring()->one();
120  Fsa::StaticAlphabet* input = new Fsa::StaticAlphabet();
121  f->setInputAlphabet(Fsa::ConstAlphabetRef(input));
122 
123  if (sausage == 0) {
124  for (Fsa::Hash<std::string, Core::StringHash>::const_iterator token =
125  tokens.begin();
126  token != tokens.end(); ++token)
127  sp->newArc(sp->id(), f->semiring()->one(), input->addSymbol(*token));
128  sp->setTags(Fsa::StateTagFinal);
129  sp->weight_ = f->semiring()->one();
130  f->setState(sp);
131  }
132  else {
133  for (u32 i = 0; i < sausage; ++i) {
134  for (Fsa::Hash<std::string, Core::StringHash>::const_iterator
135  token = tokens.begin();
136  token != tokens.end(); ++token)
137  sp->newArc(i + 1, f->semiring()->one(), input->addSymbol(*token));
138  f->setState(sp);
139  sp = new _State(i + 1);
140  }
141  }
142 
143  sp->setTags(Fsa::StateTagFinal);
144  sp->weight_ = f->semiring()->one();
145  f->setState(sp);
146  f->normalize();
147 }
