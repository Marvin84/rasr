23 /** Aggregate of instances of class T.
24  * TypedAggregate<T> collects objects strictly of the same type coming from different streams.
25  * Note:
26  * -It does not support polymoric I/O.
27  * -Aggregate is a general implementation of the same problem supporting polymorfism.
28  * Advantage of this class compared to Aggregate is that the type of the elements in known.
29  * Thus there is no need for type checking when accessing the elements.
30  */

173 template<class T>
174 bool TypedDisaggregateNode<T>::work(PortId) {
175  DataPtr<TypedAggregate<DataType>> in;
176  DataPtr<DataType> out;
177  if (!getData(0, in)) {
178  for (OutputMap::const_iterator i = outputs_.begin(); i != outputs_.end(); ++i) {
179  putData(i->second, out.get());
180  }
181  return true;
182  }
183  bool ok = true;
184  for (OutputMap::const_iterator i = outputs_.begin(); i != outputs_.end(); ++i) {
185  if (i->first >= in->size()) {
186  error("invalid component %d", i->first);
187  return false;
188  }
189  else {
190  ok = putData(i->second, (*in)[i->first].get()) && ok;
191  }
192  }
193  return ok;
194 }
