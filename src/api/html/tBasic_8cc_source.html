27 class NormalizeAutomaton : public SlaveAutomaton<_Automaton> {
28  typedef SlaveAutomaton<_Automaton> Precursor;
29 
30 public:
31  typedef typename _Automaton::State _State;
32  typedef typename _Automaton::ConstStateRef _ConstStateRef;
33  typedef typename _Automaton::ConstRef _ConstAutomatonRef;
34 
35 private:
36  struct StateIdKey {
37  u32 operator()(Fsa::StateId s) {
38  return s;
39  }
40  };
41  mutable Fsa::Hash<Fsa::StateId, StateIdKey> states_;
42 
43 public:
44  NormalizeAutomaton(_ConstAutomatonRef f)
45  : Precursor(f) {}
46  virtual Fsa::StateId initialStateId() const {
47  Fsa::StateId s = Precursor::fsa_->initialStateId();
48  if (s != Fsa::InvalidStateId)
49  return states_.insert(s);
50  return s;
51  }
52  virtual _ConstStateRef getState(Fsa::StateId s) const {
53  if (s >= states_.size())
54  return _ConstStateRef();
55  _State* sp = new _State(*Precursor::fsa_->getState(states_[s]));
56  sp->setId(s);
57  for (typename _State::iterator a = sp->begin(); a != sp->end(); ++a)
58  a->target_ = states_.insert(a->target());
59  return _ConstStateRef(sp);
60  }
61  virtual std::string describe() const {
62  return "normalize(" + Precursor::fsa_->describe() + ")";
63  }
64 
65  /** State in the original automaton.
66  * Given a state of the normalization result, return the
67  * corresponding original state of the automaton. */
68  virtual Fsa::StateId originalStateId(Fsa::StateId s) const {
69  return states_[s];
70  }
71 };

107 template<class _Automaton>
108 class TrimAutomaton : public SlaveAutomaton<_Automaton> {
109  typedef SlaveAutomaton<_Automaton> Precursor;
110 
111 public:
112  typedef typename _Automaton::State _State;
113  typedef typename _Automaton::ConstStateRef _ConstStateRef;
114  typedef typename _Automaton::ConstRef _ConstAutomatonRef;
115  typedef CoaccessibleDfsState<_Automaton> _CoaccessibleDfsState;
116 
117 private:
118  std::vector<bool> accAndCoacc_;
119 
120 public:
121  TrimAutomaton(_ConstAutomatonRef f, bool progress)
122  : Precursor(f) {
123  Core::ProgressIndicator* p = 0;
124  if (progress)
125  p = new Core::ProgressIndicator("connecting", "states");
126  _CoaccessibleDfsState v(f);
127  v.dfs(p);
128  if (p)
129  delete p;
130  accAndCoacc_.resize(v.size());
131  for (Fsa::StateId s = 0; s < accAndCoacc_.size(); ++s)
132  accAndCoacc_[s] = //(v.flags()[s] & AccessibleDfsState::FlagAccessible) &&
133  (v.flags()[s] & _CoaccessibleDfsState::FlagCoaccessible);
134  }
135  virtual Fsa::StateId initialStateId() const {
136  Fsa::StateId s = Precursor::fsa_->initialStateId();
137  if ((s < accAndCoacc_.size()) && (accAndCoacc_[s]))
138  return s;
139  return Fsa::InvalidStateId;
140  }
141  virtual _ConstStateRef getState(Fsa::StateId s) const {
142  if (accAndCoacc_[s]) {
143  _ConstStateRef _s = Precursor::fsa_->getState(s);
144  _State* sp = new _State(_s->id(), _s->tags(), _s->weight_);
145  for (typename _State::const_iterator a = _s->begin(); a != _s->end(); ++a)
146  if (accAndCoacc_[a->target()])
147  *sp->newArc() = *a;
148  sp->minimize();
149  return _ConstStateRef(sp);
150  }
151  return _ConstStateRef();
152  }
153  virtual size_t getMemoryUsed() const {
154  return Precursor::fsa_->getMemoryUsed() + accAndCoacc_.size() / 8;
155  }
156  virtual void dumpMemoryUsage(Core::XmlWriter& o) const {
157  o << Core::XmlOpen("trim");
158  Precursor::fsa_->dumpMemoryUsage(o);
159  o << Core::XmlFull("states", accAndCoacc_.size() / 8) << Core::XmlClose("trim");
160  }
161  virtual std::string describe() const {
162  return "trim(" + Precursor::fsa_->describe() + ")";
163  }
164 };
