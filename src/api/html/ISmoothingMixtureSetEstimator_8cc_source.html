60 bool ISmoothingMixtureSetEstimator::distributeISmoothingMixtureSet(const MixtureSet& mixtureSet) {
61  verify(parent_);
62  if (mixtureSet.dimension() != parent_->dimension()) {
63  return false;
64  }
65  MixtureSetEstimatorIndexMap indexMaps(*parent_);
66  if (mixtureSet.nMixtures() != parent_->nMixtures()) {
67  return false;
68  }
69  for (MixtureIndex i = 0; i < mixtureSet.nMixtures(); ++i) {
70  setIMixture((*parent_->getMixtureEstimators())[i], mixtureSet.mixture(i), weightsConstant());
71  }
72  if (mixtureSet.nDensities() != indexMaps.densityMap().size()) {
73  return false;
74  }
75  for (DensityIndex i = 0; i < mixtureSet.nDensities(); ++i) {
76  setIDensity(indexMaps.densityMap()[i], mixtureSet.density(i));
77  }
78  if (mixtureSet.nMeans() != indexMaps.meanMap().size()) {
79  return false;
80  }
81  for (MeanIndex i = 0; i < mixtureSet.nMeans(); ++i) {
82  setIMean(indexMaps.meanMap()[i], mixtureSet.mean(i), meansConstant());
83  }
84  if (mixtureSet.nCovariances() != indexMaps.covarianceMap().size()) {
85  return false;
86  }
87  CovarianceToMeanSetMap meanSetMap(indexMaps.densityMap().indexToPointerMap());
88  for (CovarianceIndex i = 0; i < mixtureSet.nCovariances(); ++i) {
89  setICovariance(indexMaps.covarianceMap()[i], mixtureSet.covariance(i), meansConstant());
90  }
91  log("distribute i-smoothing mixture set ...done");
92  return true;
93 }

98 Sum ISmoothingMixtureSetEstimator::getObjectiveFunction(const MixtureSet& previousMixtureSet,
99  const MixtureSetEstimatorIndexMap& indexMaps,

104  for (MixtureIndex i = 0; i < parent_->nMixtures(); ++i) {
105  iOfW += getMixtureObjectiveFunction(mixtureEstimators[i]);
106  }
107  log("weights-objective-function: ") << iOfW;
108 
109  Sum iOfM = 0;
110  for (CovarianceIndex i = 0; i < indexMaps.covarianceMap().size(); ++i) {
111  iOfM += getCovarianceObjectiveFunction(indexMaps.covarianceMap()[i], meanSetMap);
112  }
113  log("means-objective-function: ") << iOfM;
114  return iOfW + iOfM;
115 }
