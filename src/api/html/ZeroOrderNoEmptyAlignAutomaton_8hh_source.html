25  vertical = 2 } Transitions;

28  u32 J;

32 public:
33  // make all this abstract
34  ZeroOrderNoEmptyAlignAutomaton(Core::Configuration& config,
35  const std::string& source,
36  const std::string& target,
37  const TransitionProbs& transitionProbs,
38  const TranslationLexicon* lex,
39  const double factorLexicon = 1.0)
40  : AlignAutomaton(config, source, target, transitionProbs, factorLexicon),

44  M(3),

47  virtual ConstStateRef getState(StateId s) const {
48  State* sp = new State(s);
49 
50  u32 si(s);
51  u32 jprev(si % J);
52  u32 m(si / J % M);
53  u32 i(si / J / M / (J + 1));
54  u32 j(si / J / M % (J + 1));
55 
56  // State id is previous_source + J*(m + M*(j + i*(J+1)))
57 
58  bool doHorizontal(m == diagonal || m == horizontal);
59  bool doVertical(m == diagonal || m == vertical);
60 
61  if (i < I && doVertical) { // when not at the TOP of lattice, make vertical movement
62  if (j > 0) {
63  // in the vertical case, we move just up in the
64  // lattice while reading a target word
65  // for weighting we keep the source word we just read
66 
67  std::vector<std::string> x;
68  x.push_back(source_[jprev]);
69  x.push_back(target_[i]);
70  f64 weight(lexicon_->getProb(x) * factorLexicon_);
71  if (transitionProbs_)
72  weight += transitionProbs_.exponent * transitionProbs_.v;
73 
74  StateId targetState(si + J * ((J + 1) * M - m + vertical));
75  sp->newArc(targetState, Fsa::Weight(weight), Epsilon, outputSentence_[i]);
76  }
77  else { // j==0
78  // in this case we do not know, which of the
79  // source words will be read first.
80  //
81  // at this point we have to hypothesize all source
82  // words.
83  //
84  // later (when going diagonal for the first time,
85  // we have to ensure, that the source word we read
86  // is the same that we took for the weighting of
87  // the arc at this point
88 
89  StateId targetState(si + J * ((J + 1) * M - m + vertical) - jprev);
90  for (u32 jt = 0; jt < J; ++jt) {
91  std::vector<std::string> x;
92  x.push_back(source_[jt]);
93  x.push_back(target_[i]);
94  f64 weight(lexicon_->getProb(x) * factorLexicon_);
95  if (transitionProbs_)
96  weight += transitionProbs_.exponent * transitionProbs_.v;
97 
98  sp->newArc(targetState + jt, Fsa::Weight(weight), Epsilon, outputSentence_[i]);
99  }
100  }
101  }
102 
103  if (j < J && doHorizontal) { // when not at the RIGHT border of the lattice make horizontal movement
104  StateId targetState(si + J * (M - m + horizontal) - jprev);
105  u32 iprev = (i == 0) ? i : i - 1;
106  for (u32 jt = 0; jt < J; ++jt) {
107  std::vector<std::string> x;
108  x.push_back(source_[jt]);
109  x.push_back(target_[iprev]);
110  f64 weight(lexicon_->getProb(x) * factorLexicon_);
111  if (transitionProbs_)
112  weight += transitionProbs_.exponent * transitionProbs_.h;
113 
114  sp->newArc(targetState + jt, Fsa::Weight(weight), inputSentence_[jt], Epsilon);
115  }
116  }
117  if ((j < J) && (i < I)) { // when in the middle of the lattice, do diagonal movement
118  if (j > 0) {
119  StateId targetState(si + J * ((J + 1) * M + M - m + diagonal) - jprev);
120  for (u32 jt = 0; jt < J; ++jt) {
121  std::vector<std::string> x;
122  x.push_back(source_[jt]);
123  x.push_back(target_[i]);
124  f64 weight(lexicon_->getProb(x) * factorLexicon_);
125  if (transitionProbs_)
126  weight += transitionProbs_.exponent * transitionProbs_.d;
127 
128  sp->newArc(targetState + jt, Fsa::Weight(weight), inputSentence_[jt], outputSentence_[i]);
129  }
130  }
131  else { // j==0
132  // as mentioned above when going vertical, in the
133  // beginning of the sentence we have to take care
134  // that we now read the source word we initially
135  // hypothized
136 
137  StateId targetState(si + J * ((J + 1) * M + M - m + diagonal));
138 
139  std::vector<std::string> x;
140  x.push_back(source_[jprev]);
141  x.push_back(target_[i]);
142  f64 weight(lexicon_->getProb(x) * factorLexicon_);
143  if (transitionProbs_)
144  weight += transitionProbs_.exponent * transitionProbs_.d;
145 
146  sp->newArc(targetState, Fsa::Weight(weight), inputSentence_[jprev], outputSentence_[i]);
147  }
148  }
149 
150  if (i == I && j == J) {
151  sp->setFinal(this->semiring()->one());
152  }
153  return ConstStateRef(sp);
154  }
155  virtual std::string describe() const {
156  return std::string("simpleAlignAutomaton()");
157  }
