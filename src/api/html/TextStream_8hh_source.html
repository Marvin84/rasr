28 /**
29  * Stream for text output.
30  *
31  * Use this for writing plain text files. For XML use XmlOutoutStream.
32  *
33  * Features if TextOutputStream:
34  * - word wrap
35  * - indentation
36  * - character set conversion
37  * Remeber: all internal string are considered UTF-8
38  *
39  * TextOutputStream redefines some ASCII control characters to
40  * control text formatting. Since these codes are old teletype
41  * legacy and have no use in modern computing environments, we
42  * should get away with this.
43  */

47  enum FormattingHint {
48  // Note: In XML documents only CR, LF and HT are permitted.
49  BEL = 7,
50  BS = 8,

52  LF = 10,
53  VT = 11,
54  FF = 12,
55  CR = 13,
56 
57  indentPlus = 1, /**< increase indentation level by one (effective after next newline) */
58  indentMinus = 2, /**< decrease indentation level by one (effective after next newline) */
59  describeEncoding = 3, /**< output name of output encoding used */
60  protect = 4, /**< temporarily disable word-wrap (may be nested) */
61  unprotect = 5, /**< reenable word-wrap */
62  newWord = 6, /**< optional line break (even when protect is set) */
63  softBlank = BEL, /**< consecutive soft blanks are replaced by a single normal blank */
64  lineBreak = LF, /**< start a new line uncoditionally */
65  newLine = 14 /**< start a new line, ignore if the current position is the beginning of a line */
