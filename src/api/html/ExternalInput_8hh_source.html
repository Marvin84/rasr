27 /** Wrapper for network input port
28  * Supported features:
29  * - sets sample rate,
30  * - handles startTime, endTime.
31  */
32 template<class T>
33 class ExternalInput : public virtual Core::Component {
34 private:
35  typedef Core::Component Precursor;
36 
37 public:
38  ExternalInput(const Core::Configuration& c, Network* network)
39  : Precursor(c) {
40  require(network != 0);
41  network_ = network;
42  setSampleRate(paramExternalInputSampleRate(c));
43  configure();
44  setInput();
45  }
46  virtual ~ExternalInput() {}

48  void setSampleRate(Time sampleRate) {
49  require(sampleRate > 0);
50  sampleRate_ = sampleRate;
51  configure();
52  }

54  void setInput() {
55  if (network_->nInputs()) {
56  if (network_->nInputs() > 1)
57  warning("the first network input has been set as an input");
58  inputPortId_ = 0;
59  }
60  else
61  criticalError("network doesn't have any input port");
62  }

64  void setInput(const std::string& name) {
65  inputPortId_ = network_->getInput(name);
66  if (inputPortId_ == IllegalPortId)
67  criticalError("input port with name \"%s\" doesn't exist", name.c_str());
68  else {
69  reset();
70  configure();
71  }
72  }

74  void putData(Vector<T>* v) {
75  startTime_ = endTime_;
76  endTime_ = endTime_ + Time(v->size()) / sampleRate_;
77  v->setStartTime(startTime_);
78  v->setEndTime(endTime_);
79  network_->putData(0, v);
80  }

95  virtual bool configure() {
96  bool isConfigured = network_->configure();
97  Core::Ref<Attributes> a(new Attributes());
98  a->set("sample-rate", sampleRate_);
99  a->set("sample-size", sizeof(T) * 8);
100  a->set("datatype", Vector<T>::type()->name());
101  isConfigured = isConfigured && network_->putAttributes(0, a);
102  reset();
103  return isConfigured;
104  }
