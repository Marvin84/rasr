33 f64 slotEntropy(const PosteriorCn::Slot& slot) {
34  if (slot.size() == 1)
35  return 0.0;
36  f64 entropy = 0.0;
37  for (PosteriorCn::Slot::const_iterator itArc = slot.begin(), endArc = slot.end(); itArc != endArc; ++itArc)
38  entropy += itArc->score * ::log2(itArc->score);
39  return -entropy;
40 }

42 /**
43  * minimum entropy weighting
44  **/
45 void minEntropyWeights(WeightsList& weightsList, const ConstPosteriorCnRefList& cns, const ScoreList& _weights) {
46  const std::vector<f64> weights(_weights.begin(), _weights.end());
47  for (u32 i = 0; i < weightsList.size(); ++i) {
48  ScoreList& slotWeights = weightsList[i];
49  std::fill(slotWeights.begin(), slotWeights.end(), 0.0);
50  f64 minEntropy = Core::Type<f64>::max;
51  u32 minEntropyIdx = Core::Type<u32>::max;
52  std::vector<f64>::const_iterator itWeight = weights.begin();
53  for (u32 j = 0; j < cns.size(); ++j, ++itWeight) {
54  const PosteriorCn::Slot& slot = (i < cns[j]->size()) ? (*cns[j])[i] : EpsSlot;
55  const f64 entropy = *itWeight * slotEntropy(slot);
56  if (entropy < minEntropy) {
57  minEntropy = entropy;
58  minEntropyIdx = j;
59  }
60  }
61  slotWeights[minEntropyIdx] = 1.0;
62  }
63 }

66  * inverse, thresholded entropy weighting
67  **/
68 f64 inverseSlotEntropy(const PosteriorCn::Slot& slot) {
69  f64 entropy = slotEntropy(slot);
70  if (entropy < 0.008)
71  return 125.0;
72  return 1.0 / entropy;

75 void inverseEntropyWeights(WeightsList& weightsList, const ConstPosteriorCnRefList& cns, const ScoreList& _weights) {
76  const std::vector<f64> weights(_weights.begin(), _weights.end());
77  std::vector<f64> weightedInverseEntropies(_weights.size(), 0.0);
78  for (u32 i = 0; i < weightsList.size(); ++i) {
79  f64 sumWeightedInverseEntropy = 0.0;
80  std::vector<f64>::const_iterator itWeight = weights.begin();
81  for (u32 j = 0; j < cns.size(); ++j, ++itWeight) {
82  const PosteriorCn::Slot& slot = (i < cns[j]->size()) ? (*cns[j])[i] : EpsSlot;
83  f64& weightedInverseEntropy = weightedInverseEntropies[j];
84  weightedInverseEntropy = *itWeight * inverseSlotEntropy(slot);
85  sumWeightedInverseEntropy += weightedInverseEntropy;
86  }
87  const f64 normWeightedInverseEntropy = 1.0 / sumWeightedInverseEntropy;
88  ScoreList& slotWeights = weightsList[i];
89  ScoreList::iterator itSlotWeight = slotWeights.begin();
90  for (std::vector<f64>::const_iterator itWeightedInverseEntropy = weightedInverseEntropies.begin();
91  itWeightedInverseEntropy != weightedInverseEntropies.end(); ++itWeightedInverseEntropy, ++itSlotWeight)
92  *itSlotWeight = normWeightedInverseEntropy * *itWeightedInverseEntropy;
93  }
94 }

96 void estimateJointPosteriorCn(PosteriorCn& jointCn, const ConstPosteriorCnRefList& cns, const WeightsList& weightsList) {
97  ensure(jointCn.size() == weightsList.size());
98  for (u32 i = 0; i < jointCn.size(); ++i) {
99  SlotRangeList slotRanges(cns.size());
100  Fsa::LabelId nextLabel = Core::Type<Fsa::LabelId>::max;
101  for (u32 j = 0; j < cns.size(); ++j) {
102  const PosteriorCn::Slot& pdf = (i < cns[j]->size()) ? (*cns[j])[i] : EpsSlot;
103  slotRanges[j] = std::make_pair(pdf.begin(), pdf.end());
104  verify_(slotRanges[j].first != slotRanges[j].second);
105  verify_(slotRanges[j].first->label < Core::Type<Fsa::LabelId>::max);
106  nextLabel = std::min(nextLabel, slotRanges[j].first->label);
107  }
108  const ScoreList& weights = weightsList[i];
109  PosteriorCn::Slot& jointPdf = jointCn[i];
110  ensure_(jointPdf.empty());
111  do {
112  jointPdf.push_back(PosteriorCn::Arc(nextLabel, 0.0));
113  nextLabel = Core::Type<Fsa::LabelId>::max;
114  for (u32 j = 0; j < slotRanges.size(); ++j) {
115  SlotRange& slotRange(slotRanges[j]);
116  if (slotRange.first != slotRange.second) {
117  if (slotRange.first->label == jointPdf.back().label) {
118  jointPdf.back().score += weights[j] * slotRange.first->score;
119  if (++slotRange.first != slotRange.second)
120  nextLabel = std::min(nextLabel, slotRange.first->label);
121  }
122  else {
123  verify_(jointPdf.back().label < slotRange.first->label);
124  verify(slotRange.first->label < Core::Type<Fsa::LabelId>::max);
125  nextLabel = std::min(nextLabel, slotRange.first->label);
126  }
127  }
128  }
129  } while (nextLabel != Core::Type<Fsa::LabelId>::max);
130  }
131 }

192 private:
193  ConstPosteriorCnRef getPosteriorCn() {
194  if (!hasNext_) {
195  if (ids_.size() == 1)
196  cn_ = requestPosteriorCn(ids_.front());
197  else {
198  ConstPosteriorCnRefList cns(ids_.size());
199  u32 maxT = 0;
200  for (u32 i = 0; i < ids_.size(); ++i) {
201  cns[i] = requestPosteriorCn(ids_[i]);
202  if (!cns[i]) {
203  warning("No fCN provided at port %d, discard", i);
204  cns.clear();
205  break;
206  }
207  maxT = std::max(maxT, u32(cns[i]->size()));
208  }
209  if (!cns.empty()) {
210  PosteriorCn* cnPtr = new PosteriorCn(maxT);
211  cnPtr->alphabet = cns.front()->alphabet;
212  WeightsList weightsList(maxT, weights_);
213  switch (weightingScheme_) {
214  case WeightingSchemeStatic:
215  break;
216  case WeightingSchemeMinEntropy:
217  minEntropyWeights(weightsList, cns, weights_);
218  break;
219  case WeightingSchemeInverseEntropy:
220  inverseEntropyWeights(weightsList, cns, weights_);
221  break;
222  default:
223  defect();
224  }
225  estimateJointPosteriorCn(*cnPtr, cns, weightsList);
226  cn_ = ConstPosteriorCnRef(cnPtr);
227  }
228  }
229  hasNext_ = true;
230  }
231  return cn_;
232  }

239  virtual void init(const std::vector<std::string>& arguments) {
240  u32 n = 0;
241  for (; connected(n); ++n)
242  ;
243  if (n == 0)
244  criticalError("At least one incoming fCN at port 0 required.");
245  Core::Component::Message msg = log();
246  Score sum = 0.0;
247  for (u32 i = 0; i < n; ++i) {
248  Score weight = paramWeight(select(Core::form("lattice-%d", i)));
249  if (weight != 0.0) {
250  ids_.push_back(i);
251  weights_.push_back(weight);
252  sum += weight;
253  }
254  }
255  verify(sum != 0.0);
256  for (ScoreList::iterator it = weights_.begin(); it != weights_.end(); ++it)
257  *it /= sum;
258  msg << "Combine " << ids_.size() << " fCNs.\n";
259  for (u32 i = 0; i < ids_.size(); ++i)
260  msg << "fCN at port " << ids_[i] << " has weight " << weights_[i] << ".\n";
261  Core::Choice::Value weightingScheme = paramWeightingScheme(config);
262  if (weightingScheme == Core::Choice::IllegalValue)
263  criticalError("Unknown weighting scheme");
264  msg << "Use weighting scheme: " << choiceWeightingScheme[weightingScheme] << "\n";
265  weightingScheme_ = WeightingScheme(weightingScheme);
266  hasNext_ = false;
267  }
