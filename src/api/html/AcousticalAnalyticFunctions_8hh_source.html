81 /**
82  * Bilinear Transform.
83  * x_{B} = x - T / pi * arctan(a * sin(2 * pi / T * x) / (1 + a * cos(2 * pi / T * x))),
84  * where 'a' controls curvature, and 'T' is the periodicity of the arctan term.
85  * Hint: pont T/2 is mapped on itself.
86  */
87 struct BilinearTransform : public UnaryAnalyticFunction {
88  Argument a_;
89  Argument T_;
90 
91 public:
92  BilinearTransform(Argument a, Argument T)
93  : a_(a), T_(T) {
94  require(Core::abs(a) < 1);
95  require(T > 0);
96  }
97 
98  virtual Result value(Argument x) const {
99  return x - T_ / M_PI * atan(a_ * sin(2 * M_PI / T_ * x) / (1 + a_ * cos(2 * M_PI / T_ * x)));
100  }
101  virtual UnaryAnalyticFunctionRef invert() const {
102  return UnaryAnalyticFunctionRef(new BilinearTransform(-a_, T_));
103  }
104 };
