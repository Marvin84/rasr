58  NnMatrix& errorSignalOut, const NnMatrix& activations) {

115 template<typename T>
116 void MeanAndVarianceNormalizationPreprocessingLayer<T>::_forward(const NnMatrix& input, NnMatrix& output) {
117  require(!needInit_);

189 template<typename T>
190 PolynomialPreprocessingLayer<T>::PolynomialPreprocessingLayer(const Core::Configuration& c)
191  : Core::Component(c),
192  NeuralNetworkLayer<T>(c),
193  order_(paramOrder(c)) {
194  this->log("creating polynomial feature extraction layer with order: ") << order_;
195  u32 outputDimension = Precursor::inputDimensions_[0];
196  if (order_ > 1)
197  outputDimension += Precursor::inputDimensions_[0] * (Precursor::inputDimensions_[0] + 1) / 2;
198  if (order_ > 2)
199  outputDimension += Precursor::inputDimensions_[0] * (Precursor::inputDimensions_[0] + 1) * (Precursor::inputDimensions_[0] + 2) / 6;
200  if (order_ > 3)
201  this->error("only order up to three implemented yet!");
202 
203  if (Precursor::outputDimension_ != outputDimension) {
204  this->log("resizing output of polynomial layer to ") << outputDimension;
205  Precursor::outputDimension_ = outputDimension;
206  }
207 }
