45 void BatchFeatureScorer::init(Core::Ref<const Mm::MixtureSet> mixtureSet) {
46  log("initialize nn-batch-feature-scorer with buffer size ") << bufferSize_;
47  nClasses_ = mixtureSet->nMixtures();
48 
49  // (a) load class label wrapper
50  if (labelWrapper_)
51  delete labelWrapper_;
52  labelWrapper_ = new ClassLabelWrapper(select("class-labels"), nClasses_);
53  if (!labelWrapper_->isOneToOneMapping())
54  error("no one-to-one correspondence between network outputs and classes!");
55 
56  // (b) initialize network and do some checks
57  network_.initializeNetwork(bufferSize_);
58  require_eq(network_.getTopLayer().getOutputDimension(), labelWrapper_->nClassesToAccumulate());
59  LinearAndSoftmaxLayer<f32>* topLayer = dynamic_cast<LinearAndSoftmaxLayer<f32>*>(&network_.getTopLayer());
60  if (!topLayer)
61  error("output layer must be of type 'linear+softmax'");
62 
63  // compute only the scores
64  topLayer->setEvaluateSoftmax(false);
65  // ensure that there is only a single input stream
66  if (network_.getLayer(0).nInputActivations() != 1)
67  Core::Component::criticalError("Multiple input streams not implemented in BatchFeatureScorer.");
68 
69  inputDimension_ = network_.getLayer(0).getInputDimension(0);
70 
71  // (c) remove prior from bias
72  // read
73  if (prior_.fileName() != "")
74  prior_.read();
75  else
76  prior_.setFromMixtureSet(mixtureSet, *labelWrapper_);
77 
78  // subtract
79  network_.finishComputation();
80  topLayer->removeLogPriorFromBias(prior_);
81  network_.initComputation();
82 
83  // (d) set everything in the network to computation mode
84  f32 l1norm = network_.l1norm();
85  if (std::isinf(l1norm) || Math::isnan(l1norm))
86  warning("l1 norm of all network weights is: ") << l1norm;
87  else
88  log("l1 norm of all network weights is: ") << l1norm;
89  log("batch feature scorer uses buffer size: ") << bufferSize_;
90 }

92 void BatchFeatureScorer::setFeature(u32 position, const Mm::FeatureVector& f) const {

119 Mm::FeatureScorer::Scorer BatchFeatureScorer::getScorer(const Mm::FeatureVector& f) const {
120  require(bufferFilled());

148 Mm::Score BatchFeatureScorer::getScore(Mm::EmissionIndex e, u32 position) const {
149  require_lt(position, bufferSize_);
150  // process buffer, if score for position is not up-to-date
151  if (!scoreComputed_[position]) {
152  // set everything to computing state
153  network_.initComputation();
154  buffer_.initComputation();
155  network_.forward(buffer_);
156  // scores and buffer must be readable/writable
157  network_.getTopLayerOutput().finishComputation();
158  buffer_.finishComputation(false);
159  // mark all scores in buffer as computed
160  scoreComputed_.assign(bufferSize_, true);
161  }
162  // return score
163  Mm::Score score = 0.0;
164  if (labelWrapper_->isClassToAccumulate(e)) {
165  score = -network_.getTopLayerOutput().at(labelWrapper_->getOutputIndexFromClassIndex(e), position);
166  }
167  else {
168  score = Core::Type<Mm::Score>::max;
169  }
170  return score;
171 }
