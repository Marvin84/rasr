47 void TimeframeError::add(ConstLatticeRef lHyp, ConstLatticeRef lRef) {
48  ConstBoundariesRef bHyp = lHyp->getBoundaries();
49  ConstStateRef srHyp = lHyp->getState(lHyp->initialStateId());
50  Time hypBegin, hypEnd = bHyp->time(srHyp->id());
51  Fsa::LabelId hypLabel = Fsa::Epsilon;
52 
53  ConstBoundariesRef bRef = lRef->getBoundaries();
54  ConstStateRef srRef = lRef->getState(lRef->initialStateId());
55  Time refBegin, refEnd = bRef->time(lRef->initialStateId());
56  Fsa::LabelId refLabel = Fsa::Epsilon;
57 
58  if (hypEnd != refEnd)
59  Core::Application::us()->warning(
60  "Hypothesis starts at %.3f and reference at %.3f; assume epsilon for preceeding time frames",
61  (f32(hypEnd) / 100.0), (f32(refEnd) / 100.0));
62  hypBegin = refBegin = std::min(hypEnd, refEnd);
63  for (;;) {
64  verify(refBegin <= hypBegin);
65  Time nFrames = hypEnd - hypBegin;
66  Time nErrors = 0;
67  if (nFrames > 0) {
68  while (refEnd < hypEnd) {
69  // add error
70  if (hypLabel != refLabel)
71  nErrors += refEnd - std::max(hypBegin, refBegin);
72  refBegin = refEnd;
73  if (srRef->hasArcs()) {
74  if (srRef->nArcs() > 1)
75  Core::Application::us()->criticalError("Reference lattice \"%s\" is not linear.", lRef->describe().c_str());
76  const Arc& refArc = *srRef->begin();
77  srRef = lRef->getState(refArc.target());
78  refEnd = bRef->time(srRef->id());
79  refLabel = refArc.input();
80  }
81  else if (hypEnd == Core::Type<Time>::max) {
82  hypEnd = refEnd;
83  }
84  else {
85  Core::Application::us()->warning(
86  "Reference ends at %.3f before hypothesis; assume epsilon for remaining time frames",
87  (f32(refEnd) / 100.0));
88  refEnd = Core::Type<Time>::max;
89  refLabel = Fsa::Epsilon;
90  }
91  }
92  verify((refBegin <= hypEnd) && (hypEnd <= refEnd));
93  // add error
94  if (hypLabel != refLabel)
95  nErrors += hypEnd - std::max(hypBegin, refBegin);
96  // count statistics
97  verify(nErrors <= nFrames);
98  Score error = Score(nErrors);
99  Score duration = Score(nFrames);
100  this->nWords++;
101  this->nFrames += duration;
102  this->nFrameErrors += error;
103  // see TimeframeConfusionNetwork.hh -> FramePosteriorCnFeatures
104  this->nSmoothedFrames += FramePosteriorCnFeatures::smooth(duration, duration, alpha);
105  this->nSmoothedFrameErrors += FramePosteriorCnFeatures::smooth(error, duration, alpha);
106  }
107  if ((hypEnd == refEnd) && !srHyp->hasArcs() && !srRef->hasArcs())
108  break;
109  {
110  hypBegin = hypEnd;
111  if (srHyp->hasArcs()) {
112  if (srHyp->nArcs() > 1)
113  Core::Application::us()->criticalError("Hypothesis lattice \"%s\" is not linear.", lHyp->describe().c_str());
114  const Arc& hypArc = *srHyp->begin();
115  srHyp = lHyp->getState(hypArc.target());
116  hypEnd = bHyp->time(srHyp->id());
117  hypLabel = hypArc.input();
118  }
119  else if (refEnd == Core::Type<Time>::max) {
120  refEnd = hypEnd;
121  }
122  else {
123  Core::Application::us()->warning(
124  "Hypothesis ends at %.3f before reference; assume epsilon for remaining time frames",
125  (f32(hypEnd) / 100.0));
126  hypEnd = Core::Type<Time>::max;
127  hypLabel = Fsa::Epsilon;
128  }
129  verify(hypBegin <= refEnd);
130  }
131  }
132 }

134 void TimeframeError::add(ConstLatticeRef lHyp, ConstPosteriorCnRef cnRef) {
135  FramePosteriorCnFeaturesRef features = FramePosteriorCnFeatures::create(cnRef);
136  ConstBoundariesRef bHyp = lHyp->getBoundaries();
137  Time begin, end = bHyp->time(lHyp->initialStateId());
138  for (ConstStateRef sr = lHyp->getState(lHyp->initialStateId()), nextSr; sr->hasArcs(); sr = nextSr) {
139  if (sr->nArcs() > 1)
140  Core::Application::us()->criticalError("Hypothesis lattice \"%s\" is not linear.", lHyp->describe().c_str());
141  const Arc& arc = *sr->begin();
142  nextSr = lHyp->getState(arc.target());
143  begin = end;
144  end = bHyp->time(nextSr->id());
145  if (begin < end) {
146  Time duration = end - begin;
147  features->update(arc.input(), begin, duration);
148  nWords++;
149  nFrames += features->norm(0.0);
150  nFrameErrors += features->error(0.0);
151  nSmoothedFrames += features->norm(alpha);
152  nSmoothedFrameErrors += features->error(alpha);
153  }
154  }
155 }

336 /**
337  * min. potential based max. posterior decoding
338  *
339  * Standard lattice decoder; best hypothesis present in one of the
340  * given lattices.
341  **/
342 class UnionSearchSpace : public MinimumFrameWerDecoder::SearchSpace {
343 public:
344  static const Core::ParameterFloat paramAlpha;
345  static const Core::ParameterFloat paramNonWordAlpha;
346  static const Core::ParameterString paramConfKey;
347 
348 private:
349  Score alpha_;
350  Score nonWordAlpha_;
351  std::vector<Fsa::LabelId> nonWords_;
352  Fsa::LabelId lastNonWord_;
353  Key confidenceKey_;
354 
355 protected:
356  Score alpha(Fsa::LabelId word) const {
357  if (word == Fsa::Epsilon)
358  return nonWordAlpha_;
359  if (word > lastNonWord_)
360  return alpha_;
361  std::vector<Fsa::LabelId>::const_reverse_iterator itNonWord = nonWords_.rbegin();
362  for (; *itNonWord > word; ++itNonWord)
363  ;
364  return (*itNonWord == word) ? nonWordAlpha_ : alpha_;
365  }
366 
367  /*
368  * Cost function
369  *
370  * S(w; t_start, t_end) = sum_{t \in [t_start, t_end[} (1 - p(w|t))
371  * / (1 + alpha (t_end - t_start - 1))
372  */
373  Score calcScore(const PosteriorCn& cn, Fsa::LabelId label,
374  Time start, Time end) const {
375  if (start == end)
376  return 0.0;
377  const PosteriorCn::Arc cnArc(label, 0.0);
378  Score sum = 0.0;
379  for (Time t = start; t < end; ++t)
380  sum += cn.score(t, label);
381  Score duration = end - start;
382 
383  return (duration - sum) / (1.0 + alpha(label) * (duration - 1.0));
384  }
385 
386  /*
387  * Confidence score
388  */
389  Score calcConfidence(const PosteriorCn& cn, Fsa::LabelId label, Time start, Time end) const {
390  if (start == end)
391  return 1.0;
392  const PosteriorCn::Arc cnArc(label, 0.0);
393  Probability maxP = 0.0;
394  for (Time t = start; t < end; ++t)
395  maxP = std::max(maxP, cn.score(t, label));
396  return maxP;
397  }

421  ConstTraceRef search(const ConstLatticeRefList& lats, const PosteriorCn& cn) const {
422  Trace * currentTrace = new Trace, *bestTrace = new Trace;
423  Score bestScore = Semiring::Max;
424  Fsa::StateId bestFinal = Fsa::InvalidStateId;
425  ScoresRef bestFinalWeight = lats.front()->semiring()->one();
426  u32 bestSystem = Core::Type<u32>::max;
427  for (u32 i = 0; i < lats.size(); ++i) {
428  ConstLatticeRef l = lats[i];
429  ConstStateMapRef topologicalSort = sortTopologically(l);
430  verify(topologicalSort);
431  currentTrace->clear();
432  currentTrace->grow((*topologicalSort)[0]);
433  (*currentTrace)[(*topologicalSort)[0]].score = 0.0;
434  for (u32 j = 0; j < topologicalSort->size(); ++j) {
435  Fsa::StateId sid = (*topologicalSort)[j];
436  ConstStateRef sr = l->getState(sid);
437  Time startTime = l->boundary(sid).time();
438  Score fwdScore = (*currentTrace)[sid].score;
439  if (sr->isFinal() && (fwdScore < bestScore)) {
440  bestScore = fwdScore;
441  bestFinal = sid;
442  bestFinalWeight = sr->weight();
443  bestSystem = i;
444  }
445  for (State::const_iterator a = sr->begin(); a != sr->end(); ++a) {
446  Time endTime = l->boundary(a->target()).time();
447  Score currentScore = calcScore(cn, a->input(), startTime, endTime);
448  currentTrace->grow(a->target());
449  Hyp& currentHyp = (*currentTrace)[a->target()];
450  Score score = fwdScore + currentScore;
451  if (score < currentHyp.score) {
452  currentHyp.score = score;
453  currentHyp.label = a->input();
454  currentHyp.weight = a->weight();
455  currentHyp.bptr = sid;
456  }
457  }
458  }
459  if (bestSystem == i)
460  std::swap(bestTrace, currentTrace);
461  }
462  delete currentTrace;
463  bestTrace->score = bestScore;
464  bestTrace->final = bestFinal;
465  bestTrace->system = bestSystem;
466  return ConstTraceRef(bestTrace);
467  }

469  ConstLatticeRef trace(const ConstLatticeRefList& lats, const PosteriorCn& cn, ConstTraceRef trace) const {
470  ConstSemiringRef semiring = lats[trace->system]->semiring();
471  ScoreId confidenceId = Semiring::InvalidId;
472  if (semiring.get() == confidenceSemiring().get())
473  confidenceId = ConfidenceId;
474  else if (!confidenceKey_.empty()) {
475  confidenceId = semiring->id(confidenceKey_);
476  if (confidenceId == Semiring::InvalidId)
477  warning("Semiring \"%s\" has no dimension labeled \"%s\".",
478  semiring->name().c_str(), confidenceKey_.c_str());
479  }
480  ConstBoundariesRef boundaries = lats[trace->system]->getBoundaries();
481  StaticBoundaries* b = new StaticBoundaries;
482  StaticLattice* s = new StaticLattice;
483  s->setDescription("min-fWER(union," + semiring->name() + ")");
484  s->setType(Fsa::TypeAcceptor);
485  s->setProperties(Fsa::PropertyAcyclic | Fsa::PropertyLinear, Fsa::PropertyAll);
486  s->setInputAlphabet(lats[0]->getInputAlphabet());
487  s->setSemiring(semiring);
488  s->setBoundaries(ConstBoundariesRef(b));
489  State* lastSp = s->newState();
490  lastSp->setFinal(semiring->clone(semiring->one()));
491  b->set(lastSp->id(), boundaries->get(trace->final));
492  for (Fsa::StateId sourceSid = (*trace)[trace->final].bptr, targetSid = trace->final;
493  sourceSid != Fsa::InvalidStateId; sourceSid = (*trace)[targetSid].bptr) {
494  const Hyp& target((*trace)[targetSid]);
495  State* sp = s->newState();
496  b->set(sp->id(), boundaries->get(sourceSid));
497  ScoresRef score;
498  if (confidenceId != Semiring::InvalidId) {
499  score = semiring->clone(target.weight);
500  score->set(confidenceId, calcConfidence(
501  cn, target.label, b->time(sp->id()), b->time(lastSp->id())));
502  }
503  else
504  score = target.weight;
505  sp->newArc(lastSp->id(), score, target.label);
506  lastSp = sp;
507  targetSid = sourceSid;
508  }
509  s->setInitialStateId(lastSp->id());
510  return ConstLatticeRef(s);
511  }

558  * min. potential based max. posterior decoding
559  *
560  * Lattices are transformed to the union of their time conditioned forms.
561  * A standard lattice decoding is done on the resulting mesh.
562  **/
563 class MeshSearchSpace : public UnionSearchSpace {
564  typedef UnionSearchSpace Precursor;
565 
566 public:
567  /*
568  * Attention: the usage of more boundary information than time causes problems
569  * in conjunction with null-length arcs: loops cannot be detected/avoided easily.
570  * Allowing more boundary information would require a two-step procedure:
571  * 1. build mesh, allow loops
572  * 2. detect loops and remove arcs closing a loop (which arc to remove might be arbitrary)
573  */
574  struct BoundaryTimeHash {
575  size_t operator()(const Boundary& b) const {
576  return size_t(b.time());
577  }
578  };
579  struct BoundaryTimeEqual {
580  bool operator()(const Boundary& b1, const Boundary& b2) const {
581  return b1.time() == b2.time();
582  }
583  };
584  typedef Fsa::Hash<Boundary, BoundaryTimeHash, BoundaryTimeEqual> BoundaryTimeHashList;
585 
586 protected:
587  ConstLatticeRef buildMesh(const ConstLatticeRefList& lats) {
588  ScoresRef defaultScore;
589  ConstSemiringRef semiring = lats[0]->semiring();
590  for (u32 i = 1; i < lats.size(); ++i) {
591  if ((semiring.get() != lats[i]->semiring().get()) &&
592  !(semiring == lats[i]->semiring())) {
593  semiring = confidenceSemiring();
594  defaultScore = confidenceSemiring()->one();
595  log("Building time conditioned search space: Semirings are not equal (%s != %s), use %s",
596  lats[0]->semiring()->name().c_str(), lats[i]->semiring()->name().c_str(), semiring->name().c_str());
597  break;
598  }
599  }
600  StaticBoundaries* b = new StaticBoundaries;
601  StaticLattice* s = new StaticLattice(Fsa::TypeAcceptor);
602  s->setProperties(lats[0]->knownProperties(), lats[0]->properties());
603  s->setDescription(Core::form("mesh-search-space(%s)", semiring->name().c_str()));
604  s->setInputAlphabet(lats[0]->getInputAlphabet());
605  s->setSemiring(semiring);
606  s->setBoundaries(ConstBoundariesRef(b));
607  Core::Vector<Fsa::StateId> initialSids;
608  BoundaryTimeHashList meshSids;
609  for (u32 i = 0; i < lats.size(); ++i) {
610  ConstLatticeRef l = lats[i];
611  ConstStateMapRef topologicalSort = sortTopologically(l);
612  verify(topologicalSort);
613  const Boundary& initialB = l->boundary(topologicalSort->front());
614  std::pair<Fsa::StateId, bool> meshInitial = meshSids.insertExisting(initialB);
615  if (!meshInitial.second) {
616  State* meshInitialSp = new State(meshInitial.first);
617  s->setState(meshInitialSp);
618  initialSids.push_back(meshInitialSp->id());
619  b->set(meshInitialSp->id(), Boundary(initialB.time()));
620  }
621  for (u32 j = 0; j < topologicalSort->size(); ++j) {
622  Fsa::StateId sid = (*topologicalSort)[j];
623  ConstStateRef sr = l->getState(sid);
624  const Boundary& meshB = l->boundary(sid);
625  Fsa::StateId meshSid = meshSids.find(meshB);
626  verify(meshSid != BoundaryTimeHashList::InvalidCursor);
627  State* meshSp = s->fastState(meshSid);
628  if (sr->isFinal()) {
629  if (defaultScore)
630  meshSp->setFinal(defaultScore);
631  else {
632  if (meshSp->isFinal()) {
633  if (semiring->compare(meshSp->weight(), sr->weight()) != 0)
634  warning("Building time conditioned search space: Final scores at time %d are not equal", meshB.time());
635  }
636  else
637  meshSp->setFinal(sr->weight());
638  }
639  }
640  for (State::const_iterator a = sr->begin(); a != sr->end(); ++a) {
641  const Boundary& targetB = l->boundary(a->target());
642  if (meshB.time() < targetB.time()) {
643  std::pair<Fsa::StateId, bool> meshTarget = meshSids.insertExisting(targetB);
644  if (!meshTarget.second) {
645  State* meshTargetSp = new State(meshTarget.first);
646  s->setState(meshTargetSp);
647  b->set(meshTargetSp->id(), Boundary(targetB.time()));
648  }
649  meshSp->newArc(meshTarget.first, (defaultScore ? defaultScore : a->weight()), a->input());
650  }
651  else
652  verify(meshB.time() == targetB.time());
653  }
654  }
655  }
656  if (initialSids.size() == 1)
657  s->setInitialStateId(initialSids.front());
658  else {
659  State* meshInitialSp;
660  Boundary initialB(0, Boundary::Transit(Bliss::Phoneme::term, Bliss::Phoneme::term));
661  std::pair<Fsa::StateId, bool> meshInitial = meshSids.insertExisting(initialB);
662  if (!meshInitial.second) {
663  meshInitialSp = new State(meshInitial.first);
664  s->setState(meshInitialSp);
665  b->set(meshInitialSp->id(), Boundary(initialB.time()));
666  }
667  else
668  meshInitialSp = s->fastState(meshInitial.first);
669  for (Core::Vector<Fsa::StateId>::const_iterator itSid = initialSids.begin();
670  itSid != initialSids.end(); ++itSid)
671  if (*itSid != meshInitialSp->id())
672  meshInitialSp->newArc(*itSid, s->semiring()->one(), Fsa::Epsilon, Fsa::Epsilon);
673  s->setInitialStateId(meshInitialSp->id());
674  }
675  return ConstLatticeRef(s);
676  }
677 
678 public:
679  MeshSearchSpace(const Core::Configuration& config)
680  : Precursor(config) {
681  }
682  virtual ~MeshSearchSpace() {}
683 
684  virtual ConstLatticeRef best(const ConstLatticeRefList& lats, ConstPosteriorCnRef cn) {
685  if (lats.empty())
686  return ConstLatticeRef();
687  ConstLatticeRefList meshedLats(1, buildMesh(lats));
688  ConstTraceRef best = Precursor::search(meshedLats, *cn);
689  log("Min. score is %f.", best->score);
690  ConstLatticeRef result = Precursor::trace(meshedLats, *cn, best);
691  dynamic_cast<StaticLattice*>(const_cast<Lattice*>(result.get()))
692  ->setDescription(Core::form("min-fWER(mesh,%s)", result->semiring()->name().c_str()));
693  return result;
694  }

699 class FrameCnSearchSpace : public MinimumFrameWerDecoder::SearchSpace {
700 public:
701  static const Core::ParameterFloat paramWordPenalty;
702 
703 private:
704  Score wordPenalty_;
705 
706 protected:
707  struct Hyp {
708  Score score;
709  Fsa::LabelId label;
710  u32 bptr;
711  };
712  typedef Core::Vector<Core::Vector<Hyp>> Trace;
713 
714 public:
715  virtual ConstLatticeRef best(const ConstLatticeRefList& lats, ConstPosteriorCnRef cnRef) {
716  const PosteriorCn& cn(*cnRef);
717  Trace trace(cn.size());
718  trace[0].resize(cn.front().size());
719  for (u32 j = 0; j < trace.front().size(); ++j) {
720  Hyp& hyp(trace.front()[j]);
721  hyp.score = -::log(cn.front()[j].score);
722  hyp.label = cn.front()[j].label;
723  hyp.bptr = Core::Type<u32>::max;
724  }
725  for (u32 t = 1; t < cn.size(); ++t) {
726  const PosteriorCn::Slot& prePdf(cn[t - 1]);
727  const Core::Vector<Hyp>& preHyps(trace[t - 1]);
728  const PosteriorCn::Slot& pdf(cn[t]);
729  Core::Vector<Hyp>& hyps(trace[t]);
730  hyps.resize(pdf.size());
731  for (u32 j = 0; j < hyps.size(); ++j) {
732  const PosteriorCn::Arc& cnArc(pdf[j]);
733  Hyp& hyp(hyps[j]);
734  hyp.score = Semiring::Max;
735  for (u32 k = 0; k < preHyps.size(); ++k) {
736  Score score = preHyps[k].score - Score(::log(cnArc.score));
737  if (prePdf[k].label != cnArc.label)
738  score += wordPenalty_;
739  if (score < hyp.score) {
740  hyp.score = score;
741  hyp.label = cnArc.label;
742  hyp.bptr = k;
743  }
744  }
745  }
746  }
747  Hyp bestHyp;
748  bestHyp.score = Semiring::Max;
749  bestHyp.label = Fsa::InvalidLabelId;
750  bestHyp.bptr = Core::Type<u32>::max;
751  for (u32 j = 0; j < trace.back().size(); ++j) {
752  const Hyp& hyp(trace.back()[j]);
753  if (hyp.score < bestHyp.score) {
754  bestHyp.score = hyp.score;
755  bestHyp.bptr = j;
756  }
757  }
758  verify(bestHyp.score != Semiring::Max);
759 
760  log("Min. score is %f.", bestHyp.score);
761 
762  ConstSemiringRef semiring = confidenceSemiring();
763  StaticBoundaries* b = new StaticBoundaries;
764  StaticLattice* s = new StaticLattice;
765  s->setType(Fsa::TypeAcceptor);
766  s->setProperties(Fsa::PropertyAcyclic | Fsa::PropertyLinear, Fsa::PropertyAll);
767  s->setInputAlphabet(lats.front()->getInputAlphabet());
768  s->setSemiring(semiring);
769  s->setDescription(Core::form("min-fWER(fCN,%s)", s->semiring()->name().c_str()));
770  s->setBoundaries(ConstBoundariesRef(b));
771  State* lastSp = s->newState();
772  b->set(lastSp->id(), Boundary(trace.size()));
773  lastSp->setFinal(semiring->clone(semiring->one()));
774  u32 k = bestHyp.bptr;
775  Hyp lastHyp(trace.back()[k]);
776  Probability confScore = Semiring::Max;
777  for (s32 t = trace.size() - 1; 0 <= t; --t) {
778  const Hyp& hyp(trace[t][k]);
779  if (hyp.label != lastHyp.label) {
780  ScoresRef score = semiring->create();
781  score->set(ConfidenceId, ::exp(-confScore));
782  confScore = Semiring::Max;
783  State* sp = s->newState();
784  b->set(sp->id(), Boundary(t + 1));
785  sp->newArc(lastSp->id(), score, lastHyp.label);
786  lastHyp = hyp;
787  lastSp = sp;
788  }
789  confScore = std::min(confScore, -::log(cn[t][k].score));
790  k = hyp.bptr;
791  }
792  ScoresRef score = semiring->create();
793  score->set(ConfidenceId, ::exp(-confScore));
794  State* sp = s->newState();
795  b->set(sp->id(), Boundary(0));
796  sp->newArc(lastSp->id(), score, lastHyp.label);
797  s->setInitialStateId(sp->id());
798  return ConstLatticeRef(s);
799  }

804  wordPenalty_ = paramWordPenalty(config);
805  log("word-penalty=%f", wordPenalty_);
806  }
807  virtual ~FrameCnSearchSpace() {}
808 };
