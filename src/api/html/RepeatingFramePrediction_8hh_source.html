24 /** RepeatingFramePrediction: predicts input at given target start-times by
25  * copying the previous element.
26  */
27 class RepeatingFramePrediction {
28 public:
29  typedef Flow::Time Time;
30  typedef Flow::Timestamp Data;
31  typedef Flow::DataPtr<Data> DataPointer;
32 
33 private:
34  /** Contains the two latest element of the input stream. */
35  SlidingWindow<DataPointer> slidingWindow_;
36 
37  /** If true, prediction is only made for target times not found in the input stream.
38  * If false, prediction is made for each target time.
39  */
40  bool predictOnlyMissing_;
41 
42  /** If false (default), the output stream has the same number of frames as the target streams,
43  * but only start times are synchronized.
44  * If true, start- and end-times are synchronized, so the frames of target and output-stream have the same length
45  */
46  bool syncEndTimes_;
47 
48 private:
49  /** Seeks in the input stream until @param time is found.
50  *
51  * At the beginning of the input stream, first element is always added to the slidingWindow_.
52  * After end-of-stream the last elements in the slidingWindow_ are kept until reset() is called.
53  * After seeking start-time of the front element in the slidingWindow_
54  * is greater or equal to @param time and back one is smaller then @param time
55  * (except for the very first element, which can be larger or equal to @param time).
56  */
57  void seek(Time time);
58 
59  /** Copies the front element in the slidingWindow_ to @param out if
60  * -if predictOnlyMissing_ is false and
61  * start-time of front element is equal to @param time.
62  *
63  * @return is false if the slidingWindow_ is empty.
64  */
65  bool copyLatest(Time time, DataPointer& out, Time endTime);
66 
67  /** Factual predction: element at @param time is predicted by
68  * the previous element of the input stream.
69  *
70  * Caution: start and end-time of the predicted element is set to @param time
71  *
72  * @return is false if the slidingWindow_ is empty.
73  */
74  bool copyPrevious(Time time, DataPointer& out, Time endTime);
75 
76 protected:
77  /** override nextData to supply input data on demand */
78  virtual bool nextData(DataPointer& dataPointer) = 0;
79 
80 public:
81  static std::string name() {
82  return "signal-repeating-frame-prediction";
83  }
84  RepeatingFramePrediction();
85  virtual ~RepeatingFramePrediction() {}
86 
87  /** @return is the vector created by prediction at @param time
88  *
89  * If @param time is found in the input stream and predictOnlyMissing_ is false:
90  * start-time and end-time are delivered un-changed
91  * Else:
92  * start-time and end-time of a predicted output are both set to @param time.
93  *
94  * If @return false call lastError() to get a explanation.
95  */
96  bool work(const Flow::Timestamp& time, DataPointer& out);
97 
98  /** @see predictOnlyMissing_ */
99  void setPredictOnlyMissing(bool predictOnlyMissing) {

112  slidingWindow_.clear();
113  }
