83 void CheckPosteriorDfsState::discoverState(Fsa::ConstStateRef sp) {
84  const Speech::TimeframeIndex tbeg = wordBoundaries_->time(sp->id());
85  for (Fsa::State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
86  const Speech::TimeframeIndex tend = wordBoundaries_->time(fsa_->getState(a->target())->id());
87  for (Speech::TimeframeIndex t = tbeg; t < tend; ++t) {
88  if (!accumulators_[t]) {
89  accumulators_[t] = fsa_->semiring()->getCollector(a->weight());
90  }
91  else {
92  accumulators_[t]->feed(a->weight());
93  }
94  }
95  }
96 }

98 void CheckPosteriorDfsState::finish() {
99  f32 maxDeviation = 0;
100  for (Speech::TimeframeIndex t = 0; t < accumulators_.size() - 1; ++t) {
101  const f32 sum = exp(-f32(accumulators_[t]->get()));
102  const f32 dev = Core::abs(sum - 1);
103  maxDeviation = maxDeviation < dev ? dev : maxDeviation;
104  }
105  verify(Core::isAlmostEqualUlp(maxDeviation, 0, tolerance_));
106 }

114 ConstWordLatticeRef prune(ConstWordLatticeRef lattice, const Fsa::Weight& threshold, bool relative, bool backward, bool hasFailArcs) {
115  bool hasInitialState = (lattice->mainPart()->initialStateId() != Fsa::InvalidStateId);
116  if (f32(threshold) < Core::Type<f32>::max) {
117  WordLattice* result = new WordLattice;
118  result->setWordBoundaries(lattice->wordBoundaries());
119  if (backward) {
120  if (hasFailArcs) {
121  result->setFsa(Fsa::posteriorPrune4SpecialSymbols(lattice->mainPart(), threshold, relative), lattice->mainName());
122  }
123  else
124  result->setFsa(Fsa::prunePosterior(lattice->mainPart(), threshold, relative), lattice->mainName());
125  }
126  else
127  result->setFsa(Fsa::pruneSync(lattice->mainPart(), threshold), lattice->mainName());
128  if (result->mainPart()->initialStateId() == Fsa::InvalidStateId and hasInitialState)
129  Core::Application::us()->warning("During pruning the automaton became empty. result->mainPart()->initialStateId() == Fsa::InvalidStateId");
130  return ConstWordLatticeRef(result);
131  }
132  return lattice;
133 }
