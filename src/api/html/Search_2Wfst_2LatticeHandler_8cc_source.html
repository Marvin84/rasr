44 LatticeHandler::ConstWordLatticeRef LatticeHandler::convert(const WfstLatticeAdaptor& l) const {
45  require(lexicon());
46  if (l.empty())
47  return ::Lattice::ConstWordLatticeRef();
48  ::Lattice::WordLattice* lattice = new ::Lattice::WordLattice();
49  LmScoreLattice lmScores(*l.get(), LatticeLmScoreMapper());
50  AmScoreLattice amScores(*l.get(), LatticeAmScoreMapper());
51  Core::Ref<Fsa::StaticAutomaton> lm = OpenFst::convertToFsa(lmScores);
52  Core::Ref<Fsa::StaticAutomaton> am = OpenFst::convertToFsa(amScores);
53  Fsa::ConstAlphabetRef alphabet;
54  switch (l.get()->outputType()) {
55  case OutputLemma:
56  alphabet = lexicon()->lemmaAlphabet();
57  break;
58  case OutputLemmaPronunciation:
59  alphabet = lexicon()->lemmaPronunciationAlphabet();
60  break;
61  case OutputSyntacticToken:
62  alphabet = lexicon()->syntacticTokenAlphabet();
63  break;
64  default:
65  defect();
66  break;
67  }
68  if (lm->type() == Fsa::TypeAcceptor)
69  lm->setInputAlphabet(alphabet);
70  else
71  lm->setOutputAlphabet(alphabet);
72  lattice->setFsa(lm, ::Lattice::WordLattice::lmFsa);
73  lattice->setFsa(am, ::Lattice::WordLattice::acousticFsa);
74  const Lattice::WordBoundaries& wb = l.get()->wordBoundaries();
75  ::Lattice::WordBoundaries* boundaries = new ::Lattice::WordBoundaries;
76  for (int s = 0; s < wb.size(); ++s)
77  boundaries->set(s, ::Lattice::WordBoundary(wb[s]));
78  lattice->setWordBoundaries(Core::ref(boundaries));
79  return ::Lattice::ConstWordLatticeRef(lattice);
80 }
