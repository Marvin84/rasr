168 template<class _Automaton>
169 typename _Automaton::ConstRef sort(typename _Automaton::ConstRef f, Fsa::SortType type) {
170  switch (type) {
171  case Fsa::SortTypeNone:
172  return f;
173  case Fsa::SortTypeByArc:
174  if (f->hasProperty(Fsa::PropertySortedByArc))
175  return f;
176  else
177  return typename _Automaton::ConstRef(new SortByArcAutomaton<_Automaton>(f));
178  case Fsa::SortTypeByInput:
179  if (f->hasProperty(Fsa::PropertySortedByArc) ||
180  f->hasProperty(Fsa::PropertySortedByInput) ||
181  f->hasProperty(Fsa::PropertySortedByInputAndTarget) ||
182  f->hasProperty(Fsa::PropertySortedByInputAndOutput) ||
183  f->hasProperty(Fsa::PropertySortedByInputAndOutputAndTarget))
184  return f;
185  else
186  return typename _Automaton::ConstRef(new SortByInputAutomaton<_Automaton>(f));
187  case Fsa::SortTypeByInputAndOutput:
188  if (f->hasProperty(Fsa::PropertySortedByArc) ||
189  f->hasProperty(Fsa::PropertySortedByInputAndOutput) ||
190  f->hasProperty(Fsa::PropertySortedByInputAndOutputAndTarget))
191  return f;
192  else
193  return typename _Automaton::ConstRef(new SortByInputAndOutputAutomaton<_Automaton>(f));
194  case Fsa::SortTypeByInputAndTarget:
195  if (f->hasProperty(Fsa::PropertySortedByInputAndTarget))
196  return f;
197  else
198  return typename _Automaton::ConstRef(new SortByInputAndTargetAutomaton<_Automaton>(f));
199  case Fsa::SortTypeByInputAndOutputAndTarget:
200  if (f->hasProperty(Fsa::PropertySortedByInputAndOutputAndTarget))
201  return f;
202  else
203  return typename _Automaton::ConstRef(new SortByInputAndOutputAndTargetAutomaton<_Automaton>(f));
204  case Fsa::SortTypeByOutput:
205  if (f->hasProperty(Fsa::PropertySortedByOutput))
206  return f;
207  else
208  return typename _Automaton::ConstRef(new SortByOutputAutomaton<_Automaton>(f));
209  case Fsa::SortTypeByWeight:
210  if (f->hasProperty(Fsa::PropertySortedByWeight))
211  return f;
212  else
213  return typename _Automaton::ConstRef(new SortByWeightAutomaton<_Automaton>(f));
214  default:
215  defect();
216  }
217  return f;
218 }

222  * Helper class for the topologicallySort function.
223  * time_ always represents the finishing time of the next state to finish
224  * isCyclic_ is set to true if the automaton contains cycles (even trivial ones)

253 template<class _Automaton>
254 Fsa::StateMap topologicallySort(typename _Automaton::ConstRef f, bool progress) {
255  Core::ProgressIndicator* p = 0;
256  if (progress)
257  p = new Core::ProgressIndicator("counting", "states");
258  TopologicallySortDfsState<_Automaton> v(f);
259  v.dfs(p);
260  if (p)
261  delete p;
262  --v.time_;
263  if (v.isCyclic_)
264  v.map_.resize(0); // return empty map in case of cyclic automaton
265  // states in decreasing finishing times of dfs define topological order
266  for (Fsa::StateId i = 0; i < v.map_.size(); ++i)
267  if (v.map_[i] != Fsa::InvalidStateId)
268  v.map_[i] = v.time_ - v.map_[i];
269  return v.map_;
270 }
