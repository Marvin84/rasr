61 MixtureSet* MixtureSet::createOneMixtureCopy() const {
62  Mixture* oneMixture = new Mixture(*this);
63  MixtureSet* oneMixtureSet = new MixtureSet(dimension_);
64  oneMixtureSet->addMixture(oneMixture);
65 
66  // Deep copy the mean table
67  for (MeanIndex i = 0; i < nMeans(); ++i) {
68  Mean* newMean = new Mean(*(mean(i)));
69  oneMixtureSet->addMean(i, newMean);
70  }
71 
72  // Deep copy the covariance table
73  for (CovarianceIndex i = 0; i < nCovariances(); ++i) {
74  Covariance* newCovariance = covariance(i)->clone();
75  oneMixtureSet->addCovariance(i, newCovariance);
76  }
77 
78  // Deep copy the density table
79  for (DensityIndex i = 0; i < nDensities(); ++i) {
80  GaussDensity* newDensity = new GaussDensity(*(density(i)));
81  oneMixtureSet->addDensity(i, newDensity);
82  }
83 
84  return oneMixtureSet;
85 }

142 bool MixtureSet::write(std::ostream& o) const {

157  << nCovariances() << std::endl;

159  o << *(mixture(i));

161  o << *(density(i));

163  o << *(mean(i));
164  for (CovarianceIndex i = 0; i < nCovariances(); ++i) {
165  o << " " << (*(covariance(i)));

168  return o.good();

170 bool MixtureSet::read(std::istream& i) {
171  clear();
172  ComponentIndex dim;
173  MixtureIndex nMix;
174  DensityIndex nDns;
175  MeanIndex nMean;
176  CovarianceIndex nCov;
177  // Header
178  std::string line;
179  getline(i, line); // version
180  f32 version = (f32)atof(line.substr(10).c_str());
181  if (version > 2.0) {
182  Core::Application::us()->criticalError("version \"") << version << "\" not supported";
183  }
184  getline(i, line); // CovType
185  std::string covtype = line.substr(17);
186  if (covtype.compare("DiagonalCovariance") != 0) {
187  Core::Application::us()->error("No correct Covariance Type set: %s", covtype.c_str());
188  }
189  i >> dim >> nMix >> nDns >> nMean >> nCov;
190  setDimension(dim);
191  // Data
192  while (0 < nMix--) {
193  Mixture* mix = new Mixture();
194  mix->read(i, version);
195  addMixture(mix);
196  }
197  while (0 < nDns--) {
198  GaussDensity* dns = new GaussDensity();
199  i >> *dns;
200  addDensity(dns);
201  }
202  while (0 < nMean--) {
203  Mean* mean = new Mean();
204  i >> *mean;
205  addMean(mean);
206  }
207  while (0 < nCov--) {
208  Covariance* cov = new DiagonalCovariance();
209  i >> *cov;
210  addCovariance(cov);
211  }
212 
213  return i.good();
214 }
