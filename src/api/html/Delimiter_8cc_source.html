41 f64 Delimiter::logLikelihood(u32 ib, u32 ie) const {
42  const f64 e = 0.5;
43  f64 yy = Core::Type<f64>::max;
44  f64 s1 = ib - 1 + nFeatures() - ie;
45  f64 m1 = M_[ib - 1] + M_[nFeatures()] - M_[ie];
46  f64 qq = Q_[ib - 1] + Q_[nFeatures()] - Q_[ie];
47  f64 xx = qq / s1 - ((m1 / s1) * (m1 / s1)) / (f_ > 0 ? f_ : 1);
48  f64 xsil = s1 * ::log(std::max(xx, e));
49  f64 s2 = ie - ib + 1;
50  f64 m2 = M_[ie] - M_[ib - 1];
51  qq = Q_[ie] - Q_[ib - 1];
52  xx = qq / s2 - ((m2 / s2) * (m2 / s2)) / (f_ > 0 ? f_ : 1);
53  f64 xspe = s2 * ::log(std::max(xx, e));
54  if (((m1 / s1) < (m2 / s2)) &&
55  ((ie - ib) >= (minimumSpeechProportion_ * nFeatures())))
56  yy = xsil + xspe;
57  return yy;
58 }

61 /*
62  * taken from old standard system (aka delimit(*)):
63  *
64  * START/STOP-DETECTION USING TWO GAUSSIAN MODELS,
65  * ONE FOR SPEECH AND ONE FOR SILENCE;
66  * BRIDLE,SEDGWICK, PP.656-659, ICASSP'77.
67  *
68  * ITERATIVE APPROXIMATION BY VARYING IB AND IE SEPARATELY
69  *
70  * N
71  * F = N*log(var) = N * log( 1/N * SUM [x(n)-xx]**2 )
72  * n=1
73  *
74  * SUM [x(n)-xx]**2 = SUM [x(n)**2 - 2*xx*x(n) + xx**2]
75  * = SUM x(n)**2 - 2*xx*SUM x(n) + SUM xx**2
76  * = SUM x(n)**2 - 2*xx*N*xx + N *xx**2
77  * = SUM x(n)**2 - N*(xx**2)
78  *
79  * silence speech silence
80  * [1,...,IB-1] [IB,...,IE] [IE+1,...,NI]
81  */
82 /*****************************************************************************/
83 Delimiter::Delimitation Delimiter::getDelimitation() const {
84  if (minimumSpeechProportion_ == 1.0)
85  return std::make_pair(0, nFeatures() - 1);
86 
87  f64 x;
88  f64 x_min = Core::Type<f64>::max;
89  u32 ie, ib, iEnd = nFeatures() - 1, iBeg = 1;
90  for (u32 i = 0; i < numberOfIterations_; ++i) {
91  for (ib = 2; ib < iEnd - 1; ++ib) {
92  x = logLikelihood(ib, iEnd);
93  if (x < x_min) {
94  x_min = x;
95  iBeg = ib;
96  }
97  }
98  for (ie = iBeg + 1; ie < nFeatures() - 1; ie++) {
99  x = logLikelihood(iBeg, ie);
100  if (x < x_min) {
101  x_min = x;
102  iEnd = ie;
103  }
104  }
105  }
106 
107  if (statisticsChannel_.isOpen()) {
108  statisticsChannel_ << Core::XmlOpen("delimiter-statistics")
109  << Core::XmlFull("frames", nFeatures())
110  << Core::XmlFull("speech-begin", iBeg)
111  << Core::XmlFull("speech-end", iEnd)
112  << Core::XmlFull("score", x_min)
113  << Core::XmlClose("delimiter-statistics");
114  }
115 
116  return std::make_pair(iBeg - 1, iEnd - 1);
117 }
