50 void MixtureSetSplitter::splitMeans(MixtureSet& mixtureSet,
51  const std::vector<Weight>& weights) {
52  splittedMeans_.resize(mixtureSet.nMeans());
53  splittedDensities_.resize(mixtureSet.nDensities());
54  for (DensityIndex densityIndex = 0; densityIndex < splittedDensities_.size(); ++densityIndex) {
55  const GaussDensity* densityToSplit = mixtureSet.density(densityIndex);
56  MeanIndex meanIndex = densityToSplit->meanIndex();
57  CovarianceIndex covarianceIndex = densityToSplit->covarianceIndex();
58 
59  std::vector<MeanType> perturbation = mixtureSet.covariance(covarianceIndex)->diagonal();
60  for (std::vector<MeanType>::iterator elem = perturbation.begin(); elem != perturbation.end(); ++elem) {
61  *elem = std::sqrt(*elem) * perturbationWeight_ * Core::Type<f32>::epsilon;
62  }
63 
64  splittedMeans_[meanIndex] = (weights[meanIndex] > minMeanObservationWeight_ ? splitMean(mixtureSet, meanIndex, perturbation) : meanIndex);
65  }
66 }

68 MeanIndex MixtureSetSplitter::splitMean(MixtureSet& mixtureSet, MeanIndex index, std::vector<MeanType>& perturbation) {
69  Mean* toSplit = mixtureSet.mean(index);
70  Mean* newMean = new Mean(toSplit->size());
71 
72  std::transform(toSplit->begin(), toSplit->end(), perturbation.begin(),
73  newMean->begin(), std::minus<MeanType>());
74  std::transform(toSplit->begin(), toSplit->end(), perturbation.begin(),
75  toSplit->begin(), std::plus<MeanType>());
76  return mixtureSet.addMean(newMean);
77 }

91 void MixtureSetSplitter::splitDensities(MixtureSet& mixtureSet) {
92  splittedDensities_.resize(mixtureSet.nDensities());
93  for (DensityIndex i = 0; i < splittedDensities_.size(); ++i) {
94  const GaussDensity* toSplit = mixtureSet.density(i);
95  MeanIndex splittedMeanIndex = splittedMeans_[toSplit->meanIndex()];
96  CovarianceIndex splittedCovarianceIndex = splittedCovariances_[toSplit->covarianceIndex()];
97 
98  if (splittedMeanIndex != toSplit->meanIndex() ||
99  splittedCovarianceIndex != toSplit->covarianceIndex()) {
100  splittedDensities_[i] = mixtureSet.addDensity(
101  new GaussDensity(splittedMeanIndex, splittedCovarianceIndex));
102  }
103  else
104  splittedDensities_[i] = i;
105  }
106 }

108 void MixtureSetSplitter::splitDensitiesInMixtures(MixtureSet& mixtureSet) {
109  for (MixtureIndex i = 0; i < mixtureSet.nMixtures(); ++i) {
110  Mixture* mixture = mixtureSet.mixture(i);
111  size_t nDensitiesToSplit = mixture->nDensities();
112 
113  for (size_t dns = 0; dns < nDensitiesToSplit; ++dns) {
114  DensityIndex splittedDensityIndex = splittedDensities_[mixture->densityIndex(dns)];
115  if (splittedDensityIndex != mixture->densityIndex(dns)) {
116  mixture->addLogDensity(splittedDensityIndex, mixture->logWeight(dns));
117  }
118  }
119  if (normalizeMixtureWeights_) {
120  mixture->normalizeWeights();
121  }
122  }
