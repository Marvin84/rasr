50 bool WindowBuffer::put(const Flow::Vector<Sample>& in) {
51  if (needInit_) {

56  if (buffer_.empty()) {

99 bool WindowBuffer::flush(Flow::Vector<f32>& out) {
100  if (needInit_) {
101  init();
102  verify(!needInit_);

104 
105  verify(buffer_.size() < 2 * std::max(shift_, length_));
106 
107  if (buffer_.empty())

111  flushed_ = (shift_ >= buffer_.size());
112  else
113  flushed_ = (std::max(shift_, length_) >= buffer_.size());
114 
115  copy(out, std::min(length_, u32(buffer_.size())));
116 
117  if (flushed())
118  needInit_ = true;
119  else {
120  buffer_.erase(buffer_.begin(), buffer_.begin() + shift_);
121 
122  bufferStartTime_ += (Time)shift_ / (Time)sampleRate_;
123  }
124
