29 Fsa::LabelId TokenAlphabet::index(const std::string& sym) const {
30  Fsa::LabelId special = specialIndex(sym);
31  if (special != Fsa::InvalidLabelId)
32  return special;
33  if (sym.size() && sym[0] == '#') {
34  int di = atoi(sym.c_str() + 1);
35  if (di < 1)
36  return Fsa::InvalidLabelId;
37  return disambiguator(di - 1);
38  }
39  const Token* token = tokens_[sym];
40  if (!token)
41  return Fsa::InvalidLabelId;
42  return token->id();
43 }

143 Fsa::LabelId LemmaPronunciationAlphabet::index(const std::string& id) const {
144  Fsa::LabelId special = specialIndex(id);
145  if (special != Fsa::InvalidLabelId)
146  return special;
147 
148  std::string::size_type i = id.find(" /");
149  std::string::size_type j = i + 4;
150  if (i == std::string::npos) { // for backward compatibility
151  i = id.find("/");
152  j = i + 1;
153  }
154 
155  if (i == std::string::npos)
156  return Fsa::InvalidLabelId;
157  std::string lemmaName = id.substr(0, i);
158 
159  std::string::size_type k = id.rfind("/");
160  if (k == std::string::npos)
161  return Fsa::InvalidLabelId;
162  std::string phon = id.substr(j, k - j);
163 
164  const Lemma* lemma = lexicon_->lemma(lemmaName);
165  if (!lemma)
166  return Fsa::InvalidLabelId;
167  Lemma::PronunciationIterator lp, lp_end;
168  for (Core::tie(lp, lp_end) = lemma->pronunciations(); lp != lp_end; ++lp) {
169  if (phon == lp->pronunciation()->format(lexicon_->phonemeInventory()))
170  return ((const LemmaPronunciation*)lp)->id();
171  }
172  return Fsa::InvalidLabelId;
173 }

175 std::string LemmaPronunciationAlphabet::symbol(Fsa::LabelId id) const {
176  std::string special = specialSymbol(id);
177  if (!special.empty())
178  return special;
179  if (id >= Fsa::LabelId(lexicon_->nLemmaPronunciations()))
180  return "unknown lemma pronunciation";

182  verify(lp);
