35 /**
36  * Iterate through posterior cn,
37  * sort slot w.r.t. probability.
38  **/
39 class PosteriorCnSortingIterator {
40 private:
41  struct ProbabilityWeakOrder {
42  bool operator()(const PosteriorCn::Arc& a1, const PosteriorCn::Arc& a2) const {
43  return a1.score > a2.score;
44  }
45  };
46 
47 private:
48  ConstPosteriorCnRef cn_;
49  PosteriorCn::Slot slot_;
50  PosteriorCn::const_iterator it_, end_;
51 
52 private:
53  PosteriorCnSortingIterator(ConstPosteriorCnRef cn)
54  : cn_(cn), slot_(0) {
55  verify(cn);
56  }
57 
58  void setSlot() {
59  if (it_ != end_) {
60  slot_ = *it_;
61  std::sort(slot_.begin(), slot_.end(), ProbabilityWeakOrder());
62  }
63  else
64  slot_.clear();
65  }
66 
67 public:
68  const PosteriorCn::Slot& operator*() const {
69  return slot_;
70  }
71  bool operator!=(const PosteriorCnSortingIterator& it) const {
72  return it_ != it.it_;
73  }
74  void operator++() {
75  verify(it_ != end_);
76  ++it_;
77  setSlot();
78  }
79 
80 public:
81  static PosteriorCnSortingIterator begin(ConstPosteriorCnRef cn) {
82  PosteriorCnSortingIterator cnIt(cn);
83  cnIt.it_ = cn->begin();
84  cnIt.end_ = cn->end();
85  cnIt.setSlot();
86  return cnIt;
87  }
88 
89  static PosteriorCnSortingIterator end(ConstPosteriorCnRef cn) {
90  PosteriorCnSortingIterator cnIt(cn);
91  cnIt.it_ = cnIt.end_ = cn->end();
92  return cnIt;
93  }
94 };

111  u32 i = 0;
112  for (PosteriorCnSortingIterator itSlot = PosteriorCnSortingIterator::begin(cn),
113  endSlot = PosteriorCnSortingIterator::end(cn);
114  itSlot != endSlot; ++itSlot, ++i) {
115  os << "# slot " << (i + 1) << std::endl;
116  const PosteriorCn::Slot& slot = *itSlot;
117  for (PosteriorCn::Slot::const_iterator itArc = slot.begin(), endArc = slot.end(); itArc != endArc; ++itArc)
118  os << ((itArc->label == Fsa::Epsilon) ? "@" : alphabet->symbol(itArc->label)) << "\t"
119  << itArc->score << std::endl;
120  }
121 }

305 void writePosteriorCnAsXml(Core::XmlWriter& xml, ConstPosteriorCnRef cn, ConstSegmentRef segment, u32 ioFlag) {
306  if (!cn)
307  return;
308  {
309  Core::XmlOpen xmlCnOpen("posterior-cn");
310  xmlCnOpen + Core::XmlAttribute("version", "1.1");
311  if (segment) {
312  if (segment->hasSegmentId())
313  xmlCnOpen + Core::XmlAttribute("name", segment->segmentId());
314  if (segment->hasRecordingId())
315  xmlCnOpen + Core::XmlAttribute("recording", segment->recordingId());
316  if (segment->hasStartTime() && segment->hasEndTime())
317  xmlCnOpen + Core::XmlAttribute("start", Core::form("%.3f", segment->startTime())) + Core::XmlAttribute("end", Core::form("%.3f", segment->endTime()));
318  }
319  xml << xmlCnOpen;
320  }
321  if (ioFlag & IoFlag::WriteHead) {
322  xml << Core::XmlOpen("head");
323  xml << Core::XmlEmpty("alphabet") + Core::XmlAttribute("name", Lexicon::us()->alphabetName(Lexicon::us()->alphabetId(cn->alphabet)).c_str());
324  xml << Core::XmlClose("head");
325  }
326  if (ioFlag & IoFlag::WriteBody) {
327  Fsa::ConstAlphabetRef alphabet = cn->alphabet;
328  xml << (Core::XmlOpen("body") + Core::XmlAttribute("n", cn->size()));
329  for (PosteriorCnSortingIterator itSlot = PosteriorCnSortingIterator::begin(cn),
330  endSlot = PosteriorCnSortingIterator::end(cn);
331  itSlot != endSlot; ++itSlot) {
332  const PosteriorCn::Slot& slot = *itSlot;
333  xml << Core::XmlOpen("slot") + Core::XmlAttribute("n", slot.size());
334  for (PosteriorCn::Slot::const_iterator itArc = slot.begin(), endArc = slot.end(); itArc != endArc; ++itArc)
335  xml << Core::XmlOpen("arc")
336  << Core::XmlFull("word", alphabet->symbol(itArc->label))
337  << Core::XmlFull("probability", itArc->score)
338  << Core::XmlClose("arc");
339  xml << Core::XmlClose("slot");
340  }
341  xml << Core::XmlClose("body");
342  }
343  {
344  xml << Core::XmlClose("posterior-cn");
345  }
346 }

350 void writePosteriorCnAsFlowAlignment(Flow::Cache& cache, ConstPosteriorCnRef cn, ConstSegmentRef segment) {
351  Flow::CacheWriter* cacheWriter = cache.newWriter(segment->segmentId());
352  verify(cacheWriter && cacheWriter->isOpen());
353  Flow::Attributes* flowAttr = new Flow::Attributes;
354  Core::Ref<const Flow::Attributes> flowAttrRef = Core::Ref<const Flow::Attributes>(flowAttr);
355  flowAttr->set("datatype", "flow-alignment");
356  Flow::DataAdaptor<Speech::Alignment>* flowAlignment = new Flow::DataAdaptor<Speech::Alignment>();
357  Flow::DataPtr<Flow::Data> flowAlignmentRef = Flow::DataPtr<Flow::Data>(flowAlignment);
358  flowAlignment->setStartTime(segment->startTime());
359  flowAlignment->setEndTime(segment->endTime());
360  Speech::Alignment& alignment = flowAlignment->data();
361  Time t = 0;
362  for (PosteriorCnSortingIterator itSlot = PosteriorCnSortingIterator::begin(cn),
363  endSlot = PosteriorCnSortingIterator::end(cn);
364  itSlot != endSlot; ++itSlot, ++t) {
365  const PosteriorCn::Slot& slot = *itSlot;
366  for (PosteriorCn::Slot::const_iterator itArc = slot.begin(), endArc = slot.end(); itArc != endArc; ++itArc)
367  if ((Fsa::FirstLabelId <= itArc->label) && (itArc->label <= Fsa::LastLabelId))
368  alignment.push_back(Speech::AlignmentItem(t, itArc->label, itArc->score));
369  }
370  alignment.sortItems();
371  cacheWriter->putAttributes(flowAttrRef);
372  cacheWriter->putData(flowAlignment);
373  delete cacheWriter;
374 }
