83 void KeyedEstimator::setFeatureDescription(const Mm::FeatureDescription& description) {
84  description[featureStream_].getValue(Mm::FeatureDescription::nameDimension, featureDimension_);
85 
86  std::string key;
87  corpusKey_->resolve(key);
88  Core::StringHashMap<std::string>::const_iterator i = corpusKeyMap_.find(key);
89  if (i != corpusKeyMap_.end())
90  key = i->second;
91 
92  currentAccumulator_ = accumulatorCache_.findForWriteAccess(key);
93 
94  if (currentAccumulator_ == 0) {
95  if (featureDimension_ == 0)
96  warning("Input vector size is 0");
97 
98  if (featureDimension_ < modelDimension_)
99  warning("Model dimension larger than feature dimension");
100 
101  createAccumulator(key);
102 
103  if (!accumulatorCache_.insert(key, currentAccumulator_))
104  defect();
105  }
106 
107  verify(currentAccumulator_);
108  createAssigningFeatureScorer();
109 }

125 void KeyedEstimator::combine() {
126  verify(operation_ == combines);
127 
128  std::set<std::string> keysToAdd = accumulatorCacheToAdd_->keys();
129  if (keysToAdd.size() == 0)
130  warning("Accumulator to add is empty!");
131 
132  for (std::set<std::string>::iterator key = keysToAdd.begin();
133  key != keysToAdd.end(); ++key) {
134  const Accumulator* accumulatorToAdd =
135  accumulatorCacheToAdd_->findForReadAccess(*key);
136  verify(accumulatorToAdd);
137 
138  currentAccumulator_ = accumulatorCache_.findForWriteAccess(*key);
139  if (currentAccumulator_ == 0) {
140  featureDimension_ = (*accumulatorToAdd)->featureDimension();
141  createAccumulator(*key);
142  if (!accumulatorCache_.insert(*key, currentAccumulator_))
143  defect();
144  }
145  verify(currentAccumulator_);
146 
147  log("Combining accumulator '") << *key << "' using weight '" << paramCombinationWeight(config) << "'";
148  (*currentAccumulator_)->combine(**accumulatorToAdd, paramCombinationWeight(config));
149  }
