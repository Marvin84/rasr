53 class MappingBuilder {
54 private:
55  ConstSemiringRef source_;
56  ConstSemiringRef target_;
57 
58 public:
59  MappingBuilder(ConstSemiringRef source, ConstSemiringRef target)
60  : source_(source), target_(target) {}
61 
62  ProjectionMatrix buildLinearMapping(ScoreId offset = 0, bool scaled = true) const {
63  ProjectionMatrix scaledMapping(target_->size());
64  ProjectionMatrix::iterator itProjectionMatrix = scaledMapping.begin();
65  ScoreId id = 0;
66  for (; (itProjectionMatrix != scaledMapping.end()) && (id < offset); ++itProjectionMatrix, ++id)
67  itProjectionMatrix->resize(source_->size(), 0.0);
68  id = 0;
69  for (; (itProjectionMatrix != scaledMapping.end()) && (id < source_->size()); ++itProjectionMatrix, ++id) {
70  itProjectionMatrix->resize(source_->size(), 0.0);
71  (*itProjectionMatrix)[id] = scaled ? source_->scale(id) : Semiring::DefaultScale;
72  }
73  for (; (itProjectionMatrix != scaledMapping.end()); ++itProjectionMatrix)
74  itProjectionMatrix->resize(source_->size(), 0.0);
75  return scaledMapping;
76  }
77 
78  ProjectionMatrix buildMapping(const SymbolicMapping& symbolicMapping, bool scaled = true) const {
79  require(symbolicMapping.size() == target_->size());
80  ProjectionMatrix scaledMapping(target_->size());
81  ProjectionMatrix::iterator itProjectionMatrix = scaledMapping.begin();
82  for (SymbolicMapping::const_iterator itKeyList = symbolicMapping.begin();
83  itKeyList != symbolicMapping.end(); ++itKeyList, ++itProjectionMatrix) {
84  itProjectionMatrix->resize(source_->size(), 0.0);
85  for (KeyList::const_iterator itKey = itKeyList->begin(); itKey != itKeyList->end(); ++itKey) {
86  ScoreId id = source_->id(*itKey);
87  if (source_->hasId(id))
88  (*itProjectionMatrix)[id] = scaled ? source_->scale(id) : Semiring::DefaultScale;
89  else
90  Core::Application::us()->error("No dimension \"%s\" found in semiring \"%s\"",
91  itKey->c_str(), source_->name().c_str());
92  }
93  }
94  return scaledMapping;
95  }

334 ConstLatticeRef fromFsa(Fsa::ConstAutomatonRef fsa,
335  ConstSemiringRef semiring, ScoresRef scores, ScoreId id,
336  Lexicon::AlphabetId inputAlphabetId, Lexicon::AlphabetId outputAlphabetId) {
337  if (!semiring) {
338  semiring = Semiring::create(Fsa::SemiringTypeTropical, 1);
339  semiring->setKey(0, "weight");
340  }
341  if (!scores)
342  scores = semiring->one();
343  FsaToLattice::WeightMapRef weightMap;
344  if (id == Semiring::InvalidId)
345  weightMap = FsaToLattice::WeightMapRef(new FsaToLattice::ConstantToScoresMap(scores));
346  else
347  weightMap = FsaToLattice::WeightMapRef(new FsaToLattice::WeightToScoresMap(semiring, id, scores));
348  Lexicon::AlphabetMapRef inputMap =
349  Lexicon::us()->alphabetMap(fsa->getInputAlphabet(), inputAlphabetId);
350  verify(inputMap);
351  Lexicon::AlphabetMapRef outputMap;
352  if (fsa->type() != Fsa::TypeAcceptor) {
353  outputMap = Lexicon::us()->alphabetMap(fsa->getInputAlphabet(), inputAlphabetId);
354  verify(outputMap);
355  }
356  return ConstLatticeRef(new FsaToLattice(fsa, semiring, weightMap, inputMap, outputMap));
357 }
