53 bool FormantExtraction::calculateResonanceFrequency(_float a1, _float a2, _float& frequency) {
54  _float tmp = -a1 * (1 + a2) / (4.0 * a2);
55  frequency = acos(tmp);
56 
57  if (fabs(tmp) > 1.0) {
58  _float relative_damping;
59  if (!calculateContinuousModel(a1, a2, frequency, relative_damping)) {
60  warning("calculateResonanceFrequency failed: a1 = %f, a2 = %f", a1, a2);
61  return false;
62  }
63  warning("calculateResonanceFrequency: relative damping too high = %f fr = %f ; a1 = %f, a2 = %f",
64  relative_damping, frequency * (_float)sample_rate_ / 2.0 / M_PI, a1, a2);
65  }
66 
67  frequency *= (_float)sample_rate_ / 2.0 / M_PI;
68  return true;
69 }

71 bool FormantExtraction::calculateMaxAmplitude(_float a1, _float a2, _float& amplitude) {
72  _float tmp = a1 * a1 + (1 - a2) * (1 - a2) - (a1 * a1 * (1 + a2) * (1 + a2) / (4 * a2));
73  if (a2 == 0 || tmp <= 0.0) {
74  _float resonance_omega;
75  _float relative_damping;
76  if (!calculateContinuousModel(a1, a2, resonance_omega, relative_damping) ||
77  relative_damping < 0.7) {
78  warning("calculateMaxAmplitude failed: a1 = %f, a2 = %f", a1, a2);
79  return false;
80  }
81 
82  // take the amplitude at 0 frequency
83  tmp = (1 + a1 + a2) * (1 + a1 + a2);
84  warning("calculateMaxAmplitude: relative damping too high = %f fr = %f ; a1 = %f, a2 = %f",
85  relative_damping, resonance_omega * (_float)sample_rate_ / 2.0 / M_PI, a1, a2);
86  }
87 
88  amplitude = 1.0 / sqrt(tmp);
89  return true;
90 }

92 bool FormantExtraction::calculateBandwidth(_float a1, _float a2, _float& bandwidth) {
93  _float resonance_omega;
94  _float relative_damping;
95  if (!calculateContinuousModel(a1, a2, resonance_omega, relative_damping)) {
96  warning("calculateBandwidth failed: a1 = %f, a2 = %f", a1, a2);
97  return false;
98  }
99 
100  _float tmp1 = 1 - 2 * relative_damping * relative_damping;
101  _float tmp2 = 2 * relative_damping * sqrt(1 - relative_damping * relative_damping);
102 
103  if (tmp1 >= tmp2)
104  bandwidth = resonance_omega * (sqrt(tmp1 + tmp2) - sqrt(tmp1 - tmp2));
105  else
106  bandwidth = resonance_omega * sqrt(sqrt(2));
107 
108  bandwidth *= (sample_rate_ / 2.0 / M_PI);
109  return true;
110 }

112 bool FormantExtraction::calculateContinuousModel(_float a1,
113  _float a2,
114  _float& resonance_omega,
115  _float& relative_damping) {
116  if (a2 < 0.0) {
117  warning("calculateContinuousModel failed: a1 = %f, a2 = %f", a1, a2);
118  return false;
119  }
120  _float delta = ::log(a2) / 2.0;
121 
122  _float omega = -a1 / 2.0 / sqrt(a2);
123  if (fabs(omega) > 1.0) {
124  if (fabs(omega) < 1.2) // case of omega = 0. recover rounding error
125  {
126  warning("calculateContinuousModel omega rounded to 1(cont.: 0) a1 = %f, a2 = %f, delta = %f, omega = %f", a1, a2, delta, omega);
127  omega = omega > 0 ? 1.0 : -1.0;
128  }
129  else {
130  warning("calculateContinuousModel failed: a1 = %f, a2 = %f, delta = %f, omega = %f", a1, a2, delta, omega);
131  return false;
132  }
133  }
134  omega = acos(omega);
135 
136  resonance_omega = sqrt(delta * delta + omega * omega);
137  relative_damping = -delta / resonance_omega;
138  return true;
139 }
