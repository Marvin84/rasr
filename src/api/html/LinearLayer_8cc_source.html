117 template<typename T>
118 void LinearLayer<T>::initializeNetworkParameters() {
119  for (u32 stream = 0; stream < weights_.size(); stream++) {
120  require(!weights_[stream].isComputing());
121  }
122  require(!bias_.isComputing());
123  switch (initializationType_) {
124  case random:
125  Core::Component::log("Initialize parameters of ") << Precursor::getName() << " randomly";
126  initializeParametersRandomly();
127  break;
128  case identity:
129  Core::Component::log("Initialize parameters of ") << Precursor::getName() << " with the identity matrix";
130  initializeParametersWithIdentityMatrix();
131  break;
132  case file:
133  Core::Component::log("Initialize parameters of ") << Precursor::getName() << " from file";
134  loadNetworkParameters(parameterFile_);
135  break;
136  case zero:
137  default:
138  Core::Component::log("Initialize parameters of ") << Precursor::getName() << " with zero";
139  initializeParametersWithZero();
140  break;
141  };
142 
143  for (u32 stream = 0; stream < weights_.size(); stream++) {
144  Core::Component::log("weight matrix size: ") << weights_[stream].nRows() << " x " << weights_[stream].nColumns();
145  }
146  Core::Component::log("bias size: ") << bias_.nRows();
147 
148  // Initialization done
149  Precursor::needInit_ = false;
150 }

152 /** Initialize the weights with random values */
153 template<typename T>
154 void LinearLayer<T>::initializeParametersRandomly() {
155  // initialize the bias and weights (resize)
156  bias_.resize(Precursor::getOutputDimension(), 0, true);
157  weights_.resize(Precursor::nInputActivations());
158  for (u32 stream = 0; stream < weights_.size(); stream++)
159  weights_.at(stream).resize(Precursor::getInputDimension(stream), Precursor::getOutputDimension());
160 
161  require(weightInitializationRangeMin_ < weightInitializationRangeMax_);
162  require(biasInitializationRangeMin_ < biasInitializationRangeMax_);
163 
164  // initialization of random number generator in Tools/NnTrainer
165 
166  // ... randomize the weights ...
167  for (u32 stream = 0; stream < Precursor::nInputActivations(); stream++) {
168  for (u32 row = 0; row < weights_.at(stream).nRows(); row++) {
169  for (u32 column = 0; column < weights_.at(stream).nColumns(); ++column) {
170  weights_.at(stream).at(row, column) = ((weightInitializationRangeMax_ - weightInitializationRangeMin_) * (T)Math::rand() / RAND_MAX) + weightInitializationRangeMin_;
171  }
172  }
173  }
174 
175  // ... and randomize the bias
176  for (u32 row = 0; row < bias_.nRows(); row++)
177  bias_.at(row) = hasBias_ ? ((biasInitializationRangeMax_ - biasInitializationRangeMin_) * (T)Math::rand() / RAND_MAX) + biasInitializationRangeMin_ : 0;
178  // Initialization done
179  Precursor::needInit_ = false;
180 }

242 /** Save weights to file */
243 template<typename T>
244 inline void LinearLayer<T>::saveNetworkParameters(const std::string& filename) const {
245  // synchronization
246  bool weightsComputing = weights_.at(0).isComputing();
247  bool biasComputing = bias_.isComputing();
248  if (weightsComputing) {
249  for (u32 stream = 0; stream < weights_.size(); stream++)
250  weights_.at(stream).finishComputation();
251  }
252  if (biasComputing)
253  bias_.finishComputation();
254 
255  // save bias and weights in one file
256  // get the size of the current layer ...
257  u32 inputSize = 0;
258  for (u32 stream = 0; stream < Precursor::nInputActivations(); stream++) {
259  inputSize += Precursor::getInputDimension(stream);
260  }
261  if (hasBias_)
262  inputSize++;
263 
264  u32 outputSize = Precursor::getOutputDimension();
265 
266  Math::Matrix<T> parameters(outputSize, inputSize);
267 
268  // Convert Math::FastVector + Math::FastMatrix -> Math::Matrix
269  // bias and all weight matrices are concatenated in output file
270  for (u32 i = 0; i < outputSize; ++i) {
271  if (hasBias_)
272  parameters[i][0] = bias_.at(i);
273 
274  u32 k = hasBias_ ? 1 : 0; // start at 1 because of the bias
275  for (u32 stream = 0; stream < Precursor::nInputActivations(); stream++) {
276  for (u32 j = 0; j < weights_.at(stream).nRows(); ++j) {
277  parameters[i][k] = weights_.at(stream).at(j, i);
278  k++;
279  }
280  }
281  }
282 
283  // write matrix to file
284  Math::Module::instance().formats().write(filename, parameters, 20);
285 
286  // synchronization
287 
288  if (weightsComputing) {
289  for (u32 stream = 0; stream < weights_.size(); stream++)
290  weights_.at(stream).initComputation(false);
291  }
292  if (biasComputing)
293  bias_.initComputation(false);
294 }

296 /** Forward the input */
297 template<typename T>
298 void LinearLayer<T>::_forward(const std::vector<NnMatrix*>& input, NnMatrix& output, bool reset) {
299  // boundary check
300  require_eq(bias_.size(), output.nRows());
301  require_eq(weights_.size(), input.size());
302  timeval start, end;
303 
304  // first: (input * weight) for each weight matrix (note: first stream handled separately due to reset flag)
305  gettimeofday(&start, NULL);
306  output.addMatrixProduct(weights_[0], *(input.at(0)), (reset ? T(0) : T(1)), T(1), true, false);
307  for (u32 stream = 1; stream < weights_.size(); stream++) {
308  output.addMatrixProduct(weights_.at(stream), *(input.at(stream)), T(1), T(1), true, false);
309  }
310 
311  Math::Cuda::deviceSync(Precursor::measureTime_ && Math::CudaDataStructure::hasGpu());
312  gettimeofday(&end, NULL);
313  timeForwardLinear_ += Core::timeDiff(start, end);
314 
315  // second: add bias
316  gettimeofday(&start, NULL);
317  if (hasBias_)
318  output.addToAllColumns(bias_);
319  Math::Cuda::deviceSync(Precursor::measureTime_ && Math::CudaDataStructure::hasGpu());
320  gettimeofday(&end, NULL);
321  timeForwardBias_ += Core::timeDiff(start, end);
322 }

374 /** Set the bias and the weight parameters.
375  *
376  * Set the bias and weight parameters from a single matrix.
377  * The first column of the matrix is the bias and the rest are the weight
378  * parameters
379  *
380  * @param parameters The matrix containing the bias and the weights
381  */
382 template<typename T>
383 void LinearLayer<T>::setParameters(const Math::Matrix<T>& parameters) {
384  for (u32 stream = 0; stream < weights_.size(); stream++)
385  require(!weights_[stream].isComputing());
386  require(!bias_.isComputing());
387 
388  // resize bias/ weights
389  require_eq(parameters.nRows(), this->getOutputDimension());
390  u32 totalInputSize = 0;
391  for (u32 stream = 0; stream < this->nInputActivations(); stream++) {
392  totalInputSize += this->getInputDimension(stream);
393  }
394  u32 inputSizeFromFile = hasBias_ ? parameters.nColumns() - 1 : parameters.nColumns();
395  if (inputSizeFromFile != totalInputSize) {
396  this->error("dimension mismatch: (parameter file vs. layer-dimension) ")
397  << inputSizeFromFile << " vs. " << totalInputSize;
398  }
399  bias_.resize(parameters.nRows());
400  weights_.resize(this->nInputActivations());
401  for (u32 stream = 0; stream < this->nInputActivations(); stream++) {
402  weights_.at(stream).resize(this->getInputDimension(stream), parameters.nRows());
403  }
404 
405  // Convert Flow::Matrix -> Flow::FastMatrix
406  for (u32 row = 0; row < parameters.nRows(); row++) {
407  // first: bias (first column)
408  if (hasBias_) {
409  bias_.at(row) = parameters[row][0];
410  }
411 
412  // second: weights (all other elements, interchange row <-> column)
413  u32 column = hasBias_ ? 1 : 0;
414  for (u32 stream = 0; stream < weights_.size(); stream++) {
415  for (u32 r = 0; r < weights_.at(stream).nRows(); r++) {
416  weights_.at(stream).at(r, row) = parameters[row][column];
417  column++;
418  }
419  }
420  }
421 
422  Precursor::needInit_ = false;
423 }

497 template<typename T>
498 template<typename S>
499 void BiasLayer<T>::removeLogPriorFromBias(const Prior<S>& priors) {
500  require(this->getBias());
501  require_eq(priors.size(), this->getBias()->size());
502  T prioriScale = priors.scale();
503  if (prioriScale != S(0.0)) {
504  this->log("removing scaled log-prior from bias parameters (scale: ") << prioriScale << ")";
505  // subtract
506  if (this->getBias()->isComputing()) {
507  typename Types<S>::NnVector priorVector;
508  priorVector.resize(priors.size());
509  priors.getVector(priorVector);
510  priorVector.initComputation();
511  this->getBias()->add(priorVector, -S(prioriScale));
512  }
513  else {
514  for (u32 c = 0; c < this->getBias()->size(); c++)
515  this->getBias()->at(c) -= prioriScale * priors.at(c);
516  }
517  }
518 }

571  // boundary check
572  require_eq(bias_.size(), output.nRows());
573  require_eq(input.size(), 1); // not supported otherwise yet
574  require_eq(input[0]->nRows(), bias_.size());

576  timeval start, end;
