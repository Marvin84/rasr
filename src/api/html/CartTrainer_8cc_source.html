46 /*
47  The default score function calculates the entropy
48  over the occurring matrixes, i.e. the probability
49  of an arbitrary matrix is just calculated as its
50  relative frequency (remember: the input of the scorer
51  is list of examples, where each example consists of a
52  collection of properties and a matrix).
53 */
54 static const Core::ParameterString paramScorer(
55  "score-function",
56  "function to minimize",
57  "ID3");
58 
59 int CartTrainer::main(const std::vector<std::string>& arguments) {
60  bool train = !paramDry(config);
61 
62  if (paramScorer(config) != paramScorer.defaultValue()) {
63  error("unknown score function \"%s\"; scorer support is so far limited to \"%s\"",
64  paramScorer(config).c_str(), paramScorer.defaultValue().c_str());
65  return 1;
66  }
67  Cart::ConstScorerRef scorer = Cart::ConstScorerRef(new Cart::ID3(select("id3")));
68 
69  DecisionTreeTrainer trainer(config);
70  trainer.setScorer(scorer);
71  log("load training plan");
72  std::string trainingFilename = DecisionTreeTrainer::paramTrainingFilename(config);
73  if (!trainingFilename.empty()) {
74  trainer.loadFromFile(trainingFilename);
75  Core::Channel trainerPlainChannel(select("training"), "plain", Core::Channel::disabled);
76  if (trainerPlainChannel.isOpen())
77  trainer.write(trainerPlainChannel);
78  Core::XmlChannel trainerXmlChannel(select("training"), "xml", Core::Channel::disabled);
79  if (trainerXmlChannel.isOpen())
80  trainer.writeXml(trainerXmlChannel);
81  }
82  else {
83  train = false;
84  warning("unable to parse training plan file \"%s\"; cannot proceed training",
85  trainingFilename.c_str());
86  }
87 
88  ExampleList examples(config);
89  log("load examples");
90  std::string exampleFilename = ExampleList::paramExampleFilename(config);
91  if (!exampleFilename.empty()) {
92  examples.loadFromFile(exampleFilename);
93  Core::Channel examplePlainChannel(select("example"), "plain", Core::Channel::disabled);
94  if (examplePlainChannel.isOpen())
95  examples.write(examplePlainChannel);
96  Core::XmlChannel exampleXmlChannel(select("example"), "xml", Core::Channel::disabled);
97  if (exampleXmlChannel.isOpen())
98  examples.writeXml(exampleXmlChannel);
99  }
100  else {
101  train = false;
102  warning("unable to parse example list file \"%s\"; cannot proceed training",
103  exampleFilename.c_str());
104  }
105 
106  if (train) {
107  DecisionTree tree(config);
108  ClusterList* clusters;
109  log("train decision tree");
110  if (!(clusters = trainer.train(&tree, examples))) {
111  error("error while training decision tree");
112  return 2;
113  }
114 
115  Core::Channel treePlainChannel(select("decision-tree"), "plain", Core::Channel::disabled);
116  if (treePlainChannel.isOpen())
117  tree.write(treePlainChannel);
118  Core::XmlChannel treeXmlChannel(select("decision-tree"), "xml", Core::Channel::disabled);
119  if (treeXmlChannel.isOpen())
120  tree.writeXml(treeXmlChannel);
121  Core::Channel treeDotChannel(select("decision-tree"), "dot", Core::Channel::disabled);
122  if (treeDotChannel.isOpen())
123  tree.draw(treeDotChannel);
124  tree.writeToFile();
125 
126  Core::Channel clusterPlainChannel(select("cluster"), "plain", Core::Channel::disabled);
127  if (clusterPlainChannel.isOpen())
128  clusters->write(clusterPlainChannel);
129  Core::XmlChannel clusterXmlChannel(select("cluster"), "xml", Core::Channel::disabled);
130  if (clusterXmlChannel.isOpen())
131  clusters->writeXml(clusterXmlChannel);
132  clusters->writeToFile();
133  delete clusters;
134  }
135 
136  return 0;
137 }
