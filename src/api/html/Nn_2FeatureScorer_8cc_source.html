37 void BaseFeatureScorer::init(Core::Ref<const Mm::MixtureSet> mixtureSet) {
38  require(mixtureSet);
39  nClasses_ = mixtureSet->nMixtures();
40  if (labelWrapper_)
41  delete labelWrapper_;
42  labelWrapper_ = new ClassLabelWrapper(select("class-labels"), nClasses_);
43  if (!labelWrapper_->isOneToOneMapping())
44  error("no one-to-one correspondence between network outputs and classes!");
45 
46  network_.initializeNetwork(1);
47  require_eq(network_.getTopLayer().getOutputDimension(), labelWrapper_->nClassesToAccumulate());
48  LinearAndSoftmaxLayer<f32>* topLayer = dynamic_cast<LinearAndSoftmaxLayer<f32>*>(&network_.getTopLayer());
49  if (!topLayer)
50  error("output layer must be of type 'linear+softmax'");
51 
52  if (network_.getLayer(0).nInputActivations() != 1)
53  Core::Component::criticalError("Multiple input streams not implemented in Nn/FeatureScorer.");
54 
55  inputDimension_ = network_.getLayer(0).getInputDimension(0);
56 
57  // remove prior from bias
58  // read
59  if (prior_.fileName() != "")
60  prior_.read();
61  else
62  prior_.setFromMixtureSet(mixtureSet, *labelWrapper_);
63  // subtract
64  network_.finishComputation();
65  topLayer->removeLogPriorFromBias(prior_);
66  network_.initComputation();
67  log("l1 norm of all network weights is: ") << network_.l1norm();
68 }

102 void OnDemandFeatureScorer::init(Core::Ref<const Mm::MixtureSet> mixtureSet) {
103  Precursor::init(mixtureSet);
104  outputLayer_ = dynamic_cast<LinearAndSoftmaxLayer<f32>*>(network_.popLayer());
105  require(outputLayer_);
106  if (network_.nLayers() > 0) {
107  topLayerOutputDimension_ = network_.getTopLayer().getOutputDimension();
108  }
109  else {
110  if (outputLayer_->nInputActivations() != 1) {
111  Core::Application::us()->criticalError("Nn/OnDemandFeatureScorer is only implemented for a single input stream.");
112  }
113  topLayerOutputDimension_ = outputLayer_->getInputDimension(0);
114  }
115  // must be in main memory for getScore()
116  outputLayer_->getBias()->finishComputation();
117 }

119 void OnDemandFeatureScorer::forwardHiddenLayers(const Mm::FeatureVector& in, NnMatrix& out) const {
120  out.resize(topLayerOutputDimension_, 1);
121 
122  out.initComputation(false);
123 
124  if (network_.nLayers() > 0) {
125  network_.forward(in);
126  out.swap(network_.getTopLayerOutput());
127  }
128  else {
129  out.copy(&(in.at(0)));
130  }
131 
132  out.finishComputation();
133 }
