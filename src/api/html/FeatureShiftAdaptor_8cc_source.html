80 bool FeatureShiftAdaptor::work(Flow::PortId p) {
81  Flow::DataPtr<Flow::Vector<Mm::FeatureType>> in;
82  while (getData(0, in)) {
83  updateAlignment(*in);
84  Alignment& alignment = alignment_->data();
85  if (featureIndex_ >= alignment.size())
86  criticalError("Input stream (%d) is longer than alignment (%zd).", featureIndex_ + 1, alignment.size());
87  Mm::MixtureIndex mixture = acousticModel_->emissionIndex(alignment[featureIndex_++].emission);
88 
89  Math::Vector<Mm::FeatureType> inVec(Flow::Vector<Mm::FeatureType>((*in.get())));
90  Flow::Vector<Mm::FeatureType>* out = new Flow::Vector<Mm::FeatureType>(inVec - (static_cast<Mm::ShiftAdaptor*>(currentAdaptor_.get())
91  ->shiftVector(mixture)));
92  out->setStartTime(in->startTime());
93  out->setEndTime(in->endTime());
94  putData(0, out);
95  }
96  return putData(0, in.get());
97 }

113 bool FeatureShiftAdaptor::setKey(const std::string& key) {
114  if (key == currentKey_ || key == "")
115  return true;
116  currentKey_ = key;
117 
118  std::string mappedKey(key);
119  Core::StringHashMap<std::string>::const_iterator i = corpusKeyMap_.find(key);
120  if (i != corpusKeyMap_.end())
121  mappedKey = i->second;
122 
123  if (mappedKey == currentMappedKey_)
124  return true;
125  currentMappedKey_ = mappedKey;
126 
127  // Do the adaptation
128  log("Adaptation key: ") << currentMappedKey_;
129 
130  if (!adaptorCache_.findForReadAccess(currentMappedKey_)) {
131  if (!adaptorEstimatorCache_.findForReadAccess(currentMappedKey_)) {
132  criticalError("No adaptation data seen for key ") << currentMappedKey_ << ". Exiting!\n";
133  }
134  else {
135  adaptorCache_.insert(currentMappedKey_, new Core::IoRef<Mm::Adaptor>((*adaptorEstimatorCache_.findForReadAccess(currentMappedKey_))->adaptor()));
136  currentAdaptor_ = (*adaptorCache_.findForReadAccess(currentMappedKey_));
137  }
138  }
139  else {
140  currentAdaptor_ = (*adaptorCache_.findForReadAccess(currentMappedKey_));
141  }
142  return true;
143 }
