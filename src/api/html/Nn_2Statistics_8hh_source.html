210 template<typename T>
211 template<typename S>
212 void Statistics<T>::initialize(const NeuralNetwork<S>& neuralNetwork) {
213  layerIndexToTrainableLayerIndex_.resize(neuralNetwork.nLayers());
214  for (u32 layer = 0; layer < neuralNetwork.nLayers(); layer++)
215  layerIndexToTrainableLayerIndex_[layer] = neuralNetwork.layerIndexToTrainableLayerIndex(layer);
216 
217  for (u32 layer = 0; layer < neuralNetwork.nLayers(); layer++) {
218  /* store which layer is trainable */
219  if (neuralNetwork.getLayer(layer).isTrainable()) {
220  s32 trainableLayerIndex = layerIndexToTrainableLayerIndex_.at(layer);
221  // initialize weights/bias gradient with the corresponding size
222  if (hasGradient_) {
223  // different layer types have different structures of the
224  // trainable parameters, so let them decide on the structure
225  // of the gradient accumulators
226  gradientWeights_[trainableLayerIndex].resize(neuralNetwork.getLayer(layer).nInputActivations());
227  for (u32 stream = 0; stream < neuralNetwork.getLayer(layer).nInputActivations(); stream++) {
228  neuralNetwork.getLayer(layer).resizeWeightsGradient(gradientWeights_.at(trainableLayerIndex).at(stream), stream);
229  }
230  neuralNetwork.getLayer(layer).resizeBiasGradient(gradientBias_.at(trainableLayerIndex));
231  }
232  }
233  }
234  if (hasGradient_) {
235  gradientWeights_.resize(neuralNetwork.nTrainableLayers());
236  gradientBias_.resize(neuralNetwork.nTrainableLayers());
237  }
238 }

250 template<typename T>
251 template<typename S>
252 void Statistics<T>::add(const Statistics<S>& rhs) {
253  require(isComputing_);
254  require(rhs.isComputing_);
255  require_eq(hasBaseStatistics_, rhs.hasBaseStatistics_);
256  require_eq(hasMeanAndVariance_, rhs.hasMeanAndVariance_);
257  require_eq(hasGradient_, rhs.hasGradient_);
258  require_eq(hasClassCounts_, rhs.hasClassCounts_);
259 
260  nObservations_ += rhs.nObservations_;
261  totalWeight_ += rhs.totalWeight_;
262  // add base statistics
263  if (hasBaseStatistics_) {
264  nClassificationErrors_ += rhs.nClassificationErrors_;
265  objectiveFunction_ += rhs.objectiveFunction_;
266  }
267  // add mean and variance
268  if (hasMeanAndVariance_) {
269  sum_.add(rhs.sum_);
270  sumOfSquares_.add(rhs.sumOfSquares_);
271  }
272  // add class counts
273  if (hasClassCounts_) {
274  for (std::map<u32, u32>::const_iterator it = rhs.classCount_.begin(); it != rhs.classCount_.end(); it++) {
275  incClassCount(it->first, it->second);
276  }
277  }
278  // add gradient
279  if (hasGradient_) {
280  require_eq(gradientWeights_.size(), rhs.gradientWeights_.size());
281  for (u32 l = 0; l < gradientWeights_.size(); l++) {
282  require_eq(gradientWeights_[l].size(), rhs.gradientWeights_[l].size());
283  for (u32 s = 0; s < gradientWeights_[l].size(); s++) {
284  gradientWeights_[l][s].add(rhs.gradientWeights_[l][s]);
285  }
286  gradientBias_[l].add(rhs.gradientBias_[l]);
287  }
288  }
289 }

291 template<typename T>
292 template<typename S>
293 void Statistics<T>::copyStructure(const Statistics<S>& statistics) {
294  hasClassCounts_ = statistics.hasClassCounts_;
295  hasBaseStatistics_ = statistics.hasBaseStatistics_;
296  hasGradient_ = statistics.hasGradient_;
297  gradientWeights_.resize(statistics.gradientWeights_.size());
298  gradientBias_.resize(statistics.gradientBias_.size());
299  layerIndexToTrainableLayerIndex_ = statistics.layerIndexToTrainableLayerIndex_;
300  statisticTypes_ = statistics.statisticTypes_;
301 
302  if (hasMeanAndVariance_) {
303  sum_.resize(statistics.sum_.size());
304  sumOfSquares_.resize(statistics.sumOfSquares_.size());
305  }
306  if (hasGradient_) {
307  for (u32 layer = 0; layer < gradientWeights_.size(); layer++) {
308  gradientWeights_[layer].resize(statistics.gradientWeights_[layer].size());
309  for (u32 stream = 0; stream < gradientWeights_[layer].size(); stream++) {
310  gradientWeights_[layer][stream].resize(statistics.gradientWeights_[layer][stream].nRows(),
311  statistics.gradientWeights_[layer][stream].nColumns());
312  }
313  }
314  for (u32 layer = 0; layer < gradientBias_.size(); layer++) {
315  gradientBias_[layer].resize(statistics.gradientBias_[layer].nRows());
316  }
317  }
318 
319  reset();
320 }
