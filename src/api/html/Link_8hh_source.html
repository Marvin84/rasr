93  /** Send data packet to down-stream Node.
94  * @return true on success (Note: Won't fail anyway.)
95  */
96  inline bool putData(Data* d) {
97  /* node tried to send a packet which is not of advertised type */
98  require(d);
99  require(Data::isSentinel(d) || !datatype() || d->datatype() == datatype());
100 
101  if (is_fast_) {
102  if (queue_.isEmpty()) {
103  if (isEmpty(fast_data_)) {
104  d->increment();
105  fast_data_ = d;
106  return true;
107  }
108  else {
109  fast_data_->decrement();
110  queue_.put(fast_data_);
111  fast_data_ = sentinelEmpty();
112  }
113  }
114  queue_.put(d);
115  return true;
116  }
117  queue_.putAtomar(d);
118  return true;
119  }
