50 void BandMllrEstimator::estimateWMatrices() {
51  Math::Vector<Precursor::ZAccumulator> z;
52  Math::Vector<Precursor::GAccumulator> g;
53 
54  propagate(leafZAccumulators_, z, tree_->root());
55  propagate(leafGAccumulators_, g, tree_->root());
56 
57  Math::Vector<Matrix::Type> resultRow;
58 
59  ensure(z.size() == g.size());
60  count_.resize(z.size());
61  for (u32 id = 0; id < z.size(); ++id) { //loop over all nodes of tree_
62  count_[id] = g[id].count();
63  w_[id].resize(dimension_, dimension_ + 1);
64  if (count_[id] > minAdaptationObservations_) {
65  //solve MLLR equation row-wise
66  for (u32 row = 0; row < dimension_; ++row) {
67  resultRow = solveRowEquation(g[id].matrix(), z[id].matrix(), row);
68  w_[id][row] = resultRow;
69  }
70  }
71  else
72  w_.erase(id);
73  }
74  return;
75 }

77 Math::Vector<Matrix::Type> BandMllrEstimator::solveRowEquation(
78  const Matrix& g,
79  const Matrix& z,
80  u32 wRow) {
81  u16 nRows = 2 * nBands_ + 1;
82  const s16 offset = wRow - nBands_; //originally wRow-nBands_-1, but nRow is starts from 0
83  Math::Vector<Matrix::Type> result(dimension_ + 1, 0.0);
84  Math::Vector<Matrix::Type> x(nRows + 1);
85  Math::Vector<Matrix::Type> c(nRows + 1);
86  Matrix B(nRows + 1, nRows + 1);
87 
88  for (u16 row = 1; row < nRows + 1; ++row) {
89  if (row + offset > 0 && u16(row + offset) < dimension_ + 1) {
90  B[row][0] = g[row + offset][0];
91  B[0][row] = g[0][row + offset]; //put here for efficiency, row=col
92  for (u16 col = 1; col < nRows + 1; ++col) {
93  if (col + offset > 0 && u16(col + offset) < dimension_ + 1) {
94  B[row][col] = g[row + offset][col + offset];
95  }
96  }
97  c[row] = z[wRow][row + offset];
98  }
99  }
100 
101  c[0] = z[wRow][0];
102  B[0][0] = g[0][0];
103 
104  Matrix BInverse(B);
105  Math::Lapack::pseudoInvert(BInverse);
106  x = BInverse * c;
107  for (u16 row = 1; row < nRows + 1; ++row) {
108  if (row + offset > 0 && u16(row + offset) < dimension_ + 1) {
109  result[row + offset] = x[row];
110  }
111  }
112  result[0] = x[0];
113  return result;
114 }
