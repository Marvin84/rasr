51 bool PolinomialVectorInterpolation::checkInterpolationTime(Time time) {
52  if (slidingWindow_.maxSize() >= 2) {

63 bool PolinomialVectorInterpolation::copyControlPoint(Time time, DataPointer& out) const {
64  SlidingWindow<DataPointer>::ConstantIterator i = slidingWindow_.begin();
65  for (; i != slidingWindow_.end(); ++i) {
66  Time startTime = (*i)->startTime();
67 
68  if (Core::isSignificantlyLess(startTime, time, Flow::timeTolerance))
69  return false;
70  else if (Core::isAlmostEqual(startTime, time, Flow::timeTolerance)) {
71  out = *i;
72  return true;
73  }

85 void PolinomialVectorInterpolation::calculateParameters() {
86  verify(!equationSystemSolved_);
87  resize();
88  for (u32 row = 0; row < A_.nRows(); ++row) {
89  Data& controlPoint = **(slidingWindow_.reverseBegin() + row);
90  verify(controlPoint.size() == B_.nColumns());
91 
92  for (u32 column = 0; column < A_.nColumns(); ++column)
93  A_(row, column) = pow(controlPoint.startTime(), column);
94 
95  for (u32 dimension = 0; dimension < B_.nColumns(); ++dimension)
96  B_(row, dimension) = controlPoint[dimension];
97  }
98  if (getrf(A_, pivotIndices_) != 0 || getrs(A_, B_, pivotIndices_) != 0)
99  defect();
100  equationSystemSolved_ = true;
101 }
