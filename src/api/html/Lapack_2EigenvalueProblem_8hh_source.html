136 class GenEigenProblemWithSchurDecomposition : public Math::GeneralizedEigenvalueProblem, public EigenvalueProblem {
137 private:
138  /**
139  * If input matrices have widely varying magnitudes
140  * then they should be first balanced.
141  * The algorithm is O(n^2).
142  * Balancing a matrix pair (A,B) includes, first,
143  * permuting rows and columns to isolate eigenvalues,
144  * second, applying diagonal similarity transformation
145  * to the rows and columns to make the rows and columns
146  * as close in norm as possible. The computed reciprocal
147  * condition numbers correspond to the balanced matrix.
148  * Permuting rows and columns will not change the condition
149  * numbers (in exact arithmetic) but diagonal scaling will.
150  * For further explanation of balancing, see section 4.11.1.2
151  * of LAPACK Users' Guide.
152  */
153  enum BalancingType {
154  noneBalancing,
155  permute,
156  scale,
157  permuteAndScale
158  };
