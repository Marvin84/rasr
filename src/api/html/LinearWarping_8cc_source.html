57 Warping* LinearWarpingNode::createWarping() {
58  if (warpingFactor_() <= 0)
59  error("Cannot warp with factor %f.", warpingFactor_());
60  if (warpingLimit_ <= 0 || warpingLimit_ >= 1)
61  error("Cannot warp with limit %f.", warpingLimit_);
62  respondToDelayedErrors();
63 
64  Warping* result = new Warping;
65 
66  Math::AnalyticFunctionFactory factory(select("warping-function"));
67  factory.setSampleRate(sampleRate_);
68  factory.setDomainType(Math::AnalyticFunctionFactory::discreteDomain);
69  factory.setMaximalArgument(inputSize_ - 1);
70  Math::UnaryAnalyticFunctionRef warpingFunction = factory.createTwoPieceLinearFunction(warpingFactor_(), warpingLimit_);
71  ensure(warpingFunction);
72 
73  if (interpolateOverWarpedAxis_)
74  result->setWarpingFunction(warpingFunction, inputSize_, mergeType_, interpolationType_);
75  else
76  result->setInverseWarpingFunction(warpingFunction->invert(), inputSize_, interpolationType_);
77  return result;
78 }
