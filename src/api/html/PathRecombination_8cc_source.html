484 u32 PathRecombination::recombinationInterval(StateId a, StateId b) const {
485  verify(distancesPtr_ && offsetAndDistanceStateForStatePtr_);
486  {
487  // Try asymetric linear recombination
488  u32 linear = linearChainLength(a, b);
489  if (linear == Core::Type<u32>::max)
490  linear = linearChainLength(b, a);
491 
492  if (linear != Core::Type<u32>::max)
493  return t(linear, 0);
494  }
495 
496  u32 offsetA = offsetAndDistanceStateForStatePtr_[a].first, offsetB = offsetAndDistanceStateForStatePtr_[b].first;
497  const DistanceState& distancesA = distancesPtr_[offsetAndDistanceStateForStatePtr_[a].second];
498  const DistanceState& distancesB = distancesPtr_[offsetAndDistanceStateForStatePtr_[b].second];
499 
500  if (approximateLinearSequences_ && (offsetA || offsetB))
501  return t(offsetA, offsetB) + recombinationInterval(offsetA ? getUniqueSuccessor(a) : a, offsetB ? getUniqueSuccessor(b) : b);
502 
503  {
504  // Try using the cache
505  IntervalCache::const_iterator cacheIt = intervalCache_.find(std::make_pair(a, b));
506  if (cacheIt != intervalCache_.end())
507  return cacheIt->second;
508 
509  cacheIt = intervalCache_.find(std::make_pair(b, a));
510  if (cacheIt != intervalCache_.end())
511  return cacheIt->second;
512  }
513 
514  manager.next();
515 
516  currentVisits_ = 0;
517 
518  u32 maxInterval = 0;
519 
520  if (recombinationStateMap_[b]) {
521  maxInterval = const_cast<PathRecombination&>(*this).R(offsetA, distancesA, offsetB, distancesB, recombinationStateMap_[b], 0);
522  }
523  else {
524  for (std::vector<std::pair<u32, u32>>::const_iterator it = distancesB.directSuccessorStates.begin(); it != distancesB.directSuccessorStates.end(); ++it) {
525  if (it->first > 100)
526  std::cout << "distance to direct successor " << it->first << std::endl;
527 
528  u32 interval = const_cast<PathRecombination&>(*this).R(offsetA, distancesA, offsetB + it->first, distancesB, it->second, 0);
529 
530  if (interval > maxInterval)
531  maxInterval = interval;
532  }
533  }
534 
535  if (intervalCache_.size() > maxCacheSize_) {
536  std::cout << "clearing interval cache" << std::endl;
537  intervalCache_.clear();
538  }
539 
540  intervalCache_.insert(std::make_pair(std::make_pair(a, b), maxInterval));
541 
542  totalVisits_ += currentVisits_;
543  nVisits_ += 1;
544

549  u32 distanceIdx = &distancesA - distancesPtr_;

560  verify(distRec.distances.size() == distancesA.distances.size());
561

563 
564  u32 minOtherDist = Core::Type<u32>::max;
565  for (u32 otherRecState = 1; otherRecState < distRec.distances.size(); ++otherRecState) {
566  u32 dist = distancesA.distances[otherRecState].shortestDistance + distRec.distances[otherRecState].shortestDistance;
567  if (dist < minOtherDist)
568  minOtherDist = dist;

581 u32 PathRecombination::R(u32 offsetA, const DistanceState& distancesA, u32 distB, const DistanceState& distancesB, u32 recombinationState, u32 depth) {
582  ++currentVisits_;
583 
584  verify(recombinationState != 0);
585 
586  if (visitingRecombinationState_[recombinationState])
587  return Core::Type<u32>::max;
588 
589  visitingRecombinationState_[recombinationState] = true;
590 
591  u32 distA = offsetA + distancesA.distances[recombinationState].shortestDistance;
592 
593  u32 intervalHere = t(distA, distB);
594 
595  u32 intervalNext = 0;
596 
597  const RecombinationState& recState(recombinationStates_[recombinationState]);
598 
599  if (recState.loop || depth >= maxDepth_ || (truncateNotPromising_ && isNotPromising(recombinationState, distancesA)) || distB >= maxExactInterval_) {
600  intervalNext = Core::Type<u32>::max;
601  }
602  else {
603  for (std::vector<RecombinationState::Successor>::const_iterator succ = recState.successors.begin(); succ != recState.successors.end(); ++succ) {
604  u32 interval = R(offsetA, distancesA, distB + succ->longestDistance, distancesB, recombinationStateMap_[succ->state], depth + 1);
605  if (interval > intervalNext)
606  intervalNext = interval;
607  }
608  }
609 
610  visitingRecombinationState_[recombinationState] = false;
611 
612  return std::min(intervalHere, intervalNext);
