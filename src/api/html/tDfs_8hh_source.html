24 /**
25  * Due to the principle of on-demand computation, exploring
26  * automata becomes an important issue as states are not directly
27  * accessable. Fsa provides this general implementation of
28  * depth-first search and makes intense use of it itself. Details
29  * of the algorithm can be looked up in (Cormen et al.). It stores
30  * state information in a single vector. DfsState uses the design
31  * pattern of a state. Intended use is as follows:
32  *
33  * 1. derive a specialized depth-first search state from DfsState
34  * 2. overload callback methods appropriately
35  * 3. create an instance of your state and pass automaton to it
36  * 4. call dfs to process your automaton
37  *
38  * The use of the state design pattern allows that information
39  * gathered upon traversal of the automaton is being kept after
40  * traversal.
41  **/
42 template<class _Automaton>
43 class DfsState {
44 public:
45  typedef typename _Automaton::Arc _Arc;
46  typedef typename _Automaton::State _State;
47  typedef typename _Automaton::ConstStateRef _ConstStateRef;
48  typedef typename _Automaton::ConstRef _ConstAutomatonRef;
49  /**
50  * The possible colors of a state.
51  **/
52  typedef Fsa::StateId Color;
53  static const Color White = 0xc0000000; // unprocessed
54  static const Color Gray = 0x80000000; // discovered
55  static const Color Black = 0x40000000; // finished
56 
57 private:
58  void setColor(Fsa::StateId s, Color c) {
59  info_[s] = (info_[s] & Fsa::StateIdMask) | c;
60  }
61  void setPredecessor(Fsa::StateId s, Fsa::StateId p) {
62  info_[s] = (info_[s] & ~Fsa::StateIdMask) | p;
63  }
64 
65 protected:
66  _ConstAutomatonRef fsa_;
67  Core::Vector<Fsa::StateId> info_;
68 
69 public:
70  DfsState(_ConstAutomatonRef f)
71  : fsa_(f) {}
72  virtual ~DfsState() {}

76  * @param sp reference to state
77  **/
78  virtual void discoverState(_ConstStateRef sp) {}
79 
80  /**
81  * Called after all subtrees starting at a previously
82  * discovered state have been processed.
83  * @param s index of the state
84  **/
85  virtual void finishState(Fsa::StateId s) {}
86 
87  /**
88  * Called upon discovery of a tree arc.
89  * Tree arcs never close a loop. For this reason they are
90  * "forward" arcs.
91  * @param from reference to the start state
92  * @param a reference to the arc
93  **/
94  virtual void exploreTreeArc(_ConstStateRef from, const _Arc& a) {}
95 
96  /**
97  * Called upon discovery of a non-tree arc.
98  * Non-tree arcs may close a loop, but this is not
99  * necessarily the case. To be precise, a loop occurs if the
100  * target state is grey. Such arcs are called "backward"
101  * arcs. Trivial loops (source state is target state) are
102  * also non tree arcs.
103  * @param from reference to the start state
104  * @param a reference to the arc
105  **/
106  virtual void exploreNonTreeArc(_ConstStateRef from, const _Arc& a) {}
107 
108  /**
109  * Called after all subtrees starting at a previously
110  * explored arc have been processed. Note: This function
111  * is called only within the recursive traversal.
112  * @param from reference to the start state
113  * @param a reference to the arc
114  */
115  virtual void finishArc(_ConstStateRef from, const _Arc& a) {}
116 
117  /**
118  * Called after all states reachable from the initial state
119  * have been processed.
120  **/
121  virtual void finish() {}

123  /**
124  * Request the highest state id of the automaton.
125  * @return the state id
126  **/
127  Fsa::StateId maxStateId() const {
128  return info_.size() - 1;
129  }
130 
131  Fsa::StateId size() const {

136  * Request the color of a state.
137  * @param s the state id
138  * @return the color of the state (see above)
139  **/
140  Color color(Fsa::StateId s) const {
141  return info_[s] & ~Fsa::StateIdMask;
142  }
143 
144  /**
145  * Request the predecessor of a state.
146  * @param s the state id
147  * @return the predecessor of a state
148  **/
149  Fsa::StateId predecessor(Fsa::StateId s) const {
150  return info_[s] & Fsa::StateIdMask;
151  }

157  void dfs(Core::ProgressIndicator* p = 0);

164  **/
165  void recursiveDfs(Core::ProgressIndicator* p = 0);
166 };
