26 /** Base class for normalization algorithms.
27  * Processing:
28  * updateStatistics is called if no new data entered or old data left the sliding window
29  * finalize() called right before normalization is applied to an output vector
30  * apply() called to perform normalization
31  * Interface:
32  * Use update(..) to put new data into the sliding window and get the latest normalized output
33  * Use flush to get normalized output without any input
34  */
35 class Normalization {
36 public:
37  typedef f32 Value;
38  typedef f64 Sum;
39  typedef Flow::DataPtr<Flow::Vector<Value>> Frame;
40 
41 protected:
42  SlidingWindow<Frame> slidingWindow_;
43  u32 length_;
44  u32 right_;
45 
46  Sum sumWeight_;
47 
48  bool changed_;
49 
50 private:
51  void normalize(Frame& out);
52 
53 protected:
54  virtual bool init(size_t dimension) {
55  return true;
56  }
57  virtual void updateStatistics(const Frame& add, const Frame& remove);
58  virtual void finalize();
59  virtual void apply(Frame& out) = 0;
60 
61 public:
62  Normalization();
63  virtual ~Normalization() {}
64 
65  /** init:
66  * @param length is length of sliding window.
67  * @param right is the output point calculated from the right side of the sliding window.
68  * (Note: data flows into the sliding window from right to left.)
69  * @param dimension is dimension of the input vectors.
70  */
71  bool init(size_t length, size_t right, size_t dimension);
72 
73  /**
74  * Puts in into the sliding window and
75  * normalizes the output vector if any.
76  * @return is true if out is valid
77  */
78  bool update(const Frame& in, Frame& out);
79  /**
80  * Normalizes the next remaining output vector if any.
81  * @return is true if out is valid
82  */
83  bool flush(Frame& out);
84  virtual void reset();
85 };
