38 Operation::AutomatonRef Minimize::process() {
39  uint32 encodeFlags = 0;
40  if (paramEncodeLabels(Operation::config)) {
41  log("using encoded labels");
42  encodeFlags |= FstLib::kEncodeLabels;
43  }
44  if (paramEncodeWeights(Operation::config)) {
45  log("using encoded weights");
46  encodeFlags |= FstLib::kEncodeWeights;
47  }
48  log("minimizing");
49  if (semiring() == tropicalSemiring) {
50  log("using tropical semiring");
51  minimize(input_, encodeFlags);
52  }
53  else {
54  log("using log semiring");
55  OpenFst::LogVectorFst* l = new OpenFst::LogVectorFst();
56  FstLib::Cast(*input_, l);
57  input_->DeleteStates();
58  minimize(l, encodeFlags);
59  FstLib::Cast(*l, input_);
60  delete l;
61  }
62  return input_;
63 }

67  typedef FstLib::EncodeMapper<A> Mapper;
68  Mapper mapper(encodeFlags, FstLib::ENCODE);

236 void Relabel::getLabelMapping(const std::vector<std::string>& labels,
237  const OpenFst::SymbolTable* symbols,
238  LabelMapping* mapping) const {
239  verify(!(labels.size() % 2));
240  for (std::vector<std::string>::const_iterator s = labels.begin(); s != labels.end(); s += 2) {
241  const std::string from = *s;
242  const std::string to = *(s + 1);
243  s32 fromId = -1, toId = -1;
244  if (symbols) {
245  fromId = symbols->Find(from);
246  toId = symbols->Find(to);
247  }
248  if (fromId < 0) {
249  Core::strconv(from, fromId);
250  warning("interpreting '%s' as '%d'", from.c_str(), fromId);
251  }
252  if (toId < 0) {
253  Core::strconv(to, toId);
254  warning("interpreting '%s' as '%d'", to.c_str(), toId);
255  }
256  log("mapping %d to %d", fromId, toId);
257  mapping->push_back(LabelPair(fromId, toId));
258  }
259 }

495 Operation::AutomatonRef ScaleLabelWeights::process() {
496  if (!input_->OutputSymbols()) {
497  error("symbol table required");
498  return input_;
499  }
500  const std::string symbol = paramLabel(config);
501  OpenFst::Label label = input_->OutputSymbols()->Find(symbol);
502  log("using label '%s' = %d", symbol.c_str(), label);
503  f32 scale = paramScale(config);
504  log("using scale %f", scale);
505  u32 nModified = 0;
506  for (OpenFst::StateIterator siter(*input_); !siter.Done(); siter.Next()) {
507  OpenFst::StateId state = siter.Value();
508  for (OpenFst::MutableArcIterator aiter(input_, state); !aiter.Done(); aiter.Next()) {
509  if (aiter.Value().olabel == label) {
510  OpenFst::Arc arc = aiter.Value();
511  arc.weight = OpenFst::Weight(arc.weight.Value() * scale);
512  aiter.SetValue(arc);
513  ++nModified;
514  }
515  }
516  }
517  log("modified %d arcs", nModified);
518  return input_;
519 }
