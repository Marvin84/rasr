68 Score FFNeuralNetworkLanguageModel::score(History const& hist, Token w) const {
69  ScoreCache const* sc = reinterpret_cast<ScoreCache const*>(hist.handle());
70  size_t output_idx = lexicon_mapping_[w->id()];
71  useOutput(*sc, output_idx);
72  if (not sc->scores.empty()) {
73  return sc->scores[output_idx];
74  }
75 
76  std::vector<ScoreCache*> caches;
77  if (eager_forwarding_) {
78  NNHistoryManager* hm = dynamic_cast<NNHistoryManager*>(historyManager_);
79  NNHistoryManager::NNCacheMap const& cm = hm->getNNCacheMap();
80  for (auto const& kv : cm) {
81  ScoreCache* c = const_cast<ScoreCache*>(static_cast<ScoreCache const*>(kv.second));
82  if (c->scores.empty()) {
83  caches.push_back(c);
84  }
85  }
86  }
87  else {
88  caches.push_back(const_cast<ScoreCache*>(sc));
89  }
90 
91  size_t vec_size = expand_one_hot_ ? context_size_ * num_outputs_ : context_size_;
92  Nn::Types<f32>::NnMatrix input(vec_size, caches.size());
93  input.setToZero();
94  for (size_t c = 0ul; c < caches.size(); c++) {
95  for (size_t t = 0ul; t < context_size_; t++) {
96  if (expand_one_hot_) {
97  input.at(t * num_outputs_ + caches[c]->history->at(t), c) = 1.0f;
98  }
99  else {
100  input.at(t, c) = caches[c]->history->at(t);
101  }
102  }
103  }
104  nn_.forward(input);
105 
106  Nn::Types<f32>::NnMatrix& gpu_output = nn_.getTopLayerOutput();
107  ;
108  gpu_output.finishComputation();
109  Math::FastMatrix<f32>& cpu_output = gpu_output.asWritableCpuMatrix();
110  require_eq(cpu_output.nRows(), num_outputs_);
111 
112  for (size_t c = 0ul; c < caches.size(); c++) {
113  cpu_output.getColumn(c, caches[c]->scores);
114  for (f32& s : caches[c]->scores) {
115  s = -std::log(s);
116  }
117  }
118  gpu_output.initComputation();
119  require_eq(sc->scores.size(), num_outputs_);
120 
121  return sc->scores[output_idx];
122 }
