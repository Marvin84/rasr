35 /** LatticeSetProcessor is base class for algorithms processing lattice sets.
36  * Remark:
37  * instance of this class can be used for dry runs since it does not have
38  * any abstract functions.

42 private:

44 
45 protected:

53  /** This lattice set processor is used for cascading.

56  verify(!processor_);

61  /** Override this function to sign on to services of the corpus visitor.
62  * Note: call the signOn function of your predecessor.
63  */
64  virtual void signOn(CorpusVisitor& corpusVisitor) {

67  }

69  /** Override this function to perform pre processing before the first segment of the corpus.

111  virtual void enterSpeechSegment(Bliss::SpeechSegment* s) {

113  processor_->enterSpeechSegment(s);
114  }
115  /** Override this function to perform post processing after the last features of the segment.
116  */
117  virtual void leaveSpeechSegment(Bliss::SpeechSegment* s) {
118  if (processor_)

120  }

122  /** Override this function to implement the processing of lattice sets.
123  */
124  virtual void processWordLattice(Lattice::ConstWordLatticeRef, Bliss::SpeechSegment*);
125  /**

134 
135  /**
