30 /** normalizeCrossCorrelationEstimate: normalizes cross-correlation estimate by number of addands
31  * Leads to the so called unbiased estimate of the cross-correaltion function
32  */
33 template<class Argument>
34 class normalizeCrossCorrelationEstimate : std::binary_function<Argument, s32, Argument> {
35  s32 sizeX_;
36  s32 sizeY_;
37 
38 public:
39  normalizeCrossCorrelationEstimate(const std::vector<Argument>& x, const std::vector<Argument>& y)
40  : sizeX_((s32)x.size()), sizeY_((s32)y.size()) {}
41 
42  Argument operator()(Argument element, s32 m) {
43  s32 N = std::min(sizeX_ - m, sizeY_) - std::max(0, -m);
44  if (N > 0)
45  return element / N;
46  return 0;
47  }

50 /** normalizeCrossCorrelationUpperBound: normalizes upper bound of cross-correlation to 1
51  * |Rxy(m)| / sqrt(Rxx(0)Ryy(0)) <= 1
52  * With other words, the two sequences x, y are normalized such that the zeroth lag
53  * of their autocorrelation is 1.
54  */
55 template<class Argument>
56 class normalizeCrossCorrelationUpperBound : std::binary_function<Argument, s32, Argument> {
57  Argument upperBound_;
58 
59 public:
60  normalizeCrossCorrelationUpperBound(const std::vector<Argument>& x, const std::vector<Argument>& y) {
61  upperBound_ = sqrt(std::inner_product(x.begin(), x.end(), x.begin(), 0.0) *
62  std::inner_product(y.begin(), y.end(), y.begin(), 0.0));
63  }
64  Argument operator()(Argument element, s32) {
65  return element / upperBound_;
66  }
67 };

89  * 0 <= Dxy(m) / [sum( abs(x(t)) ) + sum( abs(y(t)) )] <= 1
90  */
91 template<class Argument>

94 
95 public:

97  const std::vector<Argument>& y) {

106  }

109 /** CrossCorrelation
110  * Rxy(m) = sum_{t = max(0, -m)}^{min(sizeX_ - m, sizeY_)} similarity(x_{t + m}, y_{t})
111  *
112  * where:
113  * "similarty" can be:
114  * multiplication: x_{t + m} * y_{t}
115  * absolute difference: |x_{t + m} - y_{t}|^power
116  *
117  * m is element of [begin..end); begin and end are give in continuous units (e.g.: sec, Hz)
118  *
119  * normalitions:
120  * Rxy(m) can be normalized by
121  * - number of addands: leads to unbiased estimate of cross-correlation
122  * - upper bound: leads to |Rxy(t)| <= 1
123  *
124  * application of fast Fourier Transform:
125  * can be applied if similarity function is the multiplication.
126  * controlled over the parameter useFastFourierTransform
127  */

158  /** CorssCorrelation: corss-correlation implemented in time domain
159  * @param product is a binary functor implementing an arbitrary similarity function
160  *
161  * remark:
162  * This implementation compares only the overlapping part of the shifted x and y.
163  * It does not extend any of the signals by zeros. This does not make any difference
164  * if applying 'multiplication' as similarity function but it do make a significant
165  * difference for absolute-difference kind of similarity functions.
166  */
167  template<class OperationProduct>
168  void crossCorrelation(const std::vector<Data>& x, const std::vector<Data>& y,
169  std::vector<Data>& Rxy, OperationProduct product) const {
170  // interval [begin..end) too broad, x and y do not overlap
171  ensure(begin_ > -((s32)y.size()) && end_ <= (s32)x.size());
172 
173  Rxy.resize(end_ - begin_);
174  std::fill(Rxy.begin(), Rxy.end(), 0);
175 
176  for (s32 m = begin_; m < end_; ++m) {
177  s32 begin = std::max(0, -m);
178  s32 end = std::min((s32)x.size() - m, (s32)y.size());
179 
180  for (s32 n = begin; n < end; ++n)
181  Rxy[m - begin_] += product(x[n + m], y[n]);
182  }
183  }
