27 /** MeanEnergyVectorNormalization: divides each element by square of frame mean energy
28  *
29  * square of frame mean energy = ( ( sum_{i=0}^{N-1} (x[i]^2) ) / N ) ^ {1/2}
30  * where N is the number of samples in frame
31  *
32  * Applied normally to amplitude like values.
33  */

42  }

77 /** AmplitudeSpectrumEnergyVectorNormalization: divides each element by square of frame energy
78  * calculated form (half) amplitude spectrum
79  *
80  * square of frame energy =
81  * ( 1 / N * (2 * sum_{omega=1}^{N/2 - 1} (x^2(omega)) + x^2(0) + x^2(N/2)) ) ^{1/2}
82  * where N is length of FFT
83  *
84  *
85  * Applied normally to amplitude like values.
86  */
87 
88 template<class T>
89 class AmplitudeSpectrumEnergyVectorNormalization {
90 public:
91  typedef T Value;
92 
93  static std::string name() {
94  return Core::Type<Value>::name + std::string("-amplitude-spectrum-energy");
95  }
96 
97  void operator()(std::vector<Value>& v) {
98  hope(!v.empty());
99 
100  Value squareEnergy = sqrt((v.front() * v.front() + v.back() * v.back() +
101  2 * std::inner_product(v.begin() + 1, v.end() - 1, v.begin() + 1, 0.0)) /
102  T((v.size() - 1) * 2));
103 
104  std::transform(v.begin(), v.end(), v.begin(),
105  std::bind2nd(std::multiplies<T>(), (Value)1 / squareEnergy));
106  }
107 };

130 template<class T>
131 class VarianceVectorNormalization {
132 public:
133  typedef T Value;
134 
135  static std::string name() {
136  return Core::Type<Value>::name + std::string("-variance");
137  }
138 
139  void operator()(std::vector<Value>& v) {
140  Value sum = std::accumulate(v.begin(), v.end(), 0.0);
141  Value sumSquare = std::inner_product(v.begin(), v.end(), v.begin(), 0.0);
142 
143  Value mean = sum / v.size();
144  Value deviation = sqrt((sumSquare - sum * sum / v.size()) / v.size());
145 
146  std::transform(v.begin(), v.end(), v.begin(),
147  std::bind2nd(std::plus<T>(), -mean));
148 
149  std::transform(v.begin(), v.end(), v.begin(),
150  std::bind2nd(std::multiplies<T>(), (Value)1 / deviation));
151  }
152 };
