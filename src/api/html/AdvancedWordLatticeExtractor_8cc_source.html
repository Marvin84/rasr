46 void TimeConditionedLatticeSetProcessor::setWordLatticeDescription(const Lattice::WordLatticeDescription& description) {
47  for (size_t i = 0; i < description.nStreams(); ++i) {
48  bool isAcousticFsa = description[i].verifyValue(Lattice::WordLatticeDescription::nameModel, std::string(Lattice::WordLattice::acousticFsa));
49  bool isWordLevel = description[i].verifyValue(Lattice::WordLatticeDescription::nameModel, std::string(Lattice::WordLattice::acousticFsa));
50  if (isAcousticFsa and isWordLevel) {
51  return;
52  }
53  }
54  criticalError("Input lattice does not have an acoustic fsa at word level");
55 }

78 void NumeratorLatticeGenerator::leaveSpeechSegment(Bliss::SpeechSegment* s) {
79  verify(recognizer_);
80  verify(orthToLemma_);
81  Core::Ref<Lattice::WordLattice> orth(new Lattice::WordLattice);
82  orth->setFsa(orthToLemma_->createLemmaAcceptor(s->orth()), Lattice::WordLattice::acousticFsa);
83  Lattice::ConstWordLatticeRef lattice = recognizer_->extract(orth, s);
84  if (lattice and lattice->nParts() == 1) {
85  processWordLattice(lattice, s);

87  else {
88  log("Skip this segment because numerator lattice could not be generated.");
