38 template<typename T>
39 PythonLayer<T>::PythonLayer(const Core::Configuration& config)
40  : Core::Component(config), Precursor(config), timeForward_(0), timeBackward_(0) {
41  pythonInitializer_.init();
42 
43  // Get us the CPython GIL. However, when we return here,
44  // it will get released and other Python threads can run.
45  Python::ScopedGIL gil;
46 
47  std::string pyModPath(paramPyModPath(config));
48  if (!pyModPath.empty())
49  Python::addSysPath(pyModPath);
50 
51  std::string pyModName(paramPyModName(config));
52  if (pyModName.empty()) {
53  pythonCriticalError("PythonLayer: need Python module name (pymod-name)");
54  return;
55  }
56 
57  Python::ObjRef pyMod;
58  pyMod.takeOver(PyImport_ImportModule(pyModName.c_str()));
59  if (!pyMod) {
60  pythonCriticalError("PythonLayer: cannot import module '%s'", pyModName.c_str());
61  return;
62  }
63 
64  std::string pyConfig(paramPyModConfig(config));
65  pyObject_.takeOver(Python::PyCallKw(pyMod, "SprintNnPythonLayer", "{s:s}", "config", pyConfig.c_str()));
66  if (!pyObject_) {
67  pythonCriticalError("PythonLayer: failed to call SprintNnPythonLayer");
68  return;
69  }
70 
71  // The output dimension is usually not changed anymore and this function would never be called otherwise.
72  setOutputDimension(Precursor::outputDimension_);
73  // Sprint will usually calculate dynamically all the input dimensions and then call setInputDimensions(),
74  // followed by initializeNetworkParameters().
75 }

172 template<typename T>
173 void PythonLayer<T>::forward(const std::vector<NnMatrix*>& input, NnMatrix& output) {
174  timeval start, end;
175  TIMER_START(start);
176  Python::ScopedGIL gil;
177  Python::ObjRef input_ls;
178  input_ls.takeOver(PyList_New(input.size()));
179  for (size_t i = 0; i < input.size(); ++i) {
180  input[i]->finishComputation(true);
181  PyObject* np_array = NULL;
182  if (!Python::nnMatrix2numpy(getPythonCriticalErrorFunc(), np_array, *input[i]))
183  return;
184  PyList_SetItem(input_ls, i, np_array); // overtake ref
185  input[i]->initComputation(false);
186  }
187  Python::ObjRef res;
188  res.takeOver(Python::PyCallKw(pyObject_, "forward", "{s:O}", "input", input_ls.obj));
189  if (!res) {
190  pythonCriticalError("PythonLayer: exception occured while calling 'forward'");
191  return;
192  }
193  output.finishComputation(false);
194  if (!Python::numpy2nnMatrix(getPythonCriticalErrorFunc(), res, output))
195  return;
196  output.initComputation(true);
197  TIMER_STOP(start, end, timeForward_);
198 }

200 template<typename T>
201 void PythonLayer<T>::backpropagateActivations(const NnMatrix& errorSignalIn, NnMatrix& errorSignalOut, const NnMatrix& activations) {
202  // First, this is called, then backpropagateWeights.
203  // Here we usually have &errorSignalIn == &errorSignalOut,
204  // i.e. we are expected to handle that inplace,
205  // and errorSignalOut still has our output-dimensions.
206  // Note that in ase this is the lowest trainable layer,
207  // backpropagateWeights might not be called anymore.
208  // Thus, we have to catch the errorSignalIn always at this point.
209  timeval start, end;
210  TIMER_START(start);
211  Python::ScopedGIL gil;
212  Python::ObjRef in;
213  errorSignalIn.finishComputation(true);
214  if (!Python::nnMatrix2numpy(getPythonCriticalErrorFunc(), in.obj, errorSignalIn))
215  return;
216  errorSignalIn.initComputation(false);
217  backpropRes_.takeOver(Python::PyCallKw(pyObject_, "backpropagate", "{s:O}", "errorSignalIn", in.obj));
218  if (!backpropRes_) {
219  pythonCriticalError("PythonLayer: exception occured while calling 'backpropagate'");
220  return;
221  }
222  if (!PyTuple_Check(backpropRes_)) {
223  pythonCriticalError("PythonLayer: 'backpropagate' did not return a tuple");
224  return;
225  }
226  TIMER_STOP(start, end, timeBackward_);
227 }

229 template<typename T>
230 void PythonLayer<T>::backpropagateWeights(const NnMatrix& errorSignalIn, std::vector<NnMatrix*>& errorSignalOut) {
231  // backpropagateActivations was called before. See comment there.
232  timeval start, end;
233  TIMER_START(start);
234  Python::ScopedGIL gil;
235  if ((size_t)PyTuple_Size(backpropRes_) != errorSignalOut.size()) {
236  pythonCriticalError("PythonLayer: 'backpropagate' returned %zd items but we expected %zu items",
237  PyTuple_Size(backpropRes_), errorSignalOut.size());
238  return;
239  }
240  for (size_t i = 0; i < errorSignalOut.size(); ++i) {
241  require(errorSignalOut[i]);
242  errorSignalOut[i]->finishComputation(false);
243  PyObject* np_array = PyTuple_GetItem(backpropRes_, i); // borrowed
244  if (!Python::numpy2nnMatrix(getPythonCriticalErrorFunc(), np_array, *errorSignalOut[i]))
245  return;
246  errorSignalOut[i]->initComputation(true);
247  }
248  backpropRes_.clear();
249  TIMER_STOP(start, end, timeBackward_);
250 }
