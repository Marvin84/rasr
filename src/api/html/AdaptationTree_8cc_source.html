48 void AdaptationTree::loadDecisionTree(const Core::Configuration& config,
49  const Am::ClassicStateModelRef stateModel,
50  Bliss::Phoneme::Id silencePhoneme) {
51  Am::DecisionTreeStateTying cartStateTying(config, stateModel);
52 
53  const Am::Allophone* siAllo = stateModel->allophoneAlphabet().allophone(Am::Allophone(silencePhoneme,
54  Am::Allophone::isInitialPhone | Am::Allophone::isFinalPhone));
55  Am::AllophoneState siAlloState = stateModel->allophoneStateAlphabet().allophoneState(siAllo, 0);
56  Mm::MixtureIndex siMixtureId = cartStateTying.classify(siAlloState);
57  silenceMixtures_.insert(siMixtureId);
58 
59  // Build tree structure as intermediate representation for getting a Core::BinaryTree
60  // deprecated: use directly Cart::DecisionTree instead of Core::BinaryTree
61  // deprecated: don't separate explicitly silence, rather verify that silence is in the pruned tree
62  Core::BinaryTree::TreeStructure structure;
63  {
64  const Cart::BinaryTree::Node& root = cartStateTying.decisionTree().root();
65  /**
66  * First question separates silence,
67  * i.e. the decision tree must have been built in the following way:
68  * the first question asks the central phoneme for being the (unique, single-state) silence phoneme,
69  * the left node (answer=YES) is a leaf, while the right node is splitted further
70  */
71  verify(root.leftChild().isLeaf() && (root.leftChild().id() == siMixtureId));
72  verify((root.leftChild().info<Cart::TrainingInformation>().order == 1) && (root.rightChild().info<Cart::TrainingInformation>().order == 2));
73  std::stack<const Cart::BinaryTree::Node*> S;
74  S.push(&root.rightChild());
75  while (!S.empty()) {
76  const Cart::BinaryTree::Node* node = S.top();
77  S.pop();
78  u32 order = node->info<Cart::TrainingInformation>().order - 2;
79  if (node->isLeaf()) {
80  Mm::MixtureIndex leafNumber = node->id_;
81  if (leafNumber > siMixtureId)
82  --leafNumber;
83  structure.push_back(Core::BinaryTree::TreeStructureEntry(order, leafNumber));
84  }
85  else {
86  structure.push_back(Core::BinaryTree::TreeStructureEntry(order, Core::BinaryTree::invalidId));
87  S.push(&node->rightChild());
88  S.push(&node->leftChild());
89  }
90  }
91  }
92  tree_ = Core::ref(new Core::BinaryTree(structure));
93  ensure(tree_);
94  leafIndex_ = tree_->prune(numberOfBaseClasses_);
95 
96  /*! \todo change if silence is handled correctly by dectree */
97  (*leafIndex_).push_back(numberOfBaseClasses_);
98  nLeafs_ = numberOfBaseClasses_ + 1;
99  //end of part to be changed
100 }
