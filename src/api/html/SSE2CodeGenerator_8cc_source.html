58 bool SSE2L2NormCodeGenerator::appendFetchA(int block, u32 offset) {
59  /* x86:
60  block 0:
61  movdqu xmm0, [esi+offset] ; copy 128 bits from mean vector in xmm0
62 
63  block 1:
64  movdqu xmm4, [esi+offset] ; copy 128 bits from mean vector in xmm0
65  */
66  /* x86-64:
67  block 0:
68  movdqu xmm0, [rdi+offset]
69 
70  block 1:
71  movdqu xmm4, [rdi+offset]
72  */
73  static const u8 code[3] = {0xf3, 0x0f, 0x6f};
74  bool ok = append(code, sizeof(code));
75 
76  if (offset == 0) {
77  // without offset
78 #if defined(PROC_x86_64)
79  static const u8 modrm[2][1] = {{0x07}, {0x27}};
80 #else
81  static const u8 modrm[2][1] = {{0x06}, {0x26}};
82 #endif
83  ok = append(modrm[block], 1) && ok;
84  }
85  else {
86  // with offset
87  if (offset >= 128) {
88  // 32-bit displacement
89 #if defined(PROC_x86_64)
90  static const u8 modrm[2][1] = {{0x87}, {0xa7}};
91 #else
92  static const u8 modrm[2][1] = {{0x86}, {0xa6}};
93 #endif
94  ok = append(modrm[block], 1) && appendDWordOffset(offset) && ok;
95  }
96  else {
97  // 8bit displacement
98 #if defined(PROC_x86_64)
99  static const u8 modrm[2][1] = {{0x47}, {0x67}};
100 #else
101  static const u8 modrm[2][1] = {{0x46}, {0x66}};
102 #endif
103  u8 byteOffset = offset;
104  ok = append(modrm[block], 1) && append(&byteOffset, 1) && ok;
105  }
106  }
107  return ok;
108 }

216  /* block = 0

324 SSE2L2NormCodeGenerator::SSE2L2NormCodeGenerator(const Core::Configuration& c, size_t d)
325  : CodeGenerator(c) {
326  const u8 blockSize = 16;
327  log("using SSE2 instructions for SIMD-diagonal-maximum feature scorer");
328  appendProlog();
329 
330  /* first block */
331  u32 offset = 0;
332 
333  appendFetchA(0, offset);
334  appendClearAccumulator();
335  appendFetchB(0, offset);
336  offset += blockSize;
337  appendDuplicate(0);
338  if (d > blockSize)
339  appendFetchA(1, offset);
340  appendSubtract(0);
341  if (d > blockSize) {
342  appendFetchB(1, offset);
343  offset += blockSize;
344  }
345  appendPrepareForUnpack(0);
346  if (d > blockSize)
347  appendDuplicate(1);
348  appendUnpackAndAdd(0);
349 
350  /* in between blocks */
351  size_t block = 0;
352  for (; offset < d; offset += blockSize, block = 1 - block) {
353  appendSubtract(1 - block);
354  appendFetchA(block, offset);
355  appendAccumulate(block);
356  appendFetchB(block, offset);
357  appendPrepareForUnpack(1 - block);
358  appendDuplicate(block);
359  appendUnpackAndAdd(1 - block);
360  }
361 
362  /* last block */
363  if (d > blockSize)
364  appendSubtract(1 - block);
365  appendAccumulate(block);
366  if (d > blockSize) {
367  appendPrepareForUnpack(1 - block);
368  appendUnpackAndAdd(1 - block);
369  appendAccumulate(1 - block);
370  }
371 
372  appendEpilog();
373  finalize();
374 }
