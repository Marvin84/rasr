253 void CorpusLexicalStatisticsVisitor::Internal::newEdge(OrthographicParser::Node from, OrthographicParser::Node to, const Lemma* lemma) {
254  if (lemma) {
255  lemmaCounts_[lemma] += 1;
256  nLemmas_ += 1;
257  nPronunciations_ += lemma->nPronunciations();
258  nEvaluationSequences_ += lemma->nEvaluationTokenSequences();
259  u32 nEvaluationTokensInLemma_ = 0;
260  Lemma::EvaluationTokenSequenceIterator e, e_end;
261  for (Core::tie(e, e_end) = lemma->evaluationTokenSequences(); e != e_end; ++e)
262  nEvaluationTokensInLemma_ += e->length();
263  nEvaluationTokens_ += f64(nEvaluationTokensInLemma_) / f64(lemma->nEvaluationTokenSequences());
264  if (from != to) {
265  nNonLoopLemmas_ += 1;
266  nNonLoopPronunciations_ += lemma->nPronunciations();
267  nNonLoopEvaluationSequences_ += lemma->nEvaluationTokenSequences();
268  nNonLoopEvaluationTokens_ += f64(nEvaluationTokensInLemma_) / f64(lemma->nEvaluationTokenSequences());
269  }
270 
271  if (lemma->nPronunciations()) {
272  Lemma::PronunciationIterator pron, pron_end;
273  u32 nPhonemesTotal = 0;
274  for (Core::tie(pron, pron_end) = lemma->pronunciations(); pron != pron_end; ++pron)
275  nPhonemesTotal += pron->pronunciation()->length();
276  nPhonemes_ += f64(nPhonemesTotal) / f64(lemma->nPronunciations());
277  }
278  }
279 
280  if (drawChannel_.isOpen()) {
281  drawChannel_ << "n" << from << " -> "
282  << "n" << to;
283  if (lemma)
284  drawChannel_ << " [label=\"" << lemma->name() << "\"]";
285  drawChannel_ << std::endl;
286  }
287 }

319 void CorpusLexicalStatisticsVisitor::writeReport(Core::XmlWriter& os) const {
320  os << Core::XmlOpen("lexical-statistics");
321 
322  internal_->writeReport(os);
323 
324  os << Core::XmlComment("occurances of known words");
325  f64 nWordMatchPositions = 0;
326  Lexicon::LemmaIterator l, l_end;
327  for (Core::tie(l, l_end) = lexicon_->lemmas(); l != l_end; ++l) {
328  const Lemma* lemma(*l);
329  if (internal_->lemmaCounts_.find(lemma) == internal_->lemmaCounts_.end())
330  continue;
331  const u32 count = const_cast<Internal::LemmaMap&>(internal_->lemmaCounts_)[lemma];
332  os << Core::XmlOpen("lemma") + Core::XmlAttribute("name", lemma->name()) + Core::XmlAttribute("count", count)
333  << Core::XmlFull("orth", lemma->preferredOrthographicForm())
334  << Core::XmlClose("lemma");
335  if (!internal_->isNonWordLemma(lemma)) {
336  nWordMatchPositions += count;
337  }
338  }
339 
340  os << (Core::XmlOpen("lemma") + Core::XmlAttribute("special", "unknown"))
341  << Core::XmlComment("occurances of unknown words");
342 
343  typedef std::vector<std::string> WordList;
344 
345  f64 nWordFailurePositions = 0;
346  WordList list;
347 
348  std::transform(internal_->unmatchedOrthCounts_.begin(),
349  internal_->unmatchedOrthCounts_.end(),
350  std::back_insert_iterator<WordList>(list),
351  Core::select1st<Internal::UnmatchedOrthMap::value_type>());
352  std::sort(list.begin(), list.end());
353  for (std::vector<std::string>::const_iterator i = list.begin(); i != list.end(); ++i) {
354  const u32 count = const_cast<Internal::UnmatchedOrthMap&>(internal_->unmatchedOrthCounts_)[*i];
355  os << Core::XmlFull("orth", *i) + Core::XmlAttribute("count", count);
356  nWordFailurePositions += count;
357  }
358  os << Core::XmlClose("lemma");
359 
360  os << Core::XmlFull("out-of-vocabulary-rate", (nWordFailurePositions / (nWordMatchPositions + nWordFailurePositions)) * 100);
361 
362  os << Core::XmlClose("lexical-statistics");
363 }
