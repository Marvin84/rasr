22 DP trapzd(const FunctorBase<DP, DP>& func, const DP a, const DP b, const int n) {
23  DP x, tnm, sum, del;
24  static DP s;
25  int it, j;
26 
27  if (n == 1) {
28  return (s = 0.5 * (b - a) * (func(a) + func(b)));
29  }
30  else {
31  for (it = 1, j = 1; j < n - 1; j++)
32  it <<= 1;
33  tnm = it;
34  del = (b - a) / tnm;
35  x = a + 0.5 * del;
36  for (sum = 0.0, j = 0; j < it; j++, x += del)
37  sum += func(x);
38  s = 0.5 * (s + (b - a) * sum / tnm);
39  return s;
40  }
41 }

43 void polint(Vec_I_DP& xa, Vec_I_DP& ya, const DP x, DP& y, DP& dy) {
44  int i, m, ns = 0;
45  DP den, dif, dift, ho, hp, w;
46 
47  int n = xa.size();
48  Vec_DP c(n), d(n);
49  dif = fabs(x - xa[0]);
50  for (i = 0; i < n; i++) {
51  if ((dift = fabs(x - xa[i])) < dif) {
52  ns = i;
53  dif = dift;
54  }
55  c[i] = ya[i];
56  d[i] = ya[i];
57  }
58  y = ya[ns--];
59  for (m = 1; m < n; m++) {
60  for (i = 0; i < n - m; i++) {
61  ho = xa[i] - x;
62  hp = xa[i + m] - x;
63  w = c[i + 1] - d[i];
64  if ((den = ho - hp) == 0.0)
65  nrerror("Error in routine polint");
66  den = w / den;
67  d[i] = hp * den;
68  c[i] = ho * den;
69  }
70  y += (dy = (2 * (ns + 1) < (n - m) ? c[ns + 1] : d[ns--]));
71  }
72 }

74 DP midpnt(const Nr::FunctorBase<DP, DP>& func, const DP a, const DP b, const int n) {
75  int it, j;
76  DP x, tnm, sum, del, ddel;
77  static DP s;
78 
79  if (n == 1) {
80  return (s = (b - a) * func(0.5 * (a + b)));
81  }
82  else {
83  for (it = 1, j = 1; j < n - 1; j++)
84  it *= 3;
85  tnm = it;
86  del = (b - a) / (3.0 * tnm);
87  ddel = del + del;
88  x = a + 0.5 * del;
89  sum = 0.0;
90  for (j = 0; j < it; j++) {
91  sum += func(x);
92  x += ddel;
93  sum += func(x);
94  x += del;
95  }
96  s = (s + (b - a) * sum / tnm) / 3.0;
97  return s;

226 void rkqs(Vec_IO_DP& y, Vec_IO_DP& dydx, DP& x, const DP htry,
227  const DP eps, Vec_I_DP& yscal, DP& hdid, DP& hnext,
228  Nr::DerivativesBase<DP, Vec_I_DP, Vec_O_DP>& derivs) {
229  const DP SAFETY = 0.9, PGROW = -0.2, PSHRNK = -0.25, ERRCON = 1.89e-4;
230  int i;
231  DP errmax, h, htemp, xnew;
232 
233  int n = y.size();
234  h = htry;
235  Vec_DP yerr(n), ytemp(n);
236  for (;;) {
237  rkck(y, dydx, x, h, ytemp, yerr, derivs);
238  errmax = 0.0;
239  for (i = 0; i < n; i++)
240  errmax = std::max(errmax, fabs(yerr[i] / yscal[i]));
241  errmax /= eps;
242  if (errmax <= 1.0)
243  break;
244  htemp = SAFETY * h * pow(errmax, PSHRNK);
245  h = (h >= 0.0 ? std::max(htemp, 0.1 * h) : std::min(htemp, 0.1 * h));
246  xnew = x + h;
247  if (xnew == x)
248  nrerror("stepsize underflow in rkqs");
249  }
250  if (errmax > ERRCON)
251  hnext = SAFETY * h * pow(errmax, PGROW);
252  else
253  hnext = 5.0 * h;
254  x += (hdid = h);
255  for (i = 0; i < n; i++)
256  y[i] = ytemp[i];
257 }
