38  CacheAutomaton(_ConstAutomatonRef f, u32 maxAge)
39  : Precursor(f), maxAge_(maxAge), age_(0) {
40  this->setProperties(Fsa::PropertyCached | Fsa::PropertyStorage, Fsa::PropertyCached);
41  }
42  virtual ~CacheAutomaton() {
43  states_.erase(states_.begin(), states_.end());
44  }
45  void age() const {
46  if (++age_ > maxAge_) {
47  for (Core::Vector<u8>::iterator i = ages_.begin(); i != ages_.end(); ++i)
48  if (!(*i >>= 1))
49  states_[i - ages_.begin()].reset();
50  age_ = 0;
51  }
52  }
53  virtual _ConstStateRef getState(Fsa::StateId s) const {
54  age();
55  states_.grow(s);
56  if (!states_[s])
57  states_[s] = Precursor::fsa_->getState(s);
58  ages_.grow(s, 0);
59  ages_[s] |= 0x80;
60  return states_[s];
61  }
62  virtual size_t getMemoryUsed() const {
63  size_t size = 0;
64  for (typename Core::Vector<_ConstStateRef>::iterator i = states_.begin(); i != states_.end(); ++i)
65  if ((*i) && ((*i)->refCount() == 1))
66  size += (*i)->getMemoryUsed();
67  return Precursor::fsa_->getMemoryUsed() + 2 * sizeof(u32) + ages_.capacity() * sizeof(u8) + states_.capacity() * sizeof(_ConstStateRef) + size;
68  }
69  virtual std::string describe() const {
70  return Core::form("cache(%s,%d)", Precursor::fsa_->describe().c_str(), maxAge_);
71  }
72 };
