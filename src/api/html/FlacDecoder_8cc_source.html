20  ((FlacData*)clientData)->samplesRead) {
21  unsigned long int blockSamples = frame->header.blocksize;
22  unsigned long int channels = ((FlacData*)clientData)->channels;
23  unsigned long int blockPos = 0;
24 
25  while ((((FlacData*)clientData)->samplesToRead >
26  ((FlacData*)clientData)->samplesRead) &&

29  ((short int*)((FlacData*)clientData)->buffer)[(channels * (((FlacData*)clientData)->samplesRead)) + channel] = (short int)buffer[channel][blockPos];
30  }
31  blockPos++;
32  ((FlacData*)clientData)->samplesRead++;
33  }
34  }
