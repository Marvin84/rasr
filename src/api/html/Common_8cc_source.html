57 std::string cleanBiLangSymbol(const std::string& s) {
58  // change *EPS* to $
59  std::string result;
60  if (s == "*EPS*") {
61  result = "$";
62  }
63  else {
64  // remove word index from symbol
65  result = s.substr(0, s.rfind("#"));
66  // escape strange characters [+~|$]
67  // + -> ++
68  // _ -> +
69  // $ -> $$
70  // ~ -> ~~
71  // | -> ~
72  }
73  return result;
74 }

88 void writeAachen(ConstAutomatonRef f, std::ostream& o, size_t sentenceNumber, bool oneToOne, double threshold) {
89  Fsa::ConstStateRef currentState(f->getState(f->initialStateId()));
90  o << "SENT: " << sentenceNumber << std::endl;
91  LabelId iPrev(0);
92  LabelId jPrev(0); // There is an error here: if reordering is used. the previous j at sentence beginning might not be "0"

94  LabelId j(currentState->begin()->input());

96  bool isEpsilon(i == Epsilon || j == Epsilon);
97  double w(currentState->begin()->weight());
98 
99  if (!oneToOne) {
100  i = (i != Epsilon ? i : iPrev);
101  j = (j != Epsilon ? j : jPrev);
102  }
103 
104  if (!(oneToOne && isEpsilon) && !(threshold > 0 && w > threshold)) {
105  o << "S "

108  << std::endl;
109  }
