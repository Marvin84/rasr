172  void visit(const Cart::BinaryTree::Node* node, u32 lowestDepth, std::set<const Cart::BinaryTree::Node*>& visiting, std::vector<std::pair<u32, float>>& emissions) {
173  if (!node || visiting.count(node))
174  return;
175 
176  u32 depth = getDepth(node);
177 
178  if (depth < minDepth)
179  return;
180 
181  if (depth < lowestDepth)
182  lowestDepth = depth;
183 
184  visiting.insert(node);
185 
186  if (node->isLeaf()) {
187  verify(depth > lowestDepth);
188  float fac = pow(factor, depth - lowestDepth);
189  if (fac >= minFactor) {
190  verify(node->id_ < nodes.size() && nodes[node->id_] == node);
191  emissions.push_back(std::make_pair(node->id_, fac));
192  ++totalEmissions;
193  }
194  }
195 
196  visit(node->father_, lowestDepth, visiting, emissions);
197  visit(node->leftChild_, lowestDepth, visiting, emissions);
198  visit(node->rightChild_, lowestDepth, visiting, emissions);
199 
200  visiting.erase(node);
201  }

215  u32 removedEmissions = 0;

220  std::sort(tiedEmissions_[i].begin(), tiedEmissions_[i].end(),
221  Core::composeBinaryFunction(std::less<f32>(),
222  Core::select2nd<std::pair<u32, f32>>(),
223  Core::select2nd<std::pair<u32, f32>>()));
224 
225  if (tiedEmissions_[i].size() > maxEmissions_) {
226  removedEmissions += maxEmissions_ - tiedEmissions_[i].size();
227  tiedEmissions_[i].resize(maxEmissions_);
228  }
229  totalEmissions += tiedEmissions_[i].size();
230 
231  for (u32 e = 0; e < tiedEmissions_[i].size(); ++e)
232  totalWeight += tiedEmissions_[i][e].second;
233  }
234 
235  log() << "average number of tied emissions for each emission: " << float(totalEmissions / tiedEmissions_.size());
236  log() << "average number of removed tied emissions for each emission: " << float(removedEmissions / tiedEmissions_.size());
237  log() << "average tied emission weight: " << totalWeight / totalEmissions;
238  log() << "training data multiplication factor: " << ((totalWeight / totalEmissions) * tiedEmissions_.size()) + 1.0;
239  }
240  else {

242  criticalError("Change of features is not allowed.");
243  }
