24 /** MixtureSetTopologyBuilder creates a mixture set with independend densities.
25  */
26 class MixtureSetBuilder {
27 private:
28  MixtureIndex nMixtures_;
29  size_t nDensitiesPerMixture_;
30 
31 protected:
32  virtual DensityIndex densityIndex(MixtureIndex mixtureIndex, DensityIndex densityInMixture) {
33  return mixtureIndex * nDensitiesPerMixture_ + densityInMixture;
34  }
35  virtual MeanIndex meanIndex(MixtureIndex mixtureIndex, DensityIndex densityInMixture) {
36  return densityIndex(mixtureIndex, densityInMixture);
37  }
38  virtual CovarianceIndex covarianceIndex(MixtureIndex mixtureIndex, DensityIndex densityInMixture) {
39  return densityIndex(mixtureIndex, densityInMixture);
40  }
41 
42 public:
43  MixtureSetBuilder(MixtureIndex nMixtures, size_t nDensitiesPerMixture = 1);
44  virtual ~MixtureSetBuilder() {}
45  void build(MixtureSet& toBuild, ComponentIndex dimension = 0);
46 };
