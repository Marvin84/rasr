83 bool PrefixFilter::reachable(StateId state, const Bliss::Lemma* lemma) {
84  if (state < reachability_.size() && reachability_[state] != -1)
85  return (bool)reachability_[state];
86 
87  if (state >= reachability_.size())
88  reachability_.resize(state + 1, -1);
89 
90  reachability_[state] = 0;
91 
92  for (HMMStateNetwork::SuccessorIterator target = tree_.structure.successors(state); target; ++target) {
93  if (!target.isLabel()) {
94  if (reachable(*target, lemma)) {
95  reachability_[state] = 1;
96  return true;
97  }
98  }
99  else {
100  if (lexicon_->lemmaPronunciation(tree_.exits[target.label()].pronunciation)->lemma() == lemma) {
101  reachability_[state] = 1;
102  return true;
103  }
104  }
105  }
106  return false;
107 }

109 bool PrefixFilter::prune(const TraceManager &trace_manager, const StateHypothesis& hyp) const {
110  verify(hyp.trace != invalidTraceId);
111 
112  const Search::Trace& traceItem(*trace_manager.traceItem(hyp.trace).trace);
113  std::vector<Bliss::Lemma*> lemmaSequence(0);
114  traceItem.getLemmaSequence(lemmaSequence);
115 
116  s32 position = 0;
117  //Iterate over the lemma sequence of the current trace
118  for (int i = 0; i < lemmaSequence.size(); i++) {
119  //Skip non word lemmas
120  if (nonWordLemmas_.count(lemmaSequence[i])) {
121  continue;
122  }
123  if (position < prefixSequence_.size()) {
124  //Trace is invalid if lemma sequence and prefix sequence do not match
125  if (lemmaSequence[i] != prefixSequence_[position] && prefixSequence_[position] != 0) {
126  return true;
127  }
128  else {
129  position++;
130  }
131  }
132  else {
133  //Stop if end of prefix sequence is reached
134  return false;
135  }
136  }
137 
138  if (position >= prefixSequence_.size())
139  return false;
140 
141  //Check reachability of non word lemmas
142  if (nonWordLemmaNodes_.count(hyp.state)) {
143  return false;
144  }
145 
146  verify(position < prefixReachability_.size());
147 
148  return !(bool)prefixReachability_[position].count(hyp.state);
149 }
