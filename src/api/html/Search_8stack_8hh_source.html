101  template<class Scorer>
102  Score expand(Slice& next, const Scorer& scorer, Time time, Core::Vector<StateId>& expandedTokens) {
103  Score minScore = Core::Type<Score>::max, minThreshold = Core::Type<Score>::max;
104  next.bestFinalTrace_ = InvalidStateId;
105  Score minFinalScore = Core::Type<Score>::max;
106  while (!tokens_.isEmpty()) {
107  Token t = tokens_.pop();
108  const State* sp = fsa_->fastState(t.state_);
109  for (State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
110  Score score = t.score_ + Score(a->weight());
111  if (a->input() != inputEpsilon_)
112  score += scorer->score(a->input());
113  Trace trace = t.trace_;
114  if (a->output() != outputEpsilon_) {
115  Score diffScore = score - Score(trace_->fastState(trace)->weight_);
116  State* traceState = trace_->newState();
117  traceState->weight_ = Weight(score); // temporary potential
118  traceState->newArc(trace, Weight(diffScore), a->output());
119  trace = traceState->id();
120  }
121  if (fsa_->fastState(a->target())->isFinal()) {
122  Score finalScore = score + Score(fsa_->fastState(a->target())->weight_);
123  if (finalScore < minFinalScore) {
124  if (finalScore < minFinalScore) {
125  minFinalScore = finalScore;
126  next.bestFinalTrace_ = trace;
127  }
128  State* traceState = trace_->fastState(trace);
129  traceState->weight_ = Weight(score);
130  Score diffScore = score - Score(traceState->weight_);
131  for (State::iterator a = traceState->begin(); a != traceState->end(); ++a)
132  a->weight_ = Weight(Score(a->weight_) + diffScore);
133  }
134  }
135  if (a->input_ != inputEpsilon_) {
136  if (score < minThreshold) {
137  if (score < minScore) {
138  minScore = score;
139  minThreshold = minScore + threshold_;
140  }
141  if (score < expandedTokens_[a->target()].score_) {
142  if (expandedTokens_[a->target()].score_ >= Core::Type<Score>::max)
143  expandedTokens.push_back(a->target());
144  expandedTokens_[a->target()].score_ = score;
145  expandedTokens_[a->target()].trace_ = trace;
146  }
147  }
148  }
149  else
150  tokens_.push(Token(a->target(), score, trace));
151  }
152  }
153  return minThreshold;
154  }

156  void prune(Slice& next, const Core::Vector<StateId>& expandedTokens, Score minThreshold) {
157  next.statesBeforePruning_ = statesBeforePruning_;
158  next.statesBeforePruning_ += expandedTokens.size();
159  next.tokens_.clear();
160  for (Vector<StateId>::const_iterator i = expandedTokens.begin(); i != expandedTokens.end(); ++i) {
161  if (expandedTokens_[*i].score_ < minThreshold)
162  next.tokens_.push(Token(*i, expandedTokens_[*i].score_, expandedTokens_[*i].trace_));
163  expandedTokens_[*i].score_ = Core::Type<Score>::max;
164  }
165  next.statesAfterPruning_ = statesAfterPruning_;
166  next.statesAfterPruning_ += next.tokens_.size();
167  }
