92 void StatePotentials64DfsState::finishState(StateId s) {
93  // remember that fsa_ contains the transposed fsa
94  ConstStateRef sp = fsa_->getState(s);
95  ConstSemiringRef sr = fsa_->semiring();
96  potentials_.grow(s, Zero);
97 
98  // We want to compute \sum_{a \in sp->args} pot_{a->target} + a->weight
99  // (and eventually + sp->weight, which we leave out for simplicity in the following calc comment).
100  // Calculation is in negative log-space, i.e. p -> -ln p (for numerical stability).
101  // Thus, we have to compute -ln(\sum_a exp( -pot_{a->target} + -a->weight )) .
102  // Also for numeric stability, we actually compute
103  // -ln(\sum_a exp( -pot_{a->target} + -a->weight ))
104  // = -ln(exp(-min) * \sum_a exp( min + -pot_{a->target} + -a->weight ))
105  // = -ln(exp(-min) * (1 + \sum_{a \neq a_min} exp( min + -pot_{a->target} + -a->weight )))
106  // = min - ln(1 + \sum_{a \neq a_min} exp( min - pot_{a->target} - a->weight )),
107  // where min = pot_{(a_min)->target} + (a_min)->weight.
108  // For ln(1+arg), we can use log1p, which is more stable for arg~=0.
109  f64 minimumScore = sp->isFinal() ? f32(sp->weight_) : Zero;
110  State::const_iterator minimumIterator = sp->end();
111  for (State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
112  f64 score = potentials_[a->target()] + f32(a->weight());
113  if (std::isinf(score))
114  score = Core::Type<f64>::max;
115  if (score <= minimumScore) {
116  minimumScore = score;
117  minimumIterator = a;
118  }
119  }
120  f64 sum = (sp->isFinal() && minimumIterator != sp->end()) ? exp(minimumScore - f32(sp->weight_)) : 0;
121  for (State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
122  if (a != minimumIterator) {
123  f64 score = potentials_[a->target()] + f32(a->weight());
124  if (std::isinf(score))
125  score = Core::Type<f64>::max;
126  sum += exp(minimumScore - score);
127  }
128  }
129  potentials_[s] = minimumScore - log1p(sum);
130  potentials_[s] = potentials_[s] < Zero ? potentials_[s] : Zero;
131 
132  if (s == fsa_->initialStateId()) {
133  totalFlow_ = potentials_[s];
134  }
135 }

251 void StatePotentialsEDfsState::finishState(StateId s) {
252  // remember that fsa_/risk_ contain the transposed fsa
253  ConstStateRef sp = fsa_->getState(s);
254  potentials_.grow(s, Zero);
255  f64 minWeight = sp->isFinal() ? f32(sp->weight_) : Zero;
256  for (State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
257  f64 weight = potentials_[a->target()] + f32(a->weight());
258  if (std::isinf(weight))
259  weight = Zero;
260  if (minWeight > weight)
261  minWeight = weight;
262  }
263  ConstStateRef rp = risk_->getState(s);
264  verify(rp);
265  generalized_.grow(s, 0);
266  f64 denSum = sp->isFinal() ? exp(minWeight - f32(sp->weight_)) : 0;
267  f64 numSum = rp->isFinal() ? f32(rp->weight_) * denSum : 0;
268  State::const_iterator r = rp->begin();
269  verify(rp->nArcs() == sp->nArcs());
270  for (State::const_iterator a = sp->begin(); a != sp->end(); ++a, ++r) {
271  f64 weight = potentials_[a->target()] + f32(a->weight());
272  if (std::isinf(weight))
273  weight = Zero;
274  const f64 p = exp(minWeight - weight);
275  numSum += p * (f32(r->weight()) + generalized_[r->target()]);
276  denSum += p;
277  }
278  if (denSum > 0) {
279  potentials_[s] = std::min(minWeight - log1p(denSum - 1), Zero);
280  generalized_[s] = std::max(std::min(numSum / denSum, Zero), -Zero);
281  }
282  if (s == fsa_->initialStateId()) {
283  totalFlow_ = generalized_[s];
284  }
285 }

347 void PosteriorEAutomaton::modifyState(State* sp) const {
348  StateId s = sp->id();
349  ConstStateRef rp = r_->getState(s);
350  if (sp->isFinal()) {
351  sp->weight_ = Weight(exp(-finalLogPosterior(sp)) * finalRisk(rp));

353  State::const_iterator r = rp->begin();
