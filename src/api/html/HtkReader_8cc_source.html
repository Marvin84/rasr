82 Fsa::Weight HtkReader::getWeight(const Link& link, Fsa::LabelId lemmaId) {
83  f64 weight = -(link.amScore * amScale_ + link.lmScore * lmScale_);
84  if (penaltyScale_ != 0.0) {
85  if ((lemmaId == silLemmaId_) && (silLemmaId_ != Fsa::InvalidLabelId))
86  weight += penaltyScale_ * silPenalty_;
87  else
88  weight += penaltyScale_ * wordPenalty_;
89  }
90  return Fsa::Weight(weight);
91 }

167 HtkReader::Status HtkReader::add(const std::string& s, StringPairList& props) {
168  const char* c = s.c_str();
169  const char *keyStart, *keyEnd, *valueStart, *valueEnd;
170 
171  while (*c != '\0') {
172  for (; ::isspace(*c); ++c)
173  ;
174  if (*c == '\0')
175  break;
176  keyStart = c;
177 
178  for (; (*c != '=') && !::isspace(*c) && (*c != '\0'); ++c)
179  ;
180  if (*c == '\0')
181  return htkPropertyParseError;
182  keyEnd = c;
183 
184  for (; ::isspace(*c); ++c)
185  ;
186  if (*c != '=')
187  return htkPropertyParseError;
188  ++c;
189 
190  for (; ::isspace(*c); ++c)
191  ;
192  if (*c == '\0')
193  return htkPropertyParseError;
194 
195  if (*c == '"') {
196  valueStart = ++c;
197 
198  for (; (*c != '"') && (*c != '\0'); ++c)
199  ;
200  if (*c == '\0')
201  return htkPropertyParseError;
202  valueEnd = c;
203  ++c;
204  }
205  else {
206  valueStart = c;
207 
208  for (; !::isspace(*c) && (*c != '\0'); ++c)
209  ;
210  valueEnd = c;
211  }
212 
213  props.push_back(StringPair(
214  std::string(keyStart, keyEnd - keyStart),
215  std::string(valueStart, valueEnd - valueStart)));
216  }
217  return htkOk;
218 }

230 HtkReader::Status HtkReader::set(const StringPairList& props, Node& node) {
231  s32 required = 1;
232  std::string variant;
233  node.label.clear();
234  node.timeframeIndex = Speech::InvalidTimeframeIndex;
235  for (StringPairList::const_iterator it = props.begin();
236  it != props.end(); ++it) {
237  switch (it->first.at(0)) {
238  case 'I':
239  if (!Core::strconv(it->second, node.id))
240  error("Unable to parse \"%s\"", it->second.c_str());
241  else
242  --required;
243  break;
244  case 'W':
245  if (!Core::strconv(it->second, node.label)) {
246  error("Unable to parse \"%s\"", it->second.c_str());
247  node.label.clear();
248  }
249  break;
250  case 'v':
251  s32 variantNo;
252  if (!Core::strconv(it->second, variantNo))
253  error("Unable to parse \"%s\"", it->second.c_str());
254  else if (variantNo > 0)
255  variant = it->second;
256  break;
257  case 't':
258  f64 time;
259  if (!Core::strconv(it->second, time))
260  error("Unable to parse \"%s\"", it->second.c_str());
261  node.timeframeIndex = timeframeIndex(time);
262  break;
263  }
264  }
265  if (!node.label.empty())
266  node.lemmas = parseLabel(node.label, variant);
267  else
268  node.lemmas = LemmaRange(LemmaIterator(), LemmaIterator());
269  return (required) ? htkParseNodeError : htkOk;
270 }

363 ConstWordLatticeRef HtkReader::read(std::istream& is) {
364  lattice_ = new Fsa::StaticAutomaton;
365  wordBoundaries_ = new WordBoundaries;
366 
367  line_ = 0;
368  status_ = htkOk;
369  u32 state = 0;
370  StringPairList _props;
371  std::string s;
372 
373  {
374  if (!checkStatus(status_ = start()))
375  return ConstWordLatticeRef();
376  }
377  {
378  // header
379  nextLine(is, s);
380  while ((state == 0) && (status_ == htkOk) && is) {
381  switch (s.at(0)) {
382  case 'I':
383  state = 1;
384  break;
385  case 'J':
386  state = 2;
387  break;
388  default:
389  status_ = add(s, _props);
390  nextLine(is, s);
391  }
392  }
393  status_ = header(_props);
394  _props.clear();
395  }
396  {
397  // nodes
398  Node _node;
399  while ((state == 1) && (status_ == htkOk) && is) {
400  switch (s.at(0)) {
401  case 'I':
402  if ((status_ = add(s, _props)) != htkOk)
403  break;
404  if ((status_ = set(_props, _node)) != htkOk)
405  break;
406  if ((status_ = node(_node)) != htkOk)
407  break;
408  _props.clear();
409  nextLine(is, s);
410  break;
411  case 'J':
412  state = 2;
413  break;
414  default:
415  status_ = htkUnexpectedLine;
416  }
417  }
418  if (!checkStatus(status_))
419  return ConstWordLatticeRef();
420  }
421  {
422  // links
423  Link _link;
424  while ((state == 2) && (status_ == htkOk) && is) {
425  switch (s.at(0)) {
426  case 'I':
427  status_ = htkUnexpectedNode;
428  break;
429  case 'J':
430  if ((status_ = add(s, _props)) != htkOk)
431  break;
432  if ((status_ = set(_props, _link)) != htkOk)
433  break;
434  if ((status_ = link(_link)) != htkOk)
435  break;
436  _props.clear();
437  nextLine(is, s);
438  break;
439  default:
440  status_ = htkUnexpectedLine;
441  }
442  }
443  if (!checkStatus(status_))
444  return ConstWordLatticeRef();
445  }
446  {
447  status_ = end();
448  if (!checkStatus(status_))
449  return ConstWordLatticeRef();
450  }
451 
452  Core::Ref<WordLattice> f(new WordLattice);
453  f->setWordBoundaries(Core::Ref<WordBoundaries>(wordBoundaries_));
454  f->setFsa(Fsa::ConstAutomatonRef(lattice_), WordLattice::totalFsa);
455  ConstWordLatticeRef l = Lattice::composeMatching(lemmaPronunciationToHtkTransducerRef_, f);
456  WordLattice* wordLattice = new WordLattice;
457  wordLattice->setWordBoundaries(l->wordBoundaries());
458  wordLattice->setFsa(
459  Fsa::staticCopy(
460  Fsa::projectInput(
461  l->part(WordLattice::totalFsa))),
462  WordLattice::totalFsa);
463 
464  lattice_ = 0;
465  wordBoundaries_ = 0;
466  timeOffset_ = Core::Type<f64>::min;
467  states_.clear();
468  return ConstWordLatticeRef(wordLattice);
469 }
