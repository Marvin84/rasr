49 const Core::Ref<Mm::MixtureSet> MixtureSetTrainer::getMixtureSet(
50  size_t nMixtures, size_t dimension) {
51  Core::Ref<Mm::MixtureSet> mixtureSet;
52  if (paramSplitFirst(config)) {
53  const std::string filename = paramOldMixtureSetFilename(config);
54  Mm::MixtureSetEstimator estimator(config);
55  if (read(filename, estimator)) {
56  Mm::MixtureSetSplitter splitter(select("splitter"));
57  mixtureSet = splitter.split(estimator);
58  }
59  else {
60  error("Failed to read mixture-set from \"%s\".", filename.c_str());
61  }
62  }
63  else {
64  mixtureSet = Mm::Module::instance().readMixtureSet(
65  paramOldMixtureSetFilename(config), select("old-mixture-set"));
66  if (!mixtureSet) {
67  error("Failed to read mixture-set.");
68  }
69  }
70  if (mixtureSet) {
71  if (mixtureSet->nMixtures() != nMixtures) {
72  error("Number of labels of alignment (%zd) do not match the number of mixtures (%d).",
73  nMixtures, mixtureSet->nMixtures());
74  }
75  if (mixtureSet->dimension() != dimension) {
76  error("Feature vector size (%zd) do not match the dimension in mixtures-set (%d).",
77  dimension, mixtureSet->dimension());
78  }
79  }
80  return mixtureSet;
81 }

84  size_t dimension,

118 bool MixtureSetTrainer::combine(const std::vector<std::string>& toCombine) {
119  verify(!estimator_);
120 
121  Core::BinaryInputStreams bis(toCombine);
122  if (!bis.areOpen()) {
123  error("Failed to open <mixture-set-files-to-combine> for reading.");
124  return false;
125  }
126 
127  const std::string newMixtureSetFilename = paramNewMixtureSetFilename(config);
128  for (u32 n = 0; n < toCombine.size(); ++n) {
129  if (newMixtureSetFilename == toCombine[n]) {
130  error("<new-mixture-set-file> must be different from <mixture-set-files-to-combine>");
131  return false;
132  }
133  log("Add \"%s\" ...", toCombine[n].c_str());
134  }
135  Core::BinaryOutputStream bos(newMixtureSetFilename);
136  if (!bos) {
137  error("Failed to open \"%s\" for writing", newMixtureSetFilename.c_str());
138  return false;
139  }
140 
141  Mm::AbstractMixtureSetEstimator* estimator = createMixtureSetEstimator();
142  bool result = estimator->accumulate(bis, bos);
143  if (!result) {
144  error("Combination failed.");
145  }
146  delete estimator;
147  return result;
148 }

154  verify(!estimator_);

222 void MixtureSetTrainer::checkCovarianceTying() {
223  Mm::MixtureSetEstimatorIndexMap indexMap(*estimator_);
224  u32 nCovariances = indexMap.covarianceMap().size();
225 
226  switch (Mm::Module_::paramCovarianceTyingType(config)) {
227  case Mm::Module_::mixtureSpecificCovariance:
228  if (nCovariances < estimator_->nMixtures()) {
229  log("creating mixture specific covariance estimators");
230  if (!estimator_->createMixtureSpecificCovarianceEstimator()) {
231  error("modification of mixture set topology failed");
232  }
233  }
234  break;
235  case Mm::Module_::pooledCovariance:
236  if (nCovariances != 1) {
237  log("creating pooled covariance estimator using %d covariance estimators", nCovariances);
238  if (!estimator_->createPooledCovarianceEstimator()) {
239  error("modification of mixture set topology failed");
240  }
241  }
242  break;
243  default:
244  warning("covariance tying not checked");
245  break;
246  }
247 }
