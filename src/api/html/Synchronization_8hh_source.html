89 /** Base class for algorithms which synchronise the input stream to a "target" stream
90  * according to the start-times
91  *
92  * Inputs:
93  * -target: series of target start-times
94  * -(any name): input elements to get synchronised
95  *
96  * Outputs:
97  * -target: elements of the target input without any changes
98  * -(any name): synchronised series of elements
99  *
100  *
101  * Algorithm has to provide the following interface:
102  * typename DataPointer
103  * typename Time
104  * void reset()
105  * bool work(Time time, DataPointer out);
106  * std::string lastError() const;
107  *
108  */
109 template<class Algorithm>
110 class SynchronizationNode : public Node, public Algorithm {
111 public:
112  typedef typename Algorithm::DataPointer DataPointer;
113 
114 private:
115  bool ignoreErrors_;
116  bool firstData_;
117  Time previousStartTime_;
118 
119 private:
120  void reset();
121 
122 protected:
123  /** Delivers next element of the input stream and checks its consistent
124  *
125  * @return is the result of the call to Node::getData
126  *
127  * Consistent checks:
128  * -Empty input stream
129  * -Null input element
130  * -Monotonously increasing start-times
131  */
132  virtual bool nextData(DataPointer& dataPointer);
133 
134 public:
135  static std::string filterName() {
136  return Algorithm::name();
137  }
138 
139  SynchronizationNode(const Core::Configuration& c);
140  virtual ~SynchronizationNode() {}
141 
142  virtual Flow::PortId getInput(const std::string& name) {
143  return name == "target" ? 1 : 0;
144  }
145  virtual Flow::PortId getOutput(const std::string& name) {

209 template<class Algorithm>
210 bool SynchronizationNode<Algorithm>::work(Flow::PortId p) {
211  DataPtr<Timestamp> interpolationTime;
212  if (!getData(1, interpolationTime)) {
213  putData(0, interpolationTime.get());
214  putData(1, interpolationTime.get());
215  return true;
216  }
217 
218  DataPointer out;
219  if (Algorithm::work(*interpolationTime, out)) {
220  verify((bool)out);
221  putData(0, out.get());
222  putData(1, interpolationTime.get());
223  return true;
224  }
225 
226  if (!ignoreErrors_)
227  this->criticalError("%s", this->lastError().c_str());
228  // Synchronization failed, typically at the end of the segment, so put eos
229  putData(0, Flow::Data::eos());
230  putData(1, interpolationTime.get());
231  return true;
232 }

234 template<class Algorithm>
235 bool SynchronizationNode<Algorithm>::nextData(DataPointer& dataPointer) {
236  bool result = Node::getData(0, dataPointer);
237  if (result) {
238  if (!dataPointer)
239  criticalError("Input is null.");
240 
241  if (!firstData_ &&
242  !Core::isSignificantlyLess(previousStartTime_, dataPointer->startTime(), timeTolerance)) {
243  criticalError("Input start-times do not increase monotonously: %f after %f",
244  dataPointer->startTime(), previousStartTime_);
245  }
246 
247  firstData_ = false;
248  previousStartTime_ = dataPointer->startTime();
249  }
250  else {
251  if (firstData_)
252  warning("Input stream is empty.");
253  }
254  return result;
255 }
