139 // network forwarding
140 void NeuralNetworkForwardNode::processBuffer() {
141  for (u32 index = 0; index < nFeatures_.size(); ++index) {
142  inputBuffer_[index].resize(nFeatures_[index], aggregatedFeatures_ ? aggregateBuffer_.size() : featureBuffer_.size());
143  }
144  if (aggregatedFeatures_) {
145  for (size_t b = 0ul; b < aggregateBuffer_.size(); b++) {
146  for (size_t i = 0ul; i < inputBuffer_.size(); i++) {
147  inputBuffer_[i].copy(*(*aggregateBuffer_[b])[i], 0, b);
148  }
149  }
150  }
151  else {
152  for (size_t b = 0ul; b < featureBuffer_.size(); b++) {
153  inputBuffer_[0].copy(*featureBuffer_[b], 0, b);
154  }
155  }
156 
157  bool result = network_.forward(inputBuffer_);
158  require(result);
159  NnMatrix const& out = network_.getTopLayerOutput();
160  totalOutputFrames_ = out.nColumns();
161  currentOutputFrame_ = 0u;
162 }

164 // send next feature from buffer to the output of the node
165 bool NeuralNetworkForwardNode::putNextFeature() {
166  require(currentOutputFrame_ < totalOutputFrames_);
167  Flow::Vector<FeatureType>* out = nullptr;
168  out = new Flow::Vector<FeatureType>(outputDimension_);
169  out->setTimestamp(aggregatedFeatures_ ? aggregateBuffer_[currentOutputFrame_] : featureBuffer_[currentOutputFrame_]);
170  column_.initComputation(false);
171  network_.getTopLayerOutput().getColumn(currentOutputFrame_, column_);
172  column_.finishComputation();
173  ++currentOutputFrame_;
174  if (checkValues_) {
175  if (!column_.isFinite()) {
176  column_.show();
177  this->error("non-finite output of neural network detected");
178  }
179  }
180 
181  Math::copy<FeatureType, FeatureType>(outputDimension_, column_.begin(), 1, &(out->at(0)), 1);
182  bool result = putData(0, out);
183  return result;
184 }

186 // overrides Flow::Node::work
187 bool NeuralNetworkForwardNode::work(Flow::PortId p) {
188  // features from the flow network (single feature stream/aggregate features)
189  Flow::DataPtr<Flow::Vector<FeatureType>> ptrFeatures;
190  Flow::DataPtr<Flow::TypedAggregate<Flow::Vector<FeatureType>>> ptrAggregateFeatures;
191 
192  if (needInit_) {
193  // get data type of the flow stream
194  Core::Ref<Flow::Attributes> attributes(new Flow::Attributes());
195  getInputAttributes(p, *attributes);
196  aggregatedFeatures_ = configureDataType(attributes, Flow::TypedAggregate<Flow::Vector<FeatureType>>::type());
197  }
198  bool endOfStream = false;
199  // there is no more output, get features and forward them
200  if (currentOutputFrame_ >= totalOutputFrames_) {
201  featureBuffer_.clear();
202  aggregateBuffer_.clear();
203  bool validData = true;
204  while (not bufferFull() && validData) {
205  // pull feature from incoming connections
206  if (aggregatedFeatures_) {
207  validData = getData(0, ptrAggregateFeatures);
208  endOfStream = ptrAggregateFeatures == Flow::Data::eos();
209  }
210  else {
211  validData = getData(0, ptrFeatures);
212  endOfStream = ptrFeatures == Flow::Data::eos();
213  }
214  // init when receiving first feature
215  if (needInit_ && validData) {
216  if (not aggregatedFeatures_) { // single feature stream
217  std::vector<u32> nSizes(1);
218  nSizes[0] = ptrFeatures.get()->size();
219  initialize(nSizes);
220  }
221  else {
222  std::vector<u32> nSizes(ptrAggregateFeatures.get()->size());
223  for (u32 i = 0; i < nSizes.size(); ++i) {
224  nSizes[i] = (*ptrAggregateFeatures.get())[i]->size();
225  }
226  initialize(nSizes);
227  }
228  }
229  // add feature to buffer
230  if (validData && aggregatedFeatures_) {
231  aggregateBuffer_.push_back(ptrAggregateFeatures);
232  }
233  else if (validData && not aggregatedFeatures_) {
234  featureBuffer_.push_back(ptrFeatures);
235  }
236  }
237  // forward features in buffer
238  if (not bufferEmpty()) {
239  processBuffer();
240  }
241  }
242  // put next feature in buffer
243  // if buffer still empty: eos has been reached or error in other flow node
244  if (currentOutputFrame_ >= totalOutputFrames_) {
245  if (endOfStream) {
246  return putData(0, Flow::Data::eos());
247  }
248  else {
249  return false;
250  }
251  }
252  else {
253  return putNextFeature();
254  }
255 
256  return false;
257 }
