65 class CachingInPlaceStateRescorer : public RescoreLattice::StateRescorer {

71 public:
72  CachingInPlaceStateRescorer(const RescoreLattice* l)

74  virtual ConstStateRef operator()(Fsa::StateId sid) const {
75  cache_.grow(sid);
76  ConstStateRef sr = cache_[sid];
77  if (!sr) {

98 RescoreLattice::RescoreLattice(ConstLatticeRef l, RescoreMode rescoreMode)
