90 void ComposedNetwork::reset() {
91  if (++resetCount_ < resetInterval_) {
92  return;
93  }
94  resetCount_ = 0;
95  r_->reset();
96  if (stateTable_)
97  log("visited states in compose fst:") << stateTable_->size();
98  delete stateTable_;
99  delete f_;
100  log("creating composed fst");
101  logMemoryUsage();
102  f_ = l_->compose(*r_, cacheSize_, &stateTable_);
103  log("composed fst. cache=%zd", cacheSize_);
104  logMemoryUsage();
105 }
