102 std::set<Phoneme::Id> PhonemeInventory::parseSelection(std::string selector) const {
103  std::set<Bliss::Phoneme::Id> selection;
104  if (!selector.empty()) {
105  for (Bliss::Token::Id phone = 1; phone < nPhonemes(); ++phone) {
106  std::string phoneName(phoneme(phone)->symbol());
107  if ((selector[0] == '*' && selector[selector.length() - 1] == '*' && phoneName.find(selector.substr(1, selector.length() - 2)) != std::string::npos) ||
108  (selector[0] == '*' && phoneName.rfind(selector.substr(1)) == phoneName.length() - (selector.length() - 1)) ||
109  (selector[selector.length() - 1] == '*' && phoneName.find(selector.substr(0, selector.length() - 1)) == 0) ||
110  selector == phoneName)
111  selection.insert(phone);
112  }
113  }
114  return selection;
115 }
