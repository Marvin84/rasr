60 template<typename T>
61 void BufferedAlignedFeatureProcessor<T>::initAcousticModel() {
62  /* acoustic model to identify labels */
63  Speech::ModelCombination modelCombination(select("model-combination"),
64  Speech::ModelCombination::useAcousticModel,
65  Am::AcousticModel::noEmissions | Am::AcousticModel::noStateTransition);
66  modelCombination.load();
67  acousticModel_ = modelCombination.acousticModel();
68  /* set silence */
69  Am::Allophone silenceAllophone(acousticModel_->silence(), Am::Allophone::isInitialPhone | Am::Allophone::isFinalPhone);
70  silence_ = classIndex(acousticModel_->allophoneStateAlphabet()->index(&silenceAllophone, 0));
71  this->log("silence index is ") << silence_;
72  this->log("silence accumulation weight is ") << paramSilenceWeight(config);
73  this->log("use alignment weights: ") << weightedAlignment_;
74 
75  u32 nClasses = acousticModel_->nEmissions();
76  this->log("number of classes of acoustic model: ") << nClasses;
77 
78  if (classLabelWrapper_)
79  delete classLabelWrapper_;
80  classLabelWrapper_ = new ClassLabelWrapper(select("class-labels"), nClasses);
81  require(classLabelWrapper_->nClassesToAccumulate() > 0);
82 
83  /* initialize class weights */
84  setClassWeights();
85 
86  acousticModelNeedInit_ = false;
87 }

89 template<typename T>
90 void BufferedAlignedFeatureProcessor<T>::setClassWeights() {
91  classWeights_.resize(classLabelWrapper_->nClassesToAccumulate(), 1.0);
92  std::string classWeightsFilename(paramClassWeightsFile(config));
93  if (classWeightsFilename != "") {
94  if (paramSilenceWeight(config) != 1.0)
95  this->error("Can not use both silence weight and class weights file");
96  this->log("reading class weights file ") << classWeightsFilename;
97  Math::Module::instance().formats().read(classWeightsFilename, classWeights_);
98  if (classWeights_.size() != classLabelWrapper_->nClassesToAccumulate()) {
99  this->error("dimension mismatch: class weights vs number of classes to accumulate")
100  << classWeights_.size() << " != " << classLabelWrapper_->nClassesToAccumulate();
101  }
102  }
103  if (classLabelWrapper_->isClassToAccumulate(silence_))
104  classWeights_.at(classLabelWrapper_->getOutputIndexFromClassIndex(silence_)) = paramSilenceWeight(config);
105 }

128 template<typename T>
129 void BufferedAlignedFeatureProcessor<T>::processAlignedFeature(Core::Ref<const Speech::Feature> f, Am::AllophoneStateIndex e, Mm::Weight w) {
130  if (acousticModelNeedInit_)
131  initAcousticModel();
132  if (PrecursorBuffer::needInit_)
133  initBuffer(f);
134  u32 labelIndex = classIndex(e);
135  if (classLabelWrapper_->isClassToAccumulate(labelIndex)) {
136  // check for buffer overflow
137  if (PrecursorBuffer::checkIsTooLongSegment())
138  return;
139  // check consistency
140  verify_eq(alignmentBuffer_.size(), BufferedFeatureExtractor<T>::featureBuffer_.at(0).nColumns());
141  alignmentBuffer_.at(PrecursorBuffer::nBufferedFeatures_) = classLabelWrapper_->getOutputIndexFromClassIndex(labelIndex);
142  if (weightedAlignment_)
143  alignmentWeightsBuffer_.at(PrecursorBuffer::nBufferedFeatures_) = w;
144  // collect the feature -> buffer (use BufferedFeatureExtractor)
145  BufferedFeatureExtractor<T>::processFeature(f);
146  }
147 }

149 template<typename T>
150 void BufferedAlignedFeatureProcessor<T>::generateMiniBatch(std::vector<NnMatrix>& miniBatch,
151  Math::CudaVector<u32>& miniBatchAlignment,
152  std::vector<f64>& miniBatchAlignmentWeights,
153  u32 batchSize) {
154  // resize mini batch alignment
155  miniBatchAlignment.resize(batchSize, 0, true);
156  // fill mini batch alignment
157  miniBatchAlignment.finishComputation(false);
158 
159  if (weightedAlignment_)
160  miniBatchAlignmentWeights.resize(batchSize, 0);
161 
162  for (u32 i = 0; i < batchSize; i++) {
163  u32 alignmentIndex = PrecursorBuffer::nProcessedFeatures_ + i;
164  if (PrecursorBuffer::shuffle_) {
165  alignmentIndex = PrecursorBuffer::shuffledIndices_.at(alignmentIndex);
166  }
167  miniBatchAlignment.at(i) = alignmentBuffer_.at(alignmentIndex);
168  if (weightedAlignment_)
169  miniBatchAlignmentWeights.at(i) = alignmentWeightsBuffer_.at(alignmentIndex);
170  }
171  PrecursorBuffer::generateMiniBatch(miniBatch, batchSize);
172 }

174 template<typename T>
175 void BufferedAlignedFeatureProcessor<T>::initTrainer(const std::vector<NnMatrix>& miniBatch) {
176  std::vector<u32> streamSizes;
177  for (u32 stream = 0; stream < miniBatch.size(); stream++) {
178  streamSizes.push_back(miniBatch.at(stream).nRows());
179  }
180  require(PrecursorBuffer::trainer_);
181  PrecursorBuffer::trainer_->initializeTrainer(PrecursorBuffer::batchSize_, streamSizes);
182  PrecursorBuffer::trainer_->setClassWeights(&classWeights_);
183  if (PrecursorBuffer::trainer_->hasClassLabelPosteriors()) {
184  require(classLabelWrapper_);
185  if (PrecursorBuffer::trainer_->getClassLabelPosteriorDimension() != classLabelWrapper_->nClassesToAccumulate()) {
186  this->warning("mismatch in number of trainer class labels (e.g. NN output layer dim) and number of classes to accumulate: ")
187  << PrecursorBuffer::trainer_->getClassLabelPosteriorDimension()
188  << " vs. "
189  << classLabelWrapper_->nClassesToAccumulate();
190  }
191  }

194 template<typename T>
195 void BufferedAlignedFeatureProcessor<T>::processBuffer() {
196  PrecursorBuffer::prepareProcessBuffer();
197  timeval startBatch, endBatch, end;
198  bool measureTime = PrecursorBuffer::trainer_->measuresTime();
199  std::vector<NnMatrix> miniBatch;
200  Math::CudaVector<u32> miniBatchAlignment;
201  std::vector<f64> miniBatchAlignmentWeights;
202  NnVector weights;
203  while (PrecursorBuffer::nProcessedFeatures_ + PrecursorBuffer::batchSize_ <= PrecursorBuffer::nBufferedFeatures_) {
204  log("Process mini-batch ") << PrecursorBuffer::nProcessedMiniBatches_ + 1 << " with " << PrecursorBuffer::batchSize_
205  << " features";
206  f64 timeMinibatch = 0, timeGenerateMiniBatch = 0;
207  TIMER_START(startBatch);
208  generateMiniBatch(miniBatch, miniBatchAlignment, miniBatchAlignmentWeights, PrecursorBuffer::batchSize_);
209  // determine weights for the mini batch features
210  weights.resize(miniBatchAlignment.size(), 0, true);
211  weights.finishComputation(false);
212  // weight vectors according to class membership
213  for (u32 index = 0; index < weights.size(); index++) {
214  weights.at(index) = classWeights_.at(miniBatchAlignment.at(index));
215  }
216  // additionally weight vectors according to alignment weights
217  if (weightedAlignment_) {
218  for (u32 index = 0; index < weights.size(); index++) {
219  weights.at(index) *= miniBatchAlignmentWeights.at(index);
220  }
221  }
222  TIMER_GPU_STOP(startBatch, end, measureTime, timeGenerateMiniBatch)
223  // initialize trainer (trainer checks if initialization is needed)
224  if (!PrecursorBuffer::trainer_->isInitialized())
225  initTrainer(miniBatch);
226 
227  // process mini batch
228  PrecursorBuffer::trainer_->processBatch_feedInput(miniBatch, &weights, PrecursorBuffer::getCurSegment());
229  PrecursorBuffer::trainer_->processBatch_finishWithAlignment(miniBatchAlignment);
230  PrecursorBuffer::nProcessedMiniBatches_++;
231  PrecursorBuffer::nProcessedFeatures_ += PrecursorBuffer::batchSize_;
232  TIMER_GPU_STOP(startBatch, endBatch, measureTime, timeMinibatch)
233  if (measureTime) {
234  log("time for generating mini-batch: ") << timeGenerateMiniBatch;
235  log("overall processing time for mini-batch: ") << timeMinibatch;
236 
237  PrecursorBuffer::trainer_->logBatchTimes();
238  }
239  }
240  // process the remaining feature with a smaller mini batch
241  // only done for algorithms where the mini batch size is not critical
242  u32 nRemainingFeatures = this->nBufferedFeatures_ - this->nProcessedFeatures_;
243  if (this->processRemainingFeatures_ && nRemainingFeatures > 0) {
244  log("Process mini-batch ") << this->nProcessedMiniBatches_ + 1 << " with " << nRemainingFeatures << " features.";
245  generateMiniBatch(miniBatch, miniBatchAlignment, miniBatchAlignmentWeights, nRemainingFeatures);
246  // determine weights for the mini batch features
247  weights.resize(miniBatchAlignment.size(), 0, true);
248  weights.finishComputation(false);
249  // weight vectors according to class membership
250  for (u32 index = 0; index < weights.size(); index++) {
251  weights.at(index) = classWeights_.at(miniBatchAlignment.at(index));
252  }
253  // additionally weight vectors according to alignment weights
254  if (weightedAlignment_) {
255  for (u32 index = 0; index < weights.size(); index++) {
256  weights.at(index) *= miniBatchAlignmentWeights.at(index);
257  }
258  }
259 
260  this->trainer_->setBatchSize(nRemainingFeatures);
261  // initialize trainer
262  if (!this->trainer_->isInitialized())
263  initTrainer(miniBatch);
264  // process mini batch
265  PrecursorBuffer::trainer_->processBatch_feedInput(miniBatch, &weights, PrecursorBuffer::getCurSegment());
266  PrecursorBuffer::trainer_->processBatch_finishWithAlignment(miniBatchAlignment);
267  this->nProcessedMiniBatches_++;
268  this->nProcessedFeatures_ += nRemainingFeatures;
269  // reset to old batch size
270  this->trainer_->setBatchSize(this->batchSize_);
271  }
272  PrecursorBuffer::finalizeProcessBuffer();
273 }

306 template<typename T>
307 void BufferedAlignedFeatureProcessor<T>::leaveCorpus(Bliss::Corpus* corpus) {
308  if (corpus->level() == 0) {
309  PrecursorBuffer::processCorpus();
310  Core::Component::log("Total number of processed mini-batches: ") << PrecursorBuffer::totalNumberOfProcessedMiniBatches_;
311  if (!PrecursorBuffer::trainer_ || !PrecursorBuffer::trainer_->isInitialized()) {
312  warning("BufferedAlignedFeatureProcessor.leaveCorpus: Trainer was not initalized. "
313  "The trainer is lazily initalized usually. "
314  "If this happens, maybe the corpus is empty or we skipped everything. "
315  "We are now initializing the trainer and directly finalizing it.");
316  if (!PrecursorBuffer::trainer_)
317  PrecursorBuffer::trainer_ = createTrainer(config);
318  if (acousticModelNeedInit_)
319  initAcousticModel(); // needed for initTrainer(), classLabelWrapper_
320  initTrainer(std::vector<NnMatrix>()); // should be ok without stream-sizes
321  }
322  PrecursorBuffer::trainer_->finalize();
323  PrecursorAligned::leaveCorpus(corpus);
324  }
