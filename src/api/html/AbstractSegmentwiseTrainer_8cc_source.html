68 void AbstractSegmentwiseTrainer::initialize(Bliss::LexiconRef lexicon) {
69  Precursor::initialize(lexicon);
70  verify(!orthToLemma_);
71  orthToLemma_ = new Bliss::OrthographicParser(select("orthographic-parser"), lexicon);
72  lemmaPronToLemma_ = lexicon->createLemmaPronunciationToLemmaTransducer();

74  Fsa::ConstAutomatonRef lemmaToEval = lexicon->createLemmaToEvaluationTokenTransducer();
