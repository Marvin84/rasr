106 bool StateTreeWriter::write(const StateTree& tree, const std::string& filename) {
107  Core::BinaryOutputStream out(filename);
108  if (!out.good()) {
109  tree.warning("failed to open state tree file");
110  return false;
111  }
112 
113  Position pos = writeHeader(out);
114  if (!writeTree(tree, out)) {
115  tree.warning("failed to write tree");
116  return false;
117  }
118  getTreeDependencies(tree);
119  writeDependenciesPosition(out, pos);
120  Position dependenciesPos = out.position();
121  out.close();
122  if (!writeDependencies(filename, dependenciesPos)) {
123  tree.warning("failed to write dependencies");
124  return false;
125  }
126  return true;
127 }

183 bool StateTreeReader::read(StateTree& tree, const std::string& filename) {
184  bool error = false;
185  Core::BinaryInputStream in(filename);
186  if (!in.isOpen()) {
187  tree.log("no state tree file to open");
188  return false;
189  }
190  Position dependencyPosition;
191  if (!checkHeader(in, dependencyPosition)) {
192  tree.warning("wrong file format in state tree file");
193  return false;
194  }
195  getTreeDependencies(tree);
196  if (!checkDependencies(tree, filename, dependencyPosition)) {
197  tree.warning("failed to read dependencies");
198  error = true;
199  }
200  else {
201  error = !readTree(tree, in);
202  }
203  return !error;
204 }
