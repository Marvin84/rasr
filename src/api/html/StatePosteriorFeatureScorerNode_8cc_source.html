40  Flow::DataPtr<Speech::Feature::FlowFeature> in;
41  if (!getData(0, in)) {
42  return putData(0, in.get());
43  }
44  fs_->setDefaultFilter();
45  Core::Ref<const Speech::Feature> featureRef = Core::ref(new Speech::Feature(in));
46  Core::Ref<const StatePosteriorScorer> scorer(required_cast(const StatePosteriorScorer*,
47  fs_->getAssigningScorer(featureRef).get()));
48  const PosteriorsAndDensities& p = scorer->posteriorsAndDensities();
49  std::vector<Mm::DensityIndex> sortedP;
50  for (PosteriorsAndDensities::const_iterator it = p.begin(); it != p.end(); ++it) {
51  sortedP.push_back(it->first);
52  }
53  std::sort(sortedP.begin(), sortedP.end());
54  ScoreVector sv(fs_->nDensities());
55  for (std::vector<Mm::DensityIndex>::const_iterator it = sortedP.begin(); it != sortedP.end(); ++it) {
56  sv.push_back(*it, p.find(*it)->second);
57  }
58  Core::TsRef<FlowScoreVector> out = Core::tsRef(new FlowScoreVector(sv));
59  out->setTimestamp(*in);
60  return putData(0, out.get());
