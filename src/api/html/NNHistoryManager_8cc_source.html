33 void NNHistoryManager::release(HistoryHandle handle) {
34  NNCacheBase* c = const_cast<NNCacheBase*>(reinterpret_cast<NNCacheBase const*>(handle));
35  require_gt(c->ref_count, 0ul);
36  c->ref_count--;
37  if (c->ref_count == 0ul) {
38  if (has_on_release_handler_) {
39  on_release_handler_(handle);
40  }
41  auto iter = nn_caches_.find(c->history.get());
42  require(iter != nn_caches_.end());
43  NNCacheBase* to_delete = iter->second;
44  nn_caches_.erase(iter);
45  delete to_delete; // might cause cascade of more deletions, thus put it at the end after we do not need iterator anymore
46  }
47 }
