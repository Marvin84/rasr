40  /** grow to size maxId+1. */
41  void grow(size_t maxId, const T& def = T()) {
42  if (maxId >= this->size()) {
43  if (maxId >= this->capacity())
44  this->reserve(std::max(2 * this->capacity(), maxId + 1));
45  this->insert(this->end(), maxId - this->size() + 1, def);
46  }
47  }

49  /** Free over-allocated storage. */
50  void yield() {
51  std::vector<T> tmp(*this);
52  this->swap(tmp);
53  ensure(this->capacity() == this->size());
54  }

56  void minimize() {
57  yield();
58  }

60  /** safe set*/
61  void set(size_t id, const T& val, const T& def = T()) {
62  grow(id, def);
63  (*this)[id] = val;
64  }

66  /** safe get*/
67  T get(size_t id, const T& def = T()) const {
68  /* Earlier, we returned `const T&` here. This is of course dangerous if we might want to return
69  * the default value (which might be a temporary object).
70  * Returning `T` instead is also faster in most cases (because of the indirection +
71  * because maybe even sizeof(void*) >= sizeof(T)).
72  * sizeof(T) <= sizeof(long) * 4 might be a somewhat sane check for this to be at least as fast.
73  * Otherwise, let's force the user of this `Vector` to code somewhat more explicitely.
74  * Note that with compiler optimizations, the compiler will most likely do the same anyway,
75  * no matter if we have `const T&` or `T` as the return here -- only in the unsafe cases, it will
76  * use safe code instead.
77  */
78  static_assert(sizeof(T) <= sizeof(long) * 4, "size too big");
79  if (id < this->size())
80  return (*this)[id];
81  return def;
82  }
83 
84  size_t getMemoryUsed() const {
85  return sizeof(T) * this->capacity() + sizeof(*this);
86  }
