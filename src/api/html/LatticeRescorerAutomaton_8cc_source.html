86 Fsa::Weight EmissionLatticeRescorerAutomaton::_score(const Bliss::Coarticulated<Bliss::LemmaPronunciation>& coarticulatedPronunciation,
87  TimeframeIndex begtime, TimeframeIndex endtime) const {
88  if (begtime < endtime) {
89  f32 score = fsa_->semiring()->one();
90  const Alignment* alignment = alignmentGenerator_->getAlignment(coarticulatedPronunciation, begtime, endtime);
91  for (std::vector<AlignmentItem>::const_iterator al = alignment->begin(); al != alignment->end(); ++al) {
92  Mm::FeatureScorer::Scorer scorer = acousticModel_->featureScorer()->getScorer((*features_)[al->time]);
93  score += scorer->score(acousticModel_->emissionIndex(al->emission));
94  }
95  return Fsa::Weight(score);
96  }
97  else {
98  Core::Application::us()->warning("score 0 assigned to arc with begin time ")
99  << begtime << " , end time " << endtime << " and label id " << coarticulatedPronunciation.object().id();
100  return fsa_->semiring()->one();

284 Fsa::Weight LmLatticeRescorerAutomaton::score(Fsa::StateId s, const Fsa::Arc& a) const {
285  if (s == fsa_->initialStateId()) {
286  histories_.grow(s);
287  histories_[s] = languageModel_->startHistory();
288  }
289  require(histories_[s].isValid());
290  Lm::History hist = histories_[s];
291  Lm::Score score = 0;
292  const Bliss::LemmaPronunciation* lp = alphabet_->lemmaPronunciation(a.input());
293  if (lp) {
294  Lm::addLemmaPronunciationScore(languageModel_, lp, pronunciationScale_, languageModel_->scale(), hist, score);
295  }
296  /*! \warning sentence end score has to be added manually */
297  if (fsa_->getState(a.target())->isFinal()) {
298  score += languageModel_->sentenceEndScore(hist);
299  hist = languageModel_->startHistory();
300  }
301 
302  histories_.grow(a.target());
303  if (!histories_[a.target()].isValid()) {
304  histories_[a.target()] = hist;
305  }
306  if (!(hist == histories_[a.target()])) {
307  languageModel_->error() << "Mismatch between lattice and language model: "
308  "ambiguous history at state '"
309  << a.target() << "'.\n"
310  "Possible causes: 1) lattice is time-conditioned,\n"
311  "2) lattice has been generated by using another language model.";
312  }
313  return Fsa::Weight(score);
314 }
