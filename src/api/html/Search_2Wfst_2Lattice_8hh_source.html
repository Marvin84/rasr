96 inline int Compare(const LatticeWeight& w, const LatticeWeight& v) {
97  f32 f1 = w.AmScore() + w.LmScore(),
98  f2 = v.AmScore() + v.LmScore();
99  if (f1 < f2)
100  return 1;
101  else if (f1 > f1)
102  return -1;
103  else if (w.LmScore() < v.LmScore())
104  return 1;
105  else if (w.LmScore() > v.LmScore())
106  return -1;
107  else
108  return 0;
109 }

151  /**
152  * Call VectorFst::DeleteStates and handle wordBoundaries_
153  */
154  virtual void DeleteStates(const std::vector<StateId>& dstates) {
155  if (!wordBoundaries_.empty()) {
156  std::vector<StateId> sorted = dstates;
157  std::sort(sorted.begin(), sorted.end());
158  sorted.erase(std::unique(sorted.begin(), sorted.end()), sorted.end());
159  std::vector<StateId>::const_iterator d = sorted.begin();
160  StateId nstates = 0;
161  for (StateId s = 0; s < wordBoundaries_.size(); ++s) {
162  if (d != sorted.end() && s == *d) {
163  ++d;
164  }
165  else {
166  if (s != nstates)
167  wordBoundaries_[nstates] = wordBoundaries_[s];
168  ++nstates;
169  }
170  }
171  wordBoundaries_.resize(nstates);
172  }
173  VectorFst::DeleteStates(dstates);
174  }
