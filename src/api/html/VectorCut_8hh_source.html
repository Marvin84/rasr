73  void apply(std::vector<Value>& in) {
74  init(in);
75 
76  u32 f = first();
77  u32 l = std::min(last(), u32(in.size()) - 1);
78  hope(!in.empty() && f <= l);
79 
80  if (f == 0) { // do not copy 'in' if first == 0
81 
82  in.resize(l + 1);
83  }
84  else {
85  std::vector<Value> out(in.begin() + f, in.begin() + l + 1);
86  in.swap(out);
87  }
88  }

157 template<class T>
158 class VectorCutRelativeSurface : public VectorCutLength<T> {
159 public:
160  typedef T Value;
161 
162 private:
163  std::vector<Value> integral_;
164 
165 protected:
166  virtual void init(const std::vector<Value>& in) {
167  integral_.resize(in.size());
168 
169  std::transform(in.begin(), in.end(), integral_.begin(), Core::absoluteValue<Value>());
170 
171  partial_sum(integral_.begin(), integral_.end(), integral_.begin());
172  }
173 
174  /** first
175  * @return first index within the interval to send
176  */
177 
178  virtual u32 first() const {
179  ensure(this->first_ >= 0.0 && this->first_ <= 1.0);
180 
181  if (this->first_ == 0.0)
182  return 0;
183 
184  T limit = integral_.back() * this->first_;
185  return std::distance(integral_.begin(),
186  find_if(integral_.begin(), integral_.end(),
187  std::bind2nd(std::greater_equal<Value>(), limit)));
188  }
189 
190  /** last

192  */
193 
194  virtual u32 last() const {
195  ensure(this->last_ >= 0.0 && this->last_ <= 1.0);
196 
197  if (this->last_ == 1.0)
198  return integral_.size() - 1;
199 
200  T limit = integral_.back() * this->last_;
201  return std::distance(integral_.begin(),
202  find_if(integral_.begin(), integral_.end(),
203  std::bind2nd(std::greater_equal<Value>(), limit)));
204  }
