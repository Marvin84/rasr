29 RescoreMode getRescoreMode(const Core::Configuration& config) {
30  RescoreMode rescoreMode = getRescoreMode(paramRescoreMode(config));
31  switch (rescoreMode) {
32  case RescoreModeInPlaceCache:
33  Core::Application::us()->warning(
34  "Cached in-place rescoring can have side effects in conjunction with non-linear command networks.");
35  break;
36  case RescoreModeInPlace:
37  Core::Application::us()->warning(
38  "In-place rescoring can have side effects. "
39  "USE WITH CARE: amongst others, side effects will show up for all nodes traversing a lattice and not caching the visited states, e.g. \"info\".");
40  break;
41  default:
42  break;
43  }
44  return rescoreMode;
45 }

109 ConstLatticeRef RescoreSingleDimensionNode::rescore(ConstLatticeRef l) {
110  ScoreId id;
111  if (append_) {
112  id = l->semiring()->size();
113  if (!lastSemiring_ || (lastSemiring_.get() != l->semiring().get())) {
114  lastSemiring_ = l->semiring();
115  lastExtendedSemiring_ = appendSemiring(lastSemiring_, appendScale_, key_);
116  if (lastExtendedSemiring_->key(lastExtendedSemiring_->size() - 1) != key_)
117  error("Failed to set key \"%s\" for dimension %zu; probably key is already in use.",
118  key_.c_str(), (lastExtendedSemiring_->size() - 1));
119  }
120  l = offsetSemiring(l, lastExtendedSemiring_, 0);
121  verify(id == l->semiring()->size() - 1);
122  }
123  else {
124  id = l->semiring()->id(key_);
125  if (!l->semiring()->hasId(id)) {
126  if (!Core::strconv(key_, id))
127  id = Semiring::InvalidId;
128  if (!l->semiring()->hasId(id)) {
129  criticalError("Could not rescore lattice \"%s\"; dimension \"%s\" not found",
130  l->describe().c_str(), key_.c_str());
131  return l;
132  }
133  }
134  }
135  return rescore(l, id);
136 }
