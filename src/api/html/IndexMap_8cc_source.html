43 void IndexMappedLm::mapToken(Token s, InternalClassIndex c) {
44  require(s);
45  require(0 <= c && c < nInternalClasses_);
46 
47  if (tokenMap_[s] != invalidClass) {
48  warning("Remapping \"%s\" to \"%s\", was \"%s\".",
49  s->symbol().str(), internalClassName(c), internalClassName(tokenMap_[s]));
50  }
51  if (classMap_[c] != 0) {
52  error("Mapping \"%s\" to \"%s\", which is already occupied by \"%s\".",
53  s->symbol().str(), internalClassName(c), classMap_[c]->symbol().str());
54  }
55  tokenMap_[s] = c;
56  classMap_[c] = s;
57  classSizes_[c] += 1;
58 }

60 void IndexMappedLm::checkForUnmappedTokens() {
61  InternalClassIndex fallback = (fallbackToken()) ? tokenMap_[fallbackToken()] : invalidClass;
62 
63  for (size_t i = 0; i < nInternalClasses(); ++i)
64  if (classMap_[i] == 0)
65  classMap_[i] = fallbackToken();
66 
67  u32 nUnknowns = 0;
68  Bliss::Lexicon::SyntacticTokenIterator s, s_end;
69  for (Core::tie(s, s_end) = lexicon()->syntacticTokens(); s != s_end; ++s) {
70  Token t = (*s);
71  if (tokenMap_[t] == invalidClass) {
72  if (fallback != invalidClass) {
73  ++nUnknowns;
74  warning("Unknown syntactic token \"%s\" mapped to \"%s\".",
75  (*s)->symbol().str(),
76  internalClassName(fallback));
77  tokenMap_[t] = fallback;
78  classSizes_[tokenMap_[t]] += 1;
79  }
80  else {
81  error("Unknown syntactic token \"%s\" could not be mapped.",
82  (*s)->symbol().str());
83  }
84  }
85  }
86  if (nUnknowns)
87  warning(
88  "%d unknown syntactic tokens mapped to \"%s\".", nUnknowns,
89  internalClassName(fallback));
90 }
