39 void EbwDiscriminativeMixtureEstimator::estimateMixtureWeights(
40  std::vector<Weight>& weights,
41  bool normalizeWeights) const {
42  /**
43  * Cambridge University mixture weight update scheme
44  */
45  if (nDensities() > 1 and getWeight() > Core::Type<Weight>::epsilon) {
46  weights.resize(nDensities());
47  Weight kMax = 0;
48  for (DensityIndex dns = 0; dns < nDensities(); ++dns) {
49  weights[dns] = previousMixtureWeights_[dns];
50  if (weights[dns] > Core::Type<f64>::epsilon) {
51  kMax = std::max(kMax, denWeights_[dns] / weights[dns]);
52  }
53  else {
54  weights[dns] = 0;
55  }
56  }
57  std::vector<Weight> k(nDensities(), 0);
58  for (DensityIndex dns = 0; dns < nDensities(); ++dns) {
59  if (weights[dns] > 0) {
60  k[dns] = kMax - denWeights_[dns] / weights[dns];
61  }
62  }
63  for (u32 iter = 0; iter < nIterations_; ++iter) {
64  for (DensityIndex dns = 0; dns < nDensities(); ++dns) {
65  if (weights[dns] > 0) {
66  weights[dns] = weight(dns) + k[dns] * weights[dns];
67  }
68  if (weights[dns] < 0) {
69  Core::Application::us()->log("iteration ")
70  << iter << ": set negative weights["
71  << dns << "]=" << weights[dns] << " to 0";
72  weights[dns] = 0;
73  }
74  }
75  if (normalizeWeights) {
76  Weight sum = std::accumulate(weights.begin(), weights.end(), Weight(0));
77  if (sum > Core::Type<Weight>::delta) {
78  std::transform(weights.begin(), weights.end(),
79  weights.begin(), std::bind2nd(std::divides<Weight>(), sum));
80  }
81  else {
82  Core::Application::us()->warning("cannot normalize mixture weights because of vanishing sum");
83  std::fill(weights.begin(), weights.end(), 1 / Weight(nDensities()));
84  }
85  }
86  }
87  }
88  else {
89  weights.resize(nDensities(), normalizeWeights ? 1 / Weight(nDensities()) : 1);
90  }
91 }

123 Mixture* EbwDiscriminativeMixtureEstimator::estimate(const ReferenceIndexMap<GaussDensityEstimator>& densityMap,
124  bool normalizeWeights) {

126 
127  std::vector<Weight> weights;

130  result->addDensity(densityMap[densityEstimators_[dns]], weights[dns]);

172 DensityIndex EbwDiscriminativeMixtureEstimator::accumulate(Core::BinaryInputStreams& is,
173  Core::BinaryOutputStream& os) {
174  DensityIndex nDensities = Precursor::accumulate(is, os);
175  for (DensityIndex dns = 0; dns < nDensities; ++dns) {
176  Weight weight;
177  is.front() >> weight;
178  for (u32 n = 1; n < is.size(); ++n) {
179  Weight _weight;
180  is[n] >> _weight;
181  weight += _weight;
182  }
183  os << weight;
184  }
185  return nDensities;
