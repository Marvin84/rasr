56 /**
57  * Represents the result of a call to one of the mapAlphabet
58  * functions. This is a separate class and not simply a vector,
59  * because:
60  *
61  * 1. we need additional information of the type of
62  * mapping (injective, one-to-one, etc.)
63  * 2. seemlessly support all special labels
64  **/
65 class AlphabetMapping : public Core::Vector<LabelId> {
66 public:
67  typedef u8 Type;
68  static const Type typeUnmapped = 0x01; /**< all symbols have the same index in FROM and TO */
69  static const Type typeComplete = 0x02; /**< all symbols in FROM are also in TO */
70  static const Type typeIdentity = 0x04; /**< FROM and TO are identical (not just equal) */
71 
72 private:
73  friend void mapAlphabet(ConstAlphabetRef, ConstAlphabetRef,
74  AlphabetMapping&, LabelId, u32);
75  ConstAlphabetRef from_, to_;
76  Type type_;
77  LabelId unkId_;

91  /**
92  * @return true if this mapping requires an automaton to be
93  * modified. False if this is a trivial mapping where all
94  * label ids remain unchanged.

97  return (type_ & (typeUnmapped | typeComplete)) != (typeUnmapped | typeComplete);

124  /**
125  * Request the target label id for a source label id. This
126  * method is aware of special label ids and does not map them.
127  * @return the target label id
128  **/
129  const LabelId operator[](LabelId i) const {
130  if (!isModifyingType())
131  return i;
132  if ((i < FirstLabelId) || (LastLabelId < i))
133  return i;
134  if (size_t(i) >= size())
135  return unkId_;
136  return Core::Vector<LabelId>::operator[](i);
