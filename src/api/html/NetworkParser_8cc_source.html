180 void NetworkParser::start_link(const XmlAttributes atts) {
181  const char* from = atts["from"];
182  const char* to = atts["to"];
183  const char* buffer = atts["buffer"];
184 
185  if (!from) {
186  error("no 'from' attribute for link");
187  return;
188  }
189  if (!to) {
190  error("no 'to' attribute for link");
191  return;
192  }
193 
194  // from
195  std::string tmp = config.resolve(from);
196  std::string::size_type pos = tmp.find(':');
197  std::string from_node = tmp.substr(0, pos);
198  std::string from_port;
199  if (pos == std::string::npos)
200  from_port = "";
201  else
202  from_port = tmp.substr(pos + 1, tmp.size());
203 
204  // to
205  tmp = config.resolve(to);
206  pos = tmp.find(':');
207  std::string to_node = tmp.substr(0, pos);
208  std::string to_port;
209  if (pos == std::string::npos)
210  to_port = "";
211  else
212  to_port = tmp.substr(pos + 1, tmp.size());
213 
214  // add link to network
215  bool status;
216  status = nodeBuilder_.currentTemplate().addLink(from_node, from_port, to_node, to_port,
217  (buffer ? atoi(buffer) : 0));
218  if (!status)
219  error("could not add link to network");
220 }

269 bool NodeBuilder::findNetworkFile(const std::string& pathname, std::string& foundPathname) const {
270  std::string pathnameWithExtension = pathname;
271  if (Core::filenameExtension(pathnameWithExtension) == "")
272  pathnameWithExtension += paramNetworkFileExtension(config);
273 
274  std::vector<std::string> paths;
275  paths.push_back(currentDirectory_);
276  paths.push_back(paramNetworkFilePath(config));
277  char* ret;
278 
279  for (std::vector<std::string>::const_iterator p = paths.begin(); p != paths.end(); ++p) {
280  std::string result = Core::joinPaths(*p, pathnameWithExtension);
281  if (Core::isRegularFile(result)) {
282  char realname[PATH_MAX + 1];
283  ret = realpath(result.c_str(), realname); // absolute path
284  foundPathname = std::string(realname);
285  return true;
286  }
287  }
288  return false;
289 }

306 AbstractNode* NodeBuilder::createNode(const NetworkTemplate::NodeAttributes& nodeAttributes) {
307  using namespace std;
308  const string name = nodeAttributes.name;
309  string filterName = nodeAttributes.filter;
310 
311  filterName = select(name).resolve(filterName);
312 
313  AbstractNode* result = 0;
314 
315  // node is a filter class
316  if ((result = createFilterNode(name, filterName)) != 0) {
317  return result;
318  }
319 
320  // node is a network from a predefined template
321  if (networkTemplates_[filterName]) {
322  Network* newNetwork = new Network(select(name), false);
323  newNetwork->setFilename(filterName);
324  networkTemplates_[filterName]->createNetwork(*newNetwork);
325  return newNetwork;
326  }
327 
328  // node is a network from an extern .flow-file
329  if ((result = createNetworkNode(name, filterName)) != 0) {
330  return result;
331  }
332 
333  error("Could not find filter \"%s\" (neither as built-in nor as network file).",
334  filterName.c_str());
335  return result;
336 }

340 void NodeBuilder::registerNodeInNetwork(AbstractNode* node, Network& network) {
341  for (AbstractNode::UnresolvedAttributes::const_iterator attr = node->unresolvedAttributes().begin();
342  attr != node->unresolvedAttributes().end(); ++attr) {
343  std::string parameterName(attr->first);
344  std::string parameterValue(attr->second);
345 
346  if ((parameterName != "name") && (parameterName != "filter")) {
347  Core::StringExpression parameterValueExpression;
348  Core::StringExpressionParser parameterParser(parameterValueExpression);
349  parameterParser.accept(parameterValue);
350 
351  if (!parameterValueExpression.isConstant()) {
352  // set default value from configuration
353  parameterValueExpression.setVariables(network.getConfiguration());
354 
355  // register as user of network parameter
356  bool res = network.addParameterUse(node, parameterName, parameterValueExpression);
357 
358  if (!res) {
359  error("Network parameter does not exist in \"%s\"=\"%s\"",
360  parameterName.c_str(), parameterValue.c_str());
361  }
362  }
363 
364  if (!node->addParameter(parameterName, parameterValueExpression)) {
365  error("Failed to add parameter in \"%s\"=\"%s\"",
366  parameterName.c_str(), parameterValue.c_str());
367  }
368  }
369  }
