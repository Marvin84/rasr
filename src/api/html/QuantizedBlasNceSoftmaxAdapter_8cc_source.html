21 float quantized_dot_16bit(size_t size, float scale, s16 const* a, s16 const* b) {
22 #ifdef __AVX2__
23  size_t i = 0ul;
24  size_t remainder = size % 16ul;
25  __m256i acc = _mm256_set1_epi32(0);
26  for (; i < size - remainder; i += 16ul) {
27  __m256i val_a = _mm256_loadu_si256(reinterpret_cast<__m256i const*>(a + i));
28  __m256i val_b = _mm256_loadu_si256(reinterpret_cast<__m256i const*>(b + i));
29  acc += _mm256_madd_epi16(val_a, val_b);
30  }
31  __m128i lower = _mm256_extracti128_si256(acc, 0);
32  __m128i upper = _mm256_extracti128_si256(acc, 1);
33  __m128i s = _mm_hadd_epi32(lower, upper);
34  s = _mm_hadd_epi32(s, s);
35  s = _mm_hadd_epi32(s, s);
36  s32 sum = _mm_extract_epi32(s, 0);
37  for (; i < size; i++) {
38  sum += static_cast<s32>(a[i]) * static_cast<s32>(b[i]);
39  }
40  return sum * scale;
41 #else
42  s32 sum = 0ul;
43  for (size_t i = 0ul; i < size; i++) {
44  sum += static_cast<s32>(a[i]) * static_cast<s32>(b[i]);
45  }
46  return sum * scale;
47 #endif
48 }

65 Score QuantizedBlasNceSoftmaxAdapter16Bit::get_score(Lm::CompressedVectorPtr<float> const& nn_out, size_t output_idx) {
66  std::vector<s16> nn_output;
67  s16 const* data;
68  float scale;
69  QuantizedFloatVector16Bits const* vec = dynamic_cast<QuantizedFloatVector16Bits const*>(nn_out.get());
70 
71  if (vec != nullptr) {
72  data = vec->data().data();
73  scale = vec->scale();
74  }
75  else {
76  float inv_scale = 1.0 / nnOutputEpsilon_;
77  float min_val = std::numeric_limits<s16>::min();
78  float max_val = std::numeric_limits<s16>::max();
79  nn_output.resize(nn_out->size());
80  std::vector<float> float_out(nn_out->size());
81  nn_out->uncompress(float_out.data(), float_out.size());
82  for (size_t i = 0ul; i < nn_output.size(); i++) {
83  float val = std::round(float_out[i] * inv_scale);
84  val = std::max(val, min_val);
85  val = std::min(val, max_val);
86  nn_output[i] = static_cast<s16>(val);
87  }
88  data = nn_output.data();
89  scale = nnOutputEpsilon_;
90  }
91 
92  return quantized_dot_16bit(nn_out->size(), weightsBiasEpsilon_ * scale, &weights_(0, output_idx), data) + bias_[output_idx];
93 }
