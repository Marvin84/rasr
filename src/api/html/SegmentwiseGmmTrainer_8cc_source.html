141 void MinimumErrorSegmentwiseGmmTrainer::processWordLattice(
142  Lattice::ConstWordLatticeRef lattice, Bliss::SpeechSegment* s) {
143  Precursor::processWordLattice(lattice, s);
144  if (!lattice) {
145  return;
146  }
147  if (!lattice->hasPart(part_)) {
148  criticalError("lattice has no \" ") << part_ << "\" part";
149  }
150  if (!lattice->hasPart(Lattice::WordLattice::accuracyFsa)) {
151  criticalError("lattice has no \" ") << Lattice::WordLattice::accuracyFsa << "\" part";
152  }
153 
154  // calculate the posterior automaton with expectation semiring of Q(Z), with Z = (P,P*A)
155  // A = accuracies, P = scaled joint probabilities
156  Lattice::ConstWordLatticeRef denominator = Lattice::getPart(lattice, part_); // denominator = P-lattice
157  Fsa::Weight expectation; // expected accuracy (objective function of current sample)
158 
159  Fsa::ConstAutomatonRef fsa = Fsa::posteriorE(Fsa::changeSemiring(denominator->mainPart(), Fsa::LogSemiring),
160  lattice->part(Lattice::WordLattice::accuracyFsa),
161  expectation, true, posteriorTolerance());
162 
163  /**
164  * accumulation of the covariance Cov(A, gradient( log(P) ) )
165  */
166  // pass over all arcs with negative arc weights (negative accuracy = denominator)
167  accumulateDenominator(Fsa::multiply(fsa, Fsa::Weight(f32(-1))), denominator->wordBoundaries());
168  // pass over all arcs with positive arc weights (positive accuracy = numerator)
169  accumulateNumerator(fsa, denominator->wordBoundaries());
170  accumulateObjectiveFunction(expectation);
171  resetAcc();
172 }
