28 u32 FastFourierTransform::setLength(u32 length) {
29  if (length == 0)
30  return length_ = 0;
31 
32  double power = std::log((double)length) / std::log((double)2);
33  if (Core::isAlmostEqual(power, rint(power)))
34  power = rint(power);
35  else
36  power = ceil(power);
37  ensure(power < 32);
38  return (length_ = (1 << (u32)power));
39 }

88 bool RealInverseFastFourierTransform::pack(std::vector<Data>& data) {
89  require_(data.size() == maximalInputSize());
90 
91  if (data[1] != (Data)0 || data[data.size() - 1] != (Data)0) {
92  lastError_ = Core::form("For real inverse FFT, imag(input[0]) and imag(input[FFT-length - 1]) need to be zero.");

94  }
95 
96  data[1] = data[data.size() - 2];
97  data.resize(data.size() - 2);
