58 void ClassMapping::load() {
59  std::string filename = paramClassFileName(config);
60  Core::CompressedInputStream* cis =
61  new Core::CompressedInputStream(filename.c_str());
62  Core::TextInputStream tis(cis);
63  tis.setEncoding(paramClassFileEncoding(config));
64  u32 nSyntTok = lexicon_->syntacticTokenInventory().size();
65  classMap_.resize(nSyntTok, std::make_pair(static_cast<ClassToken*>(0), 0.0));
66  std::string syntWord, classWord;
67  f32 classScore;
68  while (tis) {
69  std::string line;
70  Core::getline(tis, line);
71  std::istringstream iss(line);
72  if (!(iss >> std::ws).good())
73  continue;
74  iss >> syntWord >> std::ws;
75  if ((syntWord.at(0) == '#') || (syntWord.at(0) == ';'))
76  continue;
77  require(iss.good());
78  if ((iss >> classWord >> std::ws).good())
79  iss >> classScore;
80  else
81  classScore = 1.0;
82  const Bliss::SyntacticToken* syntTok = lexicon_->syntacticToken(syntWord);
83  if (!syntTok) {
84  warning("In class file: Unknown syntactic token \"%s\" found.",
85  syntWord.c_str());
86  continue;
87  }
88  if (classMap_[syntTok->id()].first) {
89  error("In class file: Duplicated entry for syntactic token \"%s\"; discard entry.",
90  syntWord.c_str());
91  continue;
92  }
93  ClassToken* classTok = static_cast<ClassToken*>(tokenInventory_[classWord]);
94  if (!classTok)
95  classTok = addToken(classWord);
96  classTok->syntacticTokens_.push_back(syntTok);
97  ensure(size_t(syntTok->id()) < classMap_.size());
98  classMap_[syntTok->id()] = std::make_pair(classTok, classScore);
99  --nSyntTok;
100  }
101  if (nSyntTok) {
102  log("No mapping specified for %d syntactic tokens; "
103  "add identity mappings.",
104  nSyntTok);
105  Bliss::Lexicon::SyntacticTokenIterator synt_tok_it, synt_tok_end;
106  for (Core::tie(synt_tok_it, synt_tok_end) = lexicon_->syntacticTokens();
107  synt_tok_it != synt_tok_end; ++synt_tok_it) {
108  const Bliss::SyntacticToken* syntTok = *synt_tok_it;
109  if (!classMap_[syntTok->id()].first) {
110  std::string classWord = syntTok->symbol().str();
111  ClassToken* classTok = dynamic_cast<ClassToken*>(tokenInventory_[classWord]);
112  if (!classTok)
113  classTok = addToken(classWord);
114  classTok->syntacticTokens_.push_back(syntTok);
115  ensure(size_t(syntTok->id()) < classMap_.size());
116  classMap_[syntTok->id()] = std::make_pair(classTok, 1.0);
117  }
118  }
119  }
120  for (Bliss::TokenInventory::Iterator itClassTok = tokenInventory_.begin();
121  itClassTok != tokenInventory_.end(); ++itClassTok) {
122  ClassToken* classTok = static_cast<ClassToken*>(*itClassTok);
123  Score sum = 0.0;
124  for (ClassToken::SyntacticTokenList::iterator itSyntTok = classTok->syntacticTokens_.begin();
125  itSyntTok != classTok->syntacticTokens_.end(); ++itSyntTok)
126  sum += classMap_[(*itSyntTok)->id()].second;
127  for (ClassToken::SyntacticTokenList::iterator itSyntTok = classTok->syntacticTokens_.begin();
128  itSyntTok != classTok->syntacticTokens_.end(); ++itSyntTok) {
129  std::pair<ClassToken*, Score>& tokenScore = classMap_[(*itSyntTok)->id()];
130  tokenScore.second = -::log(tokenScore.second / sum);
131  }
132  }
133 }

183 void ClassMapping::writeClasses(Core::XmlWriter& xml) const {
184  u32 nClass = 1;
185  for (Bliss::TokenInventory::Iterator itClassTok = tokenInventory_.begin();
186  itClassTok != tokenInventory_.end(); ++itClassTok, ++nClass) {
187  ClassToken* classTok = static_cast<ClassToken*>(*itClassTok);
188  xml << std::setw(4) << std::right << nClass << ". " << std::string(classTok->symbol()) << "={";
189  ClassToken::Iterator itSyntTok = classTok->begin();
190  xml << std::string((*itSyntTok)->symbol())
191  << "/" << classEmissionScore(*itSyntTok);
192  for (++itSyntTok; itSyntTok != classTok->end(); ++itSyntTok)
193  xml << ", " << std::string((*itSyntTok)->symbol())
194  << "/" << classEmissionScore(*itSyntTok);
195  xml << "}\n";
196  }
