47 void basicInfo(ConstLatticeRef l, Core::XmlWriter& o) {
48  o << Core::XmlFull("description", l->describe());
49  o << Core::XmlFull("fsa-type", Fsa::TypeChoice[l->type()]);
50  std::string tmp;
51  for (Core::Choice::const_iterator i = Fsa::PropertyChoice.begin(); i != Fsa::PropertyChoice.end(); ++i) {
52  if (l->knowsProperty(i->value())) {
53  if (!tmp.empty())
54  tmp += " ";
55  if (!l->hasProperty(i->value()))
56  tmp += "!";
57  tmp += i->ident();
58  }
59  }
60  o << Core::XmlFull("fsa-properties", tmp);
61  o << Core::XmlFull("semiring", l->semiring()->name());
62  o << Core::XmlFull(
63  "input-alphabet", Lexicon::us()->alphabetName(Lexicon::us()->alphabetId(l->getInputAlphabet())));
64  if (l->type() == Fsa::TypeTransducer)
65  o << Core::XmlFull(
66  "output-alphabet", Lexicon::us()->alphabetName(Lexicon::us()->alphabetId(l->getOutputAlphabet())));
67  if (l->initialStateId() != Fsa::InvalidStateId)
68  o << Core::XmlFull("initial-state-id", l->initialStateId());
69  else
70  o << Core::XmlEmpty("no-states");
71 }

141 void extendedInfo(ConstLatticeRef l, Core::XmlWriter& o, LatticeStatistics& stats, BestStatistics& bestStats) {
142  o << Core::XmlOpen("lattice-info") + Core::XmlAttribute("type", "extended");
143  basicInfo(l, o);
144  if (l->initialStateId() != Fsa::InvalidStateId) {
145  // init
146  l = persistent(l);
147  bool isAcceptor = (l->type() == Fsa::TypeAcceptor);
148  ConstSemiringRef semiring = l->semiring();
149  ConstSemiringRef posteriorSemiring = toLogSemiring(semiring);
150  Lexicon::AlphabetId inputAlphabetId = Lexicon::us()->alphabetId(l->getInputAlphabet());
151  LabelMapRef inputNonWordMap;
152  if ((inputAlphabetId == Lexicon::LemmaAlphabetId) || (inputAlphabetId == Lexicon::LemmaPronunciationAlphabetId))
153  inputNonWordMap = LabelMap::createNonWordToEpsilonMap(inputAlphabetId);
154  Lexicon::AlphabetId outputAlphabetId = inputAlphabetId;
155  LabelMapRef outputNonWordMap;
156  if (!isAcceptor) {
157  outputAlphabetId = Lexicon::us()->alphabetId(l->getOutputAlphabet());
158  if ((outputAlphabetId == Lexicon::LemmaAlphabetId) || (outputAlphabetId == Lexicon::LemmaPronunciationAlphabetId))
159  outputNonWordMap = LabelMap::createNonWordToEpsilonMap(outputAlphabetId);
160  }
161 
162  // fwd/bwd-scores, topological sort
163  std::pair<ConstLatticeRef, ConstFwdBwdRef> result = FwdBwd::build(l, posteriorSemiring);
164  l = result.first;
165  ConstFwdBwdRef fb = result.second;
166  ConstStateMapRef topologicalSort = sortTopologically(l);
167  Fsa::StateId initialSid = topologicalSort->front();
168 
169  // lattice statistics
170  Traceback traceback(topologicalSort->maxSid + 1);
171  traceback[initialSid].score = 0.0;
172  TraceElement bestTrace;
173  for (u32 i = 0; i < topologicalSort->size(); ++i) {
174  Fsa::StateId sid = (*topologicalSort)[i];
175  ConstStateRef sr = l->getState(sid);
176  Time t = l->boundary(sid).time();
177  if (t < stats.minTime)
178  stats.minTime = t;
179  if (t > stats.maxTime)
180  stats.maxTime = t;
181  const TraceElement& currentTrace = traceback[sid];
182  if (sr->isFinal()) {
183  Score score = currentTrace.score + semiring->project(sr->weight());
184  if (score < bestTrace.score) {
185  bestTrace.score = score;
186  bestTrace.bptr = sid;
187  }
188  ++stats.nFinalStates;
189  }
190  for (State::const_iterator a = sr->begin(); a != sr->end(); ++a) {
191  Score score = currentTrace.score + semiring->project(a->weight());
192  TraceElement& trace = traceback[a->target()];
193  if (score < trace.score) {
194  trace.score = score;
195  trace.bptr = sid;
196  trace.arc = *a;
197  }
198  if (a->input() == Fsa::Epsilon)
199  ++stats.nInputEpsilonArcs;
200  else if (inputNonWordMap && !(*inputNonWordMap)[a->input()].empty())
201  ++stats.nInputNonWordArcs;
202  else
203  ++stats.nInputWordArcs;
204  if (!isAcceptor) {
205  if (a->output() == Fsa::Epsilon)
206  ++stats.nOutputEpsilonArcs;
207  else if (outputNonWordMap && !(*outputNonWordMap)[a->output()].empty())
208  ++stats.nOutputNonWordArcs;
209  else
210  ++stats.nOutputWordArcs;
211  }
212  }
213  }
214  verify(bestTrace.bptr != Fsa::InvalidStateId);
215 
216  // best statistics
217  bestStats.minTime = l->boundary(initialSid).time();
218  bestStats.maxTime = l->boundary(bestTrace.bptr).time();
219  bestStats.nFinalStates = 1;
220  bestStats.scores = l->getState(bestTrace.bptr)->weight();
221  Fsa::StateId bestSid = bestTrace.bptr;
222  while (bestSid != initialSid) {
223  const TraceElement& trace = traceback[bestSid];
224  if (trace.arc.input() == Fsa::Epsilon)
225  ++bestStats.nInputEpsilonArcs;
226  else if (inputNonWordMap && !(*inputNonWordMap)[trace.arc.input()].empty())
227  ++bestStats.nInputNonWordArcs;
228  else
229  ++bestStats.nInputWordArcs;
230  if (!isAcceptor) {
231  if (trace.arc.output() == Fsa::Epsilon)
232  ++bestStats.nOutputEpsilonArcs;
233  else if (outputNonWordMap && !(*outputNonWordMap)[trace.arc.output()].empty())
234  ++bestStats.nOutputNonWordArcs;
235  else
236  ++bestStats.nOutputWordArcs;
237  }
238  bestStats.scores = semiring->extend(bestStats.scores, trace.arc.weight());
239  bestSid = trace.bptr;
240  }
241  bestStats.score = semiring->project(bestStats.scores);
242  bestStats.posteriorScore = posteriorSemiring->project(bestStats.scores) - fb->sum();
243  // verify(Core::isAlmostEqualUlp(bestTrace.score, bestStats.score, 100));
244 
245  // dump lattice statistics
246  o << Core::XmlOpen("lattice-statistics");
247  if (l->getBoundaries()->valid()) {
248  o << Core::XmlFull("min-timeframe", stats.minTime);
249  o << Core::XmlFull("max-timeframe", stats.maxTime);
250  if (stats.maxTime != stats.minTime) {
251  o << Core::XmlFull("arcs-per-second",
252  (f64(stats.nInputEpsilonArcs + stats.nInputNonWordArcs + stats.nInputWordArcs) * 100) / (stats.maxTime - stats.minTime));
253  o << Core::XmlFull("word-arcs-per-second", (f64(stats.nInputWordArcs) * 100) / (stats.maxTime - stats.minTime));
254  }
255  }
256  o << Core::XmlFull("states", topologicalSort->size());
257  o << Core::XmlFull("final-states", stats.nFinalStates);
258  o << Core::XmlFull("max-state-id", topologicalSort->maxSid);
259  u32 nArcs = stats.nInputEpsilonArcs + stats.nInputNonWordArcs + stats.nInputWordArcs;
260  o << Core::XmlFull("arcs", nArcs);
261  o << Core::XmlFull("input-epsilon-arcs", stats.nInputEpsilonArcs);
262  if (inputNonWordMap) {
263  o << Core::XmlFull("input-non-word-arcs", stats.nInputNonWordArcs);
264  o << Core::XmlFull("input-word-arcs", stats.nInputWordArcs);
265  }
266  if (!isAcceptor) {
267  o << Core::XmlFull("output-epsilon-arcs", stats.nOutputEpsilonArcs);
268  if (outputNonWordMap) {
269  o << Core::XmlFull("output-non-word-arcs", stats.nOutputNonWordArcs);
270  o << Core::XmlFull("output-word-arcs", stats.nOutputWordArcs);
271  }
272  }
273  bestStats.nNonEps = bestStats.nInputNonWordArcs + bestStats.nInputWordArcs;
274  if (bestStats.nNonEps > 0) {
275  f64 nNonEps = f64(bestStats.nNonEps);
276  bestStats.densityAllNonEps = f64(stats.nInputEpsilonArcs + stats.nInputNonWordArcs + stats.nInputWordArcs) / nNonEps;
277  bestStats.densityNonEpsNonEps = f64(stats.nInputNonWordArcs + stats.nInputWordArcs) / nNonEps;
278  }
279  o << Core::XmlFull("density", bestStats.densityAllNonEps) + Core::XmlAttribute("numerator", "all arcs") + Core::XmlAttribute("denominator", "non-epsilon arcs");
280  o << Core::XmlFull("density", bestStats.densityNonEpsNonEps) + Core::XmlAttribute("numerator", "non-epsilon arcs") + Core::XmlAttribute("denominator", "non-epsilon arcs");
281  if (inputNonWordMap) {
282  bestStats.nWords = bestStats.nInputWordArcs;
283  if (bestStats.nWords > 0) {
284  f64 nWords = f64(bestStats.nWords);
285  bestStats.densityAllWord = f64(stats.nInputEpsilonArcs + stats.nInputNonWordArcs + stats.nInputWordArcs) / nWords;
286  bestStats.densityWordWord = f64(stats.nInputWordArcs) / nWords;
287  }
288  o << Core::XmlFull("density", bestStats.densityAllWord) + Core::XmlAttribute("numerator", "all arcs") + Core::XmlAttribute("denominator", "word arcs");
289  o << Core::XmlFull("density", bestStats.densityWordWord) + Core::XmlAttribute("numerator", "word arcs") + Core::XmlAttribute("denominator", "word arcs");
290  }
291  o << Core::XmlClose("lattice-statistics");
292 
293  // dump best statistics
294  o << Core::XmlOpen("best-statistics");
295  o << Core::XmlFull("scores", semiring->describe(bestStats.scores, Fsa::HintShowDetails | HintUnscaled));
296  o << Core::XmlFull("score", bestStats.score);
297  o << Core::XmlFull("posterior-score", Core::form("%.15f", bestStats.posteriorScore));
298  o << Core::XmlFull("posterior-probability", Core::form("%.15f", ::exp(-bestStats.posteriorScore)));
299  if (l->getBoundaries()->valid()) {
300  o << Core::XmlFull("start-timeframe", bestStats.minTime);
301  o << Core::XmlFull("end-timeframe", bestStats.maxTime);
302  }
303  u32 nBestArcs = bestStats.nInputEpsilonArcs + bestStats.nInputNonWordArcs + bestStats.nInputWordArcs;
304  o << Core::XmlFull("arcs", nBestArcs);
305  o << Core::XmlFull("input-epsilon-arcs", bestStats.nInputEpsilonArcs);
306  if (inputNonWordMap) {
307  o << Core::XmlFull("input-non-word-arcs", bestStats.nInputNonWordArcs);
308  o << Core::XmlFull("input-word-arcs", bestStats.nInputWordArcs);
309  }
310  if (!isAcceptor) {
311  o << Core::XmlFull("output-epsilon-arcs", bestStats.nOutputEpsilonArcs);
312  if (outputNonWordMap) {
313  o << Core::XmlFull("output-non-word-arcs", bestStats.nOutputNonWordArcs);
314  o << Core::XmlFull("output-word-arcs", bestStats.nOutputWordArcs);
315  }
316  }
317  o << Core::XmlClose("best-statistics");
318  }
319  o << Core::XmlClose("lattice-info");
320 }

358 class InfoNode : public FilterNode {
359  friend class Network;

361 public:
362  static const Core::ParameterString paramInfoType;
363 
364 private:
365  InfoType infoType_;
366  // extended
367  u32 nLattices_;
368  u32 nBestNonEpsLattices_;
369  u32 nBestNonEpsTokens_;
370  f64 sumAllNonEps_;
371  f64 sumNonEpsNonEps_;
372  u32 nBestWordLattices_;
373  u32 nBestWordTokens_;
374  f64 sumAllWord_;
375  f64 sumTime_;
376  f64 sumWordWord_;
377 
378 protected:
379  void overallInfo(Core::XmlWriter& o) const {
380  o << Core::XmlOpen("overall-lattice-statistics");
381  o << Core::XmlFull("lattices", nLattices_);
382  o << Core::XmlOpen("best");
383  o << Core::XmlFull("lattices", nBestNonEpsLattices_) + Core::XmlAttribute("type", "has-non-eps-tokens");
384  o << Core::XmlFull("tokens", nBestNonEpsTokens_) + Core::XmlAttribute("type", "non-eps");
385  o << Core::XmlFull("lattices", nBestWordLattices_) + Core::XmlAttribute("type", "has-word-tokens");
386  o << Core::XmlFull("tokens", nBestWordTokens_) + Core::XmlAttribute("type", "word");
387  if (sumTime_) {
388  o << Core::XmlFull("arcs-per-second", sumAllWord_ / sumTime_);
389  o << Core::XmlFull("word-arcs-per-second", sumWordWord_ / sumTime_);
390  }
391  o << Core::XmlClose("best");
392  f64 densityAllNonEps = 0.0;
393  f64 densityNonEpsNonEps = 0.0;
394  if (nBestNonEpsTokens_ > 0) {
395  f64 nNonEps = f64(nBestNonEpsTokens_);
396  densityAllNonEps = sumAllNonEps_ / nNonEps;
397  densityNonEpsNonEps = sumNonEpsNonEps_ / nNonEps;
398  }
399  o << Core::XmlFull("density", densityAllNonEps) + Core::XmlAttribute("numerator", "all arcs") + Core::XmlAttribute("denominator", "non-epsilon arcs");
400  o << Core::XmlFull("density", densityNonEpsNonEps) + Core::XmlAttribute("numerator", "non-epsilon arcs") + Core::XmlAttribute("denominator", "non-epsilon arcs");
401  f64 densityAllWord = 0.0;
402  f64 densityWordWord = 0.0;
403  if (nBestWordTokens_ > 0) {
404  f64 nWords = f64(nBestWordTokens_);
405  densityAllWord = sumAllNonEps_ / nWords;
406  densityWordWord = sumWordWord_ / nWords;
407  }
408  o << Core::XmlFull("density", densityAllWord) + Core::XmlAttribute("numerator", "all arcs") + Core::XmlAttribute("denominator", "word arcs");
409  o << Core::XmlFull("density", densityWordWord) + Core::XmlAttribute("numerator", "word arcs") + Core::XmlAttribute("denominator", "word arcs");
410  o << Core::XmlClose("overall-lattice-statistics");
411  }

413  virtual ConstLatticeRef filter(ConstLatticeRef l) {
414  if (l) {
415  if (infoType_ == InfoTypeExtended) {
416  LatticeStatistics stats;
417  BestStatistics bestStats;
418  extendedInfo(l, log(), stats, bestStats);
419  ++nLattices_;
420  if (bestStats.nNonEps > 0) {
421  ++nBestNonEpsLattices_;
422  nBestNonEpsTokens_ += bestStats.nNonEps;
423  sumAllNonEps_ += f64(bestStats.nNonEps) * bestStats.densityAllNonEps;
424  sumNonEpsNonEps_ += f64(bestStats.nNonEps) * bestStats.densityNonEpsNonEps;
425  }
426  if (bestStats.nWords > 0) {
427  ++nBestWordLattices_;
428  nBestWordTokens_ += bestStats.nWords;
429  sumAllWord_ += f64(bestStats.nWords) * bestStats.densityAllWord;
430  sumWordWord_ += f64(bestStats.nWords) * bestStats.densityWordWord;
431  }
432  sumTime_ += bestStats.maxTime - bestStats.minTime;
433  }
434  else
435  info(l, log(), infoType_);
436  }
437  else
438  log() << Core::XmlOpen("lattice-info")
439  << Core::XmlEmpty("empty-lattice")
440  << Core::XmlClose("lattice-info");
441  return l;
442  }
