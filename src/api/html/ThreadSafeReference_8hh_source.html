24 /** Base class for reference-counted objects used in a multithreaded environment.
25  *
26  * To enable reference counting for a class it need to inhert this
27  * class publically.
28  *
29  * Remark: ThreadSafeReferenceCounted is by intention not derived from ReferenceCounted
30  * to forbit access of thread safe reference count by a not thread safe Core::Ref object.
31  */
32 class ThreadSafeReferenceCounted {
33  template<class T>
34  friend class TsRef;
35 
36 private:
37  mutable u32 referenceCount_;
38  mutable Core::Mutex mutex_;
39 
40 private:
41  static inline ThreadSafeReferenceCounted* sentinel() {
42  static ThreadSafeReferenceCounted sentinel_(1);
43  return &sentinel_;
44  }
45  static bool isSentinel(const ThreadSafeReferenceCounted* object) {
46  return object == sentinel();
47  }
48  static bool isNotSentinel(const ThreadSafeReferenceCounted* object) {
49  return object != sentinel();
50  }
51 
52 protected:
53  virtual void free() const {
54  verify_(isNotSentinel(this));
55  delete this;
56  }
57 
58  void lock() const {
59  mutex_.lock();
60  }
61  void release() const {
62  mutex_.release();
63  }
64  void increment() const {
65  ++referenceCount_;
66  }
67  bool decrement() const {
68  return ((--referenceCount_) == 0);
69  }
70 
71 protected:
72  explicit ThreadSafeReferenceCounted(u32 rc)
73  : referenceCount_(rc) {}
74 
75 public:
76  ThreadSafeReferenceCounted()
77  : referenceCount_(0) {}
78  ThreadSafeReferenceCounted(const ThreadSafeReferenceCounted&)
79  : referenceCount_(0) {}
80  ThreadSafeReferenceCounted& operator=(const ThreadSafeReferenceCounted&) {
81  return *this;
82  }
83  /** Constructor for static objects. */
84  virtual ~ThreadSafeReferenceCounted() {} // calling mutex_.lock() might be useful for debugging
85 
86  u32 refCount() const {
87  return referenceCount_;
88  }
89 };

91 /**
92  * Class template for smart pointers using intrusive
93  * reference-counting in a multithreaded environment.
94  *
95  * For more important comments @see Core::Ref
96  */
97 template<class T /* -> ThreadSafeReferenceCounted */>
98 class TsRef {
99 protected:
100  typedef T Object;
101  Object* object_;
102  static Object* sentinel() {
103  return static_cast<Object*>(Object::sentinel());
104  }
105 
106 protected:
107  inline void referenceObject(Object* obj) {
108  obj->lock();
109  obj->increment();
110  obj->release();
111  }
112  inline void reference() {
113  referenceObject(object_);
114  }
115  inline void unreference() {
116  object_->lock();
117  if (object_->decrement()) {
118  verify_(Object::isNotSentinel(object_));
119  object_->release();
120  object_->free();
121  (object_ = sentinel())->increment();
122  }
123  else
124  object_->release();
125  }
126 
127  /** Takes over the value of @param o.
128  * -Handles self assignment, i.e. if @param o == object_.
129  * -@see the function 'tsRef' about how to create a TsRef object
130  * conveniently from a pointer.
131  */
132  void set(Object* o) {
133  // it might happen that the object we get is also referenced by the current object
134  // thus we need to increment the ref counter on the new object first, only then
135  // unreference the old object, this happens for example in Search::SearchSpace::pruneSilenceSiblings
136  if (object_ != o) {
137  referenceObject(o);
138  unreference();
139  object_ = o;
140  }
141  }
142 
143 public:
144  TsRef()
145  : object_(sentinel()) {
146  reference();
147  }
148  explicit TsRef(Object* o)
149  : object_(o) {
150  require_(object_);
151  reference();
152  }
153  /** Copy Constructor for the Type Ref<T>
154  * @warning: Implementaion of this constructor is necesseary although the template
155  * copy constructor implements exactly the same. If this constructor is not implemented,
156  * the compiler creates a default copy constructor instead of using the template one.
157  */
158  TsRef(const TsRef& r)
159  : object_(r.object_) {
160  reference();
161  }
162  /** Template Copy Constuctor
163  * Using templates solves the problem of casting ...
164  * -to const pointer
165  * -to precursor class pointer.
166  * For all classes S which are not derived from T, this constructor will not compile.
167  */
168  template<class S>
169  TsRef(const TsRef<S>& r)
170  : object_(r._get()) {
171  reference();
172  }
173  ~TsRef() {
174  unreference();
175  }
176 
177  TsRef& operator=(const TsRef& rhs) {
178  set(rhs.object_);
179  return *this;
180  }
181  template<class S>
182  TsRef& operator=(const TsRef<S>& rhs) {
183  set(rhs._get());
184  return *this;
185  }
186 
187  /** Reset to void.
188  * Corresponds to assigning zero to normal pointer.
189  */
190  void reset() {
191  set(sentinel());
192  }
193 
194  /** Test for identity */
195  bool operator==(const TsRef& r) const {
196  return object_ == r.object_;
197  }
198  template<class S>
199  bool operator==(const TsRef<S>& r) const {
200  return object_ == r._get();
201  }
202  /** Test for non-identity */
203  bool operator!=(const TsRef& r) const {
204  return object_ != r.object_;
205  }
206  template<class S>
207  bool operator!=(const TsRef<S>& r) const {
208  return object_ != r._get();
209  }
210 
211  /** Test for non-voidness. */
212  operator bool() const {
213  return Object::isNotSentinel(object_);
214  }
215  /** Test for voidness. */
216  bool operator!() const {
217  return Object::isSentinel(object_);
218  }
219 
220  Object& operator*() const {
221  require_(Object::isNotSentinel(object_));
222  return *object_;
223  }
224  Object* operator->() const {
225  require_(Object::isNotSentinel(object_));
226  return object_;
227  }
228 
229  /** Explicit conversion to normal pointer.
230  * @warning This defeats the reference counting mechanism.
231  * Use with caution!
232  */
233  Object* get() const {
234  return Object::isNotSentinel(object_) ? object_ : 0;
235  }
236  /** Copies the value of the pointer.
237  * @warning Do not use this function. It is only used in the
238  * template copy contructor and the template operator= functions.
239  */
240  Object* _get() const {
241  return object_;
242  }
243 };
