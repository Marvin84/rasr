106  struct AnchoredAccumulator : public Ftl::Accumulator<_Weight> {
107  typedef _Weight (ConstSelf::*Function)(const _Weight& a, const _Weight& b) const;
108  const Self* self;
109  Function f;
110  _Weight w;
111  AnchoredAccumulator(const Self* self, Function f, const _Weight& init)
112  : self(self), f(f), w(init) {
113  self->acquireReference();
114  }
115  virtual ~AnchoredAccumulator() {
116  if (self->releaseReference())
117  self->free();
118  }
119  void feed(const _Weight& a) {
120  w = (self->*f)(w, a);
121  }
122  _Weight get() const {
123  return w;
124  }
125  };
