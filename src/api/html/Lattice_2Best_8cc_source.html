85 Fsa::ConstAutomatonRef NBestListExtractor::getNBestListWithoutWordBoundaries(
86  Fsa::ConstAutomatonRef fsa, f32& threshold) {
87  Fsa::ConstAutomatonRef nbl;
88  u32 nHypotheses = 0;
89  Fsa::Weight minWeight = Fsa::bestscore(fsa); // TODO: a bestscore for FAIL arcs
90  Fsa::ConstSemiringRef sr = fsa->semiring();
91  threshold = minThreshold_;
92  for (; threshold <= maxThreshold_; threshold += thresholdIncrement_) {
93  if (threshold > minThreshold_)
94  Core::Application::us()->log() << "NBestListExtractor::getNBestListWithoutWordBoundaries: threshold increased to " << threshold;
95 
96  Fsa::ConstAutomatonRef pruned;
97  if (hasFailArcs_) {
98  pruned = Fsa::posteriorPrune4SpecialSymbols(fsa, Fsa::Weight(threshold));
99  pruned = Fsa::removeFailure4SpecialSymbols(pruned);
100  }
101  else {
102  pruned = Fsa::prunePosterior(fsa, Fsa::Weight(threshold));
103  }
104  nbl = Fsa::nbest(Fsa::cache(pruned), targetNHypotheses_);
105 
106  /**
107  * Stop if number of hypotheses has not changed since the last iteration.
108  * It is very likely that the automaton does not contain more hypotheses.
109  */
110  Fsa::ConstStateRef hypotheses = nbl->getState(nbl->initialStateId());
111  bool hasChanged = nHypotheses != hypotheses->nArcs();
112  if (!hasChanged) {
113  // log("return on \"has-not-changed\"");
114  return Fsa::staticCopy(nbl);
115  }
116 
117  nHypotheses = hypotheses->nArcs();
118 
119  /**
120  * Stop if n-best list is complete. This is the case if the target number
121  * of hypotheses is reached and the least hypothesis has lower score than
122  * the relative pruning threshold.
123  */
124  bool hasEnoughHypotheses = nHypotheses == targetNHypotheses_;
125  Fsa::Weight leastWeight = nbl->getState(nbl->initialStateId())->getArc(nHypotheses - 1)->weight();
126  bool hasNBestHypotheses = sr->compare(leastWeight, sr->extend(minWeight, Fsa::Weight(threshold))) < 0;
127  if (hasEnoughHypotheses && hasNBestHypotheses) {
128  // log("return on \"enough-hypotheses\"");
129  return Fsa::staticCopy(nbl);
130  }
131  }
132 
133  // log("return on \"maximum-pruning-threshold\"");
134  return Fsa::staticCopy(nbl);
135 }

147 ConstWordLatticeRef NBestListExtractor::getNBestList(
148  ConstWordLatticeRef lattice) {
149  if (normalize_) {
150  Fsa::ConstAutomatonRef total = lattice->mainPart();
151  Fsa::Weight denNorm;
152  if (not hasFailArcs_)
153  Fsa::posterior64(Fsa::changeSemiring(total, Fsa::LogSemiring), denNorm);
154  else
155  Fsa::posterior4SpecialSymbols(Fsa::changeSemiring(total, Fsa::LogSemiring), denNorm);
156  Core::Application::us()->log() << Core::XmlFull("norm", f32(denNorm));
157  }
158 
159  if (targetNHypotheses_ == 1) {
160  return best(lattice, true);
161  }
162 
163  if (not hasFailArcs_)
164  lattice = removeEpsilons(lattice);
165  Fsa::ConstAutomatonRef total = lattice->mainPart();
166  require(total->semiring() == Fsa::TropicalSemiring);
167 
168  if (!workOnOutput_) {
169  total = mapLemmaPronToEval(Fsa::projectInput(total));
170  }
171  total = Fsa::projectOutput(total);
172  f32 threshold;
173  Fsa::ConstAutomatonRef nBestList = getNBestListWithoutWordBoundaries(total, threshold);
174  if (hasFailArcs_) { // or make compose able to handle *FAIL*
175  PruneAndRemoveFail pr(threshold);
176  lattice = apply(lattice, pr);
177  }
178  Core::Vector<Lattice::ConstWordLatticeRef> hypotheses;
179  Fsa::State::const_iterator hIt =
180  nBestList->getState(nBestList->initialStateId())->begin();
181  Fsa::State::const_iterator hEnd =
182  nBestList->getState(nBestList->initialStateId())->end();
183  for (; hIt != hEnd; ++hIt) {
184  Fsa::ConstAutomatonRef hf =
185  Fsa::partial(
186  nBestList,
187  hIt->target(),
188  hIt->weight());
189  if (!workOnOutput_) {
190  hf = mapEvalToLemmaPron(hf);
191  }
192  if (hf->initialStateId() == Fsa::InvalidStateId) {
193  Fsa::write(nBestList, "nBestList.fsa.gz");
194  Fsa::write(lattice->mainPart(), "lattice.fsa.gz");
195  Core::Application::us()->error()
196  << "Nbest extraction failed! Hypothesis " << hIt->target() << " results to empty partial automaton!"
197  << "For further diagnosis the n best list automaton is saved to nBestList.fsa.gz"
198  << " and the input lattice is saved to lattice.fsa.gz";
199  }
200  hf = Fsa::multiply(hf, Fsa::Weight(0));
201  Lattice::ConstWordLatticeRef composed = composeMatching(lattice, hf);
202  if (composed->mainPart()->initialStateId() == Fsa::InvalidStateId) {
203  Fsa::write(nBestList, "nBestList.fsa.gz");
204  Fsa::write(lattice->mainPart(), "lattice.fsa.gz");
205  Core::Application::us()->error()
206  << "Nbest extraction failed! Compose with partial automaton of hypothesis " << hIt->target() << " results to empty compose automaton!"
207  << "\n"
208  << "For further diagnosis the n best list automaton is saved to nBestList.fsa.gz"
209  << " and the input lattice is saved to lattice.fsa.gz";
210  }
211  Lattice::ConstWordLatticeRef bestR = best(composed);
212  hypotheses.push_back(
213  staticCopy(normalize(bestR)));
214  }
215  return staticCopy(unite(hypotheses));
216 }

233 ConstWordLatticeRef best(ConstWordLatticeRef lattice, bool listFormat) {
234  require(lattice->nParts() == 1);

237  if (listFormat) {
