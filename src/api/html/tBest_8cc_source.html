65  /**
66  * \warning best() does not work when there are non-trivial zero-weight loops.
67  */
68  virtual _ConstStateRef getState(Fsa::StateId s) const {
69  _ConstStateRef _sp = Precursor::fsa_->getState(s);
70  _State* sp = new _State(_sp->id(), _sp->tags(), _sp->weight_);
71  typename _State::const_iterator bestArc = _sp->end();
72  _Weight minWeight = Precursor::fsa_->semiring()->max();
73  for (typename _State::const_iterator a = _sp->begin(); a != _sp->end(); ++a) {
74  if (a->target() == s)
75  continue;
76  _Weight w = Precursor::fsa_->semiring()->extend(a->weight_, potentials_[a->target()]);
77  if (Precursor::fsa_->semiring()->compare(w, minWeight) < 0) {
78  minWeight = w;
79  bestArc = a;
80  }
81  }
82  if (sp->isFinal()) {
83  if (Precursor::fsa_->semiring()->compare(sp->weight_, minWeight) < 0) {
84  minWeight = sp->weight_;
85  bestArc = _sp->end();
86  }
87  }
88  if (bestArc != _sp->end()) {
89  sp->unsetFinal();
90  *sp->newArc() = *bestArc;
91  }
92  return _ConstStateRef(sp);
93  }
94  virtual std::string describe() const {
95  return "best(" + Precursor::fsa_->describe() + ")";
96  }

166  (*this)[t].predecessor_ = freelist_;

195  struct PriorityFunction {
196  _ConstSemiringRef semiring_;
197  const _StatePotentials& potentials_;
198  PriorityFunction(_ConstSemiringRef semiring, const _StatePotentials& potentials)
199  : semiring_(semiring), potentials_(potentials) {}
200  bool operator()(const Hyp& h1, const Hyp& h2) const {
201  return (semiring_->compare(semiring_->extend(h1.potential_, potentials_[h1.state_]),
202  semiring_->extend(h2.potential_, potentials_[h2.state_])) < 0);
203  }
204  };
205 
206 private:
207  TraceCache traces_;
208  Core::Vector<_Weight> finals_;
209 
210 public:
211  NBestTraces(_ConstAutomatonRef f, size_t n, bool bestSequences)
212  : traces_(n), finals_() {
213  // check for tropical semiring (or for at least a semiring with property 'ordered')
214  if (bestSequences) {
215  if (f->type() & Fsa::TypeAcceptor)
216  f = cache<_Automaton>(removeEpsilons<_Automaton>(f));
217  f = determinize<_Automaton>(f);
218  }
219  _ConstAutomatonRef f2 = transpose<_Automaton>(f); // also generates a single final state
220  _StatePotentials potentials(sssp<_Automaton>(f, false));
221  _ConstSemiringRef semiring = f2->semiring();
222  Core::PriorityQueue<Hyp, PriorityFunction> S(PriorityFunction(semiring, potentials));
223  Core::Vector<size_t> r_;
224 
225  finals_.reserve(n + 1);
226  finals_.push_back(semiring->one());
227  S.insert(Hyp(f2->initialStateId(), semiring->one(), traces_.newTrace(Trace())));
228  while (!S.empty()) {
229  Hyp h = S.top();
230  S.pop();
231  _ConstStateRef sp = f2->getState(h.state_);
232  if (sp->isFinal()) {
233  traces_[finals_.size()] = traces_[h.trace_];
234  finals_.push_back(semiring->extend(h.potential_, sp->weight_));
235  if (finals_.size() == n + 1)
236  break; // stop after n final traces
237  }
238  // there can only be n distinct paths through a single state
239  r_.grow(h.state_, 0);
240  if (++r_[h.state_] <= n)
241  for (typename _State::const_iterator a = sp->begin(); a != sp->end(); ++a)
242  S.insert(Hyp(a->target(), semiring->extend(h.potential_, a->weight()),
243  traces_.newTrace(Trace(a->input(), a->output(), h.trace_))));
244  traces_.deleteTrace(h.trace_);
245  }
246  }
247  size_t nTraces() const {
248  return traces_.size();
249  }
250  const Trace& trace(size_t n) const {
251  require(n < traces_.size());
252  return traces_[n];
253  }
254  size_t size() const {
255  return finals_.size() - 1;
256  }
257  const _Weight& weight(size_t n) const {
258  require(n < finals_.size());
259  return finals_[n];
260  }
261  size_t getMemoryUsed() const {
262  return traces_.getMemoryUsed() + finals_.getMemoryUsed();
263  }
264  void dumpMemoryUsage(Core::XmlWriter& o) const {
265  o << Core::XmlFull("traces", traces_.getMemoryUsed())
266  << Core::XmlFull("weights", finals_.getMemoryUsed());
267  }

327  }

409  TraceId search(TracePool& traces) {
410  typename Hyp::Priority precedes(this->semiring());
411  Core::TracedPriorityQueue<Hyp, Fsa::StateId, typename Hyp::Key, typename Hyp::Priority>
412  stack(precedes);
413  std::unordered_map<Fsa::StateId, _Weight> closed;
414 
415  Hyp newHyp;
416  newHyp.state = Precursor::fsa_->initialStateId();
417  newHyp.weight = this->semiring()->one();
418  newHyp.trace = 0;
419  stack.insert(newHyp);
420 
421  while (!stack.empty()) {
422  Hyp current = stack.top();
423  if (current.state & Fsa::StateTagFinal)
424  break;
425 
426  verify(closed.find(current.state) == closed.end());
427  closed[current.state] = current.weight;
428 
429  stack.pop();
430 
431  _ConstStateRef state = Precursor::fsa_->getState(current.state);
432  for (typename _State::const_iterator aa = state->begin(); aa != state->end(); ++aa) {
433  newHyp.state = aa->target();
434  newHyp.weight = this->semiring()->extend(current.weight, aa->weight());
435  if (closed.find(newHyp.state) == closed.end()) {
436  newHyp.trace = traces.newTrace(current.trace, current.state, aa - state->begin());
437  stack.insertOrRelax(newHyp);
438  }
439  }
440  if (state->isFinal()) {
441  newHyp.state = current.state | Fsa::StateTagFinal;
442  newHyp.weight = this->semiring()->extend(current.weight, state->weight_);
443  if (closed.find(newHyp.state) == closed.end()) {
444  newHyp.trace = traces.newTrace(current.trace, current.state, -1);
445  stack.insertOrRelax(newHyp);
446  }
447  }
448  }
449 
450  if (stack.empty())
451  return 0;
452 
453  return stack.top().trace;
454  }
