155  virtual ConstStateRef getState(Fsa::StateId sid) const {
156  std::vector<Fsa::ConstStateRef>::iterator itSr = tmpStates_.begin();
157  std::vector<Fsa::State::const_iterator>::iterator itA = tmpArcIterators_.begin();
158  for (std::vector<Fsa::ConstAutomatonRef>::const_iterator itFsa = fsas_.begin(); itFsa != fsas_.end(); ++itFsa, ++itSr, itA++) {
159  *itSr = (*itFsa)->getState(sid);
160  *itA = (*itSr)->begin();
161  }
162  State* sp = new State(sid);
163  if (tmpStates_.front()->isFinal()) {
164  ScoresRef scores = semiring_->create();
165  sp->setFinal(scores);
166  Scores::iterator itScore = scores->begin();
167  for (std::vector<Fsa::ConstStateRef>::const_iterator itSr = tmpStates_.begin(); itSr != tmpStates_.end(); ++itSr, ++itScore)
168  *itScore = Score((*itSr)->weight());
169  }
170  for (Fsa::State::const_iterator a = tmpArcIterators_.front(), a_end = tmpStates_.front()->end(); a != a_end; ++a) {
171  ScoresRef scores = semiring_->create();
172  sp->newArc(a->target(), scores, a->input(), a->output());
173  Scores::iterator itScore = scores->begin();
174  for (std::vector<Fsa::State::const_iterator>::iterator itA = tmpArcIterators_.begin(); itA != tmpArcIterators_.end(); ++itA, ++itScore) {
175  *itScore = Score((*itA)->weight());
176  ++(*itA);
177  }
178  }
179  return ConstStateRef(sp);
180  }

341 class FindTopologicalOrder : public TopologicalOrderBuilder {
342 private:
343  Fsa::StateId time_;
344 
345 protected:
346  virtual void finishState(Fsa::StateId sid) {
347  ensure(sid != Fsa::InvalidStateId);
348  map_->grow(sid, Fsa::InvalidStateId);
349  (*map_)[sid] = time_++;
350  }
351 
352 public:
353  FindTopologicalOrder(ConstLatticeRef l, Core::Ref<StateMap> map)
354  : TopologicalOrderBuilder(l, map), time_(0) {
355  dfs();
356  if (time_ > 0) {
357  if (!isCyclic()) {
358  --time_;
359  for (StateMap::iterator itOrder = map_->begin(); itOrder != map_->end(); ++itOrder)
360  if (*itOrder != Fsa::InvalidStateId)
361  *itOrder = time_ - *itOrder;
362  map_->maxSid = map_->size() - 1;
363  }
364  else
365  map_->clear();
366  }
367  }
368  virtual ~FindTopologicalOrder() {}
369 };
370 ConstStateMapRef findTopologicalOrder(ConstLatticeRef l) {
371  Core::Ref<StateMap> map;
372  if (!l->getTopologicalSort()) {
373  map = Core::Ref<StateMap>(new StateMap);
374  FindTopologicalOrder(l, map);
375  if (map->maxSid == Fsa::InvalidStateId)
376  map.reset();
377  }
378  else {
379  const StateMap& topologicalSort = *l->getTopologicalSort();
380  if (topologicalSort.maxSid == Fsa::InvalidStateId) {
381  map = Core::Ref<StateMap>(new StateMap(0));
382  map->maxSid = Fsa::InvalidStateId;
383  }
384  else {
385  map = Core::Ref<StateMap>(new StateMap(topologicalSort.maxSid + 1, Fsa::InvalidStateId));
386  StateMap& topologicalOrder = *map;
387  Fsa::StateId order = 0;
388  for (StateMap::const_iterator itSid = topologicalSort.begin(), endSid = topologicalSort.end();
389  itSid != endSid; ++itSid, ++order)
390  topologicalOrder[*itSid] = order;
391  topologicalOrder.maxSid = topologicalSort.maxSid;
392  }
393  }
394  return map;
395 }

397 struct ChronologicalWeakOrder {
398  const Boundaries& boundaries;
399  const StateMap& topologicalOrder;
400  bool operator()(Fsa::StateId sid1, Fsa::StateId sid2) const {
401  if (boundaries.time(sid1) == boundaries.time(sid2))
402  return topologicalOrder[sid1] < topologicalOrder[sid2];
403  else
404  return boundaries.time(sid1) < boundaries.time(sid2);
405  }
406  ChronologicalWeakOrder(const Boundaries& boundaries, const StateMap& topologicalOrder)
407  : boundaries(boundaries), topologicalOrder(topologicalOrder) {}
408 };
409 ConstStateMapRef sortChronologically(ConstLatticeRef l) {
410  ConstStateMapRef topologicalSort = sortTopologically(l);
411  verify(topologicalSort);
412  StateMap* chronologicalSort = new StateMap(*topologicalSort);
413  // a stable sort w/o taking the topological order (explicitly) into account should be enough here
414  ConstStateMapRef topologicalOrder = findTopologicalOrder(l);
415  ChronologicalWeakOrder lessThan(*l->getBoundaries(), *topologicalOrder);
416  std::stable_sort(chronologicalSort->begin(), chronologicalSort->end(), lessThan);
417  return ConstStateMapRef(chronologicalSort);
418 }

516 class ScoreAppendLattice : public SlaveLattice {
517  typedef SlaveLattice Precursor;
518 
519 private:
520  ConstLatticeRef appendL_;
521  u32 n1_, n2_;
522  ConstSemiringRef semiring_;
523 
524 protected:
525  ScoresRef concatenate(const Scores& scores1, const Scores& scores2) const {
526  ScoresRef scores = semiring_->create();
527  Scores::iterator itScore = scores->begin();
528  for (Scores::const_iterator itScore1 = scores1.begin(), endScore1 = scores1.begin() + n1_;
529  itScore1 != endScore1; ++itScore1, ++itScore)
530  *itScore = *itScore1;
531  for (Scores::const_iterator itScore2 = scores2.begin(), endScore2 = scores2.begin() + n2_;
532  itScore2 != endScore2; ++itScore2, ++itScore)
533  *itScore = *itScore2;
534  return scores;
535  }
536 
537 public:
538  ScoreAppendLattice(ConstLatticeRef l1, ConstLatticeRef l2, ConstSemiringRef semiring)
539  : Precursor(l1), appendL_(l2), semiring_(semiring) {
540  const Semiring& semiring1 = *l1->semiring();
541  n1_ = semiring1.size();
542  const Semiring& semiring2 = *l2->semiring();
543  n2_ = semiring2.size();
544  if (!semiring) {
545  verify(semiring1.type() == semiring2.type());
546  ScoreList scales(semiring1.scales());
547  scales.insert(scales.end(), semiring2.scales().begin(), semiring2.scales().end());
548  KeyList keys(semiring1.keys());
549  keys.insert(keys.end(), semiring2.keys().begin(), semiring2.keys().end());
550  semiring_ = Semiring::create(semiring1.type(), n1_ + n2_, scales, keys);
551  }
552  else
553  verify(semiring_->size() == n1_ + n2_);
554  }
555  virtual ~ScoreAppendLattice() {}
556 
557  ConstSemiringRef semiring() const {
558  return semiring_;
559  }
560 
561  ConstStateRef getState(Fsa::StateId sid) const {
562  State* sp = new State(*fsa_->getState(sid));
563  ConstStateRef appendSp = appendL_->getState(sid);
564  verify(appendSp);
565  if (sp->isFinal()) {
566  verify(appendSp->isFinal());
567  sp->weight_ = concatenate(*sp->weight(), *appendSp->weight());
568  }
569  State::const_iterator a_append = appendSp->begin(), end_append = appendSp->end();
570  for (State::iterator a = sp->begin(), end = sp->end(); a != end; ++a, ++a_append) {
571  verify(a_append != end_append);
572  verify(a->target() == a_append->target());
573  verify_(a->input() == a_append->input());
574  verify_(a->output() == a_append->output());
575  a->weight_ = concatenate(*a->weight(), *a_append->weight());
576  }
577  return ConstStateRef(sp);
578  }
579  std::string describe() const {
580  return Core::form("append-score(%s,%s)", fsa_->describe().c_str(), appendL_->describe().c_str());
581  }
582 };
583 ConstLatticeRef appendScores(ConstLatticeRef l1, ConstLatticeRef l2, ConstSemiringRef semiring) {
584  return ConstLatticeRef(new ScoreAppendLattice(l1, l2, semiring));
