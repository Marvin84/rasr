236 void ClassicTransducerBuilder::setupWordStart(Fsa::State* s, const PhoneBoundaryStateDescriptor& pbsd) {
237  if (pbsd.context.history.empty() && pbsd.context.future.empty() && pbsd.flag == wordStart)
238  // This should be unique.

241  s->addTags(Fsa::StateTagFinal);

246 void ClassicTransducerBuilder::setupWordEnd(Fsa::State* s, const PhoneBoundaryStateDescriptor& pbsd) {
247  if (acceptCoarticulatedSinglePronunciation_) {
248  s->addTags(Fsa::StateTagFinal);

259 Fsa::State* ClassicTransducerBuilder::phoneStartState(
260  const Allophone* allophone, bool looseCtxMark) {
261  PhoneBoundaryStateDescriptor pbsd;
262  pbsd.context = allophone->context();
263  model_->phonology()->pushFuture(pbsd.context, allophone->phoneme());
264  pbsd.flag = 0;
265  if (allophone->boundary & Allophone::isInitialPhone) {
266  pbsd.flag |= wordStart;
267  if (!pbsd.context.history.empty() && acceptCoarticulatedSinglePronunciation_)
268  pbsd.context.history.clear();
269  if (looseCtxMark)
270  pbsd.flag |= loosesCtx;
271  }
272  /**
273  * Note: non-coarticulated phonemes mostly occur as initial/final phoneme (e.i. word start/end)
274  * but there exit seldom non-coarticulated within word phonemes as well (e.g. more
275  * than one subsequent noise phonemes).
276  */
277  if (pbsd.context.history.empty())
278  pbsd.context.future.clear();
279  return phoneBoundaryState(pbsd);
280 }

282 Fsa::State* ClassicTransducerBuilder::phoneEndState(const Allophone* allophone) {
283  PhoneBoundaryStateDescriptor pbsd;
284  pbsd.context = allophone->context();
285  model_->phonology()->pushHistory(pbsd.context, allophone->phoneme());
286  pbsd.flag = 0;
287  if (allophone->boundary & Allophone::isFinalPhone) {
288  pbsd.flag |= wordEnd;
289  if (!pbsd.context.future.empty() && acceptCoarticulatedSinglePronunciation_)
290  pbsd.context.future.clear();
291  if (fixAllophoneContextAtWordBoundaries_ && pbsd.context.future.empty() && model_->phonemeInventory()->phoneme(allophone->phoneme(0))->isContextDependent())
292  // A context dependend allophone which has future context empty must either be the sequence end
293  // or followed by a context independent allophone (like silence).
294  pbsd.flag |= loosesCtx;
295  }
296  /**
297  * Note: non-coarticulated phonemes mostly occur as initial/final phoneme (e.i. word start/end)
298  * but there exit seldom non-coarticulated within word phonemes as well (e.g. more
299  * than one subsequent noise phonemes).
300  */
301  if (pbsd.context.future.empty())
302  pbsd.context.history.clear();
303  return phoneBoundaryState(pbsd);
304 }

306 void ClassicTransducerBuilder::buildAllophone(const Allophone* allophone) {
307  if (filterOutInvalidAllophones_) {
308  // Filter out some invalid allophones.
309  // They might occur if you use *.allophones.add-all=true.
310  // Note that this is important for a proper allophone-state-to-phoneme-transducer,
311  // e.g. consider reading:r eh ..., the transducer would have
312  // r{#+#}@i:r eh{#+#}@i:eh ... and r{#+eh}:r eh{r+.}:eh ...
313  // and thus will accept an invalid allophone sequence.
314  if (model_->phonemeInventory()->phoneme(allophone->phoneme(0))->isContextDependent()) {
315  if (model_->phonology()->maximumHistoryLength() > 0 && allophone->context().history.empty())
316  if (!(allophone->boundary & Allophone::isInitialPhone))
317  return;
318  if (model_->phonology()->maximumFutureLength() > 0 && allophone->context().future.empty())
319  if (!(allophone->boundary & Allophone::isFinalPhone))
320  return;
321  }
322  }
323  // nDisambiguators_ effects the application of the transition model.
324  require(!acceptCoarticulatedSinglePronunciation_ || nDisambiguators_ == 0);
325  buildAllophone(allophone, phoneStartState(allophone, false), phoneEndState(allophone));
326  if (fixAllophoneContextAtWordBoundaries_ && (allophone->boundary & Allophone::isInitialPhone) && !model_->phonemeInventory()->phoneme(allophone->phoneme(0))->isContextDependent()) { // e.g. silence
327  require(allophone->context().history.empty());
328  require(allophone->context().future.empty());
329  buildAllophone(allophone, phoneStartState(allophone, true), phoneEndState(allophone));
330  }
331 }

340 void ClassicTransducerBuilder::buildAllophone(const Allophone* allophone,
341  Fsa::State* start, Fsa::State* end) {
342  verify(allophone);
343  switch (inputType_) {
344  case inputAcceptsAllophones: {
345  const Bliss::Phoneme* phoneme = model_->phonemeInventory()->phoneme(
346  allophone->phoneme());
347  verify(phoneme);
348  verify(product_->inputAlphabet() == allophones_);
349  start->newArc(end->id(), product_->semiring()->one(),
350  allophones_->index(allophone), phoneme->id());
351  } break;
352  case inputAcceptsAllophoneStates:
353  case inputAcceptsEmissionLabels: {
354  buildAllophoneStates(allophone, start, end);

356  default:

374 size_t ClassicTransducerBuilder::hashSequence(const AllophoneSuffix& as) {
375  size_t result = as.target;
376  for (s16 state = as.allophoneState.state(); state < as.hmmTopology->nPhoneStates(); ++state) {
377  result = (result << 11) ^ (result >> 21);
378  result ^= size_t(allophoneStateLabel(allophoneStates_->allophoneState(
379  as.allophoneState.allophone(), state)));
380  }
381  result += as.hmmTopology->nSubStates();
382  result += as.subState;
383  return result;
384 }

386 int ClassicTransducerBuilder::compareSequences(const AllophoneSuffix& ll,
387  const AllophoneSuffix& rr) {
388  if (ll.target != rr.target)
389  return (ll.target < rr.target) ? -1 : 1;
390  if (ll.subState != rr.subState)
391  return (ll.subState < rr.subState) ? -1 : 1;
392  if (ll.hmmTopology->nSubStates() != rr.hmmTopology->nSubStates())
393  return (ll.hmmTopology->nSubStates() < rr.hmmTopology->nSubStates()) ? -1
394  : 1;
395  s16 lstate = ll.allophoneState.state();
396  s16 rstate = rr.allophoneState.state();
397  for (;;) {
398  if (lstate == ll.hmmTopology->nPhoneStates())
399  return (rstate == rr.hmmTopology->nPhoneStates()) ? 0 : 1;
400  if (rstate == rr.hmmTopology->nPhoneStates())
401  return -1;
402  Fsa::LabelId llLabel = allophoneStateLabel(
403  allophoneStates_->allophoneState(ll.allophoneState.allophone(),
404  lstate));
405  Fsa::LabelId rrLabel = allophoneStateLabel(
406  allophoneStates_->allophoneState(rr.allophoneState.allophone(),
407  rstate));
408  if (llLabel != rrLabel)
409  return (llLabel < rrLabel) ? -1 : 1;
410  ++lstate;
411  ++rstate;
412  }
413  return 0;
414 }

416 void ClassicTransducerBuilder::buildAllophoneStates(const Allophone* allophone,
417  Fsa::State* start, Fsa::State* end) {
418  verify((product_->outputAlphabet() == phonemes_) || (product_->outputAlphabet() == allophones_));
419  verify((product_->inputAlphabet() == allophoneStates_) || (product_->inputAlphabet() == emissions_));
420 
421  const Bliss::Phoneme* phoneme = model_->phonemeInventory()->phoneme(
422  allophone->phoneme());
423  verify(phoneme);
424 
425  AllophoneSuffix as;
426  as.hmmTopology = model_->hmmTopology(phoneme->id());
427  as.allophoneState = allophoneStates_->allophoneState(allophone, 0);
428  as.subState = 0;
429  as.target = end->id();
430 
431  Fsa::LabelId output = Fsa::Epsilon;
432  if (product_->outputAlphabet() == phonemes_)
433  output = phoneme->id();
434  else if (product_->outputAlphabet() == allophones_)
435  output = allophones_->index(allophone);
436 
437  Fsa::State* from = start;
438  while (from) {
439  Fsa::Arc* aa = from->newArc();
440  aa->weight_ = product_->semiring()->one();
441  aa->input_ = allophoneStateLabel(as.allophoneState);
442  aa->output_ = output;
443  output = Fsa::Epsilon;
444  if (++as.subState >= as.hmmTopology->nSubStates()) {
445  as.subState = 0;
446  as.allophoneState = allophoneStates_->allophoneState(
447  as.allophoneState.allophone(), as.allophoneState.state() + 1);
448  }
449  if (as.allophoneState.state() >= as.hmmTopology->nPhoneStates()) {
450  aa->target_ = end->id();
451  break;
452  }
453  else {
454  AllophoneSuffixMap::const_iterator ti = allophoneSuffixes_.find(as);
455  if (ti != allophoneSuffixes_.end()) {
456  aa->target_ = ti->second;
457  break;
458  }
459  }
460  aa->target_ = (from = product_->newState())->id();
461  allophoneSuffixes_.insert(std::make_pair(as, aa->target_));
462  }
463 }

465 void ClassicTransducerBuilder::buildWordBoundaryLinks(Fsa::State* from,
466  Fsa::State* to) {
467  verify((product_->outputAlphabet() == phonemes_) || (product_->outputAlphabet() == allophones_));
468  if (nDisambiguators_) {
469  for (u32 d = 0; d < nDisambiguators_; ++d) {
470  Fsa::Arc* a = from->newArc();
471  a->target_ = to->id();
472  a->weight_ = product_->semiring()->one();
473  switch (inputType_) {
474  case inputAcceptsAllophones:
475  verify(product_->inputAlphabet() == allophones_);
476  a->input_ = allophones_->disambiguator(d);
477  break;
478  case inputAcceptsAllophoneStates:
479  verify(product_->inputAlphabet() == allophoneStates_);
480  a->input_ = allophoneStates_->disambiguator(d);
481  break;
482  case inputAcceptsEmissionLabels:
483  verify(product_->inputAlphabet() == emissions_);
484  a->input_ = emissions_->disambiguator(d);
485  break;
486  default:
487  defect();
488  }
489  if (product_->outputAlphabet() == phonemes_)
490  a->output_ = phonemes_->disambiguator(d);
491  else if (product_->outputAlphabet() == allophones_)
492  a->output_ = allophones_->disambiguator(d);
493  }
494  }
495  else {
496  from->newArc(to->id(), product_->semiring()->one(), Fsa::Epsilon,
497  Fsa::Epsilon);
498  }
499 }

545 Fsa::State* ClassicTransducerBuilder::buildSilenceAndNoiseLoops(
546  const Bliss::PhonemeInventory& phonemeInventory,
547  const Phonology& phonology, Fsa::State* s) {
548  allophoneStates_->disambiguator(0);
549  setDisambiguators(allophoneStates_->nDisambiguators());
550 
551  Fsa::State* finalState = product_->newState();
552  buildWordBoundaryLinks(s, finalState);
553 
554  PronunciationList::const_iterator silencePronunciationIt =
555  silencesAndNoises_->begin();
556  for (; silencePronunciationIt != silencesAndNoises_->end(); ++silencePronunciationIt) {
557  if ((*silencePronunciationIt)->length() == 1) {
558  Allophone allophone(phonology(**silencePronunciationIt, 0),
559  Allophone::isFinalPhone | Allophone::isInitialPhone);
560  verify(
561  !phonemeInventory.phoneme(allophone.phoneme())->isContextDependent());
562  buildAllophone(allophones_->allophone(allophone), finalState, s);
563  }
564  else {
565  require((*silencePronunciationIt)->length() == 0);
566  }
567  }
568  return finalState;
569 }

571 void ClassicTransducerBuilder::buildPronunciation(const Bliss::Coarticulated<
572  Bliss::Pronunciation>& coarticulatedPronunciation) {
573  // require(nDisambiguators_ == 0); // nDisambiguators_ effects the application of the transition model.
574  setDisambiguators(0);
575 
576  const Bliss::Pronunciation& p(coarticulatedPronunciation.object());
577  Bliss::Phoneme::Id leftContext = coarticulatedPronunciation.leftContext();
578  Bliss::Phoneme::Id rightContext = coarticulatedPronunciation.rightContext();
579 
580  const Phonology& phonology = *(model_->phonology().get());
581  const Bliss::PhonemeInventory& phonemeInventory =
582  *(model_->phonemeInventory().get());
583 
584  createEmptyTransducer();
585  Fsa::State* s = product_->newState();
586  product_->setInitialStateId(s->id());
587 
588  if (p.length() == 1) {
589  Allophone allophone(phonology(p, 0), Allophone::isInitialPhone | Allophone::isFinalPhone);
590  if (model_->isAcrossWordModelEnabled() && phonemeInventory.phoneme(
591  allophone.phoneme())
592  ->isContextDependent()) {
593  if (leftContext != Bliss::Phoneme::term)
594  phonology.appendHistory(allophone, leftContext);
595  if (rightContext != Bliss::Phoneme::term)
596  phonology.appendFuture(allophone, rightContext);
597  }
598  s = buildAllophone(allophones_->allophone(allophone), s);
599  }

603  Allophone::isInitialPhone);

605  phonology.appendHistory(initialAllophone, leftContext);
606  }

608  }

619  phonology.appendFuture(finalAllophone, rightContext);

623  }

775 /**
776  * - union of closure of reduced HMM state sequences for each (used, unique) allophone
777  * - add disambiguation symbols
778  * - still not determinized on state level! (would result in less efficient composition)
779  *
780  * state labels and their corresponding outgoing transition probabilities:
781  * entry-m1 the (hidden) entry state of word models
782  * silence the single, unique state of the silence model
783  * phone0 the first state of a phone segment
784  * phone1 the second state of a phone segment
785  *
786  * silence.exit = silence.forward - phone?.forward but why?
787  **/
788 Fsa::ConstAutomatonRef ClassicTransducerBuilder::createEmissionLoopTransducer(
789  bool transitionModel) {
790  Fsa::StaticAutomaton* a = new Fsa::StaticAutomaton();
791  a->setType(Fsa::TypeTransducer);
792  a->setSemiring(Fsa::TropicalSemiring);
793  Core::Ref<const AllophoneStateAlphabet> asa =
794  model_->allophoneStateAlphabet();
795  EmissionAlphabet* ea = 0;
796  if (transitionModel) {
797  ea = new EmissionAlphabet(model_->nEmissions());
798  a->setInputAlphabet(Fsa::ConstAlphabetRef(ea));
799  }
800  else
801  a->setInputAlphabet(Fsa::ConstAlphabetRef(asa));
802  Core::Ref<const AllophoneAlphabet> aa = model_->allophoneAlphabet();
803  a->setOutputAlphabet(aa);
804  Fsa::State* initial = a->newState();
805  a->setInitialStateId(initial->id());
806  initial->setFinal(a->semiring()->one());
807  MinimizedStateMap minimizedStates;
808  for (AllophoneAlphabet::AllophoneList::const_iterator ai =
809  allophoneList_->begin();
810  ai != allophoneList_->end(); ++ai) {
811  Fsa::State * from = initial, *to = 0, *fromSkip = 0;
812  const ClassicHmmTopology* hmmTopology = model_->hmmTopology((*ai)->central());
813  require(hmmTopology != 0);
814  int nPhoneStates = hmmTopology->nPhoneStates();
815  int nSubStates = hmmTopology->nSubStates();
816  int nStates = nPhoneStates * nSubStates;
817  // std::cout << aa->symbol(aa->index(*ai)) << " " << nStates << std::endl;
818 
819  const StateTransitionModel* tdp = 0;
820  if ((*ai)->central() == model_->silence_)
821  tdp = (*model_->transitionModel_)[TransitionModel::silence];
822  else
823  tdp = (*model_->transitionModel_)[TransitionModel::phone0];
824 
825  f32 fw = a->semiring()->one(); // constant in case of no transition model
826  for (int s = 0; s < nStates; ++s) {
827  AllophoneState as = asa->allophoneState(*ai, s / nSubStates);
828  bool isFirstArc = (s == 0);
829  bool isLastArc = (s == nStates - 1);
830  Mm::MixtureIndex input;
831  if (transitionModel)
832  input = model_->stateTyingRef_->classify(as);
833  else
834  input = asa->index(as);
835 
836  std::vector<Mm::MixtureIndex> hmm;
837  for (int i = s; i < nStates; ++i) {
838  AllophoneState _as = asa->allophoneState(*ai, i / nSubStates);
839  if (transitionModel)
840  hmm.push_back(model_->stateTyingRef_->classify(_as));
841  else
842  hmm.push_back(asa->index(_as));
843  }
844  MinimizedState ms(hmm);
845  MinimizedStateMap::iterator j = minimizedStates.find(ms);
846 
847  if (transitionModel) {
848  // find forward state
849  if (j == minimizedStates.end()) {
850  minimizedStates[ms] = to = a->newState();
851  // loop transition
852  to->newArc(to->id(), Fsa::Weight((*tdp)[StateTransitionModel::loop]), input, Fsa::Epsilon);
853  if (isLastArc)
854  to->newArc(initial->id(), a->semiring()->one(), Fsa::Epsilon, Fsa::Epsilon);
855  }
856  else {
857  to = j->second;
858  s = nStates;
859  }
860  if (isFirstArc) {
861  fw = (*(*model_->transitionModel_)[TransitionModel::entryM1])[StateTransitionModel::forward];
862  if ((*ai)->boundary & Allophone::isFinalPhone)
863  fw += (*tdp)[StateTransitionModel::exit];
864  }
865  else {
866  fw = (*tdp)[StateTransitionModel::forward];
867  }
868  // skip transition
869  if (((*tdp)[StateTransitionModel::skip] < Core::Type<StateTransitionModel::Score>::max) && (fromSkip)) {
870  fromSkip->newArc(to->id(), Fsa::Weight((*tdp)[StateTransitionModel::skip]), input, (fromSkip == initial) ? aa->index(*ai) : Fsa::Epsilon);
871  }
872  }
873  else {
874  if (j == minimizedStates.end()) {
875  if (isLastArc)
876  to = initial;
877  else
878  minimizedStates[ms] = to = a->newState();
879  }
880  else {
881  to = j->second;
882  s = nStates;
883  }
884  }
885  // forward
886  from->newArc(to->id(), Fsa::Weight(fw), input, isFirstArc ? aa->index(*ai) : Fsa::Epsilon);
887  fromSkip = from;
888  from = to;
889  }
890  }
891  for (u32 d = 0; d < nDisambiguators_; ++d)
892  if (transitionModel)
893  initial->newArc(initial->id(), a->semiring()->one(), ea->disambiguator(d), aa->disambiguator(d));
894  else
895  initial->newArc(initial->id(), a->semiring()->one(), asa->disambiguator(d), aa->disambiguator(d));
896  return Fsa::ConstAutomatonRef(a);
897 }

941 Fsa::ConstAutomatonRef ClassicTransducerBuilder::createMinimizedContextDependencyTransducer(

944

951  allophoneList_->begin();

953  const Allophone* a = *ai;
954  maxHistory = std::max(maxHistory, a->history().size());
955  maxFuture = std::max(maxFuture, a->future().size());

968  initialCoartPhones.erase(std::unique(initialCoartPhones.begin(),
969  initialCoartPhones.end()),

972  initialNonCoartPhones.erase(std::unique(initialNonCoartPhones.begin(),
973  initialNonCoartPhones.end()),
974  initialNonCoartPhones.end());

977  Phonology::Allophone(model_->silence_), 0));
978 
979  // ensure that code only runs with triphones
980  require(maxHistory <= 1);
981  require(maxFuture <= 1);

990  PhoneContext pc(maxHistory + maxFuture, true);

992  model_->isAcrossWordModelEnabled() ? Fsa::StateTagFinal
993  : Fsa::StateTagNone,

996  _c->setInitialStateId(initial->id());
997  if (model_->isAcrossWordModelEnabled()) {

999  for (size_t d = 0; d < nDisambiguators_; ++d)
1000  initial->newArc(initial->id(), semiring->one(),
1001  allophones_->disambiguator(d), phonemes_->disambiguator(d));
1002 
1003  // non-coart phones
1004  for (std::unordered_map<Bliss::Phoneme::Id, const Allophone*>::const_iterator
1005  i = nonCoartPhones.begin();
1006  i != nonCoartPhones.end(); ++i)
1007  initial->newArc(initial->id(), semiring->one(), allophones_->index(i->second), initialPhoneAndSilenceOffset + Fsa::LabelId(i->first));

1010  // The construction of C for within- and across-word models
1011  // differs slightly. In both cases we shift the allophone input by
1012  // one phone/arc in order to make the FSA deterministic right
1013  // away. The main difference occurs at word boundaries.
1014  //
1015  // Using within-word models only, always uses the open context at
1016  // word boundaries. This would lead to a major source of
1017  // non-determinism. As disambiguation symbols only appear at word
1018  // boundaries, we output the phone disambiguators first and then
1019  // consume the appropriate allophone disambiguator at an arc of
1020  // the target state.
1021  //
1022  // Using across-word models, problems at the word boundaries are
1023  // more severe. For a particular state a-b we can have 3 possible
1024  // extensions: a-b-c, a-b|c and a-b|#. Using the same
1025  // disambiguator trick from above results in a-b-c:c, a-b|c:#1 and
1026  // a-b|#:#1, but we can't easily disambiguate the later two
1027  // cases. Instead, we can use loops at each state to pass through
1028  // disambiguators. However, then the 3 cases become a-b-c:c,
1029  // a-b|c:c and a-b|#:c. This looks even more critical, but a-b-c
1030  // and a-b|c are easy to disambiguate by using special phone
1031  // symbols for the first phone of a word. So, we already have
1032  // a-b-c:c and a-b|c:c'. For a-b|# we can utilize the
1033  // approximation of a minimum of 1 silence frame in case of no
1034  // coarticulation. Note that this approximation doesn't help when
1035  // using the disambiguator-in-sequence trick. Another trick is to
1036  // not use the coarticulated silence case.
1037  //
1038  // Construction:
1039  //
1040  // States are labeled either with a-b or a|b, so for triphones we
1041  // have to hash over 2 phones + 1 flag per state, indicating
1042  // whether there is a word boundary between the phones or not.
1043  // Then there is 1 state per non-coarticulated phone p, marked by
1044  // #-p. Additionally, for within-word models we have states that
1045  // carry over disambiguator symbols. The final state is the
1046  // (non-coarticulated) silence history #|si.
1047  //
1048  // Each allophone a*b*c generates exactly one arc from a*b to b*c
1049  // usually labeled with a*b*c:c. Special cases are the entry
1050  // structure and word boundaries as described above.
1051 
1052  // we create as many final states as there are disambiguators
1053  // passing disambiguators from phone to allophones is integrated
1054  // in the construction disambiguators on phone sequences are used
1055  // to detect word boundaries
1056  //
1057  // ASSUMES: phones still completely set to Bliss::Phoneme::term
1058  // and boundary set to true!!!
1059  std::vector<NoCoartBoundary> nonCoartBoundaries;
1060 
1061  // cycle and exit:
1062  // for each allophone a-b-c:
1063  // - if c == #, we draw an arc to the final state f and an additional silence arc to #,si
1064  // - otherwise, we draw an arc from a,b to b,c
1065  for (AllophoneAlphabet::AllophoneList::const_iterator ai =
1066  allophoneList_->begin();
1067  ai != allophoneList_->end(); ++ai) {
1068  // create phone sequence from allophone
1069  Phones phones(maxHistory + 1 + maxFuture, Bliss::Phoneme::term);
1070  std::copy((*ai)->history().begin(), (*ai)->history().end(),
1071  phones.begin() + maxHistory - (*ai)->history().size());
1072  phones[maxHistory] = (*ai)->central();
1073  std::copy((*ai)->future().begin(), (*ai)->future().end(),
1074  phones.begin() + maxHistory + 1);
1075 
1076  // create arcs (and states) for allophone
1077  if (model_->isAcrossWordModelEnabled()) {
1078  // filter out:
1079  // - unsupported silences (none of #-si-#, #|si|#)
1080  // - non-coart phones
1081  if ((*ai)->central() == silenceAllophone->central()) {
1082  if ((((*ai)->history().size() > 0) && ((*ai)->future().size() == 0)) ||
1083  (((*ai)->history().size() == 0) && ((*ai)->future().size() > 0)))
1084  continue;
1085  if ((((*ai)->boundary & Allophone::isInitialPhone) && !((*ai)->boundary & Allophone::isFinalPhone)) ||
1086  (!((*ai)->boundary & Allophone::isInitialPhone) && ((*ai)->boundary & Allophone::isFinalPhone)))
1087  continue;
1088  }
1089  if (((*ai)->boundary & Allophone::isInitialPhone) && (nonCoartPhones.find((*ai)->central()) != nonCoartPhones.end()))
1090  continue;
1091 
1092  // create "from" state
1093  Fsa::StateId sFrom = stateMap.insert(PhoneContext(phones.begin(), phones.end() - 1, (*ai)->boundary & Allophone::isInitialPhone));
1094  if (!_c->state(sFrom)) {
1095  _c->setState(new Fsa::State(sFrom));
1096 
1097  // generic disambiguator loop
1098  for (size_t d = 0; d < nDisambiguators_; ++d)
1099  _c->state(sFrom)->newArc(sFrom, semiring->one(),
1100  allophones_->disambiguator(d),
1101  phonemes_->disambiguator(d));
1102 
1103  // special treatment for coart (within-word) silence
1104  if ((!(*ai)->boundary & Allophone::isInitialPhone) && (*ai)->central() == silenceAllophone->central() && ((*ai)->history().size() == 0)) {
1105  for (std::vector<Bliss::Phoneme::Id>::const_iterator p = initialCoartPhones.begin(); p != initialCoartPhones.end(); ++p) {
1106  PhoneContext pc(maxHistory + maxFuture, false);
1107  pc[maxHistory] = Fsa::LabelId(*p);
1108  _c->state(sFrom)->newArc(stateMap.insert(pc), semiring->one(), allophones_->index(silenceAllophone), Fsa::LabelId(*p));
1109  }
1110  }
1111  }
1112  Fsa::State* sp = _c->state(sFrom).get();
1113 
1114  // create "to" states and arcs
1115  if ((*ai)->boundary & Allophone::isFinalPhone) {
1116  // Two cases here: right context is either term or
1117  // phone. In case of term we output non-coarticulated
1118  // phones, in case of phone we mark it as word-initial
1119  // phone. The third case mentioned above is handled
1120  // already by the outer else case.
1121  if (((*ai)->future().size() == 0) || ((*ai)->future()[0] == Bliss::Phoneme::term))
1122  sp->newArc(initial->id(), semiring->one(),
1123  allophones_->index(*ai), Fsa::Epsilon);
1124  else
1125  sp->newArc(stateMap.insert(PhoneContext(phones.begin() + 1, phones.end(), true)),
1126  semiring->one(),
1127  allophones_->index(*ai),
1128  initialPhoneAndSilenceOffset + phones[maxHistory + 1]);
1129  }
1130  else {
1131  // within-phrase word boundary
1132  if (((*ai)->future().size() == 0) || ((*ai)->future()[0] == Bliss::Phoneme::term)) {
1133  // to state marked #-si
1134  if ((*ai)->central() != silenceAllophone->central()) {
1135  PhoneContext pc(maxHistory + maxFuture, false);
1136  pc[maxHistory] = silenceAllophone->central();
1137  sp->newArc(stateMap.insert(pc), semiring->one(),
1138  allophones_->index(*ai),
1139  silenceAllophone->central());
1140  }
1141  else
1142  sp->newArc(sFrom, semiring->one(), allophones_->index(*ai), silenceAllophone->central());
