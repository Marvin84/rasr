119  /** LinearInterpolationInverseItem creates values of the interval (first .. end)
120  * at a relative position by linear interpolation.
121  */
122  class LinearInterpolationInverseItem : public InverseItem {
123  u32 leftIndex_;
124  /** relative position in the interval [leftIndex_..leftIndex_+1] */
125  Data relativePosition_;
126 
127  public:
128  LinearInterpolationInverseItem(u32 first, u32 last, f64 relativePosition) {
129  require(first < last);
130  require(relativePosition > 0 && relativePosition < 1.0);
131  f64 position = first + f64(last - first) * relativePosition;
132  leftIndex_ = (u32)floor(position);
133  relativePosition_ = position - leftIndex_;
134  }
135  virtual ~LinearInterpolationInverseItem() {}
136 
137  virtual Data apply(const std::vector<Data>& in) const {
138  verify_(leftIndex_ + 1 < in.size());
139  return in[leftIndex_] * (Data(1) - relativePosition_) +
140  in[leftIndex_ + 1] * relativePosition_;
141  }
142  };
