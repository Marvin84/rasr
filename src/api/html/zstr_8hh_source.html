45 /// Exception class thrown by failed zlib operations.
46 class Exception
47  : public std::exception {
48 public:
49  Exception(z_stream* zstrm_p, int ret)
50  : _msg("zlib: ") {
51  switch (ret) {
52  case Z_STREAM_ERROR: _msg += "Z_STREAM_ERROR: "; break;
53  case Z_DATA_ERROR: _msg += "Z_DATA_ERROR: "; break;
54  case Z_MEM_ERROR: _msg += "Z_MEM_ERROR: "; break;
55  case Z_VERSION_ERROR: _msg += "Z_VERSION_ERROR: "; break;
56  case Z_BUF_ERROR: _msg += "Z_BUF_ERROR: "; break;
57  default:
58  std::ostringstream oss;
59  oss << ret;
60  _msg += "[" + oss.str() + "]: ";
61  break;
62  }
63  _msg += zstrm_p->msg;
64  }
65  Exception(const std::string msg)
66  : _msg(msg) {}
67  const char* what() const noexcept {
68  return _msg.c_str();
69  }
70 
71 private:
72  std::string _msg;
73 }; // class Exception

77 class z_stream_wrapper
78  : public z_stream {
79 public:
80  z_stream_wrapper(bool _is_input = true, int _level = Z_DEFAULT_COMPRESSION)
81  : is_input(_is_input) {
82  this->zalloc = Z_NULL;
83  this->zfree = Z_NULL;
84  this->opaque = Z_NULL;
85  int ret;
86  if (is_input) {
87  this->avail_in = 0;
88  this->next_in = Z_NULL;
89  ret = inflateInit2(this, 15 + 32);
90  }
91  else {
92  ret = deflateInit2(this, _level, Z_DEFLATED, 15 + 16, 8, Z_DEFAULT_STRATEGY);
93  }
94  if (ret != Z_OK)
95  throw Exception(this, ret);
96  }
97  ~z_stream_wrapper() {
98  if (is_input) {
99  inflateEnd(this);
100  }
101  else {
102  deflateEnd(this);
103  }
104  }
105 
106 private:
107  bool is_input;
108 }; // class z_stream_wrapper

114 public:
115  istreambuf(std::streambuf* _sbuf_p, std::size_t _buff_size = default_buff_size, bool _auto_detect = true)

142  virtual std::streambuf::int_type underflow() {
143  if (this->gptr() == this->egptr()) {
144  // pointers for free region in output buffer
145  char* out_buff_free_start = out_buff;
146  do {
147  // read more input if none available
148  if (in_buff_start == in_buff_end) {
149  // empty input buffer: refill from the start
150  in_buff_start = in_buff;
151  std::streamsize sz = sbuf_p->sgetn(in_buff, buff_size);
152  in_buff_end = in_buff + sz;
153  if (in_buff_end == in_buff_start)
154  break; // end of input
155  }
156  // auto detect if the stream contains text or deflate data
157  if (auto_detect && !auto_detect_run) {
158  auto_detect_run = true;
159  unsigned char b0 = *reinterpret_cast<unsigned char*>(in_buff_start);
160  unsigned char b1 = *reinterpret_cast<unsigned char*>(in_buff_start + 1);
161  // Ref:
162  // http://en.wikipedia.org/wiki/Gzip
163  // http://stackoverflow.com/questions/9050260/what-does-a-zlib-header-look-like
164  is_text = !(in_buff_start + 2 <= in_buff_end && ((b0 == 0x1F && b1 == 0x8B) // gzip header
165  || (b0 == 0x78 && (b1 == 0x01 // zlib header
166  || b1 == 0x9C || b1 == 0xDA))));
167  }
168  if (is_text) {
169  // simply swap in_buff and out_buff, and adjust pointers
170  assert(in_buff_start == in_buff);
171  std::swap(in_buff, out_buff);
172  out_buff_free_start = in_buff_end;
173  in_buff_start = in_buff;
174  in_buff_end = in_buff;
175  }
176  else {
177  // run inflate() on input
178  if (!zstrm_p)
179  zstrm_p = new detail::z_stream_wrapper(true);
180  zstrm_p->next_in = reinterpret_cast<decltype(zstrm_p->next_in)>(in_buff_start);
181  zstrm_p->avail_in = in_buff_end - in_buff_start;
182  zstrm_p->next_out = reinterpret_cast<decltype(zstrm_p->next_out)>(out_buff_free_start);
183  zstrm_p->avail_out = (out_buff + buff_size) - out_buff_free_start;
184  int ret = inflate(zstrm_p, Z_NO_FLUSH);
185  // process return code
186  if (ret != Z_OK && ret != Z_STREAM_END)
187  throw Exception(zstrm_p, ret);
188  // update in&out pointers following inflate()
189  in_buff_start = reinterpret_cast<decltype(in_buff_start)>(zstrm_p->next_in);
190  in_buff_end = in_buff_start + zstrm_p->avail_in;
191  out_buff_free_start = reinterpret_cast<decltype(out_buff_free_start)>(zstrm_p->next_out);
192  assert(out_buff_free_start + zstrm_p->avail_out == out_buff + buff_size);
193  // if stream ended, deallocate inflator
194  if (ret == Z_STREAM_END) {
195  delete zstrm_p;
196  zstrm_p = nullptr;
197  }
198  }
199  } while (out_buff_free_start == out_buff);
200  // 2 exit conditions:
201  // - end of input: there might or might not be output available
202  // - out_buff_free_start != out_buff: output available
203  this->setg(out_buff, out_buff, out_buff_free_start);
204  }
205  return this->gptr() == this->egptr()
206  ? traits_type::eof()
207  : traits_type::to_int_type(*this->gptr());
208  }

211  std::streambuf* sbuf_p;
212  char* in_buff;

216  detail::z_stream_wrapper* zstrm_p;

218  bool auto_detect;
219  bool auto_detect_run;

223 }; // class istreambuf

225 class ostreambuf
226  : public std::streambuf {
227 public:
228  ostreambuf(std::streambuf* _sbuf_p,
229  std::size_t _buff_size = default_buff_size, int _level = Z_DEFAULT_COMPRESSION)
230  : sbuf_p(_sbuf_p),
231  zstrm_p(new detail::z_stream_wrapper(false, _level)),
232  buff_size(_buff_size) {
233  assert(sbuf_p);
234  in_buff = new char[buff_size];
235  out_buff = new char[buff_size];
236  setp(in_buff, in_buff + buff_size);
237  }
238 
239  ostreambuf(const ostreambuf&) = delete;
240  ostreambuf(ostreambuf&&) = default;
241  ostreambuf& operator=(const ostreambuf&) = delete;
242  ostreambuf& operator=(ostreambuf&&) = default;

244  int deflate_loop(int flush) {
245  while (true) {
246  zstrm_p->next_out = reinterpret_cast<decltype(zstrm_p->next_out)>(out_buff);
247  zstrm_p->avail_out = buff_size;
248  int ret = deflate(zstrm_p, flush);
249  if (ret != Z_OK && ret != Z_STREAM_END && ret != Z_BUF_ERROR)
250  throw Exception(zstrm_p, ret);
251  std::streamsize sz = sbuf_p->sputn(out_buff, reinterpret_cast<decltype(out_buff)>(zstrm_p->next_out) - out_buff);
252  if (sz != reinterpret_cast<decltype(out_buff)>(zstrm_p->next_out) - out_buff) {
253  // there was an error in the sink stream
254  return -1;
255  }
256  if (ret == Z_STREAM_END || ret == Z_BUF_ERROR || sz == 0) {
257  break;
258  }
259  }
260  return 0;
261  }
262 
263  virtual ~ostreambuf() {
264  // flush the zlib stream
265  //
266  // NOTE: Errors here (sync() return value not 0) are ignored, because we
267  // cannot throw in a destructor. This mirrors the behaviour of
268  // std::basic_filebuf::~basic_filebuf(). To see an exception on error,
269  // close the ofstream with an explicit call to close(), and do not rely
270  // on the implicit call in the destructor.
271  //
272  sync();
273  delete[] in_buff;
274  delete[] out_buff;
275  delete zstrm_p;
276  }
277  virtual std::streambuf::int_type overflow(std::streambuf::int_type c = traits_type::eof()) {
278  zstrm_p->next_in = reinterpret_cast<decltype(zstrm_p->next_in)>(pbase());
279  zstrm_p->avail_in = pptr() - pbase();
280  while (zstrm_p->avail_in > 0) {
281  int r = deflate_loop(Z_NO_FLUSH);
282  if (r != 0) {
283  setp(nullptr, nullptr);
284  return traits_type::eof();
285  }
286  }
287  setp(in_buff, in_buff + buff_size);
288  return traits_type::eq_int_type(c, traits_type::eof()) ? traits_type::eof() : sputc(c);
289  }
290  virtual int sync() {
291  // first, call overflow to clear in_buff
292  overflow();
293  if (!pptr())
294  return -1;
295  // then, call deflate asking to finish the zlib stream
296  zstrm_p->next_in = nullptr;
297  zstrm_p->avail_in = 0;
298  if (deflate_loop(Z_FINISH) != 0)
299  return -1;
300  deflateReset(zstrm_p);
301  return 0;
302  }
