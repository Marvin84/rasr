48 template<class Key, Key invalidKey, class Value, bool preventFalsePositives = false, bool useHashChain = false, bool powerOfTwoSize = true, class Hash = StandardApproxValueHash<Key>, class Combine = MinimumCombine<Value>>
49 class ApproxLinearMiniHash {
50  inline u32 constrain(u32 val) const {
51  if (powerOfTwoSize)
52  return val & mask_;
53  else
54  return val % mask_;
55  }

57 public:
58  ApproxLinearMiniHash()
59  : size_(0), mask_(0) {
60  verify(sizeof(float) == sizeof(u32)); // Expected in all of the code
61  }

63  u32 hashSize() const {
64  return sparseValues_.size();
65  }

67  void clear(u32 minHashSize = 0) {
68  size_ = 0;
69 
70  if (minHashSize == 0) {
71  sparseValues_.clear();
72  if (preventFalsePositives)
73  sparseKeys_.clear();
74  mask_ = 0;
75  return;
76  }
77 
78  u32 hashSize;
79 
80  if (powerOfTwoSize) {
81  hashSize = 1;
82 
83  while (hashSize < minHashSize)
84  hashSize <<= 1;
85 
86  verify(hashSize >= minHashSize);
87 
88  mask_ = hashSize - 1;
89  }
90  else {
91  hashSize = 89;
92 
93  while (hashSize < minHashSize)
94  hashSize = (2 * hashSize) + 1;
95 
96  verify(hashSize >= minHashSize);
97 
98  mask_ = hashSize;
99  }
100 
101  u32 fillSize = std::min(hashSize, (u32)sparseValues_.size());
102  sparseValues_.resize(hashSize, F32_MAX);
103  if (preventFalsePositives)
104  sparseKeys_.resize(hashSize, invalidKey);
105 
106  for (u32 a = 0; a < fillSize; ++a) {
107  sparseValues_[a] = F32_MAX;
108  if (preventFalsePositives)
109  sparseKeys_[a] = invalidKey;
110  }
111  }

113  // Returns the new hash-size if a resize is required
114  inline u32 checkResize(int resizeAtFraction) {
115  if (size_ >= ((sparseValues_.size() * resizeAtFraction) >> 8))
116  return hashSize() * 2;
117  return 0;
118  }

133  // Returns the number of conflicts
134  u32 insert(Key id, Value value) {
135  const u32 h = hash(id);
136  const unsigned char marker = h >> 16;
137  if (!preventFalsePositives) {
138  value = mark(value, marker);
139  verify(isMarked(value, marker));
140  }
141 
142  u32 pos = constrain(h);
143  ++size_;
144 
145  if (!useHashChain) {
146  if (sparseValues_[pos] == F32_MAX) {
147  sparseValues_[pos] = value;
148  if (preventFalsePositives)
149  sparseValues_[pos] = id;
150  return 0;
151  }
152  else {
153  if (preventFalsePositives) {
154  // hard-coded minimum-combine
155  if (sparseValues_[pos] > value) {
156  sparseValues_[pos] = value;
157  sparseKeys_[pos] = id;
158  }
159  }
160  else {
161  sparseValues_[pos] = Combine()(sparseValues_[pos], value);
162  }
163  return 1;
164  }
165  }
166  else {
167  if (sparseValues_[pos] == F32_MAX) {
168  // The bin is empty, insert just here
169  if (preventFalsePositives)
170  sparseKeys_[pos] = id;
171  sparseValues_[pos] = value;
172  return 0;
173  }
174  else {
175  u32 cnt = 0;
176 
177  // Iterate linearly, searching for a free bin
178  for (u32 p = constrain(pos + 1); p != pos; p = constrain(p + 1)) {
179  ++cnt;
180  if (sparseValues_[p] == F32_MAX) {
181  // The bin is empty, insert just here
182  if (preventFalsePositives)
183  sparseKeys_[p] = id;
184  sparseValues_[p] = value;
185  return cnt;
186  }
187  else {
188  if (preventFalsePositives)
189  verify_(sparseKeys_[p] != id); // Every item must be inserted only once
190  }
191  }
192 
193  verify(0); // Should never happen
194  }
195  }

202  inline bool get(Key id, Value& target) const {
203  const u32 pos = constrain(id);
204  if (!useHashChain) {
205  if (preventFalsePositives && sparseKeys_[pos] != id)
206  return false;
207 
208  Value t = sparseValues_[pos];
209  if (t != F32_MAX && isMarked(t, id >> 16)) {
210  target = t;
211  return true;
212  }
213  else {
214  return false;
215  }
216  }
217  else {
218  const unsigned char marker = id >> 16;
219 
220  if (sparseValues_[pos] == F32_MAX)
221  return false;
222 
223  if ((preventFalsePositives && sparseKeys_[pos] == id) || (!preventFalsePositives && isMarked(sparseValues_[pos], marker))) {
224  target = sparseValues_[pos];
225  return true;
226  }
227 
228  // The first iteration of the loop is unrolled, for performance reasons
229  u32 p = constrain(pos + 1);
230  Value item2Value = sparseValues_[p];
231 
232  if ((preventFalsePositives && sparseKeys_[p] == id) || (!preventFalsePositives && isMarked(item2Value, marker))) {
233  target = item2Value;
234  return true;
235  }
236  else if (item2Value == F32_MAX) {
237  return false; // The id wasn't found
238  }
239 
240  // Iterate linearly, searching for the bin
241  for (p = constrain(p + 1); p != pos; p = constrain(p + 1)) {
242  Value item3Value = sparseValues_[p];
243  if ((preventFalsePositives && sparseKeys_[p] == id) || (!preventFalsePositives && isMarked(item3Value, marker))) {
244  target = item3Value;
245  return true;
246  }
247  else if (item3Value == F32_MAX)
248  return false; // The id wasn't found
249  }
250 
251  // Should never happen, as the hash is never 100% full
252  verify(0);
253  }
254  }
