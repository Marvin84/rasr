35 template<typename FloatT>
36 Criterion<FloatT>* Criterion<FloatT>::create(const Core::Configuration& config) {
37  Criterion::Type criterion = (Criterion::Type)paramCriterion(config);
38  switch (criterion) {
39  case crossEntropy:
40  Core::Application::us()->log("Create cross-entropy training criterion");
41  return new CrossEntropyCriterion<FloatT>(config);
42  case squaredError:
43  Core::Application::us()->log("Create squared-error training criterion");
44  return new SquaredErrorCriterion<FloatT>(config);
45  case binaryDivergence:
46  Core::Application::us()->log("Create binary-divergence training criterion");
47  return new BinaryDivergenceCriterion<FloatT>(config);
48  case ctc:
49  Core::Application::us()->log("Create CTC training criterion");
50  return new CtcCriterion<FloatT>(config);
51  case none:
52  Core::Application::us()->log("Create 'none' training criterion");
53  return new Criterion<FloatT>(config);
54  };
55  Core::Application::us()->criticalError("Invalid criterion");
56  return NULL;
57 }

83  require(Precursor::nnOutput_);
84  require(Precursor::alignment_); // not implemented yet without fixed alignment
85 
86  switch (lastLayer.getLayerType()) {
87  case NeuralNetworkLayer<FloatT>::linearAndSoftmaxLayer:
88  case NeuralNetworkLayer<FloatT>::softmaxLayer:
89  // softmax - kronecker delta (minimization problem)
90  errorSignal.setToZero();

128 template<typename FloatT>
129 void SquaredErrorCriterion<FloatT>::getErrorSignal_naturalPairing(NnMatrix& errorSignal, NeuralNetworkLayer<FloatT>& lastLayer) {
130  require(Precursor::nnOutput_);
131  require(Precursor::alignment_); // not implemented yet without fixed alignment
132 
133  NnVector tmp;
134  NnMatrix& netOutput = *Precursor::nnOutput_;
135  auto& alignment = *Precursor::alignment_;
136 
137  switch (lastLayer.getLayerType()) {
138  case NeuralNetworkLayer<FloatT>::linearLayer:
139  errorSignal.setToZero();
140  errorSignal.add(netOutput);
141  errorSignal.addKroneckerDelta(alignment, -1.0);
142  break;
143 
144  case NeuralNetworkLayer<FloatT>::linearAndSoftmaxLayer:
145  case NeuralNetworkLayer<FloatT>::softmaxLayer:
146  // (a) (softmax - kronecker-delta) .* softmax
147  errorSignal.setToZero();
148  errorSignal.add(netOutput);
149  errorSignal.addKroneckerDelta(alignment, -1.0);
150  errorSignal.elementwiseMultiplication(netOutput);
151  // (b) store column sums in tmp vector
152  tmp.initComputation();
153  tmp.resize(errorSignal.nColumns(), 0, true);
154  tmp.setToZero();
155  tmp.addSummedRows(errorSignal);
156  // (c) redefine error signal: softmax - kronecker-delta
157  errorSignal.setToZero();
158  errorSignal.add(netOutput);
159  errorSignal.addKroneckerDelta(alignment, -1.0);
160  // (d) subtract column sums and multiply with softmax
161  errorSignal.addToAllRows(tmp, -1.0);
162  errorSignal.elementwiseMultiplication(netOutput);
163  break;
164 
165  default:
166  Core::Component::criticalError(
167  "This layer-type is not yet implemented in training. "
168  "Allowed types: linear, softmax, linear+softmax.");
169  break;
170  }
171 
172  if (Precursor::weights_)
173  errorSignal.multiplyColumnsByScalars(*Precursor::weights_);
174 }
