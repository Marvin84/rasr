50 bool DynamicProgramingSegmentEstimator::buildBackPointerMatrix() {
51  // dynamic programming
52 
53  errorMatrix__(0, 0) = 0.0;
54  for (int i = 1; i <= max_segment_value_; i++)
55  errorMatrix__(0, i) = infinite;
56 
57  for (s32 segment_end = step_; segment_end < max_segment_value_ + step_; segment_end += step_) {
58  //Force to cover the rightmost frequency
59  if (segment_end > max_segment_value_)
60  segment_end = max_segment_value_;
61 
62  for (u8 nr_segment = 1; nr_segment <= std::min((s32)max_nr_segment_, segment_end); nr_segment++) {
63  _float minimum = infinite;
64  s32 back_pointer = segment_end;
65  for (s32 segment_begin = (nr_segment - 1) * step_;
66  segment_begin < segment_end + step_; segment_begin += step_) {
67  //Force to cover the rightmost frequency
68  if (segment_begin > segment_end)
69  segment_begin = segment_end;
70 
71  f64 auxiliaryFunction = errorMatrix__(nr_segment - 1, segment_begin) + segmentError__(segment_begin, segment_end);
72 
73  if (auxiliaryFunction < minimum) {
74  minimum = auxiliaryFunction;
75  back_pointer = segment_begin;
76  }
77  }
78  errorMatrix__(nr_segment, segment_end) = minimum;
79  backPointer__(nr_segment, segment_end) = back_pointer;
80  }
81  }
82 
83  return true;
84 }

86 bool DynamicProgramingSegmentEstimator::work(std::vector<s32>& segments) {
87  if (need_realloc_ && !reallocateBuffers())
88  return false;
89  if (need_init_ && !init())
90  return false;
91 
92  std::vector<s32> tmp_segment(max_nr_segment_ + 1);
93 
94  // traceback
95  s32 traceback = max_segment_value_;
96  for (s16 nr_segment = (s16)max_nr_segment_; nr_segment >= 0; nr_segment--) {
97  tmp_segment[nr_segment] = traceback;
98  traceback = backPointer__(nr_segment, traceback);
99  }
100 
101  segments.reserve(max_nr_segment_ + 1);
102  segments.clear();
103 
104  segments.push_back(tmp_segment[0]);
105  for (u8 i = 1; i <= max_nr_segment_; i++) {
106  if (tmp_segment[i] != segments.back())
107  segments.push_back(tmp_segment[i]);
108  }
109 
110  return true;
111 }
