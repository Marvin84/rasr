92 bool EigenvalueProblem::finalizeEigenvaluesAndVectors(
93  const Matrix<ValueType>& a,
94  Vector<ValueType>& eigenvalues,
95  Matrix<ValueType>& eigenvectors) {
96  bool result = true;
97 
98  eigenvectors.normalizeColumnDirection();
99  sortEigenvalues(eigenvalues, eigenvectors);
100 
101  switch (eigenvectorNormalizationType_) {
102  case nonEigenvectorNormalization:
103  break;
104  case unityLength:
105  eigenvectors.normalizeColumns(2);
106  break;
107  case unityDiagonal:
108  if (!normalizeDiagonal(a, eigenvectors))
109  result = false;
110  break;
111  default:
112  error("Eigenvector normalization type not supported.");
113  result = false;
114  }
115  return result;
116 }

118 bool EigenvalueProblem::normalizeDiagonal(
119  const Matrix<ValueType>& a, Matrix<ValueType>& eigenvectors) {
120  bool result = true;
121  Vector<ValueType> diagonal =
122  (eigenvectors.transpose() * a * eigenvectors).diagonal();
123  for (size_t i = 0; i < diagonal.size(); ++i) {
124  if (diagonal[i] < 0 && Core::abs(diagonal[i]) > diagonalNormalizationTolerance_) {
125  error("Failed to normalize eigenvectors because "
126  "because %zu diagonal element %e is negative.",
127  i, diagonal[i]);
128  result = false;
129  }
130  }
131  if (result) {
132  diagonal.takeSquareRoot(diagonalNormalizationTolerance_);
133  Message scalingMessage(log("Eigenvector scaling factors relative to their lengths:\n"));
134  for (size_t i = 0; i < diagonal.size(); ++i) {
135  if (diagonal[i] == 0)
136  scalingMessage << "no-scaling ";
137  else
138  scalingMessage << (eigenvectors.columnNorm(i, 2) / diagonal[i]) << " ";
139  }
140  diagonal.takeReciprocal(diagonalNormalizationTolerance_);
141  eigenvectors = eigenvectors * makeDiagonalMatrix(diagonal);
142  }
143  return result;
144 }

146 bool EigenvalueProblem::isCorrectSolution(const Matrix<ValueType>& a,
147  const Vector<ValueType>& eigenvalues,
148  const Matrix<ValueType>& eigenvectors) {
149  if ((a.nColumns() != eigenvectors.nRows()) ||
150  (eigenvectors.nColumns() != eigenvalues.size())) {
151  error("Dimension mismatch while verifying solution: A[%zd, %zd], X[%zd, %zd], lambda[%zd].)",
152  a.nRows(), a.nColumns(), eigenvectors.nRows(), eigenvectors.nColumns(), eigenvalues.size());
153  return false;
154  }

156  Matrix<ValueType> eigenvectorsTimesValues(eigenvectors * makeDiagonalMatrix(eigenvalues));
157  return areEqual(aTimesEigenvectors, eigenvectorsTimesValues);

174 void EigenvalueProblem::sortEigenvalues(Vector<ValueType>& eigenvalues, Matrix<ValueType>& eigenvectors) {
175  if (eigenvalueSortType_ != noSort) {
176  typedef std::pair<ValueType, size_t> MapItem;
177  std::vector<MapItem> eigenvalueIndexMap(eigenvalues.size());
178  for (size_t i = 0; i < eigenvalues.size(); ++i)
179  eigenvalueIndexMap[i] = std::make_pair(Core::abs(eigenvalues[i]), i);
180 
181  if (eigenvalueSortType_ == decreasing) {
182  std::sort(eigenvalueIndexMap.begin(), eigenvalueIndexMap.end(),
183  Core::composeBinaryFunction(std::greater<ValueType>(),
184  select1st<MapItem>(),
185  select1st<MapItem>()));
186  }
187  else if (increasing) {
188  std::sort(eigenvalueIndexMap.begin(), eigenvalueIndexMap.end(),
189  Core::composeBinaryFunction(std::less<ValueType>(),
190  select1st<MapItem>(),
191  select1st<MapItem>()));
192  }
193  else {
194  error("undefined sort type for eigenvalues");
195  }
196 
197  std::vector<size_t> permutation(eigenvalues.size());
198  std::transform(eigenvalueIndexMap.begin(),
199  eigenvalueIndexMap.end(),
200  permutation.begin(),
201  select2nd<MapItem>());
202  eigenvalues.permuteElements(permutation);
203  eigenvectors.permuteColumns(permutation);
204  }
205 }

207 bool EigenvalueProblem::real(const Vector<std::complex<ValueType>>& c, Vector<ValueType>& r) {
208  ValueType imaginaryMaximum = Core::maxAbsoluteElement(c) * relativeImaginaryMaximum_;
209  bool success = true;
210  r.resize(c.size());
211  Vector<std::complex<ValueType>>::const_iterator ci;
212  Vector<ValueType>::iterator ri;
213  for (ci = c.begin(), ri = r.begin(); ci != c.end(); ++ci, ++ri) {
214  ValueType absImag = ci->imag();
215  if (absImag > imaginaryMaximum) {
216  error("The %zdth imaginary element (%e) is larger than imaginary-maximum (%e).",
217  size_t(ci - c.begin()), absImag, imaginaryMaximum);
218  success = false;
219  }
220  *ri = ci->real();
221  }
222  return success;
223 }

225 bool EigenvalueProblem::real(const Matrix<std::complex<ValueType>>& c, Matrix<ValueType>& r) {
226  ValueType imaginaryMaximum = Math::maxAbsoluteElement(c) * relativeImaginaryMaximum_;
227  bool success = true;
228  r.resize(c.nRows(), c.nColumns());
229  for (size_t row = 0; row < r.nRows(); ++row) {
230  for (size_t column = 0; column < r.nColumns(); ++column) {
231  ValueType absImag = Core::abs(c[row][column].imag());
232  if (absImag > imaginaryMaximum) {
233  error("The (%zd, %zd) imaginary element (%e) is larger than imaginary-maximum (%e).",
234  row, column, absImag, imaginaryMaximum);
235  success = false;
236  }
237  r[row][column] = c[row][column].real();
238  }
239  }
240  return success;
241 }
