64 TEST_F(Test, TestLinearLayer, IOBin) {

67  std::string tmpdir;

69  tmpdir = t;

73  Core::createDirectory(dirName);
74  std::string filename = "bin:" + dirName + "model";
75  layer_->saveNetworkParameters(filename);
76  Nn::NeuralNetworkLayer<f64>* layer2 = new LinearLayer<f64>(config);

79  layer2->setOutputDimension(3);

81
