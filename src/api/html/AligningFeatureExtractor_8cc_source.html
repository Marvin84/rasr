106 void AligningFeatureExtractor::processFeature(Core::Ref<const Feature> f) {
107  verify(alignment_ and !alignment_->empty());
108  if (!processWeighted_) {
109  if (currentAlignmentItem_ != alignment_->end()) {
110  if (currentAlignmentItem_->time > currentFeatureId_) {
111  warning("Alignment item further than current timeframe (alignment=%zd, current=%zd), skipping!", (size_t)currentAlignmentItem_->time, (size_t)currentFeatureId_);
112  while (currentFeatureId_ < currentAlignmentItem_->time)
113  ++currentFeatureId_;
114  }
115  alignedFeatureProcessor_.processAlignedFeature(f, currentAlignmentItem_->emission);
116  ++currentAlignmentItem_;
117  }
118  else {
119  warning("Alignment (size=%zd) shorter than the feature stream.", alignment_->size());
120  }
121  }
122  else {
123  if (!alignment2_) {
124  unaryProcessFeature(f);
125  }
126  else {
127  binaryProcessFeature(f);
128  }
129  }
130  ++currentFeatureId_;
131 }

133 void AligningFeatureExtractor::unaryProcessFeature(Core::Ref<const Feature> f) {
134  verify(processWeighted_);
135  while (currentAlignmentItem_ != alignment_->end() && currentAlignmentItem_->time == currentFeatureId_) {
136  verify(currentAlignmentItem_ == alignment_->begin() ? true : (currentAlignmentItem_ - 1)->time <= currentAlignmentItem_->time);
137  alignedFeatureProcessor_.processAlignedFeature(f, currentAlignmentItem_->emission, currentAlignmentItem_->weight);
138  ++currentAlignmentItem_;
139  }
140 }

142 void AligningFeatureExtractor::binaryProcessFeature(Core::Ref<const Feature> f) {
143  verify(alignment2_);
144  verify(!alignment2_->empty());
145  verify(processWeighted_);
146  while (currentAlignmentItem_ != alignment_->end() && currentAlignmentItem_->time == currentFeatureId_) {
147  verify(currentAlignmentItem_ == alignment_->begin() ? true : (currentAlignmentItem_ - 1)->time <= currentAlignmentItem_->time);
148  verify(currentAlignmentItem_->time == currentAlignment2Item_->time);
149  verify(currentAlignmentItem_->emission == currentAlignment2Item_->emission);
150  alignedFeatureProcessor_.processAlignedFeature(f,
151  currentAlignmentItem_->emission,
152  currentAlignmentItem_->weight,
153  currentAlignment2Item_->weight);
154  ++currentAlignmentItem_;
155  ++currentAlignment2Item_;
156  }
157 }

159 bool AligningFeatureExtractor::initializeAlignment() {
160  alignment_ = 0;
161  alignment2_ = 0;
162 
163  if (!dataSource()->getData(alignmentPortId_, alignmentRef_)) {
164  error("Failed to extract alignment.");
165  return false;
166  }
167  alignment_ = &alignmentRef_->data();
168  currentFeatureId_ = 0;
169  currentAlignmentItem_ = alignment_->begin();
170  if (alignment_->empty()) {
171  warning("Segment has been discarded because of empty alignment.");
172  return false;
173  }
174  processWeighted_ = (processWeighted_ || alignment_->hasWeights() || (currentAlignmentItem_->time != 0));
175 
176  if (alignment2PortId_ != Flow::IllegalPortId) {
177  if (!dataSource()->getData(alignment2PortId_, alignment2Ref_)) {
178  error("Failed to extract alignment.");
179  return false;
180  }
181  alignment2_ = &alignment2Ref_->data();
182  currentAlignment2Item_ = alignment2_->begin();
183  processWeighted_ = true;
184  if (alignment2_->empty()) {
185  warning("Segment has been discarded because of empty alignment.");
186  return false;
187  }
188  if (alignment_->size() != alignment2_->size()) {
189  error("Mismatch in size of alignments.");
190  return false;
191  }
192  }
193 
194  return true;
195 }
