83  MeanIndex meanIndex;

201 Covariance* CovarianceEstimator::estimate(const CovarianceToMeanSetMap& meanSetMap, VarianceType minimumVariance) {
202  if (accumulator_.weight() == 0)
203  return new DiagonalCovariance(accumulator_.sum().size());
204 
205  const CovarianceToMeanSetMap::MeanSet& meanSet = meanSetMap[Core::Ref<AbstractCovarianceEstimator>((CovarianceEstimator*)this)];
206  WeighedMeanSquareSum weighedMeanSquareSum(accumulator_.size());
207 
208  for (CovarianceToMeanSetMap::MeanSet::const_iterator meanEstimator = meanSet.begin();
209  meanEstimator != meanSet.end(); ++meanEstimator) {
210  const MeanEstimator* pMeanEstimator = dynamic_cast<const MeanEstimator*>(meanEstimator->get());
211  verify(pMeanEstimator);
212 
213  if (pMeanEstimator->weight() > 0)
214  weighedMeanSquareSum.accumulate(pMeanEstimator->sum(), pMeanEstimator->weight());
215  }
216  verify(accumulator_.weight() > 0);
217  verify(Core::isAlmostEqualUlp(accumulator_.weight(), weighedMeanSquareSum.weight(), s64(1e12)));
218  DiagonalCovariance* result = new DiagonalCovariance(accumulator_.sum().size());
219 
220  std::transform(accumulator_.sum().begin(), accumulator_.sum().end(), weighedMeanSquareSum.sum().begin(),
221  result->buffer_.begin(), normalizedMinus<Sum>(accumulator_.weight()));
222 
223  applyMinimumVariance(minimumVariance, *result);
224  return result;
225 }

227 void CovarianceEstimator::applyMinimumVariance(VarianceType minimumVariance, DiagonalCovariance& result) const {
228  if (minimumVariance == 0)
