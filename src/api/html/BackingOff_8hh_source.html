50  /**
51  * Back-off language model item.
52  *
53  * For items with non-zero token, score gives the negative
54  * natural logarithm of p(token | history):
55  * score = - ln ( p(token | history) )
56  *
57  * Items with token equals zero provide back-off weights:
58  * score = - ln ( p(w | history) / p(w | shortened_history) )
59  * Obviously this term has to be constant over all words w
60  * for which the back-off case occurs. (Actually, this could
61  * be used as the the definition of "back-off case".)
62  */
63  struct InitItem {
64  Token* history; /**< zero-terminated string, recent-most first */
65  Token token; /**< predicted word, or zero iff back-off */
66  Score score; /**< negative natural logarithm of p(token | history). */
67  };
