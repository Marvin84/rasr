55 Application::Application()
56  : Component(getConfig()),
57  channelManager_(0),
58  debugChannel_(0),
59  debugXmlChannel_(0),
60  defaultLoadConfigurationFile_(true),
61  defaultOutputXmlHeader_(true),
62  comment_("#;%") {
63  require(!app_); // There must be only one instance.
64  app_ = this;

66  char* value;
67  const char* user;
68  struct passwd* p = NULL;
69 
70  // load process environment into configuration
71  for (const char* const* env = environ; *env; ++env) {
72  const char* sep = strchr(*env, '=');
73  if (sep)
74  config.set(std::string(*env, sep - *env), std::string(sep + 1));
75  }
76 
77  // get the real (!) home directory
78  if ((user = getlogin()) != NULL)
79  p = getpwnam(user);
80  if (p != NULL)
81  value = p->pw_dir;
82  else
83  value = getenv("HOME");
84  config.set("*.home", value);
85 }

300 void Application::closeLogging(bool configAvailable) {

320 int Application::main(int argc, char* argv[]) {
321  /**
322  * @bug This is a workaround for possible bug in LIBC 2.3.2. We are waiting for an answer on our bug report.
323  */
324 #ifdef OS_linux
325  mallopt(M_TOP_PAD, 1024 * 1024); // 128*1024 default for GLIBC2.3.2 on debian sarge
326 #endif
327 
328  if (!app_) {
329  std::cerr << "oops! you did not instantiate any application class!" << std::endl;
330  return EXIT_FAILURE;
331  }
332 
333  int status = EXIT_SUCCESS;
334 
335  std::vector<std::string> arguments;
336  for (int i = 0; i < argc; ++i) {
337  arguments.push_back(argv[i]);
338  }
339 
340  setFromCommandline(arguments);
341 
342  std::string file = paramConfig(getConfig());
343  if (file.empty() && app_->defaultLoadConfigurationFile_)
344  file = app_->name() + ".config";
345  if (!file.empty())
346  setFromFile(file);
347 
348  setFromEnvironment(app_->getVariable());
349 
350  arguments = setFromCommandline(arguments);
351 
352  if (app_->name() == "UNNAMED") {
353  std::string title = app_->getBaseName();
354  title.erase(title.find("."));
355  app_->setTitle(title);
356  }
357 
358  if (paramHelp(getConfig())) {
359  std::cerr << app_->getUsage();
360  }
361  else {
362  status = app_->run(arguments);
363  }
364 
365  // Do an early reset of this variable, so that code in exit handlers can check for it.
366  // (E.g. Core::printWarning().)
367  app_ = NULL;
368 
369  return status;
370 }
