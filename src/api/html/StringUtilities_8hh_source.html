198  std::string::size_type a = s.find_first_not_of(utf8::whitespace);
199  std::string::size_type b = s.find_last_not_of(utf8::whitespace);

204  return false;
205  }
206  else {
207  return true;

348 template<typename T>
349 bool str2vector(const std::string& s, std::vector<T>& e,
350  const std::string& elemDelim) {
351  require(!elemDelim.empty());
352  std::string::size_type start = s.find_first_not_of(utf8::whitespace);
353  std::string::size_type end;
354  std::string slice;
355  T t;
356 
357  while (start != std::string::npos) {
358  end = s.find(elemDelim, start);
359  if (end == std::string::npos) {
360  slice = s.substr(start);
361  start = std::string::npos;
362  }
363  else {
364  slice = s.substr(start, end - start);
365  start = s.find_first_not_of(utf8::whitespace, end + elemDelim.size());
366  }
367  if (!strconv(slice, t))
368  return false;
369  e.push_back(t);
370  }
371  return true;
372 }

374 template<typename T>
375 bool str2vector(const std::string& s, std::vector<T>& e,
376  const std::string& elemDelim,
377  const std::string& rangeDelim,
378  const T& minValue, const T& maxValue, const u32& minSize, const u32& maxSize, const T& step) {
379  require(!elemDelim.empty());
380  require(!rangeDelim.empty());
381  std::string::size_type start = s.find_first_not_of(utf8::whitespace);
382  std::string::size_type center, end;
383  std::string slice;
384  T t, rangeStart, rangeEnd;
385 
386  while (start != std::string::npos) {
387  end = s.find(elemDelim, start);
388  if (end == std::string::npos) {
389  slice = s.substr(start);
390  start = std::string::npos;
391  }
392  else {
393  slice = s.substr(start, end - start);
394  start = s.find_first_not_of(utf8::whitespace, end + elemDelim.size());
395  }
396  center = slice.find(rangeDelim, 1);
397  if (center != std::string::npos) {
398  if (!strconv(slice.substr(0, center), rangeStart) || !strconv(slice.substr(std::min(center + rangeDelim.size(), slice.size())), rangeEnd))
399  return false;
400  rangeStart = std::max(rangeStart, minValue);
401  rangeEnd = std::min(rangeEnd, maxValue);
402  for (t = rangeStart; t <= rangeEnd; t += step)
403  e.push_back(t);
404  }
405  else {
406  if (!strconv(slice, t))
407  return false;
408  if ((minValue <= t) && (t <= maxValue))
409  e.push_back(t);
410  else {
411  std::cerr << "configuration error: ";
412  std::cerr << "the given string contains out of range values." << std::endl;
413  return false;
414  }
415  }
416  }
417  if ((e.size() >= minSize) && (e.size() <= maxSize))
418  return true;
419  else {
420  std::cerr << "configuration error: ";
421  std::cerr << "the given string has an incorrect number of elements." << std::endl;
422  return false;
423  }
424 }

426 template<typename T>
427 bool str2vector(const std::string& s, std::vector<T>& e,
428  const std::string& elemDelim,
429  const std::string& rangeDelim,
430  const std::string& assignDelim,
431  const T& minValue, const T& maxValue, const u32& minSize, const u32& maxSize, const T& step) {
432  require(!elemDelim.empty());
433  require(!rangeDelim.empty());
434  require(!assignDelim.empty());
435  std::string::size_type start = s.find_first_not_of(utf8::whitespace);
436  std::string::size_type center, end, assignPoint;
437  std::string slice, wholeSlice;
438  T t, rangeStart, rangeEnd, weight;
439 
440  while (start != std::string::npos) {
441  end = s.find(elemDelim, start);
442  if (end == std::string::npos) {
443  wholeSlice = s.substr(start);
444  start = std::string::npos;
445  }
446  else {
447  wholeSlice = s.substr(start, end - start);
448  start = s.find_first_not_of(utf8::whitespace, end + elemDelim.size());
449  }
450 
451  assignPoint = wholeSlice.find(assignDelim, 1);
452  if (assignPoint != std::string::npos) {
453  if (!strconv(wholeSlice.substr(std::min(assignPoint + assignDelim.size(), wholeSlice.size())), weight))
454  return false;
455  weight = std::max(weight, minValue);
456  weight = std::min(weight, maxValue);
457  slice = wholeSlice.substr(0, assignPoint);
458  center = slice.find(rangeDelim, 1);
459  if (center != std::string::npos) {
460  if (!strconv(slice.substr(0, center), rangeStart) || !strconv(slice.substr(std::min(center + rangeDelim.size(), slice.size())), rangeEnd))
461  return false;
462  if (rangeStart > rangeEnd)
463  return false;
464  for (t = rangeStart; t <= rangeEnd; t += step)
465  e.push_back(weight);
466  }
467  else {
468  e.push_back(weight);
469  }
470  }
471  else {
472  if (!strconv(wholeSlice, weight))
473  return false;
474  weight = std::max(weight, minValue);
475  weight = std::min(weight, maxValue);
476  e.push_back(weight);
477  }
478  }
479  if ((e.size() >= minSize) && (e.size() <= maxSize))
480  return true;
481  else {
482  std::cerr << "configuration error: ";
483  std::cerr << "the given string has an incorrect number of elements." << std::endl;
484  return false;
485  }
