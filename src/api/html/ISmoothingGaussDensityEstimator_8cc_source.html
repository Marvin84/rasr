56 ISmoothingCovarianceEstimator::Accumulator::SumType ISmoothingCovarianceEstimator::iSum(const CovarianceToMeanSetMap::MeanSet& meanSet,
57  ComponentIndex i) const {
58  if (!iSum_) {
59  iSum_ = new std::vector<Accumulator::SumType>(iCovariance().size());
60  std::copy(iCovariance().begin(), iCovariance().end(), iSum_->begin());
61  std::transform(iSum_->begin(), iSum_->end(), iSum_->begin(),
62  std::bind2nd(std::multiplies<Accumulator::SumType>(), meanSet.size()));
63  CovarianceToMeanSetMap::MeanSet::const_iterator m = meanSet.begin();
64  for (; m != meanSet.end(); ++m) {
65  const ISmoothingMeanEstimator* meanEstimator = dynamic_cast<const ISmoothingMeanEstimator*>(m->get());
66  std::transform(iSum_->begin(), iSum_->end(), meanEstimator->iMean().begin(),
67  iSum_->begin(), plusSquare<Accumulator::SumType>());
68  }
69  }
70  return (*iSum_)[i];
71 }

73 Sum ISmoothingCovarianceEstimator::getObjectiveFunction(const CovarianceToMeanSetMap& meanSetMap) {
74  verify(parent_);
75  Sum iOf = 0;
76  const CovarianceToMeanSetMap::MeanSet& meanSet = meanSetMap[parent_];
77  for (ComponentIndex i = 0; i < parent_->sum().size(); ++i) {
78  const VarianceType prevSigma = parent_->previousCovariance(i);
79  iOf += meanSet.size() * std::log(2 * M_PI * prevSigma);
80  Sum _iOf = iSum(meanSet, i);
81  CovarianceToMeanSetMap::MeanSet::const_iterator m = meanSet.begin();
82  for (; m != meanSet.end(); ++m) {
83  const MeanType prevMean = required_cast(const DiscriminativeMeanEstimator*, m->get())->previousMean(i);
84  _iOf += prevMean * (prevMean - 2 * iMean(*m)[i]);
85  }
86  iOf += _iOf / prevSigma;
87  }
88  return -0.5 * constant() * iOf;
89 }
