114  /**
115  * @param n is number of rows
116  * @param m is number of columns
117  */
118  void resize(size_t n, size_t m) {
119  StorageScheme newStorageScheme(n, m);
120  if (storageScheme_.storageSize() == newStorageScheme.storageSize())
121  return;
122  deleteBuffer();
123  if (newStorageScheme.storageSize() > 0)
124  buffer_ = new Type[(storageScheme_ = newStorageScheme).storageSize()];
125  }
126  size_t nRows() const {
127  return storageScheme_.nRows();
128  };
129  size_t nColumns() const {
130  return storageScheme_.nColumns();
131  };

134  return buffer_[storageScheme_(row, column)];
135  }
