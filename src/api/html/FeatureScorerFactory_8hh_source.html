44  /**
45  * Register a FeatureScorer class.
46  * T: FeatureScorer class.
47  * ModelType: MixtureSet class required by the feature scorer.
48  * Loader: MixtureSetLoader class used to read the model.
49  * id: unique feature scorer id
50  * name: name associated with the feature scorer type
51  *
52  * a Core::Choice is maintained for the id to name mapping.
53  */
54  template<class T, class ModelType, class Loader>
55  bool registerFeatureScorer(u32 id, const char* name) {
56  if (registry_.find(id) == registry_.end()) {
57  CreationFunction c = createInstance<T, ModelType>;
58  registry_.insert(Registry::value_type(id, c));
59  loader_.insert(LoaderRegistry::value_type(id, new Loader()));
60  names_.addChoice(name, id);
61  return true;
62  }
63  else {
64  return false;
65  }
66  }
67 
68  /**
69  * Create a FeatureScorer object.
70  * The actual class is determined from the id.
71  */
72  FeatureScorer* createFeatureScorer(u32 id,
73  const Core::Configuration& c,
74  Core::Ref<const AbstractMixtureSet> m) const {
75  CreationFunction create = getCreationFunction(id);
76  if (create) {
77  return create(c, m);
78  }
79  else {
80  return 0;
81  }
82  }
83 
84  /**
85  * Return a MixtureSetLoader to read the model (AbstractMixtureSet or derivative)
86  * for the given feature scorer id.
87  */
88  const AbstractMixtureSetLoader* getMixtureSetLoader(u32 id) const {
89  LoaderRegistry::const_iterator i = loader_.find(id);
90  if (i != loader_.end())
91  return i->second;
92  else
93  return 0;
94  }

96  /**

99  const Core::Choice& featureScorerNames() const {

103 protected:
104  CreationFunction getCreationFunction(u32 id) const {
105  Registry::const_iterator item = registry_.find(id);
106  if (item != registry_.end()) {
107  return item->second;
108  }
109  else {
110  return 0;
111  }
112  }

114  template<class T, class ModelType>
115  static FeatureScorer* createInstance(const Core::Configuration& c, Core::Ref<const AbstractMixtureSet> m) {
116  verify(m);
117  // Core::Ref::Ref(T *ptr) doesn't check if ptr is valid.
118  const ModelType* castedModel = dynamic_cast<const ModelType*>(m.get());
119  ensure(castedModel);
120  const Core::Ref<const ModelType> castedModelRef(castedModel);
121  return new T(c, castedModelRef);
122  }
