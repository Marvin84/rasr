201 /**
202  * Batched feature scorer using floats.
203  *
204  * Requires an acoustic model with a single (globally pooled) diagonal covariance.
205  * Scores are calculated using the maximum approximation (max. over densities).
206  * The distance computation uses SIMD instructions (SSE).
207  * Using floats without quantization.
208  *
209  * Data is stored in raw arrays with 16-byte alignment.
210  * (required by the SSE instructions).
211  *
212  * Mean vectors and features are preprocessed by multiplication with the inverse
213  * square root of the variance.
214  */
215 class BatchFloatFeatureScorer : public BatchFeatureScorerBase {
216 public:
217  BatchFloatFeatureScorer(const Core::Configuration& c);
218  BatchFloatFeatureScorer(const Core::Configuration& c, Core::Ref<const MixtureSet> mixtureSet);
219  virtual ~BatchFloatFeatureScorer();
220 
221  virtual void init(const MixtureSet& mixtureSet);
222 
223 protected:
224  virtual void setFeature(size_t pos, const FeatureVector& f) const;
225  virtual void fillScoreCache(EmissionIndex e, u32 featureIndex, u32 length) const;
226  template<class DensitySelector>
227  inline void fillScoreCacheTpl(EmissionIndex e, u32 featureIndex, u32 length, const DensitySelector& selector) const;
228 
229 protected:
230  /**
231  * Number of floats that are processed in one iteration
232  * of the distance computation.
233  */
234  static const size_t BlockSize;
235 
236  /**
237  * feature buffer.
238  * features are stored sequentially with paddedDimension_ floats.
239  */
240  mutable f32* features_;
241  /**
242  * pre-processed mean vectors of all densities.
243  * means are stored sequentially for each mixture.
244  * mapping from a mixture to a mean using offsets_:
245  */
246  f32* means_;
247  /**
248  * the reciprocal square root of the variance vector.
249  */
250  f32* variance_;
251  /**
252  * density weight + global normalization factor
253  * for each density.
254  */
255  f32* constants_;

276 /**
277  * Batched feature scorer using integer computations.
278  *
279  * Requires an acoustic model with a single (globally pooled) diagonal covariance.
280  * Scores are calculated using the maximum approximation (max. over densities).
281  * The distance computation uses SIMD instructions (SSE2).
282  * Uses quantization to one byte for each vector component.
283  *
284  * Data is stored in raw arrays with 16-byte alignment.

288  * square root of the variance, followed by quantization.
289  */

302  virtual void fillScoreCache(EmissionIndex e, u32 featureIndex, u32 length) const;
