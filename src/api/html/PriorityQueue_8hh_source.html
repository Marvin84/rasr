81 template<class H, class PF>
82 void PriorityQueueBase<H, PF>::downHeap(Index i) {
83  require(1 <= i && i <= Precursor::size());
84  Index j;
85  Item e = Precursor::item(i);
86  while (i <= Precursor::size() / 2) {
87  j = 2 * i;
88  if (j < Precursor::size() && precedes_(Precursor::item(j + 1), Precursor::item(j)))
89  j = j + 1;
90  if (!precedes_(Precursor::item(j), e))
91  break;
92  Precursor::move(i, j);
93  i = j;
94  }
95  Precursor::put(i, e);
96 }

98 template<class H, class PF>
99 void PriorityQueueBase<H, PF>::upHeap(Index i) {
100  require(1 <= i && i <= Precursor::size());
101  Item e = Precursor::item(i);
102  while (i > 1 && !precedes_(Precursor::item(i / 2), e)) {
103  Precursor::move(i, i / 2);
104  i /= 2;
105  }
106  Precursor::put(i, e);
107 }

286  /** Change item with the key of @c e to @c e. */
287  void update(const typename Precursor::Item& e) {
288  require(contains(key_(e)));
289  typename Precursor::Index i = Precursor::map_[key_(e)];
290  if (precedes_(e, Precursor::heap_[i])) {
291  Precursor::heap_[i] = e;
292  upHeap(i);
293  }
294  else {
295  Precursor::heap_[i] = e;
296  downHeap(i);
297  }

300  /**
301  * Conditional update with higher priority.
302  *
303  * If the stack contains an element with the same key as @c e but
304  * lower priority then this element is replaced by @c e. If their
305  * is no element with @c e's key, @c e is inserted. Otherwise
306  * nothing happens.
307  *
308  * This rather specific function is most useful for implementing
309  * best-first search like Dijkstra's algorithm or A*.
310  *
311  * @return true if given item was inserted (by replacing an
312  * existing one or by enlarging the stack). False if the
313  * given item dropped, and the stack is unchanged.
314  */
315  bool insertOrRelax(const typename Precursor::Item& e) {
316  if (Precursor::contains(Precursor::key_(e))) {
317  typename Precursor::Index i = Precursor::map_[Precursor::key_(e)];
318  if (Precursor::precedes_(e, Precursor::heap_[i])) {
319  Precursor::heap_[i] = e;
320  Precursor::upHeap(i);
321  verify_(Precursor::invariant());
322  }
323  else
324  return false;
325  }
326  else
327  insert(e);
328  return true;
329  }
