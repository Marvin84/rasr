68  Lattice::ConstWordLatticeRef rescored = work(lattice, segment);

149 Fsa::Weight AlignmentLatticeRescorerAutomaton::score(Fsa::StateId s, const Fsa::Arc& a) const {
150  const Bliss::LemmaPronunciationAlphabet* alphabet = required_cast(const Bliss::LemmaPronunciationAlphabet*, fsa_->getInputAlphabet().get());
151  const Bliss::LemmaPronunciation* pronunciation = alphabet->lemmaPronunciation(a.input());
152  const TimeframeIndex begtime = wordBoundaries_->time(s);
153  if (pronunciation && begtime != InvalidTimeframeIndex) {
154  Bliss::Coarticulated<Bliss::LemmaPronunciation> coarticulatedPronunciation(
155  *pronunciation, wordBoundaries_->transit(s).final,
156  wordBoundaries_->transit(fsa_->getState(a.target())->id()).initial);
157  const TimeframeIndex endtime = wordBoundaries_->time(fsa_->getState(a.target())->id());
158  return _score(coarticulatedPronunciation, begtime, endtime);
159  }
160  else {
161  return fsa_->semiring()->one();
162  }
163 }
164 
165 Fsa::Weight AlignmentLatticeRescorerAutomaton::_score(const Bliss::Coarticulated<Bliss::LemmaPronunciation>& coarticulatedPronunciation,
166  TimeframeIndex begtime, TimeframeIndex endtime) const {
167  if (begtime < endtime) {
168  return Fsa::Weight(alignmentGenerator_->alignmentScore(coarticulatedPronunciation, begtime, endtime));
169  }
170  else {
171  Core::Application::us()->warning("score 0 assigned to arc with begin time ")
172  << begtime << " , end time " << endtime << " and label id " << coarticulatedPronunciation.object().id();
173  return fsa_->semiring()->one();
174  }

232 Fsa::Weight LmLatticeRescorerAutomaton::score(Fsa::StateId s, const Fsa::Arc& a) const {
233  if (s == fsa_->initialStateId()) {
234  histories_.grow(s);
235  histories_[s] = languageModel_->startHistory();
236  }
237  require(histories_[s].isValid());
238  Lm::History hist = histories_[s];
239  Lm::Score score = 0;
240  const Bliss::LemmaPronunciation* lp = alphabet_->lemmaPronunciation(a.input());
241  if (lp) {
242  Lm::addLemmaPronunciationScore(
243  languageModel_, lp, pronunciationScale_, languageModel_->scale(), hist, score);
244  }
245  /*! \warning sentence end score has to be added manually */
246  if (fsa_->getState(a.target())->isFinal()) {
247  score += languageModel_->sentenceEndScore(hist);
248  hist = languageModel_->startHistory();
249  }
250 
251  histories_.grow(a.target());
252  if (!histories_[a.target()].isValid()) {
253  histories_[a.target()] = hist;
254  }
255  if (!(hist == histories_[a.target()])) {
256  languageModel_->error() << "Mismatch between lattice and language model: "
257  "ambiguous history at state '"
258  << a.target() << "' ('" << languageModel_->formatHistory(hist) << "' vs. '" << languageModel_->formatHistory(histories_[a.target()]) << "').\n"
259  "Possible causes: 1) lattice is time-conditioned,\n"
260  "2) lattice has been generated by using another language model.";
261  }
262  bool notFinite = Math::isnan(score) || score > Core::Type<f32>::max || score < Core::Type<f32>::min;
263  if (notFinite)
264  Core::Application::us()->log("inf lm score replaced by fall back score ") << score << ", " << fallBackValue_;
265  return Fsa::Weight(notFinite ? fallBackValue_ : score);
266 }

339 Lattice::ConstWordLatticeRef LatticeReader::extract(Lattice::ConstWordLatticeRef lattice, Bliss::SpeechSegment* segment) {
340  verify(archiveReader_);
341  Lattice::ConstWordLatticeRef result = archiveReader_->get(segment->fullName(), fsaPrefix_);
342  if (result && result->nParts() == 1) {
343  if (result->wordBoundaries() and (*result->wordBoundaries()) != (*lattice->wordBoundaries())) {
344  error("Mismatch between the word boundaries of the given lattice and the read lattice.");
345  }
346  }
347  else {
348  error("Failed to retrieve lattice '%s' for segment '%s'.",
349  fsaPrefix_.c_str(), segment->fullName().c_str());
