19 LstmStateManager::HistoryState LstmStateManager::initialState(StateVariables const& vars, CompressedVectorFactory<float> const& vector_factory) {
20  HistoryState result;
21  result.reserve(vars.size());
22  for (auto const& var : vars) {
23  require_gt(var.shape.size(), 0ul);
24  s64 state_size = var.shape.back();
25  require_ge(state_size, 0); // variable must not be of unknown size
26  std::vector<float> vec(state_size, 0.0f);
27  auto compression_param_estimator = vector_factory.getEstimator();
28  compression_param_estimator->accumulate(vec.data(), vec.size());
29  auto compression_params = compression_param_estimator->estimate();
30  result.emplace_back(vector_factory.compress(vec.data(), vec.size(), compression_params.get()));
31  }
32  return result;
33 }

35 void LstmStateManager::mergeStates(StateVariables const& vars,
36  std::vector<size_t>& prefix_lengths,
37  std::vector<HistoryState const*> const& prefix_states,
38  FeedDict& feed_dict,
39  TargetList& targets) {
40  require_eq(prefix_states.size(), prefix_lengths.size());
41  feed_dict.reserve(vars.size());
42  targets.reserve(vars.size());
43  Tensorflow::int64 batch_size = prefix_lengths.size();
44  for (size_t v = 0ul; v < vars.size(); v++) {
45  Tensorflow::int64 state_size = prefix_states.front()->at(v)->size();
46  Tensorflow::Tensor var_tensor = Tensorflow::Tensor::zeros<float>({batch_size, state_size});
47  float* data = var_tensor.data<f32>();
48  for (size_t b = 0ul; b < static_cast<size_t>(batch_size); b++) {
49  auto const& compressed_state = prefix_states[b]->at(v);
50  require_eq(compressed_state->size(), static_cast<size_t>(state_size));
51  compressed_state->uncompress(data + b * state_size, state_size);
52  }
53  feed_dict.emplace_back(vars[v].initial_value_name, var_tensor);
54  targets.emplace_back(vars[v].initializer_name);
55  }
56 }

58 std::vector<LstmStateManager::HistoryState> LstmStateManager::splitStates(StateVariables const& vars,
59  std::vector<size_t>& suffix_lengths,
60  std::vector<Tensorflow::Tensor> const& state_tensors,
61  CompressedVectorFactory<float> const& vector_factory) {
62  require_eq(vars.size(), state_tensors.size());
63 
64  std::vector<HistoryState> result(suffix_lengths.size());
65  for (size_t r = 0ul; r < suffix_lengths.size(); r++) {
66  result[r].reserve(vars.size());
67  suffix_lengths[r] = 1ul;
68  }
69 
70  for (size_t v = 0ul; v < vars.size(); v++) {
71  Tensorflow::Tensor const& tensor = state_tensors[v];
72  require_eq(tensor.numDims(), 2);
73  size_t batch_size = tensor.dimSize(0);
74  size_t state_size = tensor.dimSize(1);
75  require_eq(batch_size, suffix_lengths.size());
76 
77  for (size_t b = 0ul; b < batch_size; b++) {
78  auto compression_param_estimator = vector_factory.getEstimator();
79  float const* data = tensor.data<f32>(b, 0);
80  compression_param_estimator->accumulate(data, state_size);
81  auto compression_params = compression_param_estimator->estimate();
82  result[b].emplace_back(vector_factory.compress(data, state_size, compression_params.get()).release());
83  }
84  }
85 
86  return result;
87 }
