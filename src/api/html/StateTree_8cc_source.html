30 /*
31  * There are three types of transitions in the state tree:
32  * a) emitting (acoustic) transitions
33  * b) exit, or word end, transitions are taken when the word identity
34  * is known and lead into the tree copy for the corresponding new LM
35  * history.
36  * c) epsilon, pseudo exit, in-tree transit or "physical word end"
37  * transitions (not sure what the best name is) have been introduced
38  * to enforce the creation of book-keeping entries (traces aka
39  * backpointer) to record the physical word boundary time and score
40  * also when using early recombination.
41  *
42  * Emissions are coupled to the states following the principle:
43  * propagate, then emit. Thus a state's mixture can be interpreted as
44  * the emission for all transitions *entering* the state. Exit
45  * transitions are attached to their originating state, and they are
46  * taken *after* the emission.
47  *
48  * Depth is counted in acoustic (forward) transitions from beginning
49  * of word: I.e. the root state has depth 0, the first state of a word
50  * has depth 1, and pronunciations of length n end at depth n. */

222 inline const Am::Allophone* StateTree::getAllophone(const PronunciationSuffix& s) const {
223  const Am::Phonology& phonology = *(acousticModel_->phonology());

292 /**
293  * Go to the next (sub-) state in the pronciation.
294  * Successor pronuncation suffixes appended to @c out
295  * If the pronunciation ends, it is appended to @c ends
296  */
297 void StateTree::advancePronunciationSuffix(const PronunciationSuffix& _s,
298  std::back_insert_iterator<std::vector<PronunciationSuffix>> out,
299  std::back_insert_iterator<std::vector<PronunciationSuffix>> ends) const {
300  PronunciationSuffix s(_s);
301  require(!s.isEmpty());
302 
303  if (advanceSubState(s)) {
304  // next phoneme state reached
305  bool mustAdvancePhoneme = advancePhoneState(s);
306  if (mustAdvancePhoneme) {
307  // next phoneme reached
308  ++s.phoneme;
309  s.isHashValid_ = false;
310  if (s.phoneme >= s.pronunciation_->length()) {
311  // end of pronunciation reached.
312  s.isEmpty_ = true;
313  *ends++ = s;
314  return;
315  }
316 
317  if (s.phoneme > 0) // ?
318  s.predecessorPhoneme_ = Bliss::Phoneme::term;
319  }
320 
321  if (acousticModel_->isAcrossWordModelEnabled() &&
322  mustAdvancePhoneme && s.phoneme == s.pronunciation_->length() - 1) {
323  // last phoneme in the pronunciation
324  // create fan-out suffixes, i.e. create pronuncation alternatives
325  // using all initial phonemes as successor phoneme
326  if (phonemeInventory()->phoneme(s.currentPhoneme())->isContextDependent()) {
327  for (u32 r = 0; r < initialPhonemes_->size(); ++r) {
328  s.successorPhoneme_ = (*initialPhonemes_)[r];
329  *out++ = s;
330  }
331  }
332  s.successorPhoneme_ = Bliss::Phoneme::term;
333  }
334  }
335  *out++ = s;
336 }

338 u32 StateTree::nStatesRemaining(const PronunciationSuffix& _s) const {
339  u32 result = 0;
340  PronunciationSuffix s(_s);
341  while (s.phoneme < s.pronunciation_->length()) {
342  if (advanceSubState(s))
343  if (advancePhoneState(s))
344  ++s.phoneme;

346  }

350 int StateTree::compareSequences(const PronunciationSuffix& l, const PronunciationSuffix& r) const {
351  if (l.pronunciation() != r.pronunciation())
352  return (l.pronunciation() < r.pronunciation()) ? -1 : 1;
353  if (l.successorPhoneme_ != r.successorPhoneme_)
354  return (l.successorPhoneme_ < r.successorPhoneme_) ? -1 : 1;
355  if (l.phoneState != r.phoneState)
356  return l.phoneState - r.phoneState;
357  if (l.subState != r.subState)
358  return l.subState - r.subState;
359  if (l.nPhonemesRemaining() != r.nPhonemesRemaining())
360  return l.nPhonemesRemaining() - r.nPhonemesRemaining();
361  for (s32 p = 0; p < l.nPhonemesRemaining(); ++p) {
362  if ((*l.pronunciation_)[l.phoneme + p] != (*r.pronunciation_)[r.phoneme + p])
363  return (*l.pronunciation_)[l.phoneme + p] - (*r.pronunciation_)[r.phoneme + p];
364  }
365 
366  const Am::Allophone* la = getAllophone(l);
367  const Am::Allophone* ra = getAllophone(r);
368  verify(nPhoneStates(l) == nPhoneStates(r));
369  if (la != ra) {
370  for (s8 phoneState = l.phoneState; phoneState < nPhoneStates(l); ++phoneState) {
371  Am::AllophoneState las = acousticModel_->allophoneStateAlphabet()->allophoneState(la, phoneState);
372  Am::AllophoneState ras = acousticModel_->allophoneStateAlphabet()->allophoneState(ra, phoneState);
373  Mm::MixtureIndex lm = acousticModel_->emissionIndex(las);
374  Mm::MixtureIndex rm = acousticModel_->emissionIndex(ras);
375  if (lm != rm)
376  return lm - rm;
377  }
378  }
379 
380  verify_(hashSequence(l) == hashSequence(r));
381  return 0;
382 }

384 u32 StateTree::hashSequence(const PronunciationSuffix& s) const {
385  u32 result = u32(long(s.pronunciation() /* this is a pointer */));
386 
387  for (s32 p = 0; p < s.nPhonemesRemaining(); ++p) {
388  result = (result << 7) | (result >> 25);
389  result ^= u32((*s.pronunciation_)[s.phoneme + p]);
390  }
391  result ^= u32(s.successorPhoneme_) << 16;
392 
393  const Am::Allophone* a = getAllophone(s);
394  for (s16 state = s.phoneState; state < nPhoneStates(s); ++state) {
395  Am::AllophoneState as = acousticModel_->allophoneStateAlphabet()->allophoneState(a, state);
396  result = (result << 11) | (result >> 21);
397  result ^= u32(acousticModel_->emissionIndex(as));
398  }
399 
400  result += u32(s.subState);
401 
402  return result;
403 }

448  std::sort(sorted.begin(), sorted.end(), PronunciationSuffix::Ordering(model_));
449  verify(std::equal(sorted.begin(), sorted.end(), items_.begin(), PronunciationSuffix::Equality(model_)));
450 
451  u32 hash = 0;
452  for (ItemList::const_iterator i = items_.begin(); i != items_.end(); ++i) {
453  hash = (hash << 5) | (hash >> 27);
454  hash ^= PronunciationSuffix::Hash(model_)(*i);

456  verify(hash == hash_);
457  }

539 /**
540  * Split the suffix set in a set of suffix sets, such
541  * that each resulting suffix set contains pronunciation
542  * suffixes with equal first state (w.r.t the state description @see StateDesc)
543  */
544 void StateTree::SuffixSet::partition(SuffixSetList& conts) const {
545  require(isSorted());
546  require(conts.empty());
547 
548  typedef std::map<StateDesc, u32> ContsMap;
549  ContsMap contsMap;
550  for (ItemList::const_iterator i = items_.begin(); i < items_.end(); ++i) {
551  verify(!i->isEmpty());
552  StateDesc head(model_->head(*i));
553 
554  if (model_->isPathRecombinationInFanInEnabled_) {
555  ///@todo Change the hash structure, instead of abusing TransitionModelIndex
556  if ((i->phoneme == 0 && i->subState == model_->nSubStates(*i) - 1 && i->phoneState == model_->nPhoneStates(*i) - 1) || (i->phoneme == 1 && i->subState == 0 && i->phoneState == 0))
557  head.transitionModelIndex = i->currentPhoneme();
558  }
559 
560  ContsMap::iterator c = contsMap.find(head);
561  if (c == contsMap.end()) {
562  c = contsMap.insert(c, std::make_pair(head, conts.size()));
563  conts.push_back(SuffixSet(model_));
564  }
565  conts[c->second].addSorted(*i);
566  }
567 }

625 std::pair<Bliss::Phoneme::Id, Bliss::Phoneme::Id> StateTree::describeRootState(StateId s) const {
626  Bliss::Phoneme::Id final, initial;
627  if (s == root() || !coarticulationStructure_) {
628  // root state or no across word modelling
629  final = initial = Bliss::Phoneme::term;
630  }
631  else {
632  if (s < coarticulationStructure_->boundaryPhonemes.size()) {
633  final = coarticulationStructure_->boundaryPhonemes[s].final;
634  initial = coarticulationStructure_->boundaryPhonemes[s].initial;
635  }
636  else {
637  std::string text = Core::form("Unknown transit index %d. Context is set to non-coarticulated. ", s);
638  text += "It is likely that it occurs in combination with \"No active word end hypothesis at sentence end.\"";
639  warning(text.c_str());
640  final = initial = Bliss::Phoneme::term;
641  }
642  }
643  return std::make_pair(final, initial);
644 }

653  */
654 struct StateTree::BuildRequest {
655  Depth depth;
656  enum RequestType { unknown,
657  batchRequest,
658  leapRequest };
659  RequestType type;
660  BuildRequest(RequestType t)
661  : type(t) {}
662  virtual ~BuildRequest() {}
663  virtual void merge(const BuildRequest*) = 0;
664  virtual void satisfy(StateTree*) = 0;
665  virtual u32 hash() const = 0;
666  virtual bool equals(const BuildRequest*) const = 0;

668  struct DepthOrdering {
669  bool operator()(const BuildRequest* l, const BuildRequest* r) const {
670  if (l->depth >= 0) {
671  if (r->depth < 0)
672  return true;
673  }
674  else {
675  if (r->depth >= 0)
676  return false;
677  }
678  return l->depth < r->depth;
679  }
680  };
681 
682  struct Hash {
683  u32 operator()(const BuildRequest* r) const {
684  return r->hash();
685  }
686  };
687 
688  struct Equality {
689  bool operator()(const BuildRequest* l, const BuildRequest* r) const {
690  return (l == r) || l->equals(r);
691  }
692  };
693 };

695 /**
696  * Build a new batch, i.e. the sets of successors for a set of states

698 struct StateTree::BatchRequest : public BuildRequest {

703  BatchRequest(StateTree* m, StateId s)

708  }

710  virtual void merge(const BuildRequest*);
711  virtual void satisfy(StateTree* t);
712 
713  virtual u32 hash() const {

718  if (type != _r->type)
719  return false;
720  const BatchRequest* r = dynamic_cast<const BatchRequest*>(_r);
721  require(r);

723  }

999 bool StateTree::isConsistent() const {
1000  bool result = true;
1001  if (!haveSuccessorBatches_)
1002  return result;
1003  // check 2nd order consistency
1004  for (StateId s = 0; s < nStates(); ++s) {
1005  StateTree::StateId ss, ss_end, sss, sss_end, ss2, ss2_end;
1006  tie(ss2, ss2_end) = successors2(s);
1007  for (tie(ss, ss_end) = successors(s); ss < ss_end; ++ss)
1008  for (tie(sss, sss_end) = successors(ss); sss < sss_end; ++sss) {
1009  bool r = (sss == ss2++);
1010  result &= r;
1011  if (!r)
1012  warning("2nd order mesh structure s=%d, ss=%d, sss=%d, ss2=%d",
1013  s, ss, sss, ss2);
1014  }
1015  bool r = (ss2 == ss2_end);
1016  if (!r)
1017  warning("2nd order mesh structure s=%d, ss2=%d, ss2_end=%d",
1018  s, ss2, ss2_end);
1019  result &= r;
1020  }
1021 
1022  return result;
1023 }

1193 /**
1194  * Create successor tree nodes and submit respective build requests.
1195  */
1196 StateTree::BatchId StateTree::createBatch(BatchRequest* request) {
1197  SuffixSet::SuffixSetList conts;
1198  SuffixSet::ItemList ends;
1199 
1200  request->suffixes.partition(conts);
1201  require(conts.size() > 0);
1202 
1203  BatchId batch = batches_.size() - 1;
1204  verify(batches_[batch] == StateId(states_.size()));
1205  State m;
1206  m.depth = request->depth;
1207  for (SuffixSet::SuffixSetList::iterator c = conts.begin(); c != conts.end(); ++c) {
1208  m.desc = c->head(); // before c->advance(), all pronunciation suffixes have the same head
1209  c->advance(ends);
1210  createExits(ends, m.exits);
1211  if (shouldLeap(*c, m.depth)) {
1212  LeapRequest* l = new LeapRequest(this, states_.size());
1213  c->leap(l->leaps.front());
1214  c->sortAndHash();
1215  l->depth = -c->nStatesRemaining();
1216  l->suffixes = *c;
1217  m.successors = emptyBatch_;
1218  requests_->submit(l);
1219  }
1220  else if (c->size() > 0) {
1221  c->sortAndHash();
1222  BatchRequest* r = new BatchRequest(this, states_.size());
1223  r->depth = m.depth + 1;
1224  r->parentsSuffixes = request->suffixes;
1225  r->suffixes = *c;
1226  m.successors = invalidBatch;
1227  requests_->submit(r);
1228  }
1229  else {
1230  m.successors = emptyBatch_;
1231  }
1232  states_.push_back(m);
1233  m.exits.clear();
1234  }
1235  batches_.push_back(states_.size());
1236  return batch;
1237 }

1263 /**
1264  * Fill successorBatches_, the successors of batches
1265  */
1266 void StateTree::createSecondOrderBatches() {
1267  log("building second order structure...");
1268 
1269  BatchId nFirstOrderBatches = batches_.size() - 1;
1270  haveSuccessorBatches_ = true;
1271 
1272  for (BatchId b = 0; b < nFirstOrderBatches; ++b) {
1273  BatchId b2 = emptyBatch_;
1274  s32 nStates = batches_[b + 1] - batches_[b];
1275  if (nStates == 1) {
1276  b2 = states_[batches_[b]].successors;
1277  }
1278  else if (nStates > 1) {
1279  StateId begin = Core::Type<StateId>::max;
1280  StateId end = Core::Type<StateId>::min;
1281  for (StateId s = batches_[b]; s < batches_[b + 1]; ++s) {
1282  BatchId sb = states_[s].successors;
1283  if (sb == emptyBatch_)
1284  continue;

1286  begin = batches_[sb];

1357  depth, thisLevel.nStates, float(thisLevel.nChildren) / float(thisLevel.nStates), thisLevel.nExits);
