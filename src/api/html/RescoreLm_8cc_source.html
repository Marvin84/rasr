64  LexiconRef lexicon = Lexicon::us();

67  Core::Ref<const Bliss::LemmaPronunciationAlphabet> lpAlphabet;

71  lAlphabet = Lexicon::us()->lemmaAlphabet();
72  break;
73  case Lexicon::LemmaPronunciationAlphabetId:
74  lpAlphabet = Lexicon::us()->lemmaPronunciationAlphabet();
75  break;
76  default:
77  defect();
78  }
79 
80  if (dynamic_cast<Lm::ScaledLanguageModel*>(lm.get()))

92  ScoreId lmScoreId = semiring->id("lm");

95

107  typedef std::multimap<std::pair<Speech::TimeframeIndex, Fsa::StateId>, WordEndHypothesis> Hypotheses;
108  Hypotheses hypotheses;
109 
110  {
111  WordEndHypothesis initialHyp;
112  initialHyp.h = lm->startHistory();
113  initialHyp.score = 0;
114  initialHyp.arc = Arc(Fsa::InvalidStateId, ScoresRef(), Fsa::Epsilon, Fsa::Epsilon);
115  for (u32 i = 0; i < prefix.size(); ++i)
116  Lm::addLemmaScore(lm, lmScale, prefix[i], lmScale, initialHyp.h, initialHyp.score);
117  initialHyp.preState = Fsa::InvalidStateId;
118  hypotheses.insert(std::make_pair(std::make_pair(lat->boundary(lat->initialStateId()).time(), lat->initialStateId()), initialHyp));
119  }
120 
121  Speech::TimeframeIndex lastPrunedTimeframe = -1;
122 
123  while (!hypotheses.empty()) {
124  Speech::TimeframeIndex time = hypotheses.begin()->first.first;
125 
126  if (time != lastPrunedTimeframe) {
127  // Step 1: Prune hypotheses for timeframe
128  Hypotheses::iterator frameBegin = hypotheses.begin();
129  Hypotheses::iterator frameEnd = hypotheses.upper_bound(std::make_pair(time, Core::Type<Fsa::StateId>::max));
130  Score best = Core::Type<Score>::max;
131  for (Hypotheses::iterator hypIt = frameBegin; hypIt != frameEnd; ++hypIt) {
132  if (hypIt->second.score < best)
133  best = hypIt->second.score;
134  }
135  u32 count = 0, pruned = 0;
136  for (Hypotheses::iterator hypIt = frameBegin; hypIt != frameEnd;) {
137  if (hypIt->second.score > best + wordEndBeam) {
138  Hypotheses::iterator removeIt = hypIt;
139  ++hypIt;
140  ++pruned;
141  hypotheses.erase(removeIt);
142  }
143  else {
144  ++hypIt;
145  ++count;
146  }
147  }
148  // The first element may have been deleted
149  frameBegin = hypotheses.begin();
150  if (count > wordEndLimit) {
151  std::vector<std::pair<Hypotheses::iterator, Score>> scores;
152  for (Hypotheses::iterator hypIt = frameBegin; hypIt != frameEnd; ++hypIt)
153  scores.push_back(std::make_pair(hypIt, hypIt->second.score));
154 
155  std::nth_element(scores.begin(), scores.begin() + wordEndLimit, scores.end(), CompareSecond<Hypotheses::iterator, Score>());
156  for (std::vector<std::pair<Hypotheses::iterator, Score>>::const_iterator deleteIt = scores.begin() + wordEndLimit; deleteIt != scores.end(); ++deleteIt)
157  hypotheses.erase(deleteIt->first);
158  }
159  frameBegin = hypotheses.begin();
160  lastPrunedTimeframe = time;
161  }
162 
163  verify(!hypotheses.empty());
164 
165  Fsa::StateId stateId = hypotheses.begin()->first.second;
166  ConstStateRef state = lat->getState(stateId);
167 
168  Hypotheses::iterator stateEnd = hypotheses.upper_bound(std::make_pair(time, stateId));
169 
170  // Step 2: Recombine and build (create a state for each each distinct history, and build the corresponding incoming arcs)
171  std::multimap<Lm::History, WordEndHypothesis> historyHyps;
172  for (Hypotheses::iterator hypIt = hypotheses.begin(); hypIt != stateEnd; ++hypIt)
173  historyHyps.insert(std::make_pair(hypIt->second.h, hypIt->second));
174 
175  while (!historyHyps.empty()) {
176  Lm::History history = historyHyps.begin()->first;
177  std::multimap<Lm::History, WordEndHypothesis>::iterator endIt = historyHyps.upper_bound(historyHyps.begin()->first);
178 
179  State* newState = s->newState(state->tags());
180  b->set(newState->id(), lat->boundary(stateId));
181  Score best = Core::Type<Score>::max;
182 
183  if (newState->isFinal()) {
184  bool hadSentenceEnd = false;
185  if (historyHyps.begin()->second.preState != Fsa::InvalidStateId) {
186  Fsa::LabelId labelId = historyHyps.begin()->second.arc.input();
187 
188  if (Fsa::FirstLabelId <= labelId && labelId <= Fsa::LastLabelId) {
189  const Bliss::Lemma* lemma = (lAlphabet) ? lAlphabet->lemma(labelId) : lpAlphabet->lemmaPronunciation(labelId)->lemma();
190  verify(lemma);
191  if (lemma->hasSyntacticTokenSequence() && lemma->syntacticTokenSequence().size() && lemma->syntacticTokenSequence().operator[](lemma->syntacticTokenSequence().size() - 1) == lm->sentenceEndToken()) {
192  hadSentenceEnd = true;
193  }
194  }
195  }
196 
197  if (!hadSentenceEnd) {
198  appendFinalState.push_back(std::make_pair(newState->id(), history));
199  newState->setTags(newState->tags() & ~Fsa::StateTagFinal);
200  }
201  }
202 
203  for (std::multimap<Lm::History, WordEndHypothesis>::const_iterator hypIt = historyHyps.begin();
204  hypIt != endIt; ++hypIt) {
205  if (hypIt->second.score < best)
206  best = hypIt->second.score;
207 
208  if (hypIt->second.preState != Fsa::InvalidStateId)
209  const_cast<State&>(*s->getState(hypIt->second.preState)).newArc(newState->id(), hypIt->second.arc.weight(), hypIt->second.arc.input(), hypIt->second.arc.output());
210  }
211 
212  // Step 3: Create successor hypotheses
213  for (u32 arcI = 0; arcI < state->nArcs(); ++arcI) {
214  const Arc* arc = state->getArc(arcI);
215  WordEndHypothesis nextHyp;
216  nextHyp.h = history;
217  nextHyp.score = best;
218  nextHyp.preState = newState->id();
219  nextHyp.arc = *arc;
220  nextHyp.arc.setWeight(semiring->clone(nextHyp.arc.weight()));
221 
222  Fsa::LabelId labelId = arc->input();
223 
224  if (Fsa::FirstLabelId <= labelId && labelId <= Fsa::LastLabelId) {
225  const Bliss::Lemma* lemma = (lAlphabet) ? lAlphabet->lemma(labelId) : lpAlphabet->lemmaPronunciation(labelId)->lemma();
226  verify(lemma);
227 
228  Bliss::Lemma::Id id = lemma->id();
229  LmCache::const_iterator lmCacheIt = lmCache.find(std::make_pair(history.handle(), id));
230 
231  if (lmCacheIt != lmCache.end()) {
232  nextHyp.h = lmCacheIt->second.second;
233  nextHyp.arc.setScore(lmScoreId, lmCacheIt->second.first);
234  }
235  else {
236  Score rawScore = 0;
237  verify(nextHyp.h.isValid());
238  Lm::addLemmaScore(lm, 1.0, lemma, 1.0, nextHyp.h, rawScore);
239  nextHyp.arc.setScore(lmScoreId, rawScore);
240  lmCache.insert(std::make_pair(std::make_pair(history.handle(), id), std::make_pair(rawScore, nextHyp.h)));
241  }
242  }
243  else {
244  nextHyp.arc.setScore(lmScoreId, 0);
245  }
246  nextHyp.score += semiring->project(nextHyp.arc.weight());
247  hypotheses.insert(std::make_pair(std::make_pair(lat->boundary(arc->target()).time(), arc->target()), nextHyp));
248  }
249 
250  historyHyps.erase(historyHyps.begin(), endIt);
251  }
252 
253  hypotheses.erase(hypotheses.begin(), hypotheses.upper_bound(std::make_pair(time, stateId)));
254 
255  if (lmCache.size() > maxCacheSize)
256  lmCache.clear();
257  }
258 
259  s->setInitialStateId(lat->initialStateId());
260 
261  if (appendFinalState.size()) {
262  State* finalState = s->newState(Fsa::StateTagFinal);
263  Flf::Boundary finalB = s->boundary(appendFinalState.front().first);
264  finalB.setTime(finalB.time() + 1); // Be consistent with the recognizer: It also appends a length-1 epsilon arc
265  std::cout << "decoding creating final state for " << appendFinalState.size() << " predecessor states, time " << finalB.time() << std::endl;
266  b->set(finalState->id(), finalB);
267  for (std::vector<std::pair<Fsa::StateId, Lm::History>>::const_iterator it = appendFinalState.begin();
268  it != appendFinalState.end(); ++it) {
269  Flf::Arc* finalArc = const_cast<State&>(*s->getState(it->first)).newArc();
270  finalArc->setTarget(finalState->id());
271  finalArc->setInput(Fsa::Epsilon);
272  finalArc->setOutput(Fsa::Epsilon);
273  finalArc->setWeight(semiring->clone(semiring->defaultWeight()));
274 
275  Lm::History h = it->second;
276  Flf::Score s = 0;
277 
278  for (u32 i = 0; i < suffix.size(); ++i)
279  Lm::addLemmaScore(lm, 1.0, suffix[i], 1.0, h, s);
280 
281  s += lm->sentenceEndScore(h);
282 
283  finalArc->setScore(lmScoreId, s);
284  }
285  }
286 
287  s->setDescription("decode_rescore(" + lat->describe() + ")");
288 
289  std::cout << "ready" << std::endl;
290 
291  // Last step: Trim the resulting lattice, removing dead-end paths
292  StaticLatticeRef createdLattice(s);
293  trimInPlace(createdLattice);
294  return copyBoundaries(createdLattice);
295 }
