23  Core::Ref<const Am::AllophoneAlphabet> allophones = am->allophoneAlphabet();
24  for (std::vector<std::string>::const_iterator p = phones.begin(); p != phones.end(); ++p) {
25  const Bliss::Phoneme* phone = pi->phoneme(*p);
26  verify(phone);
27  Am::AllophoneIndex ai = allophones->index(
28  Am::Allophone(phone->id(), Am::Allophone::isInitialPhone | Am::Allophone::isFinalPhone));
29  const Am::Allophone* allophone = allophones->allophone(ai);
30  verify(allophone);
31  StateSequence states;
32  states.createFromAllophone(am, allophone);
33  StateSequenceList::const_iterator s = std::find(stateSequences.begin(), stateSequences.end(), states);
34  if (s == stateSequences.end()) {
35  Core::Application::us()->error("unknown non-word allophone %s (phone %s)",
36  allophones->toString(*allophone).c_str(), p->c_str());
37  return false;
38  }
39  else {
40  nonWordHmms_.insert(&(*s));
41  }
42  }
43  return true;
44 }
