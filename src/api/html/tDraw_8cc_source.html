48  DrawDotDfsState(DotDrawer<_Automaton>& dd, _ConstAutomatonRef f)
49  : Precursor(f), dd_(dd) {
50  if (dd_.hints_ & Fsa::HintMarkBest) {
51  potentials_ = sssp<_Automaton>(transpose<_Automaton>(Precursor::fsa_, false));
52  _ConstAutomatonRef tmp = best<_Automaton>(Precursor::fsa_, potentials_);
53  isOnBestPath_.resize(potentials_.size(), false);
54  std::fill(isOnBestPath_.begin(), isOnBestPath_.end(), false);
55  Fsa::Stack<Fsa::StateId> S;
56  if (tmp->initialStateId() != Fsa::InvalidStateId)
57  S.push(tmp->initialStateId());
58  while (!S.isEmpty()) {
59  Fsa::StateId s = S.pop();
60  _ConstStateRef sp = tmp->getState(s);
61  isOnBestPath_[s] = true;
62  if (sp->nArcs() == 1)
63  S.push(sp->begin()->target());
64  }
65  }
66  }

75  void discoverState(_ConstStateRef sp) {
76  dd_.os_ << "n" << sp->id() << " [label=\"";
77  if (dd_.hints_ & Fsa::HintShowDetails)
78  Precursor::fsa_->dumpState(sp->id(), dd_.os_);
79  else
80  dd_.os_ << sp->id();
81  if (sp->isFinal())
82  if (!Precursor::fsa_->semiring()->isDefault(sp->weight_))
83  dd_.os_ << "/" << Precursor::fsa_->semiring()->describe(sp->weight_, dd_.hints_);
84  dd_.os_ << "\"";
85  if (sp->isFinal())
86  dd_.os_ << ",shape=doublecircle";
87  else
88  dd_.os_ << ",shape=circle";
89  if (sp->id() == Precursor::fsa_->initialStateId())
90  dd_.os_ << ",style=bold";
91  if ((dd_.hints_ & Fsa::HintMarkBest) && isOnBestPath_[sp->id()])
92  dd_.os_ << ",color=red";
93  dd_.os_ << "]" << std::endl;
94  typename _State::const_iterator bestArc = sp->end();
95  if ((dd_.hints_ & Fsa::HintMarkBest) && isOnBestPath_[sp->id()]) {
96  _Weight minWeight = Precursor::fsa_->semiring()->max();
97  for (typename _State::const_iterator a = sp->begin(); a != sp->end(); ++a)
98  if (isOnBestPath_[a->target()]) {
99  _Weight w = Precursor::fsa_->semiring()->extend(a->weight_, potentials_[a->target()]);
100  if (Precursor::fsa_->semiring()->compare(w, minWeight) < 0) {
101  minWeight = w;
102  bestArc = a;
103  }
104  }
105  }
106  for (typename _State::const_iterator a = sp->begin(); a != sp->end(); ++a) {
107  std::string symbol;
108  dd_.os_ << "n" << sp->id() << " -> n" << a->target() << " [";
109  if (a == bestArc)
110  dd_.os_ << "color=red,";
111  dd_.os_ << "label=\"";
112  if (Precursor::fsa_->getInputAlphabet())
113  symbol = Precursor::fsa_->getInputAlphabet()->symbol(a->input());
114  if ((!Precursor::fsa_->getInputAlphabet()) || (symbol.empty()))
115  dd_.os_ << a->input();
116  else
117  dd_.os_ << replaceQuotationMarks(symbol);
118  if (Precursor::fsa_->type() == Fsa::TypeTransducer) {
119  dd_.os_ << ":";
120  if (Precursor::fsa_->getOutputAlphabet())
121  symbol = Precursor::fsa_->getOutputAlphabet()->symbol(a->output());
122  if ((!Precursor::fsa_->getOutputAlphabet()) || (symbol.empty()))
123  dd_.os_ << a->output();
124  else
125  dd_.os_ << replaceQuotationMarks(symbol);
126  }
127  if (!Precursor::fsa_->semiring()->isDefault(a->weight()))
128  dd_.os_ << "/" << Precursor::fsa_->semiring()->describe(a->weight(), dd_.hints_);
129  dd_.os_ << "\"]" << std::endl;
130  }
131  }
