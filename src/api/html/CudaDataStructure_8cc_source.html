47 void CudaDataStructure::initialize() {
48  if (initialized)
49  return;
50  initialized = true;
51 
52  UseCudaMode useCudaMode = autoUseCuda;
53  if (!Core::Application::us())
54  warning("CudaDataStructure: no application, use CUDA if available");
55  else
56  useCudaMode = (UseCudaMode)paramUseCuda(Core::Application::us()->getConfiguration());
57 
58  // check whether MODULE_CUDA is active and a GPU is available
59  int nGpus = 0;
60  bool hasCuda = false;
61  if (useCudaMode == dontUseCuda) {
62  hasGpu_ = false;
63  log("CUDA is disabled via config");
64  }
65  else {
66  cudaError_t success = Cuda::getNumberOfGpus(nGpus, hasCuda);
67  if (!hasCuda)
68  hasGpu_ = false;
69  else if (success != cudaSuccess || nGpus <= 0) {
70  // no GPU available, or some error occured
71  hasGpu_ = false;
72  std::ostringstream ss;
73  ss << "Using binary with GPU support, but no GPU available.";
74  if (success != cudaSuccess) {
75  ss << " Error code is: " << success
76  << " (" << Cuda::getErrorString(success) << ")";
77  }
78  if (nGpus != 0) {
79  // this should never occur (when no GPU is available, a non-zero error code is returned)
80  ss << " Strange, number of GPUs is: " << nGpus;
81  }
82  if (useCudaMode == autoUseCuda) {
83  log(ss.str());
84  }
85  else {
86  ss << " This is critical with use-cuda = true.";
87  criticalError(ss.str());
88  }
89  }
90  else {
91  // We have some GPUs on this system.
92  hasGpu_ = true;
93  }
94  }
95 
96  if (hasGpu_) {
97 #ifdef MODULE_CUDA
98  // find free device first
99  {
100  cudaError_t status = cudaErrorInvalidDevice;
101  std::ostringstream ss;
102 
103  for (int d = 0; d < nGpus; ++d) {
104  ss.str("");
105  ss.clear();
106  ss << "Trying to cudaSetDevice on GPU " << d;
107  log(ss.str());
108  cudaSetDevice(d); // this sometimes returns 0 on occupied GPUs
109  status = cudaFree(0); // this always fails on occupied GPUs
110  if (status == cudaSuccess)
111  break;
112  }
113  if (status != cudaSuccess)
114  criticalError("Failed to get a GPU handle.");
115  }
116 #endif
117  // initialize cuBLAS and cuRAND
118  {
119  cublasStatus_t success = Cuda::createCublasHandle(cublasHandle);
120  if (success != CUBLAS_STATUS_SUCCESS) {
121  std::ostringstream serr;
122  serr << "Failed to initialize cuBLAS library: Error code is: " << success;
123  serr << " (" << Cuda::cublasGetErrorString(success) << ")";
124  criticalError(serr.str());
125  }
126  }
127  {
128  curandStatus_t success = Cuda::createRandomNumberGenerator(randomNumberGenerator, CURAND_RNG_PSEUDO_DEFAULT);
129  if (success != CURAND_STATUS_SUCCESS) {
130  std::ostringstream serr;
131  serr << "Failed to initialize cuRAND random number generator library: Error code is: " << success;
132  serr << " (" << Cuda::curandGetErrorString(success) << ")";
133  criticalError(serr.str());
134  }
135  }
136  {
137  curandStatus_t success = Cuda::setSeed(randomNumberGenerator, (unsigned long long)Math::rand());
138  if (success != CURAND_STATUS_SUCCESS) {
139  std::ostringstream serr;
140  serr << "Failed to set seed for cuRAND random number generator: Error code is: " << success;
141  serr << " (" << Cuda::curandGetErrorString(success) << ")";
142  criticalError(serr.str());
143  }
144  }
145 
146 #ifdef MODULE_CUDA
147  // Get the current active GPU which we are going to use.
148  // Do this after CuBLAS init because otherwise it might return the wrong one.
149  // See related comment in Theano CUDA init code:
150  // https://github.com/Theano/Theano/blob/master/theano/sandbox/cuda/__init__.py
151  cudaError_t ret = cudaGetDevice(&activeGpu_);
152  if (ret != cudaSuccess) {
153  criticalError(std::string() + "Cannot query current active GPU, error " + std::to_string(ret) + " (" + Cuda::getErrorString(ret) + ")");
154  }
155  else {
156  log(std::string() + "Using GPU " + std::to_string(activeGpu_ + 1) + " (= idx + 1) of " + std::to_string(nGpus) + " GPUs");
157  }
158 #endif
159  }
160 }
