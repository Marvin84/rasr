68  XmlRegularElement* dependency = new XmlRegularElementRelay("dependency", this,

70  XmlRegularElementRelay::endHandler(&Self::endDependency));
71  collect(dependency);
72  dependency->addTransition(0, 1, collect(new XmlStringBuilderElement("value", this, XmlStringBuilderElement::handler(&Self::setDependencyValue))));

77  collect(newDependencySet);

149 bool DependencySet::operator==(const DependencySet& s) const {
150  bool isEqual = true;
151  if (dependencies_.size() != s.dependencies_.size()) {
152  Application::us()->warning("dependency sets differ in size");
153  isEqual = false;
154  }
155  for (Dependencies::const_iterator d = dependencies_.begin(); d != dependencies_.end(); ++d) {
156  Dependencies::const_iterator found = s.dependencies_.find(d->first);
157  if (found == s.dependencies_.end()) {
158  Application::us()->warning("dependency named '%s' with value '%s' generated on %s not in second set",
159  d->first.c_str(), d->second.value().c_str(),
160  d->second.date().c_str());
161  isEqual = false;
162  }
163  else {
164  if (d->second.value() != found->second.value()) {
165  Application::us()->warning("dependency named '%s' with value '%s' generated on %s differs from value '%s' generated on %s",
166  d->first.c_str(), d->second.value().c_str(),
167  d->second.date().c_str(), found->second.value().c_str(),
168  found->second.date().c_str());
169  isEqual = false;
170  }
171  }
172  }
173  return isEqual;
