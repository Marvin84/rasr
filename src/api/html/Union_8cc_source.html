29 class UnionBoundaries : public Boundaries {
30 private:
31  Fsa::StateId nLats_;
32  ConstBoundariesRefList boundaries_;
33  Boundary initialBoundary_;
34 
35 private:
36  std::pair<Fsa::StateId, Fsa::StateId> fromUnionSid(Fsa::StateId unionSid) const {
37  Fsa::StateId sid = (unionSid - 1) / nLats_;
38  return std::make_pair(unionSid - 1 - sid * nLats_, sid);
39  }
40 
41 public:
42  UnionBoundaries(const ConstLatticeRefList& lats)
43  : Boundaries(), nLats_(lats.size()), boundaries_(lats.size()) {
44  boundaries_[0] = lats[0]->getBoundaries();
45  initialBoundary_ = boundaries_[0]->get(lats[0]->initialStateId());
46  for (u32 i = 1; i < nLats_; ++i) {
47  boundaries_[i] = lats[i]->getBoundaries();
48  if (initialBoundary_.valid()) {
49  const Boundary& currentInitialBoundary(
50  boundaries_[i]->get(lats[i]->initialStateId()));
51  if (currentInitialBoundary.time() != initialBoundary_.time())
52  initialBoundary_ = InvalidBoundary;
53  else if (!(currentInitialBoundary.transit() == initialBoundary_.transit()))
54  initialBoundary_.setTransit(Boundary::Transit());
55  }
56  }
57  }
58  virtual ~UnionBoundaries() {}
59 
60  virtual bool valid() const {
61  return true;
62  }
63 
64  virtual bool valid(Fsa::StateId unionSid) const {
65  return get(unionSid).valid();
66  }
67 
68  virtual const Boundary& get(Fsa::StateId unionSid) const {
69  if (unionSid == 0)
70  return initialBoundary_;
71  else {
72  std::pair<Fsa::StateId, Fsa::StateId> sid = fromUnionSid(unionSid);
73  return boundaries_[sid.first]->get(sid.second);
74  }
75  }

122  virtual ConstStateRef getState(Fsa::StateId unionSid) const {
123  State* sp;
124  if (unionSid == 0) {
125  sp = new State(0);
126  for (u32 i = 0; i < nLats_; ++i) {
127  ConstStateRef sr = lats_[i]->getState(lats_[i]->initialStateId());
128  if (sr->isFinal())
129  sp->newArc(toUnionSid(i, lats_[i]->initialStateId()),
130  semiring_->one(), Fsa::Epsilon);
131  for (State::const_iterator a = sr->begin(); a != sr->end(); ++a)
132  sp->newArc(toUnionSid(i, a->target()), a->weight(),
133  a->input(), a->output());
134  }
135  }
136  else {
137  std::pair<Fsa::StateId, Fsa::StateId> sid = fromUnionSid(unionSid);
138  ConstStateRef sr = lats_[sid.first]->getState(sid.second);
139  sp = new State(*sr);
140  sp->setId(unionSid);
141  for (State::iterator a = sp->begin(); a != sp->end(); ++a)
142  a->target_ = toUnionSid(sid.first, a->target());
143  }
144  return ConstStateRef(sp);
145  }

156 ConstLatticeRef unite(const ConstLatticeRefList& lats, ConstSemiringRef semiring) {
157  switch (lats.size()) {
158  case 0:
159  return ConstLatticeRef();
160  case 1:
161  if (semiring)
162  return changeSemiring(lats[0], semiring);
163  else
164  return lats[0];
165  default:
166  if (!semiring) {
167  semiring = lats[0]->semiring();
168  for (u32 i = 1; i < lats.size(); ++i)
169  if (!Semiring::equal(semiring, lats[0]->semiring()))
170  Core::Application::us()->criticalError(
171  "Lattice union requires equal semirings, but \"%s\" != \"%s\".",
172  semiring->name().c_str(), lats[i]->semiring()->name().c_str());
173  }
174  return ConstLatticeRef(new UnionLattice(lats, semiring));
175  }
176 }

266 ConstLatticeRef expandTransits(ConstLatticeRef lat, Bliss::Phoneme::Id leftContext, Bliss::Phoneme::Id rightContext) {
267  LexiconRef lexicon = Lexicon::us();
268  verify(lat->getBoundaries()->valid());
269  lat = sortByTopologicalOrder(lat);
270  StaticBoundaries* b = new StaticBoundaries;
271  StaticLattice* s = new StaticLattice(Fsa::TypeAcceptor);
272  s->setProperties(lat->knownProperties(), lat->properties());
273  s->setInputAlphabet(lat->getInputAlphabet());
274  s->setSemiring(lat->semiring());
275  s->setBoundaries(ConstBoundariesRef(b));
276 
277  verify(Lexicon::us()->alphabetId(lat->getInputAlphabet()) == Lexicon::LemmaPronunciationAlphabetId);
278  Core::Ref<const Bliss::LemmaPronunciationAlphabet> lpAlphabet = Lexicon::us()->lemmaPronunciationAlphabet();
279 
280  typedef std::pair<Fsa::StateId, Flf::Arc> InArc;
281  typedef std::multimap<Fsa::StateId, InArc> Hypotheses;
282  Hypotheses inArcs;
283  inArcs.insert(std::make_pair(lat->initialStateId(), InArc(Fsa::InvalidStateId, Flf::Arc())));
284 
285  while (!inArcs.empty()) {
286  Fsa::StateId oldStateId = inArcs.begin()->first;
287  ConstStateRef oldState = lat->getState(oldStateId);
288 
289  Hypotheses::iterator inHypEnd = inArcs.upper_bound(oldStateId);
290 
291  std::set<std::pair<int, int>> transits;
292  typedef std::multimap<int, InArc> FinalPhoneHyps;
293  FinalPhoneHyps finalPhoneHyps;
294  std::multimap<int, u32> initialPhoneArcs;
295  std::set<int> finalPhones, initialPhones;
296 
297  bool hasRealInArcs = false;
298 
299  for (Hypotheses::const_iterator inHypIt = inArcs.begin();
300  inHypIt != inHypEnd; ++inHypIt) {
301  Bliss::Phoneme::Id final = Bliss::Phoneme::term;
302  if (inHypIt->second.first == Fsa::InvalidStateId || inHypIt->second.second.input() == Fsa::Epsilon) {
303  final = leftContext;
304  }
305  else {
306  const Bliss::LemmaPronunciation* pron = lpAlphabet->lemmaPronunciation(inHypIt->second.second.input());
307  if (pron && pron->pronunciation()->length()) {
308  hasRealInArcs = true;
309  final = pron->pronunciation()->phonemes()[pron->pronunciation()->length() - 1];
310  }
311  }
312  if (final != Bliss::Phoneme::term && lexicon->phonemeInventory()->phoneme(final)->isContextDependent()) {
313  finalPhones.insert(final);
314  finalPhoneHyps.insert(std::make_pair((int)final, inHypIt->second));
315  }
316  else {
317  finalPhoneHyps.insert(std::make_pair((int)Bliss::Phoneme::term, inHypIt->second));
318  }
319  }
320 
321  bool epsilonToFinal = false;
322 
323  for (u32 arcI = 0; arcI < oldState->nArcs(); ++arcI) {
324  Bliss::Phoneme::Id initial = Bliss::Phoneme::term;
325  if (oldState->getArc(arcI)->input() == Fsa::Epsilon) {
326  if (lat->getState(oldState->getArc(arcI)->target())->isFinal())
327  epsilonToFinal = true;
328  else
329  Core::Application::us()->warning() << "Found a non-initial epsilon arc, transit boundaries will not be correct, target " << oldState->getArc(arcI)->target() << " am score " << oldState->getArc(arcI)->score(0) << " lm score " << oldState->getArc(arcI)->score(1);
330  }
331  const Bliss::LemmaPronunciation* pron = lpAlphabet->lemmaPronunciation(oldState->getArc(arcI)->input());
332  if (pron && pron->pronunciation()->length())
333  initial = pron->pronunciation()->phonemes()[0];
334  if (initial != Bliss::Phoneme::term && lexicon->phonemeInventory()->phoneme(initial)->isContextDependent()) {
335  initialPhones.insert(initial);
336  initialPhoneArcs.insert(std::make_pair((int)initial, arcI));
337  }
338  else {
339  initialPhoneArcs.insert(std::make_pair((int)Bliss::Phoneme::term, arcI));
340  }
341  }
342 
343  if (rightContext != Bliss::Phoneme::term && (oldState->isFinal() || epsilonToFinal))
344  initialPhones.insert(rightContext);
345 
346  for (std::set<int>::const_iterator finalIt = finalPhones.begin(); finalIt != finalPhones.end(); ++finalIt)
347  for (std::set<int>::const_iterator initialIt = initialPhones.begin(); initialIt != initialPhones.end(); ++initialIt)
348  transits.insert(std::make_pair(*finalIt, *initialIt));
349 
350  if (transits.empty() || finalPhoneHyps.count(Bliss::Phoneme::term) || initialPhoneArcs.count(Bliss::Phoneme::term)) {
351  transits.clear();
352  transits.insert(std::make_pair<int, int>((int)Bliss::Phoneme::term, (int)Bliss::Phoneme::term));
353  }
354 
355  verify(!transits.empty());
356 
357  for (std::set<std::pair<int, int>>::iterator transitIt = transits.begin(); transitIt != transits.end(); ++transitIt) {
358  bool isContextDependent = transitIt->first != Bliss::Phoneme::term;
359 
360  State* newState = s->newState(oldState->tags(), oldState->weight());
361  b->set(newState->id(), Flf::Boundary(lat->boundary(oldStateId).time(), Flf::Boundary::Transit(transitIt->first, transitIt->second)));
362 
363  if (oldState->id() == lat->initialStateId()) {
364  verify(s->initialStateId() == Fsa::InvalidStateId);
365  s->setInitialStateId(newState->id());
366  }
367 
368  FinalPhoneHyps::const_iterator finalHypsBegin = isContextDependent ? finalPhoneHyps.lower_bound(transitIt->first) : finalPhoneHyps.begin();
369  FinalPhoneHyps::const_iterator finalHypsEnd = isContextDependent ? finalPhoneHyps.upper_bound(transitIt->first) : finalPhoneHyps.end();
370  for (FinalPhoneHyps::const_iterator finalHypIt = finalHypsBegin;
371  finalHypIt != finalHypsEnd; ++finalHypIt)
372  if (finalHypIt->second.first != Fsa::InvalidStateId)
373  const_cast<State&>(*s->getState(finalHypIt->second.first)).newArc(newState->id(), finalHypIt->second.second.weight(), finalHypIt->second.second.input(), finalHypIt->second.second.output());
374 
375  std::multimap<int, u32>::const_iterator initialArcsBegin = isContextDependent ? initialPhoneArcs.lower_bound(transitIt->second) : initialPhoneArcs.begin();
376  std::multimap<int, u32>::const_iterator initialArcsEnd = isContextDependent ? initialPhoneArcs.upper_bound(transitIt->second) : initialPhoneArcs.end();
377  for (std::multimap<int, u32>::const_iterator initialArcIt = initialArcsBegin;
378  initialArcIt != initialArcsEnd; ++initialArcIt) {
379  InArc nextHyp;
380  nextHyp.first = newState->id();
381  nextHyp.second = *oldState->getArc(initialArcIt->second);
382  inArcs.insert(std::make_pair(nextHyp.second.target(), nextHyp));
383  verify(nextHyp.second.target() > oldStateId);
384  }
385  }
386 
387  verify(inArcs.begin()->first == oldStateId);
388  inArcs.erase(inArcs.begin(), inArcs.upper_bound(oldStateId));
389  }
390 
391  s->setDescription("expand_transits(" + lat->describe() + ")");
392 
393  return ConstLatticeRef(s);
394 }

428 template<class MeshedBoundaryBuilder>
429 ConstLatticeRef buildMesh(const std::vector<MeshEntry>& entries, ConstSemiringRef semiring, const MeshedBoundaryBuilder& meshedBoundaryBuilder = MeshedBoundaryBuilder()) {
430  StaticBoundaries* b = new StaticBoundaries;
431  StaticLattice* s = new StaticLattice(Fsa::TypeAcceptor);
432  StaticLatticeRef staticLattice(s);
433  ConstLatticeRef firstLat = entries[0].lattice;
434  s->setProperties(firstLat->knownProperties(), firstLat->properties());
435  s->setInputAlphabet(firstLat->getInputAlphabet());
436  s->setSemiring(semiring);
437  s->setBoundaries(ConstBoundariesRef(b));
438 
439  ScoreId amScoreId = semiring->id("am");
440 
441  s->addProperties(Fsa::PropertySortedByInputAndTarget);
442 
443  typedef typename MeshedBoundaryBuilder::HashList BoundaryHashList;
444  BoundaryHashList meshSids;
445  for (u32 i = 0; i < entries.size(); ++i) {
446  s32 timeOffset = entries[i].timeOffset;
447  ConstLatticeRef l = entries[i].lattice;
448  ConstStateMapRef topologicalSort = sortTopologically(l);
449  verify(topologicalSort && topologicalSort->size());
450  if (entries[i].reverseOffset != Speech::InvalidTimeframeIndex) {
451  // backward lattice
452  Speech::TimeframeIndex reverseOffset = entries[i].reverseOffset;
453 
454  for (s32 j = ((s32)topologicalSort->size()) - 1; j >= 0; --j) {
455  Fsa::StateId sid = (*topologicalSort)[j];
456  ConstStateRef sr = l->getState(sid);
457  Boundary meshB = meshedBoundaryBuilder(l->boundary(sid));
458 
459  {
460  // reverse boundary
461  if (meshB.time() > reverseOffset) {
462  std::cout << "skipping state " << sid << " which can not be reversed. state time: " << meshB.time() << " reverse offset " << reverseOffset << std::endl;
463  continue;
464  }
465  meshB.setTime((reverseOffset - meshB.time()) + timeOffset);
466  Flf::Boundary::Transit transit = meshB.transit();
467  std::swap(transit.final, transit.initial);
468  meshB.setTransit(transit);
469  }
470 
471  std::pair<Fsa::StateId, bool> meshSource = meshSids.insertExisting(meshB);
472  if (!meshSource.second) {
473  State* meshSourceSp = new State(meshSource.first);
474  verify(!s->hasState(meshSourceSp->id()));
475  s->setState(meshSourceSp);
476  b->set(meshSourceSp->id(), meshB);
477  }
478 
479  State* meshSp = s->fastState(meshSource.first);
480 
481  for (State::const_iterator a = sr->begin(); a != sr->end(); ++a) {
482  if (a->input() == Fsa::Epsilon)
483  continue; // Skip non-acoustic arcs
484 
485  Boundary targetB = meshedBoundaryBuilder(l->boundary(a->target()));
486 
487  {
488  // reverse boundary
489  verify(targetB.time() <= reverseOffset);
490  targetB.setTime((reverseOffset - targetB.time()) + timeOffset);
491  Flf::Boundary::Transit transit = targetB.transit();
492  std::swap(transit.final, transit.initial);
493  targetB.setTransit(transit);
494  }
495 
496  Fsa::StateId targetSid = meshSids.find(targetB);
497  verify(targetSid != BoundaryHashList::InvalidCursor);
498 
499  if (meshSp->id() != targetSid) {
500  if ((meshB.time() <= targetB.time())) {
501  std::cout << "skipping backward arc" << std::endl;
502  Core::Application::us()->warning(
503  "Reversed lattice \"%s\" contains null/negative-length arcs; Skipping arc.",
504  l->describe().c_str())
505  << " " << targetB.time() << " -> " << meshB.time() << " (label=" << a->input() << ")";
506  continue;
507  }
508  verify(targetB.time() < meshB.time());
509  State* targetSp = s->fastState(targetSid);
510  Flf::Arc arc(meshSp->id(), a->weight(), a->input(), a->output());
511  State::iterator pos = targetSp->lower_bound(arc, Ftl::byInputAndTarget<Lattice>());
512  if ((pos == targetSp->end()) || (arc.target() != pos->target()) || (arc.input() != pos->input())) {
513  targetSp->insert(pos, arc);
514  }
515  else {
516  if (semiring->project(arc.weight()) < semiring->project(pos->weight()))
517  pos->setWeight(arc.weight());
518  }
519  }
520  }
521  }
522  }
523  else {
524  // forward lattice
525  Boundary initialB = meshedBoundaryBuilder(l->boundary(topologicalSort->front()));
526  initialB.setTime(initialB.time() + timeOffset);
527  std::pair<Fsa::StateId, bool> meshInitial = meshSids.insertExisting(initialB);
528  if (!meshInitial.second) {
529  State* meshInitialSp = new State(meshInitial.first);
530  verify(!s->hasState(meshInitialSp->id()));
531  s->setState(meshInitialSp);
532  b->set(meshInitialSp->id(), initialB);
533  }
534 
535  for (u32 j = 0; j < topologicalSort->size(); ++j) {
536  Fsa::StateId sid = (*topologicalSort)[j];
537 
538  ConstStateRef sr = l->getState(sid);
539  bool hasNonEpsilonArc = false;
540  for (State::const_iterator a = sr->begin(); a != sr->end(); ++a)
541  if (a->input() != Fsa::Epsilon)
542  hasNonEpsilonArc = true;
543 
544  if (!hasNonEpsilonArc)
545  continue;
546 
547  Boundary meshB = meshedBoundaryBuilder(l->boundary(sid));
548  meshB.setTime(meshB.time() + timeOffset);
549 
550  State* meshSp = 0;
551  std::pair<Fsa::StateId, bool> meshS = meshSids.insertExisting(meshB);
552  if (!meshS.second) {
553  meshSp = new State(meshS.first);
554  verify(!s->hasState(meshSp->id()));
555  s->setState(meshSp);
556  b->set(meshSp->id(), meshB);
557  }
558  else {
559  meshSp = s->fastState(meshS.first);
560  }
561 
562  for (State::const_iterator a = sr->begin(); a != sr->end(); ++a) {
563  if (a->input() == Fsa::Epsilon)
564  continue; // Skip non-acoustic arcs
565  Boundary targetB = meshedBoundaryBuilder(l->boundary(a->target()));
566  targetB.setTime(targetB.time() + timeOffset);
567  std::pair<Fsa::StateId, bool> meshTarget = meshSids.insertExisting(targetB);
568  if (!meshTarget.second) {
569  State* meshTargetSp = new State(meshTarget.first);
570  verify(!s->hasState(meshTargetSp->id()));
571  s->setState(meshTargetSp);
572  b->set(meshTargetSp->id(), targetB);
573  }
574 
575  if (meshSp->id() != meshTarget.first) {
576  if ((meshB.time() >= targetB.time())) {
577  std::cout << "skipping forward arc" << std::endl;
578  Core::Application::us()->warning(
579  "Lattice \"%s\" contains null/negative-length arcs; skipping arc.",
