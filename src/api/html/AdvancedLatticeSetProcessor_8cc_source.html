88 void MultiplyLatticeProcessorNode::processWordLattice(Lattice::ConstWordLatticeRef lattice, Bliss::SpeechSegment* s) {
89  Lattice::ConstWordLatticeRef result = lattice;
90  if (factors_.empty()) {
91  if (f32(factor_) != 1) {
92  result = Lattice::multiply(lattice, factor_);
93  }
94  }
95  else {
96  if (factors_.size() != lattice->nParts()) {
97  criticalError("mismatch in number of factors and number of lattice parts");
98  }
99  result = Lattice::multiply(lattice, factors_);
100  }
101  Precursor::processWordLattice(result, s);
102 }

159 void LemmaPronunciationToEvaluationToken::processWordLattice(Lattice::ConstWordLatticeRef lattice, Bliss::SpeechSegment* s) {
160  if (lattice->mainPart()->getInputAlphabet() != lexicon_->lemmaPronunciationAlphabet()) {
161  criticalError("Input alphabet must be the lemma pronuncation alphabet.");
162  }
163 
164  Fsa::ConstAutomatonRef evalFsa = Fsa::cache(
165  Fsa::projectOutput(
166  Fsa::composeMatching(
167  Fsa::composeMatching(lattice->mainPart(),
168  Fsa::multiply(lemmaPronToLemma_, Fsa::Weight(0.0))),
169  Fsa::multiply(lemmaToEval_, Fsa::Weight(0.0)))));
170  Core::Ref<Lattice::WordLattice> eval(new Lattice::WordLattice);
171  eval->setFsa(evalFsa, lattice->mainName());

173  Precursor::processWordLattice(eval, s);
174 }
