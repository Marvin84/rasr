25 void scoresToProbabilities(PosteriorCn& cn, bool normalize = false) {
26  for (u32 i = 0; i < cn.size(); ++i) {
27  PosteriorCn::Slot& pdf(cn[i]);
28  Probability sum = 0.0;
29  for (PosteriorCn::Slot::iterator itPdf = pdf.begin(); itPdf != pdf.end(); ++itPdf)
30  sum += (itPdf->score = ::exp(-itPdf->score));
31  verify(sum > 0.0);
32  if (!normalize)
33  if ((sum <= 0.99) || (1.01 <= sum))
34  Core::Application::us()->warning(
35  "Numerical instability: sum over posterior CN slot exceeds [0.99, 1.01]; "
36  "sum is %f.",
37  sum);
38  // Normalize anyway to deal with slight numerical instabilities, e.g. rounding errors
39  for (PosteriorCn::Slot::iterator itPdf = pdf.begin(); itPdf != pdf.end(); ++itPdf)
40  itPdf->score /= sum;
41  }
42 }

47 /**
48  * Calculate time frame wise word posterior distribution from
49  * arc confidence score, i.e. p(w|t) =
50  * (sum_{e \in E: begin(e) <= t < end(e) and word(e) = w} conf(e))
51  * / (sum_{e \in E: begin(e) <= t < end(e)} conf(e))
52  *
53  * The result is a list of posterior distributions p(*|t) for t = 1,...,T
54  **/
55 template<class ArcScores>
56 class FramewiseCollector : public TraverseState {
57 private:
58  ArcScores arcScores_;
59  PosteriorCn& cn_;
60  Time begin_, end_;
61 
62 protected:
63  virtual void exploreState(ConstStateRef sr) {
64  typename ArcScores::const_iterator itScore = arcScores_(sr->id()).first;
65  for (State::const_iterator a = sr->begin(), end_a = sr->end();
66  a != end_a; ++a, ++itScore) {
67  if (a->input() == Fsa::InvalidLabelId)
68  Core::Application::us()->criticalError("Frame-wise score collector: label id is invalid.");
69  PosteriorCn::Arc cnArc(a->input(), itScore->score());
70  Time t = l->boundary(sr->id()).time(), end_t = l->boundary(a->target()).time();
71  if ((t == InvalidTime) || (end_t == InvalidTime) || (t > end_t) || (end_t >= 2147483647))
72  Core::Application::us()->criticalError("Frame-wise score collector: interval [%d,%d] is invalid. Label id: %d", t, end_t, a->input());
73 
74  if (t < end_t) {
75  begin_ = std::min(begin_, t);
76  end_ = std::max(end_, end_t);
77  cn_.grow(end_t - 1);
78  for (; t < end_t; ++t) {
79  PosteriorCn::Slot& pdf = cn_[t];
80  PosteriorCn::Slot::iterator pdfIt = std::lower_bound(
81  pdf.begin(), pdf.end(), cnArc);
82  if ((pdfIt == pdf.end()) || (pdfIt->label != cnArc.label)) {
83  pdf.insert(pdfIt, cnArc);
84  }
85  else
86  pdfIt->score = ArcScores::collect(pdfIt->score, cnArc.score);
87  }
88  }
89  }
90  }
91 
92 public:
93  FramewiseCollector(
94  PosteriorCn& cn,
95  ConstLatticeRef l,
96  const ArcScores& arcScores)
97  : TraverseState(l), arcScores_(arcScores), cn_(cn), begin_(Core::Type<Time>::max), end_(0) {
98  if (!l->getBoundaries()->valid())
99  Core::Application::us()->criticalError("FramewiseCollector: Lattice \"%s\" has no time boundaries", l->describe().c_str());
100  traverse();
101  PosteriorCn::Slot epsSlot(1, PosteriorCn::Arc(Fsa::Epsilon, 0.0));
102  for (u32 i = 0; (i < cn_.size()) && cn_[i].empty(); ++i)
103  cn_[i] = epsSlot;
104  }
105  virtual ~FramewiseCollector() {}
106  Time begin() const {
107  return begin_;
108  }
109  Time end() const {
110  return end_;
111  }
112 };
