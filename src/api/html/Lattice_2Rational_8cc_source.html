22 ConstWordLatticeRef transpose(ConstWordLatticeRef l, bool progress, WordBoundary final) {
23  require(l->wordBoundaries());
24  WordLattice* result = 0;
25  for (size_t i = 0; i < l->nParts(); ++i) {
26  Fsa::ConstAutomatonRef transposed = Fsa::transpose(l->part(i), progress);
27  if (!result) {
28  if (!final.valid()) {
29  final = l->wordBoundaries()->getFinalWordBoundary();
30  }
31  Core::Ref<WordBoundaries> wordBoundaries(new WordBoundaries(*(l->wordBoundaries())));
32  wordBoundaries->set(transposed->initialStateId(), final);
33  result = new WordLattice();
34  result->setWordBoundaries(wordBoundaries);
35  }
36  result->setFsa(transposed, l->name(i));
37  }
38  return ConstWordLatticeRef(result);
39 }

48 ConstWordLatticeRef unite(const Core::Vector<ConstWordLatticeRef>& lattices) {
49  if (lattices.size() > 0) {
50  bool hasWordBoundaries = true;
51  const u32 nParts = lattices.front()->nParts();
52  Core::Vector<Core::Ref<const WordBoundaries>> wordBoundaries(lattices.size());
53  Core::Vector<Fsa::ConstMappingRef> morphisms(lattices.size());
54  Core::Ref<WordLattice> result(new WordLattice());
55  for (size_t i = 0; i < nParts; ++i) {
56  Core::Vector<Fsa::ConstAutomatonRef> tmp(lattices.size());
57  const std::string latticeName = lattices.front()->name(i);
58  for (u32 n = 0; n < lattices.size(); ++n) {
59  require(lattices[n]->name(i) == latticeName);
60  tmp[n] = lattices[n]->part(i);
61  }
62  result->setFsa(Fsa::unite(tmp), latticeName);
63  for (u32 n = 0; n < lattices.size(); ++n) {
64  if (lattices[n]->wordBoundaries()) {
65  wordBoundaries[n] = lattices[n]->wordBoundaries();
66  }
67  else {
68  hasWordBoundaries = false;
69  }
70  morphisms[n] = Fsa::mapToSubAutomaton(result->part(latticeName), n);
71  }
72  }
73  if (hasWordBoundaries) {
74  result->setWordBoundaries(Core::ref(new WordBoundaries()));
75  return resolveNaryMorphism(result, wordBoundaries, morphisms);
76  }
77  else {
78  return result;
79  }
80  }
81  return ConstWordLatticeRef();
82 }
