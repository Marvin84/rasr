86 int SpeechRecognizer::main(const std::vector<std::string>& arguments) {
87  switch (paramRecognitionMode(config)) {
88  case offlineRecognition:
89  case offlineConstrainedRecognition: {
90  Speech::CorpusProcessor* processor = 0;
91  switch (paramRecognitionMode(config)) {
92  case offlineRecognition:
93  processor = new Speech::OfflineRecognizer(config);
94  break;
95  case offlineConstrainedRecognition:
96  processor = new Speech::ConstrainedOfflineRecognizer(config);
97  break;
98  default: defect();
99  }
100  verify(processor);
101  Speech::CorpusVisitor corpusVisitor(config);
102  processor->signOn(corpusVisitor);
103  Bliss::CorpusDescription corpusDescription(select("corpus"));
104  corpusDescription.accept(&corpusVisitor);
105  delete processor;
106  } break;
107  case initOnlyRecognition: {
108  auto recognizer = Search::Module::instance().createRecognizer(static_cast<Search::SearchType>(Speech::Recognizer::paramSearch(config)), select("recognizer"));
109  auto modelCombinationRef = Speech::ModelCombinationRef(new Speech::ModelCombination(select("model-combination"), recognizer->modelCombinationNeeded(), Am::AcousticModel::noEmissions));
110  modelCombinationRef->load();
111  recognizer->setModelCombination(*modelCombinationRef);
112  recognizer->init();
113  delete recognizer;
114  } break;
115  default: defect();
116  }
117  return 0;
118 }
