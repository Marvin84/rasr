84  State* createState(Fsa::StateId s) const {
85  State* state = new State(s);
86  if (fst_->Final(s) != FstAutomaton::Weight::Zero()) {
87  state->setFinal(weightConverter_(fst_->Final(s)));
88  }
89  for (FstLib::ArcIterator<FstAutomaton> a(*fst_, s); !a.Done(); a.Next()) {
90  const typename FstAutomaton::Arc& arc = a.Value();

92  }

102  }
103  type_ = (isAcceptor ? Fsa::TypeAcceptor : Fsa::TypeTransducer);
