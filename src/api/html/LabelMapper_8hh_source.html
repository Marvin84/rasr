24 /**
25  * Transforms an acceptor to a transducer by mapping input labels
26  * which represent word (or lemma pronunciation) labels to output labels.
27  * The pseudo input labels l to be transformed are assumed to have
28  * wordLabelOffset <= l < wordLabelOffset + disambiguatorOffset
29  * The output labels are created as (l - wordLabelOffset).
30  */
31 template<class A>
32 class RestoreOutputLabelMapper {
33  typedef A Arc;
34 
35 public:
36  RestoreOutputLabelMapper(int wordLabelOffset, int disambiguatorOffset)
37  : wordLabelOffset_(wordLabelOffset), disambiguatorStart_(disambiguatorOffset) {}
38 
39  Arc operator()(const Arc& arc) const {
40  // transform only acceptors
41  verify(arc.ilabel == arc.olabel);
42  Arc newArc = arc;
43  if (arc.ilabel >= wordLabelOffset_ && arc.ilabel < disambiguatorStart_) {
44  newArc.ilabel = OpenFst::Epsilon;
45  newArc.olabel -= wordLabelOffset_;
46  }
47  else {
48  newArc.olabel = OpenFst::Epsilon;
49  }
50  return newArc;
51  }
52  FstLib::MapFinalAction FinalAction() const {
53  return FstLib::MAP_NO_SUPERFINAL;
54  }
55  FstLib::MapSymbolsAction InputSymbolsAction() const {
56  return FstLib::MAP_COPY_SYMBOLS;
57  }
58  FstLib::MapSymbolsAction OutputSymbolsAction() const {
59  return FstLib::MAP_CLEAR_SYMBOLS;
60  }
61  uint64 Properties(u64 props) const {
62  return props & FstLib::kILabelInvariantProperties & FstLib::kOLabelInvariantProperties;
63  }
64 
65 private:
66  int wordLabelOffset_;
67  int disambiguatorStart_;
68 };

72  * Relabels all arcs with
73  * disambiguatorMin <= ilabel <= disambiguatorMax.

108 template<class A>
109 void pushOutputLabels(FstLib::MutableFst<A>* f) {
110  typedef typename A::StateId StateId;
111  std::stack<StateId> state_queue;
112  std::vector<bool> visited;
113  StateId initial = f->Start();
114  visited.resize(initial + 1, false);
115  state_queue.push(initial);
116  std::vector<A> newArcs;
117  while (!state_queue.empty()) {
118  StateId s = state_queue.top();
119  state_queue.pop();
120  while (visited.size() <= s)
121  visited.push_back(false);
122  if (visited[s])
123  continue;
124  visited[s] = true;
125  typedef FstLib::ArcIterator<FstLib::Fst<A>> ConstArcIter;
126  newArcs.clear();
127  bool changedArcs = false;
128  for (ConstArcIter aiter(*f, s); !aiter.Done(); aiter.Next()) {
129  const A& arc = aiter.Value();
130  if (arc.olabel != OpenFst::Epsilon) {
131  verify(arc.ilabel == OpenFst::Epsilon);
132  for (ConstArcIter niter(*f, arc.nextstate); !niter.Done(); niter.Next()) {
133  const A& nextArc = niter.Value();
134  A newArc = nextArc;
135  newArc.weight = FstLib::Times(arc.weight, nextArc.weight);
136  verify(newArc.olabel == OpenFst::Epsilon);
137  newArc.olabel = arc.olabel;
138  newArcs.push_back(newArc);
139  changedArcs = true;
140  if (newArc.nextstate >= visited.size() || !visited[newArc.nextstate])
141  state_queue.push(newArc.nextstate);
142  }
143  }
144  else {
145  if (arc.nextstate >= visited.size() || !visited[arc.nextstate]) {
146  state_queue.push(arc.nextstate);
147  }
148  newArcs.push_back(arc);
149  }
150  }
151  if (changedArcs) {
152  f->DeleteArcs(s);
153  for (typename std::vector<A>::const_iterator a = newArcs.begin();
154  a != newArcs.end(); ++a) {
155  f->AddArc(s, *a);
156  }
157  }
158  }
159  FstLib::Connect(f);
160 }
