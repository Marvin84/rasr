24 /**
25  * Vector concatenation filter.
26  * All input packets must be vectors, which will be concatenated
27  * into a single vector.
28  * Inputs: many, dynamically generated.
29  * Outputs: one
30  * All inputs consume the same nummber of packets. The dimension of
31  * the output vector will be the sum of the dimensions of all input
32  * vectors. The time stamp of the output vector is set so that it
33  * contains the time ranges of all input vectors.
34  */
35 
36 template<typename T>
37 class VectorConcatNode : public MergerNode<Vector<T>, Vector<T>> {
38  typedef MergerNode<Vector<T>, Vector<T>> Precursor;
39 
40 private:
41  u32 output_size_;
42 
43 public:
44  static std::string filterName() {
45  return std::string("generic-vector-") + Core::NameHelper<T>() + "-concat";
46  }
47  VectorConcatNode(const Core::Configuration& c)
48  : Core::Component(c), Precursor(c), output_size_(0) {}
49  virtual ~VectorConcatNode() {}
50 
51  virtual bool configure() {
52  output_size_ = 0;
53  return Precursor::configure();
54  }
55 
56  virtual Vector<T>* merge(std::vector<DataPtr<Vector<T>>>& inputData) {
57  Flow::Vector<T>* out = new Flow::Vector<T>;
58  if (output_size_ > 0)
59  out->reserve(output_size_);
60 
61  for (u32 i = 0; i < inputData.size(); i++) {
62  Flow::DataPtr<Flow::Vector<T>>& in(inputData[i]);
63  out->insert(out->end(), in->begin(), in->end());
64  }
65  if (out->size() > output_size_)
66  output_size_ = out->size();
67  return out;
68  }
69 };
