56  void add_cache(ScoresWithContext* cache) {
57  std::vector<ScoresWithContext*> request_chain;
58  request_chain.push_back(cache);
59  ScoresWithContext* parent = const_cast<ScoresWithContext*>(reinterpret_cast<ScoresWithContext const*>(cache->parent.handle()));
60  request_chain.push_back(parent);
61  while (parent->state.empty()) {
62  parent = const_cast<ScoresWithContext*>(reinterpret_cast<ScoresWithContext const*>(parent->parent.handle()));
63  request_chain.push_back(parent);
64  }
65 
66  std::vector<size_t>* child_idxs = &roots;
67  while (not request_chain.empty()) {
68  // find root node
69  size_t child_idx = child_idxs->size();
70  for (size_t c = 0ul; c < child_idxs->size(); c++) {
71  if (entries[child_idxs->at(c)] == request_chain.back()) {
72  child_idx = c;
73  break;
74  }
75  }
76  size_t next_child_idx = 0ul;
77  if (child_idx == child_idxs->size()) {
78  child_idxs->push_back(entries.size());
79  entries.push_back(request_chain.back());
80  next_child_idx = child_idxs->at(child_idx);
81  children.emplace_back(); // can invalidate child_idxs
82  }
83  else {
84  next_child_idx = child_idxs->at(child_idx);
85  }
86  child_idxs = &children[next_child_idx];
87  request_chain.pop_back();
88  }
89  }
90 
91  void get_requests_dfs(std::vector<FwdRequest>& requests, ScoresWithContext* initial, size_t entry, size_t length) {
92  if (children[entry].empty()) {
93  requests.emplace_back(FwdRequest{initial, entries[entry], length});
94  }
95  else {
96  for (size_t e : children[entry]) {
97  get_requests_dfs(requests, initial, e, length + 1ul);
98  }
99  }
100  }
101 
102  std::vector<FwdRequest> get_requests() {
103  std::vector<FwdRequest> result;
104  for (size_t r : roots) {
105  for (size_t c : children[r]) {
106  get_requests_dfs(result, entries[r], c, 1ul);
107  }
108  }
109  return result;
110  }

113 void dump_scores(ScoresWithContext const& cache, std::string const& prefix) {
114  std::stringstream path;
115  path << prefix;
116  for (auto token : *cache.history) {
117  path << "_" << token;

119  std::ofstream out(path.str(), std::ios::out | std::ios::trunc);
120  out << "nn_output:\n";
121  std::vector<float> nn_output(cache.nn_output->size());
122  cache.nn_output->uncompress(nn_output.data(), nn_output.size());
123  for (auto nn_out : nn_output) {
124  out << nn_out << '\n';
125  }
126  for (size_t s = 0ul; s < cache.state.size(); s++) {
127  out << "state " << s << ":\n";
128  std::vector<float> state_data(cache.state[s]->size());
129  cache.state[s]->uncompress(state_data.data(), state_data.size());
130  for (auto v : state_data) {
131  out << v << '\n';
132  }
133  }

445 History TFRecurrentLanguageModel::reducedHistory(History const& hist, u32 limit) const {
446  ScoresWithContext const* sc = reinterpret_cast<ScoresWithContext const*>(hist.handle());
447  if (not allow_reduced_history_ or sc->history->size() <= limit) {
448  return hist;
449  }
450  History h = startHistory();

452  h = extendHistoryWithOutputIdx(h, sc->history->at(sc->history->size() - w));

457 Score TFRecurrentLanguageModel::score(History const& hist, Token w) const {
458  ScoresWithContext* sc = const_cast<ScoresWithContext*>(reinterpret_cast<ScoresWithContext const*>(hist.handle()));
459 
460  if (not sc->computed.load()) {
461  auto start = std::chrono::steady_clock::now();
462  if (async_) {
463  // promise should only be used once
464  to_fwd_finished_ = std::promise<History const*>();
465  std::future<History const*> future = to_fwd_finished_.get_future();
466  to_fwd_.store(&hist);
467  future.wait();
468  }
469  else {
470  forward<false>(&hist);
471  }
472  auto end = std::chrono::steady_clock::now();
473  double wait_time = std::chrono::duration<double, std::milli>(end - start).count();
474  total_wait_time_ += wait_time;
475  if (verbose_) {
476  std::cerr << "wait: " << wait_time << " " << sc->info.numStates << " " << sc->info.bestScoreOffset << std::endl;
477  }
478  }
479 
480  require(sc->computed.load());

595 template<bool async>
596 void TFRecurrentLanguageModel::forward(Lm::History const* hist) const {
597  ScoresWithContext* sc = nullptr;
598  if (hist != nullptr) {
599  sc = const_cast<ScoresWithContext*>(reinterpret_cast<ScoresWithContext const*>(hist->handle()));
600  }
601  if (async and sc != nullptr and sc->computed.load()) { // nothing to do (only happens in async case)
602  to_fwd_finished_.set_value(hist);
603  return;
604  }
605  auto start = std::chrono::steady_clock::now();
606 
607  RequestGraph request_graph;
608  if (sc != nullptr) {
609  request_graph.add_cache(const_cast<ScoresWithContext*>(sc));
610  }
611 
612  std::vector<FwdRequest> requests;
613  std::vector<Lm::History> request_histories; // make sure none of the request caches go away while we compute the scores
614  size_t max_length = 0ul;
615 
616  size_t num_pending_requests = pending_.size();
617  std::unordered_set<void const*> handles; // only relevant in async case
618  handles.reserve(pending_.size());
619  std::vector<ScoresWithContext*> early_requests;
620  std::vector<History const*> early_request_histories; // make sure none of the request caches go away while we compute the scores (only relevant in async case)
621 
622  if (async) {
623  auto process_hist = [&](History const* hist) {
624  ScoresWithContext* c = const_cast<ScoresWithContext*>(reinterpret_cast<ScoresWithContext const*>(hist->handle()));
625  ScoresWithContext* parent_c = const_cast<ScoresWithContext*>(reinterpret_cast<ScoresWithContext const*>(c->parent.handle()));
626  if (handles.find(hist->handle()) == handles.end() and not c->computed.load() and c != sc and c->parent.handle() != nullptr and c->ref_count > 1 and (not single_step_only_ or parent_c->computed.load())) {
627  early_requests.emplace_back(c);
628  early_request_histories.emplace_back(hist);
629  handles.insert(hist->handle());
630  }
631  else {
632  finished_queue_.enqueue(hist);
633  }
634  };
635 
636  std::for_each(pending_.begin(), pending_.end(), process_hist);
637  pending_.clear();
638 
639  History const* hist_buf = nullptr;
640  bool success = false;
641  bool first = true;
642  do {
643  if (first) {
644  success = fwd_queue_.wait_dequeue_timed(hist_buf, 1000);
645  }
646  else {
647  success = fwd_queue_.try_dequeue(hist_buf);
648  }
649  if (success) {
650  process_hist(hist_buf);
651  first = false;
652  }
653  } while (success);
654  }
655  else {
656  NNHistoryManager* hm = dynamic_cast<NNHistoryManager*>(historyManager_);
657  hm->visit([&](HistoryHandle h) {
658  ScoresWithContext* c = const_cast<ScoresWithContext*>(reinterpret_cast<ScoresWithContext const*>(h));
659  if (not c->computed.load() and c != sc and not(c->was_expanded and c->info.numStates == 0)) {
660  early_requests.emplace_back(c);
661  }
662  });
663  }
664 
665  size_t num_early_requests = early_requests.size();
666 
667  auto end_early_requests = std::chrono::steady_clock::now();
668 
669  if (async and sc == nullptr and early_requests.empty()) {
670  // can only happen in async case
671  return;
672  }
673 
674  // because the scores can be updated while we are sorting we need to store them, so we get a consistent view
675  std::vector<std::pair<unsigned, Score>> idxs;
676  idxs.reserve(early_requests.size());
677  for (size_t i = 0ul; i < early_requests.size(); i++) {
678  idxs.emplace_back(i, early_requests[i]->info.minLabelDistance * 1000 + early_requests[i]->info.bestScoreOffset);
679  }
680  std::sort(idxs.begin(), idxs.end(), [](std::pair<size_t, Score> const& a, std::pair<size_t, Score> const& b) {
681  return a.second < b.second;
682  });
683 
684  for (auto idx : idxs) {
685  request_graph.add_cache(early_requests[idx.first]);
686  }
687 
688  // we do not need early_requests anymore
689  early_requests.clear();
690  idxs.clear();
691 
692  requests = request_graph.get_requests();
693 
694  // prune requests
695  if (min_batch_size_ > 0ul and requests.size() > min_batch_size_) {
696  size_t i = min_batch_size_;
697  Score ref_score = requests.front().final_cache->info.bestScoreOffset + batch_pruning_threshold_;
698  if (not Math::isinf(ref_score)) {
699  while ((i + 1) < requests.size() and requests[i + 1].final_cache->info.bestScoreOffset <= ref_score) {
700  i += 1ul;
701  }
702  requests.resize(i);
703  }
704  }
705 
706  if (min_batch_size_ > 0ul and opt_batch_size_ > 0ul and requests.size() > opt_batch_size_ + min_batch_size_) {
707  requests.resize(opt_batch_size_);
708  }
709  if (max_batch_size_ > 0ul and requests.size() > max_batch_size_) {
710  requests.resize(max_batch_size_);
711  }
712 
713  Score worst_score = std::numeric_limits<Score>::min();
714  for (auto const& r : requests) {
715  max_length = std::max(max_length, r.length);
716  worst_score = std::max(worst_score, r.final_cache->info.bestScoreOffset);
717  }
718 
719  auto end_requests = std::chrono::steady_clock::now();
720 
721  // prepare the data in Sprint Datastructures
722  Math::FastMatrix<s32> words(requests.size(), max_length);
723  Math::FastVector<s32> word_lengths(requests.size());
724  for (size_t r = 0ul; r < requests.size(); r++) {
725  auto& history = *(requests[r].final_cache->history);
726  size_t offset = history.size() - requests[r].length;
727  for (size_t w = 0u; w < requests[r].length; w++) {
728  words.at(r, w) = static_cast<s32>(history[offset + w]);
729  }
730  for (size_t w = requests[r].length; w < max_length; w++) {
731  words.at(r, w) = 0;
732  }
733  word_lengths[r] = requests[r].length;
734  ScoresWithContext* initial_cache = requests[r].initial_cache;
735  require(initial_cache != nullptr);
736  require_eq(state_variables_.size(), initial_cache->state.size());
737  }
738 
739  bool full_prefix_required = state_manager_->requiresAllParentStates();
740  size_t total_prefix_length = 0ul;
741  size_t total_suffix_length = 0ul;
742 
743  std::vector<size_t> prefix_lengths(requests.size());
744  std::vector<size_t> suffix_lengths(requests.size());
745  for (size_t r = 0ul; r < requests.size(); r++) {
746  prefix_lengths[r] = requests[r].initial_cache->history->size();
747  suffix_lengths[r] = requests[r].length;
748  total_prefix_length += prefix_lengths[r];
749  total_suffix_length += suffix_lengths[r];
750  }
751 
752  std::vector<StateManager::HistoryState const*> prefix_states(full_prefix_required ? total_prefix_length : requests.size());
753  size_t current_offset = 0ul;
754  for (size_t r = 0ul; r < requests.size(); r++) {
755  ScoresWithContext* current_cache = requests[r].initial_cache;
756  if (full_prefix_required) {
757  size_t prefix_length = prefix_lengths[r];
758  for (size_t i = 0ul; i < prefix_length; i++) {
759  prefix_states[current_offset + prefix_length - i - 1] = &current_cache->state;
760  current_cache = const_cast<ScoresWithContext*>(reinterpret_cast<ScoresWithContext const*>(current_cache->parent.handle()));
761  }
762  current_offset += prefix_length;
763  }
764  else {
765  prefix_states[r] = &current_cache->state;
766  }
767  }
768 
769  auto end_prepare = std::chrono::steady_clock::now();
770 
771  // build tensors + set state variables
772  std::vector<std::pair<std::string, Tensorflow::Tensor>> inputs;
773  std::vector<std::string> targets;
774  state_manager_->mergeStates(state_variables_, prefix_lengths, prefix_states, inputs, targets);
775  std::vector<s32> state_lengths(prefix_lengths.begin(), prefix_lengths.end());
776 
777  if (dump_inputs_) {
778  std::string out = dump_inputs_prefix_ + "_" + std::to_string(dump_inputs_counter_) + "_state_";
779  for (size_t i = 0ul; i < inputs.size(); i++) {
780  inputs[i].second.save<s16>(out + std::to_string(i));
781  }
782  }
783 
784  auto end_merge_state = std::chrono::steady_clock::now();
785 
786  session_.run(inputs, targets);
787 
788  auto end_set_state = std::chrono::steady_clock::now();
789 
790  // run nn-output calculation
791  inputs.clear();
792  auto const& word_info = tensor_input_map_.get_info("word");
793  inputs.emplace_back(std::make_pair(word_info.tensor_name(), Tensorflow::Tensor::create(words)));
794  if (not word_info.seq_length_tensor_name().empty()) {
795  inputs.emplace_back(std::make_pair(word_info.seq_length_tensor_name(), Tensorflow::Tensor::create(word_lengths)));
796  }
797  if (tensor_input_map_.has_info("state-lengths")) {
798  auto const& state_lengths_info = tensor_input_map_.get_info("state-lengths");
799  inputs.emplace_back(std::make_pair(state_lengths_info.tensor_name(), Tensorflow::Tensor::create(state_lengths)));
800  }
801  std::vector<Tensorflow::Tensor> outputs;
802  session_.run(inputs, output_tensor_names_, graph_->update_ops(), outputs);
803 
804  if (dump_inputs_) {
805  std::string out = dump_inputs_prefix_ + "_" + std::to_string(dump_inputs_counter_) + "_nn_in_";
806  for (size_t i = 0ul; i < inputs.size(); i++) {
807  inputs[i].second.save<s32>(out + std::to_string(i));
808  }
809  out = dump_inputs_prefix_ + "_" + std::to_string(dump_inputs_counter_) + "_nn_out_";
810  for (size_t i = 0ul; i < outputs.size(); i++) {
811  outputs[i].save<f32>(out + std::to_string(i));
812  }
813  dump_inputs_counter_ += 1ul;
814  }
815 
816  auto end_nn_output = std::chrono::steady_clock::now();
817 
818  // store outputs in caches
819  for (size_t r = 0ul; r < requests.size(); r++) {
820  ScoresWithContext* cache = requests[r].final_cache;
821  for (size_t w = requests[r].length; w > 0;) {
822  --w;
823  cache->last_used = current_time_;
824  int num_outputs = outputs[0ul].dimSize(2);
825  auto compression_param_estimator = nn_output_comp_vec_factory_->getEstimator();
826  float const* data = outputs[0ul].data<f32>(r, w, 0);
827  compression_param_estimator->accumulate(data, num_outputs);
828  auto compression_params = compression_param_estimator->estimate();
829  cache->nn_output = nn_output_comp_vec_factory_->compress(data, num_outputs, compression_params.get());
830  cache->computed.store(true);
831  cache = const_cast<ScoresWithContext*>(reinterpret_cast<ScoresWithContext const*>(cache->parent.handle()));
832  }
833  require_eq(cache, requests[r].initial_cache);
834  }
835 
836  auto end_set_nn_output = std::chrono::steady_clock::now();
837 
838  // fetch new values of state variables, needs to be done in separate Session::run call (for GPU devices)
839  session_.run({}, read_vars_tensor_names_, {}, outputs);
840 
841  auto end_get_new_state = std::chrono::steady_clock::now();
842 
843  auto split_states = state_manager_->splitStates(state_variables_, suffix_lengths, outputs, *state_comp_vec_factory_);
844 
845  size_t output_offset = 0ul;
846  for (size_t r = 0ul; r < requests.size(); r++) {
847  ScoresWithContext* current_cache = requests[r].final_cache;
848  size_t suffix_length = suffix_lengths[r];
849  while (suffix_length > 0ul) {
850  current_cache->state = std::move(split_states[output_offset + suffix_length - 1]);
851  current_cache = const_cast<ScoresWithContext*>(reinterpret_cast<ScoresWithContext const*>(current_cache->parent.handle()));
852  suffix_length -= 1ul;
853  }
854  output_offset += suffix_lengths[r];
855  }
856 
857  auto end_split_state = std::chrono::steady_clock::now();
858 
859  std::chrono::duration<double, std::milli> duration = end_split_state - end_prepare;
860  size_t bucket = requests.size() - 1;
861  run_time_.at(bucket) += duration.count();
862  run_count_.at(bucket) += 1ul;
863 
864  if (dump_scores_) {
865  for (auto const& r : requests) {
866  dump_scores(*r.final_cache, dump_scores_prefix_);
867  }
868  }
869 
870  if (async) {
871  for (auto hist : early_request_histories) {
872  ScoresWithContext* c = const_cast<ScoresWithContext*>(reinterpret_cast<ScoresWithContext const*>(hist->handle()));
873  if (c->computed.load() or c->ref_count == 1ul or c->info.numStates == 0) {
874  finished_queue_.enqueue(hist);
875  }
876  else {
877  pending_.push_back(hist);
878  }
879  }
880  if (sc != nullptr) {
881  to_fwd_finished_.set_value(hist);
882  }
883  }
884 
885  auto end = std::chrono::steady_clock::now();
886 
887  TimeStatistics stats;
888  stats.total_duration = std::chrono::duration<double, std::milli>(end - start);
889  stats.early_request_duration = std::chrono::duration<double, std::milli>(end_early_requests - start);
890  stats.request_duration = std::chrono::duration<double, std::milli>(end_requests - end_early_requests);
891  stats.prepare_duration = std::chrono::duration<double, std::milli>(end_prepare - end_requests);
892  stats.merge_state_duration = std::chrono::duration<double, std::milli>(end_merge_state - end_prepare);
893  stats.set_state_duration = std::chrono::duration<double, std::milli>(end_set_state - end_merge_state);
894  stats.run_nn_output_duration = std::chrono::duration<double, std::milli>(end_nn_output - end_set_state);
895  stats.set_nn_output_duration = std::chrono::duration<double, std::milli>(end_set_nn_output - end_nn_output);
896  stats.get_new_state_duration = std::chrono::duration<double, std::milli>(end_get_new_state - end_set_nn_output);
897  stats.split_state_duration = std::chrono::duration<double, std::milli>(end_split_state - end_get_new_state);
898  stats.softmax_output_duration = std::chrono::duration<double, std::milli>();
899  if (verbose_) {
900  stats.write(std::cerr);
901  std::cerr << " #pr:" << num_pending_requests
902  << " #er:" << num_early_requests
903  << " #r:" << requests.size() << std::endl;
904  }
905  fwd_statistics_ += stats;
906 }
