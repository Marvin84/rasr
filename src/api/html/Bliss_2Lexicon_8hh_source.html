385 template<class Object>
386 std::string Coarticulated<Object>::format(Core::Ref<const PhonemeInventory> phonemeInventory) const {
387  const std::string l(leftContext() == Phoneme::term ? "#" : phonemeInventory->phoneme(leftContext())->symbol().str());
388  const std::string r(rightContext() == Phoneme::term ? "#" : phonemeInventory->phoneme(rightContext())->symbol().str());
389  return Core::form("%s|%s|%s", l.c_str(), object().format(phonemeInventory).c_str(), r.c_str());
390 }

465 /**
466  * The Bliss lexicon.
467  *
468  * See the <a href="../../doc/Bliss.pdf">Bliss documentation</a> and
469  * <a href="../../doc/Lexicon.pdf">Lexicon Format Reference</a>
470  * for comporehensive explanation.
471  *
472  * An entry of the lexicon are called "lemma". (The term "word"
473  * is avoided because of its ambiguity.) A lemma has one or more
474  * orthographic forms, zero or more pronunciations (aka "acoustic
475  * words") and zero or more syntactic tokens (aka "language model
476  * words").
477  *
478  * A lemma may be assigned a symbolic name, which the system can
479  * use to identify lemmas which have a special meaning to it.
480  * E.g. the silence word is is identified by the symbolic name
481  * "silence". Such lemmas a called "special lemmas".
482  */
