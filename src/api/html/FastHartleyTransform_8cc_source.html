148 void FastHartleyTransform::hartleyTransform(std::vector<Data>& fz) const {
149  // *very* fast hartley transform
150  f32 *fi, *fn, *gi;
151  u32 i, k, k1, k2, k3, k4, kx;
152 #ifdef GOOD_TRIG
153  double coswrk[20], sinwrk[20];
154 #endif
155  TRIG_VARS;
156 
157  for (k1 = 1; k1 < length(); k1++) {
158  k2 = bitReverse_[k1];
159  if (k1 > k2) {
160  f32 a = fz[k1];
161  fz[k1] = fz[k2];
162  fz[k2] = a;
163  }
164  }
165 
166  for (k = 0; u32(1 << k) < length(); k++)
167  ;
168  k &= 1;
169 
170  if (k == 0) {
171  for (fi = &fz[0], fn = &fz[length()]; fi < fn; fi += 4) {
172  double f0, f1, f2, f3;
173  f1 = fi[0] - fi[1];
174  f0 = fi[0] + fi[1];
175  f3 = fi[2] - fi[3];
176  f2 = fi[2] + fi[3];
177  fi[2] = (f0 - f2);
178  fi[0] = (f0 + f2);
179  fi[3] = (f1 - f3);
180  fi[1] = (f1 + f3);
181  }
182  }
183  else {
184  for (fi = &fz[0], fn = &fz[length()], gi = fi + 1; fi < fn; fi += 8,
185  gi += 8) {
186  double s1, c1, s2, c2, s3, c3, s4, c4, g0, f0, f1, g1, f2, g2, f3, g3;
187  c1 = fi[0] - gi[0];
188  s1 = fi[0] + gi[0];
189  c2 = fi[2] - gi[2];
190  s2 = fi[2] + gi[2];
191  c3 = fi[4] - gi[4];
192  s3 = fi[4] + gi[4];
193  c4 = fi[6] - gi[6];
194  s4 = fi[6] + gi[6];
195  f1 = (s1 - s2);
196  f0 = (s1 + s2);
197  g1 = (c1 - c2);
198  g0 = (c1 + c2);
199  f3 = (s3 - s4);
200  f2 = (s3 + s4);
201  g3 = SQRT2 * c4;
202  g2 = SQRT2 * c3;
203  fi[4] = f0 - f2;
204  fi[0] = f0 + f2;
205  fi[6] = f1 - f3;
206  fi[2] = f1 + f3;
207  gi[4] = g0 - g2;
208  gi[0] = g0 + g2;
209  gi[6] = g1 - g3;
210  gi[2] = g1 + g3;
211  }
212  }
213  if (length() < 16)
214  return;
215 
216  do {
217  double s1, c1;
218  k += 2;
219  k1 = 1 << k;
220  k2 = k1 << 1;
221  k4 = k2 << 1;
222  k3 = k2 + k1;
223  kx = k1 >> 1;
224  fi = &fz[0];
225  gi = fi + kx;
226  fn = &fz[length()];
227  do {
228  double g0, f0, f1, g1, f2, g2, f3, g3;
229  f1 = fi[0] - fi[k1];
230  f0 = fi[0] + fi[k1];
231  f3 = fi[k2] - fi[k3];
232  f2 = fi[k2] + fi[k3];
233  fi[k2] = f0 - f2;
234  fi[0] = f0 + f2;
235  fi[k3] = f1 - f3;
236  fi[k1] = f1 + f3;
237  g1 = gi[0] - gi[k1];
238  g0 = gi[0] + gi[k1];
239  g3 = SQRT2 * gi[k3];
240  g2 = SQRT2 * gi[k2];
241  gi[k2] = g0 - g2;
242  gi[0] = g0 + g2;
243  gi[k3] = g1 - g3;
244  gi[k1] = g1 + g3;
245  gi += k4;
246  fi += k4;
247  } while (fi < fn);
248 
249  TRIG_INIT(k, c1, s1);
250  for (i = 1; i < kx; i++) {
251  double c2, s2;
252  TRIG_NEXT(k, c1, s1);
253  c2 = c1 * c1 - s1 * s1;
254  s2 = 2 * (c1 * s1);
255  fn = &fz[length()];
256  fi = &fz[i];
257  gi = &fz[k1 - i];
258  do {
259  double a, b, g0, f0, f1, g1, f2, g2, f3, g3;
260  b = s2 * fi[k3] - c2 * gi[k3];
261  a = c2 * fi[k3] + s2 * gi[k3];
262  f3 = fi[k2] - a;
263  f2 = fi[k2] + a;
264  g3 = gi[k2] - b;
265  g2 = gi[k2] + b;
266  b = s2 * fi[k1] - c2 * gi[k1];
267  a = c2 * fi[k1] + s2 * gi[k1];
268  f1 = fi[0] - a;
269  f0 = fi[0] + a;
270  g1 = gi[0] - b;
271  g0 = gi[0] + b;
272  b = s1 * f2 - c1 * g3;
273  a = c1 * f2 + s1 * g3;
274  fi[k2] = f0 - a;
275  fi[0] = f0 + a;
276  gi[k3] = g1 - b;
277  gi[k1] = g1 + b;
278  b = c1 * g2 - s1 * f3;
279  a = s1 * g2 + c1 * f3;
280  gi[k2] = g0 - a;
281  gi[0] = g0 + a;
282  fi[k3] = f1 - b;
283  fi[k1] = f1 + b;
284  gi += k4;
285  fi += k4;
286  } while (fi < fn);
287  }
288  } while (k4 < length());
289 }

317 void hartleyToFourier(const std::vector<f32>& hartley, std::vector<f32>& fourier) {
318  if (hartley.empty())
319  return;
320 
321  u32 N = hartley.size();
322 
323  fourier.resize((N / 2 + 1) * 2); // N even: N + 2; N odd: N + 1
324 
325  fourier[0] = hartley[0];
326  fourier[1] = 0;
327 
328  for (u32 n = 1; n <= N / 2; ++n) {
329  fourier[2 * n] = (hartley[n] + hartley[N - n]) / 2;
330  fourier[2 * n + 1] = (hartley[n] - hartley[N - n]) / 2;
331  }
332 }

334 void hartleyToFourierAmplitude(const std::vector<f32>& hartley, std::vector<f32>& amplitude) {
335  if (hartley.empty())
336  return;
337 
338  u32 N = hartley.size();
339 
340  if (amplitude.size() < N / 2 + 1)
341  amplitude.resize(N / 2 + 1);
342 
343  amplitude[0] = Core::abs(hartley[0]);
344 
345  f32 real2, imaginary2;
346 
347  for (u32 n = 1; n <= N / 2; ++n) {
348  real2 = hartley[n] + hartley[N - n];
349  imaginary2 = hartley[n] - hartley[N - n];
350 
351  amplitude[n] = sqrt(real2 * real2 + imaginary2 * imaginary2) / 2;
352  }
353 
354  amplitude.resize(N / 2 + 1);
355 }
