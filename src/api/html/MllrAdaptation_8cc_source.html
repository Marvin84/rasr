59 void ShiftAdaptor::adaptMixtureSet(Core::Ref<Mm::MixtureSet> adaptableMixtureSet) const {
60  // Adapt references
61  log("adapting references, using ") << adaptationMatrices_.size() << " shifts.";
62  for (MixtureIndex mix = 0; mix < adaptableMixtureSet->nMixtures(); ++mix) {
63  MatrixConstIterator w = adaptationMatrices_.find(adaptationMatrixIndex_[mix]);
64  ensure(w != adaptationMatrices_.end());
65  for (size_t dns = 0; dns < adaptableMixtureSet->mixture(mix)->nDensities(); ++dns) {
66  // Extract the mean vector
67  DensityIndex densityIndex = adaptableMixtureSet->mixture(mix)->densityIndex(dns);
68  MeanIndex meanIndex = adaptableMixtureSet->density(densityIndex)->meanIndex();
69  Mm::Mean mean = *(adaptableMixtureSet->mean(meanIndex));
70 
71  // Shift
72  Math::Vector<Mm::MeanType> adaptedMean(w->second[0] + Math::Vector<Mm::MeanType>(mean));
73 
74  // Update MixtureSet
75  // Should be efficient, as only the buffer pointers are swapped
76  adaptableMixtureSet->mean(meanIndex)->swap(adaptedMean);
77  }
78  }
79 }

81 void ShiftAdaptor::adaptMixtureSet(Core::Ref<Mm::MixtureSet> adaptableMixtureSet, Core::Ref<Mm::ShiftAdaptor> preAdaptor) const {
82  if (adaptationMatrices_.size() > 0 || preAdaptor->adaptationMatrices().size() > 0) {
83  // Adapt references
84  log("adapting references, using ") << adaptationMatrices_.size() << " shifts.";
85  for (MixtureIndex mix = 0; mix < adaptableMixtureSet->nMixtures(); ++mix) {
86  Math::Vector<double> shift;
87  Math::Vector<double> preShift;
88  if (adaptationMatrices_.size() > 0)
89  shift = const_cast<ShiftAdaptor&>(*this).shiftVector(mix);
90  if (preAdaptor->adaptationMatrices().size() > 0)
91  preShift = preAdaptor->shiftVector(mix);
92  for (size_t dns = 0; dns < adaptableMixtureSet->mixture(mix)->nDensities(); ++dns) {
93  // Extract the mean vector
94  DensityIndex densityIndex = adaptableMixtureSet->mixture(mix)->densityIndex(dns);
95  MeanIndex meanIndex = adaptableMixtureSet->density(densityIndex)->meanIndex();
96  Mm::Mean mean = *(adaptableMixtureSet->mean(meanIndex));
97 
98  // Shift
99  Math::Vector<Mm::MeanType> adaptedMean;
100  if (preShift.size() == 0)
101  adaptedMean = Math::Vector<Mm::MeanType>(mean);
102  else
103  adaptedMean = Math::Vector<Mm::MeanType>(mean) - preShift;
104  if (shift.size() > 0)
105  adaptedMean += shift;
106 
107  // Update MixtureSet
108  // Should be efficient, as only the buffer pointers are swapped
109  adaptableMixtureSet->mean(meanIndex)->swap(adaptedMean);
110  }
111  }
112  }

320 void ShiftAdaptorViterbiEstimator::init() {
321  countAccumulators_.resize(nLeafs_, 0); //number of leafs in MLLR tree
322  betaAccumulators_.resize(nLeafs_, Math::Vector<Sum>(dimension_));
323  shiftAccumulators_.resize(nLeafs_, Math::Vector<Sum>(dimension_));
324 
325  std::stack<NodeId> stack;
326  stack.push(tree_->root());
327  while (!stack.empty()) {
328  Core::BinaryTree::Id id = stack.top();
329  stack.pop();
330  shift_[id] = Matrix();
331  if (tree_->right(id) != tree_->invalidId) {
332  stack.push(tree_->right(id));
333  }
334  if (tree_->left(id) != tree_->invalidId) {
335  stack.push(tree_->left(id));
336  }
337  }
338 
339  for (MatrixIterator p = shift_.begin(); p != shift_.end(); ++p) {
340  p->second = Math::Matrix<Sum>();
341  p->second.addRow(Math::Vector<Sum>(dimension_));
342  }
343 }

358 void ShiftAdaptorViterbiEstimator::estimateWMatrices() {
359  Math::Vector<Sum> count;
360  Math::Vector<Math::Vector<Sum>> beta;
361  Math::Vector<Math::Vector<Sum>> shift;
362 
363  propagate(countAccumulators_, count, tree_->root());
364  propagate(betaAccumulators_, beta, tree_->root());
365  propagate(shiftAccumulators_, shift, tree_->root());
366 
367  ensure(beta.size() == shift.size());
368  count_.resize(count.size());
369  for (u32 id = 0; id < beta.size(); ++id) { //loop over all nodes of tree_
370  count_[id] = count[id];
371  if (count_[id] > minAdaptationObservations_) {
372  for (u32 d = 0; d < dimension_; d++)
373  shift_[id][0][d] = shift[id][d] / beta[id][d];
374  }
375  else {
376  shift_.erase(id);
377  }
378  }
379 }

381 void ShiftAdaptorViterbiEstimator::accumulate(Core::Ref<const Feature::Vector> feature,
382  DensityIndex density,
383  MixtureIndex mixture,
384  Core::Ref<MixtureSet> mixtureSet) {
385  MeanIndex meanIndex = mixtureSet->density(density)->meanIndex();
386  const Mm::Mean& mean = *(mixtureSet->mean(meanIndex));
387 
388  CovarianceIndex covIndex = mixtureSet->density(density)->covarianceIndex();
389  const Mm::Covariance& cov = *(mixtureSet->covariance(covIndex));
390 
391  countAccumulators_[(*leafIndex_)[mixture]] += 1.0;
392  for (u32 d = 0; d < dimension_; d++) {
393  betaAccumulators_[(*leafIndex_)[mixture]][d] += 1.0 / cov.diagonal()[d];
394  shiftAccumulators_[(*leafIndex_)[mixture]][d] +=
395  ((*feature)[d] - mean[d]) / cov.diagonal()[d];
396  }
397 }

418 void ShiftAdaptorViterbiEstimator::accumulate(const Speech::Alignment& alignment,
419  const std::vector<Core::Ref<const Mm::Feature>>& featureSequence,
420  Core::Ref<const Am::MixtureSetAdaptor> unadaptedAdaptor,
421  Core::Ref<const Mm::AssigningFeatureScorer> unadaptedFeatureScorer) {
422  require(alignment.size() == featureSequence.size());
423  for (size_t item = 0; item < alignment.size(); ++item) {
424  if (alignment[item].weight < 0.5)
425  continue;
426  MixtureIndex mixture = unadaptedAdaptor->acousticModel()->emissionIndex(alignment[item].emission);
427  size_t density = unadaptedFeatureScorer->getAssigningScorer(featureSequence[item])->bestDensity(mixture);
428 
429  // Extract the mean vector
430  DensityIndex densityIndex = unadaptedAdaptor->mixtureSet()->mixture(mixture)->densityIndex(density);
431  MeanIndex meanIndex = unadaptedAdaptor->mixtureSet()->density(densityIndex)->meanIndex();
432  const Mm::Mean& mean = *(unadaptedAdaptor->mixtureSet()->mean(meanIndex));
433  CovarianceIndex covIndex = unadaptedAdaptor->mixtureSet()->density(densityIndex)->covarianceIndex();
434  const Mm::Covariance& cov = *(unadaptedAdaptor->mixtureSet()->covariance(covIndex));
435 
436  countAccumulators_[(*leafIndex_)[mixture]] += 1.0;
437  for (u32 d = 0; d < dimension_; d++) {
438  betaAccumulators_[(*leafIndex_)[mixture]][d] += 1.0 / cov.diagonal()[d];
439  shiftAccumulators_[(*leafIndex_)[mixture]][d] += ((*featureSequence[item]->mainStream())[d] - mean[d]) / cov.diagonal()[d];
440  }
441  }
442 }

444 Core::Ref<Adaptor> ShiftAdaptorViterbiEstimator::adaptor(void) {
445  Core::Ref<Adaptor> adaptor(new ShiftAdaptor(config));
446 
447  estimateWMatrices();
448 
449  std::vector<NodeId> adaptationMatrixIndex(leafIndex_->size());
450  std::vector<NodeId> matrixId(nLeafs_);
451  std::map<NodeId, Matrix> adaptationMatrices;
452  NodeId id;
453 
454  id = tree_->root();
455  if (count_[id] <= minAdaptationObservations_) {
456  //not enough observations in complete tree
457  ensure(id == tree_->root()); //the only case this can happen
458  log("too few observations for adaptation\n")
459  << minAdaptationObservations_ << " observations needed, "
460  << count_[id] << " seen.\n"
461  << "resetting adaptation to unity";
462  adaptationMatrices[id] = Math::Matrix<Sum>();
463  adaptationMatrices[id].addRow(Math::Vector<Sum>(dimension_));
464  for (Core::BinaryTree::LeafList::const_iterator p = tree_->leafList().begin();
465  p != tree_->leafList().end(); ++p) {
466  matrixId[tree_->leafNumber(p)] = id;
467  }
468  }
469  else {
470  for (Core::BinaryTree::LeafList::const_iterator p = tree_->leafList().begin();
471  p != tree_->leafList().end(); ++p) {
472  id = tree_->id(p);
473  NodeId leafId = tree_->leafNumber(id);
474  while (id != tree_->root() && count_[id] <= minAdaptationObservations()) {
475  id = tree_->previous(id);
476  }
477  ensure(id != Core::BinaryTree::invalidId);
478  ensure(count_[id] > minAdaptationObservations());
479  if (adaptationMatrices.find(id) == adaptationMatrices.end()) {
480  MatrixConstIterator p = shift_.find(id);
481  ensure(p != shift_.end()); //something has gone wrong in estimateWMatrices()
482  adaptationMatrices[id] = p->second;
483  }
484  matrixId[leafId] = id;
485  }
486  }
487 
488  Core::BinaryTree::Id silenceId = tree_->numberOfNodes() + tree_->numberOfLeafs();
489 
490  //estimate WMatrix for silence mixtures
491  for (std::set<MixtureIndex>::const_iterator p = silenceMixtures_.begin(); p != silenceMixtures_.end(); ++p) {
492  Core::BinaryTree::Id silenceLeaf = (*leafIndex_)[*p];
493  if (countAccumulators_[silenceLeaf] > minSilenceObservations_) {
494  adaptationMatrices[silenceId].clear();
495  adaptationMatrices[silenceId].addRow(Math::Vector<Sum>(dimension_));
496  for (u32 d = 0; d < dimension_; d++)
497  adaptationMatrices[silenceId][0][d] = shiftAccumulators_[silenceLeaf][d] / betaAccumulators_[silenceLeaf][d];
498  }
499  else {
500  log("too few observations for silence adaptation\n")
501  << minSilenceObservations_ << " observations needed, "
502  << int(countAccumulators_[silenceLeaf]) << " seen.\n"
503  << "resetting matrix to unity";
504  adaptationMatrices[silenceId].clear();
505  adaptationMatrices[silenceId].addRow(Math::Vector<Sum>(dimension_));
506  }
507  matrixId[silenceLeaf] = silenceId;
508  ++silenceId;
509  }
510 
511  log("Estimated ") << u32(adaptationMatrices.size()) << " adaptation shifts";
512  if (adaptationDumpChannel_.isOpen()) {
513  adaptationDumpChannel_ << Core::XmlOpen("mixture-shift-mapping");
514  }
515  for (u32 mixture = 0; mixture < leafIndex_->size(); mixture++) {
516 #ifndef __w_all__
517  adaptationMatrixIndex[mixture] = matrixId[(*leafIndex_)[mixture]];
518 #else
519  log("w_all estimation");
520  adaptationMatrixIndex[mixture] = matrixId[tree_->root()];
521 #endif
522  if (adaptationDumpChannel_.isOpen()) {
523  adaptationDumpChannel_ << Core::XmlEmpty("shift") + Core::XmlAttribute("shift-id", mixture) + Core::XmlAttribute("shift-id", adaptationMatrixIndex[mixture]);
524  }
525  }
526  if (adaptationDumpChannel_.isOpen()) {
527  adaptationDumpChannel_ << Core::XmlClose("mixture-shift-mapping");
528  }
529  adaptor->setMatrix(adaptationMatrices, adaptationMatrixIndex);
530  return adaptor;
531 }

671 void FullAdaptorViterbiEstimator::estimateWMatrices() {
672  Math::Vector<ZAccumulator> z;
673  Math::Vector<GAccumulator> g;
674 
675  propagate(leafZAccumulators_, z, tree_->root());
676  propagate(leafGAccumulators_, g, tree_->root());
677 
678  ensure(z.size() == g.size());
679  count_.resize(z.size());
680  for (u32 id = 0; id < z.size(); ++id) { //loop over all nodes of tree_
681  count_[id] = g[id].count();
682  if (count_[id] > minAdaptationObservations_) {
683  Matrix gInverse(g[id].matrix());
684  pseudoInvert(gInverse);
685  w_[id] = z[id].matrix() * gInverse;
686  }
687  else {
688  w_.erase(id);
689  }
690  }
691 }

741 Core::Ref<Adaptor> FullAdaptorViterbiEstimator::adaptor(void) {
742  Core::Ref<Adaptor> adaptor(new FullAdaptor(config));
743 
744  estimateWMatrices();
745 
746  std::vector<NodeId> adaptationMatrixIndex(leafIndex_->size());
747  std::vector<NodeId> matrixId(nLeafs_);
748  std::map<NodeId, Matrix> adaptationMatrices;
749  NodeId id;
750 
751  id = tree_->root();
752  if (count_[id] <= minAdaptationObservations_) {
753  //not enough observations in complete tree
754  ensure(id == tree_->root()); //the only case this can happen
755  log("too few observations for adaptation\n")
756  << minAdaptationObservations_ << " observations needed, "
757  << count_[id] << " seen.\n"
758  << "resetting matrix to unity";
759  adaptationMatrices[id] = adaptationUnitMatrix(dimension_);
760  for (Core::BinaryTree::LeafList::const_iterator p = tree_->leafList().begin();
761  p != tree_->leafList().end(); ++p) {
762  matrixId[tree_->leafNumber(p)] = id;
763  }
764  }
765  else {
766  for (Core::BinaryTree::LeafList::const_iterator p = tree_->leafList().begin();
767  p != tree_->leafList().end(); ++p) {
768  id = tree_->id(p);
769  NodeId leafId = tree_->leafNumber(id);
770  while (id != tree_->root() && count_[id] <= minAdaptationObservations()) {
771  id = tree_->previous(id);
772  }
773  ensure(id != Core::BinaryTree::invalidId);
774  ensure(count_[id] > minAdaptationObservations());
775  if (adaptationMatrices.find(id) == adaptationMatrices.end()) {
776  MatrixConstIterator p = w_.find(id);
777  ensure(p != w_.end()); //something has gone wrong in estimateWMatrices()
778  adaptationMatrices[id] = p->second;
779  }
780  matrixId[leafId] = id;
781  }
782  }
783 
784  /*! \todo change if silence is handled correctly by dectree */
785  Core::BinaryTree::Id silenceId = tree_->numberOfNodes() + tree_->numberOfLeafs();
786 
787  //estimate WMatrix for silence mixtures
788  for (std::set<MixtureIndex>::const_iterator p = silenceMixtures_.begin();
789  p != silenceMixtures_.end(); ++p) {
790  Core::BinaryTree::Id silenceLeaf = (*leafIndex_)[*p];
791  if (leafGAccumulators_[silenceLeaf].count() > minSilenceObservations_) {
792  Matrix gInverse(leafGAccumulators_[silenceLeaf].matrix());
793  pseudoInvert(gInverse);
794  adaptationMatrices[silenceId] = leafZAccumulators_[silenceLeaf].matrix() * gInverse;
795  }
796  else {
797  log("too few observations for silence adaptation\n")
798  << minSilenceObservations_ << " observations needed, "
799  << leafGAccumulators_[silenceLeaf].count() << " seen.\n"
800  << "resetting matrix to unity";
801  adaptationMatrices[silenceId] = adaptationUnitMatrix(dimension_);
802  }
803  matrixId[silenceLeaf] = silenceId;
804  ++silenceId;
805  }
806 
807  log("Estimated ") << u32(adaptationMatrices.size()) << " adaptation matrices";
808  if (adaptationDumpChannel_.isOpen()) {
809  adaptationDumpChannel_ << Core::XmlOpen("mixture-matrix-mapping");
810  }
811  for (u32 mixture = 0; mixture < leafIndex_->size(); mixture++) {
812 #ifndef __w_all__
813  adaptationMatrixIndex[mixture] = matrixId[(*leafIndex_)[mixture]];
814 #else
815  log("w_all estimation");
816  adaptationMatrixIndex[mixture] = matrixId[tree_->root()];
817 #endif
818  if (adaptationDumpChannel_.isOpen()) {
819  adaptationDumpChannel_ << Core::XmlEmpty("mixture") + Core::XmlAttribute("mixture-id", mixture) + Core::XmlAttribute("matrix-id", adaptationMatrixIndex[mixture]);
820  }
821  }
822  if (adaptationDumpChannel_.isOpen()) {
823  adaptationDumpChannel_ << Core::XmlClose("mixture-matrix-mapping");
824  }
825  adaptor->setMatrix(adaptationMatrices, adaptationMatrixIndex);
826  return adaptor;
827 }
