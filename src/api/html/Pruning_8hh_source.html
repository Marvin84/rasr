237  inline bool prune(TraceManager &trace_manager, StateHypothesis const& hyp) {
238  const uintptr_t invalidPruningMark = root_ptr ^ -1;
239  if (live_traces.count(hyp.trace)) {
240  return false;
241  }
242  if (dead_traces.count(hyp.trace)) {
243  return true;
244  }
245 
246  TraceItem &currentItem = trace_manager.traceItem(hyp.trace);
247  Trace* current = currentItem.trace.get();
248  std::vector<Trace*> chain;
249  bool should_prune = true;
250  while (current) {
251  chain.push_back(current);
252  if (current->pruningMark == root_ptr) {
253  should_prune = false;
254  break;
255  }
256  else if (current->pruningMark == invalidPruningMark) {
257  break;
258  }
259  current = current->predecessor.get();
260  }
261  if (should_prune) {
262  dead_traces.insert(hyp.trace);
263  for (Trace* e : chain) {
264  e->pruningMark = invalidPruningMark;
265  }
266  }
267  else {
268  live_traces.insert(hyp.trace);
269  for (Trace* e : chain) {
270  e->pruningMark = root_ptr;
271  }
272  }
273  return should_prune;
274  }
