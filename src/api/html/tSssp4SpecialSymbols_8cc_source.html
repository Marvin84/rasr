237  template<class _AddArc>
238  _Weight processState(const Fsa::StateId s, DirectHits& directHits) {
239  _Accumulator* collector = semiring_->getCollector(semiring_->zero());
240  _ConstStateRef state = this->fsa_->getState(s);
241  bool oldDirectHitsEmpty = directHits.empty();
242 
243  if (not sssp4SpecialSymbolsHelper::stateHasFailArc<_Automaton>(state))
244  if (oldDirectHitsEmpty)
245  for (typename _State::const_iterator arc = state->begin(); arc != state->end(); ++arc) {
246  _AddArc::regularArcWithoutDirectHits(&(*arc), collector, this);
247  }
248  else {
249  typename DirectHits::iterator darc = directHits.begin();
250  for (typename _State::const_iterator arc = state->begin(); arc != state->end(); ++arc)
251  _AddArc::regularArcWithDirectHits(&(*arc), directHits, darc, collector, this);
252  }
253  else {
254  bool noNoneFailArc = true;
255  typename DirectHits::iterator darc = directHits.begin();
256 
257  for (typename _State::const_iterator arc = state->begin(); arc != state->end(); ++arc) {
258  if (noNoneFailArc or not(arc->output() == Fsa::Failure or arc->input() == Fsa::Failure)) {
259  noNoneFailArc = false;
260 
261  if (oldDirectHitsEmpty) {
262  _AddArc::regularArcWithoutDirectHits(&(*arc), collector, this);
263  directHits.push_back(&(*arc));
264  }
265  else {
266  _AddArc::regularArcWithDirectHits(&(*arc), directHits, darc, collector, this);
267  }
268  }
269  else {
270  verify(arc->output() == arc->input()); // assume that we only have arcs with fail on both sides
271  _Weight failWeight = _AddArc::failArc(&(*arc), directHits, collector, oldDirectHitsEmpty, this);
272  // only first processState in recursion update fail potential
273  if (oldDirectHitsEmpty)
274  failPotential_.set(s, failWeight);
275  } // end else Failure
276  } // end for arc
277  }
278 
279  if (state->isFinal()) {
280  collector->feed(semiring_->extend(sssp4SpecialSymbolsHelper::convertWeight<_AutomataWeight, _Weight>(state->weight()), semiring_->one()));
281  }
282 
283  _Weight result = collector->get();
284  delete collector;
285  return result;
286  }
287 
288 protected:
289  void finishState(Fsa::StateId s) {
290  processedStates_.push(s);
291  DirectHits directHits;
292  _Weight potential;
293  if (sssp4SpecialSymbolsHelper::isTropicalSemiring<_ConstSemiringRef>(semiring_))
294  potential = processState<AddArcTropical>(s, directHits);
295  else if (semiring_->hasInvCollect())
296  potential = processState<AddArcNegCollect>(s, directHits);
297  else
298  potential = processState<AddArc>(s, directHits);
299  potential_.set(s, potential);
300  }

384  /**
385  * move forwardAccumulatorsPos_ to given arc
386  * if arc is not found forwardAccumulatorsPos_ is
387  * at position where arc should be included
388  * return: true if arc was found
389  */
390  bool walk2(const _Arc& arc, const Sorter& sorter) const {
391  if (RSorter::cmp(arc, forwardAccumulators_->front().first)) {
392  forwardAccumulatorsPos_ = forwardAccumulators_->begin();
393  return false;
394  }
395 
396  if (LSorter::cmp(forwardAccumulators_->back().first, arc)) {
397  forwardAccumulatorsPos_ = forwardAccumulators_->end();
398  return false;
399  }
400 
401  if (forwardAccumulatorsPos_ == forwardAccumulators_->end())
402  --forwardAccumulatorsPos_;
403 
404  // while forwardPotentialsPos > arc
405  while ((forwardAccumulatorsPos_ != forwardAccumulators_->begin()) and (RSorter::cmp(arc, forwardAccumulatorsPos_->first))) {
406  --forwardAccumulatorsPos_;
407  }
408  // while forwardPotentialsPos < arc
409  while (LSorter::cmp(forwardAccumulatorsPos_->first, arc)) {
410  ++forwardAccumulatorsPos_;
411  if (forwardAccumulatorsPos_ == forwardAccumulators_->end()) {
412  return false;
413  }
414  }
415  return not(RSorter::cmp(arc, forwardAccumulatorsPos_->first));
416  }

426  void feedForwardPotential(const _Arc& arc, const Sorter& sorter, const _Weight& w) {
427  if (forwardAccumulators_ == NULL) {
428  forwardAccumulators_ = new _ForwardAccumulators();
429  forwardAccumulatorsPos_ = forwardAccumulators_->begin();
430  _FPArc forwardArc(arc.target(), semiring_->zero(), arc.input(), arc.output());
431  _ArcAccumulatorPair pair(forwardArc, semiring_->getCollector(w));
432  forwardAccumulatorsPos_ = forwardAccumulators_->insert(forwardAccumulatorsPos_, pair);
433  }
434  else {
435  if (walk2(arc, sorter)) {
436  // forwardPotentials_ has arc
437  _ArcAccumulatorPair& pair = *forwardAccumulatorsPos_;
438  verify(pair.first.output() == arc.output());
439  if (pair.second == NULL)
440  pair.second = semiring_->getCollector(pair.first.weight());
441  pair.second->feed(w);
442  }
443  else {
444  // no element arc in forwardPotentials_
445  _FPArc forwardArc(arc.target(), semiring_->zero(), arc.input(), arc.output());
446  _ArcAccumulatorPair pair(forwardArc, semiring_->getCollector(w));
447  forwardAccumulatorsPos_ = forwardAccumulators_->insert(forwardAccumulatorsPos_, pair);
448  }

607  template<class _AddArc>
608  void processState(_ConstStateRef state, DirectHits& directHits, _Weight sourceForwardPotential) {
609  Fsa::StateId s = state->id();
610 
611  bool oldDirectHitsEmpty = directHits.empty();
612  ForwardPotential& fpot = this->getForwardPotential(s);
613 
614  if (not sssp4SpecialSymbolsHelper::stateHasFailArc<_Automaton>(state))
615  if (oldDirectHitsEmpty) {
616  for (typename _State::const_iterator arc = state->begin(); arc != state->end(); ++arc)
617  _AddArc::regularArcWithoutDirectHits(&(*arc), sourceForwardPotential, fpot, this);
618  }
619  else {
620  typename DirectHits::iterator darc = directHits.begin();
621  for (typename _State::const_iterator arc = state->begin(); arc != state->end(); ++arc)
622  _AddArc::regularArcWithDirectHits(&(*arc), directHits, darc, sourceForwardPotential, fpot, this);
623  }
624  else { // state has fail arc
625  bool noNoneFailArc = true;
626  typename DirectHits::iterator darc = directHits.begin();
627 
628  for (typename _State::const_iterator arc = state->begin(); arc != state->end(); ++arc) {
629  if (noNoneFailArc or not(arc->output() == Fsa::Failure or arc->input() == Fsa::Failure)) {
630  noNoneFailArc = false;
631 
632  if (oldDirectHitsEmpty) {
633  _AddArc::regularArcWithoutDirectHits(&(*arc), sourceForwardPotential, fpot, this);
634  directHits.push_back(&(*arc));
635  }
636  else {
637  _AddArc::regularArcWithDirectHits(&(*arc), directHits, darc, sourceForwardPotential, fpot, this);
638  }
639  }
640  else {
641  verify(arc->output() == arc->input()); // assume that we only have arcs with fail on both sides
642  _AddArc::failArc(&(*arc), directHits, sourceForwardPotential, oldDirectHitsEmpty, this);
643  } // end else Failure
644  } // end arc
645  }
646  }

648  void finishStates(const _Stack states) {
649  Fsa::StateId initialStateId = this->fsa_->initialStateId();
650  if (initialStateId == Fsa::InvalidStateId)
651  return;
652  _Accumulator* fwdFlowAcc = semiring_->getCollector();
653  Fsa::StateId firstState = *(states.begin());
654  require(firstState == initialStateId);
655 
656  this->getForwardPotential(initialStateId).feedForwardPotential(semiring_->one());
657 
658  for (_Stack::const_iterator s = states.begin(); s != states.end(); ++s) {
659  const ForwardPotential& fpot = this->getFinalizedForwardPotential(*s);
660  DirectHits directHits;
661  _ConstStateRef state = this->fsa_->getState(*s);
662  _Weight fStatePot = fpot.getForwardPotential();
663  if (semiring_->hasInvCollect())
664  processState<AddArcNegCollect>(state, directHits, fStatePot);
665  else
666  processState<AddArc>(state, directHits, fStatePot);
667 
668  if (state->isFinal()) {
669  fwdFlowAcc->feed(semiring_->extend(fStatePot, sssp4SpecialSymbolsHelper::convertWeight<_AutomataWeight, _Weight>(state->weight())));
670  }
671  }
672 
673  for (_Stack::const_iterator s = states.begin(); s != states.end(); ++s) {
674  this->getFinalizedForwardPotential(*s);
675  }
676 
677  totalFlow_ = fwdFlowAcc->get();
678  delete fwdFlowAcc;
679  }

682  SsspForward4SpecialSymbols(_ConstAutomatonRef f, const _Stack states)
683  : fsa_(sort<_Automaton>(f, Fsa::SortTypeByInputAndOutput)),
684  semiring_(sssp4SpecialSymbolsHelper::convertSemiring<_ConstSemiringRef>(this->fsa_->semiring())),
685  totalFlow_(semiring_->zero()) {
686  if (not f->hasProperty(Fsa::PropertyAcyclic))
687  Core::Application::us()->warning("SsspForward4SpecialSymbols: automaton does not have the property 'acyclic'");
688  finishStates(states);
689  }

877  bool bestArcFound = false;
