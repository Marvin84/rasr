24 /** Filter collects a vector sequence into an aggregate
25  * Input is sequence of vectors
26  * Output is an aggregate with vectors as aggregate streams
27  * Input vectors are collected with preceding and succeeding vectors.
28  * Parameters
29  * -expand-timestamp: if true timestamp of output encompasses the timestamps of its element;
30  * if false timestamp of the "present" element is copied to timestamp of output.
31  * -for further parameters @see DelayNode
32  * Note: in output vector oldest frames are first and recent frames last.
33  */

92 template<class T>
93 bool VectorSequenceAggregation<T>::putData() {
94  // collect sequence packets
95  std::vector<Flow::DataPtr<DataType>> inputData;
96  Flow::Timestamp* dummy = new Flow::Timestamp();
97  dummy->invalidateTimestamp();
98 
99  for (int relativeIndex = -(int)maxPastSize(); relativeIndex <= (int)maxFutureSize(); ++relativeIndex) {
100  Flow::DataPtr<DataType> o;
101  get(relativeIndex, o);
102  inputData.push_back(o);
103  if (expandTimestamp_)
104  dummy->expandTimestamp(*o);
105  else if (relativeIndex == 0)
106  dummy->setTimestamp(*o);
107  }
108 
109  // create new TypedAggegrate
110  Flow::TypedAggregate<DataType>* out = new Flow::TypedAggregate<DataType>(inputData);
111  out->invalidateTimestamp();
112  out->setTimestamp(*dummy);
113  delete dummy;
114  return Node::putData(0, out);
115 }
