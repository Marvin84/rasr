253  * Make static copy of lattice and
254  * calculate normalized fwd./bwd. scores.
255  */
256  ConstLatticeRef operator()(ConstLatticeRef l, const FwdBwd::Parameters& params, FwdBwd* fb = 0) {
257  if (!l || (l->initialStateId() == Fsa::InvalidStateId))
258  return ConstLatticeRef();
259  ConstSemiringRef posteriorSemiring = params.posteriorSemiring;
260  if (!posteriorSemiring)
261  posteriorSemiring = toLogSemiring(l->semiring(), params.alpha);
262 
263  bool hasRisk = (params.riskId != Semiring::InvalidId);
264  bool hasScores = ((params.scoreId != Semiring::InvalidId) || (params.riskId != Semiring::InvalidId));
265  /*
266  * Static lattice properties and skeleton
267  */
268  StaticLatticeRef s = StaticLatticeRef(new StaticLattice(l->type()));
269  s->setSemiring(l->semiring());
270  s->setInputAlphabet(l->getInputAlphabet());
271  if (l->type() != Fsa::TypeAcceptor)
272  s->setOutputAlphabet(l->getOutputAlphabet());
273  StaticBoundariesRef b = StaticBoundariesRef(new StaticBoundaries);
274  s->setBoundaries(b);
275  s->setDescription("static(" + l->describe() + ")");
276  s->setProperties(l->knownProperties(), l->properties());
277  /*
278  * Make lattice persistent and calculate some statistics
279  */
280  Properties properties;
281  TraverseLattice traverse(l, properties, *s, *b);
282  ConstStateMapRef topologicalSort = properties.topologicalSort;
283  s->setInitialStateId(topologicalSort->front());
284  /*
285  * Data structures
286  */
287  // normalized fwd./bwd. scores
288  FwdBwd::Internal* fbScores = 0;
289  FwdBwd::Arc * nextFwdBwdArc = 0, *endFwdBwdArc = 0;
290  if (fb) {
291  fbScores = fb->internal_ = new FwdBwd::Internal(ConstSemiringRefList(1, posteriorSemiring));
292  fbScores->states = new FwdBwd::State[topologicalSort->maxSid + 1];
293  fbScores->arcs = new FwdBwd::Arc[properties.nArcs];
294  nextFwdBwdArc = fbScores->arcs;
295  endFwdBwdArc = fbScores->arcs + properties.nArcs;
296  }
297  // temporary data structure
298  ScoreState* stateScores = new ScoreState[topologicalSort->maxSid + 1];
299  ScoreArc* fwdArcScores = new ScoreArc[properties.nArcs];
300  ScoreArc* bwdArcScores = new ScoreArc[properties.nArcs];
301  /*
302  * Calculate fwd./bwd. probabilities
303  */
304  Collector* col = createCollector(Fsa::SemiringTypeLog);
305  CostCollector* colCost = (hasRisk) ? CostCollector::create() : 0;
306  ScoreArc * nextFwdArcScores = fwdArcScores, *endFwdArcScores = fwdArcScores + properties.nArcs;
307  ScoreArc * nextBwdArcScores = bwdArcScores, *endBwdArcScores = bwdArcScores + properties.nArcs;
308  /*
309  * bwd. scores; build up bwd arc structure
310  */
311  for (StateMap::const_reverse_iterator itSid = topologicalSort->rbegin(),
312  endSid = topologicalSort->rend();
313  itSid != endSid; ++itSid) {
314  Fsa::StateId sid = *itSid;
315  ScoreState& stateScore = stateScores[sid];
316  std::pair<u32, u32>& fanInOut = properties.fanInOut[sid];
317  stateScore.fwdBegin = stateScore.fwdEnd = nextFwdArcScores;
318  nextFwdArcScores += fanInOut.first;
319  verify(nextFwdArcScores <= endFwdArcScores);
320  stateScore.bwdBegin = stateScore.bwdEnd = nextBwdArcScores;
321  nextBwdArcScores += fanInOut.second;
322  verify(nextBwdArcScores <= endBwdArcScores);
323  const Flf::State* sp = s->fastState(sid);
324  if (!sp->hasArcs()) {
325  verify(sp->isFinal());
326  stateScore.bwdScore = posteriorSemiring->project(sp->weight());
327  if (hasRisk)
328  stateScore.cost = stateScore.genBwdScore = sp->weight()->get(params.costId);
329  }
330  else {
331  stateScore.bwdScore = stateScore.cost = 0.0;
332  for (Flf::State::const_iterator a = sp->begin(), a_end = sp->end(); a != a_end; ++a) {
333  Fsa::StateId targetSid = a->target();
334  stateScore.fwdEnd->target = targetSid;
335  ScoreState& targetStateScore = stateScores[targetSid];
336  targetStateScore.bwdEnd->target = sid;
337  f64 score = posteriorSemiring->project(a->weight());
338  stateScore.fwdEnd->score = score;
339  targetStateScore.bwdEnd->score = score;
340  f64 bwdScore = targetStateScore.bwdScore + score;
341 
342  col->feed(bwdScore);
343  if (hasRisk) {
344  f64 cost = a->weight()->get(params.costId);
345  stateScore.fwdEnd->cost = targetStateScore.bwdEnd->cost = cost;
346  f64 genBwdScore = targetStateScore.genBwdScore + cost;
347  colCost->feed(bwdScore, genBwdScore);
348  }
349  ++stateScore.fwdEnd;
350  ++targetStateScore.bwdEnd;
351  }
352  stateScore.bwdScore = col->get();
353  col->reset();
354  if (hasRisk) {
355  stateScore.genBwdScore = colCost->get(stateScore.bwdScore);
356  colCost->reset();
357  }
358  }
359  }
360  /*
361  * fwd. scores
362  */
363  stateScores[topologicalSort->front()].fwdScore = 0.0;
364  for (StateMap::const_iterator itSid = topologicalSort->begin() + 1,
365  endSid = topologicalSort->end();
366  itSid != endSid; ++itSid) {
367  Fsa::StateId sid = *itSid;
368  ScoreState& stateScore = stateScores[sid];
369  for (const ScoreArc *a = stateScore.bwdBegin, *a_end = stateScore.bwdEnd; a != a_end; ++a) {
370  ScoreState& targetStateScore = stateScores[a->target];
371  f64 fwdScore = targetStateScore.fwdScore + a->score;
372 
373  col->feed(fwdScore);
374  if (hasRisk) {
375  f64 genFwdScore = targetStateScore.genFwdScore + a->cost;
376  colCost->feed(fwdScore, genFwdScore);
377  }
378  }
379  stateScore.fwdScore = col->get();
380  col->reset();
381  if (hasRisk) {
382  stateScore.genFwdScore = colCost->get(stateScore.fwdScore);
383  colCost->reset();
384  }
385  }
386  /*
387  * complete fwd. sums for final states;
388  * calculate fwd./bwd. sum
389  */
390  f64 bwdSum = stateScores[topologicalSort->front()].bwdScore;
391  f64 genBwdSum = (hasRisk) ? stateScores[topologicalSort->front()].genBwdScore : 0.0;
392  for (Core::Vector<Fsa::StateId>::const_iterator itSid = properties.finalStateIds.begin(), endSid = properties.finalStateIds.end();
393  itSid != endSid; ++itSid) {
394  ScoreState& stateScore = stateScores[*itSid];
395  stateScore.fwdScore += stateScore.bwdScore;
396 
397  col->feed(stateScore.fwdScore);
398  if (hasRisk) {
399  stateScore.genFwdScore += stateScore.cost;
400  colCost->feed(stateScore.fwdScore, stateScore.genFwdScore);
401  }
402  }
403 
404  f64 fwdSum = col->get();
405  col->reset();
406  {
407  f64 deviation = 0.5 * (fwdSum - bwdSum);
408  if ((deviation <= OOneInterval.first) || (OOneInterval.second <= deviation))
409  Core::Application::us()->warning(
410  "Numerical instability: fwd./bwd. scores are not equal; "
411  "fwd-score=%f, bwd-score=%f, probability quotient is %f and exceeds [0.99, 1.01]; ",
412  fwdSum, bwdSum, ::exp(-deviation));
413  }
414  f64 genFwdSum = 0.0;
415  if (hasRisk) {
416  genFwdSum = colCost->get(fwdSum);
417  colCost->reset();
418  f64 deviation = 0.5 * (genFwdSum - genBwdSum);
419  if ((deviation <= OOneInterval.first) || (OOneInterval.second <= deviation))
420  Core::Application::us()->warning(
421  "Numerical instability: generalized fwd./bwd. scores are not equal; "
422  "gen-fwd-score=%f, gen-bwd-score=%f, probability quotient is %f and exceeds [0.99, 1.01]; ",
423  genFwdSum, genBwdSum, ::exp(-deviation));
424  }
425 
426  /*
427  * fwd./bwd. probabilities
428  */
429  f64 normScore = 0.5 * (fwdSum + bwdSum);
430 
431  if (hasRisk) {
432  Core::Application::us()->log("Expected cost for lattice \"%s\" is %f.", l->describe().c_str(), 0.5 * (genFwdSum + genBwdSum));
433  }
434  f64 genNormScore = params.normRisk ? 0.5 * (genFwdSum + genBwdSum) : 0.0;
435  if (hasScores) {
436  const Semiring& semiring = *s->semiring();
437  for (StateMap::const_iterator itSid = topologicalSort->begin(), endSid = topologicalSort->end();
438  itSid != endSid; ++itSid) {
439  Fsa::StateId sid = *itSid;
440  ScoreState& stateScore = stateScores[sid];
441  Flf::State::iterator a = s->fastState(sid)->begin();
442  for (const ScoreArc *fa = stateScore.fwdBegin, *fa_end = stateScore.fwdEnd; fa != fa_end; ++fa, ++a) {
443  ScoreState& targetStateScore = stateScores[fa->target];
444  ScoresRef scores = a->weight_ = semiring.clone(a->weight());
445  Score fbScore = stateScore.fwdScore + fa->score + targetStateScore.bwdScore - normScore;
446  if (params.scoreId != Semiring::InvalidId)
447  scores->set(params.scoreId, fbScore);
448  if (hasRisk) {
449  f64 genFbScore = stateScore.genFwdScore + fa->cost + targetStateScore.genBwdScore - genNormScore;
450  Score risk = (genFbScore >= 0.0) ? ::exp(::log(genFbScore) - fbScore) : -::exp(::log(-genFbScore) - fbScore);
451  scores->set(params.riskId, risk);
452  }
453  }
454  verify_(a == s->getState(sid)->end());
455  }
456  }
457  if (fbScores) {
458  fbScores->sum = normScore;
459  for (StateMap::const_iterator itSid = topologicalSort->begin(), endSid = topologicalSort->end();
460  itSid != endSid; ++itSid) {
461  Fsa::StateId sid = *itSid;
462  ScoreState& stateScore = stateScores[sid];
463  FwdBwd::State& fbState = fbScores->states[sid];
464  fbState.fwdScore = stateScore.fwdScore;
465  fbState.bwdScore = stateScore.bwdScore;
466  fbState.normScore = normScore;
467  fbState.begin_ = fbState.end_ = nextFwdBwdArc;
468  for (const ScoreArc *fa = stateScore.fwdBegin, *fa_end = stateScore.fwdEnd; fa != fa_end; ++fa, ++fbState.end_) {
469  fbState.end_->arcScore = fa->score;
470  fbState.end_->fbScore = stateScore.fwdScore + fa->score + stateScores[fa->target].bwdScore;
471  fbState.end_->normScore = normScore;
472  Score posteriorScore = fbState.end_->fbScore - fbState.end_->normScore;
473  if (posteriorScore > fbScores->max)
474  fbScores->max = posteriorScore;
475  if (posteriorScore < fbScores->min)
476  fbScores->min = posteriorScore;
477  }
478  nextFwdBwdArc = fbState.end_;
479  verify(nextFwdBwdArc <= endFwdBwdArc);
480  }
481  }
482  /*
483  * clean up temporary data structure
484  */
485  delete[] stateScores;
486  stateScores = 0;
487  delete[] fwdArcScores;
488  fwdArcScores = 0;
489  delete[] bwdArcScores;
490  bwdArcScores = 0;
491  delete col;
492  col = 0;
493  delete colCost;
494  colCost = 0;
495  /*
496  * we are done
497  */
498  return s;
499  }

501  /*
502  * Build union of all lattices and
503  * calculate combined, normalized fwd./bwd. scores.
504  */
505  ConstLatticeRef operator()(const ConstLatticeRefList& _lats, const FwdBwd::CombinationParameters& params, FwdBwd* fb = 0) {
506  std::vector<u32> indexMap;
507  ConstLatticeRefList lats;
508  ConstSemiringRefList posteriorSemirings;
509  ScoreList weights;
510  /*
511  * Normalize weights, exclude lattices with weight zero; find posterior semiring
512  */
513  {
514  Score normWeight = 0.0;
515  for (u32 i = 0; i < _lats.size(); ++i) {
516  Score weight = params.weights.empty() ? 1.0 : params.weights[i];
517  if ((weight > 0.0) && _lats[i] && (_lats[i]->initialStateId() != Fsa::InvalidStateId)) {
518  indexMap.push_back(i);
519  lats.push_back(_lats[i]);
520  weights.push_back(weight);
521  posteriorSemirings.push_back(
522  (!params.posteriorSemirings.empty() && params.posteriorSemirings[i]) ? params.posteriorSemirings[i] : toLogSemiring(_lats[i]->semiring(), (params.alphas.empty() ? Score(0.0) : params.alphas[i])));
523  }
524  normWeight += weight;
525  }
526  if (lats.empty())
527  return ConstLatticeRef();
528  verify(normWeight != 0.0);
529  normWeight = 1.0 / normWeight;
530  for (ScoreList::iterator itWeight = weights.begin(); itWeight != weights.end(); ++itWeight)
531  *itWeight *= normWeight;
532  }
533  const SemiringCombinationHelper& semiringCombo = *params.combination->semiringCombination();
534  /*
535  * Union lattice properties and skeleton
536  */
537  bool hasSystemLabels = bool(params.systemAlphabet);
538  StaticLatticeRef unionL = StaticLatticeRef(new StaticLattice);
539  if (hasSystemLabels) {
540  unionL->setType(Fsa::TypeTransducer);
541  unionL->setInputAlphabet(params.combination->inputAlphabet());
542  unionL->setOutputAlphabet(params.systemAlphabet);
543  }
544  else {
545  unionL->setType(params.combination->type());
546  unionL->setInputAlphabet(params.combination->inputAlphabet());
547  if (params.combination->type() != Fsa::TypeAcceptor)
548  unionL->setOutputAlphabet(params.combination->outputAlphabet());
549  }
550  unionL->setSemiring(semiringCombo.semiring());
551  unionL->setProperties(Fsa::PropertyAll, Fsa::PropertyAcyclic);
552  unionL->setDescription(Core::form("union(n=%zu)", lats.size()));
553  StaticBoundariesRef unionB = StaticBoundariesRef(new StaticBoundaries);
554  unionL->setBoundaries(unionB);
555 
556  /*
557  * Build unified lattice and calculate some statistics
558  */
559  Core::Vector<Properties> propertiesList(lats.size());
560  u32 maxStates = 0, maxArcs = 0;
561  u32 nUnionStates = 2, nInitialUnionArcs = lats.size(), nUnionArcs = lats.size();
562  Time unionStartTime = Core::Type<Time>::max, unionEndTime = Core::Type<Time>::min;
563  for (u32 i = 0; i < lats.size(); ++i) {
564  Properties& properties = propertiesList[i];
565  properties.offset = nUnionStates;
566  TraverseSubLattice traverse(lats[i], properties, *unionL, *unionB);
567  maxStates = std::max(maxStates, properties.topologicalSort->maxSid + 1);
568  maxArcs = std::max(maxArcs, properties.nArcs);
569  nUnionStates += properties.topologicalSort->maxSid + 1;
570  nUnionArcs += properties.nArcs + properties.finalStateIds.size();
571  if (properties.startTime < unionStartTime)
572  unionStartTime = properties.startTime;
573  if (properties.endTime > unionEndTime)
574  unionEndTime = properties.endTime;
575  }
576  Flf::State* unionInitialSp = new Flf::State(0);
577  unionL->setState(unionInitialSp);
578  unionL->setInitialStateId(unionInitialSp->id());
579  unionB->set(unionInitialSp->id(), unionStartTime);
580  Flf::State* unionFinalSp = new Flf::State(1);
581  unionFinalSp->setFinal(unionL->semiring()->clone(unionL->semiring()->one()));
582  unionL->setState(unionFinalSp);
583  unionB->set(unionFinalSp->id(), unionEndTime);
584 
585  /*
586  * Fwd/Bwd data structures
587  */
588  // normalized fwd./bwd. scores
589  FwdBwd::Internal* fbScores = 0;
590  FwdBwd::Arc * nextFwdBwdArc = 0, *endFwdBwdArc = 0;
591  if (fb) {
592  fbScores = fb->internal_ = new FwdBwd::Internal(posteriorSemirings);
593  fbScores->sum = 0.0;
594  fbScores->states = new FwdBwd::State[nUnionStates];
595  fbScores->arcs = new FwdBwd::Arc[nUnionArcs];
596  FwdBwd::State& initialFbState = fbScores->states[unionInitialSp->id()];
597  initialFbState.fwdScore = 0.0;
598  initialFbState.bwdScore = 0.0;
599  initialFbState.normScore = 0.0;
600  initialFbState.begin_ = fbScores->arcs;
601  initialFbState.end_ = fbScores->arcs + nInitialUnionArcs;
602  FwdBwd::State& finalFbState = fbScores->states[unionFinalSp->id()];
603  finalFbState.fwdScore = 0.0;
604  finalFbState.bwdScore = 0.0;
605  finalFbState.normScore = 0.0;
606  nextFwdBwdArc = fbScores->arcs + nInitialUnionArcs;
607  endFwdBwdArc = fbScores->arcs + nUnionArcs;
608  }
609 
610  /*
611  * Temporary data structure
612  */
613  ScoreState* stateScores = new ScoreState[maxStates];
614  ScoreArc* fwdArcScores = new ScoreArc[maxArcs];
615  ScoreArc* bwdArcScores = new ScoreArc[maxArcs];
616 
617  /*
618  * Calculate fwd./bwd. probabilities
619  */
620  Collector* col = createCollector(Fsa::SemiringTypeLog);
621  for (u32 i = 0; i < lats.size(); ++i) {
622  Properties& properties = propertiesList[i];
623  Fsa::StateId offsetSid = properties.offset;
624  ConstStateMapRef topologicalSort = properties.topologicalSort;
625  ScoreArc * nextFwdArcScores = fwdArcScores, *endFwdArcScores = fwdArcScores + properties.nArcs;
626  ScoreArc * nextBwdArcScores = bwdArcScores, *endBwdArcScores = bwdArcScores + properties.nArcs;
627  ConstSemiringRef posteriorSemiring = posteriorSemirings[i];
628  const Semiring& semiring = *semiringCombo.semiring();
629  u32 paramIndex = indexMap[i];
630  Fsa::LabelId systemLabel = hasSystemLabels ? params.systemLabels[paramIndex] : Fsa::Epsilon;
631  // DEPR begin
632  ScoreId normId = params.normIds[paramIndex];
633  ScoreId weightId = params.weightIds[paramIndex];
634  Score normT = Score(1.0) / (Score(properties.endTime - unionStartTime));
635  // DEPR end
636 
637  /*
638  * bwd. scores; build up bwd arc structure
639  */
640  for (StateMap::const_reverse_iterator itSid = topologicalSort->rbegin(),
641  endSid = topologicalSort->rend();
642  itSid != endSid; ++itSid) {
643  Fsa::StateId sid = *itSid;
644  ScoreState& stateScore = stateScores[sid];
645  std::pair<u32, u32>& fanInOut = properties.fanInOut[sid];
646  stateScore.fwdBegin = stateScore.fwdEnd = nextFwdArcScores;
647  nextFwdArcScores += fanInOut.first;
648  verify(nextFwdArcScores <= endFwdArcScores);
649  stateScore.bwdBegin = stateScore.bwdEnd = nextBwdArcScores;
650  nextBwdArcScores += fanInOut.second;
651  verify(nextBwdArcScores <= endBwdArcScores);
652  Fsa::StateId unionSid = sid + offsetSid;
653  const Flf::State* unionSp = unionL->fastState(unionSid);
654  if (!unionSp->hasArcs()) {
655  verify(unionSp->isFinal());
656  stateScore.bwdScore = posteriorSemiring->project(unionSp->weight());
657  }
658  else {
659  stateScore.score = 0.0;
660  for (Flf::State::const_iterator a = unionSp->begin(), a_end = unionSp->end(); a != a_end; ++a) {
661  Fsa::StateId targetSid = a->target() - offsetSid;
662  f64 score = posteriorSemiring->project(a->weight());
663  stateScore.fwdEnd->target = targetSid;
664  stateScore.fwdEnd->score = score;
665  ++stateScore.fwdEnd;
666  ScoreState& targetStateScore = stateScores[targetSid];
667  col->feed(targetStateScore.bwdScore + score);
668  targetStateScore.bwdEnd->target = sid;
669  targetStateScore.bwdEnd->score = score;
670  ++targetStateScore.bwdEnd;
671  }
672  stateScore.bwdScore = col->get();
673  col->reset();
674  }
675  }
676  /*
677  * fwd. scores
678  */
679  stateScores[topologicalSort->front()].fwdScore = 0.0;
680  for (StateMap::const_iterator itSid = topologicalSort->begin() + 1, endSid = topologicalSort->end();
681  itSid != endSid; ++itSid) {
682  Fsa::StateId sid = *itSid;
683  ScoreState& stateScore = stateScores[sid];
684  for (const ScoreArc *a = stateScore.bwdBegin, *a_end = stateScore.bwdEnd; a != a_end; ++a)
685  col->feed(stateScores[a->target].fwdScore + a->score);
686  stateScore.fwdScore = col->get();
687  col->reset();
688  }
689  /*
690  * complete final state;
691  * calculate fwd./bwd. sums
692  */
693  for (Core::Vector<Fsa::StateId>::const_iterator itSid = properties.finalStateIds.begin(), endSid = properties.finalStateIds.end();
694  itSid != endSid; ++itSid) {
695  ScoreState& stateScore = stateScores[*itSid];
696  col->feed(stateScore.fwdScore + stateScore.bwdScore);
697  // stateScore.fwdScore += stateScore.score; // DEPR
698  }
699  f64 fwdSum = col->get();
700  col->reset();
701  f64 bwdSum = stateScores[topologicalSort->front()].bwdScore;
702  {
703  f64 deviation = 0.5 * (fwdSum - bwdSum);
704  if ((deviation <= OOneInterval.first) || (OOneInterval.second <= deviation))
705  Core::Application::us()->warning(
706  "Numerical instability: fwd./bwd. scores are not equal; "
707  "fwd-score=%f, bwd-score=%f, probability quotient is %f and exceeds [0.99, 1.01]; ",
708  fwdSum, bwdSum, ::exp(-deviation));
709  }
710  /*
711  * fwd./bwd. probabilities
712  */
713  f64 fbSum = 0.5 * (fwdSum + bwdSum);
714  f64 normScore = fbSum + ::log(weights[i]);
715 
716  // DEPR begin
717  f64 systemNormScore = 0.0;
718  if (normId != Semiring::InvalidId) {
719  if (params.fsaNorms[paramIndex]) {
720  Fsa::ConstAutomatonRef total = Fsa::staticCopy(
721  Fsa::changeSemiring(
722  toFsa(
723  lats[i],
724  posteriorSemiring->scales()),
725  Fsa::getSemiring(Fsa::SemiringTypeLog)));
726  Fsa::Weight totalInv;
727  Fsa::posterior64(total, totalInv);
728  systemNormScore = -f64(totalInv);
729  }
730  else
731  systemNormScore = fbSum;
732  }
733  // DEPR end
734 
735  if (fbScores) {
736  fbScores->sum += normScore;
737  // union initial state
738  {
739  FwdBwd::State& fbState = fbScores->states[unionInitialSp->id()];
740  FwdBwd::Arc& fbArc = fbState.begin_[i];
741  fbArc.arcScore = 0.0;
742  fbArc.fbScore = fbSum;
743  fbArc.normScore = normScore;
744  }
745  // sub-lattice states
746  for (StateMap::const_iterator itSid = topologicalSort->begin(),
747  endSid = topologicalSort->end();
748  itSid != endSid; ++itSid) {
749  Fsa::StateId sid = *itSid;
750  ScoreState& stateScore = stateScores[sid];
751  FwdBwd::State& fbState = fbScores->states[sid + offsetSid];
752  fbState.fwdScore = stateScore.fwdScore;
753  fbState.bwdScore = stateScore.bwdScore;
754  fbState.normScore = normScore;
755  fbState.begin_ = fbState.end_ = nextFwdBwdArc;
756  for (const ScoreArc *fa = stateScore.fwdBegin, *fa_end = stateScore.fwdEnd; fa != fa_end; ++fa, ++fbState.end_) {
757  fbState.end_->arcScore = fa->score;
758  fbState.end_->fbScore = stateScore.fwdScore + fa->score + stateScores[fa->target].bwdScore;
759  fbState.end_->normScore = normScore;
760  Score posteriorScore = fbState.end_->fbScore - fbState.end_->normScore;
761  if (posteriorScore > fbScores->max)
762  fbScores->max = posteriorScore;
763  if (posteriorScore < fbScores->min)
764  fbScores->min = posteriorScore;
765  }
766  nextFwdBwdArc = fbState.end_;
767  verify(nextFwdBwdArc <= endFwdBwdArc);
768  }
769  // final states
770  for (Core::Vector<Fsa::StateId>::const_iterator itSid = properties.finalStateIds.begin(), endSid = properties.finalStateIds.end();
771  itSid != endSid; ++itSid) {
772  Fsa::StateId sid = *itSid;
773  // ScoreState &stateScore = stateScores[sid];
774  FwdBwd::State& fbState = fbScores->states[sid + offsetSid];
775  fbState.begin_ = nextFwdBwdArc;
776  fbState.begin_->arcScore = fbState.bwdScore; // i.e. final state score
777  fbState.begin_->fbScore = fbState.fwdScore + fbState.bwdScore;
778  fbState.begin_->normScore = normScore;
779  fbState.end_ = ++nextFwdBwdArc;
780  verify(nextFwdBwdArc <= endFwdBwdArc);
781  }
782  }
783  /*
784  * Connect union initial state to sub-initial state and set fwd/bwd scores for initial arcs
785  */
786  {
787  ScoresRef scores = semiring.clone(semiring.one());
788  if (params.scoreId != Semiring::InvalidId)
789  scores->set(params.scoreId, -::log(weights[i]));
790  unionInitialSp->newArc(topologicalSort->front() + offsetSid, scores, Fsa::Epsilon, systemLabel);
791  }
792  /*
793  * Connect sub-final states to union final state and set fwd/bwd scores for remaining arcs
794  */
795  for (StateMap::const_iterator itSid = topologicalSort->begin(),
796  endSid = topologicalSort->end();
797  itSid != endSid; ++itSid) {
798  Fsa::StateId sid = *itSid;
799  ScoreState& stateScore = stateScores[sid];
800  Time beginT = unionB->time(sid + offsetSid);
801  Flf::State* sp = unionL->fastState(sid + offsetSid);
802  Flf::State::iterator a = sp->begin();
803  if (sp->isFinal()) {
804  verify(!sp->hasArcs());
805  ScoresRef scores = semiring.clone(semiring.one());
806  semiringCombo.set(scores, i, sp->weight());
807  sp->unsetFinal();
808  FwdBwd::State& fbState = fbScores->states[sid + offsetSid];
809  if (params.scoreId != Semiring::InvalidId)
810  scores->set(params.scoreId, fbState.fwdScore + fbState.bwdScore - normScore);
811  sp->newArc(unionFinalSp->id(), scores, Fsa::Epsilon, systemLabel);
812  }
813  else {
814  for (const ScoreArc *fa = stateScore.fwdBegin, *fa_end = stateScore.fwdEnd; fa != fa_end; ++fa, ++a) {
815  if (hasSystemLabels)
816  a->output_ = systemLabel;
817  ScoresRef scores = semiring.clone(semiring.one());
818  semiringCombo.set(scores, i, a->weight());
819  if (params.scoreId != Semiring::InvalidId) {
820  Score fbScore = stateScore.fwdScore + fa->score + stateScores[fa->target].bwdScore - normScore;
821  scores->set(params.scoreId, fbScore);
822  }
823 
824  // DEPR begin
825  if ((normId != Semiring::InvalidId) || (weightId != Semiring::InvalidId)) {
826  Score normDuration = normT * Score(unionB->time(a->target()) - beginT);
827  if (normId != Semiring::InvalidId)
828  scores->set(normId, -systemNormScore * normDuration);
829  if (weightId != Semiring::InvalidId)
830  scores->set(weightId, normDuration);
831  }
832  // DEPR end
833 
834  a->weight_ = scores;
835  }
836  verify_(a == unionL->fastState(sid + offsetSid)->end());
837  }
838  }
839  }
840  /*
841  * clean up temporary data structure
842  */
843  delete[] stateScores;
844  stateScores = 0;
845  delete[] fwdArcScores;
846  fwdArcScores = 0;
847  delete[] bwdArcScores;
848  bwdArcScores = 0;
849  /*
850  * Verify union fwd./bwd. sum
851  */
852  if (fbScores) {
853  f64 unionFwdBwdSum = 0.0;
854  {
855  FwdBwd::State& fbState = fbScores->states[unionInitialSp->id()];
856  verify(u32(fbState.end_ - fbState.begin_) == unionInitialSp->nArcs());
857  for (FwdBwd::Arc* itArc = fbState.begin_; itArc != fbState.end_; ++itArc)
858  col->feed(itArc->fbScore - itArc->normScore);
859  }
860  unionFwdBwdSum = col->get();
861  col->reset();
862  if ((unionFwdBwdSum <= OOneInterval.first) || (OOneInterval.second <= unionFwdBwdSum))
863  Core::Application::us()->warning(
864  "Numerical instability: sum over initial arcs exceeds [0.99, 1.01]; "
865  "normalized fwd. sum is %f.",
866  ::exp(-unionFwdBwdSum));
867  for (u32 i = 0; i < lats.size(); ++i) {
868  Properties& properties = propertiesList[i];
869  for (Core::Vector<Fsa::StateId>::const_iterator itSid = properties.finalStateIds.begin(), endSid = properties.finalStateIds.end();
870  itSid != endSid; ++itSid) {
871  const Flf::State* sp = unionL->fastState(*itSid + properties.offset);
872  verify(!sp->isFinal() && (sp->nArcs() == 1));
873  FwdBwd::State& fbState = fbScores->states[sp->id()];
874  verify(fbState.end_ - fbState.begin_ == 1);
875  col->feed(fbState.begin_->fbScore - fbState.begin_->normScore);
876  }
877  }
878  unionFwdBwdSum = col->get();
879  col->reset();
880  if ((unionFwdBwdSum <= OOneInterval.first) || (OOneInterval.second <= unionFwdBwdSum))
881  Core::Application::us()->warning(
882  "Numerical instability: sum over final arcs exceeds [0.99, 1.01]; "
883  "normalized fwd. sum is %f.",
884  ::exp(-unionFwdBwdSum));
885  }
886  /*
887  * Change to posterior semiring, preserving the scales used for FB calculation
888  */
889  if (params.setPosteriorSemiring) {
890  ScoreList unionPosteriorScales(unionL->semiring()->size(), 0.0);
891  for (ScoreId i = 0; i < lats.size(); ++i) {
892  u32 paramIndex = indexMap[i];
893  for (ScoreId j = 0; j < posteriorSemirings[i]->size(); ++j) {
894  ScoreId unionId = semiringCombo.subId(i, j);
895  if (unionId != Semiring::InvalidId)
896  unionPosteriorScales[unionId] = posteriorSemirings[i]->scale(j);
897  }
898 
899  // DEPR begin
900  ScoreId normId = params.normIds[paramIndex];
901  if (normId != Semiring::InvalidId)
902  unionPosteriorScales[normId] = 1.0;
903  ScoreId weightId = params.weightIds[paramIndex];
904  if (weightId != Semiring::InvalidId)
905  unionPosteriorScales[weightId] = -::log(weights[i]);
906  // DEPR end
907  }
908  ConstSemiringRef unionPosteriorSemiring = Semiring::create(
909  Fsa::SemiringTypeLog, unionL->semiring()->size(), unionPosteriorScales, unionL->semiring()->keys());
910  unionL->setSemiring(unionPosteriorSemiring);
911  }
912  delete col;
913  col = 0;
914  return unionL;
915  }
