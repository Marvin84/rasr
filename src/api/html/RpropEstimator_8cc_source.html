79 template<typename T>
80 void RpropEstimator<T>::estimate(NeuralNetwork<T>& network, Statistics<T>& statistics) {
81  require(statistics.hasGradient());
82 
83  if (needInit_)
84  initialize(network);
85 
86  verify(stepSizes_);
87 
88  // do everything on CPU (currently Rprop is only used in batch mode)
89  network.finishComputation();
90 
91  statistics.finishComputation();
92  stepSizes_->finishComputation();
93  if (prevStatistics_)
94  prevStatistics_->finishComputation();
95 
96  // update step sizes (except of first iteration)
97  if (prevStatistics_) {
98  for (u32 layer = 0; layer < network.nLayers(); layer++) {
99  if (network.getLayer(layer).isTrainable()) {
100  for (u32 stream = 0; stream < network.getLayer(layer).nInputActivations(); stream++) {
101  NnMatrix& gradient = statistics.gradientWeights(layer).at(stream);
102  NnMatrix& prevGradient = prevStatistics_->gradientWeights(layer).at(stream);
103  NnMatrix& stepSizes = stepSizes_->gradientWeights(layer).at(stream);
104  for (u32 i = 0; i < gradient.nRows(); i++) {
105  for (u32 j = 0; j < gradient.nColumns(); j++) {
106  updateStepSize(prevGradient.at(i, j), gradient.at(i, j), stepSizes.at(i, j));
107  }
108  }
109  }
110  NnVector& gradient = statistics.gradientBias(layer);
111  NnVector& prevGradient = prevStatistics_->gradientBias(layer);
112  NnVector& stepSizes = stepSizes_->gradientBias(layer);
113  for (u32 i = 0; i < gradient.nRows(); i++)
114  updateStepSize(prevGradient.at(i), gradient.at(i), stepSizes.at(i));
115  }
116  }
117  }
118  // update parameters
119  for (u32 layer = 0; layer < network.nLayers(); layer++) {
120  if (network.getLayer(layer).isTrainable()) {
121  for (u32 stream = 0; stream < network.getLayer(layer).nInputActivations(); stream++) {
122  NnMatrix* weights = network.getLayer(layer).getWeights(stream);
123  require(weights);
124  NnMatrix& gradient = statistics.gradientWeights(layer).at(stream);
125  NnMatrix& stepSizes = stepSizes_->gradientWeights(layer).at(stream);
126  // update weights
127  for (u32 i = 0; i < gradient.nRows(); i++) {
128  for (u32 j = 0; j < gradient.nColumns(); j++)
129  updateParameter(stepSizes.at(i, j), gradient.at(i, j), weights->at(i, j));
130  }
131  }
132  NnVector* bias = network.getLayer(layer).getBias();
133  require(bias);
134  NnVector& gradient = statistics.gradientBias(layer);
135  NnVector& stepSizes = stepSizes_->gradientBias(layer);
136 
137  // update bias
138  for (u32 i = 0; i < gradient.nRows(); i++)
139  updateParameter(stepSizes.at(i), gradient.at(i), bias->at(i));
140  }
141  }
142  if (this->logStepSize_ && this->statisticsChannel_.isOpen()) {
143  T normStepSize = 0;
144  for (u32 layer = 0; layer < network.nLayers(); layer++) {
145  if (network.getLayer(layer).isTrainable()) {
146  for (u32 stream = 0; stream < network.getLayer(layer).nInputActivations(); stream++) {
147  NnMatrix& stepSizes = stepSizes_->gradientWeights(layer).at(stream);
148  stepSizes.initComputation();
149  normStepSize += stepSizes.l1norm();
150  stepSizes.finishComputation(false);
151  }
152  NnVector& stepSizes = stepSizes_->gradientBias(layer);
153  stepSizes.initComputation();
154  normStepSize += stepSizes.l1norm();
155  stepSizes.finishComputation(false);
156  }
157  }
158  this->statisticsChannel_ << "step-size: " << normStepSize;
159  }
160 
161  if (!this->fullBatchMode_) {
162  if (prevStatistics_)
163  delete prevStatistics_;
164  prevStatistics_ = new Statistics<T>(statistics);
165  }
166  else {
167  if (!stepSizes_->write(paramStepSizesNew(this->config)))
168  this->error("failed to write step sizes to ") << paramStepSizesNew(this->config);
169  }
170 }

172 template<typename T>
173 void RpropEstimator<T>::initialize(const NeuralNetwork<T>& network) {
174  if (needInit_) {
175  verify(!stepSizes_);
176  verify(!prevStatistics_);
177 
178  if (!network.isComputing())
179  network.initComputation();
180  T parameterNorm = network.l1norm(true);
181 
182  // determine minimum of block averages
183  T blockMin = Core::Type<T>::max;
184  for (u32 layer = 0; layer < network.nLayers(); layer++) {
185  if (network.getLayer(layer).isTrainable()) {
186  for (u32 stream = 0; stream < network.getLayer(layer).nInputActivations(); stream++) {
187  const NnMatrix* weights = network.getLayer(layer).getWeights(stream);
188  T avgBlockNorm = weights->l1norm() / weights->size();
189  blockMin = std::min(blockMin, avgBlockNorm);
190  }
191  const NnVector* bias = network.getLayer(layer).getBias();
192  T avgBlockNorm = bias->l1norm() / bias->size();
193  blockMin = std::min(blockMin, avgBlockNorm);
194  }
195  }
196 
197  u32 nParameters = network.nFreeParameters();
198  T avgParameterNorm = parameterNorm / nParameters;
199  this->log("average norm of trainable parameters: ") << avgParameterNorm;
200 
201  // step sizes are stored in gradient-field of statistics, which has exactly the same structure
202  stepSizes_ = new Statistics<T>(network.nLayers(), Statistics<T>::GRADIENT);
203  stepSizes_->initialize(network);
204 
205  std::string filename = paramStepSizesOld(this->config);
206  if (filename != "") {
207  if (!stepSizes_->read(paramStepSizesOld(this->config)))
208  this->error("failed to read step size file ") << filename;
209 
210  require(paramPreviousStatistics(this->config) != ""); // need previous gradient (except of the initial step)
211 
212  u32 statisticsTypeFile = 0;
213  bool dummy;
214  if (!Nn::Statistics<T>::getTypeFromFile(paramPreviousStatistics(this->config), statisticsTypeFile, dummy))
215  this->error("could not read header from file: ") << paramPreviousStatistics(this->config);
216  u32 requiredStats = requiredStatistics();
217  u32 statisticsType = statisticsTypeFile | requiredStats; // union
218  prevStatistics_ = new Statistics<T>(network.nLayers(), statisticsType);
219  prevStatistics_->initialize(network);
220  if (!prevStatistics_->read(paramPreviousStatistics(this->config)))
221  this->error("failed to read previous statistics from file ") << paramPreviousStatistics(this->config);
222  }
223  else {
224  stepSizes_->initComputation(false);
225  for (u32 layer = 0; layer < network.nLayers(); layer++) {
226  if (network.getLayer(layer).isTrainable()) {
227  for (u32 stream = 0; stream < network.getLayer(layer).nInputActivations(); stream++) {
228  T initialStepSize = 0;
229  const NnMatrix* weights = network.getLayer(layer).getWeights(stream);
230  if (initializationType_ == constant)
231  initialStepSize = relativeInitialStepSize_ != 0 ? relativeInitialStepSize_ * avgParameterNorm : this->initialLearningRate_;
232  else if (initializationType_ == minBlockAverage)
233  initialStepSize = blockMin * relativeInitialStepSize_;
234  else if (initializationType_ == blockwiseAverage)
235  initialStepSize = weights->l1norm() / weights->size() * relativeInitialStepSize_;
236  stepSizes_->gradientWeights(layer).at(stream).fill(initialStepSize);
237  }
238  T initialStepSize = 0;
239  const NnVector* bias = network.getLayer(layer).getBias();
240  if (initializationType_ == constant)
241  initialStepSize = relativeInitialStepSize_ != 0 ? relativeInitialStepSize_ * avgParameterNorm : this->initialLearningRate_;
242  else if (initializationType_ == minBlockAverage)
243  initialStepSize = blockMin * relativeInitialStepSize_;
244  else if (initializationType_ == blockwiseAverage)
245  initialStepSize = bias->l1norm() / bias->size() * relativeInitialStepSize_;
246  stepSizes_->gradientBias(layer).fill(initialStepSize);
247  }
248  }
249  stepSizes_->finishComputation();
250  }
251  stepSizes_->initComputation();
252  this->log("average norm of step sizes: ") << (stepSizes_->gradientL1Norm() / nParameters);
253  stepSizes_->finishComputation(false);
254  network.finishComputation(false);
255 
256  needInit_ = false;
257  }
258 }

260 template<typename T>
261 inline void RpropEstimator<T>::updateStepSize(const T& prevGradient, const T& gradient, T& stepSize) const {
262  if (stepSize < 0)
263  stepSize *= -1.0; // un-set hold
264  else if (prevGradient * gradient >= 0) // no sign change in gradient
265  stepSize *= increasingFactor_;
266  else // sign change, negative sign marks hold, i.e. no step is done
267  stepSize *= -decreasingFactor_;
268 }

270 template<typename T>
271 inline void RpropEstimator<T>::updateParameter(const T& stepSize, const T& gradient, T& parameter) const {
272  if (stepSize > 0)
273  parameter -= stepSize * signum(gradient); // minimization problem formulation -> go into direction of negative gradient
274 }
