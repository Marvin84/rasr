26 /**
27  * Iterate through cn.
28  * If cn is normalized, then the slot is sorted in descending order by scores[posteriorId].
29  **/
30 class CnSortingIterator {
31 private:
32  struct ProbabilityWeakOrder {
33  ScoreId posteriorId;
34  ProbabilityWeakOrder(ScoreId posteriorId)
35  : posteriorId(posteriorId) {}
36  bool operator()(const ConfusionNetwork::Arc& a1, const ConfusionNetwork::Arc& a2) const {
37  return a1.scores->get(posteriorId) > a2.scores->get(posteriorId);
38  }
39  };
40 
41 private:
42  ConstConfusionNetworkRef cn_;
43  ScoreId posteriorId_;
44  bool isNormalized_;
45  const ConfusionNetwork::Slot* slot_;
46  ConfusionNetwork::Slot sortedSlot_;
47  ConfusionNetwork::const_iterator it_, end_;
48 
49 private:
50  CnSortingIterator(ConstConfusionNetworkRef cn)
51  : cn_(cn), posteriorId_(Semiring::InvalidId), slot_(0) {
52  verify(cn);
53  if (cn->isNormalized()) {
54  posteriorId_ = cn->normalizedProperties->posteriorId;
55  verify(posteriorId_ != Semiring::InvalidId);
56  }
57  }
58 
59  void setSlot() {
60  if (it_ != end_) {
61  if (posteriorId_ != Semiring::InvalidId) {
62  sortedSlot_ = *it_;
63  std::sort(sortedSlot_.begin(), sortedSlot_.end(), ProbabilityWeakOrder(posteriorId_));
64  slot_ = &sortedSlot_;
65  }
66  else
67  slot_ = &*it_;
68  }
69  else
70  slot_ = 0;
71  }
72 
73 public:
74  const ConfusionNetwork::Slot& operator*() const {
75  return *slot_;
76  }
77  bool operator!=(const CnSortingIterator& it) const {
78  return it_ != it.it_;
79  }
80  void operator++() {
81  verify(it_ != end_);
82  ++it_;
83  setSlot();
84  }
85 
86 public:
87  static CnSortingIterator begin(ConstConfusionNetworkRef cn) {
88  CnSortingIterator cnIt(cn);
89  cnIt.it_ = cn->begin();
90  cnIt.end_ = cn->end();
91  cnIt.setSlot();
92  return cnIt;
93  }
94  static CnSortingIterator end(ConstConfusionNetworkRef cn) {
95  CnSortingIterator cnIt(cn);
96  cnIt.it_ = cnIt.end_ = cn->end();
97  return cnIt;
98  }
99 };

107 /**
108  * Text-Format:
109  * # oracle
110  * #word begin duration <key[0]>/<scale[0]> <key[1]>/<scale[0]> ...
111  * # slot
112  * #word begin duration <key[0]>/<scale[0]> <key[1]>/<scale[0]> ...

115  * <word> <begin> <duration> <score[0]> <score[1]> ...

118  * ...

123 void writeConfusionNetworkHeaderAsText(std::ostream& os, Fsa::ConstAlphabetRef alphabet, ConstSemiringRef semiring) {
124  os << "#word\tbegin\tduration";
125  for (u32 i = 0; i < semiring->size(); ++i)
126  os << "\t" << semiring->keys()[i] << "/" << semiring->scales()[i];
127  os << std::endl;
128 }

130 void writeConfusionNetworkArcAsText(std::ostream& os, Fsa::ConstAlphabetRef alphabet, ConstSemiringRef semiring, const ConfusionNetwork::Arc& arc) {
131  if (arc.begin != Speech::InvalidTimeframeIndex)
132  os << arc.begin;
133  else
134  os << "inf";
135  if (arc.duration != Speech::InvalidTimeframeIndex)
136  os << "\t" << arc.duration;
137  else
138  os << "\t"
139  << "inf";
140  os << "\t" << ((arc.label == Fsa::Epsilon) ? "@" : alphabet->symbol(arc.label));
141  for (Scores::const_iterator itScore = arc.scores->begin(),
142  endScore = arc.scores->begin() + semiring->size();
143  itScore != endScore; ++itScore)
144  os << "\t" << *itScore;
145  os << std::endl;
146 }

149 void writeConfusionNetworkAsText(std::ostream& os, ConstConfusionNetworkRef cn, ConstSegmentRef segment) {
150  if (segment)
151  printSegmentHeader(os, segment);
152  if (!cn)
153  return;
154  if (cn->isNormalized())
155  os << "# Normalized CN (posterior key is \"" << cn->semiring->key(cn->normalizedProperties->posteriorId) << "\")." << std::endl;
156  if (cn->isNBestAlignment())
157  os << "# " << cn->nBestAlignmentProperties->n << "-best CN" << std::endl;
158  Fsa::ConstAlphabetRef alphabet = cn->alphabet;
159  ConstSemiringRef semiring = cn->semiring;
160  ConfusionNetwork::ConstOracleAlignmentPropertiesRef oracleProps;
161  if (cn->isOracleAlignment()) {
162  os << "oracle" << std::endl;
163  oracleProps = cn->oracleAlignmentProperties;
164  writeConfusionNetworkHeaderAsText(os, alphabet, oracleProps->semiring);
165  }
166  os << "slot" << std::endl;
167  writeConfusionNetworkHeaderAsText(os, alphabet, semiring);
168  u32 i = 0;
169  for (CnSortingIterator itSlot = CnSortingIterator::begin(cn),
170  endSlot = CnSortingIterator::end(cn);
171  itSlot != endSlot; ++itSlot, ++i) {
172  if (oracleProps) {
173  os << "# oracle " << (i + 1) << std::endl;
174  writeConfusionNetworkArcAsText(os, alphabet, oracleProps->semiring, oracleProps->alignment[i]);
175  }
176  os << "# slot " << (i + 1) << std::endl;
177  const ConfusionNetwork::Slot& slot = *itSlot;
178  for (ConfusionNetwork::Slot::const_iterator itArc = slot.begin(), endArc = slot.end(); itArc != endArc; ++itArc)
179  writeConfusionNetworkArcAsText(os, alphabet, semiring, *itArc);
180  }
181 }

256  void startRoot(const Core::XmlAttributes atts) {
257  const char* c = atts["version"];
258  if (c) {

313  void startHeader(const Core::XmlAttributes atts) {
314  normalizedPosteriorKey_.clear();
315  oracleAlignment_ = 0;

317  cn_->nBestAlignmentProperties.reset();
318  cn_->oracleAlignmentProperties.reset();

321  void endHeader() {
322  cn_->semiring = semiringXmlElement_->semiring();
323  if (!normalizedPosteriorKey_.empty()) {
324  ScoreId posteriorId = cn_->semiring->id(normalizedPosteriorKey_);
325  if (posteriorId == Semiring::InvalidId)
326  parser()->criticalError("Semiring \"%s\" has no dimension labeled \"%s\".",
327  cn_->semiring->name().c_str(), normalizedPosteriorKey_.c_str());
328  cn_->normalizedProperties = ConfusionNetwork::ConstNormalizedPropertiesRef(
329  new ConfusionNetwork::NormalizedProperties(posteriorId));
330  }
331  init();
332  }

335  verify(cn_->alphabet && cn_->semiring);
336  const char* c = atts["n"];

345  cn_->push_back(ConfusionNetwork::Slot());

347  if (c) {
348  s32 n;

356  std::sort(cn_->back().begin(), cn_->back().end());

360  scores_ = semiring_->clone(semiring_->one());

553 void writeConfusionNetworkAsXml(Core::XmlWriter& xml, ConstConfusionNetworkRef cn, ConstSegmentRef segment, u32 ioFlag) {
554  if (!cn)
555  return;
556  {
557  Core::XmlOpen xmlCnOpen("cn");
558  xmlCnOpen + Core::XmlAttribute("version", "1.1");
559  if (segment) {
560  if (segment->hasSegmentId())
561  xmlCnOpen + Core::XmlAttribute("name", segment->segmentId());
562  if (segment->hasRecordingId())
563  xmlCnOpen + Core::XmlAttribute("recording", segment->recordingId());
564  if (segment->hasStartTime() && segment->hasEndTime())
565  xmlCnOpen + Core::XmlAttribute("start", Core::form("%.3f", segment->startTime())) + Core::XmlAttribute("end", Core::form("%.3f", segment->endTime()));
566  }
567  xml << xmlCnOpen;
568  }
569  if (ioFlag & IoFlag::WriteHead) {
570  xml << Core::XmlOpen("head");
571  xml << Core::XmlEmpty("alphabet") + Core::XmlAttribute("name", Lexicon::us()->alphabetName(Lexicon::us()->alphabetId(cn->alphabet)).c_str());
572  cn->semiring->write(xml);
573  if (cn->isNormalized())
574  xml << Core::XmlEmpty("normalized") + Core::XmlAttribute("posterior-key", cn->semiring->key(cn->normalizedProperties->posteriorId));
575  if (cn->isNBestAlignment())
576  xml << Core::XmlEmpty("n-best") + Core::XmlAttribute("n", cn->nBestAlignmentProperties->n);
577  if (cn->isOracleAlignment()) {
578  xml << Core::XmlOpen("oracle");
579  cn->oracleAlignmentProperties->semiring->write(xml);
580  xml << Core::XmlClose("oracle");
581  }
582  xml << Core::XmlClose("head");
583  }
584  if (ioFlag & IoFlag::WriteBody) {
585  Fsa::ConstAlphabetRef alphabet = cn->alphabet;
586  ConstSemiringRef semiring = cn->semiring;
587  ConstSemiringRef oracleSemiring;
588  const ConfusionNetwork::OracleAlignmentProperties::OracleAlignment* oracleAlignment = 0;
589  if (cn->isOracleAlignment()) {
590  oracleSemiring = cn->oracleAlignmentProperties->semiring;
591  oracleAlignment = &cn->oracleAlignmentProperties->alignment;
592  }
593  xml << (Core::XmlOpen("body") + Core::XmlAttribute("n", cn->size()));
594  u32 i = 0;
