108 bool ScatterDiagonalNormalization::work(const ScatterMatrix& scatter) {
109  bool success = true;
110  Math::Vector<ScatterMatrix::Type> diagonal = scatter.diagonal();
111  for (size_t i = 0; i < diagonal.size(); ++i) {
112  if (diagonal[i] < 0 && Core::abs(diagonal[i]) > tolerance_) {
113  error("Normalization failed because %zu-th diagonal element %f is negative.", i, diagonal[i]);
114  success = false;
115  }
116  }
117  if (success) {
118  diagonal.takeSquareRoot(tolerance_);
119  diagonal.takeReciprocal(tolerance_);
120  Message scalingMessage(log("Scaling factors:\n"));
121  for (size_t i = 0; i < diagonal.size(); ++i) {
122  if (diagonal[i] == 0)
123  scalingMessage << "no-scaling ";
124  else
125  scalingMessage << diagonal[i] << " ";
126  }
127  Math::Vector<TransformationMatrix::Type> results = diagonal;
128  transform_ = makeDiagonalMatrix(results);
129  if (!Precursor::work())
130  success = false;
131  }
132  return success;
133 }

156 bool ScatterThresholding::work() {
157  ScatterMatrix scatterMatrix;
158  readScatterMatrix(inputScatterFilename_, totalScatterType, scatterMatrix);
159  for (size_t i = 0; i < scatterMatrix.nRows(); ++i) {
160  for (size_t j = 0; j < scatterMatrix.nColumns(); ++j) {
161  if (scatterMatrix[i][j] < threshold_)
162  scatterMatrix[i][j] = threshold_;
163  }
164  }
165  scatterMatrix_ = scatterMatrix;
166  return true;
