56  void convert() {
57  for (u32 a = 0; a < tree->states_.size(); ++a) {
58  StateId created = subtrees.allocateTreeNode(masterTreeIndex);
59  verify(a + 1 == created);
60  }
61 
62  for (int a = ((int)tree->states_.size()) - 1; a >= 0; --a) {
63  StateTree::StateId state = a;
64  StateId node = a + 1;
65  convert(state, node);
66  }
67 
68  std::set<StateTree::StateId> coarticulatedRootStates;
69  for (StateTree::StateId state = 0; state < tree->nStates(); ++state) {
70  for (std::list<StateTree::Exit>::const_iterator it = tree->state(state).exits.begin(); it != tree->state(state).exits.end(); ++it) {
71  if ((*it).transitEntry != tree->root())
72  coarticulatedRootStates.insert((*it).transitEntry);
73  }
74  }
75 
76  ///Make sure a node is created for every single state, so that also the coarticulated roots are respected
77 
78  for (std::set<StateTree::StateId>::iterator stateIt = coarticulatedRootStates.begin(); stateIt != coarticulatedRootStates.end(); ++stateIt) {
79  StateTree::StateId state = *stateIt;
80 
81  verify(nodesForStates.find(state) != nodesForStates.end());
82  coarticulatedRootNodes.insert(nodesForStates[state]);
83  rootTransitDescriptions[nodesForStates[state]] = tree->describeRootState(state);
84  }
85 
86  verify(nodesForStates.find(tree->root()) != nodesForStates.end());
87 
88  rootTransitDescriptions[nodesForStates[tree->root()]] = tree->describeRootState(tree->root());
89  rootTransitDescriptions[nodesForStates[tree->ciRoot()]] = tree->describeRootState(tree->ciRoot());
90 
91  /// Care about outputs
92 
93  for (std::unordered_map<StateTree::StateId, StateId>::const_iterator it = nodesForStates.begin(); it != nodesForStates.end(); ++it) {
94  StateId node = (*it).second;
95  StateTree::StateId state = (*it).first;
96  verify(node == state + 1);
97 
98  std::set<u32> exitIndices;
99  const StateTree::State& realState(tree->state(state));
100 
101  for (std::list<StateTree::Exit>::const_iterator it = realState.exits.begin(); it != realState.exits.end(); ++it) {
102  std::map<StateTree::Exit, u32>::iterator exitEntry = exits.find(*it);
103 
104  if (exitEntry == exits.end()) {
105  PersistentStateTree::Exit e;
106  std::unordered_map<StateTree::StateId, StateId>::iterator nodeIt = nodesForStates.find(it->transitEntry);
107  verify(nodeIt != nodesForStates.end());
108 
109  if (it->pronunciation) {
110  e.pronunciation = it->pronunciation->id();
111  verify(it->pronunciation);
112  }
113  else {
114  e.pronunciation = Bliss::LemmaPronunciation::invalidId;
115  }
116  e.transitState = nodeIt->second;
117 
118  exitVector.push_back(e);
119  exitEntry = exits.insert(std::make_pair(*it, exitVector.size() - 1)).first;
120  }
121  exitIndices.insert(exitEntry->second);
122  }
123 
124  //Add connections to the attached outputs/exits
125  for (std::set<u32>::iterator it = exitIndices.begin(); it != exitIndices.end(); ++it)
126  subtrees.addOutputToEdge(subtrees.state(node).successors, *it);
127  }
128  }

130 private:
131  void convert(StateTree::StateId stateId, StateId node) {
132  if (stateId == tree->root())
133  rootSubTree = node;
134  if (stateId == tree->ciRoot())
135  ciRootNode = node;
136 
137  if (nodesForStates.find(stateId) != nodesForStates.end()) {
138  verify(nodesForStates[stateId] == node);
139  verify(statesForNodes[node] = stateId);
140  return;
141  }
142 
143  verify(stateId + 1 == node);
144 
145  nodesForStates[stateId] = node;
146 
147  statesForNodes[node] = stateId;
148 
149  const StateTree::StateDesc& state = tree->stateDesc(stateId);
150 
151  subtrees.state(node).stateDesc = state;
152 
153  //Build successor structure
154  std::pair<StateTree::SuccessorIterator, StateTree::SuccessorIterator> successors = tree->successors(stateId);
155 
156  StateId current = node; //Just to verify the order
157 
158  for (; successors.first != successors.second; ++successors.first) {
159  std::unordered_map<StateTree::StateId, StateId>::iterator nodeIt = nodesForStates.find(*successors.first);
160  verify(nodeIt != nodesForStates.end());
161  verify(nodeIt->second > current);
162  current = nodeIt->second;
163 
164  subtrees.addNodeToEdge(subtrees.state(node).successors, current);
165  }
166  }

340 void PersistentStateTree::removeOutputs() {
341  Core::Application::us()->log() << "removing outputs from the search network";
342 
343  for (StateId node = 1; node < structure.stateCount(); ++node) {
344  HMMStateNetwork::ChangePlan change = structure.change(node);
345 
346  for (HMMStateNetwork::SuccessorIterator it = structure.successors(node); it; ++it)
347  if (it.isLabel())
348  change.removeSuccessor(*it);
349  ;
350 
351  change.apply();
352  }
353 
354  std::list<StateId> rootsList;
355 
356  {
357  std::set<StateId> roots = coarticulatedRootStates;
358  roots.insert(rootState);
359  roots.insert(ciRootState);
360 
361  // Also collect all transition-successors as coarticulated roots
362  for (StateId node = 1; node < structure.stateCount(); ++node) {
363  for (HMMStateNetwork::SuccessorIterator target = structure.successors(node); target; ++target)
364  if (target.isLabel())
365  roots.insert(exits[target.label()].transitState);
366  }
367 
368  for (std::set<StateId>::iterator it = roots.begin(); it != roots.end(); ++it)
369  rootsList.push_back(*it);
370  }
371 
372  HMMStateNetwork::CleanupResult cleanupResult = structure.cleanup(rootsList, masterTree, false, true);
373 
374  for (std::unordered_map<StateId, StateId>::const_iterator it = cleanupResult.nodeMap.begin(); it != cleanupResult.nodeMap.end(); ++it) {
375  if (it->first != it->second)
376  std::cout << "mapped " << it->first << " to " << it->second << std::endl;
377  verify(it->first == it->second);
378  }
379  for (std::unordered_map<TreeIndex, TreeIndex>::const_iterator it = cleanupResult.treeMap.begin(); it != cleanupResult.treeMap.end(); ++it)
380  verify(it->first == it->second);

383 HMMStateNetwork::CleanupResult PersistentStateTree::cleanup(bool cleanupExits) {
384  Core::Application::us()->log() << "cleaning up the search network";
385 
386  if (cleanupExits) {
387  std::vector<Exit> newExits;
388 
389  for (StateId node = 1; node < structure.stateCount(); ++node) {
390  std::set<u32> removeList;
391  std::set<u32> addList;
392 
393  for (HMMStateNetwork::SuccessorIterator it = structure.successors(node); it; ++it) {
394  if (it.isLabel()) {
395  removeList.insert(it.label());
396  addList.insert(newExits.size());
397  newExits.push_back(exits[it.label()]);
398  }
399  }
400 
401  for (std::set<u32>::iterator it = removeList.begin(); it != removeList.end(); ++it)
402  structure.removeOutputFromNode(node, *it);
403 
404  for (std::set<u32>::iterator it = addList.begin(); it != addList.end(); ++it)
405  structure.addOutputToEdge(structure.state(node).successors, *it);
406  }
407 
408  Core::Application::us()->log() << "changed number of exits from " << exits.size() << " to " << newExits.size();
409  exits.swap(newExits);
410  }
411 
412  std::list<StateId> rootsList;
413 
414  for (std::set<StateId>::iterator it = unpushedCoarticulatedRootStates.begin(); it != unpushedCoarticulatedRootStates.end(); ++it)
415  verify(coarticulatedRootStates.count(*it));
416 
417  {
418  std::set<StateId> roots = coarticulatedRootStates;
419  roots.insert(rootState);
420  roots.insert(ciRootState);
421 
422  // Also collect all transition-successors as coarticulated roots
423  for (StateId node = 1; node < structure.stateCount(); ++node) {
424  for (HMMStateNetwork::SuccessorIterator target = structure.successors(node); target; ++target)
425  if (target.isLabel())
426  roots.insert(exits[target.label()].transitState);
427  }
428 
429  for (std::set<StateId>::iterator it = roots.begin(); it != roots.end(); ++it)
430  rootsList.push_back(*it);
431  }
432 
433  ///@todo Go through the search tree, and collect the required coarticulated root nodes
434 
435  HMMStateNetwork::CleanupResult cleanupResult = structure.cleanup(rootsList, masterTree);
436 
437  Core::HashMap<StateId, StateId>::const_iterator targetNodeIt;
438  if (rootState) {
439  verify(cleanupResult.nodeMap.find(rootState) != cleanupResult.nodeMap.end()); //Root-node must stay unchanged
440  verify(cleanupResult.nodeMap.find(rootState)->second == rootState);
441  targetNodeIt = cleanupResult.nodeMap.find(rootState);
442  verify(targetNodeIt != cleanupResult.nodeMap.end());
443  rootState = (*targetNodeIt).second;
444 
445  targetNodeIt = cleanupResult.nodeMap.find(ciRootState);
446  verify(targetNodeIt != cleanupResult.nodeMap.end());
447  ciRootState = (*targetNodeIt).second;
448  }
449 
450  std::set<StateId> tempCoarticulatedRootNodes;
451  RootTransitDescriptions tempRootTransitDescriptions;
452  for (std::set<StateId>::iterator it = coarticulatedRootStates.begin(); it != coarticulatedRootStates.end(); ++it) {
453  targetNodeIt = cleanupResult.nodeMap.find(*it);
454 
455  if (targetNodeIt != cleanupResult.nodeMap.end()) {
456  tempCoarticulatedRootNodes.insert((*targetNodeIt).second);
457 
458  std::map<StateId, std::pair<Bliss::Phoneme::Id, Bliss::Phoneme::Id>>::iterator oldIt = rootTransitDescriptions.find(*it);
459  verify(oldIt != rootTransitDescriptions.end());
460 
461  verify(tempRootTransitDescriptions.find((*targetNodeIt).second) == tempRootTransitDescriptions.end());
462 
463  tempRootTransitDescriptions.insert(std::make_pair((*targetNodeIt).second, (*oldIt).second));
464  }
465  }
466 
467  if (rootState)
468  tempRootTransitDescriptions.insert(*rootTransitDescriptions.find(rootState)); ///@todo Also add the overall root node to coarticulatedRootNodes, then this is not needed
469 
470  Core::Application::us()->log() << "deleted " << tempCoarticulatedRootNodes.size() - coarticulatedRootStates.size() << " coarticulated root nodes";
471 
472  verify(coarticulatedRootStates.size() == tempCoarticulatedRootNodes.size());
473  verify(rootTransitDescriptions.size() == tempRootTransitDescriptions.size());
474 
475  coarticulatedRootStates.swap(tempCoarticulatedRootNodes);
476  rootTransitDescriptions.swap(tempRootTransitDescriptions);
477 
478  for (std::vector<Exit>::iterator exitIt = exits.begin(); exitIt != exits.end(); ++exitIt) {
479  targetNodeIt = cleanupResult.nodeMap.find(exitIt->transitState);
480  verify(targetNodeIt != cleanupResult.nodeMap.end());
481  exitIt->transitState = (*targetNodeIt).second;
482  }
483 
484  Core::HashMap<TreeIndex, TreeIndex>::const_iterator targetTreeIt;
485 
486  targetTreeIt = cleanupResult.treeMap.find(masterTree);
487  verify(targetTreeIt != cleanupResult.treeMap.end());
488 
489  masterTree = (*targetTreeIt).second;
490 
491  // pushedWordEndNodes = cleanupResult.mapNodes( pushedWordEndNodes );
492  uncoarticulatedWordEndStates = cleanupResult.mapNodes(uncoarticulatedWordEndStates);
493  // uncoarticulatedPushedWordEndNodes = cleanupResult.mapNodes( uncoarticulatedPushedWordEndNodes );
494  unpushedCoarticulatedRootStates = cleanupResult.mapNodes(unpushedCoarticulatedRootStates);
495 
496  return cleanupResult;
497 }
