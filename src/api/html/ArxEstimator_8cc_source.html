22 void ArxEstimator::convolution() {
23  //C_tilde and F_tilde does not contain the first (1) element !
24 
25  CF_tilde_.resize(F_tilde_.size() + C_tilde_.size());
26 
27  u8 t, tau;
28 
29  for (t = 0; t < F_tilde_.size() + C_tilde_.size(); t++)
30  CF_tilde_[t] = 0.0;
31 
32  for (t = 0; t < F_tilde_.size(); t++) {
33  CF_tilde_[t] += F_tilde_[t];
34  for (tau = 0; tau < t && tau < C_tilde_.size(); tau++)
35  CF_tilde_[t] += C_tilde_[tau] * F_tilde_[t - tau - 1];
36  }
37 
38  for (t = F_tilde_.size(); t < F_tilde_.size() + C_tilde_.size(); t++) {
39  CF_tilde_[t - F_tilde_.size()] += C_tilde_[t - F_tilde_.size()];
40  for (tau = t - F_tilde_.size(); tau < t && tau < C_tilde_.size(); tau++)
41  CF_tilde_[t] += C_tilde_[tau] * F_tilde_[t - tau - 1];
42  }
43 }

45 bool ArxEstimator::prepare(const std::vector<_float>* u,
46  const std::vector<_float>* y,
47  const std::vector<_float>* y0) {
48  u32 nr_sample = y->size();
49  u32 i;
50 
51  const std::vector<_float>* u_work = u;
52  const std::vector<_float>* y_work = y;
53 
54  if (D_tilde_.size() || CF_tilde_.size()) {
55  if (u_ == 0 || y_ == 0 || y0_ == 0)
56  return false;
57 
58  u_work = u_;
59  y_work = y_;
60 
61  // D / FC * u[t]
62  if (u) {
63  u_work = u_;
64 
65  for (u32 t = 0; t < nr_sample; t++) {
66  (*u_)[t] = (*u)[t];
67 
68  for (i = 1; i <= D_tilde_.size(); i++)
69  (*u_)[t] += D_tilde_[i - 1] * (t >= i ? (*u)[t - i] : 0.0);
70 
71  for (i = 1; i <= CF_tilde_.size(); i++)
72  (*u_)[t] -= CF_tilde_[i - 1] * (t >= i ? (*u_)[t - i] : 0.0);
73  }
74  }
75 
76  // D / C * y0[t]
77  if (y0) {
78  for (u32 t = 0; t < nr_sample; t++) {
79  (*y0_)[t] = 0.0;
80 
81  //input for (D/AC)*y0
82  for (i = t + 1; i <= F_tilde_.size(); i++)
83  (*y0_)[t] -= F_tilde_[i - 1] * (*y0)[F_tilde_.size() - (i - t)];
84 
85  for (i = 1; i <= D_tilde_.size(); i++)
86  (*y0_)[t] += D_tilde_[i - 1] * (t >= i ? (*y0_)[t - i] : 0.0);
87 
88  for (i = 1; i <= CF_tilde_.size(); i++)
89  (*y0_)[t] -= CF_tilde_[i - 1] * (t >= i ? (*y0_)[t - i] : 0.0);
90  }
91  }
92 
93  // D / FC * y[t]
94  for (u32 t = 0; t < nr_sample; t++) {
95  (*y_)[t] = (*y)[t];
96 
97  for (i = 1; i <= D_tilde_.size(); i++)
98  (*y_)[t] += D_tilde_[i - 1] * (t >= i ? (*y)[t - i] : 0.0);
99 
100  for (i = 1; i <= CF_tilde_.size(); i++)
101  (*y_)[t] -= CF_tilde_[i - 1] * (t >= i ? (*y_)[t - i] : 0.0);
102  }
103  }
104 
105  return least_squares_builder_.work(u_work, y_work, y0);
106 }

108 bool ArxEstimator::work(const std::vector<_float>* u,
109  const std::vector<_float>* y,
110  const std::vector<_float>* y0,
111  _float* estimation_error,
112  std::vector<_float>* B_tilde, std::vector<_float>* A_tilde) {
113  if (y == 0 || (order_B_ > 0 && u == 0) || (y0 != 0 && y0->size() < F_tilde_.size()))
114  return false;
115 
116  std::vector<_float> theta;
117 
118  if ((need_init_ && !init()) ||
119  !prepare(u, y, y0) ||
120  !least_squares_.work(estimation_error, B_tilde || A_tilde ? &theta : 0))
121  return false;
122 
123  u8 i;
124  u8 k = 0;
125  if (order_B_ && B_tilde) {
126  B_tilde->resize(order_B_);
127  for (i = 0; i < order_B_; i++)
128  (*B_tilde)[i] = theta[k++];
129  }
130  else
131  k += order_B_;
132 
133  if (order_A_ && A_tilde) {
134  A_tilde->resize(order_A_);
135  for (i = 0; i < order_A_; i++)
136  (*A_tilde)[i] = theta[k++];
137  }
138 
139  return true;
140 }
