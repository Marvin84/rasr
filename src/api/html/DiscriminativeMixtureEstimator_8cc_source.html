34 void DiscriminativeMixtureEstimator::removeDensitiesWithLowWeight(Weight minObservationWeight, Weight minRelativeWeight) {
35  DensityIndex densityMax = densityIndexWithMaxWeight();
36  Weight minWeight = minObservationWeight;
37  for (DensityIndex dns = 0; dns < densityEstimators_.size();) {
38  bool hasEnoughWeight = weights_[dns] >= minWeight;
39  bool hasEnoughRelativeWeight = previousMixtureWeight(dns) >= minRelativeWeight;
40  bool isDeletable = dns != densityMax;
41  if ((!hasEnoughWeight || !hasEnoughRelativeWeight) && isDeletable) {
42  removeDensity(dns);
43  --densityMax;
44  }
45  else {
46  ++dns;
47  }
48  }
