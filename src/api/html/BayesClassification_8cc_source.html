121 bool BayesClassification::getScores(Flow::Vector<Score>& scores) {
122  if (needInit_)
123  init(0);
124  if (!newData_)
125  return false;
126  scores.resize(nClasses());
127  scores.setStartTime(firstStartTime_);
128  scores.setEndTime(lastEndTime_);
129  newData_ = false;
130  for (u32 classIndex = 0; classIndex < nClasses(); ++classIndex)
131  scores[classIndex] = (*aprioriProbability_)[classIndex] + (*likelihoodFunction_)[classIndex];
132  return true;
133 }

135 const std::string& BayesClassification::argMin() const {
136  verify(nClasses() > 0);
137 
138  Score minScore = Core::Type<Score>::max;
139  u32 minClassIndex = Core::Type<u32>::max;
140  Core::Statistics<Score> statistics("score-statistics");
141 
142  for (u32 classIndex = 0; classIndex < nClasses(); ++classIndex) {
143  Score score = (*aprioriProbability_)[classIndex];
144  if (useSlidingWindow_) {
145  for (SlidingWindow<ScoreVector>::ConstantIterator i = scoreWindow_.begin(); i != scoreWindow_.end(); ++i)
146  score += (*i)[classIndex];
147  }
148  else {
149  score += (*likelihoodFunction_)[classIndex];
150  }
151  if (statisticsChannel_.isOpen())
152  statistics += score;
153  if (score < minScore) {
154  minScore = score;
155  minClassIndex = classIndex;
156  }
157  }
158  if (statisticsChannel_.isOpen())
159  writeStatistics(statisticsChannel_, minClassIndex, statistics);
160  return classLabels_[minClassIndex];
161 }

365  if (nInputs() >= 2) {

384 bool BayesClassificationNode::work(Flow::PortId p) {
385  Flow::DataPtr<Flow::Vector<f32>> featureVector;
386  Flow::DataPtr<Flow::String> classLabel(new Flow::String);
387  do {
388  if (!getData(0, featureVector)) {
389  // end of stream reached
390  if (classify(*classLabel))
391  // send the recognized class labe
392  putData(0, classLabel.get());
393  // pass the EOS
394  return putData(0, featureVector.get());
395  }
396  } while (!classify(*classLabel, *featureVector, featureScoreWeight(*featureVector)));
397 
398  if (!needMoreFeatureVectors()) {
399  // features for classification have allready been read
400  // read from stream until end to get the correct end time
401  Time endTime = classLabel->endTime();
402  while (getData(0, featureVector))
403  endTime = featureVector->endTime();
404  classLabel->setEndTime(endTime);
405  return putData(0, classLabel.get()) && putData(0, featureVector.get());
406  }
407  else {
408  // continuous classification, delay <= nFeatures_
409  return putData(0, classLabel.get());
410  }
411 }
