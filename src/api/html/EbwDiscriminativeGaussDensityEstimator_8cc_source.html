73 Mean* EbwDiscriminativeMeanEstimator::estimate() {
74  Weight weight = dtWeight() + iterationConstant();
75  if (weight > 0) {
76  std::vector<Sum> buffer(accumulator_.size());
77  const std::vector<Accumulator::SumType> dSum = dtSum();
78  std::transform(dSum.begin(), dSum.end(), previousMean().begin(),
79  buffer.begin(), plusWeighted<Sum>(iterationConstant()));
80  Mean* result = new Mean(accumulator_.size());
81  std::transform(buffer.begin(), buffer.end(), result->begin(),
82  std::bind2nd(std::divides<Accumulator::SumType>(), weight));
83  return result;
84  }
85  else {
86  Mean* result = new Mean(previousMean());
87  return result;
88  }
89 }

165 Covariance* EbwDiscriminativeCovarianceEstimator::estimate(const CovarianceToMeanSetMap& meanSetMap,
166  VarianceType minimumVariance) {
167  const ComponentIndex dimension = accumulator_.size();
168  const CovarianceToMeanSetMap::MeanSet& meanSet = meanSetMap[Core::Ref<AbstractCovarianceEstimator>(this)];
169  std::vector<Sum> buffer = dtSum(meanSet);
170  Weight weight = 0;
171  CovarianceToMeanSetMap::MeanSet::const_iterator m = meanSet.begin();
172  for (; m != meanSet.end(); ++m) {
173  const Core::Ref<EbwDiscriminativeMeanEstimator> meanEstimator(
174  required_cast(EbwDiscriminativeMeanEstimator*, m->get()));
175 
176  Weight tmpWeight = meanEstimator->dtWeight() + meanEstimator->iterationConstant();
177  weight += tmpWeight;
178 
179  const Mean* mean = getMean(meanEstimator);
180  for (ComponentIndex i = 0; i < dimension; ++i) {
181  Sum previousSum = previousCovariance(i);
182  previousSum += meanEstimator->previousMean(i) * meanEstimator->previousMean(i);
183  previousSum *= meanEstimator->iterationConstant();
184  Sum newSum = tmpWeight * (*mean)[i] * (*mean)[i];
185  buffer[i] += previousSum - newSum;
186  }
187  delete mean;
188  }
189  verify(weight > 0);
190  std::transform(buffer.begin(), buffer.end(),
191  buffer.begin(), std::bind2nd(std::divides<Sum>(), weight));
192  std::vector<VarianceType> buffer32(dimension);
193  std::copy(buffer.begin(), buffer.end(), buffer32.begin());
194  DiagonalCovariance* result = new DiagonalCovariance(buffer32);
195  verify(result->isPositiveDefinite());
196  applyMinimumVariance(minimumVariance, *result);
197  return result;
198 }
