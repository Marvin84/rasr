21 /**
22  * Calculates the composition result of two input transducers
23  * using the matching epsilon filter. This is not guaranteed to do
24  * the same than what the AT&T FSM library matching composition
25  * filter does although we haven't seen a case yet where results
26  * were different.
27  * @param fl "left" automaton
28  * @param fr "right" automaton
29  * @param reportUnknowns tell the user (on stdout) wether mapping
30  * of alphabet symbols reveiled unknowns in the "right" automaton
31  * @return the result automaton of the composition (an acceptor in
32  * the case of two input acceptors, a transducer in all other
33  * cases)
34  **/
35 template<class _Automaton>
36 typename _Automaton::ConstRef composeMatching(typename _Automaton::ConstRef fl, typename _Automaton::ConstRef fr, bool reportUnknowns = true);
