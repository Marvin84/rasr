76  out += Core::form("%f ", interpolation_weights_[i]);

78  case posteriorCombInvEntropy:
79  out += "inverse-entropy combination of input streams (need to be normalized posteriors!)";

83  out += Core::form("gamma=%f", gamma_);
84  break;

86  this->log(out.c_str());

137 /** Forward the input */
138 template<typename T>
139 void OperationLayer<T>::_forward(const std::vector<NnMatrix*>& input, NnMatrix& output, bool reset) {
140  // boundary check
141  require_eq(Precursor::inputDimensions_.size(), interpolation_weights_.size());
142  timeval start, end;
143 
144  gettimeofday(&start, NULL);
145  NnVector norm;
146  NnVector* weight;
147  T Hmax;
148  output.fill(0);
149 
150  switch (operation_) {
151  case streamLinearCombine:
152  for (u32 stream = 0; stream < Precursor::inputDimensions_.size(); stream++) {
153  output.add(*(input.at(stream)), (T)interpolation_weights_[stream]);
154  }
155  break;
156 
157  case posteriorCombInvEntropy:
158  if (frame_weights_.size() != Precursor::inputDimensions_.size()) {
159  frame_weights_.resize(Precursor::inputDimensions_.size());
160  for (u32 stream = 0; stream < Precursor::inputDimensions_.size(); stream++) {
161  frame_weights_[stream] = new NnVector(input.at(stream)->nColumns());
162  }
163  }
164  for (u32 stream = 0; stream < Precursor::inputDimensions_.size(); stream++) {
165  weight = frame_weights_[stream];
166  weight->initComputation(false);
167  weight->columnEntropy(*(input.at(stream)));
168  weight->pow(-1.0);
169  }
170  norm.resize(input.at(0)->nColumns());
171  norm.initComputation(false);
172  norm.setToZero();
173  for (u32 stream = 0; stream < Precursor::inputDimensions_.size(); stream++) {
174  norm.add(*(frame_weights_[stream]));
175  output.addWithColumnWeights(*(input.at(stream)), *(frame_weights_[stream]));
176  }
177  output.divideColumnsByScalars(norm);
178  if (applyLog_)
179  output.log();
180 
181  break;
182 
183  case posteriorCombDS:
184  require_eq(Precursor::inputDimensions_.size(), 2);
185  if (frame_weights_.size() != Precursor::inputDimensions_.size()) {
186  frame_weights_.resize(Precursor::inputDimensions_.size());
187  for (u32 stream = 0; stream < Precursor::inputDimensions_.size(); stream++) {
188  frame_weights_[stream] = new NnVector(input.at(stream)->nColumns());
189  }
190  }
191  Hmax = std::log(input.at(0)->nRows());
192  for (u32 stream = 0; stream < Precursor::inputDimensions_.size(); stream++) {
193  weight = frame_weights_[stream];
194  weight->initComputation(false);
195  weight->columnEntropy(*(input.at(stream)));
196  weight->scale(-1.0 / Hmax);
197  weight->addConstantElementwise(1.0);
198  weight->pow(gamma_);
199  }
200  input.at(0)->finishComputation(false);
201  NnMatrix tmp(*input.at(0));
202  input.at(0)->initComputation(false);
203  tmp.initComputation(false);
204  tmp.elementwiseMultiplication(*(input.at(1)));
205  tmp.add(*(input.at(0)), T(-1.0));
206  tmp.add(*(input.at(1)), T(-1.0));
207  tmp.multiplyColumnsByScalars(*(frame_weights_[0]));
208  tmp.multiplyColumnsByScalars(*(frame_weights_[1]));
209 
210  for (u32 stream = 0; stream < Precursor::inputDimensions_.size(); stream++) {
211  output.addWithColumnWeights(*(input.at(stream)), *(frame_weights_[stream]));
212  }
213  output.add(tmp);
214  if (applyLog_) {
215  output.ensureMinimalValue(1e-20);
216  output.log();
217  }
218  break;
219  }
220 
221  Math::Cuda::deviceSync(Precursor::measureTime_ && Math::CudaDataStructure::hasGpu());
222  gettimeofday(&end, NULL);
223  timeForwardLinear_ += Core::timeDiff(start, end);
224 }
