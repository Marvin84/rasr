60 void AffineFeatureTransformEstimator::postProcess() {
61  std::string outputDirectory = paramTransformDirectory(config);
62  if (paramEstimationIterations(config) > 0)
63  Core::createDirectory(outputDirectory);
64 
65  bool initialDirectory(paramInitialTransformDirectory(config) != "");
66  bool initialFile(paramInitialTransform(config) != "");
67 
68  ConcreteAccumulator::Transform initialTransform;
69 
70  if (initialDirectory && initialFile)
71  warning("Value for 'initial-transform-directory' overrides value for 'initial-transform'.");
72  else if (initialFile)
73  if (!Math::Module::instance().formats().read(paramInitialTransform(config), initialTransform))
74  error("Failed to read matrix from file '%s'.",
75  paramInitialTransform(config).c_str());
76 
77  std::set<std::string> seenKeys = accumulatorCache_.keys();
78  log("number of keys: %zu", seenKeys.size());
79 
80  for (std::set<std::string>::iterator key = seenKeys.begin(); key != seenKeys.end(); ++key) {
81  /*! @todo this is a more or less diry hack.
82  what we want to do here is to modify the object
83  without writing the modification back to cache
84  */
85  currentAccumulator_ = const_cast<Accumulator*>(accumulatorCache_.findForReadAccess(*key));
86  verify(currentAccumulator_);
87 
88  ConcreteAccumulator* concreteCurrentAccumulator =
89  dynamic_cast<ConcreteAccumulator*>(&**currentAccumulator_);
90 
91  concreteCurrentAccumulator->finalize();
92  if (paramEstimationIterations(config) > 0) {
93  ConcreteAccumulator::Transform transform;
94  if (initialDirectory) {
95  if (!Math::Module::instance().formats().read(Core::joinPaths(
96  paramInitialTransformDirectory(config),
97  *key + transformExtension()),
98  initialTransform)) {
99  error("Failed to read matrix from file '%s'.",
100  Core::joinPaths(paramInitialTransformDirectory(config),
101  *key + transformExtension())
102  .c_str());
103  initialTransform.clear();
104  }
105  transform =
106  concreteCurrentAccumulator->estimate(paramEstimationIterations(config),
107  paramMinObservationWeight(config),
108  static_cast<ConcreteAccumulator::Criterion>(paramOptimizationCriterion(config)),
109  initialTransform);
110  }
111  else if (initialFile) {
112  transform =
113  concreteCurrentAccumulator->estimate(paramEstimationIterations(config),
114  paramMinObservationWeight(config),
115  static_cast<ConcreteAccumulator::Criterion>(paramOptimizationCriterion(config)),
116  initialTransform);
117  }
118  else {
119  transform =
120  concreteCurrentAccumulator->estimate(paramEstimationIterations(config),
121  paramMinObservationWeight(config),
122  static_cast<ConcreteAccumulator::Criterion>(paramOptimizationCriterion(config)));
123  }
124  std::string filename = Core::joinPaths(outputDirectory, *key + transformExtension());
125  if (!Core::createDirectory(Core::directoryName(filename))) {
126  error("Cannot create directory '%s'", Core::directoryName(filename).c_str());
127  }
128  Core::XmlOutputStream os(filename);
129  if (!os.good()) {
130  error("Cannot write to file '%s'", filename.c_str());
131  }
132  os << transform;
133  }
134  concreteCurrentAccumulator->compact();
135  }
136 }

138 void AffineFeatureTransformEstimator::scoreTransforms() {
139  std::string transformDirectory = paramTransformDirectory(config);
140 
141  std::vector<std::pair<std::string, ConcreteAccumulator::Transform>> transforms;
142 
143  Core::DirectoryFileIterator::FileExtensionFilter filter(transformExtension());
144 
145  for (Core::DirectoryFileIterator file(transformDirectory, &filter); file; ++file) {
146  ConcreteAccumulator::Transform transform;
147  Math::Module::instance().formats().read(Core::joinPaths(transformDirectory, file.path()), transform);
148  std::string fileKey(file.path().substr(0, file.path().size() - std::string(transformExtension()).size()));
149  transforms.push_back(std::make_pair(fileKey, transform));
150  }
151 
152  std::set<std::string> seenKeys = accumulatorCache_.keys();
153  for (std::set<std::string>::iterator key = seenKeys.begin(); key != seenKeys.end(); ++key) {
154  currentAccumulator_ = const_cast<Accumulator*>(accumulatorCache_.findForReadAccess(*key));
155  verify(currentAccumulator_);
156 
157  ConcreteAccumulator* concreteCurrentAccumulator =
158  dynamic_cast<ConcreteAccumulator*>(&**currentAccumulator_);
159 
160  concreteCurrentAccumulator->finalize();
161 
162  for (size_t i = 0; i < transforms.size(); i++) {
163  // TODO: Nicer logging!
164  Mm::Sum score = concreteCurrentAccumulator->score(transforms[i].second,
165  static_cast<ConcreteAccumulator::Criterion>(paramOptimizationCriterion(config)));
166  std::cerr << *key << " " << transforms[i].first << " " << score << std::endl;
167  log("Score, corpus-key=\"%s\" transform=\"%s\": %f", key->c_str(), transforms[i].first.c_str(), score);
168  }
