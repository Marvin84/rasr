65 Operation::AutomatonRef BuildGrammar::process() {
66  Fsa::ConstAutomatonRef fsaG = resources_.languageModel()->getFsa();
67  Fsa::AutomatonCounts cnt = Fsa::count(fsaG);
68  log("G synt: %d states %zu arcs", cnt.nStates_, cnt.nArcs_);
69  Lm::Token se = resources_.languageModel()->unscaled()->sentenceEndToken();
70  const Fsa::LabelId sentenceEnd = se->id();
71  Lm::Token sb = resources_.languageModel()->unscaled()->sentenceBeginToken();
72  const Fsa::LabelId sentenceBegin = sb->id();
73  resources_.deleteLanguageModel();
74  bool addSentenceBegin = paramAddSentenceBegin(Operation::config);
75  bool addSentenceEnd = paramAddSentenceEnd(Operation::config);
76  log("add sentence begin: %d, add sentence end: %d", addSentenceBegin, addSentenceEnd);
77  if (paramAddSentenceBoundaries(Operation::config))
78  addSentenceBegin = addSentenceEnd = true;
79  if (addSentenceBegin || addSentenceEnd) {
80  log("adding sentence boundaries");
81  Core::Ref<Fsa::StaticAutomaton> sg = Fsa::staticCopy(fsaG);
82  fsaG.reset();
83  addSentenceBoundaries(sg, sentenceBegin, sentenceEnd,
84  addSentenceBegin, addSentenceEnd);
85  fsaG = sg;
86  }
87 
88  u32 nDisambiguators = countDisambiguators(fsaG->getInputAlphabet());
89  AutomatonRef g = OpenFst::convertFromFsa<Fsa::Automaton, Automaton>(fsaG);
90  fsaG.reset();
91 
92  if (outputType() != outputSyntacticTokens) {
93  g = mapSymbols(g);
94  }
95  g->setAttribute(Automaton::attrNumDisambiguators, nDisambiguators);
96  return g;
97 }

99 Operation::AutomatonRef BuildGrammar::mapSymbols(AutomatonRef g) const {
100  u32 nDisambiguators = 0;
101  require(outputType() != outputSyntacticTokens);
102  bool addEps = paramAddEmptySyntacticTokens(Operation::config);
103  log("creating lemma to syntactic token transducer ")
104  << (addEps ? "with" : "without") << "empty syntactic tokens";
105  Core::Ref<const Bliss::Lexicon> lexicon = resources_.lexicon();
106  FstLib::StdProjectFst * result = 0, *intermediate = 0;
107  OpenFst::VectorFst * l2s = 0, *lp2l = 0;
108 
109  l2s = OpenFst::convertFromFsa(lexicon->createLemmaToSyntacticTokenTransducer(addEps, nDisambiguators));
110  log("projecting to lemmas");
111  FstLib::ArcSort(l2s, FstLib::StdOLabelCompare());
112  result = new FstLib::StdProjectFst(FstLib::StdComposeFst(*l2s, *g), FstLib::PROJECT_INPUT);
113  if (outputType() == outputLemmaPronunciations) {
114  lp2l = OpenFst::convertFromFsa(lexicon->createLemmaPronunciationToLemmaTransducer(nDisambiguators));
115  log("projecting to lemma pronunciations");
116  Mm::Score pronunciationScale = resources_.pronunciationScale();
117  if (pronunciationScale != 1.0) {
118  log("applying pronunciation scale %f", pronunciationScale);
119  OpenFst::scaleWeights(lp2l, pronunciationScale);
120  }
121  intermediate = result;
122  result = new FstLib::StdProjectFst(FstLib::StdComposeFst(*lp2l, *intermediate), FstLib::PROJECT_INPUT);
123  }
124  AutomatonRef staticFst = new Automaton(*result);
125  delete g;
126  delete intermediate;
127  delete l2s;
128  delete lp2l;
129  delete result;
130  return staticFst;
131 }

133 void BuildGrammar::addSentenceBoundaries(Core::Ref<Fsa::StaticAutomaton> g,
134  Fsa::LabelId sentenceBegin,
135  Fsa::LabelId sentenceEnd,
136  bool addBegin, bool addEnd) const {
137  if (addEnd) {
138  Fsa::State* finalState = g->newState();
139  log("new final state: %d", finalState->id());
140  finalState->setFinal(g->semiring()->one());
141  Fsa::StateId final = finalState->id();
142  DBG(1) << "final: " << final << std::endl;
143  for (Fsa::StateId s = 0; s <= g->maxStateId(); ++s) {
144  Fsa::State::Ref state = g->state(s);
145  if (!state) {
146  warning("invalid state id: ") << s;
147  continue;
148  }
149  verify(state);
150  if (!state->isFinal() || s == final)
151  continue;
152  DBG(1) << "final state: " << state->id() << " " << (float)state->weight() << std::endl;
153  Fsa::Weight weight = state->weight();
154  state->newArc(final, weight, sentenceEnd);
155  state->unsetFinal();
156  state->setWeight(g->semiring()->zero());
157  DBG(1) << state->id() << " -> " << final << std::endl;
158  }
159  }
160  if (addBegin) {
161  Fsa::State* initial = g->newState();
162  log("new initial state: %d", initial->id());
163  initial->newArc(g->initialStateId(), g->semiring()->one(), sentenceBegin);
164  g->setInitialStateId(initial->id());
165  }
166  g->unsetProperties(Fsa::PropertySortedByInput);
167 }

280 Operation::AutomatonRef LemmaMapping::process() {
281  Core::Ref<const Bliss::Lexicon> lexicon = resources_.lexicon();
282  MapType mType = static_cast<MapType>(paramMapType(config));
283  nDisambiguators_ = std::max(nDisambiguators_, 0);
284  log("using %d disambiguators", nDisambiguators_);
285  Fsa::ConstAutomatonRef m;
286  switch (mType) {
287  case LemmaPronunciationToLemma:
288  log("lemma pronunciation to lemma mapping");
289  m = lexicon->createLemmaPronunciationToLemmaTransducer(nDisambiguators_);
290  break;
291  case LemmaToSyntacticToken:
292  log("lemma to syntactic token mapping");
293  m = lexicon->createLemmaToSyntacticTokenTransducer(true, nDisambiguators_);
294  break;
295  default:
296  defect();
297  break;
298  }
299  f32 scale = paramScale(config);
300  if (scale != 1.0) {
301  log("applying scale %f", scale);
302  m = Fsa::multiply(m, Fsa::Weight(scale));
303  }
304  AutomatonRef result = OpenFst::convertFromFsa<Fsa::Automaton, Automaton>(m);
305  m.reset();
306  return result;

312 Operation::AutomatonRef AddPronunciationWeight::process() {
313  nDisambiguators_ = std::max(nDisambiguators_, 0);
314  log("using %d disambiguators", nDisambiguators_);
315 
316  AutomatonRef result = new Automaton();
317  OpenFst::StateId state = result->AddState();
318  result->SetFinal(state, OpenFst::Weight::One());
319  result->SetStart(state);
320  Core::Ref<const Bliss::LemmaPronunciationAlphabet> lpa = resources_.lexicon()->lemmaPronunciationAlphabet();
321  Bliss::Lexicon::PronunciationIterator piBegin, piEnd;
322  Core::tie(piBegin, piEnd) = resources_.lexicon()->pronunciations();
323  for (Bliss::Lexicon::PronunciationIterator pi = piBegin; pi != piEnd; ++pi) {
324  Bliss::Pronunciation::LemmaIterator liBegin, liEnd;
325  Core::tie(liBegin, liEnd) = (*pi)->lemmas();
326  for (Bliss::Pronunciation::LemmaIterator li = liBegin; li != liEnd; ++li) {
327  const Bliss::LemmaPronunciation* lemmaPron = li;
328  OpenFst::Weight weight = OpenFst::Weight(lemmaPron->pronunciationScore());
329  OpenFst::Label label = OpenFst::convertLabelFromFsa(lemmaPron->id());
330  result->AddArc(state, OpenFst::Arc(label, label, weight, state));
331  }
332  }
333  for (s32 d = 0; d < nDisambiguators_; ++d) {
334  OpenFst::Label label = OpenFst::convertLabelFromFsa(lpa->disambiguator(d));
335  result->AddArc(state, OpenFst::Arc(label, label, OpenFst::Weight::One(), state));
336  }
337  OpenFst::SymbolTable* symbols = OpenFst::convertAlphabet(lpa, "lemma-pronunciations");
338  result->SetInputSymbols(symbols);
339  result->SetOutputSymbols(symbols);
340  FstLib::ArcSort(result, FstLib::StdILabelCompare());
341  f32 scale = paramScale(config);
342  if (scale != 1.0) {
343  log("applying scale %f", scale);
344  OpenFst::scaleWeights(result, scale);
345  }
346  return result;
347 }

442 void AddNonWordTokens::renormalizeWeights(OpenFst::StateId s) {
443  typedef FstLib::LogWeight LogWeight;
444  LogWeight sum = LogWeight::Zero();
445  for (OpenFst::ArcIterator aiter(*input_, s); !aiter.Done(); aiter.Next())
446  sum = FstLib::Plus(sum, LogWeight(aiter.Value().weight.Value()));
447  for (OpenFst::MutableArcIterator aiter(input_, s); !aiter.Done(); aiter.Next()) {
448  OpenFst::Arc arc = aiter.Value();
449  arc.weight = OpenFst::Weight(FstLib::Divide(LogWeight(arc.weight.Value()), sum).Value());
450  aiter.SetValue(arc);
451  }

454 OpenFst::Label AddNonWordTokens::getLabel(const Bliss::Lemma* lemma) const {
455  Fsa::LabelId label = Fsa::InvalidLabelId;
456  switch (outputType()) {
457  case outputLemmas: {
458  label = lemma->id();
459  } break;
460  case outputLemmaPronunciations: {
461  verify(lemma->nPronunciations() == 1);
462  Bliss::Lemma::LemmaPronunciationRange pron = lemma->pronunciations();
463  label = resources_.lexicon()->lemmaPronunciationAlphabet()->index(
464  pron.first);
465  } break;
466  case outputSyntacticTokens: {
467  const Bliss::SyntacticTokenSequence& tokens = lemma->syntacticTokenSequence();
468  if (tokens.isEpsilon()) {
469  warning("adding empty syntactic token for %s", lemma->symbol().str());
470  label = Fsa::Epsilon;
471  }
472  else {
473  verify(tokens.length() == 1);
474  label = (*tokens.begin())->id();
475  }
476  } break;
477  default:
478  defect();
479  }
480  return OpenFst::convertLabelFromFsa(label);
481 }

510 OpenFst::StateId AddNonWordTokens::getUnigramState() const {
511  OpenFst::StateId state = input_->Start();
512  OpenFst::StateId prevState = state;
513  while (input_->NumInputEpsilons(state)) {
514  for (OpenFst::ArcIterator aiter(*input_, state); !aiter.Done(); aiter.Next()) {
515  const OpenFst::Arc& arc = aiter.Value();
516  if (arc.ilabel == OpenFst::Epsilon) {
517  state = arc.nextstate;
518  break;
519  }
520  }
521  if (state == prevState) {
522  error("epsilon loop found");
523  return OpenFst::InvalidStateId;
524  }
525  prevState = state;
526  }
527  return state;
528 }

572 Operation::AutomatonRef RemoveEmptyPath::process() {
573  log("removing empty path");
574  bool moreThanOne = false;
575  OpenFst::StateId final = OpenFst::findFinalState(*input_, &moreThanOne);
576  if (moreThanOne)
577  error("expected only one final state");
578  if (final == OpenFst::InvalidStateId)
579  error("no final state found");

582  OpenFst::StateId firstUnigram = input_->AddState();
583  log("new state: %d", firstUnigram);
584 
585  // modify arcs of initial state:
586  // - remove epsilon transition to final state
587  // - redirect epsilon arc to new "first unigram state"
588  std::vector<OpenFst::Arc> newArcs;
589  newArcs.reserve(input_->NumArcs(state));
590  OpenFst::Arc backoffArc;
591  backoffArc.nextstate = OpenFst::InvalidStateId;

593  for (OpenFst::MutableArcIterator aiter(input_, state); !aiter.Done(); aiter.Next()) {
594  const OpenFst::Arc& arc = aiter.Value();
595  if (arc.nextstate != final) {
596  if (arc.ilabel == OpenFst::Epsilon)
597  backoffArc = arc;
598  else
599  newArcs.push_back(arc);
600  }
601  else {
602  verify(arc.ilabel == OpenFst::Epsilon);
603  }
604  }
605  input_->DeleteArcs(state);
606  verify(backoffArc.nextstate != OpenFst::InvalidStateId);
607  unigram = backoffArc.nextstate;
608  log("unigram state: %d", unigram);
609  backoffArc.nextstate = firstUnigram;
610  input_->AddArc(state, backoffArc);
611  OpenFst::addArcs(input_, state, newArcs);
612 
613  // add arcs from unigram state to "first unigram state"
614  // except for the epsilon arc to the final state (sentence end)
615  verify(input_->NumInputEpsilons(unigram) <= 1);
616  for (OpenFst::MutableArcIterator aiter(input_, unigram); !aiter.Done(); aiter.Next()) {
617  const OpenFst::Arc& arc = aiter.Value();
618  if (arc.nextstate != final) {
619  verify(arc.ilabel != OpenFst::Epsilon);
620  input_->AddArc(firstUnigram, arc);
621  }
622  }
623  return input_;

684  const std::string subwordFile = paramSubwordList(Operation::config);
685  log("reading subword list: %s", subwordFile.c_str());
686  if (!readSubwordList(subwordFile)) {
687  error("cannot read subword list");

691  log("%zd subword tokens", subwordTokens_.size());
692  }
693  const OpenFst::SymbolTable* symbols = input_->OutputSymbols();
694  if (!symbols) {
695  error("symbol table is required");
696  return input_;
697  }
698  OpenFst::StateId oldStart = input_->Start();
699  OpenFst::StateId newStart = input_->AddState();
700  input_->SetStart(newStart);
701  input_->SetFinal(newStart, OpenFst::Weight::One());
702  Core::Ref<const Fsa::Alphabet> alphabet;
703  switch (outputType()) {
704  case outputSyntacticTokens:
705  alphabet = resources_.lexicon()->syntacticTokenAlphabet();
706  break;
707  case outputLemmas:
708  alphabet = resources_.lexicon()->lemmaAlphabet();
709  break;
710  case outputLemmaPronunciations:
711  alphabet = resources_.lexicon()->lemmaPronunciationAlphabet();
712  break;
713  }
714  const std::string transitionSymbol = paramTransitionSymbol(Operation::config);
715  Fsa::LabelId transitionLabel = alphabet->index(transitionSymbol);
716  if (transitionLabel == Fsa::InvalidLabelId) {
717  error("unknown transition symbol '%s'", transitionSymbol.c_str());
718  return input_;
719  }
720 
721  for (Fsa::Alphabet::const_iterator t = alphabet->begin(); t != alphabet->end(); ++t) {
722  Fsa::LabelId id = t;
723  const std::string symbol = *t;
724  const bool isSubword = subwordTokens_.count(id);
725  if (!isSubword && !symbol.empty() && id != transitionLabel) {
726  OpenFst::Label label = OpenFst::convertLabelFromFsa(id);
727  if (input_->InputSymbols() && input_->InputSymbols()->Find(label).empty())
728  warning("symbol not in symbol table: %s %d", symbol.c_str(), label);
729  input_->AddArc(newStart, OpenFst::Arc(label, label, OpenFst::Weight::One(), newStart));
730  }
731  }
732  for (OpenFst::StateIterator siter(*input_); !siter.Done(); siter.Next()) {
733  OpenFst::StateId state = siter.Value();
734  if (state == newStart)
735  continue;
736  if (OpenFst::isFinalState(*input_, state)) {
737  input_->AddArc(state, OpenFst::Arc(OpenFst::Epsilon, OpenFst::Epsilon, OpenFst::Weight::One(), newStart));
738  input_->SetFinal(state, OpenFst::Weight::Zero());
739  }
740  for (OpenFst::MutableArcIterator aiter(input_, state); !aiter.Done(); aiter.Next()) {
741  OpenFst::Arc arc = aiter.Value();
742  arc.ilabel = OpenFst::Epsilon;
743  aiter.SetValue(arc);
744  }
745  }
746 
747  OpenFst::Label l = OpenFst::convertLabelFromFsa(transitionLabel);
748  log("using transition label '%s' %d", transitionSymbol.c_str(), l);
749  input_->AddArc(newStart, OpenFst::Arc(l, OpenFst::Epsilon, OpenFst::Weight::One(), oldStart));
750  return input_;
751 }

850  std::string stateSequencesFile = paramStateSequences(config);
851  log("loading state sequences: %s", stateSequencesFile.c_str());
852  StateSequenceList ssl;
853  if (!ssl.read(stateSequencesFile))
854  criticalError("cannot read state sequences");
855  log("# state sequences: %d", u32(ssl.size()));
856  u32 nSpecialTokens = numSpecialSymbols();
857  u32 specialTokenOffset = ssl.size() - nSpecialTokens;
858  TiedStateSequenceMap newLabels;
859 
860  OpenFst::InDegree<OpenFst::Arc> inDegree(*input_);
861  std::stack<OpenFst::StateId> queue;
862  queue.push(input_->Start());
863  std::vector<bool> visited(input_->NumStates(), false);
864  visited[queue.top()] = true;
865  while (!queue.empty()) {
866  OpenFst::StateId s = queue.top();
867  queue.pop();
868  for (OpenFst::MutableArcIterator ai(input_, s); !ai.Done(); ai.Next()) {
869  OpenFst::Arc arc = ai.Value();
870  StateSequence seq;
871  s32 seqIndex = -1;
872  if (arc.ilabel != OpenFst::Epsilon) {
873  seqIndex = OpenFst::convertLabelToFsa(arc.ilabel);
874  seq = ssl[seqIndex];
875  }
876  OpenFst::StateId ns = arc.nextstate;
877  OpenFst::Label output = arc.olabel;
878  OpenFst::Label newOutput = output;
879  OpenFst::Weight weight = arc.weight;
880  while (inDegree[ns] == 1 && !seq.isFinal() && seqIndex < specialTokenOffset) {
881  if (OpenFst::isFinalState(*input_, ns))
882  break;
883  if (input_->NumArcs(ns) != 1)
884  break;
885  const OpenFst::Arc nextArc = OpenFst::ArcIterator(*input_, ns).Value();
886  if (nextArc.olabel != OpenFst::Epsilon) {
887  if (newOutput != OpenFst::Epsilon)
888  break;
889  else
890  newOutput = nextArc.olabel;
891  }
892 
893  if (nextArc.ilabel != OpenFst::Epsilon) {
894  u32 nextSeqIndex = OpenFst::convertLabelToFsa(nextArc.ilabel);
895  if (nextSeqIndex >= specialTokenOffset)
896  break;
897  StateSequence nextSeq = ssl[nextSeqIndex];
898  if (nextSeq.isFinal())
899  seq.setFinal();
900  if (nextSeq.isInitial()) {
901  seq.setInitial();
902  verify_eq(seq.nStates(), 0);
903  }
904  for (u32 si = 0; si < nextSeq.nStates(); ++si)
905  seq.appendState(nextSeq.state(si).emission_, nextSeq.state(si).transition_);
906  }
907  output = newOutput;
908  weight = FstLib::Times(weight, nextArc.weight);
909  ns = nextArc.nextstate;
910  }
911  arc.nextstate = ns;
912  if (seq.nStates()) {
913  if (seqIndex < specialTokenOffset)
914  arc.ilabel = OpenFst::convertLabelFromFsa(newLabels.index(seq));
915  else
916  arc.ilabel = -1 - (seqIndex - specialTokenOffset);
917  }
918  arc.olabel = output;
919  ai.SetValue(arc);
920  if (!visited[ns]) {
921  visited[ns] = true;
922  queue.push(ns);
923  }
924  }
925  }
926  FstLib::Connect(input_);
927 
928  StateSequenceList newList;
929  newLabels.createStateSequenceList(newList);
930  OpenFst::LabelMapping labelMapping;
931  if (addNonWords_) {
932  // add special sequences to the end of the list
933  s32 l = -1;
934  for (u32 i = specialTokenOffset; i < ssl.size(); ++i, --l) {
935  newList.push_back(ssl[i]);
936  labelMapping.push_back(std::make_pair(l, newList.size()));
937  }
938  }
939  if (!labelMapping.empty())
940  FstLib::Relabel(input_, labelMapping, OpenFst::LabelMapping());
941  stateSequencesFile = paramNewStateSequences(config);
942  log("writing %d state sequences: %s", u32(newList.size()), stateSequencesFile.c_str());
943  newList.write(stateSequencesFile);
944  return input_;
945 }
