53 /**
54  * Distribute recording time over segments:
55  * In case of overlaps, the begining of the next next segment is cut
56  **/
57 BoundedSegmentList buildBoundedSegments(const RecordingSegmentation& recSeg) {
58  BoundedSegmentList boundedSegs(recSeg.size());
59  f32 recStartTime = recSeg.desc->startTime(), recEndTime = recSeg.desc->endTime();
60  switch (recSeg.size()) {
61  case 0:
62  break;
63  case 1:
64  verify((recStartTime < recSeg.front()->endTime()) && (recSeg.back()->startTime() < recEndTime));
65  boundedSegs[0].segment = recSeg[0];
66  boundedSegs[0].offset = 0;
67  boundedSegs[0].startFrame = -nFrames(recStartTime, recSeg[0]->startTime());
68  boundedSegs[0].endFrame = nFrames(recSeg[0]->startTime(), recEndTime);
69  break;
70  default:
71  verify((recStartTime < recSeg.front()->endTime()) && (recSeg.back()->startTime() < recEndTime));
72  RecordingSegmentation::const_iterator itSegL = recSeg.begin(), itSeg = recSeg.begin() + 1, itSegR = recSeg.begin() + 2;
73  BoundedSegmentList::iterator itBoundedSegL = boundedSegs.begin(), itBoundedSeg = boundedSegs.begin() + 1;
74  itBoundedSegL->segment = *itSegL;
75  itBoundedSegL->offset = 0;
76  itBoundedSegL->startFrame = -nFrames(recStartTime, (*itSegL)->startTime());
77  itBoundedSegL->endFrame = nFrames((*itSegL)->startTime(), (*itSegL)->endTime());
78  for (; itSegR != recSeg.end(); ++itSeg, ++itSegR) {
79  if ((*itSeg)->endTime() <= (*itSegL)->endTime()) {
80  Core::Application::us()->warning(
81  "Segment \"%s\" lies in \"%s\", discard first segment.",
82  (*itSeg)->segmentIdOrDie().c_str(), (*itSegL)->segmentIdOrDie().c_str());
83  }
84  else {
85  itBoundedSeg->segment = *itSeg;
86  itBoundedSeg->offset = itBoundedSegL->offset + (itBoundedSegL->endFrame - itBoundedSegL->startFrame);
87  itBoundedSeg->startFrame = -(nFrames(recStartTime, (*itSeg)->startTime()) - itBoundedSeg->offset);
88  itBoundedSeg->endFrame = nFrames((*itSeg)->startTime(), (*itSeg)->endTime());
89  itSegL = itSeg;
90  ++itBoundedSegL, ++itBoundedSeg;
91  }
92  }
93  if ((*itSeg)->endTime() <= (*itSegL)->endTime()) {
94  Core::Application::us()->warning(
95  "Segment \"%s\" lies in \"%s\", discard first segment.",
96  (*itSeg)->segmentIdOrDie().c_str(), (*itSegL)->segmentIdOrDie().c_str());
97  --itBoundedSeg;
98  }
99  else {
100  itBoundedSeg->segment = *itSeg;
101  itBoundedSeg->offset = itBoundedSegL->offset + (itBoundedSegL->endFrame - itBoundedSegL->startFrame);
102  itBoundedSeg->startFrame = -(nFrames(recStartTime, (*itSeg)->startTime()) - itBoundedSeg->offset);
103  itBoundedSeg->endFrame = nFrames((*itSeg)->startTime(), recEndTime);
104  }
105  boundedSegs.erase(itBoundedSeg + 1, boundedSegs.end());
106  break;
107  }
108  return boundedSegs;
109 }

171  bool init_(NetworkCrawler& crawler, const std::vector<std::string>& arguments) {
172  if (!connected(0) || !connected(1))
173  criticalError("Need segment sources at port 0 and port 1.");
174  if (!crawler.init(in()[0].node, arguments) || !crawler.init(in()[1].node, arguments))
175  return false;
176  NetworkCrawlerRef recSegCrawler = Module::instance().processor()->newCrawler();
177  for (bool _good = true; _good;) {
178  ConstSegmentRef segment = requestSegment(1);
179  if (!recSegSet_.insertRecording(segment))
180  criticalError("Could not add recording \"%s\", "
181  "because a recording with the same name does already exist.",
182  segment->recordingId().c_str());
183  _good = recSegCrawler->sync(in()[1].node);
184  recSegCrawler->reset();
185  }
186  for (bool _good = true; _good;) {
187  ConstSegmentRef segment = requestSegment(0);
188  if (!recSegSet_.addSegment(segment))
189  warning("Discard segment \"%s\", "
190  "because no recording name \"%s\" exists.",
191  segment->segmentId().c_str(),
192  segment->recordingId().c_str());
193  _good = recSegCrawler->sync(in()[0].node);
194  recSegCrawler->reset();
195  }
196  itRecSegSet_ = recSegSet_.begin();
197  endRecSegSet_ = recSegSet_.end();
198  if (itRecSegSet_ == endRecSegSet_)
199  return false;
200  if (dump_.isOpen()) {
201  for (RecordingSegmentationSet::const_iterator itRecSeg = recSegSet_.begin(); itRecSeg != recSegSet_.end(); ++itRecSeg) {
202  const RecordingSegmentation& recSeg = **itRecSeg;
203  dump_ << "Recording [" << recSeg.desc->recordingId() << ":"
204  << Core::form("%.2f-%.2f", recSeg.desc->startTime(), recSeg.desc->endTime()) << "]" << std::endl;
205  for (RecordingSegmentation::const_iterator itSeg = recSeg.begin(); itSeg != recSeg.end(); ++itSeg)
206  dump_ << "\tSegment [" << (*itSeg)->segmentId() << ":"
207  << Core::form("%.2f-%.2f", (*itSeg)->startTime(), (*itSeg)->endTime()) << "]" << std::endl;
208  dump_ << std::endl;
209  }
210  }
211  init(arguments);
212  return good();
213  }

216  ++itRecSegSet_;
217  if (itRecSegSet_ == endRecSegSet_)
218  return false;
219  sync();
220  return good();
221  }
