101 TheanoCommunicator::TheanoCommunicator(const Core::Configuration& c)
102  : Core::Component(c) {
103  key_t shMemKey = paramSharedMemKey(c);
104  if (shMemKey == -1) {
105  criticalError("sharedMemKey not specified");
106  }
107  log() << "allocating " << SHARED_MEM_SIZE << " bytes of shared memory...";
108  shId_ = shmget(shMemKey, SHARED_MEM_SIZE, IPC_CREAT | 0660);
109  if (shId_ < 0) {
110  criticalError("failed to allocated shared memory (shmget failed)");
111  }
112  shMem_ = shmat(shId_, 0, 0);
113  if (shMem_ == (void*)-1) {
114  criticalError("failed to allocate shared memory (shmat failed)");
115  }
116  //shmctl(shId_, IPC_RMID, 0); //mark for deletion, will only be deleted after detaching
117  //note: seems not to work like this..., so do it in destructor instead
118  shMemStatus() = IDLE;
119  log() << "shared memory allocation was successfull";
120 }

145 const Math::Matrix<f32>& TheanoCommunicator::getPosteriorsForSegment(const Bliss::SpeechSegment* segment) {
146  std::string segmentName = getSegmentName(segment);
147  if (segmentName == currentSegmentName_) {
148  /* use caching only a single time per segment
149  assumption: each segment is requested 2 times in a row,
150  once for rescoring and once for accumulation
151  after this the same segment might be requested again
152  but then we want to get new posteriors from theano */
153  currentSegmentName_ = "";
154  return posteriors_;
155  }
156 
157  timeval start;
158  TIMER_START(start);
159 
160  //shared memory transmission
161  std::string shMemSegmentName = &shMem<char>(SEGMENT_NAME_BEGIN);
162  if (segmentName != shMemSegmentName) {
163  criticalError("segment names do not match: ") << segmentName << " " << shMemSegmentName;
164  }
165  int rows = shMemRows();
166  int cols = shMemCols();
167  posteriors_.resize(rows, cols);
168  for (int i = 0; i < rows; ++i) {
169  for (int j = 0; j < cols; ++j) {
170  posteriors_[i][j] = shMemData(i * cols + j);
171  }
172  }
173  shMemStatus() = IDLE;
174 
175  currentSegmentName_ = segmentName;
176  timeval end;
177  f64 diff = 0;
178  TIMER_STOP(start, end, diff);
179  log() << "time getting posteriors: " << diff;
180  return posteriors_;
181 }

183 void TheanoCommunicator::writeErrorSignalForSegment(const Bliss::SpeechSegment* segment, float loss, const Math::Matrix<f32>& errSig) {
184  std::string segmentName = getSegmentName(segment);
185  log() << "writing error signal for segment " << segmentName;
186  if (shMemStatus() != IDLE) {

190  shMemCols() = static_cast<float>(errSig.nColumns());
191  shMemLoss() = loss;

194  for (u32 j = 0; j < errSig.nColumns(); ++j) {
195  shMemData(i * errSig.nColumns() + j) = errSig[i][j];
196  }
197  }
198

200  log() << "done writing error signal";
201 }
