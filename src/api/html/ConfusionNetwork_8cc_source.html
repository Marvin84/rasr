100 ConstConfusionNetworkRef sausageLattice2cn(ConstLatticeRef l, bool keepEpsArcs) {
101  if (!l || (l->initialStateId() == Fsa::InvalidStateId))
102  return ConstConfusionNetworkRef();
103  ConstBoundariesRef b = l->getBoundaries();
104  ConfusionNetwork* cn = new ConfusionNetwork;
105  cn->alphabet = l->getInputAlphabet();
106  cn->semiring = l->semiring();
107  u32 stateClusterId = 0;
108  Time begin = b->time(l->initialStateId());
109  for (ConstStateRef sr = l->getState(l->initialStateId()), nextSr; sr->hasArcs(); sr = nextSr) {
110  nextSr = l->getState(sr->begin()->target());
111  Time duration = b->time(nextSr->id()) - begin;
112  bool hasSlot = false;
113  for (State::const_iterator a = sr->begin(), end_a = sr->end(); a != end_a; ++a) {
114  if (a->target() != nextSr->id())
115  Core::Application::us()->criticalError(
116  "\"%s\" has not a \"sausage\" topology.", l->describe().c_str());
117  if (keepEpsArcs || (a->input() != Fsa::Epsilon)) {
118  if (!hasSlot) {
119  cn->push_back(ConfusionNetwork::Slot());
120  ++stateClusterId;
121  hasSlot = true;
122  }
123  cn->back().push_back(
124  ConfusionNetwork::Arc(
125  a->input(), a->weight(),
126  begin, duration,
127  stateClusterId - 1, stateClusterId));
128  }
129  }
130  begin += duration;
131  }
132  return ConstConfusionNetworkRef(cn);
133 }

140  Score calcConfidence(ScoreList::const_iterator itP, ScoreList::const_iterator endP) const {
141  switch (u32(endP - itP)) {
142  case 0:
143  return 1.0;
144  case 1:
145  return *itP;
146  default:
147  Score maxP = 0.0;
148  for (; itP != endP; ++itP)
149  if (*itP > maxP)
150  maxP = *itP;
151  return maxP;
152  }

158  ConstLatticeRef convert(ConstConfusionNetworkRef cn, bool approximateBoundaryTimes = false) const {
159  if (!cn)
160  return ConstLatticeRef();
161  StaticLattice* s = new StaticLattice;
162  s->setDescription("cn");
163  s->setType(Fsa::TypeAcceptor);
164  s->setProperties(Fsa::PropertyAcyclic, Fsa::PropertyAll);
165  s->setInputAlphabet(cn->alphabet);
166  s->setSemiring(cn->semiring);
167  State* sp = s->newState();
168  s->setInitialStateId(sp->id());
169  Flf::StaticBoundaries* bounds = approximateBoundaryTimes ? new StaticBoundaries : 0;
170  for (ConfusionNetwork::const_iterator itSlot = cn->begin(), endSlot = cn->end();
171  itSlot != endSlot; ++itSlot) {
172  State* nextSp = s->newState();
173  Score bestArcScore = Core::Type<Score>::max;
174  for (ConfusionNetwork::Slot::const_iterator itArc = itSlot->begin(), endArc = itSlot->end();
175  itArc != endArc; ++itArc) {
176  sp->newArc(nextSp->id(), itArc->scores, itArc->label);
177  if (bounds) {
178  Score score = itArc->scores->project(cn->semiring->scales());
179  if (score && score < bestArcScore) {
180  bounds->set(sp->id(), Boundary(itArc->begin));
181  bounds->set(nextSp->id(), Boundary(itArc->begin + itArc->duration));
182  bestArcScore = score;
183  }
184  }
185  }
186  sp = nextSp;
187  }
188  ScoresRef scores = cn->semiring->one();
189  if (cn->isNormalized()) {
190  scores = cn->semiring->clone(scores);
191  scores->set(cn->normalizedProperties->posteriorId, 1.0);
192  }
193  sp->setFinal(scores);
194  s->setBoundaries(ConstBoundariesRef(bounds));
195  return ConstLatticeRef(s);
196  }

209  ConstConfusionNetworkRef normalize(ConstConfusionNetworkRef cn, ScoreId posteriorId) const {
210  if (!cn)
211  return ConstConfusionNetworkRef();
212  if (posteriorId == Semiring::InvalidId)
213  Core::Application::us()->criticalError(
214  "No dimension storing a posterior probability distribution is given; "
215  "in order to normalize a CN a prob. dist. over all arcs in the slot is required, where lacking prob. mass is assigned to eps.");
216  ConfusionNetwork* normCn = new ConfusionNetwork(cn->size());
217  normCn->alphabet = cn->alphabet;
218  normCn->semiring = cn->semiring;
219  normCn->normalizedProperties = ConfusionNetwork::ConstNormalizedPropertiesRef(
220  new ConfusionNetwork::NormalizedProperties(posteriorId));
221  ConstSemiringRef semiring = normCn->semiring;
222  ArcPtrList arcPtrs;
223  for (u32 i = 0; i < normCn->size(); ++i) {
224  // Squeeze words -> all arcs with same label are merged into one arc
225  const ConfusionNetwork::Slot& slot = (*cn)[i];
226  ConfusionNetwork::Slot& normSlot = (*normCn)[i];
227  Score sum = 0.0;
228  if (!slot.empty()) {
229  if (arcPtrs.size() < slot.size())
230  arcPtrs.resize(slot.size(), 0);
231  ArcPtrList::iterator itArcPtr = arcPtrs.begin();
232  for (ConfusionNetwork::Slot::const_iterator itArc = slot.begin(), endArc = slot.end(); itArc != endArc; ++itArc, ++itArcPtr)
233  *itArcPtr = &*itArc;
234  std::sort(arcPtrs.begin(), arcPtrs.begin() + slot.size(), ArcPtrList::SortByLabel());
235  Score sumPosterior, maxPosterior;
236  Score sumBegin, sumDuration;
237  Fsa::LabelId lastLabel;
238  {
239  const ConfusionNetwork::Arc& arc = *arcPtrs.front();
240  Score posterior = arc.scores->get(posteriorId);
241  normSlot.push_back(ConfusionNetwork::Arc(arc.label, arc.scores));
242  lastLabel = arc.label;
243  sumBegin = posterior * Score(arc.begin);
244  sumDuration = posterior * Score(arc.duration);
245  sumPosterior = maxPosterior = posterior;
246  }
247  for (ArcPtrList::const_iterator itArcPtr = arcPtrs.begin() + 1, endArcPtr = arcPtrs.begin() + slot.size();
248  itArcPtr != endArcPtr; ++itArcPtr) {
249  const ConfusionNetwork::Arc& arc = **itArcPtr;
250  Score posterior = arc.scores->get(posteriorId);
251  if (arc.label != lastLabel) {
252  {
253  ConfusionNetwork::Arc& normArc = normSlot.back();
254  normArc.scores = semiring->clone(normArc.scores);
255  normArc.scores->set(posteriorId, sumPosterior);
256  normArc.begin = Time(Core::round(sumBegin / sumPosterior));
257  normArc.duration = Time(Core::round(sumDuration / sumPosterior));
258  }
259  sum += sumPosterior;
260  normSlot.push_back(ConfusionNetwork::Arc(arc.label, arc.scores));
261  lastLabel = arc.label;
262  sumBegin = posterior * Score(arc.begin);
263  sumDuration = posterior * Score(arc.duration);
264  sumPosterior = maxPosterior = posterior;
265  }
266  else {
267  if (posterior > maxPosterior) {
268  ConfusionNetwork::Arc& normArc = normSlot.back();
269  normArc.scores = arc.scores;
270  maxPosterior = posterior;
271  }
272  sumBegin += posterior * Score(arc.begin);
273  sumDuration += posterior * Score(arc.duration);
274  sumPosterior += posterior;
275  }
276  }
277  {
278  ConfusionNetwork::Arc& normArc = normSlot.back();
279  normArc.scores = semiring->clone(normArc.scores);
280  normArc.scores->set(posteriorId, sumPosterior);
281  normArc.begin = Time(Core::round(sumBegin / sumPosterior));
282  normArc.duration = Time(Core::round(sumDuration / sumPosterior));
283  sum += sumPosterior;
284  }
285  // Add missing proability mass to epsilon arc
286  if (sum != 1.0) {
287  bool normalize = false;
288  Score dEpsPosterior = 1.0 - sum;
289  if (dEpsPosterior < 0.0) {
290  // if (dEpsPosterior < -0.001)
291  if (dEpsPosterior < -0.01)
292  Core::Application::us()->warning(
293  "Normalize CN: Expected 1.0, got %f", (1.0 - dEpsPosterior));
294  normalize = true;
295  }
296  else {
297  if (normSlot.front().label == Fsa::Epsilon) {
298  normSlot.front().scores->add(posteriorId, dEpsPosterior);
299  }
300  else if (dEpsPosterior >= 0.001) {
301  ScoresRef scores = semiring->clone(semiring->one());
302  scores->set(posteriorId, dEpsPosterior);
303  normSlot.insert(normSlot.begin(), ConfusionNetwork::Arc(
304  Fsa::Epsilon,
305  scores,
306  (normSlot.front().begin + normSlot.front().duration) / 2, 0));
307  }
308  else
309  normalize = true;
310  }
311  if (normalize) {
312  Score norm = 1.0 / sum;
313  for (ConfusionNetwork::Slot::iterator itArc = normSlot.begin(), endArc = normSlot.end(); itArc != endArc; ++itArc)
314  itArc->scores->multiply(posteriorId, norm);
315  }
316  }
317  }
318  else {
319  ScoresRef scores = semiring->clone(semiring->one());
320  scores->set(posteriorId, 1.0);
321  normSlot.push_back(ConfusionNetwork::Arc(
322  Fsa::Epsilon,
323  scores,
324  Speech::InvalidTimeframeIndex, 0));
325  }
326  }
327  if (cn->hasMap()) {
328  ConfusionNetwork::MapProperties* mapProperties = new ConfusionNetwork::MapProperties(*cn->mapProperties);
329  mapProperties->reduce(); // normalized CN contains only "lattice-state+arc -> slot" information
330  normCn->mapProperties = ConfusionNetwork::ConstMapPropertiesRef(mapProperties);
331  }
332  return ConstConfusionNetworkRef(normCn);
333  }

335  ConstLatticeRef decode(ConstConfusionNetworkRef cn, ScoreId posteriorId) const {
336  if (!cn)
337  return ConstLatticeRef();
338  if (posteriorId == Semiring::InvalidId) {
339  verify(cn->isNormalized());
340  posteriorId = cn->normalizedProperties->posteriorId;
341  }
342  else if (!cn->isNormalized()) {
343  cn = normalize(cn, posteriorId);
344  }
345  verify((posteriorId != Semiring::InvalidId) && cn->isNormalized() && (cn->normalizedProperties->posteriorId == posteriorId));
346  ConstSemiringRef semiring = cn->semiring;
347  StaticBoundaries* b = new StaticBoundaries;
348  StaticLattice* s = new StaticLattice;
349  s->setDescription("decode-cn");
350  s->setType(Fsa::TypeAcceptor);
351  s->setProperties(Fsa::PropertyAcyclic | Fsa::PropertyLinear, Fsa::PropertyAll);
352  s->setInputAlphabet(cn->alphabet);
353  s->setSemiring(cn->semiring);
354  s->setBoundaries(ConstBoundariesRef(b));
355  s->setInitialStateId(0);
356  Flf::State* sp = new Flf::State(0);
357  s->setState(sp);
358  Time minEndTime = 0, preferredEndTime = 0; // Attention: preferredEndTime >= minEndTime
359  ScoreList pendingEpsPosteriors;
360  for (u32 i = 0; i < cn->size(); ++i) {
361  const ConfusionNetwork::Slot& pdf = (*cn)[i];
362  Score maxPosterior = 0.0;
363  ConfusionNetwork::Slot::const_iterator itBestPdf = pdf.end();
364  for (ConfusionNetwork::Slot::const_iterator itPdf = pdf.begin(); itPdf != pdf.end(); ++itPdf) {
365  Score posterior = itPdf->scores->get(posteriorId);
366  if (posterior > maxPosterior) {
367  maxPosterior = posterior;
368  itBestPdf = itPdf;
369  }
370  }
371  verify(itBestPdf != pdf.end());
372  if (itBestPdf->label != Fsa::Epsilon) {
373  if (preferredEndTime < itBestPdf->begin) {
374  b->set(sp->id(), Boundary(preferredEndTime));
375  ScoresRef scores = semiring->clone(semiring->one());
376  scores->set(posteriorId, calcConfidence(pendingEpsPosteriors.begin(), pendingEpsPosteriors.end()));
377  pendingEpsPosteriors.clear();
378  sp->newArc(sp->id() + 1, scores, Fsa::Epsilon, Fsa::Epsilon);
379  sp = new Flf::State(sp->id() + 1);
380  s->setState(sp);
381  preferredEndTime = itBestPdf->begin;
382  }
383  else
384  preferredEndTime = std::max(minEndTime, (preferredEndTime + itBestPdf->begin) / 2);
385  b->set(sp->id(), Boundary(preferredEndTime));
386  sp->newArc(sp->id() + 1, itBestPdf->scores, itBestPdf->label, itBestPdf->label);
387  sp = new Flf::State(sp->id() + 1);
388  s->setState(sp);
389  minEndTime = preferredEndTime + 1;
390  preferredEndTime = std::max(minEndTime, itBestPdf->begin + itBestPdf->duration);
391  }
392  else
393  pendingEpsPosteriors.push_back(maxPosterior);
394  }
395  b->set(sp->id(), Boundary(preferredEndTime));
396  ScoresRef scores = semiring->one();
397  if (!pendingEpsPosteriors.empty()) {
398  scores = semiring->clone(scores);
399  scores->set(posteriorId, calcConfidence(pendingEpsPosteriors.begin(), pendingEpsPosteriors.end()));
400  pendingEpsPosteriors.clear();
401  }
402  sp->setFinal(scores);
403  return ConstLatticeRef(s);
404  }

542 class CnFeatures : public Core::ReferenceCounted {
543 public:
544  static const Score CorCost;
545  static const Score SubCost;

549 private:
550  CnFeatureConfiguration cnConfig_;
551  bool needsModification_;
552  bool needsOracle_;
553  bool needsPosterior_;
554  ConstConfusionNetworkRef normalizedCn_;
555  std::vector<Fsa::StateId> slotIdToNonEpsilonSlotIdMap_;
556 
557 public:
558  CnFeatures(CnFeatureConfiguration cnConfig)
559  : cnConfig_(cnConfig) {
560  if (!cnConfig_.cn || !cnConfig_.cn->hasMap())
561  Core::Application::us()->criticalError("Confusion Network with lattice/CN map required.");
562  if (
563  (cnConfig_.confidenceId != Semiring::InvalidId) ||
564  (cnConfig_.scoreId != Semiring::InvalidId) ||
565  (cnConfig_.errId != Semiring::InvalidId) ||
566  (cnConfig_.oracleOutput) ||
567  (cnConfig_.slotEntropyId != Semiring::InvalidId) ||
568  (cnConfig_.slotId != Semiring::InvalidId) ||
569  (cnConfig_.nonEpsSlotId != Semiring::InvalidId)) {
570  needsModification_ = true;
571  // oracle CN required
572  if ((cnConfig_.errId != Semiring::InvalidId) || cnConfig_.oracleOutput) {
573  needsOracle_ = true;
574  if (!cnConfig_.cn->isOracleAlignment())
575  Core::Application::us()->criticalError("Confusion Network with oracle alignment required.");
576  }
577  // normalized CN required
578  if ((cnConfig_.confidenceId != Semiring::InvalidId) || (cnConfig_.scoreId != Semiring::InvalidId) || (cnConfig_.slotEntropyId != Semiring::InvalidId) || (cnConfig_.nonEpsSlotId != Semiring::InvalidId)) {
579  if (!cnConfig_.cn->isNormalized()) {
580  if (cnConfig_.cnPosteriorId == Semiring::InvalidId)
581  Core::Application::us()->criticalError("Posterior dimension for confusion network normalization required.");
582  normalizedCn_ = normalizeCn(cnConfig_.cn, cnConfig_.cnPosteriorId);
583  }
584  else {
585  cnConfig_.cnPosteriorId = cnConfig_.cn->normalizedProperties->posteriorId;
586  normalizedCn_ = cnConfig_.cn;
587  }
588  needsPosterior_ =
589  (cnConfig_.confidenceId != Semiring::InvalidId) ||
590  (cnConfig_.scoreId != Semiring::InvalidId);
591  }
592  // mapping to eps-slot free CN required
593  if (cnConfig_.nonEpsSlotId != Semiring::InvalidId) {
594  slotIdToNonEpsilonSlotIdMap_.resize(cnConfig_.cn->size(), Fsa::InvalidStateId);
595  Fsa::StateId nonEpsSlotId = 0;
596  std::vector<Fsa::StateId>::iterator itNonEpsilonSlotId = slotIdToNonEpsilonSlotIdMap_.begin();
597  for (ConfusionNetwork::const_iterator itSlot = cnConfig_.cn->begin(), endSlot = cnConfig_.cn->end(), itNormalizedSlot = normalizedCn_->begin();
598  itSlot != endSlot; ++itSlot, ++itNormalizedSlot, ++itNonEpsilonSlotId) {
599  if ((itNormalizedSlot->front().label == Fsa::Epsilon) && (itNormalizedSlot->front().scores->get(cnConfig_.cnPosteriorId) >= cnConfig_.epsSlotThreshold))
600  continue;

648  void modify(Arc& arc, Fsa::StateId slotId, Fsa::StateId fromSlotId, Fsa::StateId toSlotId) const {
649  if (!needsModification_)
650  return;
651  arc.weight_ = cnConfig_.semiring->clone(arc.weight_);
652  if (slotId == Fsa::InvalidStateId) {
653  if (cnConfig_.confidenceId != Semiring::InvalidId)
654  arc.weight()->set(cnConfig_.confidenceId, 1.0);
655  if (cnConfig_.scoreId != Semiring::InvalidId)
656  arc.weight()->set(cnConfig_.scoreId, Semiring::One);
657  if (cnConfig_.errId != Semiring::InvalidId)
658  arc.weight()->set(cnConfig_.errId, 0.0);
659  if (cnConfig_.oracleOutput)
660  arc.output_ = Fsa::InvalidLabelId;
661  if (cnConfig_.slotEntropyId != Semiring::InvalidId)
662  arc.weight()->set(cnConfig_.slotEntropyId, Semiring::Invalid);
663  if (cnConfig_.slotId != Semiring::InvalidId)
664  arc.weight()->set(cnConfig_.slotId, Semiring::Invalid);
665  if (cnConfig_.nonEpsSlotId != Semiring::InvalidId)
666  arc.weight()->set(cnConfig_.nonEpsSlotId, Semiring::Invalid);
667  return;
668  }
669  verify((fromSlotId != Fsa::InvalidStateId) && (toSlotId != Fsa::InvalidStateId));
670  verify((fromSlotId <= slotId) && (slotId <= toSlotId) && (toSlotId < cnConfig_.cn->size()));
671  // oracle based features
672  if (needsOracle_) {
673  const ConfusionNetwork::OracleAlignmentProperties& props = oracleProperties();
674  // error over spanned slots
675  if (cnConfig_.errId != Semiring::InvalidId) {
676  Score err = 0.0;
677  Fsa::StateId thisSlotId = fromSlotId;
678  for (; thisSlotId < slotId; ++thisSlotId)
679  err += cost(Fsa::Epsilon, props.alignment[thisSlotId].label);
680  err += cost(arc.input(), props.alignment[thisSlotId].label);
681  for (++thisSlotId; thisSlotId <= toSlotId; ++thisSlotId)
682  err += cost(Fsa::Epsilon, props.alignment[thisSlotId].label);
683  arc.weight()->set(cnConfig_.errId, err);
684  }
685  // oracle output
686  if (cnConfig_.oracleOutput) {
687  arc.output_ = props.alignment[slotId].label;
688  }
689  }
690  // slot entropy
691  if (cnConfig_.slotEntropyId != Semiring::InvalidId) {
692  const ConfusionNetwork::Slot& slot = (*normalizedCn_)[slotId];
693  Score e = 0.0;
694  for (ConfusionNetwork::Slot::const_iterator itArc = slot.begin(), endArc = slot.end(); itArc != endArc; ++itArc) {
695  Score p = itArc->scores->get(cnConfig_.cnPosteriorId);
696  e += p * ::log(p);
697  }
698  e = -e;
699  arc.weight()->set(cnConfig_.slotEntropyId, e);
700  }
701  // posterior based features
702  if (needsPosterior_) {
703  const ConfusionNetwork::NormalizedProperties& props = normalizedProperties();
704  // slot confidence
705  if (cnConfig_.confidenceId != Semiring::InvalidId) {
706  Score posterior = props.posteriorScore((*normalizedCn_)[slotId], arc.input());
707  arc.weight()->set(cnConfig_.confidenceId, posterior);
708  }
709  // score over spanned slots
710  if (cnConfig_.scoreId != Semiring::InvalidId) {
711  Score score = 0.0;
712  for (Fsa::StateId thisSlotId = fromSlotId; thisSlotId <= toSlotId; ++thisSlotId) {
713  Score posterior = props.posteriorScore(
714  (*normalizedCn_)[thisSlotId],
715  ((thisSlotId == slotId) ? arc.input() : Fsa::Epsilon));
716  if (posterior == 0.0) {
717  score = Semiring::Max;
718  break;
719  }
720  else
721  score += -::log(posterior);
722  }
723  arc.weight()->set(cnConfig_.scoreId, score);
724  }
725  }
726  // slot id
727  if (cnConfig_.slotId != Semiring::InvalidId) {
728  arc.weight()->set(cnConfig_.slotId, Score(slotId));
729  }
730  // non-eps slot id
731  if (cnConfig_.nonEpsSlotId != Semiring::InvalidId) {
732  if (arc.input() == Fsa::Epsilon) {
733  //verify_(((*cn_)[slotId].begin() + arcId)->label == Fsa::Epsilon);
734  arc.weight()->set(cnConfig_.nonEpsSlotId, Semiring::Invalid);
735  }
736  else {
737  //verify_(((*cn_)[slotId].begin() + arcId)->label != Fsa::Epsilon);
738  ScoreId nonEpsSlotId = slotIdToNonEpsilonSlotIdMap_[slotId];
739  if (nonEpsSlotId == Semiring::InvalidId) {
740  arc.weight()->set(cnConfig_.nonEpsSlotId, Semiring::Invalid);
741  arc.input_ = Fsa::Epsilon;
742  }
743  else {
744  arc.weight()->set(cnConfig_.nonEpsSlotId, Score(nonEpsSlotId));
745  }
746  }
747  }
748  }

761 ConstLatticeRef composeAndAddCnFeatures(ConstLatticeRef l, const CnFeatures& cnFeatures) {
762  if (!l)
763  return ConstLatticeRef();
764  StaticLattice* s = new StaticLattice;
765  StaticBoundaries* b = new StaticBoundaries;
766  s->setBoundaries(ConstBoundariesRef(b));
767  copy(l, s, b);
768  if (cnFeatures.cnConfig().oracleOutput) {
769  s->setType(Fsa::TypeTransducer);
770  s->setOutputAlphabet(s->getInputAlphabet());
771  }
772  std::vector<Fsa::StateId> minOut(s->size(), Core::Type<Fsa::StateId>::max);
773  for (Fsa::StateId sid = 0; sid < s->size(); ++sid) {
774  const State* sp = s->fastState(sid);
775  if (!sp)
776  continue;
777  if (sp->hasArcs()) {
778  ConfusionNetwork::MapProperties::Map::const_iterator itMap = cnFeatures.mapProperties().state(sp->id());
779  for (State::const_iterator a = sp->begin(), end = sp->end(); a != end; ++a, ++itMap) {
780  Fsa::StateId slotId = itMap->sid;
781  if (slotId != Fsa::InvalidStateId)
782  if (slotId < minOut[sid])
783  minOut[sid] = slotId;
784  }
785  }
786  else
787  minOut[sid] = cnFeatures.cn().size();
788  }
789  const Semiring& semiring = *s->semiring();
790  if (!cnFeatures.cnConfig().compose) {
791  s->setDescription(Core::form("addCnFeatures(%s)", s->describe().c_str()));
792  for (Fsa::StateId sid = 0; sid < s->size(); ++sid) {
793  State* sp = s->fastState(sid);
794  if (!sp)
795  continue;
796  Fsa::StateId fromSlotId = minOut[sid];
797  ConfusionNetwork::MapProperties::Map::const_iterator itMap = cnFeatures.mapProperties().state(sp->id());
798  for (State::iterator a = sp->begin(), end = sp->end(); a != end; ++a, ++itMap) {
799  Fsa::StateId slotId = itMap->sid;
800  Fsa::StateId nextSlotId = minOut[a->target()];
801  cnFeatures.modify(
802  *a, slotId,
803  (fromSlotId == Core::Type<Fsa::StateId>::max) ? slotId : fromSlotId,
804  (nextSlotId == Core::Type<Fsa::StateId>::max) ? slotId : nextSlotId - 1);
805  }
806  }
807  }
808  else {
809  s->setDescription(Core::form("composeAndAddCnFeatures(%s)", s->describe().c_str()));
810  bool duplicateOutput = cnFeatures.cnConfig().duplicateOutput;
811  Fsa::StateId endSid = s->size(), nextSid = s->size();
812  for (Fsa::StateId sid = 0; sid < endSid; ++sid) {
813  State* sp = s->fastState(sid);
814  if (!sp)
815  continue;
816  Fsa::StateId fromSlotId = minOut[sid];
817  verify(fromSlotId != Core::Type<Fsa::StateId>::max); // can happen, if lattice has (many) arcs of length 0
818  ConfusionNetwork::MapProperties::Map::const_iterator itMap = cnFeatures.mapProperties().state(sp->id());
819  for (State::iterator a = sp->begin(), end = sp->end(); a != end; ++a, ++itMap) {
820  Fsa::StateId slotId = itMap->sid;
821  verify(slotId != Fsa::InvalidStateId);
822  Fsa::StateId toSlotId = minOut[a->target()];
823  verify(toSlotId != Core::Type<Fsa::StateId>::max); // can happen, if lattice has (many) arcs of length 0
824  --toSlotId;
825  cnFeatures.modify(*a, slotId);
826  verify((fromSlotId <= slotId) && (slotId <= toSlotId));
827  if ((fromSlotId != slotId) || (slotId != toSlotId)) {
828  const Arc tmpArc = *a;
829  Fsa::LabelId epsOutputLabel = (duplicateOutput && (l->type() != Fsa::TypeAcceptor)) ? tmpArc.output() : Fsa::Epsilon;
830  State * thisSp = sp, *nextSp = 0;
831  Arc* thisArc = &*a;
832  Fsa::StateId thisSlotId = fromSlotId;
833  if (thisSlotId < slotId) {
834  const Boundary fromBoundary = b->get(sid);
835  nextSp = new Flf::State(nextSid++);
836  s->setState(nextSp);
837  b->set(nextSp->id(), fromBoundary);
838  thisArc->target_ = nextSp->id();
839  thisArc->weight_ = semiring.one();
840  thisArc->input_ = Fsa::Epsilon;
841  a->output_ = epsOutputLabel;
842  cnFeatures.modify(*thisArc, thisSlotId);
843  thisSp = nextSp;
844  for (++thisSlotId; thisSlotId < slotId; ++thisSlotId) {
845  nextSp = new Flf::State(nextSid++);
846  s->setState(nextSp);
847  b->set(nextSp->id(), fromBoundary);
848  thisArc = thisSp->newArc(nextSp->id(), semiring.one(), Fsa::Epsilon, epsOutputLabel);
849  cnFeatures.modify(*thisArc, thisSlotId);
850  thisSp = nextSp;
851  }
852  *(thisArc = thisSp->newArc()) = tmpArc;
853  }
854  if (thisSlotId < toSlotId) {
855  const Boundary toBoundary = b->get(tmpArc.target());
856  for (++thisSlotId; thisSlotId <= toSlotId; ++thisSlotId) {
857  nextSp = new Flf::State(nextSid++);
858  s->setState(nextSp);
859  b->set(nextSp->id(), toBoundary);
860  thisArc->target_ = nextSp->id();
861  thisArc = nextSp->newArc(Fsa::InvalidStateId, semiring.one(), Fsa::Epsilon, epsOutputLabel);
862  cnFeatures.modify(*thisArc, thisSlotId);
863  }
864  thisArc->target_ = tmpArc.target();
865  }
866  }
867  }
868  }
869  }
870  return ConstLatticeRef(s);
871 }
