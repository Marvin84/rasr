54 void SimpleAlignAutomaton::generateDiagonalOrHorizontalArcs_(State* s,
55  const unsigned targetIndexIncrement,
56  const unsigned nCoveredTargetWords,
57  const unsigned nCoveredSourceWords,
58  const std::vector<unsigned>& sourcePositionHistory) const {
59  // get "offset" cost defined by the transition
60  double cost = factorTransition_ * (transitionProbs_[1 - targetIndexIncrement]);
61 
62  // loop over all possible input words and create arcs with the proper probability
63  for (unsigned currentSourceWordIndex = 0; currentSourceWordIndex < J_ - 1; ++currentSourceWordIndex) {
64  // initialize lexicon key and index vector with current source word
65  std::vector<Fsa::LabelId> diagonalKey(1, mappedSourceSentence_[currentSourceWordIndex]);
66  std::vector<unsigned> diagonalJ(order_, 0);
67 
68  if (order_ > 0) {
69  diagonalJ[0] = currentSourceWordIndex;
70  }
71 
72  Fsa::LabelId currentTargetWordAlphabetIndex;
73 
74  // new source position j=0, ..., J_-1
75  // new target position i=i+1
76  if (targetIndexIncrement == 1) {
77  // Diagonal movement, we increase the target word index by 1
78  // nCoveredTargetWords is the number of target words covered so far.
79  // the arcs leavin this state are labeled with the word that is the first
80  // uncovered target word. since the target sentence vector is indexed
81  // starting with 0, the index of the next target word is the same as
82  // nCoveredTargetWords
83  currentTargetWordAlphabetIndex = mappedTargetSentence_[nCoveredTargetWords];
84  }
85  else {
86  currentTargetWordAlphabetIndex = nullWordIndex_;
87  }
88  diagonalKey.push_back(currentTargetWordAlphabetIndex);
89 
90  // loop copies the reduced source and target context
91  // stops at order_-1 (dont copy oldest history elements)
92  for (unsigned n = 0; n < order_; ++n) {
93  if (n < nCoveredSourceWords) {
94  diagonalKey.push_back(mappedSourceSentence_[sourcePositionHistory[n]]);
95  }
96  else {
97  diagonalKey.push_back(sentenceBeginPaddingSymbolIndex_);
98  }
99 
100  // next target position i+1, going back in history with each n
101 
102  // when going back in the target history, we ave to take care of the
103  // sentence boundary. when we move past the first wordposition, words
104  // are padded by sentence begin symbols
105  if (nCoveredTargetWords > 0) {
106  diagonalKey.push_back(mappedTargetSentence_[nCoveredTargetWords - n - 1]);
107  }
108  else {
109  diagonalKey.push_back(sentenceBeginPaddingSymbolIndex_);
110  }
111  if (order_ > 1) {
112  // update index vector
113  diagonalJ[n + 1] = sourcePositionHistory[n];
114  }
115  }
116 
117  // get StateId
118  Fsa::StateId arcTarget = indices2StateId(nCoveredTargetWords + targetIndexIncrement, nCoveredSourceWords + 1, diagonalJ);
119  diagonalKey[0] = mappedSourceSentence_[currentSourceWordIndex];
120 
121  if (targetIndexIncrement == 1) {
122  // Diagonal movement, include target label
123  s->newArc(arcTarget, Fsa::Weight(cost + factorLexicon_ * lexicon_->getReverseCost(1 - targetIndexIncrement, diagonalKey)), inputSentence_[currentSourceWordIndex], outputSentence_[nCoveredTargetWords]);
124  }
125  else {
126  s->newArc(arcTarget, Fsa::Weight(cost + factorLexicon_ * lexicon_->getReverseCost(1 - targetIndexIncrement, diagonalKey)), inputSentence_[currentSourceWordIndex], Fsa::Epsilon);
127  }
128  }
129 }

131 ConstStateRef SimpleAlignAutomaton::getState(StateId s) const {
132  //! create new state to be returned
133  State* result = new State(s);
134  unsigned nCoveredTargetWords;
135  std::vector<unsigned> sourcePositionHistory(order_ + 1, sentenceBeginPaddingSymbolIndex_);
136  unsigned nCoveredSourceWords;
137  this->stateId2Indices(s, nCoveredTargetWords, nCoveredSourceWords, sourcePositionHistory);
138 
139  StateId arcTarget;
140 
141  // Diagonal Movement
142 
143  if (nCoveredTargetWords < I_ - 1 && nCoveredSourceWords < J_ - 1) {
144  this->generateDiagonalOrHorizontalArcs_(result, 1, nCoveredTargetWords, nCoveredSourceWords, sourcePositionHistory);
145  }
146 
147  // Horizontal Movement
148 
149  if (nCoveredSourceWords < J_ - 1) {
150  this->generateDiagonalOrHorizontalArcs_(result, 0, nCoveredTargetWords, nCoveredSourceWords, sourcePositionHistory);
151  }
152 
153  // Vertical Movement
154  // just the target index changes, this means, we just have to add 1 to the state index
155 
156  if (nCoveredTargetWords < I_ - 1) {
157  arcTarget = s + 1;
158 
159  std::vector<Fsa::LabelId> verticalKey;
160 
161  verticalKey.push_back(nullWordIndex_);
162  verticalKey.push_back(mappedTargetSentence_[nCoveredTargetWords]);
163 
164  // loop copies the full source context (it is not changed in a vertical movement)
165  for (unsigned n = 0; n < order_; ++n) {
166  // need to check if order ist too long for history
167  verticalKey.push_back(mappedSourceSentence_[sourcePositionHistory[n]]);
168  if (nCoveredTargetWords < n + 1) {
169  verticalKey.push_back(sentenceBeginPaddingSymbolIndex_);
170  }
171  else {
172  verticalKey.push_back(mappedTargetSentence_[nCoveredTargetWords - n - 1]);
173  }
174  }
175 
176  double cost = factorLexicon_ * lexicon_->getReverseCost(2, verticalKey);
177  cost += factorTransition_ * transitionProbs_[2];
178  result->newArc(arcTarget, Fsa::Weight(cost), Epsilon, outputSentence_[nCoveredTargetWords]);
179  }
180 
181  if (nCoveredTargetWords == I_ - 1 && nCoveredSourceWords == J_ - 1) {
182  result->setFinal(this->semiring()->one());
183  }
184 
185  return ConstStateRef(result);
186 }
