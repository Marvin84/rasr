183 void LatticeSetGenerator::appendAcousticRescorers() {
184  std::vector<std::string> acousticNames = paramAcousticExtractors(config);
185  for (std::vector<std::string>::iterator name = acousticNames.begin();
186  name != acousticNames.end(); ++name) {
187  switch (paramSearchType(select(*name))) {
188  case exactMatch: {
189  AcousticLatticeRescorer* rescorer;
190  if (paramShareAcousticModel(config)) {
191  rescorer = new AlignmentLatticeRescorer(select(*name));
192  }
193 #ifdef MODULE_SPEECH_DT_ADVANCED
194  else {
195  CombinedAcousticLatticeRescorer* r =
196  new CombinedAcousticLatticeRescorer(select(*name));
197  r->setSegmentwiseFeatureExtractor(segmentwiseFeatureExtractor());
198  rescorer = r;
199  }
200 #endif
201  rescorer->setAlignmentGenerator(alignmentGenerator());
202  extractors_.push_back(rescorer);
203  log("\"%s\" appended (acoustic-rescorer, exact match)", name->c_str());
204  } break;
205 #ifdef MODULE_SPEECH_DT_ADVANCED
206  case fullSearch: {
207  RecognizerWithConstrainedLanguageModel* rescorer =
208  new RecognizerWithConstrainedLanguageModel(select(*name), lexicon_);
209  rescorer->setSegmentwiseFeatureExtractor(segmentwiseFeatureExtractor());
210  extractors_.push_back(rescorer);
211  log("\"%s\" appended (acoustic-rescorer, full search)", name->c_str());
212  } break;
213 #endif
214  }
215  }
216 }

422 void LatticeSetGenerator::processWordLattice(
423  Lattice::ConstWordLatticeRef lattice, Bliss::SpeechSegment* s) {
424  if (!segmentwiseFeatureExtractor_ || segmentwiseFeatureExtractor_->valid()) {
425  timeval start, end;
426  int i = 0;
427  TIMER_START(start)
428  Core::Ref<Lattice::WordLattice> rescored(new Lattice::WordLattice);
429  for (LatticeExtractors::iterator it = extractors_.begin(); it != extractors_.end(); ++it, ++i) {
430  timeval startRescorer, endRescorer;
431  TIMER_START(startRescorer)
432  Lattice::ConstWordLatticeRef r = (*it)->extract(lattice, s);
433  if (!r)
434  continue;
435  if (!rescored->wordBoundaries()) {
436  rescored->setWordBoundaries(r->wordBoundaries());
437  }
438  rescored->setFsa(r->mainPart(), (*it)->name());
439  TIMER_GPU_STOP(startRescorer, endRescorer, true, timeRescorers_[i])
440  }
441  TIMER_GPU_STOP(start, end, true, timeProcessSegment_)
442  Precursor::processWordLattice(rescored, s);
443  }
444  else {
445  warning("invalid segmentwise feature extractor");
446  Precursor::processWordLattice(Lattice::ConstWordLatticeRef(), s);
447  }
448 }

613 void LatticeSetReader::leaveSpeechSegment(Bliss::SpeechSegment* s) {
614  require(archiveReader_);
615  timeval start, end;
616  TIMER_START(start)
