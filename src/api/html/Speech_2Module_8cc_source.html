122 MixtureSetTrainer* Module_::createMixtureSetTrainer(const Core::Configuration& configuration) const {
123  Speech::MixtureSetTrainer* result = 0;
124  switch (Mm::Module_::paramEstimatorType(configuration)) {
125  case Mm::Module_::maximumLikelihood:
126  result = new MlMixtureSetTrainer(configuration);
127  break;
128 #ifdef MODULE_SPEECH_DT
129  case Mm::Module_::discriminative:
130  case Mm::Module_::discriminativeWithISmoothing:
131 #endif
132 #if defined(MODULE_SPEECH_DT)
133  result = createDiscriminativeMixtureSetTrainer(configuration);
134  break;
135 #endif
136  default:
137  defect();
138  break;
139  }
140  return result;
141 }
